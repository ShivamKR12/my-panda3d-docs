269 // the thread itself, so we shouldn't try to call join().

319void PStatClientImpl::
320new_frame(int thread_index, int frame_number) {
321 double frame_start = get_real_time();
322
323 nassertv(thread_index >= 0 && thread_index < _client->_num_threads);
324
325 PStatClient::InternalThread *pthread = _client->get_thread_ptr(thread_index);
326 nassertv(pthread != nullptr);
327
328 // If we're the main thread, we should exchange control packets with the
329 // server.
330#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
331 if (thread_index == 0 && _thread == nullptr) {
332#else
333 if (thread_index == 0) {
334#endif
335 transmit_control_data();
336 }
337
338 // If we've got the UDP port by the time the frame starts, it's time to
339 // become active and start actually tracking data.
340 if (_got_udp_port) {
341 pthread->_is_active = true;
342 }
343
344 if (!pthread->_is_active) {
345 return;
346 }
347
348 PStatFrameData frame_data;
349
350 if (!pthread->_frame_data.is_empty()) {
351 // Collector 0 is the whole frame.
352 _client->stop(0, thread_index, frame_start);
353
354 // Fill up the level data for all the collectors who have level data for
355 // this pthread.
356 int num_collectors = _client->_num_collectors;
357 PStatClient::CollectorPointer *collectors =
358 (PStatClient::CollectorPointer *)_client->_collectors;
359 for (int i = 0; i < num_collectors; i++) {
360 const PStatClient::PerThreadData &ptd =
361 collectors[i]->_per_thread[thread_index];
362 if (ptd._has_level) {
363 pthread->_frame_data.add_level(i, ptd._level);
364 }
365 }
366 pthread->_frame_data.swap(frame_data);
367 if (frame_number == -1) {
368 frame_number = pthread->_frame_number;
369 }
370
371 // Record the number of context switches on this thread.
372 if (_thread_profiling) {
373 size_t total, involuntary;
374 PT(Thread) thread = pthread->_thread.lock();
375 if (thread != nullptr && thread->get_context_switches(total, involuntary)) {
376 size_t total_this_frame = total - pthread->_context_switches;
377 pthread->_context_switches = total;
378 frame_data.add_level(_cswitch_pcollector.get_index(), total_this_frame);
379
380 if (involuntary != 0) {
381 size_t involuntary_this_frame = involuntary - pthread->_involuntary_context_switches;
382 pthread->_involuntary_context_switches = involuntary;
383 frame_data.add_level(_cswitch_involuntary_pcollector.get_index(), involuntary_this_frame);
384 }
385 }
386 _client->clear_level(_cswitch_sleep_pcollector.get_index(), thread_index);
387 _client->clear_level(_cswitch_yield_pcollector.get_index(), thread_index);
388 _client->clear_level(_cswitch_cvar_pcollector.get_index(), thread_index);
389 }
390 }
391 else if (_thread_profiling) {
392 // Record the initial number of context switches.
393 PT(Thread) thread = pthread->_thread.lock();
394 if (thread != nullptr) {
395 thread->get_context_switches(pthread->_context_switches,
396 pthread->_involuntary_context_switches);
397 }
398 _client->clear_level(_cswitch_sleep_pcollector.get_index(), thread_index);
399 _client->clear_level(_cswitch_yield_pcollector.get_index(), thread_index);
400 _client->clear_level(_cswitch_cvar_pcollector.get_index(), thread_index);
401 }
402
403 pthread->_frame_data.clear();
404 pthread->_frame_number = frame_number + 1;
405 _client->start(0, thread_index, frame_start);
406
407 // Also record the time for the PStats operation itself.
408 int current_thread_index = Thread::get_current_thread()->get_pstats_index();
409 int pstats_index = PStatClient::_pstats_pcollector.get_index();
410 _client->start(pstats_index, current_thread_index, frame_start);
411
412 if (!frame_data.is_empty()) {
413 enqueue_frame_data(thread_index, frame_number, std::move(frame_data));
414 }
415 _client->stop(pstats_index, current_thread_index, get_real_time());
416}

478void PStatClientImpl::
479enqueue_frame_data(int thread_index, int frame_number,
480 PStatFrameData &&frame_data) {
481#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
482 if (_thread != nullptr) {
483 int max_size = pstats_max_queue_size;
484 _thread_lock.lock();
485 if (max_size < 0 || _frame_queue.size() < (size_t)max_size) {
486 _frame_queue.emplace_back(thread_index, frame_number);
487 frame_data.swap(_frame_queue.back()._frame_data);
488 }
489 _thread_cvar.notify();
490 _thread_lock.unlock();
491 return;
492 }
493#endif
494
495 // We don't have a thread, so transmit it directly.
496 if (_is_connected) {
497 transmit_frame_data(thread_index, frame_number, frame_data);
498 }
499}

501#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
505void PStatClientImpl::
506thread_main() {
507 MutexHolder holder(_thread_lock);
508 transmit_control_data();
509
510 while (!_thread_should_shutdown) {
511 while (_frame_queue.empty() && !_thread_should_shutdown) {
512 _thread_cvar.wait();
513 }
514
515 while (!_frame_queue.empty()) {
516 // Dequeue up to 8 at a time, to decrease the amount of times we need to
517 // hold the lock.
518 QueuedFrame frames[8];
519 int num_frames = 0;
520
521 while (!_frame_queue.empty() && num_frames < 8) {
522 QueuedFrame &qf = _frame_queue.front();
523 frames[num_frames]._thread_index = qf._thread_index;
524 frames[num_frames]._frame_number = qf._frame_number;
525 frames[num_frames]._frame_data.swap(qf._frame_data);
526 ++num_frames;
527 _frame_queue.pop_front();
528 }
529 _thread_lock.unlock();
530
531 transmit_control_data();
532
533 if (num_frames > 0) {
534 for (int i = 0; i < num_frames; ++i) {
535 QueuedFrame &qf = frames[i];
536 transmit_frame_data(qf._thread_index, qf._frame_number, qf._frame_data);
537 }
538 }
539
540 _thread_lock.lock();
541 }
542 }
543
544 _thread = nullptr;
545}
546#endif

552void PStatClientImpl::
553transmit_frame_data(int thread_index, int frame_number,
554 const PStatFrameData &frame_data) {
555 nassertv(thread_index >= 0 && thread_index < _client->_num_threads);
556 PStatClient::InternalThread *thread = _client->get_thread_ptr(thread_index);
557 nassertv(thread != nullptr);
558
559 if (_is_connected && thread->_is_active) {
560
561 // We don't want to send too many packets in a hurry and flood the server.
562 // Check that enough time has elapsed for us to send a new packet. If
563 // not, we'll drop this packet on the floor and send a new one next time
564 // around.
565 double now = get_real_time();
566 if (now >= thread->_next_packet) {
567 // We don't want to send more than _max_rate UDP-size packets per
568 // second, per thread.
569 double packet_delay = 1.0 / _max_rate;
570
571 // Send new data.
572 NetDatagram datagram;
573 // We always start with a zero byte, to differentiate it from a control
574 // message.
575 datagram.add_uint8(0);
576
577 datagram.add_uint16(thread_index);
578 datagram.add_uint32(frame_number);
579
580 bool sent;
581
582 if (!frame_data.write_datagram(datagram, _client)) {
583 // Too many events to fit in a single datagram. Maybe it was a long
584 // frame load or something. Just drop the datagram.
585 sent = false;
586
587 } else if (_writer.is_valid_for_udp(datagram)) {
588 if (_udp_count * _udp_count_factor < _tcp_count * _tcp_count_factor) {
589 // Send this one as a UDP packet.
590 nassertv(_got_udp_port);
591 sent = _writer.send(datagram, _udp_connection, _server);
592 _udp_count++;
593
594 if (_udp_count == 0) {
595 // Wraparound!
596 _udp_count = 1;
597 _tcp_count = 1;
598 }
599
600 } else {
601 // Send this one as a TCP packet.
602 sent = _writer.send(datagram, _tcp_connection);
603 _tcp_count++;
604
605 if (_tcp_count == 0) {
606 // Wraparound!
607 _udp_count = 1;
608 _tcp_count = 1;
609 }
610 }
611
612 } else {
613 sent = _writer.send(datagram, _tcp_connection);
614 // If our packets are so large that we must ship them via TCP, then
615 // artificially slow down the packet rate even further.
616 int packet_ratio =
617 (datagram.get_length() + maximum_udp_datagram - 1) /
618 maximum_udp_datagram;
619 packet_delay *= (double)packet_ratio;
620 }
621
622 thread->_next_packet = now + packet_delay;
623
624 if (!sent) {
625 if (pstats_cat.is_debug()) {
626 pstats_cat.debug()
627 << "Couldn't send packet.\n";
628 }
629 }
630 }
631 }

638void PStatClientImpl::
639transmit_control_data() {
640 // Check for new messages from the server.
641 while (_is_connected && _reader.data_available()) {
642 NetDatagram datagram;
643
644 if (_reader.get_data(datagram)) {
645 PStatServerControlMessage message;
646 if (message.decode(datagram)) {
647 handle_server_control_message(message);
648
649 } else {
650 pstats_cat.error()
651 << "Got unexpected message from server.\n";
652 }
653 }
654 }
655
656 if (_is_connected) {
657 report_new_collectors();
658 report_new_threads();
659 }

708void PStatClientImpl::
709report_new_collectors() {
710 // Empirically, we determined that you can't send more than about 1400
711 // collectors at once without exceeding the 64K limit on a single datagram.
712 // So we limit ourselves here to sending only half that many.
713 static const int max_collectors_at_once = 700;
714
715 while (_is_connected && _collectors_reported < _client->_num_collectors) {
716 PStatClientControlMessage message;
717 message._type = PStatClientControlMessage::T_define_collectors;
718 int i = 0;
719 while (_collectors_reported < _client->_num_collectors &&
720 i < max_collectors_at_once) {
721 message._collectors.push_back(_client->get_collector_def(_collectors_reported));
722 _collectors_reported++;
723 i++;
724 }
725
726 Datagram datagram;
727 message.encode(datagram);
728 _writer.send(datagram, _tcp_connection, true);
729 }
730}
