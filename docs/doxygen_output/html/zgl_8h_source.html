254static inline int gl_clipcode(PN_stdfloat x,PN_stdfloat y,PN_stdfloat z,PN_stdfloat w1)
255{
256 PN_stdfloat w;
257
258 w = w1 * (1.0f + CLIP_EPSILON);
259 return (x<-w) |
260 ((x>w)<<1) |
261 ((y<-w)<<2) |
262 ((y>w)<<3) |
263 ((z<-w)<<4) |
264 ((z>w)<<5) ;
265}
