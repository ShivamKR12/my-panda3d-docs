460void GraphicsWindow::
461process_events() {
462 if (_requested_properties.is_any_specified()) {
463 // We don't bother to grab the mutex until after we have already checked
464 // whether any properties have been specified. This is technically
465 // sloppy, but it ought to be o.k. since it's just a bitmask after all.
466 WindowProperties properties;
467 {
468 LightReMutexHolder holder(_properties_lock);
469 properties = _requested_properties;
470 _requested_properties.clear();
471
472 set_properties_now(properties);
473 if (properties.is_any_specified()) {
474 display_cat.info()
475 << "Unable to set window properties: " << properties << "\n";
476 _rejected_properties.add_properties(properties);
477 }
478 }
479 }

496void GraphicsWindow::
497set_properties_now(WindowProperties &properties) {
498 if (properties.has_open() &&
499 properties.get_open() != _properties.get_open()) {
500 // Open or close a new window. In this case we can get all of the
501 // properties at once.
502
503 _properties.add_properties(properties);
504 properties.clear();
505
506 if (_properties.get_open()) {
507 if (open_window()) {
508 // When the window is first opened, force its size to be broadcast to
509 // its display regions.
510 _is_valid = true;
511 set_size_and_recalc(_properties.get_x_size(),
512 _properties.get_y_size());
513 } else {
514 // Since we can't even open the window, tag the _rejected_properties
515 // with all of the window properties that failed.
516 _rejected_properties.add_properties(_properties);
517
518 // And mark the window closed.
519 _properties.set_open(false);
520 _is_valid = false;
521 }
522
523 } else {
524 // We used to resist closing a window before its GSG has been released.
525 // Now it seems we never release a GSG, so go ahead and close the
526 // window.
527 close_window();
528 _is_valid = false;
529 }
530 return;
531 }
532
533 if (!_properties.get_open()) {
534 // The window is not currently open; we can set properties at will.
535 _properties.add_properties(properties);
536 properties.clear();
537 return;
538 }
539
540 properties.clear_open();
541
542 // The window is already open; we are limited to what we can change on the
543 // fly.
544
545 if (properties.has_size() || properties.has_origin()) {
546 // Consider changing the window's size andor position.
547 WindowProperties reshape_props;
548 if (properties.has_size()) {
549 reshape_props.set_size(properties.get_x_size(), properties.get_y_size());
550 } else {
551 reshape_props.set_size(_properties.get_x_size(), _properties.get_y_size());
552 }
553
554 if (properties.has_origin() && !is_fullscreen()) {
555 reshape_props.set_origin(properties.get_x_origin(), properties.get_y_origin());
556 } else if (_properties.has_origin()) {
557 reshape_props.set_origin(_properties.get_x_origin(), _properties.get_y_origin());
558 }
559
560 bool has_origin = reshape_props.has_origin();
561 int x_origin = 0, y_origin = 0;
562 if (has_origin) {
563 x_origin = reshape_props.get_x_origin();
564 y_origin = reshape_props.get_y_origin();
565 }
566
567 if (reshape_props.get_x_size() != _properties.get_x_size() ||
568 reshape_props.get_y_size() != _properties.get_y_size() ||
569 (has_origin && (x_origin != _properties.get_x_origin() ||
570 y_origin != _properties.get_y_origin()))) {
571 if (do_reshape_request(x_origin, y_origin, has_origin,
572 reshape_props.get_x_size(),
573 reshape_props.get_y_size())) {
574 properties.clear_size();
575 properties.clear_origin();
576 }
577 } else {
578 properties.clear_size();
579 properties.clear_origin();
580 }
581 }
582
583 if (properties.has_fullscreen() &&
584 properties.get_fullscreen() == _properties.get_fullscreen()) {
585 // Fullscreen property specified, but unchanged.
586 properties.clear_fullscreen();
587 }
588 if (properties.has_mouse_mode() &&
589 properties.get_mouse_mode() == _properties.get_mouse_mode()) {
590 // Mouse mode specified, but unchanged.
591 properties.clear_mouse_mode();
592 }
593}

672 if (display_cat.is_debug()) {

675 }
676
677 LightReMutexHolder holder(_properties_lock);
678
