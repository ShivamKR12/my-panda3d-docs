233void PStatClient::
234main_tick() {
235 // We have code here to report the memory usage. We can't put this code
236 // inside the MemoryUsage class, where it fits a little better, simply
237 // because MemoryUsage is a very low-level class that doesn't know about
238 // PStatClient.
239
240#ifdef DO_MEMORY_USAGE
241 if (is_connected()) {
242 _heap_total_size_pcollector.set_level(MemoryUsage::get_total_size());
243 _heap_overhead_size_pcollector.set_level(MemoryUsage::get_panda_heap_overhead());
244 _heap_single_size_pcollector.set_level(MemoryUsage::get_panda_heap_single_size());
245 _heap_array_size_pcollector.set_level(MemoryUsage::get_panda_heap_array_size());
246 _heap_external_size_pcollector.set_level(MemoryUsage::get_external_size());
247
248
249 _mmap_size_pcollector.set_level(MemoryUsage::get_panda_mmap_size());
250
251 TypeRegistry *type_reg = TypeRegistry::ptr();
252 int num_typehandles = type_reg->get_num_typehandles();
253
254 while ((int)type_handle_cols.size() < num_typehandles) {
255 type_handle_cols.push_back(TypeHandleCollector());
256 }
257
258 size_t single_total_usage = 0;
259 size_t array_total_usage = 0;
260 size_t dc_active_total_usage = 0;
261 size_t dc_inactive_total_usage = 0;
262 int i;
263 for (i = 0; i < num_typehandles; ++i) {
264 TypeHandle type = type_reg->get_typehandle(i);
265 for (int mi = 0; mi < (int)TypeHandle::MC_limit; ++mi) {
266 TypeHandle::MemoryClass mc = (TypeHandle::MemoryClass)mi;
267 size_t usage = type.get_memory_usage(mc);
268
269 switch (mc) {
270 case TypeHandle::MC_singleton:
271 single_total_usage += usage;
272 break;
273
274 case TypeHandle::MC_array:
275 array_total_usage += usage;
276 break;
277
278 case TypeHandle::MC_deleted_chain_active:
279 dc_active_total_usage += usage;
280 break;
281
282 case TypeHandle::MC_deleted_chain_inactive:
283 dc_inactive_total_usage += usage;
284 break;
285
286 case TypeHandle::MC_limit:
287 // Not used.
288 break;
289 }
290 }
291 }
292 size_t min_usage = (single_total_usage + array_total_usage + dc_active_total_usage + dc_inactive_total_usage) / 1024;
293 if (!pstats_mem_other) {
294 min_usage = 0;
295 }
296 size_t single_other_usage = single_total_usage;
297 size_t array_other_usage = array_total_usage;
298 size_t dc_active_other_usage = dc_active_total_usage;
299 size_t dc_inactive_other_usage = dc_inactive_total_usage;
300
301 for (i = 0; i < num_typehandles; ++i) {
302 TypeHandle type = type_reg->get_typehandle(i);
303 for (int mi = 0; mi < (int)TypeHandle::MC_limit; ++mi) {
304 TypeHandle::MemoryClass mc = (TypeHandle::MemoryClass)mi;
305 PStatCollector &col = type_handle_cols[i]._mem_class[mi];
306 size_t usage = type.get_memory_usage(mc);
307 if (usage > min_usage || col.is_valid()) {
308 // We have some memory usage on this TypeHandle. See if we have a
309 // collector for it.
310 if (!col.is_valid()) {
311 const char *category = "";
312 switch (mc) {
313 case TypeHandle::MC_singleton:
314 category = "Heap:Single";
315 break;
316
317 case TypeHandle::MC_array:
318 category = "Heap:Array";
319 break;
320
321 case TypeHandle::MC_deleted_chain_active:
322 category = "MMap:NeverFree:Active";
323 break;
324
325 case TypeHandle::MC_deleted_chain_inactive:
326 category = "MMap:NeverFree:Inactive";
327 break;
328
329 case TypeHandle::MC_limit:
330 // Not used.
331 break;
332 }
333 std::ostringstream strm;
334 strm << "System memory:" << category << ":" << type;
335 col = PStatCollector(strm.str());
336 }
337 col.set_level(usage);
338
339 switch (mc) {
340 case TypeHandle::MC_singleton:
341 single_other_usage -= usage;
342 break;
343
344 case TypeHandle::MC_array:
345 array_other_usage -= usage;
346 break;
347
348 case TypeHandle::MC_deleted_chain_active:
349 dc_active_other_usage -= usage;
350 break;
351
352 case TypeHandle::MC_deleted_chain_inactive:
353 dc_inactive_other_usage -= usage;
354 break;
355
356 case TypeHandle::MC_limit:
357 // Not used.
358 break;
359 }
360 }
361 }
362 }
363
364 _mmap_nf_unused_size_pcollector.set_level(NeverFreeMemory::get_total_unused());
365
366 // The remaining amount--all collectors smaller than 0.1% of the total--go
367 // into "other".
368 _heap_single_other_size_pcollector.set_level(single_other_usage);
369 _heap_array_other_size_pcollector.set_level(array_other_usage);
370 _mmap_dc_active_other_size_pcollector.set_level(dc_active_other_usage);
371 _mmap_dc_inactive_other_size_pcollector.set_level(dc_inactive_other_usage);
372 }
373#endif // DO_MEMORY_USAGE
374
375 get_global_pstats()->client_main_tick();
376}

400client_main_tick() {
401 ReMutexHolder holder(_lock);
402 if (has_impl()) {
403 if (!_impl->client_is_connected()) {
404 client_disconnect();
405 return;
406 }
407
408 ClockObject *clock = ClockObject::get_global_clock();
409
410 _impl->client_main_tick();
411
412 MultiThingsByName::const_iterator ni =
413 _threads_by_sync_name.find("Main");
414 if (ni != _threads_by_sync_name.end()) {
415 const vector_int &indices = (*ni).second;
416 for (vector_int::const_iterator vi = indices.begin();
417 vi != indices.end();
418 ++vi) {
419 InternalThread *thread = get_thread_ptr(*vi);
420 nassertd(thread != nullptr) continue;
421 _impl->new_frame(*vi, thread->_frame_number);
422 thread->_frame_number = clock->get_frame_count(get_thread_object(*vi));
423 }
424 }
425 }

476void PStatClient::
477client_disconnect() {
478 ReMutexHolder holder(_lock);
479 if (has_impl()) {
480 _impl->client_disconnect();
481 delete _impl;
482 _impl = nullptr;
483 }
484
485 // These can be relaxed loads because we hold the lock.
486 ThreadPointer *threads = _threads.load(std::memory_order_relaxed);
487 for (int ti = 0; ti < get_num_threads(); ++ti) {
488 InternalThread *thread = threads[ti];
489 if (thread != nullptr) {
490 thread->_frame_number = 0;
491 thread->_is_active = false;
492 thread->_next_packet = 0.0;
493 thread->_frame_data.clear();
494 }
495 }
496
497 CollectorPointer *collectors = _collectors.load(std::memory_order_relaxed);
498 for (int ci = 0; ci < get_num_collectors(); ++ci) {
499 Collector *collector = collectors[ci];
500 for (PerThreadData &per_thread : collector->_per_thread) {
501 per_thread._nested_count = 0;
502 }
503 }
504}

558PStatCollector PStatClient::
559make_collector_with_relname(int parent_index, string relname) {
560 ReMutexHolder holder(_lock);
561
562 if (relname.empty()) {
563 relname = "Unnamed";
564 }
565
566 // Skip any colons at the beginning of the name.
567 size_t start = 0;
568 while (start < relname.size() && relname[start] == ':') {
569 start++;
570 }
571
572 // If the name contains a colon (after the initial colon), it means we are
573 // making a nested collector.
574 size_t colon = relname.find(':', start);
575 while (colon != string::npos) {
576 string parent_name = relname.substr(start, colon - start);
577 PStatCollector parent_collector =
578 make_collector_with_name(parent_index, parent_name);
579 parent_index = parent_collector._index;
580 relname = relname.substr(colon + 1);
581 start = 0;
582 colon = relname.find(':');
583 }
584
585 string name = relname.substr(start);
586 return make_collector_with_name(parent_index, name);
587}

595PStatCollector PStatClient::
596make_collector_with_name(int parent_index, const string &name) {
597 ReMutexHolder holder(_lock);
598
599 int num_collectors = get_num_collectors();
600 nassertr(parent_index >= 0 && parent_index < num_collectors,
601 PStatCollector());
602
603 Collector *parent = get_collector_ptr(parent_index);
604
605 // A special case: if we asked for a child the same name as its parent, we
606 // really meant the parent. That is, "Frame:Frame" is really the same
607 // collector as "Frame".
608 if (parent->get_name() == name) {
609 return PStatCollector(this, parent_index);
610 }
611
612 ThingsByName::const_iterator ni = parent->_children.find(name);
613
614 if (ni != parent->_children.end()) {
615 // We already had a collector by this name; return it.
616 int index = (*ni).second;
617 nassertr(index >= 0 && index < num_collectors, PStatCollector());
618 return PStatCollector(this, (*ni).second);
619 }
620
621 // Create a new collector for this name.
622 parent->_children.insert(ThingsByName::value_type(name, num_collectors));
623
624 Collector *collector = new Collector(parent_index, name);
625 // collector->_def = new PStatCollectorDef(num_collectors, name);
626 // collector->_def->set_parent(*_collectors[parent_index]._def);
627 // initialize_collector_def(this, collector->_def);
628
629 // We need one PerThreadData for each thread.
630 while ((int)collector->_per_thread.size() < get_num_threads()) {
631 collector->_per_thread.push_back(PerThreadData());
632 }
633 add_collector(collector);
634
635 return PStatCollector(this, num_collectors);
636}

667PStatThread PStatClient::
668do_make_thread(Thread *thread) {
669 int thread_index = thread->get_pstats_index();
670 if (thread_index != -1) {
671 return PStatThread((PStatClient *)this, thread_index);
672 }
673
674 MultiThingsByName::const_iterator ni =
675 _threads_by_name.find(thread->get_name());
676
677 if (ni != _threads_by_name.end()) {
678 // We have seen a thread with this name before. Can we re-use any of
679 // them?
680 const vector_int &indices = (*ni).second;
681 for (vector_int::const_iterator vi = indices.begin();
682 vi != indices.end();
683 ++vi) {
684 int index = (*vi);
685 nassertr(index >= 0 && index < get_num_threads(), PStatThread());
686 ThreadPointer *threads = _threads.load(std::memory_order_relaxed);
687 InternalThread *pthread = threads[index];
688 if (pthread->_thread.was_deleted() &&
689 pthread->_sync_name == thread->get_sync_name()) {
690 // Yes, re-use this one.
691 pthread->_thread = thread;
692 thread->set_pstats_index(index);
693 thread->set_pstats_callback(this);
694 if (pthread->_sync_name == "Main") {
695 ClockObject *clock = ClockObject::get_global_clock();
696 pthread->_frame_number = clock->get_frame_count(thread);
697 }
698 return PStatThread(this, index);
699 }
700 }
701 }
702
703 // Create a new PStatsThread for this thread pointer.
704 int new_index = get_num_threads();
705 thread->set_pstats_index(new_index);
706 thread->set_pstats_callback(this);
707
708 InternalThread *pthread = new InternalThread(thread);
709 if (pthread->_sync_name == "Main") {
710 ClockObject *clock = ClockObject::get_global_clock();
711 pthread->_frame_number = clock->get_frame_count(thread);
712 }
713 add_thread(pthread);
714
715 return PStatThread(this, new_index);
716}

1112void PStatClient::
1113add_collector(PStatClient::Collector *collector) {
1114 int num_collectors = get_num_collectors();
1115 if (num_collectors >= (int)_collectors_size) {
1116 // We need to grow the array. We have to be careful here, because there
1117 // might be clients accessing the array right now who are not protected by
1118 // the lock.
1119 size_t new_collectors_size = (_collectors_size == 0) ? 128 : _collectors_size * 2;
1120 CollectorPointer *old_collectors = _collectors.load(std::memory_order_relaxed);
1121 CollectorPointer *new_collectors = new CollectorPointer[new_collectors_size];
1122 if (old_collectors != nullptr) {
1123 memcpy(new_collectors, old_collectors, num_collectors * sizeof(CollectorPointer));
1124 }
1125 _collectors_size = new_collectors_size;
1126 _collectors.store(new_collectors, std::memory_order_release);
1127
1128 // Now, we still have the old array, which we allow to leak. We should
1129 // delete it, but there might be a thread out there that's still trying to
1130 // access it, so we can't safely delete it; and it doesn't really matter
1131 // much, since it's not a big leak. (We will only reallocate the array so
1132 // many times in an application, and then no more.)
1133
1134 new_collectors[num_collectors] = collector;
1135 }
1136 else {
1137 CollectorPointer *collectors = _collectors.load(std::memory_order_relaxed);
1138 collectors[num_collectors] = collector;
1139 }
1140
1141 _num_collectors.fetch_add(1, std::memory_order_release);

1159 ThreadPointer *old_threads = _threads.load(std::memory_order_relaxed);
1160 ThreadPointer *new_threads = new ThreadPointer[new_threads_size];
1161 if (old_threads != nullptr) {
1162 memcpy(new_threads, old_threads, num_threads * sizeof(ThreadPointer));
1163 }

1245void PStatClient::
1246delete_hook(Thread *thread) {
1247 int thread_index = thread->get_pstats_index();
1248 if (thread_index < 0) {
1249 return;
1250 }
1251
1252 PStatClientImpl *impl;
1253 InternalThread *ithread;
1254 {
1255 ReMutexHolder holder(_lock);
1256 impl = _impl;
1257 if (impl == nullptr) {
1258 return;
1259 }
1260
1261 if (!impl->client_is_connected()) {
1262 return;
1263 }
1264
1265 MultiThingsByName::iterator ni;
1266
1267 ni = _threads_by_name.find(thread->get_name());
1268 if (ni != _threads_by_name.end()) {
1269 ni->second.erase(std::remove(ni->second.begin(), ni->second.end(), thread_index));
1270 }
1271
1272 ni = _threads_by_sync_name.find(thread->get_sync_name());
1273 if (ni != _threads_by_sync_name.end()) {
1274 ni->second.erase(std::remove(ni->second.begin(), ni->second.end(), thread_index));
1275 }
1276
1277 // This load can be relaxed because we hold the lock.
1278 ThreadPointer *threads = _threads.load(std::memory_order_relaxed);
1279 ithread = threads[thread_index];
1280 ithread->_is_active = false;
1281 ithread->_thread_active = false;
1282 threads[thread_index] = nullptr;
1283 }
1284
1285 impl->remove_thread(thread_index);
