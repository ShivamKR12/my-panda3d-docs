158void PStatReader::
159receive_datagram(const NetDatagram &datagram) {
160 Connection *connection = datagram.get_connection();
161
162 if (connection == _tcp_connection) {
163 PStatClientControlMessage message;
164 if (message.decode(datagram, _client_data)) {
165 handle_client_control_message(message);
166
167 } else if (message._type == PStatClientControlMessage::T_datagram) {
168 handle_client_udp_data(datagram);
169
170 } else {
171 nout << "Got unexpected message from client.\n";
172 }
173
174 } else if (connection == _udp_connection) {
175 handle_client_udp_data(datagram);
176
177 } else {
178 nout << "Got datagram from unexpected socket.\n";
179 }

186void PStatReader::
187handle_client_control_message(const PStatClientControlMessage &message) {
188 switch (message._type) {
189 case PStatClientControlMessage::T_hello:
190 {
191 _client_data->set_version(message._major_version, message._minor_version);
192 int server_major_version = get_current_pstat_major_version();
193 int server_minor_version = get_current_pstat_minor_version();
194
195 if (message._major_version != server_major_version ||
196 (message._major_version == server_major_version &&
197 message._minor_version > server_minor_version)) {
198 _monitor->bad_version(message._client_hostname, message._client_progname,
199 message._client_pid,
200 message._major_version, message._minor_version,
201 server_major_version, server_minor_version);
202 _monitor->close();
203 } else {
204 _monitor->hello_from(message._client_hostname, message._client_progname,
205 message._client_pid);
206 }
207 }
208 break;
209
210 case PStatClientControlMessage::T_define_collectors:
211 {
212 for (int i = 0; i < (int)message._collectors.size(); i++) {
213 _client_data->add_collector(message._collectors[i]);
214 _monitor->new_collector(message._collectors[i]->_index);
215 }
216 }
217 break;
218
219 case PStatClientControlMessage::T_define_threads:
220 {
221 // See if we can clean up old threads, so that we don't clutter up the
222 // view if we are creating many threads.
223 for (int thread_index = 0; thread_index < _client_data->get_num_threads(); ++thread_index) {
224 if (_client_data->has_thread(thread_index) && !_client_data->is_thread_alive(thread_index)) {
225 PStatThreadData *thread_data = (PStatThreadData *)_client_data->get_thread_data(thread_index);
226 if (thread_data->prune_history(_client_data->get_latest_time())) {
227 _client_data->remove_thread(thread_index);
228 _monitor->remove_thread(thread_index);
229 }
230 }
231 }
232
233 for (int i = 0; i < (int)message._names.size(); i++) {
234 int thread_index = message._first_thread_index + i;
235 std::string name = message._names[i];
236 _client_data->define_thread(thread_index, name, true);
237 _monitor->new_thread(thread_index);
238 }
239 }
240 break;
241
242 case PStatClientControlMessage::T_expire_thread:
243 if (_client_data->has_thread(message._first_thread_index)) {
244 // Remove the thread right away if it has no recent data.
245 PStatThreadData *thread_data = (PStatThreadData *)_client_data->get_thread_data(message._first_thread_index);
246 if (thread_data->prune_history(_client_data->get_latest_time())) {
247 _client_data->remove_thread(message._first_thread_index);
248 _monitor->remove_thread(message._first_thread_index);
249 } else {
250 // Otherwise, just mark it as expired, and we'll remove it later.
251 _client_data->expire_thread(message._first_thread_index);
252 }
253 }
254 break;
255
256 default:
257 nout << "Invalid control message received from client.\n";
258 }
259}

265void PStatReader::
266handle_client_udp_data(const Datagram &datagram) {
267 if (!_monitor->is_client_known()) {
268 // If we haven't heard a "hello" from the client yet, we don't know what
269 // version data it will be sending us, so we can't decode the data.
270 // Chances are good we can't display it sensibly yet anyway. Ignore frame
271 // data until we get that hello.
272 return;
273 }
274
275 DatagramIterator source(datagram);
276
277 if (_client_data->is_at_least(2, 1)) {
278 // Throw away the zero byte at the beginning.
279 int initial_byte = source.get_uint8();
280 nassertv(initial_byte == 0);
281 }
282
283 if (!_queued_frame_data.full()) {
284 FrameData data;
285 data._thread_index = source.get_uint16();
286 data._frame_number = source.get_uint32();
287 data._frame_data = new PStatFrameData;
288 data._frame_data->read_datagram(source, _client_data);
289
290 // Queue up the data till we're ready to handle it in a single-threaded
291 // way.
292 _queued_frame_data.push_back(data);
293 }
294}

300void PStatReader::
301dequeue_frame_data() {
302 if (_queued_frame_data.empty()) {
303 return;
304 }
305
306 do {
307 const FrameData &data = _queued_frame_data.front();
308 nassertv(_client_data != nullptr);
309
310 // Check to see if any new collectors have level data.
311 int num_levels = data._frame_data->get_num_levels();
312 for (int i = 0; i < num_levels; i++) {
313 int collector_index = data._frame_data->get_level_collector(i);
314 if (!_client_data->get_collector_has_level(collector_index, data._thread_index)) {
315 // This collector is now reporting level data, and it wasn't before.
316 _client_data->set_collector_has_level(collector_index, data._thread_index, true);
317 _monitor->new_collector(collector_index);
318 }
319 }
320
321 _client_data->record_new_frame(data._thread_index,
322 data._frame_number,
323 data._frame_data);
324 _monitor->new_data(data._thread_index, data._frame_number);
325
326 _queued_frame_data.pop_front();
327 }
328 while (!_queued_frame_data.empty());
329
330 // Clean up old threads.
331 for (int thread_index = 0; thread_index < _client_data->get_num_threads(); ++thread_index) {
332 if (_client_data->has_thread(thread_index) && !_client_data->is_thread_alive(thread_index)) {
333 PStatThreadData *thread_data = (PStatThreadData *)_client_data->get_thread_data(thread_index);
334 if (thread_data->prune_history(_client_data->get_latest_time())) {
335 _client_data->remove_thread(thread_index);
336 _monitor->remove_thread(thread_index);
337 }
338 }
339 }
