106bool EggData::
107read(istream &in) {
108 // First, dispense with any children we had previously. We will replace
109 // them with the new data.
110 clear();
111
112 // Create a temporary EggData structure to read into. We initialize it with
113 // a copy of ourselves, so that it will get our _coordsys value, if the user
114 // set it.
115 PT(EggData) data = new EggData(*this);
116
117 bool success;
118 {
119 EggLexerState lexer_state;
120 egg_init_lexer_state(lexer_state, in, get_egg_filename());
121 success = egg_parse(lexer_state, data, data);
122 egg_cleanup_lexer_state(lexer_state);
123 }
124
125 data->post_read();
126
127 steal_children(*data);
128 (*this) = *data;
129
130 return success;
131}

251void EggData::
252set_coordinate_system(CoordinateSystem new_coordsys) {
253 if (new_coordsys == CS_default) {
254 new_coordsys = get_default_coordinate_system();
255 }
256 if (new_coordsys != _coordsys &&
257 (_coordsys != CS_default && _coordsys != CS_invalid)) {
258 // Time to convert the data.
259 LMatrix4d mat = LMatrix4d::convert_mat(_coordsys, new_coordsys);
260 LMatrix4d inv = LMatrix4d::convert_mat(new_coordsys, _coordsys);
261
262 r_transform(mat, inv, new_coordsys);
263 r_transform_vertices(mat);
264
265 // Now we have to update the under_flags to ensure that all the cached
266 // relative matrices are correct.
267 update_under(0);
268 }
269
270 _coordsys = new_coordsys;
271}

289void EggData::
290post_read() {
291 CoordinateSystem old_coordsys = _coordsys;
292 _coordsys = find_coordsys_entry();
293
294 if (_coordsys == CS_default) {
295 // If the egg file didn't contain a <CoordinateSystem> entry, assume it's
296 // Y-up, by convention.
297 _coordsys = CS_yup_right;
298
299 } else if (_coordsys == CS_invalid) {
300 egg_cat.warning()
301 << "Contradictory <CoordinateSystem> entries encountered.\n";
302 _coordsys = CS_yup_right;
303 }
304
305 r_mark_coordsys(_coordsys);
306
307 if (old_coordsys != CS_default) {
308 // Now if we had a previous definition, enforce it. This might convert
309 // the data to the given coordinate system.
310 set_coordinate_system(old_coordsys);
311 }
312
313 // Fill this in before we automatically resolve pathnames.
314 _had_absolute_pathnames = has_absolute_pathnames();
315
316 if (get_auto_resolve_externals()) {
317 // Resolve filenames that are relative to the egg file.
318 DSearchPath dir;
319 dir.append_directory(get_egg_filename().get_dirname());
320 resolve_filenames(dir);
321 }

329void EggData::
330pre_write() {
331 // Pull out all of the texture definitions in the file and massage them a
332 // bit.
333 EggTextureCollection textures;
334 textures.extract_textures(this);
335
336 // Remove any textures that aren't being used.
337 textures.remove_unused_textures(this);
338
339 // Collapse out any textures that are completely equivalent. For this
340 // purpose, we consider two textures with identical properties but different
341 // tref names to be different.
342 textures.collapse_equivalent_textures(~0, this);
343
344 // Make sure all of the textures have unique TRef names.
345 textures.uniquify_trefs();
346 textures.sort_by_tref();
347
348 // Do the same thing with the materials.
349 EggMaterialCollection materials;
350 materials.extract_materials(this);
351 materials.remove_unused_materials(this);
352 materials.collapse_equivalent_materials(~0, this);
353 materials.uniquify_mrefs();
354 materials.sort_by_mref();
355
356 // Now put them all back at the head of the file, after any initial comment
357 // records.
358 iterator ci = begin();
359 while (ci != end() && (*ci)->is_of_type(EggComment::get_class_type())) {
360 ++ci;
361 }
362 textures.insert_textures(this, ci);
363 materials.insert_materials(this, ci);
364
365 // Also make sure that the vertex pools are uniquely named. This also
366 // checks textures and materials, which is kind of redundant since we just
367 // did that, but we don't mind.
368 EggPoolUniquifier pu;
369 pu.uniquify(this);
370}
