202void WinStatsLabelStack::
203replace_labels(WinStatsMonitor *monitor, WinStatsGraph *graph,
204 int thread_index, const vector_int &collector_indices,
205 bool use_fullname) {
206
207 _ideal_width = 0;
208
209 // First skip the part of the stack that hasn't changed.
210 size_t li = 0;
211 size_t ci = 0;
212 while (ci < collector_indices.size() && li < _labels.size()) {
213 WinStatsLabel *label = _labels[li];
214 if (collector_indices[ci] != label->get_collector_index()) {
215 // Mismatch.
216 break;
217 }
218 _ideal_width = std::max(_ideal_width, label->get_ideal_width());
219 ++ci;
220 ++li;
221 }
222
223 if (ci == collector_indices.size()) {
224 if (ci == _labels.size()) {
225 // Perfect, nothing changed.
226 return;
227 }
228
229 // Simple case, just delete the rest.
230 while (li < _labels.size()) {
231 delete _labels[li++];
232 }
233 _labels.resize(ci);
234 return;
235 }
236
237 int yp = _height;
238 if (li > 0) {
239 WinStatsLabel *label = _labels[li - 1];
240 yp = label->get_y() - label->get_height();
241 }
242
243 // Make a map of remaining labels.
244 std::map<int, WinStatsLabel *> label_map;
245 for (size_t li2 = li; li2 < _labels.size(); ++li2) {
246 WinStatsLabel *label = _labels[li2];
247 label_map[label->get_collector_index()] = label;
248 }
249
250 _labels.resize(collector_indices.size());
251
252 while (ci < collector_indices.size()) {
253 int collector_index = collector_indices[ci++];
254
255 WinStatsLabel *label;
256 auto it = label_map.find(collector_index);
257 if (it == label_map.end()) {
258 // It's not in the map. Create a new label.
259 label = new WinStatsLabel(monitor, graph, thread_index, collector_index, use_fullname);
260 if (_window) {
261 label->setup(_window);
262 }
263 } else {
264 // Erase it from the map, so that it's not deleted.
265 label = it->second;
266 label_map.erase(it);
267 }
268 if (_window) {
269 label->set_pos(0, yp - _scroll, _width);
270 }
271 _ideal_width = std::max(_ideal_width, label->get_ideal_width());
272 yp -= label->get_height();
273
274 _labels[li++] = label;
275 }
276
277 // Anything that's remaining in the label map should be deleted.
278 for (auto it = label_map.begin(); it != label_map.end(); ++it) {
279 delete it->second;
280 }
281
282 recalculate_label_positions();
283}

324void WinStatsLabelStack::
325recalculate_label_positions() {
326 int total_height = 0;
327 for (WinStatsLabel *label : _labels) {
328 total_height += label->get_height();
329 }
330 total_height += _bottom_margin + _top_margin;
331 int yp;
332 if (total_height <= _height) {
333 // Fits. Align to bottom and reset scroll.
334 yp = _height - _bottom_margin;
335 _scroll = 0;
336 } else {
337 // Doesn't fit. Align to top.
338 yp = total_height - _bottom_margin;
339 _scroll = (std::min)(_scroll, total_height - _height);
340 _scroll = (std::max)(_scroll, 0);
341 }
342 for (WinStatsLabel *label : _labels) {
343 label->set_pos(0, yp - _scroll, _width);
344 yp -= label->get_height();
345 }
