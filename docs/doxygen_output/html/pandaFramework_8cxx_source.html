301void PandaFramework::
302define_key(const string &event_name, const string &description,
303 EventHandler::EventCallbackFunction *function,
304 void *data) {
305 if (_event_handler.has_hook(event_name)) {
306 // If there is already a hook for the indicated keyname, we're most likely
307 // replacing a previous definition of a key. Search for the old
308 // definition and remove it.
309 KeyDefinitions::iterator di;
310 di = _key_definitions.begin();
311 while (di != _key_definitions.end() && (*di)._event_name != event_name) {
312 ++di;
313 }
314 if (di != _key_definitions.end()) {
315 _key_definitions.erase(di);
316 }
317 }
318
319 // Now add a new hook for the keyname, and also add the new description.
320 _event_handler.add_hook(event_name, function, data);
321
322 if (!description.empty()) {
323 KeyDefinition keydef;
324 keydef._event_name = event_name;
325 keydef._description = description;
326 _key_definitions.push_back(keydef);
327 }

384WindowFramework *PandaFramework::
385open_window() {
386 GraphicsPipe *pipe = get_default_pipe();
387 if (pipe == nullptr) {
388 // Can't get a pipe.
389 return nullptr;
390 }
391
392 WindowFramework *wf = open_window(pipe, nullptr);
393 if (wf == nullptr) {
394 // Ok, the default graphics pipe failed; try a little harder.
395 GraphicsPipeSelection *selection = GraphicsPipeSelection::get_global_ptr();
396 selection->load_aux_modules();
397
398 int num_pipe_types = selection->get_num_pipe_types();
399 for (int i = 0; i < num_pipe_types; i++) {
400 TypeHandle pipe_type = selection->get_pipe_type(i);
401 if (pipe_type != _default_pipe->get_type()) {
402 PT(GraphicsPipe) new_pipe = selection->make_pipe(pipe_type);
403 if (new_pipe != nullptr) {
404 wf = open_window(new_pipe, nullptr);
405 if (wf != nullptr) {
406 // Here's the winner!
407 _default_pipe = new_pipe;
408 return wf;
409 }
410 }
411 }
412 }
413
414 // Too bad; none of the pipes could open a window. Fall through and
415 // return NULL.
416 }
417
418 return wf;
419}

447WindowFramework *PandaFramework::
448open_window(const WindowProperties &props, int flags,
449 GraphicsPipe *pipe, GraphicsStateGuardian *gsg) {
450 if (pipe == nullptr) {
451 pipe = get_default_pipe();
452 if (pipe == nullptr) {
453 // Can't get a pipe.
454 return nullptr;
455 }
456 }
457
458 nassertr(_is_open, nullptr);
459 PT(WindowFramework) wf = make_window_framework();
460 wf->set_wireframe(get_wireframe());
461 wf->set_texture(get_texture());
462 wf->set_two_sided(get_two_sided());
463 wf->set_lighting(get_lighting());
464 wf->set_perpixel(get_perpixel());
465 wf->set_background_type(get_background_type());
466
467 GraphicsOutput *win = wf->open_window(props, flags, get_graphics_engine(),
468 pipe, gsg);
469 _engine->open_windows();
470 if (win != nullptr && !win->is_valid()) {
471 // The window won't open.
472 _engine->remove_window(win);
473 wf->close_window();
474 win = nullptr;
475 }
476
477 if (win == nullptr) {
478 // Oops, couldn't make a window or buffer.
479 framework_cat.error()
480 << "Unable to create window.\n";
481 return nullptr;
482 }
483
484 _windows.push_back(wf);
485 return wf;

597void PandaFramework::
598report_frame_rate(std::ostream &out) const {
599 double now = ClockObject::get_global_clock()->get_frame_time();
600 double delta = now - _start_time;
601
602 int frame_count = ClockObject::get_global_clock()->get_frame_count();
603 int num_frames = frame_count - _frame_count;
604 if (num_frames > 0) {
605 out << num_frames << " frames in " << delta << " seconds.\n";
606 double fps = ((double)num_frames) / delta;
607 out << fps << " fps average (" << 1000.0 / fps << "ms)\n";
608 }

708int PandaFramework::
709hide_collision_solids(NodePath node) {
710 int num_changed = 0;
711
712 if (node.node()->is_of_type(CollisionNode::get_class_type()) ||
713 node.node()->is_of_type(OccluderNode::get_class_type())) {
714 if (!node.is_hidden()) {
715 node.hide();
716 num_changed++;
717 }
718 }
719
720 int num_children = node.get_num_children();
721 for (int i = 0; i < num_children; i++) {
722 num_changed += hide_collision_solids(node.get_child(i));
723 }
724
725 return num_changed;

911void PandaFramework::
912event_esc(const Event *event, void *data) {
913 if (event->get_num_parameters() == 1) {
914 EventParameter param = event->get_parameter(0);
915 WindowFramework *wf;
916 DCAST_INTO_V(wf, param.get_ptr());
917
918 PT(GraphicsOutput) win = wf->get_graphics_output();
919
920 PandaFramework *self = (PandaFramework *)data;
921 self->close_window(wf);
922
923 // Also close any other WindowFrameworks on the same window.
924 int window_index = self->find_window(win);
925 while (window_index != -1) {
926 self->close_window(window_index);
927 window_index = self->find_window(win);
928 }
929
930 // Free up the mouse for that window.
931 self->remove_mouse(win);
932
933 // Make sure the close request propagates through the system.
934 self->_engine->open_windows();
935
936 // If we closed the last window, shut down.
937 if (self->all_windows_closed()) {
938 self->_exit_flag = true;
939 }
940 }

1378void PandaFramework::
1379event_window_event(const Event *event, void *data) {
1380 PandaFramework *self = (PandaFramework *)data;
1381 if (event->get_num_parameters() == 1) {
1382 // The parameter of the window event is the window itself, rather than the
1383 // window framework object (which is the parameter of all of the keyboard
1384 // events).
1385 EventParameter param = event->get_parameter(0);
1386 const GraphicsOutput *win;
1387 DCAST_INTO_V(win, param.get_ptr());
1388
1389 // Is this a window we've heard about?
1390 int window_index = self->find_window(win);
1391 if (window_index == -1) {
1392 if (framework_cat.is_debug()) {
1393 framework_cat.debug()
1394 << "Ignoring message from unknown window.\n";
1395 }
1396
1397 } else {
1398 if (!win->is_valid()) {
1399 // The window has been closed.
1400 int window_index = self->find_window(win);
1401 while (window_index != -1) {
1402 self->close_window(window_index);
1403 window_index = self->find_window(win);
1404 }
1405
1406 // Free up the mouse for that window.
1407 self->remove_mouse(win);
1408
1409 // If the last window was closed, exit the application.
1410 if (self->all_windows_closed() && !self->_exit_flag) {
1411 framework_cat.info()
1412 << "Last window was closed by user.\n";
1413 self->_exit_flag = true;
1414 }
1415 } else {
1416 // Adjust aspect ratio.
1417 for (int n = 0; n < (int)self->_windows.size(); n++) {
1418 if (self->_windows[n]->get_graphics_output() == win) {
1419 self->_windows[n]->adjust_dimensions();
1420 }
1421 }
1422 }
1423 }
1424 }
