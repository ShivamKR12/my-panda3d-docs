140bool OccluderNode::
141cull_callback(CullTraverser *trav, CullTraverserData &data) {
142 // Normally, an OccluderNode is invisible. But if someone shows it, we will
143 // draw a visualization, a checkerboard-textured polygon.
144 trav->get_cull_handler()->record_object(CullableObject(
145 get_occluder_viz(trav, data), get_occluder_viz_state(trav, data),
146 data.get_internal_transform(trav)), trav);
147
148 // Also get the frame.
149 nassertr(_frame_viz != nullptr, false);
150 trav->get_cull_handler()->record_object(CullableObject(
151 _frame_viz, get_frame_viz_state(trav, data),
152 data.get_internal_transform(trav)), trav);
153
154 // Now carry on to render our child nodes.
155 return true;
156}

173void OccluderNode::
174compute_internal_bounds(CPT(BoundingVolume) &internal_bounds,
175 int &internal_vertices,
176 int pipeline_stage,
177 Thread *current_thread) const {
178 // First, get ourselves a fresh, empty bounding volume.
179 PT(BoundingVolume) bound = new BoundingSphere;
180 GeometricBoundingVolume *gbv = DCAST(GeometricBoundingVolume, bound);
181
182 // Now actually compute the bounding volume by putting it around all of our
183 // vertices.
184 if (!_vertices.empty()) {
185 const LPoint3 *vertices_begin = &_vertices[0];
186 const LPoint3 *vertices_end = vertices_begin + _vertices.size();
187 gbv->around(vertices_begin, vertices_end);
188 }
189
190 internal_bounds = bound;
191 internal_vertices = 0;
192}

198PT(Geom) OccluderNode::
199get_occluder_viz(CullTraverser *trav, CullTraverserData &data) {
200 if (_occluder_viz == nullptr) {
201 nassertr(_vertices.size() == 4, nullptr);
202
203 if (pgraph_cat.is_debug()) {
204 pgraph_cat.debug()
205 << "Recomputing viz for " << *this << "\n";
206 }
207
208 PT(GeomVertexData) vdata = new GeomVertexData
209 (get_name(), GeomVertexFormat::get_v3n3t2(), Geom::UH_static);
210
211 // Compute the polygon normal from the first three vertices.
212 LPlane plane(_vertices[0], _vertices[1], _vertices[2]);
213 LVector3 poly_normal = plane.get_normal();
214
215 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
216 GeomVertexWriter normal(vdata, InternalName::get_normal());
217 GeomVertexWriter texcoord(vdata, InternalName::get_texcoord());
218 vertex.add_data3(_vertices[0]);
219 normal.add_data3(poly_normal);
220 texcoord.add_data2(0.0, 0.0);
221
222 vertex.add_data3(_vertices[1]);
223 normal.add_data3(poly_normal);
224 texcoord.add_data2(8.0, 0.0);
225
226 vertex.add_data3(_vertices[2]);
227 normal.add_data3(poly_normal);
228 texcoord.add_data2(8.0, 8.0);
229
230 vertex.add_data3(_vertices[3]);
231 normal.add_data3(poly_normal);
232 texcoord.add_data2(0.0, 8.0);
233
234 PT(GeomTriangles) triangles = new GeomTriangles(Geom::UH_static);
235 triangles->add_vertices(0, 1, 2);
236 triangles->close_primitive();
237 triangles->add_vertices(0, 2, 3);
238 triangles->close_primitive();
239
240 _occluder_viz = new Geom(vdata);
241 _occluder_viz->add_primitive(triangles);
242
243 PT(GeomLinestrips) lines = new GeomLinestrips(Geom::UH_static);
244 lines->add_vertices(0, 1, 2, 3);
245 lines->add_vertex(0);
246 lines->close_primitive();
247
248 _frame_viz = new Geom(vdata);
249 _frame_viz->add_primitive(lines);
250 }
251
252 return _occluder_viz;
253}

258CPT(RenderState) OccluderNode::
259get_occluder_viz_state(CullTraverser *trav, CullTraverserData &data) {
260 if (_viz_tex == nullptr) {
261 // Create a default texture. We set it up as a 2x2 graytone checkerboard,
262 // since that's real easy, and it doesn't look like a CollisionPolygon.
263 _viz_tex = new Texture("occluder_viz");
264 _viz_tex->setup_2d_texture(2, 2, Texture::T_unsigned_byte, Texture::F_luminance);
265 PTA_uchar image;
266 image.set_data("\x20\x80\x80\x20");
267 _viz_tex->set_ram_image(image);
268 _viz_tex->set_minfilter(SamplerState::FT_nearest);
269 _viz_tex->set_magfilter(SamplerState::FT_nearest);
270 }
271
272 static CPT(RenderState) viz_state;
273 if (viz_state == nullptr) {
274 viz_state = RenderState::make
275 (ColorAttrib::make_flat(LVecBase4(1.0f, 1.0f, 1.0f, 0.5f)),
276 TransparencyAttrib::make(TransparencyAttrib::M_alpha),
277 DepthOffsetAttrib::make(),
278 TextureAttrib::make(_viz_tex));
279 viz_state = viz_state->adjust_all_priorities(1);
280 }
281
282 CPT(RenderState) state = viz_state;
283 if (is_double_sided()) {
284 state = viz_state->set_attrib(CullFaceAttrib::make(CullFaceAttrib::M_cull_none), 1);
285 } else {
286 state = viz_state->set_attrib(CullFaceAttrib::make(CullFaceAttrib::M_cull_clockwise), 1);
287 }
288
289 return state->compose(viz_state);
290}
