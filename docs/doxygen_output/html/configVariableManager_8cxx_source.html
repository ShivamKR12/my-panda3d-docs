49make_variable(const string &name) {
50 VariablesByName::iterator ni;
51 ni = _variables_by_name.find(name);
52 if (ni != _variables_by_name.end()) {
53 return (*ni).second;
54 }
55
56 ConfigVariableCore *variable = nullptr;
57
58 // See if there's a template that matches this name.
59 VariableTemplates::const_iterator ti;
60 for (ti = _variable_templates.begin();
61 ti != _variable_templates.end() && variable == nullptr;
62 ++ti) {
63 const GlobPattern &pattern = (*ti).first;
64 ConfigVariableCore *templ = (*ti).second;
65 if (pattern.matches(name)) {
66 variable = new ConfigVariableCore(*templ, name);

68 }

70 if (variable == nullptr) {

72 }

107 VariableTemplates::const_iterator ti = _variable_templates.find(gp);
108 if (ti != _variable_templates.end()) {
109 core = (*ti).second;
110
111 } else {
112 core = new ConfigVariableCore(pattern);
113 _variable_templates[gp] = core;
114 }
115
116 if (value_type != ConfigFlags::VT_undefined) {
117 core->set_value_type(value_type);
118 }
119 if (!default_value.empty() ||
120 core->get_default_value() == nullptr) {
121 core->set_default_value(default_value);
122 }
123 if (!description.empty()) {
124 core->set_description(description);
125 }
126 if (flags != 0) {
127 core->set_flags(flags);
128 }
129 core->set_used();
130
131 // Also apply the same changes to any previously-defined variables that
132 // match the pattern.
133 Variables::iterator vi;
134 for (vi = _variables.begin(); vi != _variables.end(); ++vi) {
135 ConfigVariableCore *variable = (*vi);
136 if (gp.matches(variable->get_name())) {
137 if (value_type != ConfigFlags::VT_undefined) {
138 variable->set_value_type(value_type);
139 }
140 if (!default_value.empty() ||
141 variable->get_default_value() == nullptr) {
142 variable->set_default_value(default_value);
143 }
144 if (!description.empty()) {
145 variable->set_description(description);

313void ConfigVariableManager::
314list_variable(const ConfigVariableCore *variable,
315 bool include_descriptions) const {
316 if (!variable->is_used()) {
317 // If the variable is unused, just show its name.
318 nout << variable->get_name() << " not used";
319 if (variable->get_num_references() > 0) {
320 nout << " (referenced in "
321 << variable->get_reference(0)->get_page()->get_name()
322 << ")";
323 }
324 nout << "\n";
325
326 } else {
327 // If the variable is used--it's been defined somewhere--show its name,
328 // its type, its current and default values, and if available, its
329 // description.
330
331 nout << variable->get_name() << " "
332 << variable->get_value_type() << "\n";
333
334 const ConfigDeclaration *decl;
335
336 if (variable->get_value_type() == ConfigVariableCore::VT_list ||
337 variable->get_value_type() == ConfigVariableCore::VT_search_path) {
338 // We treat a "list" variable as a special case: list all of its values.
339 nout << " current value:\n";
340 size_t num_references = variable->get_num_trusted_references();
341 for (size_t i = 0; i < num_references; ++i) {
342 decl = variable->get_trusted_reference(i);
343 nout << " " << decl->get_string_value()
344 << " (from " << decl->get_page()->get_name() << ")\n";
345 }
346
347 } else {
348 // An ordinary, non-list variable gets one line for its current value
349 // (if it has one) and another line for its default value.
350 decl = variable->get_declaration(0);
351 if (decl != variable->get_default_value()) {
352 nout << " current value = " << decl->get_string_value();
353 if (!decl->get_page()->is_special()) {
354 nout << " (from " << decl->get_page()->get_name() << ")\n";
355 } else {
356 nout << " (defined locally)\n";
357 }
358 }
359
360 decl = variable->get_default_value();
361 if (decl != nullptr) {
362 nout << " default value = " << decl->get_string_value() << "\n";
363 }
364 }
365
366 if (!variable->get_description().empty() && include_descriptions) {
367 nout << " " << variable->get_description() << "\n";
368 }
369 }
370
371 nout << "\n";
372}
