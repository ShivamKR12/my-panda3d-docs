30bool Socket_Address::
31set_host(const std::string &hostname, unsigned short port) {
32 // hmm inet_addr4 does not resolve 255.255.255.255 on ME98 ?? * HACK * ??
33 if (hostname == "255.255.255.255") {
34 return set_broadcast(port);
35 }
36
37 struct addrinfo hints, *res = nullptr;
38 memset(&hints, 0, sizeof(hints));
39
40
41 hints.ai_flags = AI_ADDRCONFIG;
42 hints.ai_family = support_ipv6 ? AF_UNSPEC : AF_INET;
43
44 if (getaddrinfo(hostname.c_str(), nullptr, &hints, &res)) {
45 return false;
46 }
47
48 nassertr(res->ai_addrlen <= sizeof(_storage), false);
49 memcpy(&_storage, res->ai_addr, res->ai_addrlen);
50 freeaddrinfo(res);
51
52 _addr4.sin_port = htons(port);
53 return true;
54}

60bool Socket_Address::
61set_host(const std::string &hostname) {
62 std::string::size_type pos = hostname.rfind(':');
63 if (pos == std::string::npos) {
64 return false;
65 }
66
67 std::string::size_type host_begin = 0;
68 std::string::size_type host_end = pos;
69
70 // Strip spaces.
71 while (host_begin < host_end && isspace(hostname[host_begin])) {
72 ++host_begin;
73 }
74
75 while (host_begin < host_end && isspace(hostname[host_end - 1])) {
76 --host_end;
77 }
78
79 if (host_begin < host_end && hostname[host_begin] == '[') {
80 // Looks like an IPv6 address; extract from the brackets.
81 host_begin += 1;
82 if (hostname[host_end - 1] == ']') {
83 host_end -= 1;
84 } else {
85 return false;
86 }
87 }
88
89 std::string host = hostname.substr(host_begin, host_end - host_begin);
90 std::string port = hostname.substr(pos + 1, 100);
91
92 unsigned short port_dig = (unsigned short)atoi(port.c_str());
93 return set_host(host, port_dig);
94}

121std::string Socket_Address::
122get_ip_port() const {
123 char buf[100]; // 100 is more than enough for any ip address:port combo..
124 buf[0] = 0;
125
126 if (_storage.ss_family == AF_INET) {
127 getnameinfo(&_addr, sizeof(sockaddr_in), buf, sizeof(buf), nullptr, 0, NI_NUMERICHOST);
128 sprintf(buf + strlen(buf), ":%hu", get_port());
129
130 } else if (_storage.ss_family == AF_INET6) {
131 // Protect the IPv6 address within square brackets.
132 buf[0] = '[';
133 getnameinfo(&_addr, sizeof(sockaddr_in6), buf + 1, sizeof(buf) - 1, nullptr, 0, NI_NUMERICHOST);
134 sprintf(buf + strlen(buf), "]:%hu", get_port());
135
136 } else {
137 nassert_raise("unsupported address family");
138 }
139
140 return std::string(buf);
141}

147unsigned long Socket_Address::
148GetIPAddressRaw() const {
149 if (_addr.sa_family == AF_INET) {
150 return _addr4.sin_addr.s_addr;
151 }
152 if (_addr.sa_family == AF_INET6) {
153 // Okay, if we got here, something probably went wrong, but let's see if
154 // we can offer a meaningful translation for mapped addresses.
155 uint32_t *parts = (uint32_t *)&_addr6.sin6_addr;
156 if (parts[0] == 0 && parts[1] == 0 && (parts[2] == 0 || parts[2] == htonl(0xffff))) {
157 if (parts[2] == 0 && parts[3] == htonl(1)) {
158 // Special exception for localhost.
159 return 0x1000007f;
160 } else {
161 return parts[3];
162 }
163 }
164 }
165 nassert_raise("GetIPAddressRaw() can only be called on an IPv4 address");
166 return 0;
