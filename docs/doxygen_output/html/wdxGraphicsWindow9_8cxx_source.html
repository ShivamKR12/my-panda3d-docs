169int wdxGraphicsWindow9::
170verify_window_sizes(int numsizes, int *dimen) {
171 // unfortunately this only works AFTER you make the window initially, so its
172 // really mostly useful for resizes only
173 nassertr(IS_VALID_PTR(_dxgsg), 0);
174
175 int num_valid_modes = 0;
176
177 wdxGraphicsPipe9 *dxpipe;
178 DCAST_INTO_R(dxpipe, _pipe, 0);
179
180 // not requesting same refresh rate since changing res might not support
181 // same refresh rate at new size
182
183 int *pCurDim = dimen;
184
185 for (int i = 0; i < numsizes; i++, pCurDim += 2) {
186 int x_size = pCurDim[0];
187 int y_size = pCurDim[1];
188
189 bool bIsGoodMode = false;
190 bool CouldntFindAnyValidZBuf;
191 D3DFORMAT newPixFmt = D3DFMT_UNKNOWN;
192
193 if (dxpipe->special_check_fullscreen_resolution(_wcontext, x_size, y_size)) {
194 // bypass the test below for certain cards we know have valid modes
195 bIsGoodMode = true;
196
197 } else {
198 if (_wcontext._is_low_memory_card) {
199 bIsGoodMode = ((x_size == 640) && (y_size == 480));
200 } else {
201 dxpipe->search_for_valid_displaymode
202 (_wcontext, x_size, y_size, _wcontext._presentation_params.EnableAutoDepthStencil != false,
203 IS_STENCIL_FORMAT(_wcontext._presentation_params.AutoDepthStencilFormat),
204 &_wcontext._supported_screen_depths_mask,
205 &CouldntFindAnyValidZBuf, &newPixFmt, dx_force_16bpp_zbuffer, true);
206 bIsGoodMode = (newPixFmt != D3DFMT_UNKNOWN);
207 }
208 }
209
210 if (bIsGoodMode) {
211 num_valid_modes++;
212 } else {
213 // tell caller the mode is invalid
214 pCurDim[0] = 0;
215 pCurDim[1] = 0;
216 }
217
218 if (wdxdisplay9_cat.is_spam()) {
219 wdxdisplay9_cat.spam()
220 << "Fullscrn Mode (" << x_size << ", " << y_size << ")\t"
221 << (bIsGoodMode ? "V" : "Inv") << "alid\n";
222 }
223 }
224
225 return num_valid_modes;
226}

254bool wdxGraphicsWindow9::
255open_window() {
256 PT(DXGraphicsDevice9) dxdev;
257 WindowProperties props;
258
259 // For now, let's make this configurable. If this is true, then you can't
260 // open multiple different windows with the same GSG, but you may have more
261 // luck opening different windows with different GSG's.
262 static ConfigVariableBool always_discard_device("always-discard-device", true);
263 bool discard_device = always_discard_device;
264
265 if (_gsg == nullptr) {
266 _dxgsg = new DXGraphicsStateGuardian9(_engine, _pipe);
267 _gsg = _dxgsg;
268 } else {
269 DCAST_INTO_R(_dxgsg, _gsg, false);
270 }
271
272 if (!choose_device()) {
273 return false;
274 }
275
276 // Ensure the window properties get set to the actual size of the window.
277 {
278 WindowProperties resized_props;
279 resized_props.set_size(_wcontext._display_mode.Width,
280 _wcontext._display_mode.Height);
281 _properties.add_properties(resized_props);
282 }
283
284 if (wdxdisplay9_cat.is_debug()) {
285 wdxdisplay9_cat.debug() << "_wcontext._window is " << _wcontext._window << "\n";
286 }
287 if (!WinGraphicsWindow::open_window()) {
288 return false;
289 }
290 _wcontext._window = _hWnd;
291
292 if (wdxdisplay9_cat.is_debug()) {
293 wdxdisplay9_cat.debug() << "_wcontext._window is " << _wcontext._window << "\n";
294 }
295
296 // Here check if a device already exists. If so, then this open_window call
297 // may be an extension to create multiple windows on same device In that
298 // case just create an additional swapchain for this window
299
300 while (true) {
301 if (_dxgsg->get_pipe()->get_device() == nullptr || discard_device) {
302 if (wdxdisplay9_cat.is_debug()) {
303 wdxdisplay9_cat.debug() << "device is null or fullscreen\n";
304 }
305
306 // If device exists, free it
307 if (_dxgsg->get_pipe()->get_device()) {
308 _dxgsg->dx_cleanup();
309 }
310
311 if (wdxdisplay9_cat.is_debug()) {
312 wdxdisplay9_cat.debug() << "device width " << _wcontext._display_mode.Width << "\n";
313 }
314 if (!create_screen_buffers_and_device(_wcontext, dx_force_16bpp_zbuffer)) {
315 wdxdisplay9_cat.error() << "Unable to create window with specified parameters.\n";
316 close_window();
317 return false;
318 }
319 _dxgsg->get_pipe()->make_device((void*)(&_wcontext));
320 _dxgsg->copy_pres_reset(&_wcontext);
321 _dxgsg->create_swap_chain(&_wcontext);
322 break;
323
324 } else {
325 // fill in the DXScreenData from dxdevice here and change the reference
326 // to _window.
327 if (wdxdisplay9_cat.is_debug()) {
328 wdxdisplay9_cat.debug() << "device is not null\n";
329 }
330
331 dxdev = (DXGraphicsDevice9*)_dxgsg->get_pipe()->get_device();
332 memcpy(&_wcontext, &dxdev->_Scrn, sizeof(DXScreenData));
333
334 _wcontext._presentation_params.Windowed = !is_fullscreen();
335 _wcontext._presentation_params.hDeviceWindow = _wcontext._window = _hWnd;
336 _wcontext._presentation_params.BackBufferWidth = _wcontext._display_mode.Width = _properties.get_x_size();
337 _wcontext._presentation_params.BackBufferHeight = _wcontext._display_mode.Height = _properties.get_y_size();
338
339 if (wdxdisplay9_cat.is_debug()) {
340 wdxdisplay9_cat.debug() << "device width " << _wcontext._presentation_params.BackBufferWidth << "\n";
341 }
342 if (!_dxgsg->create_swap_chain(&_wcontext)) {
343 discard_device = true;
344 continue; // try again
345 }
346 init_resized_window();
347 break;
348 }
349 }
350 if (wdxdisplay9_cat.is_debug()) {
351 wdxdisplay9_cat.debug() << "swapchain is " << _wcontext._swap_chain << "\n";
352 }
353 return true;
354}

439bool wdxGraphicsWindow9::
440do_fullscreen_resize(int x_size, int y_size) {
441 if (!WinGraphicsWindow::do_fullscreen_resize(x_size, y_size)) {
442 return false;
443 }
444
445 bool bCouldntFindValidZBuf;
446 D3DFORMAT pixFmt;
447 bool bNeedZBuffer = (_wcontext._presentation_params.EnableAutoDepthStencil != false);
448 bool bNeedStencilBuffer = IS_STENCIL_FORMAT(_wcontext._presentation_params.AutoDepthStencilFormat);
449
450 wdxGraphicsPipe9 *dxpipe;
451 DCAST_INTO_R(dxpipe, _pipe, false);
452
453 bool bIsGoodMode = false;
454 bool bResizeSucceeded = false;
455
456 if (!dxpipe->special_check_fullscreen_resolution(_wcontext, x_size, y_size)) {
457 // bypass the lowvidmem test below for certain "lowmem" cards we know have
458 // valid modes
459
460 if (_wcontext._is_low_memory_card && (!((x_size == 640) && (y_size == 480)))) {
461 wdxdisplay9_cat.error() << "resize() failed: will not try to resize low vidmem device #" << _wcontext._card_id << " to non-640x480!\n";
462 return bResizeSucceeded;
463 }
464 }
465
466 // must ALWAYS use search_for_valid_displaymode even if we know a-priori
467 // that res is valid so we can get a valid pixfmt
468 dxpipe->search_for_valid_displaymode(_wcontext, x_size, y_size,
469 bNeedZBuffer, bNeedStencilBuffer,
470 &_wcontext._supported_screen_depths_mask,
471 &bCouldntFindValidZBuf,
472 &pixFmt, dx_force_16bpp_zbuffer, true);
473 bIsGoodMode = (pixFmt != D3DFMT_UNKNOWN);
474
475 if (!bIsGoodMode) {
476 wdxdisplay9_cat.error() << "resize() failed: "
477 << (bCouldntFindValidZBuf ? "Couldnt find valid zbuffer format to go with FullScreen mode" : "No supported FullScreen modes")
478 << " at " << x_size << "x" << y_size << " for device #" << _wcontext._card_id << endl;
479 return bResizeSucceeded;
480 }
481
482 // reset_device_resize_window handles both windowed & fullscrn, so need to
483 // set new displaymode manually here
484 _wcontext._display_mode.Width = x_size;
485 _wcontext._display_mode.Height = y_size;
486 _wcontext._display_mode.Format = pixFmt;
487 _wcontext._display_mode.RefreshRate = D3DPRESENT_RATE_DEFAULT;
488 // keep the previous setting for
489 // _wcontext._presentation_params.BackBufferFormat
490
491 bResizeSucceeded = reset_device_resize_window(x_size, y_size);
492
493 if (!bResizeSucceeded) {
494 wdxdisplay9_cat.error() << "resize() failed with OUT-OF-MEMORY error!\n";
495
496 if ((!IS_16BPP_DISPLAY_FORMAT(_wcontext._presentation_params.BackBufferFormat)) &&
497 (_wcontext._supported_screen_depths_mask & (R5G6B5_FLAG|X1R5G5B5_FLAG))) {
498 // fallback strategy, if we trying >16bpp, fallback to 16bpp buffers
499 _wcontext._display_mode.Format = ((_wcontext._supported_screen_depths_mask & R5G6B5_FLAG) ? D3DFMT_R5G6B5 : D3DFMT_X1R5G5B5);
500 dx_force_16bpp_zbuffer = true;
501 if (wdxdisplay9_cat.info())
502 wdxdisplay9_cat.info() << "CreateDevice failed with out-of-vidmem, retrying w/16bpp buffers on device #" << _wcontext._card_id << endl;
503
504 bResizeSucceeded = reset_device_resize_window(x_size, y_size); // create the new resized rendertargets
505 }
506 }
507
508 return bResizeSucceeded;
509}

517bool wdxGraphicsWindow9::
518create_screen_buffers_and_device(DXScreenData &display, bool force_16bpp_zbuffer) {
519 wdxGraphicsPipe9 *dxpipe;
520 DCAST_INTO_R(dxpipe, _pipe, false);
521
522 DWORD dwRenderWidth = display._display_mode.Width;
523 DWORD dwRenderHeight = display._display_mode.Height;
524 DWORD dwBehaviorFlags = 0x0;
525 LPDIRECT3D9 _d3d9 = display._d3d9;
526 D3DCAPS9 *pD3DCaps = &display._d3dcaps;
527 D3DPRESENT_PARAMETERS* presentation_params = &display._presentation_params;
528 RECT view_rect;
529 UINT adapter;
530 D3DDEVTYPE device_type;
531 HRESULT hr;
532
533 adapter = display._card_id;
534 device_type = D3DDEVTYPE_HAL;
535
536 // NVIDIA NVPerfHUD
537 if (dx_use_nvperfhud) {
538 UINT adapter_id;
539 UINT total_adapters;
540
541 total_adapters = _d3d9 -> GetAdapterCount ( );
542 for (adapter_id = 0; adapter_id < total_adapters; adapter_id++) {
543 D3DADAPTER_IDENTIFIER9 identifier;
544
545 _d3d9 -> GetAdapterIdentifier (adapter_id, 0, &identifier);
546 if (strcmp ("NVIDIA NVPerfHUD", identifier.Description) == 0) {
547 adapter = adapter_id;
548 device_type = D3DDEVTYPE_REF;
549 break;
550 }
551 }
552 }
553
554 if (wdxdisplay9_cat.is_debug()) {
555 wdxdisplay9_cat.debug() << "Display Width " << dwRenderWidth << " and PresParam Width " << _wcontext._presentation_params.BackBufferWidth << "\n";
556 }
557
558 // BUGBUG: need to change panda to put frame buffer properties with
559 // GraphicsWindow, not GSG!! Update: Did I fix the bug? - Josh
560 bool bWantStencil = (_fb_properties.get_stencil_bits() > 0);
561 bool bWantAlpha = (_fb_properties.get_alpha_bits() > 0);
562
563 PRINT_REFCNT(wdxdisplay9, _d3d9);
564
565 nassertr(_d3d9 != nullptr, false);
566 nassertr(pD3DCaps->DevCaps & D3DDEVCAPS_HWRASTERIZATION, false);
567
568 bool do_sync = sync_video;
569
570 if (do_sync && !(pD3DCaps->Caps & D3DCAPS_READ_SCANLINE)) {
571 wdxdisplay9_cat.info()
572 << "HW doesnt support syncing to vertical refresh, ignoring sync-video\n";
573 do_sync = false;
574 }
575
576 presentation_params->BackBufferFormat = display._display_mode.Format;
577
578 // check for D3DFMT_A8R8G8B8 first
579 if (bWantAlpha) {
580 if (!FAILED(_d3d9->CheckDeviceFormat(adapter, device_type, display._display_mode.Format, D3DUSAGE_RENDERTARGET,
581 D3DRTYPE_SURFACE, D3DFMT_A8R8G8B8))) {
582 if (!FAILED(_d3d9->CheckDeviceType(adapter, device_type, display._display_mode.Format, D3DFMT_A8R8G8B8,
583 is_fullscreen()))) {
584 // We can accept D3DFMT_A8R8G8B8, so use it.
585 presentation_params->BackBufferFormat = D3DFMT_A8R8G8B8;
586 }
587 }
588 }
589
590 // check for same format as display_mode verify the rendertarget fmt
591 if (FAILED(_d3d9->CheckDeviceFormat(adapter, device_type, display._display_mode.Format, D3DUSAGE_RENDERTARGET,
592 D3DRTYPE_SURFACE, presentation_params->BackBufferFormat))) {
593 wdxdisplay9_cat.error() << "adapter #" << adapter << " CheckDeviceFmt failed for surface fmt " << D3DFormatStr(presentation_params->BackBufferFormat) << endl;
594 goto Fallback_to_16bpp_buffers;
595 }
596
597 if (FAILED(_d3d9->CheckDeviceType(adapter, device_type, display._display_mode.Format, presentation_params->BackBufferFormat,
598 is_fullscreen()))) {
599 wdxdisplay9_cat.error() << "adapter #" << adapter << " CheckDeviceType failed for surface fmt " << D3DFormatStr(presentation_params->BackBufferFormat) << endl;
600 goto Fallback_to_16bpp_buffers;
601 }
602
603 if (display._presentation_params.EnableAutoDepthStencil) {
604 if (!dxpipe->find_best_depth_format(display, display._display_mode,
605 &display._presentation_params.AutoDepthStencilFormat,
606 bWantStencil, false)) {
607 wdxdisplay9_cat.error()
608 << "find_best_depth_format failed in CreateScreenBuffers for device #"
609 << adapter << endl;
610 goto Fallback_to_16bpp_buffers;
611 }
612 _depth_buffer_bpp = D3DFMT_to_DepthBits(display._presentation_params.AutoDepthStencilFormat);
613 } else {
614 _depth_buffer_bpp = 0;
615 }
616
617 presentation_params->Windowed = !is_fullscreen();
618
619 int supported_multisamples = 0;
620 if (framebuffer_multisample.get_value()){
621 supported_multisamples = multisamples.get_value();
622 }
623
624 while (supported_multisamples > 1){
625 // need to check both rendertarget and zbuffer fmts
626 hr = _d3d9->CheckDeviceMultiSampleType(adapter, D3DDEVTYPE_HAL, display._display_mode.Format,
627 is_fullscreen(), D3DMULTISAMPLE_TYPE(supported_multisamples), nullptr);
628 if (FAILED(hr)) {
629 supported_multisamples--;
630 continue;
631 }
632
633 if (display._presentation_params.EnableAutoDepthStencil) {
634 hr = _d3d9->CheckDeviceMultiSampleType(adapter, D3DDEVTYPE_HAL, display._presentation_params.AutoDepthStencilFormat,
635 is_fullscreen(), D3DMULTISAMPLE_TYPE(supported_multisamples), nullptr);
636 if (FAILED(hr)) {
637 supported_multisamples--;
638 continue;
639 }
640 }
641
642 presentation_params->MultiSampleType = D3DMULTISAMPLE_TYPE(supported_multisamples);
643
644 if (wdxdisplay9_cat.is_info())
645 wdxdisplay9_cat.info() << "adapter #" << adapter << " using multisample antialiasing level: " << supported_multisamples <<
646 ". Requested level was: " << multisamples.get_value() << endl;
647 break;
648 }
649
650 presentation_params->BackBufferCount = 1;
651 presentation_params->Flags = 0x0;
652 presentation_params->hDeviceWindow = display._window;
653 presentation_params->BackBufferWidth = display._display_mode.Width;
654 presentation_params->BackBufferHeight = display._display_mode.Height;
655
656 if (_wcontext._is_tnl_device) {
657 dwBehaviorFlags |= D3DCREATE_HARDWARE_VERTEXPROCESSING;
658 // note: we could create a pure device in this case if I eliminated the
659 // GetRenderState calls in dxgsg
660
661 // also, no software vertex processing available since I specify
662 // D3DCREATE_HARDWARE_VERTEXPROCESSING and not
663 // D3DCREATE_MIXED_VERTEXPROCESSING
664 } else {
665 dwBehaviorFlags |= D3DCREATE_SOFTWARE_VERTEXPROCESSING;
666 }
667
668 if (dx_preserve_fpu_state)
669 dwBehaviorFlags |= D3DCREATE_FPU_PRESERVE;
670
671 // if window is not foreground in exclusive mode, ddraw thinks you are 'not
672 // active', so it changes your WM_ACTIVATEAPP from true to false, causing us
673 // to go into a 'wait-for WM_ACTIVATEAPP true' loop, and the event never
674 // comes so we hang in fullscreen wait. also doing this for windowed mode
675 // since it was requested.
676 if (!SetForegroundWindow(display._window)) {
677 wdxdisplay9_cat.warning() << "SetForegroundWindow() failed!\n";
678 }
679
680 if (dx_use_multithread) {
681 dwBehaviorFlags |= D3DCREATE_MULTITHREADED;
682 }
683 if (dx_use_puredevice) {
684 dwBehaviorFlags |= D3DCREATE_PUREDEVICE;
685 }
686
687 if (dx_disable_driver_management) {
688 dwBehaviorFlags |= D3DCREATE_DISABLE_DRIVER_MANAGEMENT;
689 }
690 if (dx_disable_driver_management_ex) {
691 dwBehaviorFlags |= D3DCREATE_DISABLE_DRIVER_MANAGEMENT_EX;
692 }
693
694 if (is_fullscreen()) {
695 // CREATE FULLSCREEN BUFFERS
696
697 presentation_params->SwapEffect = D3DSWAPEFFECT_DISCARD; // we don't care about preserving contents of old frame
698 presentation_params->PresentationInterval = (do_sync ? D3DPRESENT_INTERVAL_ONE : D3DPRESENT_INTERVAL_IMMEDIATE);
699 presentation_params->FullScreen_RefreshRateInHz = display._display_mode.RefreshRate;
700
701 ClearToBlack(display._window, get_properties());
702
703 hr = _d3d9->CreateDevice(adapter, device_type, _hWnd,
704 dwBehaviorFlags, presentation_params, &display._d3d_device);
705
706 if (FAILED(hr)) {
707 wdxdisplay9_cat.fatal() << "D3D CreateDevice failed for adapter #" << adapter << ", " << D3DERRORSTRING(hr);
708
709 if (hr == D3DERR_OUTOFVIDEOMEMORY)
710 goto Fallback_to_16bpp_buffers;
711 else
712 return false;
713 }
714
715 SetRect(&view_rect, 0, 0, dwRenderWidth, dwRenderHeight);
716
717 } else {
718 // CREATE WINDOWED BUFFERS
719
720 D3DDISPLAYMODE dispmode;
721 hr = display._d3d9->GetAdapterDisplayMode(adapter, &dispmode);
722
723 if (FAILED(hr)) {
724 wdxdisplay9_cat.fatal()
725 << "GetAdapterDisplayMode failed" << D3DERRORSTRING(hr);
726 return false;
727 }
728
729 if (dispmode.Format == D3DFMT_P8) {
730 wdxdisplay9_cat.fatal()
731 << "Can't run windowed in an 8-bit or less display mode" << endl;
732 return false;
733 }
734
735 // From d3d8caps.h D3DPRESENT_INTERVAL_DEFAULT = 0x00000000L #define
736 // D3DPRESENT_INTERVAL_ONE 0x00000001L Next line is really sloppy,
737 // should either be D3DPRESENT_INTERVAL_DEFAULT or D3DPRESENT_INTERVAL_ONE
738 // not a direct number! but I'm not going to touch it because it's
739 // working as is. Zhao 12152011
740 presentation_params->PresentationInterval = 0;
741
742 // ATI 5450 doesn't like D3DSWAPEFFECT_FLIP
743 presentation_params->SwapEffect = D3DSWAPEFFECT_DISCARD;
744 if (do_sync == false) {
745 presentation_params->PresentationInterval = D3DPRESENT_INTERVAL_IMMEDIATE;
746 }
747/*
748 * if (supported_multisamples<2) { if (do_sync) { It turns out that COPY_VSYNC
749 * has real performance problems on many nVidia cards--it syncs at some random
750 * interval, possibly skipping over several video syncs. Screw it, we'll
751 * effectively disable sync-video with windowed mode using DirectX8.
752 * presentation_params->SwapEffect = D3DSWAPEFFECT_COPY_VSYNC;
753 * presentation_params->SwapEffect = D3DSWAPEFFECT_DISCARD; } else {
754 * presentation_params->SwapEffect = D3DSWAPEFFECT_DISCARD; }
755 */
756
757/*
758 * override presentation parameters for windowed mode, render and display at
759 * maximum speed if (do_sync == false) { presentation_params->SwapEffect =
760 * D3DSWAPEFFECT_FLIP; presentation_params->PresentationInterval =
761 * D3DPRESENT_INTERVAL_IMMEDIATE; } } else { presentation_params->SwapEffect =
762 * D3DSWAPEFFECT_DISCARD; }
763 */
764
765 // assert((dwRenderWidth ==
766 // presentation_params->BackBufferWidth)&&(dwRenderHeight ==
767 // presentation_params->BackBufferHeight));
768
769 hr = _d3d9->CreateDevice(adapter, device_type, _hWnd,
770 dwBehaviorFlags, presentation_params, &display._d3d_device);
771
772 if (FAILED(hr)) {
773 wdxdisplay9_cat.warning() << "presentation_params->BackBufferWidth : " << presentation_params->BackBufferWidth << endl;
774 wdxdisplay9_cat.warning() << "presentation_params->BackBufferHeight : " << presentation_params->BackBufferHeight << endl;
775 wdxdisplay9_cat.warning() << "presentation_params->BackBufferFormat : " << presentation_params->BackBufferFormat << endl;
776 wdxdisplay9_cat.warning() << "presentation_params->BackBufferCount : " << presentation_params->BackBufferCount << endl;
777 wdxdisplay9_cat.warning() << "D3D CreateDevice failed for adapter #" << adapter << D3DERRORSTRING(hr);
778 goto Fallback_to_16bpp_buffers;
779 }
780 } // end create windowed buffers
781
782 // ========================================================
783
784 PRINT_REFCNT(wdxdisplay9, _wcontext._d3d_device);
785
786 if (presentation_params->EnableAutoDepthStencil) {
787 int depth_bits;
788 int stencil_bits;
789
790 depth_bits = 1;
791 stencil_bits = 0;
792 switch (presentation_params->AutoDepthStencilFormat)
793 {
794 case D3DFMT_D16_LOCKABLE:
795 depth_bits = 16;
796 break;
797 case D3DFMT_D32:
798 depth_bits = 32;
799 break;
800 case D3DFMT_D15S1:
801 depth_bits = 15;
802 stencil_bits = 1;
803 break;
804 case D3DFMT_D24S8:
805 depth_bits = 24;
806 stencil_bits = 8;
807 break;
808 case D3DFMT_D24X8:
809 depth_bits = 24;
810 break;
811 case D3DFMT_D24X4S4:
812 depth_bits = 24;
813 stencil_bits = 4;
814 break;
815 case D3DFMT_D32F_LOCKABLE:
816 depth_bits = 32;
817 break;
818 case D3DFMT_D24FS8:
819 depth_bits = 24;
820 stencil_bits = 8;
821 break;
822 case D3DFMT_D16:
823 depth_bits = 16;
824 break;
825 default:
826 wdxdisplay9_cat.error() << "unknown depth stencil format " << presentation_params->AutoDepthStencilFormat;
827 break;
828 }
829
830 _fb_properties.set_stencil_bits(stencil_bits);
831 _fb_properties.set_depth_bits(depth_bits);
832 } else {
833 _fb_properties.set_depth_bits(0);
834 _fb_properties.set_stencil_bits(0);
835 }
836
837 init_resized_window();
838
839 return true;
840
841 Fallback_to_16bpp_buffers:
842 if ((!IS_16BPP_DISPLAY_FORMAT(presentation_params->BackBufferFormat)) &&
843 (display._supported_screen_depths_mask & (R5G6B5_FLAG|X1R5G5B5_FLAG))) {
844 // fallback strategy, if we trying >16bpp, fallback to 16bpp buffers
845
846 display._display_mode.Format = ((display._supported_screen_depths_mask & R5G6B5_FLAG) ? D3DFMT_R5G6B5 : D3DFMT_X1R5G5B5);
847
848 if (wdxdisplay9_cat.info()) {
849 wdxdisplay9_cat.info()
850 << "CreateDevice failed with out-of-vidmem or invalid BackBufferFormat, retrying w/16bpp buffers on adapter #"
851 << adapter << endl;
852 }
853 return create_screen_buffers_and_device(display, true);
854 // return;
855
856 } else if (!((dwRenderWidth == 640)&&(dwRenderHeight == 480))) {
857 if (wdxdisplay9_cat.info())
858 wdxdisplay9_cat.info() << "CreateDevice failed w/out-of-vidmem, retrying at 640x480 w/16bpp buffers on adapter #" << adapter << endl;
859 // try final fallback to 640x480x16
860 display._display_mode.Width = 640;
861 display._display_mode.Height = 480;
862 return create_screen_buffers_and_device(display, true);
863 // return;
864
865 } else {
866 wdxdisplay9_cat.fatal()
867 << "Can't create any screen buffers, bailing out.\n";
868 return false;
869 }

878bool wdxGraphicsWindow9::
879choose_device() {
880 HRESULT hr;
881
882 wdxGraphicsPipe9 *dxpipe;
883 DCAST_INTO_R(dxpipe, _pipe, false);
884
885 int num_adapters = dxpipe->__d3d9->GetAdapterCount();
886 DXDeviceInfoVec device_infos;
887
888 for (int i = 0; i < num_adapters; i++) {
889 D3DADAPTER_IDENTIFIER9 adapter_info;
890 ZeroMemory(&adapter_info, sizeof(D3DADAPTER_IDENTIFIER9));
891 hr = dxpipe->__d3d9->GetAdapterIdentifier(i, 0, &adapter_info);
892 if (FAILED(hr)) {
893 wdxdisplay9_cat.fatal()
894 << "D3D GetAdapterID(" << i << ") failed: "
895 << D3DERRORSTRING(hr) << endl;
896 continue;
897 }
898
899 LARGE_INTEGER *DrvVer = &adapter_info.DriverVersion;
900
901 wdxdisplay9_cat.info()
902 << "D3D9 Adapter[" << i << "]: " << adapter_info.Description
903 << ", Driver: " << adapter_info.Driver << ", DriverVersion: ("
904 << HIWORD(DrvVer->HighPart) << "." << LOWORD(DrvVer->HighPart) << "."
905 << HIWORD(DrvVer->LowPart) << "." << LOWORD(DrvVer->LowPart)
906 << ")\nVendorID: 0x" << std::hex << adapter_info.VendorId
907 << " DeviceID: 0x" << adapter_info.DeviceId
908 << " SubsysID: 0x" << adapter_info.SubSysId
909 << " Revision: 0x" << adapter_info.Revision << std::dec << endl;
910
911 HMONITOR _monitor = dxpipe->__d3d9->GetAdapterMonitor(i);
912 if (_monitor == nullptr) {
913 wdxdisplay9_cat.info()
914 << "D3D9 Adapter[" << i << "]: seems to be disabled, skipping it\n";
915 continue;
916 }
917
918 DXDeviceInfo devinfo;
919 ZeroMemory(&devinfo, sizeof(devinfo));
920 memcpy(&devinfo.guidDeviceIdentifier, &adapter_info.DeviceIdentifier,
921 sizeof(GUID));
922 strncpy(devinfo.szDescription, adapter_info.Description,
923 MAX_DEVICE_IDENTIFIER_STRING);
924 strncpy(devinfo.szDriver, adapter_info.Driver,
925 MAX_DEVICE_IDENTIFIER_STRING);
926 devinfo.VendorID = adapter_info.VendorId;
927 devinfo.DeviceID = adapter_info.DeviceId;
928 devinfo._driver_version = adapter_info.DriverVersion;
929 devinfo._monitor = _monitor;
930 devinfo.cardID = i;
931
932 device_infos.push_back(devinfo);
933 }
934
935 if (device_infos.empty()) {
936 wdxdisplay9_cat.error()
937 << "No available D3D9 devices found.\n";
938 return false;
939 }
940
941 // Since some adapters may have been disabled, we should re-obtain the
942 // number of available adapters.
943 num_adapters = (int)device_infos.size();
944
945 // Now choose a suitable adapter.
946
947 int adapter_num = D3DADAPTER_DEFAULT;
948
949 // Eventually, we should have some interface for specifying a device index
950 // interactively, instead of only via Configrc.
951 if (dx_preferred_device_id != -1) {
952 if (dx_preferred_device_id < 0 || dx_preferred_device_id >= num_adapters) {
953 wdxdisplay9_cat.error()
954 << "invalid 'dx-preferred-device-id', valid values are 0-"
955 << num_adapters - 1 << ", using default adapter instead.\n";
956 } else {
957 adapter_num = dx_preferred_device_id;
958 }
959 }
960
961 // Try to select the default or requested device.
962 if (adapter_num >= 0 && adapter_num < (int)device_infos.size()) {
963 if (consider_device(dxpipe, &device_infos[adapter_num])) {
964 wdxdisplay9_cat.info()
965 << "Selected device " << adapter_num << " (of "
966 << device_infos.size() << ", zero-based)\n";
967 return true;
968 }
969 wdxdisplay9_cat.info()
970 << "Could not select device " << adapter_num << "\n";
971 }
972
973 // Iterate through all available devices to find the first suitable one.
974 for (UINT devnum = 0; devnum < device_infos.size(); ++devnum) {
975 if (consider_device(dxpipe, &device_infos[devnum])) {
976 wdxdisplay9_cat.info()
977 << "Chose device " << devnum << " (of "
978 << device_infos.size() << ", zero-based): first one that works.\n";
979 return true;
980 }
981 }
982
983 wdxdisplay9_cat.error() << "no usable display devices.\n";
984 return false;
985}

992consider_device(wdxGraphicsPipe9 *dxpipe, DXDeviceInfo *device_info) {
993
994 nassertr(dxpipe != nullptr, false);
995 WindowProperties properties = get_properties();
996 DWORD dwRenderWidth = properties.get_x_size();
997 DWORD dwRenderHeight = properties.get_y_size();
998 HRESULT hr;
999 LPDIRECT3D9 _d3d9 = dxpipe->__d3d9;
1000
1001 nassertr(_dxgsg != nullptr, false);
1002 _wcontext._d3d9 = _d3d9;
1003 _wcontext._card_id = device_info->cardID; // could this change by end?

1006

1009 if (FAILED(hr)) {

1013 }
1014
1015 D3DCAPS9 _d3dcaps;
1016 hr = _d3d9->GetDeviceCaps(device_info->cardID, D3DDEVTYPE_HAL, &_d3dcaps);

1039 // do much
1040 if (_d3dcaps.MaxStreams == 0) {
1041 if (wdxdisplay9_cat.is_debug()) {
1042 wdxdisplay9_cat.debug()

1046 UINT IDnum;

1049 for (IDnum = 0; IDnum < dxpipe->_card_ids.size(); IDnum++) {
1050 if ((device_info->VendorID == dxpipe->_card_ids[IDnum].VendorID) &&
1051 (device_info->DeviceID == dxpipe->_card_ids[IDnum].DeviceID) &&
1052 (device_info->_monitor == dxpipe->_card_ids[IDnum]._monitor))

1054 }
1055
1056 if (IDnum < dxpipe->_card_ids.size()) {
1057 _wcontext._max_available_video_memory = dxpipe->_card_ids[IDnum]._max_available_video_memory;
1058 _wcontext._is_low_memory_card = dxpipe->_card_ids[IDnum]._is_low_memory_card;
1059 } else {

1061 << "Error: couldnt find a CardID match in DX7 info, assuming card is not a lowmem card\n";
1062 }
1063 }

1065 if ((bWantStencil) && (_d3dcaps.StencilCaps == 0x0)) {
1066 wdxdisplay9_cat.fatal()
1067 << "Stencil ability requested, but device #" << device_info->cardID
1068 << " (" << _wcontext._dx_device_id.Description
1069 << "), has no stencil capability!\n";
1070 return false;
1071 }
1072

1074 // (see GF2) for this case, you probably want MIXED processing to use HW for

1078 _wcontext._can_use_hw_vertex_shaders =
1079 (_d3dcaps.VertexShaderVersion >= D3DVS_VERSION(1, 0));
1080 _wcontext._can_use_pixel_shaders =
1081 (_d3dcaps.PixelShaderVersion >= D3DPS_VERSION(1, 0));

1084 ((!(_d3dcaps.RasterCaps & D3DPRASTERCAPS_ZBUFFERLESSHSR )) &&

1087 _wcontext._presentation_params.EnableAutoDepthStencil = bNeedZBuffer;

1091 if (is_fullscreen()) {
1092 bool bCouldntFindValidZBuf;
1093
1094 dxpipe->search_for_valid_displaymode(_wcontext, dwRenderWidth, dwRenderHeight,
1095 bNeedZBuffer, bWantStencil,
1096 &_wcontext._supported_screen_depths_mask,
1097 &bCouldntFindValidZBuf,
1098 &pixFmt, dx_force_16bpp_zbuffer, true);
1099
1100 // note I'm not saving refresh rate, will just use adapter default at
1101 // given res for now
1102
1103 if (pixFmt == D3DFMT_UNKNOWN) {
1104 wdxdisplay9_cat.error()
1105 << (bCouldntFindValidZBuf ? "Couldnt find valid zbuffer format to go with FullScreen mode" : "No supported FullScreen modes")
1106 << " at " << dwRenderWidth << "x" << dwRenderHeight << " for device #" << _wcontext._card_id << endl;
1107 return false;
1108 }
1109 } else {
1110 // Windowed Mode
1111
1112 D3DDISPLAYMODE dispmode;
1113 hr = _d3d9->GetAdapterDisplayMode(device_info->cardID, &dispmode);
1114 if (FAILED(hr)) {
1115 wdxdisplay9_cat.error()
1116 << "GetAdapterDisplayMode(" << device_info->cardID
1117 << ") failed" << D3DERRORSTRING(hr);
1118 return false;
1119 }
1120 pixFmt = dispmode.Format;
1121 }
1122
1123 _wcontext._display_mode.Width = dwRenderWidth;
1124 _wcontext._display_mode.Height = dwRenderHeight;
1125 _wcontext._display_mode.Format = pixFmt;
1126 _wcontext._display_mode.RefreshRate = D3DPRESENT_RATE_DEFAULT;
1127 _wcontext._monitor = device_info->_monitor;
1128
1129 if (strcmp(device_info->szDriver, "igdumd32.dll") == 0 &&
1130 device_info->_driver_version.QuadPart <= 0x0007000e000affffLL &&
1131 dx_intel_compressed_texture_bug) {
1132 // Disable compressed textures for this buggy driver (7.14.10.65535 and
1133 // earlier--I don't know whether any other drivers also exhibit the bug).
1134 _wcontext._intel_compressed_texture_bug = true;
1135 }
1136
1137 return true;
