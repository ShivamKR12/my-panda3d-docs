85bool CocoaGLGraphicsBuffer::
86open_buffer() {
87 CocoaGraphicsPipe *cocoa_pipe;
88 DCAST_INTO_R(cocoa_pipe, _pipe, false);
89
90 // GSG CreationInitialization
91 CocoaGLGraphicsStateGuardian *cocoagsg;
92 if (_gsg == nullptr) {
93 // There is no old gsg. Create a new one.
94 cocoagsg = new CocoaGLGraphicsStateGuardian(_engine, _pipe, nullptr);
95 cocoagsg->choose_pixel_format(_fb_properties, cocoa_pipe->get_display_id(), false);
96 _gsg = cocoagsg;
97 } else {
98 // If the old gsg has the wrong pixel format, create a new one that shares
99 // with the old gsg.
100 DCAST_INTO_R(cocoagsg, _gsg, false);
101 if (cocoagsg->get_engine() != _engine ||
102 !cocoagsg->get_fb_properties().subsumes(_fb_properties)) {
103 cocoagsg = new CocoaGLGraphicsStateGuardian(_engine, _pipe, cocoagsg);
104 cocoagsg->choose_pixel_format(_fb_properties, cocoa_pipe->get_display_id(), false);
105 _gsg = cocoagsg;
106 }
107 }
108
109 if (cocoagsg->_context == nil) {
110 // Could not obtain a proper context.
111 _gsg.clear();
112 return false;
113 }
114
115 FrameBufferProperties desired_props(_fb_properties);
116
117 // Lock the context, so we can safely operate on it.
118 cocoagsg->lock_context();
119
120 // Make the context current and initialize what we need.
121 [cocoagsg->_context makeCurrentContext];
122 [cocoagsg->_context update];
123 cocoagsg->reset_if_new();
124
125 // These properties are determined by choose_pixel_format.
126 _fb_properties.set_force_hardware(cocoagsg->_fbprops.get_force_hardware());
127 _fb_properties.set_force_software(cocoagsg->_fbprops.get_force_software());
128
129 bool success = GLGraphicsBuffer::open_buffer();
130 if (success) {
131 rebuild_bitplanes();
132 if (_needs_rebuild) {
133 // If it still needs rebuild, then something must have gone wrong.
134 success = false;
135 }
136 }
137
138 if (success && !_fb_properties.verify_hardware_software
139 (desired_props, cocoagsg->get_gl_renderer())) {
140 GLGraphicsBuffer::close_buffer();
141 success = false;
142 }
143
144 // Release the context.
145 cocoagsg->unlock_context();
146
147 if (!success) {
148 return false;
149 }
150
151 return true;
152}
