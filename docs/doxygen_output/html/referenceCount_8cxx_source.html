24bool ReferenceCount::
25do_test_ref_count_integrity() const {
26 int ref_count = _ref_count.load(std::memory_order_relaxed);
27
28 // If this assertion fails, we're trying to delete an object that was just
29 // deleted. Possibly you used a real pointer instead of a PointerTo at some
30 // point, and the object was deleted when the PointerTo went out of scope.
31 // Maybe you tried to create an automatic (local variable) instance of a
32 // class that derives from ReferenceCount. Or maybe your headers are out of
33 // sync, and you need to make clean in direct or some higher tree.
34 nassertr(ref_count != deleted_ref_count, false);
35
36 // If this assertion fails, the reference counts are all screwed up
37 // altogether. Maybe some errant code stomped all over memory somewhere.
38 nassertr(ref_count >= 0, false);
39
40 return true;
41}
