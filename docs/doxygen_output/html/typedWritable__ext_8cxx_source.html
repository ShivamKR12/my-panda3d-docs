156register_python_type(TypeRegistry *registry, PyTypeObject *cls) {
157 TypeHandle handle = TypeHandle::none();
158
159 if (cls->tp_bases != nullptr) {
160 Py_ssize_t count = PyTuple_GET_SIZE(cls->tp_bases);
161 for (Py_ssize_t i = 0; i < count; ++i) {
162 PyObject *base = PyTuple_GET_ITEM(cls->tp_bases, count);
163 TypeHandle base_handle = register_python_type(registry, (PyTypeObject *)base);
164 if (base_handle != TypeHandle::none()) {
165 if (handle == TypeHandle::none()) {
166 handle = registry->register_dynamic_type(cls->tp_name);

168 registry->record_derivation(handle, base_handle);
169 return handle;
170 }
171 }

181__new__(PyTypeObject *cls) {
182 if (cls == (PyTypeObject *)&Dtool_TypedWritable) {
183 return Dtool_Raise_TypeError("cannot init abstract class");
184 }
185
186 PyObject *self = cls->tp_alloc(cls, 0);
187 ((Dtool_PyInstDef *)self)->_signature = PY_PANDA_SIGNATURE;
188 ((Dtool_PyInstDef *)self)->_My_Type = &Dtool_TypedWritable;

190 // We expect the user to override this method.
191 PyObject *class_type = PyObject_CallMethod((PyObject *)cls, "get_class_type", nullptr);
192 if (class_type == nullptr) {
193 return nullptr;
194 }
195
196 // Check that it returned a TypeHandle, and that it is actually different
197 // from the one on the base class (which might mean that the user didn't
198 // actually define a custom get_class_type() method).
199 TypeHandle *handle = nullptr;
200 if (!DtoolInstance_GetPointer(class_type, handle, Dtool_TypeHandle) ||
201 *handle == TypedWritable::get_class_type() ||
202 *handle == TypedObject::get_class_type() ||
203 !handle->is_derived_from(TypedWritable::get_class_type())) {
204 Dtool_Raise_TypeError("get_class_type() must be overridden to return a unique TypeHandle that indicates derivation from TypedWritable");
205 return nullptr;
206 }
207
208 // Make sure that the bindings know how to obtain a wrapper for this type.
209 TypeRegistry *registry = TypeRegistry::ptr();
210 PyTypeObject *cls_ref = (PyTypeObject *)Py_NewRef((PyObject *)cls);
211 registry->record_python_type(*handle, cls_ref, &wrap_typed_writable);
212
213 // Note that we don't increment the reference count here, because that would
214 // create a memory leak. The TypedWritableProxy gets deleted when the Python
215 // object reaches a reference count of 0.
216 TypedWritableProxy *proxy = new TypedWritableProxy;
217 proxy->_self = self;
218 proxy->_type = *handle;

245PyObject *Extension<TypedWritable>::
246__reduce_persist__(PyObject *self, PyObject *pickler) const {
247 // We should return at least a 2-tuple, (Class, (args)): the necessary class
248 // object whose constructor we should call (e.g. this), and the arguments
249 // necessary to reconstruct this object.
250
251 // Check that we have a decode_from_bam_stream python method. If not, we
252 // can't use this interface.
253 PyObject *method = PyObject_GetAttrString(self, "decode_from_bam_stream");
254 if (method == nullptr) {
255 std::ostringstream stream;
256 stream << "Cannot pickle objects of type " << _this->get_type() << "\n";
257 std::string message = stream.str();
258 PyErr_SetString(PyExc_TypeError, message.c_str());
259 return nullptr;
260 }
261 Py_DECREF(method);
262
263 BamWriter *writer = nullptr;
264 if (pickler != nullptr) {
265 PyObject *py_writer = PyObject_GetAttrString(pickler, "bamWriter");
266 if (py_writer == nullptr) {
267 // It's OK if there's no bamWriter.
268 PyErr_Clear();
269 } else {
270 DtoolInstance_GetPointer(py_writer, writer, Dtool_BamWriter);
271 Py_DECREF(py_writer);
272 }
273 }
274
275 // First, streamify the object, if possible.
276 vector_uchar bam_stream;
277 if (!_this->encode_to_bam_stream(bam_stream, writer)) {
278 std::ostringstream stream;
279 stream << "Could not bamify object of type " << _this->get_type() << "\n";
280 std::string message = stream.str();
281 PyErr_SetString(PyExc_TypeError, message.c_str());
282 return nullptr;
283 }
284
285 // Start by getting this class object.
286 PyObject *this_class = PyObject_Type(self);
287 if (this_class == nullptr) {
288 return nullptr;
289 }
290
291 PyObject *func;
292 if (writer != nullptr) {
293 // The modified pickle support: call the "persistent" version of this
294 // function, which receives the unpickler itself as an additional
295 // parameter.
296 func = find_global_decode(this_class, "py_decode_TypedWritable_from_bam_stream_persist");
297 if (func == nullptr) {
298 PyErr_SetString(PyExc_TypeError, "Couldn't find py_decode_TypedWritable_from_bam_stream_persist()");
299 Py_DECREF(this_class);
300 return nullptr;
301 }
302
303 } else {
304 // The traditional pickle support: call the non-persistent version of this
305 // function.
306 func = find_global_decode(this_class, "py_decode_TypedWritable_from_bam_stream");
307 if (func == nullptr) {
308 PyErr_SetString(PyExc_TypeError, "Couldn't find py_decode_TypedWritable_from_bam_stream()");
309 Py_DECREF(this_class);
310 return nullptr;
311 }
312 }
313
314 // PyTuple_SET_ITEM conveniently borrows the reference it is passed.
315 PyObject *args = PyTuple_New(2);
316 PyTuple_SET_ITEM(args, 0, this_class);
317 PyTuple_SET_ITEM(args, 1, Dtool_WrapValue(bam_stream));
318
319 PyObject *tuple = PyTuple_New(2);
320 PyTuple_SET_ITEM(tuple, 0, func);
321 PyTuple_SET_ITEM(tuple, 1, args);
322 return tuple;
323}

334PyObject *Extension<TypedWritable>::
335find_global_decode(PyObject *this_class, const char *func_name) {
336 // Get the module in which BamWriter is defined.
337 PyObject *module_name = PyObject_GetAttrString((PyObject *)&Dtool_BamWriter, "__module__");
338 if (module_name != nullptr) {
339 PyObject *module = PyImport_GetModule(module_name);
340 Py_DECREF(module_name);
341 if (module != nullptr) {
342 PyObject *func = PyObject_GetAttrString(module, (char *)func_name);
343 Py_DECREF(module);
344 if (func != nullptr) {
345 return func;
346 }
347 }
348 }
349
350 PyObject *bases = PyObject_GetAttrString(this_class, "__bases__");
351 if (bases != nullptr) {
352 {
353 PyObject *tuple = PySequence_Tuple(bases);
354 Py_DECREF(bases);
355 if (tuple == nullptr) {
356 PyErr_Clear();
357 return nullptr;
358 }
359 bases = tuple;
360 }
361
362 Py_ssize_t size = PyTuple_GET_SIZE(bases);
363 for (Py_ssize_t i = 0; i < size; ++i) {
364 PyObject *base = PyTuple_GET_ITEM(bases, i);
365 if (base != nullptr) {
366 PyObject *func = find_global_decode(base, func_name);
367 if (func != nullptr) {
368 Py_DECREF(bases);
369 return func;
370 }
371 }
372 }
373 Py_DECREF(bases);
374 }
375
376 return nullptr;
377}

399PyObject *
400py_decode_TypedWritable_from_bam_stream_persist(PyObject *pickler, PyObject *this_class, const vector_uchar &data) {
401
402 PyObject *py_reader = nullptr;
403 if (pickler != nullptr) {
404 py_reader = PyObject_GetAttrString(pickler, "bamReader");
405 if (py_reader == nullptr) {
406 // It's OK if there's no bamReader.
407 PyErr_Clear();
408 }
409 }
410
411 // We need the function PandaNode::decode_from_bam_stream or
412 // TypedWritableReferenceCount::decode_from_bam_stream, which invokes the
413 // BamReader to reconstruct this object. Since we use the specific object's
414 // class as the pointer, we get the particular instance of
415 // decode_from_bam_stream appropriate to this class.
416
417 PyObject *func = PyObject_GetAttrString(this_class, "decode_from_bam_stream");
418 if (func == nullptr) {
419 Py_XDECREF(py_reader);
420 return nullptr;
421 }
422
423 PyObject *bytes = Dtool_WrapValue(data);
424 if (bytes == nullptr) {
425 Py_DECREF(func);
426 Py_XDECREF(py_reader);
427 return nullptr;
428 }
429
430 PyObject *result;
431 if (py_reader != nullptr) {
432 result = PyObject_CallFunctionObjArgs(func, bytes, py_reader, nullptr);
433 Py_DECREF(py_reader);
434 } else {
435 result = PyObject_CallOneArg(func, bytes);
436 }
437 Py_DECREF(bytes);
438 Py_DECREF(func);
439
440 if (result == nullptr) {
441 return nullptr;
442 }
443
444 if (result == Py_None) {
445 Py_DECREF(result);
446 PyErr_SetString(PyExc_ValueError, "Could not unpack bam stream");
447 return nullptr;
448 }
449
450 return result;
451}
