134void WinStatsLabel::
135update_color() {
136 if (_bg_brush != 0) {
137 DeleteObject(_bg_brush);
138 }
139 if (_highlight_bg_brush != 0) {
140 DeleteObject(_highlight_bg_brush);
141 }
142
143 LRGBColor rgb = _monitor->get_collector_color(_collector_index);
144 int r = (int)encode_sRGB_uchar((float)rgb[0]);
145 int g = (int)encode_sRGB_uchar((float)rgb[1]);
146 int b = (int)encode_sRGB_uchar((float)rgb[2]);
147 _bg_brush = CreateSolidBrush(RGB(r, g, b));
148
149 // Calculate the color when it is highlighted.
150 int hr = (int)encode_sRGB_uchar((float)rgb[0] * 0.75f);
151 int hg = (int)encode_sRGB_uchar((float)rgb[1] * 0.75f);
152 int hb = (int)encode_sRGB_uchar((float)rgb[2] * 0.75f);
153 _highlight_bg_brush = CreateSolidBrush(RGB(hr, hg, hb));
154
155 // Should our foreground be black or white?
156 double bright =
157 rgb[0] * 0.2126 +
158 rgb[1] * 0.7152 +
159 rgb[2] * 0.0722;
160
161 if (bright >= 0.5) {
162 _fg_color = RGB(0, 0, 0);
163 } else {
164 _fg_color = RGB(255, 255, 255);
165 }
166 if (bright * 0.75 >= 0.5) {
167 _highlight_fg_color = RGB(0, 0, 0);
168 } else {
169 _highlight_fg_color = RGB(255, 255, 255);
170 }
171
172 if (_window) {
173 InvalidateRect(_window, nullptr, TRUE);
174 }
175}

305LONG WinStatsLabel::
306window_proc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam) {
307 switch (msg) {
308 case WM_LBUTTONDBLCLK:
309 _graph->on_click_label(_collector_index);
310 return 0;
311
312 case WM_CONTEXTMENU:
313 _graph->on_popup_label(_collector_index);
314 return 0;
315
316 case WM_MOUSEMOVE:
317 {
318 // When the mouse enters the label area, highlight the label.
319 if (!_mouse_within) {
320 set_mouse_within(true);
321 _graph->on_enter_label(_collector_index);
322 }
323
324 // Now we want to get a WM_MOUSELEAVE when the mouse leaves the label.
325 TRACKMOUSEEVENT tme = {
326 sizeof(TRACKMOUSEEVENT),
327 TME_LEAVE,
328 _window,
329 0
330 };
331 TrackMouseEvent(&tme);
332 }
333 break;
334
335 case WM_MOUSELEAVE:
336 if (_mouse_within) {
337 set_mouse_within(false);
338 _graph->on_leave_label(_collector_index);
339 }
340 break;
341
342 case WM_PAINT:
343 {
344 PAINTSTRUCT ps;
345 HDC hdc = BeginPaint(hwnd, &ps);
346
347 RECT rect = { 0, 0, _width, _height };
348 FillRect(hdc, &rect, (_highlight || _mouse_within) ? _highlight_bg_brush : _bg_brush);
349
350 HFONT hfnt = _monitor->get_font();
351 SelectObject(hdc, hfnt);
352 SetTextAlign(hdc, TA_LEFT | TA_TOP | TA_NOUPDATECP);
353
354 SetBkMode(hdc, TRANSPARENT);
355 SetTextColor(hdc, (_highlight || _mouse_within) ? _highlight_fg_color : _fg_color);
356
357 if (_width > 8) {
358 UINT format = DT_END_ELLIPSIS | DT_SINGLELINE;
359 if (_align_right) {
360 format |= DT_RIGHT;
361 } else {
362 format |= DT_LEFT;
363 }
364
365 RECT margins = { _left_margin, _top_margin, _width - _right_margin, _height - _bottom_margin };
366 DrawText(hdc, _text.data(), _text.length(), &margins, format);
367 }
368
369 EndPaint(hwnd, &ps);
370 return 0;
371 }
372
373 case WM_NOTIFY:
374 switch (((LPNMHDR)lparam)->code) {
375 case TTN_GETDISPINFO:
376 {
377 NMTTDISPINFO &info = *(NMTTDISPINFO *)lparam;
378 _tooltip_text = _graph->get_label_tooltip(_collector_index);
379 info.lpszText = (char *)_tooltip_text.c_str();
380 }
381 return 0;
382 }
383 break;
384
385 default:
386 break;
387 }
388
389 return DefWindowProc(hwnd, msg, wparam, lparam);
