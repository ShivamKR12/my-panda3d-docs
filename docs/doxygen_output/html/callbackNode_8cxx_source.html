91bool CallbackNode::
92cull_callback(CullTraverser *trav, CullTraverserData &data) {
93 CallbackObject *cbobj = get_cull_callback();
94 if (cbobj != nullptr) {
95 NodeCullCallbackData cbdata(trav, data);
96 cbobj->do_callback(&cbdata);
97
98 // No further cull: the callback takes care of all of it.
99 return false;
100 }
101
102 // Recurse onto the node's children.
103 return true;
104}

113 if (pgraph_cat.is_spam()) {
