313PNMFileTypeTIFF::Reader::
314Reader(PNMFileType *type, istream *file, bool owns_file, string magic_number) :
315 PNMReader(type, file, owns_file)
316{
317 bool grayscale = false;
318 int numcolors;
319 int i;
320 unsigned short* redcolormap;
321 unsigned short* greencolormap;
322 unsigned short* bluecolormap;
323
324 // Hope we can putback() more than one character.
325 for (string::reverse_iterator mi = magic_number.rbegin();
326 mi != magic_number.rend();
327 mi++) {
328 _file->putback(*mi);
329 }
330 if (_file->fail()) {
331 pnmimage_tiff_cat.error()
332 << "Unable to put back magic number.\n";
333 _is_valid = false;
334 }
335
336 if (_is_valid) {
337 tif = TIFFClientOpen("TIFF file", "r",
338 (thandle_t) _file,
339 istream_read, istream_dont_write,
340 (TIFFSeekProc)istream_seek,
341 iostream_dont_close, istream_size,
342 iostream_map, iostream_unmap);
343
344 if ( tif == nullptr ) {
345 _is_valid = false;
346 }
347 }
348
349 if (_is_valid) {
350 if ( ! TIFFGetField( tif, TIFFTAG_SAMPLEFORMAT, &sample_format ) )
351 sample_format = SAMPLEFORMAT_UINT;
352 if ( ! TIFFGetField( tif, TIFFTAG_BITSPERSAMPLE, &bps ) )
353 bps = 1;
354 if ( ! TIFFGetField( tif, TIFFTAG_SAMPLESPERPIXEL, &spp ) )
355 spp = 1;
356
357 if ( ! TIFFGetField( tif, TIFFTAG_PHOTOMETRIC, &photomet ) ) {
358 pnmimage_tiff_cat.error()
359 << "Error getting photometric from TIFF file.\n";
360 _is_valid = false;
361
362 } else if (sample_format == SAMPLEFORMAT_IEEEFP) {
363 // Floating-point TIFF. We only accept 32-bit floats.
364 if (bps != 32) {
365 pnmimage_tiff_cat.error()
366 << "Can only read 32-bit float TIFF files, not " << bps << "-bit.\n";
367 _is_valid = false;
368 }
369
370 } else if (sample_format != SAMPLEFORMAT_UINT) {
371 // Can't understand other kinds of sample formats.
372 pnmimage_tiff_cat.error()
373 << "Can't understand sample format\n";
374 _is_valid = false;
375 }
376 }
377
378 if (_is_valid) {
379 unsigned short num_extra_samples;
380 unsigned short *extra_samples = nullptr;
381
382 if (!TIFFGetField(tif, TIFFTAG_EXTRASAMPLES, &num_extra_samples,
383 &extra_samples)) {
384 num_extra_samples = 0;
385 }
386 _num_channels = spp - num_extra_samples;
387 unassoc_alpha_sample = 0;
388 assoc_alpha_sample = 0;
389
390 if (_num_channels == 1 || _num_channels == 3) {
391 // Look for an alpha channel in one of the extra samples, if any.
392 bool got_alpha = false;
393 for (unsigned short s = 0; s < num_extra_samples && !got_alpha; s++) {
394 if (extra_samples[s] == EXTRASAMPLE_UNASSALPHA) {
395 unassoc_alpha_sample = s + _num_channels;
396 _num_channels++;
397 got_alpha = true;
398
399 } else if (extra_samples[s] == EXTRASAMPLE_ASSOCALPHA) {
400 assoc_alpha_sample = s + _num_channels;
401 _num_channels++;
402 got_alpha = true;
403 }
404 }
405
406 // Unfortunately, Photoshop seems to write EXTRASAMPLE_UNSPECIFIED into
407 // the EXTRASAMPLES field for its alpha channels. If we have exactly
408 // one extra channel and it's an UNSPECIFIED channel, assume it's meant
409 // to be alpha.
410 if (!got_alpha && num_extra_samples == 1 &&
411 extra_samples[0] == EXTRASAMPLE_UNSPECIFIED) {
412 unassoc_alpha_sample = _num_channels;
413 _num_channels++;
414 }
415
416 } else if ((_num_channels == 2 || _num_channels == 4) && num_extra_samples == 0) {
417 // If we have a 2- or 4-channel image but the extra samples are not
418 // declared, assume it was written out by a broken TIFF implementation
419 // and that the extra channel is really meant to be alpha.
420 unassoc_alpha_sample = _num_channels - 1;
421 if (pnmimage_tiff_cat.is_debug()) {
422 pnmimage_tiff_cat.debug()
423 << "Assuming last channel of " << spp
424 << "-color image is meant to represent alpha.\n";
425 }
426
427 } else {
428 pnmimage_tiff_cat.error()
429 << "Cannot handle " << spp << "-color image (with "
430 << num_extra_samples << " extra channels).\n";
431 _is_valid = false;
432 }
433 }
434
435 if (_is_valid) {
436 (void) TIFFGetField( tif, TIFFTAG_IMAGEWIDTH, &_x_size );
437 (void) TIFFGetField( tif, TIFFTAG_IMAGELENGTH, &_y_size );
438
439 if (pnmimage_tiff_cat.is_debug()) {
440 pnmimage_tiff_cat.debug()
441 << "Reading TIFF image: " << _x_size << " x " << _y_size << "\n"
442 << bps << " bits/sample, " << spp << " samples/pixel\n";
443 }
444
445 _maxval = ( 1 << bps ) - 1;
446 if ( _maxval == 1 && _num_channels == 1 ) {
447 if (pnmimage_tiff_cat.is_debug()) {
448 pnmimage_tiff_cat.debug(false)
449 << "monochrome\n";
450 }
451 grayscale = true;
452 } else {
453 switch ( photomet ) {
454 case PHOTOMETRIC_MINISBLACK:
455 if (pnmimage_tiff_cat.is_debug()) {
456 pnmimage_tiff_cat.debug(false)
457 << _maxval + 1 << " graylevels (min is black)\n";
458 }
459 grayscale = true;
460 break;
461
462 case PHOTOMETRIC_MINISWHITE:
463 if (pnmimage_tiff_cat.is_debug()) {
464 pnmimage_tiff_cat.debug(false)
465 << _maxval + 1 << " graylevels (min is white)\n";
466 }
467 grayscale = true;
468 break;
469
470 case PHOTOMETRIC_PALETTE:
471 if (pnmimage_tiff_cat.is_debug()) {
472 pnmimage_tiff_cat.debug(false)
473 << " colormapped\n";
474 }
475 if ( ! TIFFGetField( tif, TIFFTAG_COLORMAP, &redcolormap, &greencolormap, &bluecolormap ) ) {
476 pnmimage_tiff_cat.error()
477 << "Error getting colormap from TIFF file.\n";
478 _is_valid = false;
479 } else {
480 numcolors = _maxval + 1;
481 if ( numcolors > TIFF_COLORMAP_MAXCOLORS ) {
482 pnmimage_tiff_cat.error()
483 << "Cannot read TIFF file with " << numcolors
484 << " in colormap; max supported is " << TIFF_COLORMAP_MAXCOLORS << "\n";
485 _is_valid = false;
486 } else {
487 _maxval = PNM_MAXMAXVAL;
488 grayscale = false;
489 for ( i = 0; i < numcolors; ++i ) {
490 xelval r, g, b;
491 r = (xelval)(_maxval * (double)(redcolormap[i] / 65535.0));
492 g = (xelval)(_maxval * (double)(greencolormap[i] / 65535.0));
493 b = (xelval)(_maxval * (double)(bluecolormap[i] / 65535.0));
494 PPM_ASSIGN( colormap[i], r, g, b );
495 }
496 }
497 }
498 break;
499
500 case PHOTOMETRIC_RGB:
501 if (pnmimage_tiff_cat.is_debug()) {
502 pnmimage_tiff_cat.debug(false)
503 << "truecolor\n";
504 }
505 grayscale = false;
506 break;
507
508 case PHOTOMETRIC_MASK:
509 pnmimage_tiff_cat.error()
510 << "Don't know how to handle TIFF image with PHOTOMETRIC_MASK.\n";
511 _is_valid = false;
512 break;
513
514 case PHOTOMETRIC_DEPTH:
515 pnmimage_tiff_cat.error()
516 << "Don't know how to handle TIFF image with PHOTOMETRIC_DEPTH.\n";
517 _is_valid = false;
518 break;
519
520 default:
521 pnmimage_tiff_cat.error()
522 << "Unknown photometric " << photomet << " in TIFF image.\n";
523 _is_valid = false;
524 break;
525 }
526 }
527 }
528
529 if (_is_valid ) {
530 if ( _maxval > PNM_MAXMAXVAL ) {
531 pnmimage_tiff_cat.error()
532 << "Cannot read TIFF file with maxval of " << _maxval << "\n";
533 _is_valid = false;
534 }
535 }
536
537 if (_is_valid) {
538 if (grayscale && !is_grayscale()) {
539 _num_channels = (has_alpha()) ? 2 : 1;
540 } else if (!grayscale && is_grayscale()) {
541 _num_channels = (has_alpha()) ? 4 : 3;
542 }
543
544 current_row = 0;
545 }
546}

572bool PNMFileTypeTIFF::Reader::
573read_pfm(PfmFile &pfm) {
574 if (!is_valid()) {
575 return false;
576 }
577
578 int x_size = get_x_size();
579 int y_size = get_y_size();
580 int num_channels = get_num_channels();
581
582 pfm.clear(x_size, y_size, num_channels);
583 vector_float table;
584 pfm.swap_table(table);
585
586 for (int yi = 0; yi < y_size; ++yi) {
587 float *row = &table[(yi * x_size) * _num_channels];
588
589 if (TIFFReadScanline(tif, row, yi, 0 ) < 0 ) {
590 pnmimage_tiff_cat.error()
591 << "failed a scanline read on row " << yi << "\n";
592 pfm.swap_table(table);
593 return false;
594 }
595 }
596
597 pfm.swap_table(table);
598 return true;
599}

618bool PNMFileTypeTIFF::Reader::
619read_row(xel *row_data, xelval *alpha_data, int x_size, int) {
620 if (!is_valid()) {
621 return false;
622 }
623
624 size_t scanline_size = (size_t)TIFFScanlineSize(tif);
625 unsigned char *buf = (unsigned char*) alloca(scanline_size);
626
627 int col;
628 xelval gray, sample;
629 xelval r, g, b;
630
631 if ( TIFFReadScanline( tif, buf, current_row, 0 ) < 0 ) {
632 pnmimage_tiff_cat.error()
633 << "Bad data read on line " << current_row << " of TIFF image.\n";
634 return false;
635 }
636
637 unsigned char *buf_ptr = buf;
638 unsigned s;
639 int bits_left = 8;
640
641 // Get a pointer to a function that extracts the next bps-bit sample from
642 // the bitarray. There are a handful of different functions, which are
643 // optimized for different values of bps.
644 xelval (PNMFileTypeTIFF::Reader::*next_sample)(unsigned char *&buf_ptr, int &bits_left) const;
645
646 if (bps < 8) {
647 next_sample = &PNMFileTypeTIFF::Reader::next_sample_lt_8;
648
649 } else if (bps == 8) {
650 next_sample = &PNMFileTypeTIFF::Reader::next_sample_8;
651
652 } else if (bps == 16) {
653 next_sample = &PNMFileTypeTIFF::Reader::next_sample_16;
654
655 } else if (bps == 32) {
656 // Actually, it's not likely that a 32-bit sample will fit within a
657 // xelval. Deal with this when we come to it.
658 next_sample = &PNMFileTypeTIFF::Reader::next_sample_32;
659
660 } else {
661 next_sample = &PNMFileTypeTIFF::Reader::next_sample_general;
662 }
663
664 switch ( photomet ) {
665 case PHOTOMETRIC_MINISBLACK:
666 for ( col = 0; col < x_size; ++col )
667 {
668 sample = (this->*next_sample)(buf_ptr, bits_left);
669 gray = sample;
670
671 for (s = 1; s < spp; s++) {
672 sample = (this->*next_sample)(buf_ptr, bits_left);
673 if (s == unassoc_alpha_sample) {
674 alpha_data[col] = sample;
675
676 } else if (s == assoc_alpha_sample) {
677 alpha_data[col] = sample;
678 if (sample != 0) {
679 gray = (xelval)((float)gray * _maxval / (float)sample);
680 }
681 }
682 }
683 PPM_PUTB(row_data[col], gray);
684 }
685 break;
686
687 case PHOTOMETRIC_MINISWHITE:
688 for ( col = 0; col < x_size; ++col )
689 {
690 sample = (this->*next_sample)(buf_ptr, bits_left);
691 gray = _maxval - sample;
692 for (s = 1; s < spp; s++) {
693 sample = (this->*next_sample)(buf_ptr, bits_left);
694 sample = _maxval - sample;
695
696 if (s == unassoc_alpha_sample) {
697 alpha_data[col] = sample;
698
699 } else if (s == assoc_alpha_sample) {
700 alpha_data[col] = sample;
701 if (sample != 0) {
702 gray = (xelval)((float)gray * _maxval / (float)sample);
703 }
704 }
705 }
706
707 PPM_PUTB(row_data[col], gray);
708 }
709 break;
710
711 case PHOTOMETRIC_PALETTE:
712 for ( col = 0; col < x_size; ++col )
713 {
714 sample = (this->*next_sample)(buf_ptr, bits_left);
715 row_data[col] = colormap[sample];
716
717 for (s = 1; s < spp; s++) {
718 sample = (this->*next_sample)(buf_ptr, bits_left);
719 if (s == unassoc_alpha_sample) {
720 alpha_data[col] = sample;
721
722 } else if (s == assoc_alpha_sample) {
723 alpha_data[col] = sample;
724 if (sample != 0) {
725 r = PPM_GETR(row_data[col]);
726 g = PPM_GETG(row_data[col]);
727 b = PPM_GETB(row_data[col]);
728 r = (xelval)((float)r * _maxval / (float)sample);
729 g = (xelval)((float)g * _maxval / (float)sample);
730 b = (xelval)((float)b * _maxval / (float)sample);
731 PPM_ASSIGN(row_data[col], r, g, b);
732 }
733 }
734 }
735 }
736 break;
737
738 case PHOTOMETRIC_RGB:
739 for ( col = 0; col < x_size; ++col ) {
740 sample = (this->*next_sample)(buf_ptr, bits_left);
741 r = sample;
742 sample = (this->*next_sample)(buf_ptr, bits_left);
743 g = sample;
744 sample = (this->*next_sample)(buf_ptr, bits_left);
745 b = sample;
746
747 for (s = 3; s < spp; s++) {
748 sample = (this->*next_sample)(buf_ptr, bits_left);
749 if (s == unassoc_alpha_sample) {
750 alpha_data[col] = sample;
751
752 } else if (s == assoc_alpha_sample) {
753 alpha_data[col] = sample;
754 if (sample != 0) {
755 r = (xelval)((float)r * _maxval / (float)sample);
756 g = (xelval)((float)g * _maxval / (float)sample);
757 b = (xelval)((float)b * _maxval / (float)sample);
758 }
759 }
760 }
761
762 PPM_ASSIGN(row_data[col], r, g, b);
763 }
764 break;
765
766 default:
767 pnmimage_tiff_cat.error()
768 << "Internal error: unsupported photometric " << photomet << "\n";
769 return false;
770 }
771
772 nassertr(buf_ptr <= buf + scanline_size, false);
773
774 current_row++;
775 return true;
776}

818next_sample_32(unsigned char *&buf_ptr, int &bits_left) const {
819 // The TIFF library has already byte-swapped the values if necessary. Thus,
820 // we only need to treat it as an array of longs.
821 unsigned long result = *(unsigned long *)buf_ptr;
822 buf_ptr += 2;
823 return (xelval)result;
824}

830xelval PNMFileTypeTIFF::Reader::
831next_sample_general(unsigned char *&buf_ptr, int &bits_left) const {
832 unsigned int result = 0;
833 int bits_needed = bps;
834
835 while (bits_needed > 0) {
836 nassertr(bits_left >= 0, 0);
837 if (bits_left == 0) {
838 ++buf_ptr;
839 bits_left = 8;
840 }
841
842 if (bits_needed <= bits_left) {
843 bits_left -= bits_needed;
844 unsigned int mask = (1 << bits_needed) - 1;
845 result |= ((*buf_ptr) >> bits_left) & mask;
846 bits_needed = 0;
847
848 } else {
849 bits_needed -= bits_left;
850 unsigned int mask = (1 << bits_left) - 1;
851 result |= ((*buf_ptr) & mask) << bits_needed;
852 bits_left = 0;
853 }
854 }
855
856 return result;
857}

931 const float *row = &table[(yi * x_size) * _num_channels];

962int PNMFileTypeTIFF::Writer::
963write_data(xel *array, xelval *alpha) {
964 colorhist_vector chv = nullptr;
965 colorhash_table cht;
966 unsigned short
967 red[TIFF_COLORMAP_MAXCOLORS],
968 grn[TIFF_COLORMAP_MAXCOLORS],
969 blu[TIFF_COLORMAP_MAXCOLORS];
970 int row, colors, i;
971 int col;
972 int grayscale = false;
973 struct tiff * tif;
974 short photometric = 0;
975 short samplesperpixel = 0;
976 unsigned short extra_samples[1] = { EXTRASAMPLE_UNASSALPHA };
977 short bitspersample = 0;
978 int bytesperrow = 0;
979 unsigned char* buf;
980 unsigned char* tP;
981
982 switch ( get_color_type() ) {
983 case CT_color:
984 // This call is a bit of fakery to convert our proper 2-d array of xels to
985 // an indirect 2-d array of pixels. We make it look like a single row of
986 // _x_size * _y_size pixels.
987
988 // We can't actually write palettes bigger than 256 colors, regardless of
989 // the number of colors we can read.
990 chv = ppm_computecolorhist( (pixel **)&array, _x_size * _y_size, 1,
991 256, &colors );
992 if ( chv == nullptr ) {
993 pnmimage_tiff_cat.debug()
994 << colors << " colors found; too many for a palette.\n"
995 << "Writing a 24-bit RGB file.\n";
996 grayscale = false;
997 } else {
998 pnmimage_tiff_cat.debug()
999 << colors << " colors found; writing an 8-bit palette file.\n";
1000 grayscale = true;
1001 for ( i = 0; i < colors; ++i ) {
1002 xelval r, g, b;
1003
1004 r = PPM_GETR( chv[i].color );
1005 g = PPM_GETG( chv[i].color );
1006 b = PPM_GETB( chv[i].color );
1007 if ( r != g || g != b ) {
1008 grayscale = false;
1009 break;
1010 }
1011 }
1012 }
1013 break;
1014
1015 case CT_two_channel: // We don't yet support two-channel output for TIFF's.
1016 case CT_four_channel:
1017 chv = nullptr;
1018 grayscale = false;
1019 break;
1020
1021 case CT_grayscale:
1022 chv = nullptr;
1023 grayscale = true;
1024 break;
1025
1026 default:
1027 break;
1028 }
1029
1030 /* Open output file. */
1031 tif = TIFFClientOpen("TIFF file", "w",
1032 (thandle_t) _file,
1033 ostream_dont_read, ostream_write,
1034 (TIFFSeekProc)ostream_seek,
1035 iostream_dont_close, ostream_size,
1036 iostream_map, iostream_unmap);
1037 if ( tif == nullptr ) {
1038 return 0;
1039 }
1040
1041 /* Figure out TIFF parameters. */
1042 switch ( get_color_type() ) {
1043 case CT_color:
1044 case CT_four_channel:
1045 if ( chv == nullptr ) {
1046 samplesperpixel = _num_channels;
1047 bitspersample = 8;
1048 photometric = PHOTOMETRIC_RGB;
1049 bytesperrow = _x_size * samplesperpixel;
1050 } else if ( grayscale ) {
1051 samplesperpixel = 1;
1052 bitspersample = std::min(8, pm_maxvaltobits(_maxval));
1053 photometric = PHOTOMETRIC_MINISBLACK;
1054 i = 8 / bitspersample;
1055 bytesperrow = ( _x_size + i - 1 ) / i;
1056 } else {
1057 samplesperpixel = 1;
1058 bitspersample = std::min(8, pm_maxvaltobits(_maxval));
1059 photometric = PHOTOMETRIC_PALETTE;
1060 bytesperrow = _x_size;
1061 }
1062 break;
1063
1064 case CT_grayscale:
1065 case CT_two_channel:
1066 samplesperpixel = _num_channels;
1067 bitspersample = pm_maxvaltobits( _maxval );
1068 photometric = PHOTOMETRIC_MINISBLACK;
1069 i = 8 / bitspersample;
1070 bytesperrow = ( _x_size + i - 1 ) / i;
1071 break;
1072
1073 default:
1074 break;
1075 }
1076
1077 buf = (unsigned char*) alloca( bytesperrow );
1078 if ( buf == nullptr ) {
1079 pnmimage_tiff_cat.error()
1080 << "Can't allocate memory for row buffer\n";
1081 return 0;
1082 }
1083
1084 /* Set TIFF parameters. */
1085 TIFFSetField( tif, TIFFTAG_IMAGEWIDTH, _x_size );
1086 TIFFSetField( tif, TIFFTAG_IMAGELENGTH, _y_size );
1087 TIFFSetField( tif, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_UINT );
1088 TIFFSetField( tif, TIFFTAG_BITSPERSAMPLE, bitspersample );
1089 TIFFSetField( tif, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT );
1090 TIFFSetField( tif, TIFFTAG_COMPRESSION, tiff_compression );
1091 if ( tiff_compression == COMPRESSION_CCITTFAX3 && tiff_g3options != 0 )
1092 TIFFSetField( tif, TIFFTAG_GROUP3OPTIONS, tiff_g3options );
1093 if ( tiff_compression == COMPRESSION_LZW && tiff_predictor != 0 )
1094 TIFFSetField( tif, TIFFTAG_PREDICTOR, tiff_predictor );
1095 TIFFSetField( tif, TIFFTAG_PHOTOMETRIC, photometric );
1096 TIFFSetField( tif, TIFFTAG_FILLORDER, tiff_fillorder );
1097 // TIFFSetField( tif, TIFFTAG_DOCUMENTNAME, "TIFF Image File");
1098 TIFFSetField( tif, TIFFTAG_IMAGEDESCRIPTION,
1099 "Generated via pnmimage.\n" );
1100 TIFFSetField( tif, TIFFTAG_SAMPLESPERPIXEL, samplesperpixel );
1101 if (has_alpha()) {
1102 TIFFSetField(tif, TIFFTAG_EXTRASAMPLES, 1, extra_samples);
1103 }
1104 TIFFSetField( tif, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG );
1105
1106 if ( chv == nullptr ) {
1107 cht = nullptr;
1108 } else {
1109 /* Make TIFF colormap. */
1110 for ( i = 0; i < colors; ++i ) {
1111 red[i] = (unsigned short) (PPM_GETR( chv[i].color ) * 65535L / _maxval);
1112 grn[i] = (unsigned short) (PPM_GETG( chv[i].color ) * 65535L / _maxval);
1113 blu[i] = (unsigned short) (PPM_GETB( chv[i].color ) * 65535L / _maxval);
1114 }
1115 TIFFSetField( tif, TIFFTAG_COLORMAP, red, grn, blu );
1116
1117 /* Convert color vector to color hash table, for fast lookup. */
1118 cht = ppm_colorhisttocolorhash( chv, colors );
1119 ppm_freecolorhist( chv );
1120 }
1121
1122 /* Now write the TIFF data. */
1123 for ( row = 0; row < _y_size; ++row ) {
1124 xel *row_data = array + row*_x_size;
1125 xelval *alpha_data = alpha + row*_x_size;
1126
1127 if ( !is_grayscale() && ! grayscale ) {
1128 if ( cht == nullptr ) {
1129 tP = buf;
1130 for ( col = 0; col < _x_size; ++col ) {
1131 *tP++ = (unsigned char)(255 * PPM_GETR(row_data[col]) / _maxval);
1132 *tP++ = (unsigned char)(255 * PPM_GETG(row_data[col]) / _maxval);
1133 *tP++ = (unsigned char)(255 * PPM_GETB(row_data[col]) / _maxval);
1134 if (samplesperpixel==4) {
1135 *tP++ = (unsigned char)(255 * alpha_data[col] / _maxval);
1136 }
1137 }
1138 } else {
1139 tP = buf;
1140 for ( col = 0; col < _x_size; ++col ) {
1141 int s;
1142
1143 s = ppm_lookupcolor( cht, (pixel *)(&row_data[col]) );
1144 if ( s == -1 ) {
1145 pnmimage_tiff_cat.error()
1146 << "Internal error: color not found?!? row=" << row
1147 << " col=" << col << "\n";
1148 if (cht != nullptr) {
1149 ppm_freecolorhash(cht);
1150 }
1151 return 0;
1152 }
1153 *tP++ = (unsigned char) s;
1154 if (samplesperpixel==2) {
1155 *tP++ = (unsigned char)(255 * alpha_data[col] / _maxval);
1156 }
1157 }
1158 }
1159 } else {
1160 xelval bigger_maxval;
1161 int bitshift;
1162 unsigned char byte;
1163 xelval s;
1164
1165 bigger_maxval = pm_bitstomaxval( bitspersample );
1166 bitshift = 8 - bitspersample;
1167 byte = 0;
1168 tP = buf;
1169 for ( col = 0; col < _x_size; ++col ) {
1170 s = PPM_GETB(row_data[col]);
1171 if ( _maxval != bigger_maxval )
1172 s = (xelval)((long) s * bigger_maxval / _maxval);
1173 byte |= s << bitshift;
1174 bitshift -= bitspersample;
1175 if ( bitshift < 0 ) {
1176 *tP++ = byte;
1177 bitshift = 8 - bitspersample;
1178 byte = 0;
1179 }
1180 }
1181 if ( bitshift != 8 - bitspersample )
1182 *tP++ = byte;
1183 }
1184
1185 if ( TIFFWriteScanline( tif, buf, row, 0 ) < 0 ) {
1186 pnmimage_tiff_cat.error()
1187 << "failed a scanline write on row " << row << "\n";
1188 return row;
1189 }
1190 }
1191 TIFFFlushData( tif );
1192 TIFFClose( tif );
1193
1194 if (cht != nullptr) {
1195 ppm_freecolorhash(cht);
1196 }
1197
1198 return _y_size;
1199}
