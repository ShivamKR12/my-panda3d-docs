114void CompassEffect::
115cull_callback(CullTraverser *trav, CullTraverserData &data,
116 CPT(TransformState) &node_transform,
117 CPT(RenderState) &) const {
118 if (_properties == 0) {
119 // Nothing to do.
120 return;
121 }
122
123 if (data._instances == nullptr) {
124 CPT(TransformState) true_net_transform = data.get_net_transform(trav);
125 CPT(TransformState) want_net_transform = true_net_transform;
126 adjust_transform(want_net_transform, node_transform, data.node());
127
128 // Now compute the transform that will convert true_net_transform to
129 // want_transform. This is inv(true_net_transform) * want_transform.
130 CPT(TransformState) compass_transform =
131 true_net_transform->invert_compose(want_net_transform);
132
133 // And modify our local node's apparent transform so that
134 // true_net_transform->compose(new_node_transform) produces the same result
135 // we would have gotten had we actually computed
136 // want_transform->compose(orig_node_transform).
137 node_transform = compass_transform->compose(node_transform);
138 }
139 else {
140 // Compute the billboard effect for every instance individually.
141 InstanceList *instances = new InstanceList(*data._instances);
142 data._instances = instances;
143
144 CPT(TransformState) parent_net_transform = data.get_net_transform(trav);
145 CPT(TransformState) invert_net_transform = parent_net_transform->get_inverse();
146
147 // We make use of the fact that we know adjust_transform() does not modify
148 // its node_transform parameter.
149 CPT(TransformState) node_transform_copy = node_transform;
150 if (node_transform_copy->is_identity()) {
151 // Slightly optimized case.
152 for (InstanceList::Instance &instance : *instances) {
153 CPT(TransformState) true_net_transform = parent_net_transform->compose(instance.get_transform());
154 CPT(TransformState) want_net_transform = true_net_transform;
155 adjust_transform(want_net_transform, node_transform_copy, data.node());
156
157 instance.set_transform(invert_net_transform->compose(want_net_transform));
158 }
159 }
160 else {
161 // We apply the node_transform to the instances.
162 node_transform = TransformState::make_identity();
163
164 for (InstanceList::Instance &instance : *instances) {
165 CPT(TransformState) true_net_transform = parent_net_transform->compose(instance.get_transform());
166 CPT(TransformState) want_net_transform = true_net_transform;
167 adjust_transform(want_net_transform, node_transform_copy, data.node());
168
169 instance.set_transform(invert_net_transform->compose(want_net_transform)->compose(node_transform_copy));
170 }
171 }
172 }

193void CompassEffect::
194adjust_transform(CPT(TransformState) &net_transform,
195 CPT(TransformState) &node_transform,
196 const PandaNode *) const {
197 if (_properties == 0) {
198 // Nothing to do.
199 return;
200 }
201
202 // The reference transform: where we are acting as if we inherit from.
203 // Either the root node (identity) or the specified reference node.
204 CPT(TransformState) ref_transform;
205 if (_reference.is_empty()) {
206 ref_transform = TransformState::make_identity();
207 } else {
208 ref_transform = _reference.get_net_transform();
209 }
210
211 // Now compute the net transform we actually want to achieve. This is all
212 // of the components from the net transform we want to inherit normally from
213 // our parent, with all of the components from the ref transform we want to
214 // inherit from our reference.
215 CPT(TransformState) want_net_transform;
216 if (_properties == P_all) {
217 // If we want to steal the whole transform, that's easy.
218 want_net_transform = ref_transform;
219
220 } else {
221 // How much of the pos do we want to steal? We can always determine a
222 // transform's pos, even if it's nondecomposable.
223 LVecBase3 want_pos = net_transform->get_pos();
224 const LVecBase3 &ref_pos = ref_transform->get_pos();
225 if ((_properties & P_x) != 0) {
226 want_pos[0] = ref_pos[0];
227 }
228 if ((_properties & P_y) != 0) {
229 want_pos[1] = ref_pos[1];
230 }
231 if ((_properties & P_z) != 0) {
232 want_pos[2] = ref_pos[2];
233 }
234
235 if ((_properties & ~P_pos) == 0) {
236 // If we only want to steal the pos, that's pretty easy.
237 want_net_transform = net_transform->set_pos(want_pos);
238
239 } else if ((_properties & (P_rot | P_scale)) == (P_rot | P_scale)) {
240 // If we want to steal everything *but* the pos, also easy.
241 want_net_transform = ref_transform->set_pos(want_pos);
242
243 } else {
244 // For any other combination, we have to be able to decompose both
245 // transforms.
246 if (!net_transform->has_components() ||
247 !ref_transform->has_components()) {
248 // If we can't decompose, just do the best we can: steal everything
249 // but the pos.
250 want_net_transform = ref_transform->set_pos(want_pos);
251
252 } else {
253 // If we can decompose, then take only the components we want.
254 LQuaternion want_quat = net_transform->get_quat();
255 if ((_properties & P_rot) != 0) {
256 want_quat = ref_transform->get_quat();
257 }
258
259 LVecBase3 want_scale = net_transform->get_scale();
260 const LVecBase3 &ref_scale = ref_transform->get_scale();
261 if ((_properties & P_sx) != 0) {
262 want_scale[0] = ref_scale[0];
263 }
264 if ((_properties & P_sy) != 0) {
265 want_scale[1] = ref_scale[1];
266 }
267 if ((_properties & P_sz) != 0) {
268 want_scale[2] = ref_scale[2];
269 }
270
271 want_net_transform =
272 TransformState::make_pos_quat_scale(want_pos, want_quat, want_scale);
273 }
274 }
275 }
276
277 net_transform = want_net_transform;
278}
