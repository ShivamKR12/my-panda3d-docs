113int TransformState::
114compare_to(const TransformState &other, bool uniquify_matrix) const {
115 static const int significant_flags =
116 (F_is_invalid | F_is_identity | F_components_given | F_hpr_given | F_quat_given | F_is_2d);
117
118 int flags = (_flags & significant_flags);
119 int other_flags = (other._flags & significant_flags);
120 if (flags != other_flags) {
121 return flags < other_flags ? -1 : 1;
122 }
123
124 if ((_flags & (F_is_invalid | F_is_identity)) != 0) {
125 // All invalid transforms are equivalent to each other, and all identity
126 // transforms are equivalent to each other.
127 return 0;
128 }
129
130 if ((_flags & F_components_given) != 0) {
131 // If the transform was specified componentwise, compare them
132 // componentwise.
133 int c = _pos.compare_to(other._pos);
134 if (c != 0) {
135 return c;
136 }
137
138 if ((_flags & F_hpr_given) != 0) {
139 c = _hpr.compare_to(other._hpr);
140 if (c != 0) {
141 return c;
142 }
143 } else if ((_flags & F_quat_given) != 0) {
144 c = _quat.compare_to(other._quat);
145 if (c != 0) {
146 return c;
147 }
148 }
149
150 c = _scale.compare_to(other._scale);
151 if (c != 0) {
152 return c;
153 }
154
155 c = _shear.compare_to(other._shear);
156 return c;
157 }
158
159 // Otherwise, compare the matrices . . .
160 if (uniquify_matrix) {
161 // . . . but only if the user thinks that's a worthwhile comparison.
162 return get_mat().compare_to(other.get_mat());
163
164 } else {
165 // If not, we just compare the pointers.
166 if (this != &other) {
167 return (this < &other) ? -1 : 1;
168 }
169 return 0;
170 }

185 (F_is_invalid | F_is_identity | F_components_given | F_hpr_given | F_quat_given | F_is_2d);
186

190 return false;

194 // All invalid transforms are equivalent to each other, and all identity
195 // transforms are equivalent to each other.
196 return true;
197 }
198

200 // If the transform was specified componentwise, compare them
201 // componentwise.
202 if (_pos != other._pos) {
203 return false;
204 }
205

207 if (_hpr != other._hpr) {
208 return false;
209 }

211 if (_quat != other._quat) {
212 return false;
213 }
214 }
215
216 if (_scale != other._scale) {
217 return false;
218 }
219

221 }
222
223 // Otherwise, compare the matrices . . .
224 if (_uniquify_matrix) {
225 // . . . but only if the user thinks that's a worthwhile comparison.

550CPT(TransformState) TransformState::
551compose(const TransformState *other) const {
552 // We handle identity as a trivial special case.
553 if (is_identity()) {
554 return other;
555 }
556 if (other->is_identity()) {
557 return this;
558 }
559
560 // If either transform is invalid, the result is invalid.
561 if (is_invalid()) {
562 return this;
563 }
564 if (other->is_invalid()) {
565 return other;
566 }
567
568 if (!transform_cache) {
569 return do_compose(other);
570 }
571
572 LightReMutexHolder holder(*_states_lock);
573
574 // Is this composition already cached?
575 int index = _composition_cache.find(other);
576 if (index != -1) {
577 const Composition &comp = _composition_cache.get_data(index);
578 if (comp._result != nullptr) {
579 // Success!
580 _cache_stats.inc_hits();
581 return comp._result;
582 }
583 }
584
585 // Not in the cache. Compute a new result. It's important that we don't
586 // hold the lock while we do this, or we lose the benefit of
587 // parallelization.
588 CPT(TransformState) result = do_compose(other);
589
590 if (index != -1) {
591 Composition &comp = _composition_cache.modify_data(index);
592 // Well, it wasn't cached already, but we already had an entry (probably
593 // created for the reverse direction), so use the same entry to store
594 // the new result.
595 comp._result = result;
596
597 if (result != (const TransformState *)this) {
598 // See the comments below about the need to up the reference count
599 // only when the result is not the same as this.
600 result->cache_ref();
601 }
602 // Here's the cache!
603 _cache_stats.inc_hits();
604 return result;
605 }
606 _cache_stats.inc_misses();
607
608 // We need to make a new cache entry, both in this object and in the other
609 // object. We make both records so the other TransformState object will
610 // know to delete the entry from this object when it destructs, and vice-
611 // versa.
612
613 // The cache entry in this object is the only one that indicates the result;
614 // the other will be NULL for now.
615 _cache_stats.add_total_size(1);
616 _cache_stats.inc_adds(_composition_cache.is_empty());
617
618 _composition_cache[other]._result = result;
619
620 if (other != this) {
621 _cache_stats.add_total_size(1);
622 _cache_stats.inc_adds(other->_composition_cache.is_empty());
623 other->_composition_cache[this]._result = nullptr;
624 }
625
626 if (result != (TransformState *)this) {
627 // If the result of do_compose() is something other than this, explicitly
628 // increment the reference count. We have to be sure to decrement it
629 // again later, when the composition entry is removed from the cache.
630 result->cache_ref();
631
632 // (If the result was just this again, we still store the result, but we
633 // don't increment the reference count, since that would be a self-
634 // referential leak.)
635 }
636
637 _cache_stats.maybe_report("TransformState");
638
639 return result;
640}

649CPT(TransformState) TransformState::
650invert_compose(const TransformState *other) const {
651 // This method isn't strictly const, because it updates the cache, but we
652 // pretend that it is because it's only a cache which is transparent to the
653 // rest of the interface.
654
655 // We handle identity as a trivial special case.
656 if (is_identity()) {
657 return other;
658 }

660 // trivial for invert_compose().
661
662 // If either transform is invalid, the result is invalid.
663 if (is_invalid()) {
664 return this;
665 }
666 if (other->is_invalid()) {
667 return other;
668 }
669

725

727 _cache_stats.add_total_size(1);
728 _cache_stats.inc_adds(other->_invert_composition_cache.is_empty());
729 other->_invert_composition_cache[this]._result = nullptr;
730 }
731

734 // increment the reference count. We have to be sure to decrement it
735 // again later, when the composition entry is removed from the cache.

752bool TransformState::
753unref() const {
754 if (garbage_collect_states || !transform_cache) {
755 // If we're not using the cache at all, or if we're relying on garbage
756 // collection, just allow the pointer to unref normally.
757 return ReferenceCount::unref();
758 }
759
760 // Here is the normal refcounting case, with a normal cache, and without
761 // garbage collection in effect. In this case we will pull the object out
762 // of the cache when its reference count goes to 0.
763
764 // We always have to grab the lock, since we will definitely need to be
765 // holding it if we happen to drop the reference count to 0. Having to grab
766 // the lock at every call to unref() is a big limiting factor on
767 // parallelization.
768 LightReMutexHolder holder(*_states_lock);
769
770 if (auto_break_cycles && uniquify_transforms) {
771 if (get_cache_ref_count() > 0 &&
772 get_ref_count() == get_cache_ref_count() + 1) {
773 // If we are about to remove the one reference that is not in the cache,
774 // leaving only references in the cache, then we need to check for a
775 // cycle involving this TransformState and break it if it exists.
776 ((TransformState *)this)->detect_and_break_cycles();
777 }
778 }
779
780 if (ReferenceCount::unref()) {
781 // The reference count is still nonzero.
782 return true;
783 }
784
785 // The reference count has just reached zero. Make sure the object is
786 // removed from the global object pool, before anyone else finds it and
787 // tries to ref it.
788 ((TransformState *)this)->release_new();
789 ((TransformState *)this)->remove_cache_pointers();
790
791 return false;
792}

798bool TransformState::
799validate_composition_cache() const {
800 LightReMutexHolder holder(*_states_lock);
801
802 size_t size = _composition_cache.get_num_entries();
803 for (size_t i = 0; i < size; ++i) {
804 const TransformState *source = _composition_cache.get_key(i);
805 if (source != nullptr) {
806 // Check that the source also has a pointer back to this one. We always
807 // add entries to the composition cache in pairs.
808 int ri = source->_composition_cache.find(this);
809 if (ri == -1) {
810 // Failure! There is no back-pointer.
811 pgraph_cat.error()
812 << "TransformState::composition cache is inconsistent!\n";
813 pgraph_cat.error(false)
814 << *this << " compose " << *source << "\n";
815 pgraph_cat.error(false)
816 << "but no reverse\n";
817 return false;
818 }
819 }
820 }
821
822 size = _invert_composition_cache.get_num_entries();
823 for (size_t i = 0; i < size; ++i) {
824 const TransformState *source = _invert_composition_cache.get_key(i);
825 if (source != nullptr) {
826 // Check that the source also has a pointer back to this one. We always
827 // add entries to the composition cache in pairs.
828 int ri = source->_invert_composition_cache.find(this);
829 if (ri == -1) {
830 // Failure! There is no back-pointer.
831 pgraph_cat.error()
832 << "TransformState::invert composition cache is inconsistent!\n";
833 pgraph_cat.error(false)
834 << *this << " invert compose " << *source << "\n";
835 pgraph_cat.error(false)
836 << "but no reverse\n";

838 }
839 }

975int TransformState::
976get_num_unused_states() {
977 LightReMutexHolder holder(*_states_lock);
978
979 // First, we need to count the number of times each TransformState object is
980 // recorded in the cache. We could just trust get_cache_ref_count(), but
981 // we'll be extra cautious for now.
982 typedef pmap<const TransformState *, int> StateCount;
983 StateCount state_count;
984
985 size_t size = _states.get_num_entries();
986 for (size_t si = 0; si < size; ++si) {
987 const TransformState *state = _states.get_key(si);
988
989 std::pair<StateCount::iterator, bool> ir =
990 state_count.insert(StateCount::value_type(state, 1));
991 if (!ir.second) {
992 // If the above insert operation fails, then it's already in the
993 // cache; increment its value.
994 (*(ir.first)).second++;
995 }
996
997 size_t i;
998 size_t cache_size = state->_composition_cache.get_num_entries();
999 for (i = 0; i < cache_size; ++i) {
1000 const TransformState *result = state->_composition_cache.get_data(i)._result;
1001 if (result != nullptr && result != state) {
1002 // Here's a TransformState that's recorded in the cache. Count it.
1003 std::pair<StateCount::iterator, bool> ir =
1004 state_count.insert(StateCount::value_type(result, 1));
1005 if (!ir.second) {
1006 // If the above insert operation fails, then it's already in the
1007 // cache; increment its value.
1008 (*(ir.first)).second++;
1009 }
1010 }
1011 }
1012 cache_size = state->_invert_composition_cache.get_num_entries();
1013 for (i = 0; i < cache_size; ++i) {
1014 const TransformState *result = state->_invert_composition_cache.get_data(i)._result;
1015 if (result != nullptr && result != state) {
1016 std::pair<StateCount::iterator, bool> ir =
1017 state_count.insert(StateCount::value_type(result, 1));
1018 if (!ir.second) {
1019 (*(ir.first)).second++;
1020 }
1021 }
1022 }
1023 }
1024
1025 // Now that we have the appearance count of each TransformState object, we
1026 // can tell which ones are unreferenced outside of the TransformState cache,
1027 // by comparing these to the reference counts.
1028 int num_unused = 0;
1029
1030 StateCount::iterator sci;
1031 for (sci = state_count.begin(); sci != state_count.end(); ++sci) {
1032 const TransformState *state = (*sci).first;
1033 int count = (*sci).second;
1034 nassertr(count == state->get_cache_ref_count(), num_unused);
1035 nassertr(count <= state->get_ref_count(), num_unused);
1036 if (count == state->get_ref_count()) {
1037 num_unused++;
1038
1039 if (pgraph_cat.is_debug()) {
1040 pgraph_cat.debug()
1041 << "Unused state: " << (void *)state << ":"
1042 << state->get_ref_count() << " =\n";
1043 state->write(pgraph_cat.debug(false), 2);
1044 }
1045 }
1046 }
1047
1048 return num_unused;
1049}

1065int TransformState::
1066clear_cache() {
1067 LightReMutexHolder holder(*_states_lock);
1068
1069 PStatTimer timer(_cache_update_pcollector);
1070 int orig_size = _states.get_num_entries();
1071
1072 // First, we need to copy the entire set of states to a temporary vector,
1073 // reference-counting each object. That way we can walk through the copy,
1074 // without fear of dereferencing (and deleting) the objects in the map as we
1075 // go.
1076 {
1077 typedef pvector< CPT(TransformState) > TempStates;
1078 TempStates temp_states;
1079 temp_states.reserve(orig_size);
1080
1081 size_t size = _states.get_num_entries();
1082 for (size_t si = 0; si < size; ++si) {
1083 const TransformState *state = _states.get_key(si);
1084 temp_states.push_back(state);
1085 }
1086
1087 // Now it's safe to walk through the list, destroying the cache within
1088 // each object as we go. Nothing will be destructed till we're done.
1089 TempStates::iterator ti;
1090 for (ti = temp_states.begin(); ti != temp_states.end(); ++ti) {
1091 TransformState *state = (TransformState *)(*ti).p();
1092
1093 size_t i;
1094 size_t cache_size = state->_composition_cache.get_num_entries();
1095 for (i = 0; i < cache_size; ++i) {
1096 const TransformState *result = state->_composition_cache.get_data(i)._result;
1097 if (result != nullptr && result != state) {
1098 result->cache_unref();
1099 nassertr(result->get_ref_count() > 0, 0);
1100 }
1101 }
1102 _cache_stats.add_total_size(-(int)state->_composition_cache.get_num_entries());
1103 state->_composition_cache.clear();
1104
1105 cache_size = state->_invert_composition_cache.get_num_entries();
1106 for (i = 0; i < cache_size; ++i) {
1107 const TransformState *result = state->_invert_composition_cache.get_data(i)._result;
1108 if (result != nullptr && result != state) {
1109 result->cache_unref();
1110 nassertr(result->get_ref_count() > 0, 0);
1111 }
1112 }
1113 _cache_stats.add_total_size(-(int)state->_invert_composition_cache.get_num_entries());
1114 state->_invert_composition_cache.clear();
1115 }
1116
1117 // Once this block closes and the temp_states object goes away, all the
1118 // destruction will begin. Anything whose reference was held only within
1119 // the various objects' caches will go away.
1120 }
1121
1122 int new_size = _states.get_num_entries();
1123 return orig_size - new_size;
1124}

1132int TransformState::
1133garbage_collect() {
1134 if (!garbage_collect_states) {
1135 return 0;
1136 }
1137
1138 LightReMutexHolder holder(*_states_lock);
1139
1140 PStatTimer timer(_garbage_collect_pcollector);
1141 size_t orig_size = _states.get_num_entries();
1142
1143 // How many elements to process this pass?
1144 size_t size = orig_size;
1145 size_t num_this_pass = std::max(0, int(size * garbage_collect_states_rate));
1146 if (num_this_pass <= 0) {
1147 return 0;
1148 }
1149
1150 bool break_and_uniquify = (auto_break_cycles && uniquify_transforms);
1151
1152 size_t si = _garbage_index;
1153 if (si >= size) {
1154 si = 0;
1155 }
1156
1157 num_this_pass = std::min(num_this_pass, size);
1158 size_t stop_at_element = (si + num_this_pass) % size;
1159
1160 do {
1161 TransformState *state = (TransformState *)_states.get_key(si);
1162 if (break_and_uniquify) {
1163 if (state->get_cache_ref_count() > 0 &&
1164 state->get_ref_count() == state->get_cache_ref_count()) {
1165 // If we have removed all the references to this state not in the
1166 // cache, leaving only references in the cache, then we need to
1167 // check for a cycle involving this TransformState and break it if
1168 // it exists.
1169 state->detect_and_break_cycles();
1170 }
1171 }
1172
1173 if (!state->unref_if_one()) {
1174 // This state has recently been unreffed to 1 (the one we added when
1175 // we stored it in the cache). Now it's time to delete it. This is
1176 // safe, because we're holding the _states_lock, so it's not possible
1177 // for some other thread to find the state in the cache and ref it
1178 // while we're doing this. Also, we've just made sure to unref it to 0,
1179 // to ensure that another thread can't get it via a weak pointer.
1180 state->release_new();
1181 state->remove_cache_pointers();
1182 state->cache_unref_only();
1183 delete state;
1184
1185 // When we removed it from the hash map, it swapped the last element
1186 // with the one we just removed. So the current index contains one we
1187 // still need to visit.
1188 --size;
1189 --si;
1190 if (stop_at_element > 0) {
1191 --stop_at_element;
1192 }
1193 }
1194
1195 si = (si + 1) % size;
1196 } while (si != stop_at_element);
1197 _garbage_index = si;
1198
1199 nassertr(_states.get_num_entries() == size, 0);
1200
1201#ifdef _DEBUG
1202 nassertr(_states.validate(), 0);
1203#endif
1204
1205 // If we just cleaned up a lot of states, see if we can reduce the table in
1206 // size. This will help reduce iteration overhead in the future.
1207 _states.consider_shrink_table();
1208
1209 return (int)orig_size - (int)size;
1210}

1226list_cycles(ostream &out) {
1227 LightReMutexHolder holder(*_states_lock);
1228
1229 typedef pset<const TransformState *> VisitedStates;
1230 VisitedStates visited;
1231 CompositionCycleDesc cycle_desc;
1232
1233 size_t size = _states.get_num_entries();
1234 for (size_t si = 0; si < size; ++si) {
1235 const TransformState *state = _states.get_key(si);
1236
1237 bool inserted = visited.insert(state).second;
1238 if (inserted) {
1239 ++_last_cycle_detect;
1240 if (r_detect_cycles(state, state, 1, _last_cycle_detect, &cycle_desc)) {
1241 // This state begins a cycle.
1242 CompositionCycleDesc::reverse_iterator csi;
1243
1244 out << "\nCycle detected of length " << cycle_desc.size() + 1 << ":\n"
1245 << "state " << (void *)state << ":" << state->get_ref_count()
1246 << " =\n";

1249 const CompositionCycleDescEntry &entry = (*csi);
1250 if (entry._inverted) {
1251 out << "invert composed with ";
1252 } else {
1253 out << "composed with ";

1260 visited.insert(entry._result);
1261 }
1262
1263 cycle_desc.clear();
1264 } else {
1265 ++_last_cycle_detect;
1266 if (r_detect_reverse_cycles(state, state, 1, _last_cycle_detect, &cycle_desc)) {
1267 // This state begins a cycle.

1269
1270 out << "\nReverse cycle detected of length " << cycle_desc.size() + 1 << ":\n"

1278 << entry._obj->get_ref_count() << " =\n";

1280 visited.insert(entry._result);
1281 }
1282 out << (void *)state << ":"
1283 << state->get_ref_count() << " =\n";
1284 state->write(out, 2);

1286 cycle_desc.clear();
1287 }

1317bool TransformState::
1318validate_states() {
1319 PStatTimer timer(_transform_validate_pcollector);
1320
1321 LightReMutexHolder holder(*_states_lock);
1322 if (_states.is_empty()) {
1323 return true;
1324 }
1325
1326 if (!_states.validate()) {
1327 pgraph_cat.error()
1328 << "TransformState::_states cache is invalid!\n";
1329 return false;
1330 }
1331
1332 size_t size = _states.get_num_entries();
1333 size_t si = 0;
1334 nassertr(si < size, false);
1335 nassertr(_states.get_key(si)->get_ref_count() >= 0, false);
1336 size_t snext = si;
1337 ++snext;
1338 while (snext < size) {
1339 nassertr(_states.get_key(snext)->get_ref_count() >= 0, false);
1340 const TransformState *ssi = _states.get_key(si);
1341 if (!ssi->validate_composition_cache()) {
1342 return false;
1343 }
1344 const TransformState *ssnext = _states.get_key(snext);
1345 bool c = (*ssi) == (*ssnext);
1346 bool ci = (*ssnext) == (*ssi);
1347 if (c != ci) {
1348 pgraph_cat.error()
1349 << "TransformState::operator == () not defined properly!\n";
1350 pgraph_cat.error(false)
1351 << "(a, b): " << c << "\n";
1352 pgraph_cat.error(false)
1353 << "(b, a): " << ci << "\n";
1354 ssi->write(pgraph_cat.error(false), 2);
1355 ssnext->write(pgraph_cat.error(false), 2);
1356 return false;
1357 }
1358 si = snext;
1359 ++snext;
1360 }

1455CPT(TransformState) TransformState::
1456return_unique(TransformState *state) {
1457 nassertr(state != nullptr, state);
1458
1459 if (!transform_cache) {
1460 return state;
1461 }
1462
1463#ifndef NDEBUG
1464 if (paranoid_const) {
1465 nassertr(validate_states(), state);
1466 }
1467#endif
1468
1469 PStatTimer timer(_transform_new_pcollector);
1470
1471 LightReMutexHolder holder(*_states_lock);
1472
1473 if (state->_saved_entry != -1) {
1474 // This state is already in the cache. nassertr(_states.find(state) ==
1475 // state->_saved_entry, state);
1476 return state;
1477 }
1478
1479 // Save the state in a local PointerTo so that it will be freed at the end
1480 // of this function if no one else uses it.
1481 CPT(TransformState) pt_state = state;
1482
1483 int si = _states.find(state);
1484 if (si != -1) {
1485 // There's an equivalent state already in the set. Return it.
1486 return _states.get_key(si);
1487 }
1488
1489 // Not already in the set; add it.
1490 if (garbage_collect_states) {
1491 // If we'll be garbage collecting states explicitly, we'll increment the
1492 // reference count when we store it in the cache, so that it won't be
1493 // deleted while it's in it.
1494 state->cache_ref();
1495 }
1496 si = _states.store(state, nullptr);
1497
1498 // Save the index and return the input state.
1499 state->_saved_entry = si;

1507CPT(TransformState) TransformState::
1508do_compose(const TransformState *other) const {
1509 PStatTimer timer(_transform_compose_pcollector);
1510
1511 nassertr((_flags & F_is_invalid) == 0, this);
1512 nassertr((other->_flags & F_is_invalid) == 0, other);
1513
1514 if (compose_componentwise &&
1515 has_uniform_scale() &&
1516 !has_nonzero_shear() && !other->has_nonzero_shear() &&
1517 ((components_given() && other->has_components()) ||
1518 (other->components_given() && has_components()))) {
1519 // We will do this operation componentwise if *either* transform was given
1520 // componentwise (and there is no non-uniform scale in the way).
1521
1522 CPT(TransformState) result;
1523 if (is_2d() && other->is_2d()) {
1524 // Do a 2-d compose.
1525 LVecBase2 pos = get_pos2d();
1526 PN_stdfloat rotate = get_rotate2d();
1527 LQuaternion quat = get_norm_quat();
1528 PN_stdfloat scale = get_uniform_scale();
1529
1530 LPoint3 op = quat.xform(other->get_pos());
1531 pos += LVecBase2(op[0], op[1]) * scale;
1532
1533 rotate += other->get_rotate2d();
1534 LVecBase2 new_scale = other->get_scale2d() * scale;
1535
1536 result = make_pos_rotate_scale2d(pos, rotate, new_scale);
1537
1538 } else {
1539 // A normal 3-d compose.
1540 LVecBase3 pos = get_pos();
1541 LQuaternion quat = get_norm_quat();
1542 PN_stdfloat scale = get_uniform_scale();
1543
1544 pos += quat.xform(other->get_pos()) * scale;
1545 quat = other->get_norm_quat() * quat;
1546 LVecBase3 new_scale = other->get_scale() * scale;
1547
1548 result = make_pos_quat_scale(pos, quat, new_scale);
1549 }
1550
1551#ifndef NDEBUG
1552 if (paranoid_compose) {
1553 // Now verify against the matrix.
1554 LMatrix4 new_mat;
1555 new_mat.multiply(other->get_mat(), get_mat());
1556 if (!new_mat.almost_equal(result->get_mat(), 0.1)) {
1557 CPT(TransformState) correct = make_mat(new_mat);
1558 pgraph_cat.warning()
1559 << "Componentwise composition of " << *this << " and " << *other
1560 << " produced:\n"
1561 << *result << "\n instead of:\n" << *correct << "\n";
1562 result = correct;
1563 }
1564 }
1565#endif // NDEBUG
1566
1567 return result;
1568 }
1569
1570 // Do the operation with matrices.
1571 if (is_2d() && other->is_2d()) {
1572 LMatrix3 new_mat = other->get_mat3() * get_mat3();
1573 return make_mat3(new_mat);
1574 } else {
1575 LMatrix4 new_mat;
1576 new_mat.multiply(other->get_mat(), get_mat());
1577 return make_mat(new_mat);

1584CPT(TransformState) TransformState::
1585do_invert_compose(const TransformState *other) const {
1586 PStatTimer timer(_transform_invert_pcollector);
1587
1588 nassertr((_flags & F_is_invalid) == 0, this);
1589 nassertr((other->_flags & F_is_invalid) == 0, other);
1590
1591 if (compose_componentwise &&
1592 has_uniform_scale() &&
1593 !has_nonzero_shear() && !other->has_nonzero_shear() &&
1594 ((components_given() && other->has_components()) ||
1595 (other->components_given() && has_components()))) {
1596 // We will do this operation componentwise if *either* transform was given
1597 // componentwise (and there is no non-uniform scale in the way).
1598
1599 CPT(TransformState) result;
1600 if (is_2d() && other->is_2d()) {
1601 // Do a 2-d invert compose.
1602 LVecBase2 pos = get_pos2d();
1603 PN_stdfloat rotate = get_rotate2d();
1604 LQuaternion quat = get_norm_quat();
1605 PN_stdfloat scale = get_uniform_scale();
1606
1607 // First, invert our own transform.
1608 if (scale == 0.0f) {
1609 ((TransformState *)this)->_flags |= F_is_singular | F_singular_known;
1610 return _invalid_state;
1611 }
1612 scale = 1.0f / scale;
1613 quat.invert_in_place();
1614 rotate = -rotate;
1615 LVecBase3 mp = quat.xform(-LVecBase3(pos[0], pos[1], 0.0f));
1616 pos = LVecBase2(mp[0], mp[1]) * scale;
1617 LVecBase2 new_scale(scale, scale);
1618
1619 // Now compose the inverted transform with the other transform.
1620 if (!other->is_identity()) {
1621 LPoint3 op = quat.xform(other->get_pos());
1622 pos += LVecBase2(op[0], op[1]) * scale;
1623
1624 rotate += other->get_rotate2d();
1625 new_scale = other->get_scale2d() * scale;
1626 }
1627
1628 result = make_pos_rotate_scale2d(pos, rotate, new_scale);
1629
1630 } else {
1631 // Do a normal, 3-d invert compose.
1632 LVecBase3 pos = get_pos();
1633 LQuaternion quat = get_norm_quat();
1634 PN_stdfloat scale = get_uniform_scale();
1635
1636 // First, invert our own transform.
1637 if (scale == 0.0f) {
1638 ((TransformState *)this)->_flags |= F_is_singular | F_singular_known;
1639 return _invalid_state;
1640 }
1641 scale = 1.0f / scale;
1642 quat.invert_in_place();
1643 pos = quat.xform(-pos) * scale;
1644 LVecBase3 new_scale(scale, scale, scale);
1645
1646 // Now compose the inverted transform with the other transform.
1647 if (!other->is_identity()) {
1648 pos += quat.xform(other->get_pos()) * scale;
1649 quat = other->get_norm_quat() * quat;
1650 new_scale = other->get_scale() * scale;
1651 }
1652
1653 result = make_pos_quat_scale(pos, quat, new_scale);
1654 }
1655
1656#ifndef NDEBUG
1657 if (paranoid_compose) {
1658 // Now verify against the matrix.
1659 if (is_singular()) {
1660 pgraph_cat.warning()
1661 << "Unexpected singular matrix found for " << *this << "\n";
1662 } else {
1663 nassertr(_inv_mat != nullptr, _invalid_state);
1664 LMatrix4 new_mat;
1665 new_mat.multiply(other->get_mat(), *_inv_mat);
1666 if (!new_mat.almost_equal(result->get_mat(), 0.1)) {
1667 CPT(TransformState) correct = make_mat(new_mat);
1668 pgraph_cat.warning()
1669 << "Componentwise invert-composition of " << *this << " and " << *other
1670 << " produced:\n"
1671 << *result << "\n instead of:\n" << *correct << "\n";
1672 result = correct;
1673 }
1674 }
1675 }
1676#endif // NDEBUG
1677
1678 return result;
1679 }
1680
1681 if (is_singular()) {
1682 return _invalid_state;
1683 }
1684
1685 // Now that is_singular() has returned false, we can assume that _inv_mat
1686 // has been allocated and filled in.
1687 nassertr(_inv_mat != nullptr, _invalid_state);
1688
1689 if (is_2d() && other->is_2d()) {
1690 const LMatrix4 &i = *_inv_mat;
1691 LMatrix3 inv3(i(0, 0), i(0, 1), i(0, 3),
1692 i(1, 0), i(1, 1), i(1, 3),
1693 i(3, 0), i(3, 1), i(3, 3));
1694 if (other->is_identity()) {
1695 return make_mat3(inv3);
1696 } else {
1697 return make_mat3(other->get_mat3() * inv3);
1698 }
1699 } else {
1700 if (other->is_identity()) {
1701 return make_mat(*_inv_mat);
1702 } else {
1703 return make_mat(other->get_mat() * (*_inv_mat));
1704 }
1705 }
1706}

1746bool TransformState::
1747r_detect_cycles(const TransformState *start_state,
1748 const TransformState *current_state,
1749 int length, UpdateSeq this_seq,
1750 TransformState::CompositionCycleDesc *cycle_desc) {
1751 if (current_state->_cycle_detect == this_seq) {
1752 // We've already seen this state; therefore, we've found a cycle.
1753
1754 // However, we only care about cycles that return to the starting state
1755 // and involve more than two steps. If only one or two nodes are
1756 // involved, it doesn't represent a memory leak, so no problem there.
1757 return (current_state == start_state && length > 2);
1758 }
1759 ((TransformState *)current_state)->_cycle_detect = this_seq;
1760
1761 size_t i;
1762 size_t cache_size = current_state->_composition_cache.get_num_entries();
1763 for (i = 0; i < cache_size; ++i) {
1764 const TransformState *result = current_state->_composition_cache.get_data(i)._result;
1765 if (result != nullptr) {
1766 if (r_detect_cycles(start_state, result, length + 1,
1767 this_seq, cycle_desc)) {
1768 // Cycle detected.
1769 if (cycle_desc != nullptr) {
1770 const TransformState *other = current_state->_composition_cache.get_key(i);
1771 CompositionCycleDescEntry entry(other, result, false);
1772 cycle_desc->push_back(entry);
1773 }
1774 return true;
1775 }
1776 }
1777 }
1778
1779 cache_size = current_state->_invert_composition_cache.get_num_entries();
1780 for (i = 0; i < cache_size; ++i) {
1781 const TransformState *result = current_state->_invert_composition_cache.get_data(i)._result;
1782 if (result != nullptr) {
1783 if (r_detect_cycles(start_state, result, length + 1,
1784 this_seq, cycle_desc)) {
1785 // Cycle detected.
1786 if (cycle_desc != nullptr) {
1787 const TransformState *other = current_state->_invert_composition_cache.get_key(i);
1788 CompositionCycleDescEntry entry(other, result, true);
1789 cycle_desc->push_back(entry);
1790 }
1791 return true;
1792 }
1793 }
1794 }
1795
1796 // No cycle detected.
1797 return false;

1895void TransformState::
1896remove_cache_pointers() {
1897 nassertv(_states_lock->debug_is_locked());
1898
1899 // Fortunately, since we added CompositionCache records in pairs, we know
1900 // exactly the set of TransformState objects that have us in their cache:
1901 // it's the same set of TransformState objects that we have in our own
1902 // cache.
1903
1904/*
1905 * We do need to put considerable thought into this loop, because as we clear
1906 * out cache entries we'll cause other TransformState objects to destruct,
1907 * which could cause things to get pulled out of our own _composition_cache
1908 * map. We want to allow this (so that we don't encounter any just-destructed
1909 * pointers in our cache), but we don't want to get bitten by this cascading
1910 * effect. Instead of walking through the map from beginning to end,
1911 * therefore, we just pull out the first one each time, and erase it.
1912 */
1913
1914#ifdef DO_PSTATS
1915 if (_composition_cache.is_empty() && _invert_composition_cache.is_empty()) {
1916 return;
1917 }
1918 PStatTimer timer(_cache_update_pcollector);
1919#endif // DO_PSTATS
1920
1921 // There are lots of ways to do this loop wrong. Be very careful if you
1922 // need to modify it for any reason.
1923 size_t i = 0;
1924 while (!_composition_cache.is_empty()) {
1925 // It is possible that the "other" TransformState object is currently
1926 // within its own destructor. We therefore can't use a PT() to hold its
1927 // pointer; that could end up calling its destructor twice. Fortunately,
1928 // we don't need to hold its reference count to ensure it doesn't destruct
1929 // while we process this loop; as long as we ensure that no *other*
1930 // TransformState objects destruct, there will be no reason for that one
1931 // to.
1932 TransformState *other = (TransformState *)_composition_cache.get_key(i);
1933
1934 // We hold a copy of the composition result so we can dereference it
1935 // later.
1936 Composition comp = _composition_cache.get_data(i);
1937
1938 // Now we can remove the element from our cache. We do this now, rather
1939 // than later, before any other TransformState objects have had a chance
1940 // to destruct, so we are confident that our iterator is still valid.
1941 _composition_cache.remove_element(i);
1942 _cache_stats.add_total_size(-1);
1943 _cache_stats.inc_dels();
1944
1945 if (other != this) {
1946 int oi = other->_composition_cache.find(this);
1947
1948 // We may or may not still be listed in the other's cache (it might be
1949 // halfway through pulling entries out, from within its own destructor).
1950 if (oi != -1) {
1951 // Hold a copy of the other composition result, too.
1952 Composition ocomp = other->_composition_cache.get_data(oi);
1953
1954 other->_composition_cache.remove_element(oi);
1955 _cache_stats.add_total_size(-1);
1956 _cache_stats.inc_dels();
1957
1958 // It's finally safe to let our held pointers go away. This may have
1959 // cascading effects as other TransformState objects are destructed,
1960 // but there will be no harm done if they destruct now.
1961 if (ocomp._result != nullptr && ocomp._result != other) {
1962 cache_unref_delete(ocomp._result);
1963 }
1964 }
1965 }
1966
1967 // It's finally safe to let our held pointers go away. (See comment
1968 // above.)
1969 if (comp._result != nullptr && comp._result != this) {
1970 cache_unref_delete(comp._result);
1971 }
1972 }
1973
1974 // A similar bit of code for the invert cache.
1975 i = 0;
1976 while (!_invert_composition_cache.is_empty()) {
1977 TransformState *other = (TransformState *)_invert_composition_cache.get_key(i);
1978 nassertv(other != this);
1979 Composition comp = _invert_composition_cache.get_data(i);
1980 _invert_composition_cache.remove_element(i);
1981 _cache_stats.add_total_size(-1);
1982 _cache_stats.inc_dels();
1983 if (other != this) {
1984 int oi = other->_invert_composition_cache.find(this);
1985 if (oi != -1) {
1986 Composition ocomp = other->_invert_composition_cache.get_data(oi);
1987 other->_invert_composition_cache.remove_element(oi);
1988 _cache_stats.add_total_size(-1);
1989 _cache_stats.inc_dels();
1990 if (ocomp._result != nullptr && ocomp._result != other) {
1991 cache_unref_delete(ocomp._result);
1992 }
1993 }
1994 }
1995 if (comp._result != nullptr && comp._result != this) {
1996 cache_unref_delete(comp._result);
1997 }
1998 }

2004void TransformState::
2005calc_hash() const {
2006 // It's OK not to grab the lock here, because (1) this does not depend on
2007 // cached values (only given components are considered), and (2) the hash
2008 // itself is set atomically.
2009 PStatTimer timer(_transform_hash_pcollector);
2010 AtomicAdjust::Integer hash = 0;
2011
2012 static const int significant_flags =
2013 (F_is_invalid | F_is_identity | F_components_given | F_hpr_given | F_is_2d);
2014
2015 int flags = (_flags & significant_flags);
2016 hash = int_hash::add_hash(hash, flags);
2017
2018 nassertv((flags & (F_is_invalid | F_is_identity)) == 0);
2019
2020 // Only bother to put the rest of the stuff in the hash if the transform
2021 // is not invalid or empty.
2022
2023 if ((_flags & F_components_given) != 0) {
2024 // If the transform was specified componentwise, hash it componentwise.
2025 hash = _pos.add_hash(hash);
2026 if ((_flags & F_hpr_given) != 0) {
2027 hash = _hpr.add_hash(hash);
2028 }
2029 else if ((_flags & F_quat_given) != 0) {
2030 hash = _quat.add_hash(hash);
2031 }
2032
2033 hash = _scale.add_hash(hash);
2034 hash = _shear.add_hash(hash);
2035 }
2036 else {
2037 // Otherwise, hash the matrix . . .
2038 if (_uniquify_matrix) {
2039 // . . . but only if the user thinks that's worthwhile.
2040 check_mat();
2041 hash = _mat.add_hash(hash);
2042 }
2043 else {
2044 // Otherwise, hash the pointer only--any two different matrix-based
2045 // TransformStates are considered to be different, even if their
2046 // matrices have the same values.
2047
2048 hash = pointer_hash::add_hash(hash, this);
2049 }
2050 }
2051
2052 if (hash == H_unknown || hash == H_identity || hash == H_invalid) {
2053 // Arbitrarily offset the hash not to conflict with these special values.
2054 hash += 0x10000;
2055 }
2056
2057 // We don't care if some other thread set this in the meantime, since every
2058 // thread should have computed the same hash.
2059 AtomicAdjust::set(_hash, hash);
2060}
