135bool LODNode::
136cull_callback(CullTraverser *trav, CullTraverserData &data) {
137 if (is_any_shown()) {
138 return show_switches_cull_callback(trav, data);
139 }
140
141 consider_verify_lods(trav, data);
142
143 CDReader cdata(_cycler);
144
145 CPT(TransformState) rel_transform = get_rel_transform(trav, data);
146 PN_stdfloat lod_scale = cdata->_lod_scale *
147 trav->get_scene()->get_camera_node()->get_lod_scale();
148
149 Children children = get_children();
150 int num_children = std::min(children.get_num_children(), cdata->_switch_vector.size());
151
152 if (data._instances == nullptr || cdata->_got_force_switch) {
153 LPoint3 center = cdata->_center * rel_transform->get_mat();
154 PN_stdfloat dist2 = center.dot(center);
155
156 for (int index = 0; index < num_children; ++index) {
157 const Switch &sw = cdata->_switch_vector[index];
158 bool in_range;
159 if (cdata->_got_force_switch) {
160 in_range = (cdata->_force_switch == index);
161 } else {
162 in_range = sw.in_range_2(dist2 * lod_scale);
163 }
164
165 if (in_range) {
166 // This switch level is in range. Draw its children.
167 const PandaNode::DownConnection &child = children.get_child_connection(index);
168 trav->traverse_down(data, child);
169 }
170 }
171 }
172 else {
173 // Figure out which instances in which switch levels should be visible.
174 size_t num_instances = data._instances->size();
175 std::unique_ptr<BitArray[]> in_range(new BitArray[num_children]);
176
177 for (size_t ii = 0; ii < num_instances; ++ii) {
178 LPoint3 inst_center = cdata->_center *
179 rel_transform->compose((*data._instances)[ii].get_transform())->get_mat();
180 PN_stdfloat dist2 = inst_center.dot(inst_center);
181
182 for (int index = 0; index < num_children; ++index) {
183 const Switch &sw = cdata->_switch_vector[index];
184 if (!sw.in_range_2(dist2 * lod_scale)) {
185 in_range[index].set_bit(ii);
186 }
187 }
188 }
189
190 for (int index = 0; index < num_children; ++index) {
191 CPT(InstanceList) instances = data._instances->without(in_range[index]);
192 if (!instances->empty()) {
193 // At least one instance is visible in this switch level.
194 const PandaNode::DownConnection &child = children.get_child_connection(index);
195 trav->traverse_down(data, child);
196 }
197 }
198 }
199
200 // Now return false indicating that we have already taken care of the
201 // traversal from here.
202 return false;
203}

340int LODNode::
341compute_child(CullTraverser *trav, CullTraverserData &data) {
342 if (data.get_net_transform(trav)->is_singular()) {
343 // If we're under a singular transform, we can't compute the LOD; select
344 // none of them instead.
345 return -1;
346 }
347
348 CDReader cdata(_cycler);
349
350 if (cdata->_got_force_switch) {
351 return cdata->_force_switch;
352 }
353
354 PN_stdfloat lod_scale = cdata->_lod_scale *
355 trav->get_scene()->get_camera_node()->get_lod_scale();
356
357 CPT(TransformState) rel_transform = get_rel_transform(trav, data);
358 LPoint3 center;
359
360 if (data._instances == nullptr) {
361 center = cdata->_center * rel_transform->get_mat();
362 }
363 else {
364 // Can't really do much with instancing in FadeLODNode; let's instead
365 // just calculate the centroid of the visible instances.
366 center = LPoint3(0);
367 for (const InstanceList::Instance &instance : *data._instances) {
368 center += cdata->_center *
369 rel_transform->compose(instance.get_transform())->get_mat();
370 }
371 center *= 1.0 / data._instances->size();
372 }
373
374 PN_stdfloat dist2 = center.dot(center);
375
376 for (int index = 0; index < (int)cdata->_switch_vector.size(); ++index) {
377 if (cdata->_switch_vector[index].in_range_2(dist2 * lod_scale)) {
378 if (pgraph_cat.is_debug()) {
379 pgraph_cat.debug()
380 << data.get_node_path() << " at distance " << sqrt(dist2)
381 << ", selected child " << index << "\n";
382 }
383
384 return index;
385 }
386 }
387
388 if (pgraph_cat.is_debug()) {
389 pgraph_cat.debug()
390 << data.get_node_path() << " at distance " << sqrt(dist2)
391 << ", no children in range.\n";
392 }
393
394 return -1;
395}

462void LODNode::
463compute_internal_bounds(CPT(BoundingVolume) &internal_bounds,
464 int &internal_vertices,
465 int pipeline_stage,
466 Thread *current_thread) const {
467 // First, get ourselves a fresh, empty bounding volume.
468 PT(BoundingVolume) bound = new BoundingSphere;
469
470 // If we have any visible rings, those count in the bounding volume.
471 if (is_any_shown()) {
472 // Now actually compute the bounding volume by putting it around all of
473 // our geoms' bounding volumes.
474 pvector<const BoundingVolume *> child_volumes;
475 pvector<PT(BoundingVolume) > pt_volumes;
476
477 CDStageReader cdata(_cycler, pipeline_stage, current_thread);
478
479 SwitchVector::const_iterator si;
480 for (si = cdata->_switch_vector.begin();
481 si != cdata->_switch_vector.end();
482 ++si) {
483 const Switch &sw = (*si);
484 if (sw.is_shown()) {
485 PT(BoundingVolume) sphere = new BoundingSphere(cdata->_center, sw.get_in());
486 child_volumes.push_back(sphere);
487 pt_volumes.push_back(sphere);
488 }
489 }
490
491 const BoundingVolume **child_begin = &child_volumes[0];
492 const BoundingVolume **child_end = child_begin + child_volumes.size();
493
494 bound->around(child_begin, child_end);
495 }
496
497 internal_bounds = bound;
498 internal_vertices = 0;
499}

505CPT(TransformState) LODNode::
506get_rel_transform(CullTraverser *trav, CullTraverserData &data) {
507 // Get a pointer to the camera node.
508 Camera *camera = trav->get_scene()->get_camera_node();
509
510 // Get the camera space transform.
511 CPT(TransformState) rel_transform;
512
513 NodePath lod_center = camera->get_lod_center();
514 if (!lod_center.is_empty()) {
515 rel_transform =
516 lod_center.get_net_transform()->invert_compose(data.get_net_transform(trav));
517 } else {
518 NodePath cull_center = camera->get_cull_center();
519 if (!cull_center.is_empty()) {
520 rel_transform =
521 cull_center.get_net_transform()->invert_compose(data.get_net_transform(trav));
522 } else {
523 rel_transform = data.get_modelview_transform(trav);
524 }
525 }
526
527 return rel_transform;
528}

563bool LODNode::
564do_verify_child_bounds(const LODNode::CData *cdata, int index,
565 PN_stdfloat &suggested_radius) const {
566 suggested_radius = 0.0f;
567
568 if (index < get_num_children()) {
569 const Switch &sw = cdata->_switch_vector[index];
570 PandaNode *child = get_child(index);
571 if (child != nullptr) {
572 UpdateSeq seq;
573 CPT(BoundingVolume) bv = child->get_bounds(seq);
574
575 if (seq == sw._bounds_seq) {
576 // We previously verified this child, and it hasn't changed since
577 // then.
578 return sw._verify_ok;
579 }
580
581 ((Switch &)sw)._bounds_seq = seq;
582 ((Switch &)sw)._verify_ok = true;
583
584 if (bv->is_empty()) {
585 // This child has no geometry, so no one cares anyway.
586 return true;
587 }
588 if (bv->is_infinite()) {
589 // To be strict, we ought to look closer if the child has an infinite
590 // bounding volume, but in practice this is probably just a special
591 // case (e.g. the child contains the camera) that we don't really
592 // want to check.
593 return true;
594 }
595
596 const Switch &sw = cdata->_switch_vector[index];
597
598 const GeometricBoundingVolume *gbv;
599 DCAST_INTO_R(gbv, bv, false);
600 BoundingSphere sphere(cdata->_center, sw.get_in());
601 sphere.local_object();
602
603 int flags = sphere.contains(gbv);
604 if ((flags & BoundingVolume::IF_all) != 0) {
605 // This child's radius completely encloses its bounding volume.
606 // Perfect. (And this is the most common case.)
607 return true;
608 }
609
610 if (flags == 0) {
611 // This child's radius doesn't even come close to containing its
612 // volume.
613 nassertr(!gbv->is_infinite(), false);
614 sphere.extend_by(gbv);
615 suggested_radius = sphere.get_radius();
616 ((Switch &)sw)._verify_ok = false;
617 return false;
618 }
619
620 // This child's radius partially encloses its (loose) bounding volume.
621 // We have to look closer to determine whether it, in fact, fully
622 // encloses its geometry.
623 LPoint3 min_point(0.0f, 0.0f, 0.0f);
624 LPoint3 max_point(0.0f, 0.0f, 0.0f);
625
626 bool found_any = false;
627 child->calc_tight_bounds(min_point, max_point, found_any,
628 TransformState::make_identity(),
629 Thread::get_current_thread());
630 if (!found_any) {
631 // Hmm, the child has no geometry after all.
632 return true;
633 }
634
635 // Now we have a bounding box. Define the largest sphere we can that
636 // fits within this box. All we can say about this sphere is that it
637 // should definitely fit entirely within a bounding sphere that contains
638 // all the points of the child.
639 LPoint3 box_center = (min_point + max_point) / 2.0f;
640 PN_stdfloat box_radius = std::min(std::min(max_point[0] - box_center[0],
641 max_point[1] - box_center[1]),
642 max_point[2] - box_center[2]);
643
644 BoundingSphere box_sphere(box_center, box_radius);
645 box_sphere.local_object();
646
647 // So if any part of this inscribed sphere is outside of the radius,
648 // then the radius is bad.
649 flags = sphere.contains(&box_sphere);
650 if ((flags & BoundingVolume::IF_all) == 0) {
651 // No good.
652 if (gbv->is_infinite()) {
653 sphere.extend_by(&box_sphere);
654 } else {
655 sphere.extend_by(gbv);
656 }
657 suggested_radius = sphere.get_radius();
658 ((Switch &)sw)._verify_ok = false;
659 return false;
660 }
661 }
662 }
663
664 return true;
665}

670void LODNode::
671do_auto_verify_lods(CullTraverser *trav, CullTraverserData &data) {
672 UpdateSeq seq;
673 get_bounds(seq);
674 CDLockedReader cdata(_cycler);
675
676 if (cdata->_got_force_switch) {
677 // If we're forcing a particular switch, don't verify the LOD sizes, since
678 // they don't really apply anymore anyway. Assume the user knows what
679 // he's doing.
680 return;
681 }
682
683 if (seq != cdata->_bounds_seq) {
684 // Time to validate the children again.
685 for (int index = 0; index < (int)cdata->_switch_vector.size(); ++index) {
686 PN_stdfloat suggested_radius;
687 if (!do_verify_child_bounds(cdata, index, suggested_radius)) {
688 const Switch &sw = cdata->_switch_vector[index];
689 std::ostringstream strm;
690 strm
691 << "Level " << index << " geometry of " << data.get_node_path()
692 << " is larger than its switch radius; suggest radius of "
693 << suggested_radius << " instead of " << sw.get_in()
694 << " (configure verify-lods 0 to ignore this error)";
695 nassert_raise(strm.str());
696 }
697 }
698 CDWriter cdataw(_cycler, cdata);
699 cdataw->_bounds_seq = seq;
700 }

837void LODNode::Switch::
838compute_ring_viz() {
839 // We render the ring as a series of concentric ring-shaped triangle strips,
840 // each of which has num_slices quads.
841 static const int num_slices = 50;
842 static const int num_rings = 1;
843
844 // There are also two more triangle strips, one for the outer edge, and one
845 // for the inner edge.
846 static const PN_stdfloat edge_ratio = 0.1; // ratio of edge height to diameter.
847
848 const GeomVertexFormat *format = GeomVertexFormat::get_v3n3c();
849 PT(GeomVertexData) vdata = new GeomVertexData("LOD_ring", format, Geom::UH_static);
850
851 // Fill up the vertex table with all of the vertices.
852 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
853 GeomVertexWriter normal(vdata, InternalName::get_normal());
854 GeomVertexWriter color(vdata, InternalName::get_color());
855
856 // First, the vertices for the flat ring.
857 int ri, si;
858 for (ri = 0; ri <= num_rings; ++ri) {
859 // r is in the range [0.0, 1.0].
860 PN_stdfloat r = (PN_stdfloat)ri / (PN_stdfloat)num_rings;
861
862 // d is in the range [_out, _in].
863 PN_stdfloat d = r * (_in - _out) + _out;
864
865 for (si = 0; si < num_slices; ++si) {
866 // s is in the range [0.0, 1.0).
867 PN_stdfloat s = (PN_stdfloat)si / (PN_stdfloat)num_slices;
868
869 // t is in the range [0.0, 2pi).
870 PN_stdfloat t = MathNumbers::pi * 2.0f * s;
871
872 PN_stdfloat x = ccos(t);
873 PN_stdfloat y = csin(t);
874 vertex.add_data3(x * d, y * d, 0.0f);
875 normal.add_data3(0.0f, 0.0f, 1.0f);
876 color.add_data4(_show_color);
877 }
878 }
879
880 // Next, the vertices for the inner and outer edges.
881 for (ri = 0; ri <= 1; ++ri) {
882 PN_stdfloat r = (PN_stdfloat)ri;
883 PN_stdfloat d = r * (_in - _out) + _out;
884
885 for (si = 0; si < num_slices; ++si) {
886 PN_stdfloat s = (PN_stdfloat)si / (PN_stdfloat)num_slices;
887 PN_stdfloat t = MathNumbers::pi * 2.0f * s;
888
889 PN_stdfloat x = ccos(t);
890 PN_stdfloat y = csin(t);
891
892 vertex.add_data3(x * d, y * d, 0.5f * edge_ratio * d);
893 normal.add_data3(x, y, 0.0f);
894 color.add_data4(_show_color);
895 }
896
897 for (si = 0; si < num_slices; ++si) {
898 PN_stdfloat s = (PN_stdfloat)si / (PN_stdfloat)num_slices;
899 PN_stdfloat t = MathNumbers::pi * 2.0f * s;
900
901 PN_stdfloat x = ccos(t);
902 PN_stdfloat y = csin(t);
903
904 vertex.add_data3(x * d, y * d, -0.5f * edge_ratio * d);
905 normal.add_data3(x, y, 0.0f);
906 color.add_data4(_show_color);
907 }
908 }
909
910 // Now create the triangle strips. One tristrip for each ring.
911 PT(GeomTristrips) strips = new GeomTristrips(Geom::UH_static);
912 for (ri = 0; ri < num_rings; ++ri) {
913 for (si = 0; si < num_slices; ++si) {
914 strips->add_vertex(ri * num_slices + si);
915 strips->add_vertex((ri + 1) * num_slices + si);
916 }
917 strips->add_vertex(ri * num_slices);
918 strips->add_vertex((ri + 1) * num_slices);
919 strips->close_primitive();
920 }
921
922 // And then one triangle strip for each of the inner and outer edges.
923 for (ri = 0; ri <= 1; ++ri) {
924 for (si = 0; si < num_slices; ++si) {
925 strips->add_vertex((num_rings + 1 + ri * 2) * num_slices + si);
926 strips->add_vertex((num_rings + 1 + ri * 2 + 1) * num_slices + si);
927 }
928 strips->add_vertex((num_rings + 1 + ri * 2) * num_slices);
929 strips->add_vertex((num_rings + 1 + ri * 2 + 1) * num_slices);
930 strips->close_primitive();
931 }
932
933 PT(Geom) ring_geom = new Geom(vdata);
934 ring_geom->add_primitive(strips);
935
936 PT(GeomNode) geom_node = new GeomNode("ring");
937 geom_node->add_geom(ring_geom);
938
939 // Get a material for two-sided lighting.
940 PT(Material) material = new Material();
941 material->set_twoside(true);
942 material = MaterialPool::get_material(material);
943
944 CPT(RenderState) viz_state =
945 RenderState::make(CullFaceAttrib::make(CullFaceAttrib::M_cull_none),
946 TextureAttrib::make_off(),
947 ShaderAttrib::make_off(),
948 MaterialAttrib::make(material),
949 RenderState::get_max_priority());
950 if (_show_color[3] != 1.0f) {
951 viz_state = viz_state->add_attrib(TransparencyAttrib::make(TransparencyAttrib::M_alpha),
952 RenderState::get_max_priority());
953 }
954
955 geom_node->set_state(viz_state);
956
957 _ring_viz = geom_node.p();
958}

964void LODNode::Switch::
965compute_spindle_viz() {
966 // We render the spindle as a cylinder, which consists of num_rings rings
967 // stacked vertically, each of which is a triangle strip of num_slices
968 // quads. The scale is -10 .. 10 vertically, with a radius of 1.0.
969 static const int num_slices = 10;
970 static const int num_rings = 10;
971
972 const GeomVertexFormat *format = GeomVertexFormat::get_v3n3c();

976 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
977 GeomVertexWriter normal(vdata, InternalName::get_normal());
978 GeomVertexWriter color(vdata, InternalName::get_color());

982 // r is in the range [0.0, 1.0].
983 PN_stdfloat r = (PN_stdfloat)ri / (PN_stdfloat)num_rings;
984
985 // z is in the range [100.0, -100.0]
986 PN_stdfloat z = 100.0f - r * 200.0f;

989 // s is in the range [0.0, 1.0).
990 PN_stdfloat s = (PN_stdfloat)si / (PN_stdfloat)num_slices;
991

993 PN_stdfloat t = MathNumbers::pi * 2.0f * s;
994
995 PN_stdfloat x = ccos(t);
996 PN_stdfloat y = csin(t);

1020

1023 TextureAttrib::make_off(),
1024 ShaderAttrib::make_off(),
