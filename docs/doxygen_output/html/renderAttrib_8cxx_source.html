103bool RenderAttrib::
104unref() const {
105 if (!state_cache || garbage_collect_states) {
106 // If we're not using the cache at all, or if we're relying on garbage
107 // collection, just allow the pointer to unref normally.
108 return ReferenceCount::unref();
109 }
110
111 // Here is the normal refcounting case, with a normal cache, and without
112 // garbage collection in effect. In this case we will pull the object out
113 // of the cache when its reference count goes to 0.
114
115 // We always have to grab the lock, since we will definitely need to be
116 // holding it if we happen to drop the reference count to 0. Having to grab
117 // the lock at every call to unref() is a big limiting factor on
118 // parallelization.
119 LightReMutexHolder holder(*_attribs_lock);
120
121 if (ReferenceCount::unref()) {
122 // The reference count is still nonzero.
123 return true;
124 }
125
126 // The reference count has just reached zero. Make sure the object is
127 // removed from the global object pool, before anyone else finds it and
128 // tries to ref it.
129 ((RenderAttrib *)this)->release_new();
130
131 return false;
132}

181int RenderAttrib::
182garbage_collect() {
183 // This gets called periodically, so use this opportunity to reload the value
184 // from the Config.prc file.
185 _uniquify_attribs = uniquify_attribs && state_cache;
186
187 if (!garbage_collect_states) {
188 return 0;
189 }
190 LightReMutexHolder holder(*_attribs_lock);
191
192 PStatTimer timer(_garbage_collect_pcollector);
193 size_t orig_size = _attribs.get_num_entries();
194
195#ifdef _DEBUG
196 nassertr(_attribs.validate(), 0);
197#endif
198
199 // How many elements to process this pass?
200 size_t size = orig_size;
201 size_t num_this_pass = std::max(0, int(size * garbage_collect_states_rate));
202 if (num_this_pass <= 0) {
203 return 0;
204 }
205
206 size_t si = _garbage_index;
207 if (si >= size) {
208 si = 0;
209 }
210
211 num_this_pass = std::min(num_this_pass, size);
212 size_t stop_at_element = (si + num_this_pass) % size;
213
214 do {
215 RenderAttrib *attrib = (RenderAttrib *)_attribs.get_key(si);
216 if (!attrib->unref_if_one()) {
217 // This attrib has recently been unreffed to 1 (the one we added when
218 // we stored it in the cache). Now it's time to delete it. This is
219 // safe, because we're holding the _attribs_lock, so it's not possible
220 // for some other thread to find the attrib in the cache and ref it
221 // while we're doing this. Also, we've just made sure to unref it to 0,
222 // to ensure that another thread can't get it via a weak pointer.
223 attrib->release_new();
224 delete attrib;
225
226 // When we removed it from the hash map, it swapped the last element
227 // with the one we just removed. So the current index contains one we
228 // still need to visit.
229 --size;
230 --si;
231 if (stop_at_element > 0) {
232 --stop_at_element;
233 }
234 }
235
236 si = (si + 1) % size;
237 } while (si != stop_at_element);
238 _garbage_index = si;
239
240 nassertr(_attribs.get_num_entries() == size, 0);
241
242#ifdef _DEBUG
243 nassertr(_attribs.validate(), 0);
244#endif
245
246 // If we just cleaned up a lot of attribs, see if we can reduce the table in
247 // size. This will help reduce iteration overhead in the future.
248 _attribs.consider_shrink_table();
249
250 return (int)orig_size - (int)size;
251}

258bool RenderAttrib::
259validate_attribs() {
260 LightReMutexHolder holder(*_attribs_lock);
261 if (_attribs.is_empty()) {
262 return true;
263 }
264
265 if (!_attribs.validate()) {
266 pgraph_cat.error()
267 << "RenderAttrib::_attribs cache is invalid!\n";
268
269 size_t size = _attribs.get_num_entries();
270 for (size_t si = 0; si < size; ++si) {
271 const RenderAttrib *attrib = _attribs.get_key(si);
272 //cerr << si << ": " << attrib << "\n";
273 attrib->write(std::cerr, 2);
274 }
275
276 return false;
277 }
278
279 size_t size = _attribs.get_num_entries();
280 size_t si = 0;
281 nassertr(si < size, false);
282 nassertr(_attribs.get_key(si)->get_ref_count() >= 0, false);
283 size_t snext = si;
284 ++snext;
285 while (snext < size) {
286 nassertr(_attribs.get_key(snext)->get_ref_count() >= 0, false);
287 const RenderAttrib *ssi = _attribs.get_key(si);
288 const RenderAttrib *ssnext = _attribs.get_key(snext);
289 int c = ssi->compare_to(*ssnext);
290 int ci = ssnext->compare_to(*ssi);
291 if ((ci < 0) != (c > 0) ||
292 (ci > 0) != (c < 0) ||
293 (ci == 0) != (c == 0)) {
294 pgraph_cat.error()
295 << "RenderAttrib::compare_to() not defined properly!\n";
296 pgraph_cat.error(false)
297 << "(a, b): " << c << "\n";
298 pgraph_cat.error(false)
299 << "(b, a): " << ci << "\n";
300 ssi->write(pgraph_cat.error(false), 2);
301 ssnext->write(pgraph_cat.error(false), 2);
302 return false;
303 }
304 si = snext;
305 ++snext;
306 }
307
308 return true;
309}

314CPT(RenderAttrib) RenderAttrib::
315do_uniquify(const RenderAttrib *attrib) {
316 LightReMutexHolder holder(*_attribs_lock);
317
318 if (attrib->_saved_entry != -1) {
319 // This attrib is already in the cache.
320 //nassertr(_attribs.find(attrib) == attrib->_saved_entry, attrib);
321 return attrib;
322 }
323
324 int si = _attribs.find(attrib);
325 if (si != -1) {
326 // There's an equivalent attrib already in the set. Return it. If this
327 // is a newly created RenderAttrib, though, be sure to delete it.
328 if (attrib->get_ref_count() == 0) {
329 delete attrib;
330 }
331 return _attribs.get_key(si);
332 }
333
334 // Not already in the set; add it.
335 if (garbage_collect_states) {
336 // If we'll be garbage collecting attribs explicitly, we'll increment the
337 // reference count when we store it in the cache, so that it won't be
338 // deleted while it's in it.
339 attrib->ref();
340 }
341 si = _attribs.store(attrib, nullptr);
342
343 // Save the index and return the input attrib.
344 attrib->_saved_entry = si;
345 return attrib;
346}

502TypedWritable *RenderAttrib::
503change_this(TypedWritable *old_ptr, BamReader *manager) {
504 // First, uniquify the pointer.
505 RenderAttrib *attrib = DCAST(RenderAttrib, old_ptr);
506 CPT(RenderAttrib) pointer = return_unique(attrib);
507
508 // But now we have a problem, since we have to hold the reference count and
509 // there's no way to return a TypedWritable while still holding the
510 // reference count! We work around this by explicitly upping the count, and
511 // also setting a finalize() callback to down it later.
512 if (pointer == attrib) {
513 pointer->ref();
514 manager->register_finalize(attrib);
515 }
516
517 // We have to cast the pointer back to non-const, because the bam reader
518 // expects that.
519 return (RenderAttrib *)pointer.p();
520}

527void RenderAttrib::
528finalize(BamReader *) {
529 // Unref the pointer that we explicitly reffed in change_this().
530 unref();
531
532 // We should never get back to zero after unreffing our own count, because
533 // we expect to have been stored in a pointer somewhere. If we do get to
534 // zero, it's a memory leak; the way to avoid this is to call unref_delete()
535 // above instead of unref(), but this is dangerous to do from within a
536 // virtual function.
537 nassertv(get_ref_count() != 0);
538}
