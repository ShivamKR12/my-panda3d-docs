122bool PStatMonitor::
123read(const Filename &fn) {
124 close();
125
126 DatagramInputFile dif;
127 if (!dif.open(fn)) {
128 nout << "Failed to open " << fn << " for reading.\n";
129 return false;
130 }
131 string header;
132 if (!dif.read_header(header, 8) || header != session_file_header) {
133 nout << "Session file contains invalid header.\n";
134 return false;
135 }
136 Datagram dg;
137 dg.set_stdfloat_double(false);
138
139 if (!dif.get_datagram(dg)) {
140 nout << "Failed to read datagram from session file.\n";
141 return false;
142 }
143
144 DatagramIterator scan(dg);
145 int version = scan.get_uint16();
146 if (version != 1) {
147 nout << "Unsupported session file version " << version << ".\n";
148 return false;
149 }
150 // Room for a minor version number
151 scan.get_uint16();
152
153 read_datagram(scan);
154 dif.close();
155
156 idle();
157
158 _read_filename = fn;
159 _client_data->clear_dirty();
160
161 return true;

175 if (!dif.read_header(header, 8) || header != layout_file_header) {
176 nout << "Layout file contains invalid header.\n";

190 int version = scan.get_uint16();
191 if (version != 1) {
192 nout << "Unsupported layout file version " << version << ".\n";
193 open_strip_chart(0, 0, false);
194 return;
195 }
196 // Room for a minor version number
197 scan.get_uint16();
198
199 size_t num_colors = scan.get_uint32();
200 for (size_t i = 0; i < num_colors; ++i) {
201 int key = scan.get_int32();
202 LRGBColor &color = _colors[key];
203 color[0] = scan.get_float32();
204 color[1] = scan.get_float32();
205 color[2] = scan.get_float32();
206 }

231 int width = scan.get_int32();
232 int height = scan.get_int32();
233 bool minimized = scan.get_bool();
234 bool maximized = scan.get_bool();

236 graph = open_strip_chart(thread_index, collector_index, show_level);

238 }
239 }

243 std::string thread_name = scan.get_string();
244 std::string collector_name = scan.get_string();
245 int thread_index = _client_data->find_thread(thread_name);
246 int collector_index = collector_name.empty() ? -1 : _client_data->find_collector(collector_name);
247 int x = scan.get_int32();
248 int y = scan.get_int32();
249 int width = scan.get_int32();
250 int height = scan.get_int32();
251 bool minimized = scan.get_bool();
252 bool maximized = scan.get_bool();
253 if (thread_index != -1) {
254 graph = open_flame_graph(thread_index, collector_index);
255 graph->set_window_state(x, y, width, height, minimized, maximized);
256 }
257 }

259 size_t num_piano_rolls = scan.get_uint16();
260 for (size_t i = 0; i < num_piano_rolls; ++i) {
261 std::string thread_name = scan.get_string();
262 int thread_index = _client_data->find_thread(thread_name);
263 int x = scan.get_int32();

272 }
273 }
274
275 dif.close();
276 if (num_timelines + num_strip_charts + num_flame_graphs + num_piano_rolls == 0) {
277 open_strip_chart(0, 0, false);

407const LRGBColor &PStatMonitor::
408get_collector_color(int collector_index) {
409 Colors::iterator ci;
410 ci = _colors.find(collector_index);
411 if (ci != _colors.end()) {
412 return (*ci).second;
413 }
414
415 // Ask the client data about the color.
416 if (!_client_data.is_null() &&
417 _client_data->has_collector(collector_index)) {
418 const PStatCollectorDef &def =
419 _client_data->get_collector_def(collector_index);
420
421 LRGBColor sc(def._suggested_color.r,
422 def._suggested_color.g,
423 def._suggested_color.b);
424 if (sc != LRGBColor::zero()) {
425 ci = _colors.insert(Colors::value_type(collector_index, sc)).first;
426 return (*ci).second;
427 }
428
429 // Use the fullname of the collector as a hash to seed the random number
430 // generator (consulted below), so we get the same color for a given name
431 // across sessions.
432 string fullname = _client_data->get_collector_fullname(collector_index);
433 unsigned int hash = 0;
434 for (string::const_iterator ci = fullname.begin(); ci != fullname.end(); ++ci) {
435 hash = hash * 37 + (unsigned int)(*ci);
436 }
437 srand(hash);
438 }
439
440 // We didn't have a color for the collector; make one up.
441 LRGBColor random_color;
442 random_color[0] = (double)rand() / (double)RAND_MAX;
443 random_color[1] = (double)rand() / (double)RAND_MAX;
444 random_color[2] = (double)rand() / (double)RAND_MAX;
445
446 ci = _colors.insert(Colors::value_type(collector_index, random_color)).first;
447 return (*ci).second;
448}
