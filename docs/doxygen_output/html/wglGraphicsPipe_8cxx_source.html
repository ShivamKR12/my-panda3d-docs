244format_pfd_flags(DWORD pfd_flags) {
245 struct FlagDef {
246 DWORD flag;
247 const char *name;
248 };
249 static FlagDef flag_def[] = {
250 { PFD_DRAW_TO_WINDOW, "PFD_DRAW_TO_WINDOW" },
251 { PFD_DRAW_TO_BITMAP, "PFD_DRAW_TO_BITMAP" },
252 { PFD_SUPPORT_GDI, "PFD_SUPPORT_GDI" },
253 { PFD_SUPPORT_OPENGL, "PFD_SUPPORT_OPENGL" },
254 { PFD_GENERIC_ACCELERATED, "PFD_GENERIC_ACCELERATED" },
255 { PFD_GENERIC_FORMAT, "PFD_GENERIC_FORMAT" },
256 { PFD_NEED_PALETTE, "PFD_NEED_PALETTE" },
257 { PFD_NEED_SYSTEM_PALETTE, "PFD_NEED_SYSTEM_PALETTE" },
258 { PFD_DOUBLEBUFFER, "PFD_DOUBLEBUFFER" },
259 { PFD_STEREO, "PFD_STEREO" },
260 { PFD_SWAP_LAYER_BUFFERS, "PFD_SWAP_LAYER_BUFFERS" },
261 { PFD_SWAP_COPY, "PFD_SWAP_COPY" },
262 { PFD_SWAP_EXCHANGE, "PFD_SWAP_EXCHANGE" },
263 };
264 static const int num_flag_defs = sizeof(flag_def) / sizeof(FlagDef);
265
266 std::ostringstream out;
267
268 const char *sep = "";
269 bool got_any = false;
270 for (int i = 0; i < num_flag_defs; i++) {
271 if (pfd_flags & flag_def[i].flag) {
272 out << sep << flag_def[i].name;
273 pfd_flags &= ~flag_def[i].flag;
274 sep = "|";
275 got_any = true;
276 }
277 }
278
279 if (pfd_flags != 0 || !got_any) {
280 out << sep << std::hex << "0x" << pfd_flags << std::dec;
281 }
282
283 return out.str();
