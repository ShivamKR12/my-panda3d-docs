49void
50pm_error(const char *format, ...) {
51 va_list ap;
52 va_start(ap, format);
53
54 static const size_t buffer_size = 1024;
55 char buffer[buffer_size + 1];
56
57 vsnprintf(buffer, buffer_size, format, ap);
58 buffer[buffer_size] = 0;
59 va_end(ap);
60
61 pnmimage_cat.error() << buffer << "\n";
62
63 // Now we're supposed to exit. Inconvenient if we were running Panda
64 // interactively, but that's the way it is.
