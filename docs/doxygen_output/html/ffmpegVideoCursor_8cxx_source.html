268
269 if (_thread_status == TS_stopped && _max_readahead_frames > 0) {
270 // Get a unique name for the thread's sync name.

295void FfmpegVideoCursor::
296stop_thread() {
297 if (_thread_status != TS_stopped) {
298 PT(GenericThread) thread = _thread;
299 {
300 MutexHolder holder(_lock);
301 if (_thread_status != TS_stopped) {
302 _thread_status = TS_shutdown;
303 }
304 _action_cvar.notify();
305 _thread = nullptr;
306 }
307
308 // Now that we've released the lock, we can join the thread.
309 thread->join();
310 }
311
312 // This is a good time to clean up all of the allocated frame objects. It's
313 // not really necessary to be holding the lock, since the thread is gone,
314 // but we'll grab it anyway just in case someone else starts the thread up
315 // again.
316 MutexHolder holder(_lock);
317
318 _readahead_frames.clear();
319}

333bool FfmpegVideoCursor::
334set_time(double timestamp, int loop_count) {
335 int frame = (int)(timestamp / _video_timebase + 0.5);
336
337 if (_eof_known) {
338 if (loop_count == 0) {
339 frame = frame % (_eof_frame + 1);
340 } else {
341 int last_frame = (_eof_frame + 1) * loop_count;
342 if (frame < last_frame) {
343 frame = frame % (_eof_frame + 1);
344 } else {
345 frame = _eof_frame;
346 }
347 }
348 }
349
350 // No point in trying to position before the first frame.
351 frame = std::max(frame, _initial_dts);
352
353 if (ffmpeg_cat.is_spam() && frame != _current_frame) {
354 ffmpeg_cat.spam()
355 << "set_time(" << timestamp << "): " << frame
356 << ", loop_count = " << loop_count << "\n";
357 }
358
359 _current_frame = frame;
360 if (_current_frame_buffer != nullptr) {
361 // If we've previously returned a frame, don't bother asking for a next
362 // one if that frame is still valid.
363 return (_current_frame >= _current_frame_buffer->_end_frame ||
364 _current_frame < _current_frame_buffer->_begin_frame);
365 }
366
367 // If our last request didn't return a frame, try again.
368 return true;
369}

374PT(MovieVideoCursor::Buffer) FfmpegVideoCursor::
375fetch_buffer() {
376 MutexHolder holder(_lock);
377
378 // If there was an error at any point, just return NULL.
379 if (_format_ctx == nullptr) {
380 return nullptr;
381 }
382
383 PT(FfmpegBuffer) frame;
384 if (_thread_status == TS_stopped) {
385 // Non-threaded case. Just get the next frame directly.
386 advance_to_frame(_current_frame);
387 if (_frame_ready) {
388 frame = do_alloc_frame();
389 export_frame(frame);
390 }
391
392 } else {
393 // Threaded case. Wait for the thread to serve up the required frames.
394 if (!_readahead_frames.empty()) {
395 frame = _readahead_frames.front();
396 _readahead_frames.pop_front();
397 _action_cvar.notify();
398 while (frame->_end_frame < _current_frame && !_readahead_frames.empty()) {
399 // This frame is too old. Discard it.
400 if (ffmpeg_cat.is_debug()) {
401 ffmpeg_cat.debug()
402 << "ffmpeg for " << _filename.get_basename()
403 << " at frame " << _current_frame << ", discarding frame at "
404 << frame->_begin_frame << "\n";
405 }
406 frame = _readahead_frames.front();
407 _readahead_frames.pop_front();
408 }
409 if (frame->_begin_frame > _current_frame) {
410 // This frame is too new. Empty all remaining frames and seek
411 // backwards.
412 if (ffmpeg_cat.is_debug()) {
413 ffmpeg_cat.debug()
414 << "ffmpeg for " << _filename.get_basename()
415 << " at frame " << _current_frame << ", encountered too-new frame at "
416 << frame->_begin_frame << "\n";
417 }
418 do_clear_all_frames();
419 if (_thread_status == TS_wait || _thread_status == TS_seek || _thread_status == TS_readahead) {
420 _thread_status = TS_seek;
421 _seek_frame = _current_frame;
422 _action_cvar.notify();
423 }
424 }
425 }
426 if (frame == nullptr || frame->_end_frame < _current_frame) {
427 // No frame available, or the frame is too old. Seek.
428 if (_thread_status == TS_wait || _thread_status == TS_seek || _thread_status == TS_readahead) {
429 _thread_status = TS_seek;
430 _seek_frame = _current_frame;
431 _action_cvar.notify();
432 }
433 }
434 }
435
436 if (frame != nullptr) {
437 bool too_old = (frame->_end_frame < _current_frame && !ffmpeg_show_seek_frames);
438 bool too_new = frame->_begin_frame > _current_frame;
439 if (too_old || too_new) {
440 // The frame is too old or too new. Just recycle it.
441 frame = nullptr;
442 }
443 }
444
445 if (frame != nullptr) {
446 _current_frame_buffer = frame;
447 if (ffmpeg_cat.is_debug()) {
448 ffmpeg_cat.debug()
449 << "ffmpeg for " << _filename.get_basename()
450 << " at frame " << _current_frame << ", returning frame at "
451 << frame->_begin_frame << "\n";
452 }
453 } else {
454 if (ffmpeg_cat.is_debug()) {
455 ffmpeg_cat.debug()
456 << "ffmpeg for " << _filename.get_basename()
457 << " at frame " << _current_frame << ", returning NULL\n";
458 }
459 }
460 return frame;
461}

475bool FfmpegVideoCursor::
476open_stream() {
477 nassertr(!_ffvfile.is_open(), false);
478
479 // Hold the global lock while we open the file and create avcodec objects.
480 ReMutexHolder av_holder(_av_lock);
481
482 if (!_source->get_subfile_info().is_empty()) {
483 // Read a subfile.
484 if (!_ffvfile.open_subfile(_source->get_subfile_info())) {
485 ffmpeg_cat.info()
486 << "Couldn't open " << _source->get_subfile_info() << "\n";
487 close_stream();
488 return false;
489 }
490
491 } else {
492 // Read a filename.
493 if (!_ffvfile.open_vfs(_filename)) {
494 ffmpeg_cat.info()
495 << "Couldn't open " << _filename << "\n";
496 close_stream();
497 return false;
498 }
499 }
500
501 nassertr(_format_ctx == nullptr, false);
502 _format_ctx = _ffvfile.get_format_context();
503 nassertr(_format_ctx != nullptr, false);
504
505 if (avformat_find_stream_info(_format_ctx, nullptr) < 0) {
506 ffmpeg_cat.info()
507 << "Couldn't find stream info\n";
508 close_stream();
509 return false;
510 }
511
512 nassertr(_video_ctx == nullptr, false);
513
514 // As of libavformat version 57.41.100, AVStream.codec is deprecated in favor
515 // of AVStream.codecpar. Fortunately, the two structures have
516 // similarly-named members, so we can just switch out the declaration.
517#if LIBAVFORMAT_VERSION_INT >= AV_VERSION_INT(57, 41, 100)
518 AVCodecParameters *codecpar;
519#else
520 AVCodecContext *codecpar;
521#endif
522
523 // Find the video stream
524 AVStream *stream = nullptr;
525 for (int i = 0; i < (int)_format_ctx->nb_streams; ++i) {
526#if LIBAVFORMAT_VERSION_INT >= AV_VERSION_INT(57, 41, 100)
527 codecpar = _format_ctx->streams[i]->codecpar;
528#else
529 codecpar = _format_ctx->streams[i]->codec;
530#endif
531 if (codecpar->codec_type == AVMEDIA_TYPE_VIDEO) {
532 _video_index = i;
533 stream = _format_ctx->streams[i];
534 break;
535 }
536 }
537
538 if (stream == nullptr) {
539 ffmpeg_cat.info()
540 << "Couldn't find stream\n";
541 close_stream();
542 return false;
543 }
544
545 _video_timebase = av_q2d(stream->time_base);
546 _min_fseek = (int)(3.0 / _video_timebase);
547
548 const AVCodec *pVideoCodec = nullptr;
549 if (ffmpeg_prefer_libvpx) {
550#if LIBAVCODEC_VERSION_INT >= AV_VERSION_INT(55, 0, 0)
551 if (codecpar->codec_id == AV_CODEC_ID_VP9) {
552 pVideoCodec = avcodec_find_decoder_by_name("libvpx-vp9");
553 } else
554#endif
555 if (codecpar->codec_id == AV_CODEC_ID_VP8) {
556 pVideoCodec = avcodec_find_decoder_by_name("libvpx");
557 }
558 }
559 if (pVideoCodec == nullptr) {
560 pVideoCodec = avcodec_find_decoder(codecpar->codec_id);
561 }
562 if (pVideoCodec == nullptr) {
563 ffmpeg_cat.info()
564 << "Couldn't find codec\n";
565 close_stream();
566 return false;
567 }
568
569 _video_ctx = avcodec_alloc_context3(pVideoCodec);
570
571 if (_video_ctx == nullptr) {
572 ffmpeg_cat.info()
573 << "Couldn't allocate _video_ctx\n";
574 close_stream();
575 return false;
576 }
577
578#if LIBAVFORMAT_VERSION_INT >= AV_VERSION_INT(57, 41, 100)
579 avcodec_parameters_to_context(_video_ctx, codecpar);
580#else
581 avcodec_copy_context(_video_ctx, codecpar);
582#endif
583
584 if (avcodec_open2(_video_ctx, pVideoCodec, nullptr) < 0) {
585 ffmpeg_cat.info()
586 << "Couldn't open codec\n";
587 close_stream();
588 return false;
589 }
590
591 _size_x = _video_ctx->width;
592 _size_y = _video_ctx->height;
593 _num_components = 3;
594 _length = (double)_format_ctx->duration / (double)AV_TIME_BASE;
595 _can_seek = true;
596 _can_seek_fast = true;
597
598 return true;
599}

608
609 if (_video_ctx && _video_ctx->codec) {

676void FfmpegVideoCursor::
677thread_main() {
678 if (ffmpeg_cat.is_spam()) {
679 ffmpeg_cat.spam()
680 << "ffmpeg thread for " << _filename.get_basename() << " starting.\n";
681 }
682
683 // First, push the first frame onto the readahead queue.
684 if (_frame_ready) {
685 PT(FfmpegBuffer) frame = do_alloc_frame();
686 export_frame(frame);
687 MutexHolder holder(_lock);
688 _readahead_frames.push_back(frame);
689 }
690
691 // Now repeatedly wait for something interesting to do, until we're told to
692 // shut down.
693 MutexHolder holder(_lock);
694 while (_thread_status != TS_shutdown) {
695 nassertv(_thread_status != TS_stopped);
696 _action_cvar.wait();
697
698 while (do_poll()) {
699 // Keep doing stuff as long as there's something to do.
700 _lock.release();
701 PStatClient::thread_tick();
702 Thread::consider_yield();
703 _lock.acquire();
704 }
705 }
706
707 _thread_status = TS_stopped;
708 if (ffmpeg_cat.is_spam()) {
709 ffmpeg_cat.spam()
710 << "ffmpeg thread for " << _filename.get_basename() << " stopped.\n";
711 }
712}

719bool FfmpegVideoCursor::
720do_poll() {
721 switch (_thread_status) {
722 case TS_stopped:
723 case TS_seeking:
724 // This shouldn't be possible while the thread is running.
725 nassertr(false, false);
726 return false;
727
728 case TS_wait:
729 // The video hasn't started playing yet.
730 return false;
731
732 case TS_readahead:
733 if ((int)_readahead_frames.size() < _max_readahead_frames) {
734 // Time to read the next frame.
735 PT(FfmpegBuffer) frame = do_alloc_frame();
736 nassertr(frame != nullptr, false);
737 _lock.release();
738 fetch_frame(-1);
739 if (_frame_ready) {
740 export_frame(frame);
741 _lock.acquire();
742 _readahead_frames.push_back(frame);
743 } else {
744 // No frame.
745 _lock.acquire();
746 }
747 return true;
748 }
749
750 // No room for the next frame yet. Wait for more.
751 return false;
752
753 case TS_seek:
754 // Seek to a specific frame.
755 {
756 int seek_frame = _seek_frame;
757 _thread_status = TS_seeking;
758 PT(FfmpegBuffer) frame = do_alloc_frame();
759 nassertr(frame != nullptr, false);
760 _lock.release();
761 if (seek_frame != _begin_frame) {
762 advance_to_frame(seek_frame);
763 }
764 if (_frame_ready) {
765 export_frame(frame);
766 _lock.acquire();
767 do_clear_all_frames();
768 _readahead_frames.push_back(frame);
769 } else {
770 _lock.acquire();
771 do_clear_all_frames();
772 }
773
774 if (_thread_status == TS_seeking) {
775 // After seeking, we automatically transition to readahead.
776 _thread_status = TS_readahead;
777 }
778 }
779 return true;
780
781 case TS_shutdown:
782 // Time to stop the thread.
783 return false;
784 }
785
786 return false;
787}

873void FfmpegVideoCursor::
874fetch_frame(int frame) {
875 PStatTimer timer(_fetch_buffer_pcollector);
876
877 int finished = 0;
878
879 if (_packet_frame <= frame) {
880 finished = 0;
881
882 // Get the next packet. The first packet beyond the frame we're looking
883 // for marks the point to stop.
884 while (_packet_frame <= frame) {
885 PStatTimer timer(_seek_pcollector);
886
887 // Decode the previous packet, and get the next one.
888 decode_frame(finished);
889 _begin_frame = _packet_frame;
890 if (fetch_packet(frame)) {
891 _end_frame = _packet_frame;
892 _frame_ready = false;
893 return;
894 }
895 }
896
897 } else {
898 // Just get the next frame.
899 finished = 0;
900 while (!finished && _packet->data) {
901 decode_frame(finished);
902 _begin_frame = _packet_frame;
903 fetch_packet(_begin_frame + 1);
904 }
905 }
906
907 _end_frame = _packet_frame;
908 _frame_ready = true;
909}

972void FfmpegVideoCursor::
973do_seek(int frame, bool backward) {
974 int64_t target_ts = (int64_t)frame;
975 if (target_ts < (int64_t)(_initial_dts)) {
976 // Attempts to seek before the first packet will fail.
977 target_ts = _initial_dts;
978 }
979 int flags = 0;
980 if (backward) {
981 flags = AVSEEK_FLAG_BACKWARD;
982 }
983
984 if (av_seek_frame(_format_ctx, _video_index, target_ts, flags) < 0) {
985 if (ffmpeg_cat.is_spam()) {
986 ffmpeg_cat.spam()
987 << "Seek failure.\n";
988 }
989
990 if (backward) {
991 // Now try to seek forward.
992 reset_stream();
993 seek(frame, false);
994 return;
995 }
996
997 // Try a binary search to get a little closer.
998 if (binary_seek(_initial_dts, frame, frame, 1) < 0) {
999 if (ffmpeg_cat.is_spam()) {
1000 ffmpeg_cat.spam()
1001 << "Seek double failure.\n";
1002 }
1003 reset_stream();
1004 return;
1005 }
1006 }
1007
1008 fetch_packet(0);
1009 fetch_frame(-1);
1010}

1016int FfmpegVideoCursor::
1017binary_seek(int min_frame, int max_frame, int target_frame, int num_iterations) {
1018 int try_frame = (min_frame + max_frame) / 2;
1019 if (num_iterations > 5 || try_frame >= max_frame) {
1020 // Success.
1021 return 0;
1022 }
1023
1024 if (av_seek_frame(_format_ctx, _video_index, try_frame, AVSEEK_FLAG_BACKWARD) < 0) {
1025 // Failure. Try lower.
1026 if (binary_seek(min_frame, try_frame - 1, target_frame, num_iterations + 1) < 0) {
1027 return -1;
1028 }
1029 } else {
1030 // Success. Try higher.
1031 if (binary_seek(try_frame + 1, max_frame, target_frame, num_iterations + 1) < 0) {
1032 return -1;
1033 }
1034 }
1035 return 0;
1036}

1065void FfmpegVideoCursor::
1066advance_to_frame(int frame) {
1067 PStatTimer timer(_fetch_buffer_pcollector);
1068
1069 if (frame < _begin_frame) {
1070 // Frame is in the past.
1071 if (ffmpeg_cat.is_spam()) {
1072 ffmpeg_cat.spam()
1073 << "Seeking backward to " << frame << " from " << _begin_frame << "\n";
1074 }
1075 seek(frame, true);
1076 if (_begin_frame > frame) {
1077 if (ffmpeg_cat.is_spam()) {
1078 ffmpeg_cat.spam()
1079 << "Ended up at " << _begin_frame << ", not far enough back!\n";
1080 }
1081 reset_stream();
1082 if (ffmpeg_cat.is_spam()) {
1083 ffmpeg_cat.spam()
1084 << "Reseek to 0, got " << _begin_frame << "\n";
1085 }
1086 }
1087 if (frame > _end_frame) {
1088 if (ffmpeg_cat.is_spam()) {
1089 ffmpeg_cat.spam()
1090 << "Now sliding forward to " << frame << " from " << _begin_frame << "\n";
1091 }
1092 fetch_frame(frame);
1093 }
1094
1095 } else if (frame < _end_frame) {
1096 // Frame is in the present: already have the frame.
1097 if (ffmpeg_cat.is_spam()) {
1098 ffmpeg_cat.spam()
1099 << "Currently have " << frame << " within " << _begin_frame << " .. " << _end_frame << "\n";
1100 }
1101
1102 } else if (frame < _end_frame + _min_fseek) {
1103 // Frame is in the near future.
1104 if (ffmpeg_cat.is_spam()) {
1105 ffmpeg_cat.spam()
1106 << "Sliding forward to " << frame << " from " << _begin_frame << "\n";
1107 }
1108 fetch_frame(frame);
1109
1110 } else {
1111 // Frame is in the far future. Seek forward, then read. There's a danger
1112 // here: because keyframes are spaced unpredictably, trying to seek
1113 // forward could actually move us backward in the stream! This must be
1114 // avoided. So the rule is, try the seek. If it hurts us by moving us
1115 // backward, we increase the minimum threshold distance for forward-
1116 // seeking in the future.
1117
1118 if (ffmpeg_cat.is_spam()) {
1119 ffmpeg_cat.spam()
1120 << "Jumping forward to " << frame << " from " << _begin_frame << "\n";
1121 }
1122 int base = _begin_frame;
1123 seek(frame, false);
1124 if (_begin_frame < base) {
1125 _min_fseek += (base - _begin_frame);
1126 if (ffmpeg_cat.is_spam()) {
1127 ffmpeg_cat.spam()
1128 << "Wrong way! Increasing _min_fseek to " << _min_fseek << "\n";
1129 }
1130 }
1131 if (frame > _end_frame) {
1132 if (ffmpeg_cat.is_spam()) {
1133 ffmpeg_cat.spam()
1134 << "Correcting, sliding forward to " << frame << " from " << _begin_frame << "\n";
1135 }
1136 fetch_frame(frame);
1137 }
1138 }
1139
1140 if (ffmpeg_cat.is_spam()) {
1141 ffmpeg_cat.spam()
1142 << "Wanted " << frame << ", got " << _begin_frame << "\n";
1143 }
1144}

1151export_frame(FfmpegBuffer *buffer) {
1152 PStatTimer timer(_export_frame_pcollector);
1153
1154 if (!_frame_ready) {
1155 // No frame data ready, just fill with black.
1156 if (ffmpeg_cat.is_spam()) {
1157 ffmpeg_cat.spam()

1159 << ", no frame available.\n";
1160 }

1162 return;
1163 }
1164
1165 _frame_out->data[0] = buffer->_block + ((_size_y - 1) * _size_x * _num_components);
1166 _frame_out->linesize[0] = _size_x * -_num_components;
1167 buffer->_begin_frame = _begin_frame;
1168 buffer->_end_frame = _end_frame;
1169
1170#ifdef HAVE_SWSCALE
1171 nassertv(_convert_ctx != nullptr && _frame != nullptr);
1172 if (ffmpeg_global_lock) {
1173 ReMutexHolder av_holder(_av_lock);
1174 sws_scale(_convert_ctx, _frame->data, _frame->linesize, 0, _size_y, _frame_out->data, _frame_out->linesize);
1175 } else {
1176 sws_scale(_convert_ctx, _frame->data, _frame->linesize, 0, _size_y, _frame_out->data, _frame_out->linesize);
1177 }
1178#else
1179 nassertv(_frame != nullptr);
1180 uint8_t const *src = _frame->data[0];
1181 uint8_t *dst = _frame_out->data[0];
1182 int src_stride = _frame->linesize[0];
1183 int dst_stride = _frame_out->linesize[0];

1185 nassertv(copy_size <= (size_t)std::abs(src_stride));
1186
1187 for (int y = 0; y < _size_y; ++y) {
1188 memcpy(dst, src, copy_size);

1190 dst += dst_stride;
1191 }
1192#endif
