158void GtkStatsChartMenu::
159do_update() {
160 PStatView &view = _monitor->get_view(_thread_index);
161 _last_level_index = view.get_level_index();
162
163 const PStatClientData *client_data = _monitor->get_client_data();
164 if ((size_t)client_data->get_num_collectors() > _collector_items.size()) {
165 _collector_items.resize(client_data->get_num_collectors(), std::make_pair(nullptr, nullptr));
166 }
167
168 // The menu item(s) for the thread's frame time goes second.
169 const PStatViewLevel *view_level = view.get_top_level();
170 if (_thread_index == 0) {
171 if (add_view(_menu, view_level, false, _time_items_end)) {
172 ++_time_items_end;
173 ++_level_items_end;
174 }
175 } else {
176 for (int c = 0; c < view_level->get_num_children(); ++c) {
177 if (add_view(_menu, view_level->get_child(c), false, _time_items_end)) {
178 ++_time_items_end;
179 ++_level_items_end;
180 }
181 }
182 }
183
184 // And then the menu item(s) for each of the level values.
185 int num_toplevel_collectors = client_data->get_num_toplevel_collectors();
186 for (int tc = 0; tc < num_toplevel_collectors; tc++) {
187 int collector = client_data->get_toplevel_collector(tc);
188 if (client_data->has_collector(collector) &&
189 client_data->get_collector_has_level(collector, _thread_index)) {
190
191 PStatView &level_view = _monitor->get_level_view(collector, _thread_index);
192 add_view(_menu, level_view.get_top_level(), true, _level_items_end);
193 }
194 }
195}

201bool GtkStatsChartMenu::
202add_view(GtkWidget *parent_menu, const PStatViewLevel *view_level,
203 bool show_level, int insert_at) {
204 int collector = view_level->get_collector();
205
206 GtkWidget *&menu_item = _collector_items[collector].first;
207 GtkWidget *&menu = _collector_items[collector].second;
208
209 const PStatClientData *client_data = _monitor->get_client_data();
210
211 int num_children = view_level->get_num_children();
212 if (menu == nullptr && num_children == 0) {
213 // For a collector without children, no point in making a submenu. We just
214 // have the item open a strip chart directly (no point in creating a flame
215 // graph if there are no children).
216 if (menu_item != nullptr) {
217 // Already exists.
218 return false;
219 }
220
221 std::string collector_name = client_data->get_collector_name(collector);
222 menu_item = make_menu_item(
223 collector_name.c_str(), collector, GtkStatsMonitor::CT_strip_chart, show_level);
224 gtk_menu_shell_insert(GTK_MENU_SHELL(parent_menu), menu_item, insert_at);
225 return true;
226 }
227 else if (menu_item != nullptr && menu == nullptr) {
228 // Unhook the signal handler, we are creating a submenu.
229 GtkStatsMonitor::MenuDef smd(GtkStatsMonitor::CT_strip_chart, _thread_index, collector, -1, show_level);
230 const GtkStatsMonitor::MenuDef *menu_def = _monitor->add_menu(smd);
231
232 g_signal_handlers_disconnect_by_data(G_OBJECT(menu_item), (void *)menu_def);
233 }
234
235 // Create a submenu.
236 bool added_item = false;
237 if (menu_item == nullptr) {
238 std::string collector_name = client_data->get_collector_name(collector);
239 menu_item = gtk_menu_item_new_with_label(collector_name.c_str());
240 gtk_widget_show(menu_item);
241 gtk_menu_shell_insert(GTK_MENU_SHELL(parent_menu), menu_item, insert_at);
242 added_item = true;
243 }
244
245 if (menu == nullptr) {
246 menu = gtk_menu_new();
247 gtk_widget_show(menu);
248 gtk_menu_item_set_submenu(GTK_MENU_ITEM(menu_item), menu);
249
250 gtk_menu_shell_append(GTK_MENU_SHELL(menu),
251 make_menu_item("Open Strip Chart", collector, GtkStatsMonitor::CT_strip_chart, show_level));
252
253 if (!show_level) {
254 if (collector == 0) {
255 collector = -1;
256 }
257
258 gtk_menu_shell_append(GTK_MENU_SHELL(menu),
259 make_menu_item("Open Flame Graph", collector, GtkStatsMonitor::CT_flame_graph));
260 }
261
262 GtkWidget *sep = gtk_separator_menu_item_new();
263 gtk_widget_show(sep);
264 gtk_menu_shell_append(GTK_MENU_SHELL(menu), sep);
265 }
266
267 for (int c = 0; c < num_children; ++c) {
268 add_view(menu, view_level->get_child(c), show_level, 3 + !show_level);
269 }
270
271 return added_item;
272}
