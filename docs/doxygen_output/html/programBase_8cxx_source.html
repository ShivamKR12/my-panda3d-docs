188void ProgramBase::
189write_man_page(std::ostream &out) {
190 string prog = _program_name.get_basename_wo_extension();
191 out << ".\\\" Automatically generated by " << prog << " -write-man\n";
192
193 // Format the man page title as the uppercase version of the program name,
194 // as per the UNIX manual conventions.
195 out << ".TH ";
196 string::const_iterator si;
197 for (si = _name.begin(); si != _name.end(); ++si) {
198 out << (char)toupper(*si);
199 }
200
201 // Generate a date string for inclusion into the footer.
202 char date_str[256];
203 date_str[0] = 0;
204 time_t current_time;
205 tm *today = nullptr;
206
207 // This variable overrides the time we write to the footer.
208 const char *source_date_epoch = getenv("SOURCE_DATE_EPOCH");
209 if (source_date_epoch == nullptr || source_date_epoch[0] == 0 ||
210 (current_time = (time_t)strtoll(source_date_epoch, nullptr, 10)) <= 0) {
211 current_time = time(nullptr);
212 if (current_time != (time_t)-1) {
213 today = localtime(&current_time);
214 }
215 }
216 else {
217 // Format as UTC to avoid inconsistency being introduced due to timezones.
218 today = gmtime(&current_time);
219 }
220
221 if (today == nullptr || 0 == strftime(date_str, 256, "%d %B %Y", today)) {
222 date_str[0] = 0;
223 }
224
225 out << " 1 \"" << date_str << "\" \""
226 << PandaSystem::get_version_string() << "\" Panda3D\n";
227
228 out << ".SH NAME\n";
229 if (_brief.empty()) {
230 out << _name << "\n";
231 } else {
232 out << _name << " \\- " << _brief << "\n";
233 }
234
235 out << ".SH SYNOPSIS\n";
236 Runlines::const_iterator ri = _runlines.begin();
237 if (ri != _runlines.end()) {
238 out << "\\fB" << prog << "\\fR " << *ri << "\n";
239 ++ri;
240 }
241
242 for (; ri != _runlines.end(); ++ri) {
243 out << ".br\n";
244 out << "\\fB" << prog << "\\fR " << *ri << "\n";
245 }
246
247 out << ".SH DESCRIPTION\n";
248 string::const_iterator di;
249 char prev = 0;
250 for (di = _description.begin(); di != _description.end(); ++di) {
251 if ((*di) == '-') {
252 // We have to escape hyphens.
253 out << "\\-";
254 } else if (prev == '\n' && (*di) == '\n') {
255 // Indicate the start of a new paragraph.
256 out << ".PP\n";
257 } else {
258 out << (char)(*di);
259 }
260 prev = (*di);
261 }
262 out << "\n";
263
264 out << ".SH OPTIONS\n";
265 sort_options();
266 OptionsByIndex::const_iterator oi;
267 for (oi = _options_by_index.begin(); oi != _options_by_index.end(); ++oi) {
268 const Option &opt = *(*oi);
269 out << ".TP\n";
270
271 if (opt._parm_name.empty()) {
272 out << ".B \\-" << opt._option << "\n";
273 } else {
274 out << ".BI \"\\-" << opt._option << " \" \"" << opt._parm_name << "\"\n";
275 }
276 out << opt._description << "\n";
277 }

286ProgramBase::ExitCode ProgramBase::
287parse_command_line(int argc, char **argv, bool exit_on_complete) {
288 if (!exit_on_complete) {
289 // The help option should not be available in programmatic environments.
290 remove_option("h");
291 }
292
293 preprocess_argv(argc, argv);
294
295 // Setting this variable to zero reinitializes the options parser This is
296 // only necessary for processing multiple command lines in the same program
297 // (mainly the MaxToEgg converter plugin)
298 extern int optind;
299 optind = 0;
300
301#if !defined(HAVE_GETOPT) || !defined(HAVE_GETOPT_LONG_ONLY)
302 // We're using a Panda implementation of getopt. Let's reset that as well.
303 pgetopt_reset();
304#endif
305
306 _program_name = Filename::from_os_specific(argv[0]);
307 int i;
308 for (i = 1; i < argc; i++) {
309 _program_args.push_back(argv[i]);
310 }
311
312 if (_name.empty()) {
313 _name = _program_name.get_basename_wo_extension();
314 }
315
316 // Catch a special hidden option: -write-man, which causes the tool to
317 // generate a manual page.
318 if (argc > 1 && strcmp(argv[1], "-write-man") == 0) {
319 if (argc == 2) {
320 write_man_page(cout);
321
322 } else if (argc == 3) {
323 if (strlen(argv[2]) == 1 && argv[2][0] == '-') {
324 write_man_page(cout);
325
326 } else {
327 pofstream man_out(argv[2], std::ios::out | std::ios::trunc);
328 if (!man_out) {
329 cerr << "Failed to open output file " << argv[2] << "!\n";
330 }
331 write_man_page(man_out);
332 man_out.close();
333 }
334 } else {
335 cerr << "Invalid number of options for -write-man!\n";
336 if (exit_on_complete) {
337 exit(1);
338 }
339 return ExitCode::EC_failure;
340 }
341
342 if (exit_on_complete) {
343 exit(0);
344 }
345 return ExitCode::EC_clean_exit;
346 }
347
348 // Build up the long options list and the short options string for
349 // getopt_long_only().
350 pvector<struct option> long_options;
351 string short_options;
352
353 // We also need to build a temporary map of int index numbers to Option
354 // pointers. We'll pass these index numbers to GNU's getopt_long() so we
355 // can tell one option from another.
356 typedef pmap<int, const Option *> Options;
357 Options options;
358
359 OptionsByName::const_iterator oi;
360 int next_index = 256;
361
362 // Let's prefix the option string with "-" to tell getopt that we want it to
363 // tell us the post-option arguments, instead of trying to meddle with ARGC
364 // and ARGV (which we aren't using directly).
365 short_options = "-";
366
367 for (oi = _options_by_name.begin(); oi != _options_by_name.end(); ++oi) {
368 const Option &opt = (*oi).second;
369
370 int index;
371 if (opt._option.length() == 1) {
372 // This is a "short" option; its option string consists of only one
373 // letter. Its index is the letter itself.
374 index = (int)opt._option[0];
375
376 short_options += opt._option;
377 if (!opt._parm_name.empty()) {
378 // This option takes an argument.
379 short_options += ':';
380 }
381 } else {
382 // This is a "long" option; we'll assign it the next available index.
383 index = ++next_index;
384 }
385
386 // Now add it to the GNU data structures.
387 struct option gopt;
388 gopt.name = (char *)opt._option.c_str();
389 gopt.has_arg = (opt._parm_name.empty()) ?
390 no_argument : required_argument;
391 gopt.flag = nullptr;
392
393 // Return an index into the _options_by_index array, offset by 256 so we
394 // don't confuse it with '?'.
395 gopt.val = index;
396
397 long_options.push_back(gopt);
398
399 options[index] = &opt;
400 }
401
402 // Finally, add one more structure, all zeroes, to indicate the end of the
403 // options.
404 struct option gopt;
405 memset(&gopt, 0, sizeof(gopt));
406 long_options.push_back(gopt);
407
408 // We'll use this vector to save the non-option arguments. Generally, these
409 // will all be at the end, but with the GNU extensions, they need not be.
410 Args remaining_args;
411
412 // Now call getopt_long() to actually parse the arguments.
413 extern char *optarg;
414 const struct option *long_opts = &long_options[0];
415
416 int flag =
417 getopt_long_only(argc, argv, short_options.c_str(), long_opts, nullptr);
418 while (flag != EOF) {
419 string arg;
420 if (optarg != nullptr) {
421 arg = optarg;
422 }
423
424 switch (flag) {
425 case '?':
426 // Invalid option or parameter.
427 show_usage();
428 if (exit_on_complete) {
429 exit(1);
430 }
431 return ExitCode::EC_failure;
432
433 case '\x1':
434 // A special return value from getopt() indicating a non-option
435 // argument.
436 remaining_args.push_back(arg);
437 break;
438
439 default:
440 {
441 // A normal option. Figure out which one it is.
442 Options::const_iterator ii;
443 ii = options.find(flag);
444 if (ii == options.end()) {
445 nout << "Internal error! Invalid option index returned.\n";
446 abort();
447 }
448
449 const Option &opt = *(*ii).second;
450 bool okflag = true;
451 if (opt._option_function != (OptionDispatchFunction)nullptr) {
452 okflag = (*opt._option_function)(opt._option, arg, opt._option_data);
453 }
454 if (opt._option_method != (OptionDispatchMethod)nullptr) {
455 okflag = (*opt._option_method)(this, opt._option, arg, opt._option_data);
456 }
457 if (opt._bool_var != nullptr) {
458 (*opt._bool_var) = true;
459 }
460
461 if (!okflag) {
462 show_usage();
463 if (exit_on_complete) {
464 exit(1);
465 }
466 return ExitCode::EC_failure;
467 }
468 }
469 }
470
471 flag =
472 getopt_long_only(argc, argv, short_options.c_str(), long_opts, nullptr);
473 }
474
475 if (!handle_args(remaining_args)) {
476 show_usage();
477 if (exit_on_complete) {
478 exit(1);
479 }
480 return ExitCode::EC_failure;
481 }
482
483 if (!post_command_line()) {
484 show_usage();
485 if (exit_on_complete) {
486 exit(1);
487 }
488 return ExitCode::EC_failure;
489 }
490
491 return ExitCode::EC_not_exited;
492}

498string ProgramBase::
499get_exec_command() const {
500 string command;
501
502 command = _program_name.get_basename_wo_extension();
503 Args::const_iterator ai;
504 for (ai = _program_args.begin(); ai != _program_args.end(); ++ai) {
505 const string &arg = (*ai);
506
507 // First, check to see if the string is shell-acceptable.
508 bool legal = true;
509 string::const_iterator si;
510 for (si = arg.begin(); legal && si != arg.end(); ++si) {
511 switch (*si) {
512 case ' ':
513 case '\n':
514 case '\t':
515 case '*':
516 case '?':
517 case '\\':
518 case '(':
519 case ')':
520 case '|':
521 case '&':
522 case '<':
523 case '>':
524 case '"':
525 case ';':
526 case '$':
527 legal = false;
528 }
529 }
530
531 if (legal) {
532 command += " " + arg;
533 } else {
534 command += " '" + arg + "'";
535 }
536 }
537
538 return command;
539}

1328void ProgramBase::
1329format_text(std::ostream &out, bool &last_newline,
1330 const string &prefix, int indent_width,
1331 const string &text, int line_width) {
1332 indent_width = min(indent_width, line_width - 20);
1333 int indent_amount = indent_width;
1334 bool initial_break = false;
1335
1336 if (!prefix.empty()) {
1337 out << prefix;
1338 indent_amount = indent_width - prefix.length();
1339 if ((int)prefix.length() + 1 > indent_width) {
1340 out << "\n";
1341 initial_break = true;
1342 indent_amount = indent_width;
1343 }
1344 }
1345
1346 size_t p = 0;
1347
1348 // Skip any initial whitespace and newlines.
1349 while (p < text.length() && isspace(text[p])) {
1350 if (text[p] == '\r' ||
1351 (p > 0 && text[p] == '\n' && text[p - 1] == '\n') ||
1352 (p == 0 && text[p] == '\n' && last_newline)) {
1353 if (!initial_break) {
1354 // Here's an initial paragraph break, however.
1355 out << "\n";
1356 initial_break = true;
1357 }
1358 indent_amount = indent_width;
1359
1360 } else if (text[p] == '\n') {
1361 // Largely ignore an initial newline.
1362 indent_amount = indent_width;
1363
1364 } else if (text[p] == ' ') {
1365 // Do count up leading spaces.
1366 indent_amount++;
1367 }
1368 p++;
1369 }
1370
1371 last_newline = (!text.empty() && text[text.length() - 1] == '\n');
1372
1373 while (p < text.length()) {
1374 // Look for the paragraph or line break--the next newline character, if
1375 // any.
1376 size_t par = text.find_first_of("\n\r", p);
1377 bool is_paragraph_break = false;
1378 if (par == string::npos) {
1379 par = text.length();
1380 /*
1381 This shouldn't be necessary.
1382 } else {
1383 is_paragraph_break = (text[par] == '\r');
1384 */
1385 }
1386
1387 indent(out, indent_amount);
1388
1389 size_t eol = p + (line_width - indent_width);
1390 if (eol >= par) {
1391 // The rest of the paragraph fits completely on the line.
1392 eol = par;
1393
1394 } else {
1395 // The paragraph doesn't fit completely on the line. Determine the best
1396 // place to break the line. Look for the last space before the ideal
1397 // eol.
1398 size_t min_eol = max((int)p, (int)eol - 25);
1399 size_t q = eol;
1400 while (q > min_eol && !isspace(text[q])) {
1401 q--;
1402 }
1403 // Now roll back to the last non-space before this one.
1404 while (q > min_eol && isspace(text[q])) {
1405 q--;
1406 }
1407
1408 if (q != min_eol) {
1409 // Here's a good place to stop!
1410 eol = q + 1;
1411
1412 } else {
1413 // The line cannot be broken cleanly. Just let it keep going; don't
1414 // try to wrap it.
1415 eol = par;
1416 }
1417 }
1418 out << text.substr(p, eol - p) << "\n";
1419 p = eol;
1420
1421 // Skip additional whitespace between the lines.
1422 while (p < text.length() && isspace(text[p])) {
1423 if (text[p] == '\r' ||
1424 (p > 0 && text[p] == '\n' && text[p - 1] == '\n')) {
1425 is_paragraph_break = true;
1426 }
1427 p++;
1428 }
1429
1430 if (eol == par && is_paragraph_break) {
1431 // Print the paragraph break as a blank line.
1432 out << "\n";
1433 if (p >= text.length()) {
1434 // If we end on a paragraph break, don't try to insert a new one in
1435 // the next pass.
1436 last_newline = false;
1437 }
1438 }
1439
1440 indent_amount = indent_width;
1441 }
1442}
