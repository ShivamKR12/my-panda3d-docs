35bool TypeRegistryNode::
36is_derived_from(const TypeRegistryNode *child, const TypeRegistryNode *base) {
37 // This function is the basis for TypedObject::is_of_type(), which gets used
38 // quite frequently within Panda, often in inner-loop code. Therefore, we
39 // go through some pains to make this function as efficient as possible.
40
41 // First, compare the subtree tops. If they are the same, then this node
42 // and the base node are within the same single-inheritance subtree, and we
43 // can use our bitmask trick to determine the relationship with no
44 // additional work. (See r_build_subtrees()).
45
46 if (child->_inherit._top == base->_inherit._top) {
47 assert(child->_inherit._top != nullptr);
48
49 bool derives =
50 Inherit::is_derived_from(child->_inherit, base->_inherit);
51
52#ifndef NDEBUG
53 if (_paranoid_inheritance) {
54 bool paranoid_derives = check_derived_from(child, base);
55 if (derives != paranoid_derives) {
56 std::cerr
57 << "Inheritance test for " << child->_name
58 << " from " << base->_name << " failed!\n"
59 << "Result: " << derives << " should have been: "
60 << paranoid_derives << "\n"
61 << "Classes are in the same single inheritance subtree, children of "
62 << child->_inherit._top->_name << "\n"
63 << std::hex
64 << child->_name << " has mask " << child->_inherit._mask
65 << " and bits " << child->_inherit._bits << "\n"
66 << base->_name << " has mask " << base->_inherit._mask
67 << " and bits " << base->_inherit._bits << "\n"
68 << std::dec;
69 return paranoid_derives;
70 }
71 }
72#endif
73
74 /*
75 cerr << "trivial: " << child->_name << " vs. " << base->_name
76 << " (" << child->_inherit._top->_name << ") = " << derives << "\n";
77 */
78
79 return derives;
80 }
81
82 // The two nodes are not within the same single-inheritance subtree. This
83 // complicates things a bit.
84
85 // First, we should check whether the subtree tops of the two nodes inherit
86 // from each other.
87 TypeRegistryNode *child_top = child->_inherit._top;
88 TypeRegistryNode *base_top = base->_inherit._top;
89
90 bool derives = false;
91
92 // If child_top does not inherit from base_top, it follows that child does
93 // not inherit from base.
94 TopInheritance::const_iterator ti =
95 lower_bound(child_top->_top_inheritance.begin(),
96 child_top->_top_inheritance.end(),
97 Inherit(base_top, 0, 0));
98
99 while (ti != child_top->_top_inheritance.end() &&
100 (*ti)._top == base_top &&
101 !derives) {
102 // If child_top *does* inherit from base_top, then child may or may not
103 // inherit from base. This depends on the exact path of inheritance.
104 // Since there might be multiple paths from child_top to base_top, we have
105 // to examine all of them.
106 const Inherit &connection = (*ti);
107
108 // Here is one inheritance from child_top to base_top. If the connecting
109 // node inherits from base, then child also inherits from base. If the
110 // connecting node does not inherit from base, we must keep looking.
111 derives = Inherit::is_derived_from(connection, base->_inherit);
112
113 ++ti;
114 }
115
116#ifndef NDEBUG
117 if (_paranoid_inheritance) {
118 bool paranoid_derives = check_derived_from(child, base);
119 if (derives != paranoid_derives) {
120 std::cerr
121 << "Inheritance test for " << child->_name
122 << " from " << base->_name << " failed!\n"
123 << "Result: " << derives << " should have been: "
124 << paranoid_derives << "\n"
125 << child->_name << " is a descendent of "
126 << child_top->_name << "\n"
127 << base->_name << " is a descendent of "
128 << base_top->_name << "\n";
129 return paranoid_derives;
130 }
131 }
132#endif
133
134 /*
135 cerr << "complex: " << child->_name << " (" << child->_inherit._top->_name
136 << ") vs. " << base->_name << " (" << base->_inherit._top->_name
137 << ") = " << derives << "\n";
138 */
139 return derives;
140}

146TypeHandle TypeRegistryNode::

202void TypeRegistryNode::
203r_build_subtrees(TypeRegistryNode *top, int bit_count,
204 TypeRegistryNode::SubtreeMaskType bits) {
205 // The idea with these bits is to optimize the common case of a single-
206 // inheritance graph (that is, an inheritance tree), or a single-inheritance
207 // subgraph of the full multiple-inheritance graph (i.e. a subtree of the
208 // inheritance graph).
209
210/*
211 * When we have just single inheritance, we can define a unique number for
212 * each node in the inheritance tree that allows us to immediately determine
213 * the inheritance relationship between any two nodes in the tree. We choose
214 * a number such that for a given node whose number has n bits, each child
215 * node has m + n bits where the low-order n bits are the same as the parent
216 * node's bits, and the high-order m bits are unique among each sibling. The
217 * node at the top of the tree has zero bits.
218 */
219
220 // That way, we can simply compare bitmasks to determine if class A inherits
221 // from class B. If the low-order bits are the same, they have some
222 // ancestry in common. The highest-order bit that still matches corresponds
223 // to the lowest node in the tree that they have in common; i.e. the node
224 // from which they both inherit.
225
226 // To put it more formally, let count(A) be the number of bits in A's
227 // number, and count(B) be the number of bits in B's number. A inherits
228 // from B if and only if count(B) <= count(A), and the lower count(B) bits
229 // of A's number are the same as those in B's number.
230
231 // This algorithm breaks down in the presence of multiple inheritance, since
232 // we can't make up a single number for each node any more. We still take
233 // advantage of the algorithm by considering each single-inheritance
234 // subgraph separately.
235
236 // To handle multiple inheritance, we reset the numbers to zero every time
237 // we come across a multiple-inheritance node (this begins a new subtree).
238 // There are relatively few of these "subtree top" nodes, and we record the
239 // explicit inheritance of each one from all of its ancestor "subtree top"
240 // nodes within the node itself.
241
242 if (top != this && _parent_classes.size() != 1) {
243 assert(!_parent_classes.empty());
244
245 // This class multiply inherits; it therefore begins a new subtree.
246
247 // Copy in the inheritance relations from our parent subtree tops.
248 _top_inheritance.insert(_top_inheritance.end(),
249 top->_top_inheritance.begin(),
250 top->_top_inheritance.end());
251 _top_inheritance.push_back(Inherit(top, bit_count, bits));
252
253 _visit_count++;
254 if (_visit_count == (int)_parent_classes.size()) {
255 // This is the last time we'll visit this node, so continue the
256 // recursion now.
257 assert(_inherit._top == nullptr);
258 sort(_top_inheritance.begin(), _top_inheritance.end());
259 define_subtree();
260 }
261
262 } else {
263 // This class singly inherits, so this had better be the only time this
264 // function is called on it since clear_subtree().
265 assert(_inherit._top == nullptr);
266
267 assert(bit_count < (int)(sizeof(SubtreeMaskType) * 8));
268
269 _inherit = Inherit(top, bit_count, bits);
270
271 // Now, how many more bits do we need to encode each of our children?
272 int num_children = (int)_child_classes.size();
273 int more_bits = 0;
274 int i = num_children - 1;
275 while (i > 0) {
276 more_bits++;
277 i >>= 1;
278 }
279
280 // We need at least one bit, even if there is only one child, so we can
281 // differentiate parent from child.
282 more_bits = std::max(more_bits, 1);
283
284 assert(more_bits < (int)(sizeof(SubtreeMaskType) * 8));
285
286 if (bit_count + more_bits > (int)(sizeof(SubtreeMaskType) * 8)) {
287 // Too many bits; we need to start a new subtree right here. This node
288 // becomes a subtree top node, even though it's not a multiple-
289 // inheritance node.
290 assert(top != this);
291 _top_inheritance = top->_top_inheritance;
292 _top_inheritance.push_back(_inherit);
293 sort(_top_inheritance.begin(), _top_inheritance.end());
294 _inherit = Inherit();
295 define_subtree();
296
297 } else {
298 // Still plenty of bits, so keep going.
299 for (i = 0; i < num_children; i++) {
300 TypeRegistryNode *child = _child_classes[i];
301 SubtreeMaskType next_bits = ((SubtreeMaskType)i << bit_count);
302
303 child->r_build_subtrees(top, bit_count + more_bits,
304 bits | next_bits);
305 }
306 }
307 }
