244void GtkStatsFlameGraph::
245draw_bar(int depth, int from_x, int to_x, int collector_index, int parent_index) {
246 double bottom = get_ysize() - depth * _pixel_scale * 5;
247 double top = bottom - _pixel_scale * 5;
248
249 bool is_highlighted = collector_index == _highlighted_index;
250 cairo_set_source(_cr, get_collector_pattern(collector_index, is_highlighted));
251
252 if (to_x < from_x + 3) {
253 // It's just a tiny sliver. This is a more reliable way to draw it.
254 cairo_rectangle(_cr, from_x, top, to_x - from_x, bottom - top);
255 cairo_fill(_cr);
256 }
257 else {
258 double radius = std::min((double)_pixel_scale, (to_x - from_x) / 2.0);
259 cairo_new_sub_path(_cr);
260 cairo_arc(_cr, to_x - radius, top + radius, radius, -0.5 * M_PI, 0.0);
261 cairo_arc(_cr, to_x - radius, bottom - radius, radius, 0.0, 0.5 * M_PI);
262 cairo_arc(_cr, from_x + radius, bottom - radius, radius, 0.5 * M_PI, M_PI);
263 cairo_arc(_cr, from_x + radius, top + radius, radius, M_PI, 1.5 * M_PI);
264 cairo_close_path(_cr);
265 cairo_fill(_cr);
266
267 if ((to_x - from_x) >= _pixel_scale * 4) {
268 // Only bother drawing the text if we've got some space to draw on.
269 int left = std::max(from_x, 0) + _pixel_scale / 2;
270 int right = std::min(to_x, get_xsize()) - _pixel_scale / 2;
271
272 const PStatClientData *client_data = GtkStatsGraph::_monitor->get_client_data();
273 const PStatCollectorDef &def = client_data->get_collector_def(collector_index);
274
275 // Choose a suitable foreground color.
276 LRGBColor fg = get_collector_text_color(collector_index, is_highlighted);
277 cairo_set_source_rgb(_cr, fg[0], fg[1], fg[2]);
278
279 PangoLayout *layout = gtk_widget_create_pango_layout(_graph_window, def._name.c_str());
280 pango_layout_set_attributes(layout, _pango_attrs);
281 pango_layout_set_ellipsize(layout, PANGO_ELLIPSIZE_END);
282 pango_layout_set_width(layout, (right - left) * PANGO_SCALE);
283 pango_layout_set_height(layout, -1);
284
285 if (!pango_layout_is_ellipsized(layout)) {
286 // We have room for more. Show the collector's actual parent, if it's
287 // different than the block it's shown above.
288 if (def._parent_index > 0 && def._parent_index != parent_index) {
289 const PStatCollectorDef &parent_def = client_data->get_collector_def(def._parent_index);
290 std::string long_name = parent_def._name + ":" + def._name;
291 pango_layout_set_text(layout, long_name.c_str(), long_name.size());
292 if (pango_layout_is_ellipsized(layout)) {
293 // Nope, it's too long, go back.
294 pango_layout_set_text(layout, def._name.c_str(), def._name.size());
295 }
296 }
297 else if (collector_index == 0 && get_frame_number() >= 0) {
298 char text[32];
299 sprintf(text, "Frame %d", get_frame_number());
300 pango_layout_set_text(layout, text, -1);
301 if (pango_layout_is_ellipsized(layout)) {
302 // Nope, it's too long, go back.
303 pango_layout_set_text(layout, def._name.c_str(), def._name.size());
304 }
305 }
306 }
307
308 int width, height;
309 pango_layout_get_pixel_size(layout, &width, &height);
310
311 // Center the text vertically in the bar.
312 cairo_move_to(_cr, left, top + (bottom - top - height) / 2);
313 pango_cairo_show_layout(_cr, layout);
314
315 g_object_unref(layout);
316 }
317 }

566gboolean GtkStatsFlameGraph::
567handle_motion(int graph_x, int graph_y) {
568 if (_drag_mode == DM_none && _potential_drag_mode == DM_none &&
569 graph_x >= 0 && graph_y >= 0 && graph_x < get_xsize() && graph_y < get_ysize()) {
570 // When the mouse is over a color bar, highlight it.
571 int depth = pixel_to_depth(graph_y);
572 int collector_index = get_bar_collector(depth, graph_x);
573 on_enter_label(collector_index);
574 }
575 else {
576 // If the mouse is in some drag mode, stop highlighting.
577 _label_stack.highlight_label(-1);
578 on_leave_label(_highlighted_index);
579 }
580
581 if (_drag_mode == DM_new_guide_bar) {
582 // We haven't created the new guide bar yet; we won't until the mouse
583 // comes within the graph's region.
584 if (graph_x >= 0 && graph_x < get_xsize()) {
585 set_drag_mode(DM_guide_bar);
586 _drag_guide_bar = add_user_guide_bar(pixel_to_height(graph_x));
587 return TRUE;
588 }
589 }
590 else if (_drag_mode == DM_guide_bar) {
591 move_user_guide_bar(_drag_guide_bar, pixel_to_height(graph_x));
592 return TRUE;
593 }
594
595 return GtkStatsGraph::handle_motion(graph_x, graph_y);

664void GtkStatsFlameGraph::
665draw_guide_label(cairo_t *cr, const PStatGraph::GuideBar &bar) {
666 switch (bar._style) {
667 case GBS_target:
668 cairo_set_source_rgb(cr, rgb_light_gray[0], rgb_light_gray[1], rgb_light_gray[2]);
669 break;
670
671 case GBS_user:
672 cairo_set_source_rgb(cr, rgb_user_guide_bar[0], rgb_user_guide_bar[1], rgb_user_guide_bar[2]);
673 break;
674
675 default:
676 cairo_set_source_rgb(cr, rgb_dark_gray[0], rgb_dark_gray[1], rgb_dark_gray[2]);
677 break;
678 }
679
680 int x = height_to_pixel(bar._height);
681 const std::string &label = bar._label;
682
683 PangoLayout *layout = gtk_widget_create_pango_layout(_scale_area, label.c_str());
684 int width, height;
685 pango_layout_get_pixel_size(layout, &width, &height);
686
687 if (bar._style != GBS_user) {
688 double from_height = pixel_to_height(x - width * _cr_scale);
689 double to_height = pixel_to_height(x + width * _cr_scale);
690 if (find_user_guide_bar(from_height, to_height) >= 0) {
691 // Omit the label: there's a user-defined guide bar in the same space.
692 g_object_unref(layout);
693 return;
694 }
695 }
696
697 if (x >= 0 && x < get_xsize()) {
698 // Now convert our x to a coordinate within our drawing area.
699 int junk_y;
700
701 x /= _cr_scale;
702
703 // The x coordinate comes from the graph_window.
704 gtk_widget_translate_coordinates(_graph_window, _scale_area,
705 x, 0,
706 &x, &junk_y);
707
708 GtkAllocation allocation;
709 gtk_widget_get_allocation(_scale_area, &allocation);
710
711 int this_x = x - width / 2;
712 if (this_x >= 0 && this_x + width < allocation.width) {
713 cairo_move_to(cr, this_x, allocation.height - height);
714 pango_cairo_show_layout(cr, layout);
715 }
716 }
717
718 g_object_unref(layout);
719}
