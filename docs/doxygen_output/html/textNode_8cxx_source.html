376void TextNode::
377apply_attribs_to_vertices(const AccumulatedAttribs &attribs, int attrib_types,
378 GeomTransformer &transformer) {
379
380 Thread *current_thread = Thread::get_current_thread();
381 OPEN_ITERATE_CURRENT_AND_UPSTREAM(_cycler, current_thread) {
382 CDStageWriter cdata(_cycler, pipeline_stage, current_thread);
383
384 if ((attrib_types & SceneGraphReducer::TT_transform) != 0) {
385 const LMatrix4 &mat = attribs._transform->get_mat();
386 cdata->_transform *= mat;
387
388 if ((cdata->_flags & F_needs_measure) == 0) {
389 // If we already have a measure, transform it too. We don't need to
390 // invalidate the 2-d parts, since that's not affected by the transform
391 // anyway.
392 cdata->_ul3d = cdata->_ul3d * mat;
393 cdata->_lr3d = cdata->_lr3d * mat;
394 }
395 }
396 if ((attrib_types & SceneGraphReducer::TT_color) != 0) {
397 if (attribs._color != nullptr) {
398 const ColorAttrib *ca = DCAST(ColorAttrib, attribs._color);
399 if (ca->get_color_type() == ColorAttrib::T_flat) {
400 const LColor &c = ca->get_color();
401 TextProperties::set_text_color(c);
402 TextProperties::set_shadow_color(c);
403 cdata->_frame_color = c;
404 cdata->_card_color = c;
405 invalidate_no_measure(cdata);
406 }
407 }
408 }
409 if ((attrib_types & SceneGraphReducer::TT_color_scale) != 0) {
410 if (attribs._color_scale != nullptr) {
411 const ColorScaleAttrib *csa = DCAST(ColorScaleAttrib, attribs._color_scale);
412 const LVecBase4 &s = csa->get_scale();
413 if (s != LVecBase4(1.0f, 1.0f, 1.0f, 1.0f)) {
414 LVecBase4 tc = get_text_color();
415 tc.componentwise_mult(s);
416 TextProperties::set_text_color(tc);
417
418 LVecBase4 sc = get_shadow_color();
419 sc.componentwise_mult(s);
420 TextProperties::set_shadow_color(sc);
421
422 cdata->_frame_color.componentwise_mult(s);
423 cdata->_card_color.componentwise_mult(s);
424
425 invalidate_no_measure(cdata);
426 }
427 }
428 }
429
430 // Now propagate the attributes down to our already-generated geometry, if
431 // we have any.
432 if ((cdata->_flags & F_needs_rebuild) == 0 && cdata->_internal_geom != nullptr) {
433 SceneGraphReducer gr;
434 gr.apply_attribs(cdata->_internal_geom, attribs, attrib_types, transformer);
435 }
436 }
437 CLOSE_ITERATE_CURRENT_AND_UPSTREAM(_cycler);
438}

503void TextNode::
504compute_internal_bounds(CPT(BoundingVolume) &internal_bounds,
505 int &internal_vertices,
506 int pipeline_stage,
507 Thread *current_thread) const {
508 // First, get ourselves a fresh, empty bounding volume.
509 PT(BoundingVolume) bound = new BoundingSphere;
510
511 GeometricBoundingVolume *gbv = DCAST(GeometricBoundingVolume, bound);
512
513 // Now enclose the bounding box around the text. We can do this without
514 // actually generating the text, if we have at least measured it.
515 LPoint3 vertices[8];
516 {
517 LPoint3 ul3d, lr3d;
518 CDLockedReader cdata(_cycler);
519 if (do_needs_measure(cdata)) {
520 CDWriter cdataw(((TextNode *)this)->_cycler, cdata, false);
521 ((TextNode *)this)->do_measure(cdataw);
522
523 ul3d = cdataw->_ul3d;
524 lr3d = cdataw->_lr3d;
525 } else {
526 ul3d = cdata->_ul3d;
527 lr3d = cdata->_lr3d;
528 }
529
530 vertices[0].set(ul3d[0], ul3d[1], ul3d[2]);
531 vertices[1].set(ul3d[0], ul3d[1], lr3d[2]);
532 vertices[2].set(ul3d[0], lr3d[1], ul3d[2]);
533 vertices[3].set(ul3d[0], lr3d[1], lr3d[2]);
534 vertices[4].set(lr3d[0], ul3d[1], ul3d[2]);
535 vertices[5].set(lr3d[0], ul3d[1], lr3d[2]);
536 vertices[6].set(lr3d[0], lr3d[1], ul3d[2]);
537 vertices[7].set(lr3d[0], lr3d[1], lr3d[2]);
538 }
539
540 gbv->around(vertices, vertices + 8);
541
542 internal_bounds = bound;
543 internal_vertices = 0; // TODO: estimate this better.
544}

588PT(PandaNode) TextNode::
589do_generate(CData *cdata) {
590 PStatTimer timer(_text_generate_pcollector);
591 if (text_cat.is_debug()) {
592 text_cat.debug()
593 << "Rebuilding " << get_type() << " " << get_name()
594 << " with '" << cdata->_text << "'\n";
595 }
596
597 // The strategy here will be to assemble together a bunch of letters,
598 // instanced from the letter hierarchy of font_def, into our own little
599 // hierarchy.
600
601 // There will be one root over the whole text block, that contains the
602 // transform passed in. Under this root there will be another node for each
603 // row, that moves the row into the right place horizontally and vertically,
604 // and for each row, there is another node for each character.
605
606 cdata->_ul3d.set(0.0f, 0.0f, 0.0f);
607 cdata->_lr3d.set(0.0f, 0.0f, 0.0f);
608
609 // Now build a new sub-tree for all the text components.
610 string name = cdata->_text;
611 size_t newline = name.find('\n');
612 if (newline != string::npos) {
613 name = name.substr(0, newline);
614 }
615 PT(PandaNode) root = new PandaNode(name);
616
617 if (cdata->_wtext.empty()) {
618 return root;
619 }
620
621 TextFont *font = get_font();
622 if (font == nullptr) {
623 return root;
624 }
625
626 // Compute the overall text transform matrix. We build the text in a Z-up
627 // coordinate system and then convert it to whatever the user asked for.
628 LMatrix4 mat =
629 LMatrix4::convert_mat(CS_zup_right, cdata->_coordinate_system) *
630 cdata->_transform;
631
632 CPT(TransformState) transform = TransformState::make_mat(mat);
633 root->set_transform(transform);
634
635 // Assemble the text.
636 TextAssembler assembler(this);
637 assembler.set_properties(*this);
638 assembler.set_max_rows(cdata->_max_rows);
639 assembler.set_usage_hint(cdata->_usage_hint);
640 assembler.set_dynamic_merge((cdata->_flatten_flags & FF_dynamic_merge) != 0);
641 bool all_set = assembler.set_wtext(cdata->_wtext);
642 if (all_set) {
643 // No overflow.
644 cdata->_flags &= ~F_has_overflow;
645 } else {
646 // Overflow.
647 cdata->_flags |= F_has_overflow;
648 }
649
650 PT(PandaNode) text_root = assembler.assemble_text();
651 cdata->_text_ul = assembler.get_ul();
652 cdata->_text_lr = assembler.get_lr();
653 cdata->_num_rows = assembler.get_num_rows();
654 cdata->_wordwrapped_wtext = assembler.get_wordwrapped_wtext();
655
656 // Parent the text in.
657 PT(PandaNode) text = new PandaNode("text");
658 root->add_child(text, get_draw_order() + 2);
659 text->add_child(text_root);
660
661 // Save the bounding-box information about the text in a form friendly to
662 // the user.
663 const LVector2 &ul = assembler.get_ul();
664 const LVector2 &lr = assembler.get_lr();
665 cdata->_ul3d.set(ul[0], 0.0f, ul[1]);
666 cdata->_lr3d.set(lr[0], 0.0f, lr[1]);
667
668 cdata->_ul3d = cdata->_ul3d * cdata->_transform;
669 cdata->_lr3d = cdata->_lr3d * cdata->_transform;
670
671 // Incidentally, that means we don't need to measure the text now.
672 cdata->_flags &= ~F_needs_measure;
673
674 // Now flatten our hierarchy to get rid of the transforms we put in,
675 // applying them to the vertices.
676
677 NodePath root_np(root);
678 if (cdata->_flatten_flags & FF_strong) {
679 root_np.flatten_strong();
680 }
681 else if (cdata->_flatten_flags & FF_medium) {
682 root_np.flatten_medium();
683 }
684 else if (cdata->_flatten_flags & FF_light) {
685 root_np.flatten_light();
686 }
687
688 // Now deal with the decorations.
689
690 if (cdata->_flags & F_has_card) {
691 PT(PandaNode) card_root;
692 if (cdata->_flags & F_has_card_border) {
693 card_root = do_make_card_with_border(cdata);
694 } else {
695 card_root = do_make_card(cdata);
696 }
697 card_root->set_transform(transform);
698 card_root->set_attrib(ColorAttrib::make_flat(cdata->_card_color));
699 if (cdata->_card_color[3] != 1.0f) {
700 card_root->set_attrib(TransparencyAttrib::make(TransparencyAttrib::M_alpha));
701 }
702 if (cdata->_flags & F_has_card_texture) {
703 card_root->set_attrib(TextureAttrib::make(cdata->_card_texture));
704 }
705
706 if (has_bin()) {
707 card_root->set_attrib(CullBinAttrib::make(get_bin(), get_draw_order()));
708 }
709
710 // We always apply attribs down to the card vertices.
711 SceneGraphReducer gr;
712 gr.apply_attribs(card_root);
713
714 // In order to decal the text onto the card, the card must become the
715 // parent of the text.
716 card_root->add_child(root);
717 root = card_root;
718
719 if (cdata->_flags & F_card_decal) {
720 card_root->set_effect(DecalEffect::make());
721 }
722 }
723
724 if (cdata->_flags & F_has_frame) {
725 PT(PandaNode) frame_root = do_make_frame(cdata);
726 frame_root->set_transform(transform);
727 root->add_child(frame_root, get_draw_order() + 1);
728 frame_root->set_attrib(ColorAttrib::make_flat(cdata->_frame_color));
729 if (cdata->_frame_color[3] != 1.0f) {
730 frame_root->set_attrib(TransparencyAttrib::make(TransparencyAttrib::M_alpha));
731 }
732
733 if (has_bin()) {
734 frame_root->set_attrib(CullBinAttrib::make(get_bin(), get_draw_order() + 1));
735 }
736
737 SceneGraphReducer gr;
738 gr.apply_attribs(frame_root);
739 }
740
741 return root;
742}

767PT(PandaNode) TextNode::
768do_make_frame(const CData *cdata) {
769 nassertr((cdata->_flags & F_needs_measure) == 0, nullptr);
770
771 PT(GeomNode) frame_node = new GeomNode("frame");
772
773 PN_stdfloat left = cdata->_frame_ul[0];
774 PN_stdfloat right = cdata->_frame_lr[0];
775 PN_stdfloat bottom = cdata->_frame_lr[1];
776 PN_stdfloat top = cdata->_frame_ul[1];
777
778 if (cdata->_flags & F_frame_as_margin) {
779 left = cdata->_text_ul[0] - left;
780 right = cdata->_text_lr[0] + right;
781 bottom = cdata->_text_lr[1] - bottom;
782 top = cdata->_text_ul[1] + top;
783 }
784
785 CPT(RenderAttrib) thick = RenderModeAttrib::make(RenderModeAttrib::M_unchanged, cdata->_frame_width);
786 CPT(RenderState) state = RenderState::make(thick);
787
788 PT(GeomVertexData) vdata = new GeomVertexData
789 ("text", GeomVertexFormat::get_v3(), cdata->_usage_hint);
790 vdata->unclean_set_num_rows(4);
791 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
792
793 vertex.set_data3(left, 0.0f, top);
794 vertex.set_data3(left, 0.0f, bottom);
795 vertex.set_data3(right, 0.0f, bottom);
796 vertex.set_data3(right, 0.0f, top);
797
798 PT(GeomLinestrips) frame = new GeomLinestrips(cdata->_usage_hint);
799 frame->add_consecutive_vertices(0, 4);
800 frame->add_vertex(0);
801 frame->close_primitive();
802
803 PT(Geom) geom = new Geom(vdata);
804 geom->add_primitive(frame);
805 frame_node->add_geom(geom, state);
806
807 if (cdata->_flags & F_frame_corners) {
808 PT(GeomPoints) corners = new GeomPoints(cdata->_usage_hint);
809 corners->add_consecutive_vertices(0, 4);
810 PT(Geom) geom2 = new Geom(vdata);
811 geom2->add_primitive(corners);
812 frame_node->add_geom(geom2, state);
813 }
814
815 return frame_node;
816}
