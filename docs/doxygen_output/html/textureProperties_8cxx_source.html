245void TextureProperties::
246fully_define() {
247 if (!_got_num_channels || _force_format) {
248 switch (_format) {
249 case EggTexture::F_rgba:
250 case EggTexture::F_rgbm:
251 case EggTexture::F_rgba12:
252 case EggTexture::F_rgba8:
253 case EggTexture::F_rgba4:
254 case EggTexture::F_rgba5:
255 case EggTexture::F_srgb_alpha:
256 _num_channels = 4;
257 break;
258
259 case EggTexture::F_unspecified:
260 case EggTexture::F_rgb:
261 case EggTexture::F_rgb12:
262 case EggTexture::F_rgb8:
263 case EggTexture::F_rgb5:
264 case EggTexture::F_rgb332:
265 case EggTexture::F_srgb:
266 _num_channels = 3;
267 break;
268
269 case EggTexture::F_luminance_alpha:
270 case EggTexture::F_luminance_alphamask:
271 case EggTexture::F_sluminance_alpha:
272 _num_channels = 2;
273 break;
274
275 case EggTexture::F_red:
276 case EggTexture::F_green:
277 case EggTexture::F_blue:
278 case EggTexture::F_alpha:
279 case EggTexture::F_luminance:
280 case EggTexture::F_sluminance:
281 _num_channels = 1;
282 break;
283 }
284 _got_num_channels = true;
285 }
286
287 _effective_num_channels = _num_channels;
288
289 // Respect the _generic_format flag. If this is set, it means the user has
290 // indicated that we should strip off any bitcount-specific formats and
291 // replace them with the more generic equivalents.
292 if (_generic_format) {
293 switch (_format) {
294 case EggTexture::F_unspecified:
295 case EggTexture::F_rgba:
296 case EggTexture::F_rgbm:
297 case EggTexture::F_rgb:
298 case EggTexture::F_red:
299 case EggTexture::F_green:
300 case EggTexture::F_blue:
301 case EggTexture::F_alpha:
302 case EggTexture::F_luminance:
303 case EggTexture::F_luminance_alpha:
304 case EggTexture::F_luminance_alphamask:
305 case EggTexture::F_srgb:
306 case EggTexture::F_srgb_alpha:
307 case EggTexture::F_sluminance:
308 case EggTexture::F_sluminance_alpha:
309 break;
310
311 case EggTexture::F_rgba12:
312 case EggTexture::F_rgba8:
313 case EggTexture::F_rgba4:
314 case EggTexture::F_rgba5:
315 _format = EggTexture::F_rgba;
316 break;
317
318 case EggTexture::F_rgb12:
319 case EggTexture::F_rgb8:
320 case EggTexture::F_rgb5:
321 case EggTexture::F_rgb332:
322 _format = EggTexture::F_rgb;
323 break;
324 }
325 }
326
327 // Make sure the format reflects the number of channels, although we accept
328 // a format that ignores an alpha channel.
329 if (!_force_format && !_keep_format) {
330 switch (_num_channels) {
331 case 1:
332 switch (_format) {
333 case EggTexture::F_red:
334 case EggTexture::F_green:
335 case EggTexture::F_blue:
336 case EggTexture::F_alpha:
337 case EggTexture::F_luminance:
338 case EggTexture::F_sluminance:
339 break;
340
341 // These formats suggest an alpha channel; they are quietly replaced
342 // with non-alpha equivalents.
343 case EggTexture::F_luminance_alpha:
344 case EggTexture::F_luminance_alphamask:
345 _format = EggTexture::F_luminance;
346 break;
347
348 case EggTexture::F_sluminance_alpha:
349 _format = EggTexture::F_sluminance;
350 break;
351
352 default:
353 if (_srgb) {
354 _format = EggTexture::F_sluminance;
355 } else {
356 _format = EggTexture::F_luminance;
357 }
358 }
359 break;
360
361 case 2:
362 switch (_format) {
363 case EggTexture::F_luminance_alpha:
364 case EggTexture::F_luminance_alphamask:
365 case EggTexture::F_sluminance_alpha:
366 break;
367
368 // These formats implicitly reduce the number of channels to 1.
369 case EggTexture::F_red:
370 case EggTexture::F_green:
371 case EggTexture::F_blue:
372 case EggTexture::F_alpha:
373 case EggTexture::F_luminance:
374 case EggTexture::F_sluminance:
375 break;
376
377 default:
378 if (_srgb) {
379 _format = EggTexture::F_sluminance_alpha;
380 } else {
381 _format = EggTexture::F_luminance_alpha;
382 }
383 }
384 break;
385
386 case 3:
387 switch (_format) {
388 case EggTexture::F_rgb:
389 case EggTexture::F_rgb12:
390 case EggTexture::F_rgb8:
391 case EggTexture::F_rgb5:
392 case EggTexture::F_rgb332:
393 case EggTexture::F_srgb:
394 break;
395
396 // These formats suggest an alpha channel; they are quietly replaced
397 // with non-alpha equivalents.
398 case EggTexture::F_rgba8:
399 _format = EggTexture::F_rgb8;
400 break;
401
402 case EggTexture::F_rgba5:
403 case EggTexture::F_rgba4:
404 _format = EggTexture::F_rgb5;
405 break;
406
407 // These formats implicitly reduce the number of channels to 1.
408 case EggTexture::F_red:
409 case EggTexture::F_green:
410 case EggTexture::F_blue:
411 case EggTexture::F_alpha:
412 case EggTexture::F_luminance:
413 case EggTexture::F_sluminance:
414 break;
415
416 default:
417 if (_srgb) {
418 _format = EggTexture::F_srgb;
419 } else {
420 _format = EggTexture::F_rgb;
421 }
422 }
423 break;
424
425 case 4:
426 switch (_format) {
427 case EggTexture::F_rgba:
428 case EggTexture::F_rgbm:
429 case EggTexture::F_rgba12:
430 case EggTexture::F_rgba8:
431 case EggTexture::F_rgba4:
432 case EggTexture::F_rgba5:
433 case EggTexture::F_srgb_alpha:
434 break;
435
436 // These formats implicitly reduce the number of channels to 3.
437 case EggTexture::F_rgb:
438 case EggTexture::F_rgb12:
439 case EggTexture::F_rgb8:
440 case EggTexture::F_rgb5:
441 case EggTexture::F_rgb332:
442 case EggTexture::F_srgb:
443 _effective_num_channels = 3;
444 break;
445
446 // These formats implicitly reduce the number of channels to 2.
447 case EggTexture::F_luminance_alpha:
448 case EggTexture::F_luminance_alphamask:
449 case EggTexture::F_sluminance_alpha:
450 _effective_num_channels = 2;
451 break;
452
453 // These formats implicitly reduce the number of channels to 1.
454 case EggTexture::F_red:
455 case EggTexture::F_green:
456 case EggTexture::F_blue:
457 case EggTexture::F_alpha:
458 case EggTexture::F_luminance:
459 case EggTexture::F_sluminance:
460 _effective_num_channels = 1;
461 break;
462
463 default:
464 if (_srgb) {
465 _format = EggTexture::F_srgb_alpha;
466 } else {
467 _format = EggTexture::F_rgba;
468 }
469 }
470 }
471 }
472
473 // Respect the _srgb flag. If this is set, it means the texture is in sRGB
474 // color space and the format should be changed to reflect that.
475 if (_srgb) {
476 switch (_num_channels) {
477 case 1:
478 // Don't respect sRGB for textures using the F_alpha format, which
479 // indicates that the image represents an alpha channel, not a color
480 // channel.
481 if (_format != EggTexture::F_alpha) {
482 _format = EggTexture::F_sluminance;
483 }
484 break;
485 case 2:
486 _format = EggTexture::F_sluminance_alpha;
487 break;
488 case 3:
489 _format = EggTexture::F_srgb;
490 break;
491 case 4:
492 default:
493 _format = EggTexture::F_srgb_alpha;
494 break;
495 }
496 }
497
498 switch (_minfilter) {
499 case EggTexture::FT_unspecified:
500 _minfilter = EggTexture::FT_linear;
501 break;
502
503 default:
504 break;
505 }
506
507 switch (_magfilter) {
508 case EggTexture::FT_unspecified:
509 case EggTexture::FT_nearest_mipmap_nearest:
510 case EggTexture::FT_linear_mipmap_nearest:
511 case EggTexture::FT_nearest_mipmap_linear:
512 case EggTexture::FT_linear_mipmap_linear:
513 _magfilter = EggTexture::FT_linear;
514 break;
515
516 default:
517 break;
518 }
519
520 if (_color_type == nullptr) {
521 _color_type = pal->_color_type;
522 _alpha_type = pal->_alpha_type;
523 }
524}

906void TextureProperties::
907fillin(DatagramIterator &scan, BamReader *manager) {
908 TypedWritable::fillin(scan, manager);
909 _got_num_channels = scan.get_bool();
910 _num_channels = scan.get_int32();
911 _effective_num_channels = _num_channels;
912 if (Palettizer::_read_pi_version >= 9) {
913 _effective_num_channels = scan.get_int32();
914 }
915 _format = (EggTexture::Format)scan.get_int32();
916 _force_format = scan.get_bool();
917 _generic_format = false;
918 if (Palettizer::_read_pi_version >= 9) {
919 _generic_format = scan.get_bool();
920 }
921 _keep_format = false;
922 if (Palettizer::_read_pi_version >= 13) {
923 _keep_format = scan.get_bool();
924 }
925 _minfilter = (EggTexture::FilterType)scan.get_int32();
926 _magfilter = (EggTexture::FilterType)scan.get_int32();
927 if (Palettizer::_read_pi_version >= 18) {
928 _quality_level = (EggTexture::QualityLevel)scan.get_int32();
929 }
930 _anisotropic_degree = scan.get_int32();
931
932 if (Palettizer::_read_pi_version >= 21) {
933 _srgb = scan.get_bool();
934 }
935
936 manager->read_pointer(scan); // _color_type
937 manager->read_pointer(scan); // _alpha_type
938}
