123void BillboardEffect::
124cull_callback(CullTraverser *trav, CullTraverserData &data,
125 CPT(TransformState) &node_transform,
126 CPT(RenderState) &) const {
127 CPT(TransformState) modelview_transform = data.get_modelview_transform(trav);
128 if (modelview_transform->is_singular()) {
129 // If we're under a singular transform, never mind.
130 return;
131 }
132
133 // Since the "modelview" transform from the cull traverser already includes
134 // the inverse camera transform, the camera transform is identity.
135 CPT(TransformState) camera_transform = TransformState::make_identity();
136
137 // But if we're rotating to face something other than the camera, we have to
138 // compute the "camera" transform to compensate for that.
139 if (!_look_at.is_empty()) {
140 camera_transform = trav->get_camera_transform()->invert_compose(_look_at.get_net_transform());
141 }
142
143 if (data._instances == nullptr) {
144 compute_billboard(node_transform, modelview_transform, camera_transform);
145 }
146 else {
147 // Compute the billboard effect for every instance individually.
148 InstanceList *instances = new InstanceList(*data._instances);
149 data._instances = instances;
150
151 for (InstanceList::Instance &instance : *instances) {
152 CPT(TransformState) inst_node_transform = node_transform;
153 CPT(TransformState) inst_modelview_transform = modelview_transform->compose(instance.get_transform());
154 compute_billboard(inst_node_transform, inst_modelview_transform, camera_transform);
155
156 instance.set_transform(instance.get_transform()->compose(inst_node_transform));
157 }
158
159 // We've already applied this onto the instances.
160 node_transform = TransformState::make_identity();
161 }
162}

252void BillboardEffect::
253compute_billboard(CPT(TransformState) &node_transform,
254 const TransformState *net_transform,
255 const TransformState *camera_transform) const {
256 // First, extract out just the translation component of the node's local
257 // transform. This gets applied to the net transform, to compute the look-
258 // at direction properly.
259 CPT(TransformState) translate = TransformState::make_pos(node_transform->get_pos());
260
261 // And then the translation gets removed from the node, but we keep its
262 // rotation etc., which gets applied after the billboard operation.
263 node_transform = node_transform->set_pos(LPoint3(0.0f, 0.0f, 0.0f));
264
265 CPT(TransformState) rel_transform =
266 net_transform->compose(translate)->invert_compose(camera_transform);
267 if (!rel_transform->has_mat()) {
268 // Never mind.
269 return;
270 }
271
272 const LMatrix4 &rel_mat = rel_transform->get_mat();
273
274 // Determine the look_at point in the camera space.
275 LVector3 camera_pos, up;
276
277 // If this is an eye-relative Billboard, then (a) the up vector is relative
278 // to the camera, not to the world, and (b) the look direction is towards
279 // the plane that contains the camera, perpendicular to the forward
280 // direction, not directly to the camera.
281
282 if (_eye_relative) {
283 up = _up_vector * rel_mat;
284 camera_pos = LVector3::forward() * rel_mat;
285
286 } else {
287 up = _up_vector;
288 camera_pos = -(_look_at_point * rel_mat);
289 }
290
291 // Now determine the rotation matrix for the Billboard.
292 LMatrix4 rotate;
293 if (_axial_rotate) {
294 heads_up(rotate, camera_pos, up);
295 } else {
296 look_at(rotate, camera_pos, up);
297 }
298
299 // Also slide the billboard geometry towards the camera according to the
300 // offset factor.
301 if (_offset != 0.0f || _fixed_depth) {
302 LVector3 translate(rel_mat(3, 0), rel_mat(3, 1), rel_mat(3, 2));
303 LPoint3 pos;
304 if (_fixed_depth) {
305 pos = translate / rel_mat(3, 3);
306 } else {
307 pos.fill(0.0f);
308 }
309 translate.normalize();
310 translate *= _offset;
311 rotate.set_row(3, pos + translate);
312 }
313
314 node_transform = translate->compose(TransformState::make_mat(rotate))->compose(node_transform);
315}
