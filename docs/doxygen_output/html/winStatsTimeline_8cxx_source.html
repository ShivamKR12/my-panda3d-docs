129void WinStatsTimeline::
130draw_bar(int row, int from_x, int to_x, int collector_index,
131 const std::string &collector_name) {
132
133 int top = row_to_pixel(row);
134 int bottom = row_to_pixel(row + 1);
135
136 bool is_highlighted = row == _highlighted_row && _highlighted_x >= from_x && _highlighted_x < to_x;
137 HBRUSH brush = get_collector_brush(collector_index, is_highlighted);
138
139 if (to_x < from_x + 2) {
140 // It's just a tiny sliver. This is a more reliable way to draw it.
141 RECT rect = {from_x, top + 1, from_x + 1, bottom - 1};
142 FillRect(_bitmap_dc, &rect, brush);
143
144 //if (to_x <= from_x + 2) {
145 // // Draw an arrow pointing to it, if it's so small.
146 // POINT vertices[] = {{to_x, bottom}, {to_x - _pixel_scale, bottom + _pixel_scale * 2}, {to_x + _pixel_scale, bottom + _pixel_scale * 2}};
147 // Polygon(_bitmap_dc, vertices, 3);
148 //}
149 }
150 else {
151 SelectObject(_bitmap_dc, brush);
152 RoundRect(_bitmap_dc,
153 std::max(from_x, -_pixel_scale - 1),
154 top,
155 std::min(std::max(to_x, from_x + 1), get_xsize() + _pixel_scale),
156 bottom,
157 _pixel_scale,
158 _pixel_scale);
159
160 if ((to_x - from_x) >= _pixel_scale * 4) {
161 // Only bother drawing the text if we've got some space to draw on.
162 // Choose a suitable foreground color.
163 SetTextColor(_bitmap_dc, get_collector_text_color(collector_index, is_highlighted));
164
165 // Make sure that the text doesn't run off the chart.
166 SIZE size;
167 GetTextExtentPoint32(_bitmap_dc, collector_name.data(), collector_name.size(), &size);
168 int center = (from_x + to_x) / 2;
169 int left = std::max(from_x, 0) + _pixel_scale / 2;
170 int right = std::min(to_x, get_xsize()) - _pixel_scale / 2;
171
172 if (size.cx >= right - left) {
173 if (right - left < _pixel_scale * 6) {
174 // It's a really tiny space. Draw a single letter.
175 RECT rect = {left, top, right, bottom};
176 DrawText(_bitmap_dc, collector_name.data(), 1,
177 &rect, DT_CENTER | DT_SINGLELINE | DT_VCENTER);
178 } else {
179 // It's going to be tricky to fit it, let Windows figure it out via
180 // the more expensive DrawText call.
181 RECT rect = {left, top, right, bottom};
182 DrawText(_bitmap_dc, collector_name.data(), collector_name.size(),
183 &rect, DT_CENTER | DT_END_ELLIPSIS | DT_SINGLELINE | DT_VCENTER);
184 }
185 }
186 else {
187 int text_top = top + (bottom - top - size.cy) / 2;
188 if (center - size.cx / 2 < 0) {
189 // Put it against the left-most edge.
190 TextOut(_bitmap_dc, _pixel_scale, text_top,
191 collector_name.data(), collector_name.length());
192 }
193 else if (center + size.cx / 2 >= get_xsize()) {
194 // Put it against the right-most edge.
195 TextOut(_bitmap_dc, get_xsize() - _pixel_scale - size.cx, text_top,
196 collector_name.data(), collector_name.length());
197 }
198 else {
199 // It fits just fine, center it.
200 TextOut(_bitmap_dc, center - size.cx / 2, text_top,
201 collector_name.data(), collector_name.length());
202 }
203 }
204 }
205 }

289LONG WinStatsTimeline::
290graph_window_proc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam) {
291 switch (msg) {
292 case WM_LBUTTONDOWN:
293 if (_potential_drag_mode == DM_none) {
294 set_drag_mode(DM_pan);
295 int16_t x = LOWORD(lparam);
296 _drag_start_x = x;
297 _scroll_speed = 0.0;
298 _zoom_center = pixel_to_timestamp(x);
299 SetCapture(_graph_window);
300 return 0;
301 }
302 break;
303
304 case WM_MOUSEMOVE:
305 // Make sure we can accept keyboard events, except if we're inactive.
306 if (GetActiveWindow() == _window) {
307 SetFocus(hwnd);
308 }
309
310 if (_drag_mode == DM_none && _potential_drag_mode == DM_none) {
311 // When the mouse is over a color bar, highlight it.
312 int x = LOWORD(lparam);
313 int y = HIWORD(lparam);
314 double time = pixel_to_timestamp(x);
315
316 int row = pixel_to_row(y);
317
318 if (row != _highlighted_row) {
319 clear_graph_tooltip();
320 }
321 else if (_highlighted_row >= 0) {
322 // Is the mouse on the same bar? If not, clear the tooltip.
323 ColorBar bar;
324 if (find_bar(row, x, bar)) {
325 double prev_time = pixel_to_timestamp(_highlighted_x);
326 if (prev_time < bar._start || prev_time > bar._end) {
327 clear_graph_tooltip();
328 }
329 } else {
330 clear_graph_tooltip();
331 }
332 }
333
334 std::swap(_highlighted_x, x);
335 std::swap(_highlighted_row, row);
336
337 if (row >= 0) {
338 PStatTimeline::force_redraw(row, x, x);
339 }
340 PStatTimeline::force_redraw(_highlighted_row, _highlighted_x, _highlighted_x);
341
342 if ((_keys_held & (F_w | F_s)) != 0) {
343 // Update the zoom center if we move the mouse while zooming with the
344 // keyboard.
345 _zoom_center = time;
346 }
347
348 // Now we want to get a WM_MOUSELEAVE when the mouse leaves the graph
349 // window.
350 TRACKMOUSEEVENT tme = {
351 sizeof(TRACKMOUSEEVENT),
352 TME_LEAVE,
353 _graph_window,
354 0
355 };
356 TrackMouseEvent(&tme);
357 }
358 else {
359 // If the mouse is in some drag mode, stop highlighting.
360 if (_highlighted_row != -1) {
361 int row = _highlighted_row;
362 _highlighted_row = -1;
363 PStatTimeline::force_redraw(row, _highlighted_x, _highlighted_x);
364 clear_graph_tooltip();
365 }
366 }
367
368 if (_drag_mode == DM_pan) {
369 int16_t x = LOWORD(lparam);
370 int delta = _drag_start_x - x;
371 set_horizontal_scroll(get_horizontal_scroll() + pixel_to_height(delta));
372 _drag_start_x = x;
373 return 0;
374 }
375 break;
376
377 case WM_MOUSELEAVE:
378 // When the mouse leaves the graph, stop highlighting.
379 if (_highlighted_row != -1) {
380 int row = _highlighted_row;
381 _highlighted_row = -1;
382 PStatTimeline::force_redraw(row, _highlighted_x, _highlighted_x);
383 clear_graph_tooltip();
384 }
385 SetFocus(nullptr);
386 break;
387
388 case WM_LBUTTONUP:
389 if (_drag_mode == DM_pan) {
390 set_drag_mode(DM_none);
391 ReleaseCapture();
392 return 0;
393 }
394 break;
395
396 case WM_LBUTTONDBLCLK:
397 {
398 // Double-clicking on a color bar in the graph will zoom the graph into
399 // that collector.
400 int16_t x = LOWORD(lparam);
401 int16_t y = HIWORD(lparam);
402 int row = pixel_to_row(y);
403 ColorBar bar;
404 if (find_bar(row, x, bar)) {
405 double width = bar._end - bar._start;
406 zoom_to(width * 1.5, pixel_to_timestamp(x));
407 scroll_to(bar._start - width / 4.0);
408 } else {
409 // Double-clicking the white area zooms out.
410 _zoom_speed -= 100.0;
411 }
412 start_animation();
413 return 0;
414 }
415 break;
416
417 case WM_CONTEXTMENU:
418 {
419 // Right-clicking a color bar brings up a context menu.
420 POINT point;
421 if (GetCursorPos(&point)) {
422 POINT graph_point = point;
423 if (ScreenToClient(_graph_window, &graph_point)) {
424 int row = pixel_to_row(graph_point.y);
425 ColorBar bar;
426 if (find_bar(row, graph_point.x, bar)) {
427 _popup_bar = bar;
428
429 HMENU popup = CreatePopupMenu();
430
431 std::string label = get_bar_tooltip(row, graph_point.x);
432 if (!label.empty()) {
433 AppendMenu(popup, MF_STRING | MF_DISABLED, 0, label.c_str());
434 }
435 AppendMenu(popup, MF_STRING, 101, "Zoom To");
436 AppendMenu(popup, MF_STRING, 102, "Open Strip Chart");
437 AppendMenu(popup, MF_STRING, 103, "Open Flame Graph");
438 AppendMenu(popup, MF_STRING, 104, "Open Piano Roll");
439 AppendMenu(popup, MF_STRING | MF_SEPARATOR, 0, nullptr);
440 AppendMenu(popup, MF_STRING, 105, "Change Color...");
441 AppendMenu(popup, MF_STRING, 106, "Reset Color");
442 TrackPopupMenu(popup, TPM_LEFTBUTTON, point.x, point.y, 0, _graph_window, nullptr);
443 }
444 }
445 }
446 return 0;
447 }
448 break;
449
450 case WM_COMMAND:
451 switch (LOWORD(wparam)) {
452 case 101:
453 {
454 double width = _popup_bar._end - _popup_bar._start;
455 zoom_to(width * 1.5, (_popup_bar._end + _popup_bar._start) / 2.0);
456 scroll_to(_popup_bar._start - width / 4.0);
457 start_animation();
458 }
459 return 0;
460
461 case 102:
462 WinStatsGraph::_monitor->open_strip_chart(_popup_bar._thread_index, _popup_bar._collector_index, false);
463 return 0;
464
465 case 103:
466 WinStatsGraph::_monitor->open_flame_graph(_popup_bar._thread_index, _popup_bar._collector_index, _popup_bar._frame_number);
467 return 0;
468
469 case 104:
470 WinStatsGraph::_monitor->open_piano_roll(_popup_bar._thread_index);
471 return 0;
472
473 case 105:
474 WinStatsGraph::_monitor->choose_collector_color(_popup_bar._collector_index);
475 return 0;
476
477 case 106:
478 WinStatsGraph::_monitor->reset_collector_color(_popup_bar._collector_index);
479 return 0;
480 }
481 break;
482
483 case WM_MOUSEWHEEL:
484 {
485 if (GET_KEYSTATE_WPARAM(wparam) & MK_CONTROL) {
486 // Zoom in/out around the cursor position.
487 POINT point;
488 if (GetCursorPos(&point) && ScreenToClient(_graph_window, &point)) {
489 int delta = GET_WHEEL_DELTA_WPARAM(wparam);
490 zoom_by(delta / 120.0, pixel_to_timestamp(point.x));
491 start_animation();
492 }
493 } else {
494 int delta = GET_WHEEL_DELTA_WPARAM(wparam);
495 delta = (delta * _pixel_scale * 5) / 120;
496 int new_scroll = _scroll - delta;
497 if (_threads.empty()) {
498 new_scroll = 0;
499 } else {
500 new_scroll = (std::min)(new_scroll, get_num_rows() * _pixel_scale * 5 + _pixel_scale * 2 - get_ysize());
501 new_scroll = (std::max)(new_scroll, 0);
502 }
503 delta = new_scroll - _scroll;
504 if (delta != 0) {
505 _scroll = new_scroll;
506 _threads_changed = true;
507 PStatTimeline::force_redraw();
508 }
509 }
510 return 0;
511 }
512 break;
513
514 case WM_MOUSEHWHEEL:
515 {
516 int delta = GET_WHEEL_DELTA_WPARAM(wparam);
517 _scroll_speed += delta / 12.0;
518 start_animation();
519 return 0;
520 }
521 break;
522
523 case WM_KEYDOWN:
524 case WM_KEYUP:
525 {
526 int flag = 0;
527 int vsc = (lparam & 0xff0000) >> 16;
528 if ((lparam & 0x1000000) == 0) {
529 // Accept WASD based on their position rather than their mapping
530 switch (vsc) {
531 case 17:
532 flag = F_w;
533 break;
534 case 30:
535 flag = F_a;
536 break;
537 case 31:
538 flag = F_s;
539 break;
540 case 32:
541 flag = F_d;
542 break;
543 }
544 }
545 if (flag == 0) {
546 switch (wparam) {
547 case VK_LEFT:
548 flag = F_left;
549 break;
550 case VK_RIGHT:
551 flag = F_right;
552 break;
553 case 'W':
554 flag = F_w;
555 break;
556 case 'A':
557 flag = F_a;
558 break;
559 case 'S':
560 flag = F_s;
561 break;
562 case 'D':
563 flag = F_d;
564 break;
565 }
566 }
567 if (flag != 0) {
568 if (msg == WM_KEYDOWN) {
569 if (flag & (F_w | F_s)) {
570 POINT point;
571 if (GetCursorPos(&point) && ScreenToClient(_graph_window, &point)) {
572 _zoom_center = pixel_to_timestamp(point.x);
573 } else {
574 _zoom_center = get_horizontal_scroll() + get_horizontal_scale() / 2.0;
575 }
576 }
577 if (_keys_held == 0) {
578 start_animation();
579 }
580 _keys_held |= flag;
581 }
582 else if (_keys_held != 0) {
583 _keys_held &= ~flag;
584 }
585 }
586 }
587 break;
588
589 case WM_KILLFOCUS:
590 _keys_held = 0;
591 break;
592
593 default:
594 break;
595 }
596
597 return WinStatsGraph::graph_window_proc(hwnd, msg, wparam, lparam);
598}
