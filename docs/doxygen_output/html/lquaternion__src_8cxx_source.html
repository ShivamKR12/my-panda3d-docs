33FLOATNAME(LQuaternion) FLOATNAME(LQuaternion)::
34__pow__(FLOATTYPE power) const {
35 if (IS_NEARLY_ZERO(power)) {
36 return FLOATNAME(LQuaternion)(1, 0, 0, 0);
37 }
38
39 FLOATTYPE l = length();
40 FLOATTYPE norm = _v(0) / l;
41 if (IS_NEARLY_EQUAL(cabs(norm), (FLOATTYPE)1)) {
42 return FLOATNAME(LQuaternion)(cpow(_v(0), power), 0, 0, 0);
43 }
44
45 FLOATTYPE angle = acos(norm);
46 FLOATTYPE angle2 = angle * power;
47 FLOATTYPE mag = cpow(l, power - 1);
48 FLOATTYPE mult = mag * (sin(angle2) / sin(angle));
49 return FLOATNAME(LQuaternion)(cos(angle2) * mag * l, _v(1) * mult, _v(2) * mult, _v(3) * mult);
50}

56extract_to_matrix(FLOATNAME(LMatrix3) &m) const {
57 FLOATTYPE N = this->dot(*this);
58 FLOATTYPE s = (N == 0.0f) ? 0.0f : (2.0f / N);

60

63 xx = _v(1) * xs; xy = _v(1) * ys; xz = _v(1) * zs;
64 yy = _v(2) * ys; yz = _v(2) * zs; zz = _v(3) * zs;
65
66 m.set((1.0f - (yy + zz)), (xy + wz), (xz - wy),
67 (xy - wz), (1.0f - (xx + zz)), (yz + wx),

95void FLOATNAME(LQuaternion)::
96set_hpr(const FLOATNAME(LVecBase3) &hpr, CoordinateSystem cs) {
97 FLOATNAME(LQuaternion) quat_h, quat_p, quat_r;
98
99 FLOATNAME(LVector3) v;
100 FLOATTYPE a, s, c;
101
102 v = FLOATNAME(LVector3)::up(cs);
103 a = deg_2_rad(hpr[0] * 0.5f);
104 csincos(a, &s, &c);
105 quat_h.set(c, v[0] * s, v[1] * s, v[2] * s);
106 v = FLOATNAME(LVector3)::right(cs);
107 a = deg_2_rad(hpr[1] * 0.5f);
108 csincos(a, &s, &c);
109 quat_p.set(c, v[0] * s, v[1] * s, v[2] * s);
110 v = FLOATNAME(LVector3)::forward(cs);
111 a = deg_2_rad(hpr[2] * 0.5f);
112 csincos(a, &s, &c);
113 quat_r.set(c, v[0] * s, v[1] * s, v[2] * s);
114
115 if (is_right_handed(cs)) {
116 (*this) = quat_r * quat_p * quat_h;
117 } else {
118 (*this) = invert(quat_h * quat_p * quat_r);
119 }
120
121#ifndef NDEBUG
122 if (paranoid_hpr_quat) {
123 FLOATNAME(LMatrix3) mat;
124 compose_matrix(mat, FLOATNAME(LVecBase3)(1.0f, 1.0f, 1.0f), hpr, cs);
125 FLOATNAME(LQuaternion) compare;
126 compare.set_from_matrix(mat);
127 if (!compare.almost_equal(*this) && !compare.almost_equal(-(*this))) {
128 linmath_cat.warning()
129 << "hpr-to-quat of " << hpr << " computed " << *this
130 << " instead of " << compare << "\n";
131 (*this) = compare;
132 }
133 }
134#endif // NDEBUG

140FLOATNAME(LVecBase3) FLOATNAME(LQuaternion)::
141get_hpr(CoordinateSystem cs) const {
142 if (cs == CS_default) {
143 cs = get_default_coordinate_system();
144 }
145
146 FLOATNAME(LVecBase3) hpr;
147
148 if (cs == CS_zup_right) {
149 FLOATTYPE N =
150 (_v(0) * _v(0)) +
151 (_v(1) * _v(1)) +
152 (_v(2) * _v(2)) +
153 (_v(3) * _v(3));
154 FLOATTYPE s = (N == 0.0f) ? 0.0f : (2.0f / N);
155 FLOATTYPE xs, ys, zs, wx, wy, wz, xx, xy, xz, yy, yz, zz, c1, c2, c3, c4;
156 FLOATTYPE cr, sr, cp, sp, ch, sh;
157
158 xs = _v(1) * s; ys = _v(2) * s; zs = _v(3) * s;
159 wx = _v(0) * xs; wy = _v(0) * ys; wz = _v(0) * zs;
160 xx = _v(1) * xs; xy = _v(1) * ys; xz = _v(1) * zs;
161 yy = _v(2) * ys; yz = _v(2) * zs; zz = _v(3) * zs;
162 c1 = xz - wy;
163 c2 = 1.0f - (xx + yy);
164 c3 = 1.0f - (yy + zz);
165 c4 = xy + wz;
166
167 if (c1 == 0.0f) { // (roll = 0 or 180) or (pitch = +/- 90)
168 if (c2 >= 0.0f) {
169 hpr[2] = 0.0f;
170 ch = c3;
171 sh = c4;
172 cp = c2;
173 } else {
174 hpr[2] = 180.0f;
175 ch = -c3;
176 sh = -c4;
177 cp = -c2;
178 }
179 } else {
180 // this should work all the time, but the above saves some trig
181 // operations
182 FLOATTYPE roll = catan2(-c1, c2);
183 csincos(roll, &sr, &cr);
184 hpr[2] = rad_2_deg(roll);
185 ch = (cr * c3) + (sr * (xz + wy));
186 sh = (cr * c4) + (sr * (yz - wx));
187 cp = (cr * c2) - (sr * c1);
188 }
189 sp = yz + wx;
190 hpr[0] = rad_2_deg(catan2(sh, ch));
191 hpr[1] = rad_2_deg(catan2(sp, cp));
192
193 } else {
194 // The code above implements quat-to-hpr for CS_zup_right only. For other
195 // coordinate systems, someone is welcome to extend the implementation;
196 // I'm going to choose the lazy path till then.
197 FLOATNAME(LMatrix3) mat;
198 extract_to_matrix(mat);
199 FLOATNAME(LVecBase3) scale;
200 decompose_matrix(mat, scale, hpr, cs);
201 return hpr;
202 }
203
204
205#ifndef NDEBUG
206 if (paranoid_hpr_quat) {
207 FLOATNAME(LMatrix3) mat;
208 extract_to_matrix(mat);
209 FLOATNAME(LVecBase3) scale, compare_hpr;
210 decompose_matrix(mat, scale, compare_hpr, cs);
211 if (!compare_hpr.almost_equal(hpr)) {
212 linmath_cat.warning()
213 << "quat-to-hpr of " << *this << " computed " << hpr << " instead of "
214 << compare_hpr << "\n";
215 hpr = compare_hpr;
216 }
217 }
218#endif // NDEBUG
219
220 return hpr;
221}

228void FLOATNAME(LQuaternion)::
229set_from_matrix(const FLOATNAME(LMatrix3) &m) {
230 FLOATTYPE m00, m01, m02, m10, m11, m12, m20, m21, m22;
231
232 m00 = m(0, 0);
233 m10 = m(1, 0);
234 m20 = m(2, 0);
235 m01 = m(0, 1);
236 m11 = m(1, 1);
237 m21 = m(2, 1);
238 m02 = m(0, 2);
239 m12 = m(1, 2);
240 m22 = m(2, 2);
241
242 FLOATTYPE trace = m00 + m11 + m22;
243
244 if (trace > 0.0f) {
245 // The easy case.
246 FLOATTYPE S = csqrt(trace + 1.0f);
247 _v(0) = S * 0.5f;
248 S = 0.5f / S;
249 _v(1) = (m12 - m21) * S;
250 _v(2) = (m20 - m02) * S;
251 _v(3) = (m01 - m10) * S;
252
253 } else {
254 // The harder case. First, figure out which column to take as root. This
255 // will be the column with the largest value.
256
257 // It is tempting to try to compare the absolute values of the diagonal
258 // values in the code below, instead of their normal, signed values.
259 // Don't do it. We are actually maximizing the value of S, which must
260 // always be positive, and is therefore based on the diagonal whose actual
261 // value--not absolute value--is greater than those of the other two.
262
263 // We already know that m00 + m11 + m22 <= 0 (because we are here in the
264 // harder case).
265
266 if (m00 > m11 && m00 > m22) {
267 // m00 is larger than m11 and m22.
268 FLOATTYPE S = 1.0f + m00 - (m11 + m22);
269 nassertv(S > 0.0f);
270 S = csqrt(S);
271 _v(1) = S * 0.5f;
272 S = 0.5f / S;
273 _v(2) = (m01 + m10) * S;
274 _v(3) = (m02 + m20) * S;
275 _v(0) = (m12 - m21) * S;
276
277 } else if (m11 > m22) {
278 // m11 is larger than m00 and m22.
279 FLOATTYPE S = 1.0f + m11 - (m22 + m00);
280 nassertv(S > 0.0f);
281 S = csqrt(S);
282 _v(2) = S * 0.5f;
283 S = 0.5f / S;
284 _v(3) = (m12 + m21) * S;
285 _v(1) = (m10 + m01) * S;
286 _v(0) = (m20 - m02) * S;
287
288 } else {
289 // m22 is larger than m00 and m11.
290 FLOATTYPE S = 1.0f + m22 - (m00 + m11);
291 nassertv(S > 0.0f);
292 S = csqrt(S);
293 _v(3) = S * 0.5f;
294 S = 0.5f / S;
295 _v(1) = (m20 + m02) * S;
296 _v(2) = (m21 + m12) * S;
297 _v(0) = (m01 - m10) * S;
298 }
299 }
