152void Extension<PandaNode>::
153clear_python_tag(PyObject *key) {
154 if (_this->_python_tag_data == nullptr) {
155 return;
156 }
157
158 PyObject *dict = do_get_python_tags();
159 Py_BEGIN_CRITICAL_SECTION(dict);
160 if (PyDict_Contains(dict, key)) {
161 PyDict_DelItem(dict, key);
162 }
163
164 if (PyDict_Size(dict) == 0 && Py_REFCNT(dict) == 1) {
165 // This was the last tag, and do_get_python_tags() made sure we have a
166 // unique reference to the tags, so clear the tag object.
167 _this->_python_tag_data.clear();
168 }
169 Py_END_CRITICAL_SECTION();
170}

188int Extension<PandaNode>::
189__traverse__(visitproc visit, void *arg) {
190 // To fully implement cycle breaking, we also have to recurse into all of
191 // the node's children. However, this seems like it would be potentially
192 // quite expensive, so I'd rather not do it unless we had some optimization
193 // that would allow us to quickly find out whether there are children with
194 // Python tags.
195 if (_this->_python_tag_data != nullptr) {
196 Py_VISIT(((PythonTagDataImpl *)_this->_python_tag_data.p())->_dict);
197 }
198 return 0;
