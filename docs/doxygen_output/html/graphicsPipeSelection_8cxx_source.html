150PT(GraphicsPipe) GraphicsPipeSelection::
151make_pipe(const string &type_name, const string &module_name) {
152 TypeRegistry *type_reg = TypeRegistry::ptr();
153
154 // First, see if the type is already available.
155 TypeHandle type = type_reg->find_type(type_name);
156
157 // If it isn't, try the named module.
158 if (type == TypeHandle::none()) {
159 if (!module_name.empty()) {
160 load_named_module(module_name);
161 type = type_reg->find_type(type_name);
162 }
163 }
164
165 // If that didn't help, try the default module.
166 if (type == TypeHandle::none()) {
167 load_default_module();
168 type = type_reg->find_type(type_name);
169 }
170
171 // Still not enough, try all modules.
172 if (type == TypeHandle::none()) {
173 load_aux_modules();
174 type = type_reg->find_type(type_name);
175 }
176
177 if (type == TypeHandle::none()) {
178 return nullptr;
179 }
180
181 return make_pipe(type);
182}

194 for (const PipeType &ptype : _pipe_types) {

256PT(GraphicsPipe) GraphicsPipeSelection::
257make_default_pipe() {
258 load_default_module();
259
260 LightMutexHolder holder(_lock);
261
262 if (!_default_pipe_name.empty()) {
263 // First, look for an exact match of the default type name from the
264 // Config.prc file (excepting case and hyphen / underscore).
265 for (const PipeType &ptype : _pipe_types) {
266 if (cmp_nocase_uh(ptype._type.get_name(), _default_pipe_name) == 0) {
267 // Here's an exact match.
268 PT(GraphicsPipe) pipe = (*ptype._constructor)();
269 if (pipe != nullptr) {
270 return pipe;
271 }
272 }
273 }
274
275 // No match; look for a substring match.
276 string preferred_name = downcase(_default_pipe_name);
277 for (const PipeType &ptype : _pipe_types) {
278 string ptype_name = downcase(ptype._type.get_name());
279 if (ptype_name.find(preferred_name) != string::npos) {
280 // Here's a substring match.
281 PT(GraphicsPipe) pipe = (*ptype._constructor)();
282 if (pipe != nullptr) {
283 return pipe;
284 }
285 }
286 }
287 }
288
289 // Look for the preferred type of the default display module.
290 if (_default_pipe_type != TypeHandle::none()) {
291 for (const PipeType &ptype : _pipe_types) {
292 if (ptype._type == _default_pipe_type) {
293 PT(GraphicsPipe) pipe = (*ptype._constructor)();
294 if (pipe != nullptr) {
295 return pipe;
296 }
297 }
298 }
299 }
300
301 // Couldn't find a matching pipe type; choose the first one on the list.
302 for (const PipeType &ptype : _pipe_types) {
303 PT(GraphicsPipe) pipe = (*ptype._constructor)();
304 if (pipe != nullptr) {
305 return pipe;
306 }
307 }
308
309 // Nothing. Probably the list was empty.
310 return nullptr;
311}

401TypeHandle GraphicsPipeSelection::
402load_named_module(const string &name) {
403 LightMutexHolder holder(_loaded_modules_lock);
404
405 LoadedModules::iterator mi = _loaded_modules.find(name);
406 if (mi != _loaded_modules.end()) {
407 // We have previously loaded this module. Don't attempt to re-load it.
408 return (*mi).second._default_pipe_type;
409 }
410
411 // We have not yet loaded this module. Load it now.
412 Filename dlname = Filename::dso_filename("lib" + name + ".so");
413 display_cat.info()
414 << "loading display module: " << dlname.to_os_specific() << std::endl;
415 void *handle = load_dso(get_plugin_path().get_value(), dlname);
416 if (handle == nullptr) {
417 std::string error = load_dso_error();
418 display_cat.warning()
419 << "Unable to load " << dlname.to_os_specific() << ": " << error << std::endl;
420 return TypeHandle::none();
421 }
422
423 // Now get the module's recommended pipe type. This requires calling a
424 // specially-named function that should have been exported from the module.
425 string symbol_name = "get_pipe_type_" + name;
426 void *dso_symbol = get_dso_symbol(handle, symbol_name);
427 if (display_cat.is_debug()) {
428 display_cat.debug()
429 << "symbol of " << symbol_name << " = " << dso_symbol << "\n";
430 }
431
432 TypeHandle pipe_type = TypeHandle::none();
433
434 if (dso_symbol == nullptr) {
435 // Couldn't find the module function.
436 display_cat.warning()
437 << "Unable to find " << symbol_name << " in " << dlname.to_os_specific()
438 << "\n";
439
440 } else {
441 // We successfully loaded the module, and we found the get_pipe_type_*
442 // recommendation function. Call it to figure out what pipe type we
443 // should expect.
444 typedef int FuncType();
445 int pipe_type_index = (*(FuncType *)dso_symbol)();
446 if (display_cat.is_debug()) {
447 display_cat.debug()
448 << "pipe_type_index = " << pipe_type_index << "\n";
449 }
450
451 if (pipe_type_index != 0) {
452 TypeRegistry *type_reg = TypeRegistry::ptr();
453 pipe_type = type_reg->find_type_by_id(pipe_type_index);
454 if (display_cat.is_debug()) {
455 display_cat.debug()
456 << "pipe_type = " << pipe_type << "\n";
457 }
458 }
459 }
460
461 if (pipe_type == TypeHandle::none()) {
462 // The recommendation function returned a bogus type index, or the
463 // function didn't work at all. We can't safely unload the module,
464 // though, because it may have assigned itself into the
465 // GraphicsPipeSelection table. So we carry on.
466 display_cat.warning()
467 << "No default pipe type available for " << dlname.to_os_specific()
468 << "\n";
469 }
470
471 LoadedModule &module = _loaded_modules[name];
472 module._module_name = name;
473 module._module_handle = handle;
474 module._default_pipe_type = pipe_type;
475
476 return pipe_type;
477}
