447void x11GraphicsPipe::
448send_startup_notification() {
449 if (_sent_startup_notification || _startup_id.empty()) {
450 return;
451 }
452
453 // Allocate enough room for the message, with room for escape characters.
454 char *message = (char *)alloca(_startup_id.size() * 2 + 14);
455 memcpy(message, "remove: ID=\"", 12);
456
457 char *p = message + 12;
458 for (char c : _startup_id) {
459 if (c == '"' || c == '\\') {
460 *p++ = '\\';
461 }
462 *p++ = c;
463 }
464 *p++ = '\"';
465 *p++ = '\0';
466
467 if (x11display_cat.is_debug()) {
468 x11display_cat.debug()
469 << "Sending startup info message: " << message << "\n";
470 }
471
472 // It doesn't *strictly* seem to be necessary to create a window for this
473 // (just passing in the root window works too) but the spec says we should
474 // and GTK does it too, so why not?
475 XSetWindowAttributes attrs;
476 attrs.override_redirect = True;
477 attrs.event_mask = PropertyChangeMask | StructureNotifyMask;
478 X11_Window xwin = XCreateWindow(_display, _root, -100, -100, 1, 1, 0,
479 CopyFromParent, CopyFromParent, CopyFromParent,
480 CWOverrideRedirect | CWEventMask, &attrs);
481
482 XEvent xevent;
483 xevent.xclient.type = ClientMessage;
484 xevent.xclient.message_type = _net_startup_info_begin;
485 xevent.xclient.display = _display;
486 xevent.xclient.window = xwin;
487 xevent.xclient.format = 8;
488
489 const char *src = message;
490 const char *src_end = message + strlen(message) + 1;
491
492 char *dest, *dest_end;
493 while (src != src_end) {
494 dest = &xevent.xclient.data.b[0];
495 dest_end = dest + 20;
496
497 while (dest != dest_end && src != src_end) {
498 *dest = *src;
499 ++dest;
500 ++src;
501 }
502
503 while (dest != dest_end) {
504 *dest = '\0';
505 ++dest;
506 }
507
508 XSendEvent(_display, _root, False, PropertyChangeMask, &xevent);
509 xevent.xclient.message_type = _net_startup_info;
510 }
511
512 XDestroyWindow(_display, xwin);
513 XFlush(_display);
514
515 _sent_startup_notification = true;
516}

622 y = info->y;

647 // thread, which we didn't intend (though the global _x_mutex may allow them

709error_handler(X11_Display *display, XErrorEvent *error) {
710 ++_x_error_count;
711
712 static const int msg_len = 80;
713 char msg[msg_len];
714 XGetErrorText(display, error->error_code, msg, msg_len);
715
716 if (!_x_error_messages_enabled) {
717 if (x11display_cat.is_debug()) {
718 x11display_cat.debug()
719 << msg << "\n";
720 }
721 return 0;
722 }

724 x11display_cat.error()
725 << msg << "\n";
726
727 if (x_error_abort) {

731 // We return to allow the application to continue running, unlike the
732 // default X error handler which exits.
733 return 0;
