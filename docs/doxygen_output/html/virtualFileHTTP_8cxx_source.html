137istream *VirtualFileHTTP::
138open_read_file(bool auto_unwrap) const {
139 if (_status_only) {
140 return nullptr;
141 }
142
143 // We pre-download the file into a StringStream, then return a buffer to
144 // that. It seems safer, since we can guarantee the file comes all at once
145 // without timeouts along the way.
146 StringStream *strstream = new StringStream;
147 if (!fetch_file(strstream)) {
148 delete strstream;
149 return nullptr;
150 }
151
152 strstream->seekg(0);
153
154 return return_file(strstream, auto_unwrap);
155}

206bool VirtualFileHTTP::
207fetch_file(ostream *buffer_stream) const {
208 _channel->download_to_stream(buffer_stream, false);
209 if (!_channel->is_download_complete()) {
210 // Failure to download fully. Try again to download more.
211
212 URLSpec url(_mount->get_root());
213 url.set_path(_mount->get_root().get_path() + _local_filename.c_str());
214
215 size_t bytes_downloaded = _channel->get_bytes_downloaded();
216 size_t last_byte = bytes_downloaded;
217
218 while (bytes_downloaded > 0 && !_channel->is_download_complete()) {
219 _channel->get_subdocument(url, last_byte, 0);
220 _channel->download_to_stream(buffer_stream, true);
221 bytes_downloaded = _channel->get_bytes_downloaded();
222 last_byte = _channel->get_last_byte_delivered();
223 }
224 }
225
226 return _channel->is_download_complete() && _channel->is_valid();
227}

233istream *VirtualFileHTTP::
234return_file(istream *buffer_stream, bool auto_unwrap) const {
235 // Will we be automatically unwrapping a .pz file?
236 bool do_unwrap = (_implicit_pz_file || (auto_unwrap && _local_filename.get_extension() == "pz"));
237
238 istream *result = buffer_stream;
239#ifdef HAVE_ZLIB
240 if (result != nullptr && do_unwrap) {
241 // We have to slip in a layer to decompress the file on the fly.
242 IDecompressStream *wrapper = new IDecompressStream(result, true);
243 result = wrapper;
244 }
245#endif // HAVE_ZLIB
246
247 return result;
248}
