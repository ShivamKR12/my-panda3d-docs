50void DaeMaterials::add_material_instance(const FCDMaterialInstance* instance) {
51 nassertv(instance != nullptr);
52 const string semantic (FROM_FSTRING(instance->GetSemantic()));
53 if (_materials.count(semantic) > 0) {
54 daeegg_cat.warning() << "Ignoring duplicate material with semantic " << semantic << endl;
55 return;
56 }
57 _materials[semantic] = new DaeMaterial();
58
59 // Load in the uvsets
60 for (size_t vib = 0; vib < instance->GetVertexInputBindingCount(); ++vib) {
61 const FCDMaterialInstanceBindVertexInput* mivib = instance->GetVertexInputBinding(vib);
62 assert(mivib != nullptr);
63 PT(DaeVertexInputBinding) bvi = new DaeVertexInputBinding();
64 bvi->_input_set = mivib->inputSet;
65#if FCOLLADA_VERSION >= 0x00030005
66 bvi->_input_semantic = mivib->GetInputSemantic();
67 bvi->_semantic = *mivib->semantic;
68#else
69 bvi->_input_semantic = mivib->inputSemantic;
70 bvi->_semantic = FROM_FSTRING(mivib->semantic);
71#endif
72 _materials[semantic]->_uvsets.push_back(bvi);
73 }
74
75 // Handle the material stuff
76 if (daeegg_cat.is_spam()) {
77 daeegg_cat.spam() << "Trying to process material with semantic " << semantic << endl;
78 }
79 PT_EggMaterial egg_material = new EggMaterial(semantic);
80 pvector<PT_EggTexture> egg_textures;
81 const FCDEffect* effect = instance->GetMaterial()->GetEffect();
82 if (effect == nullptr) {
83 if (daeegg_cat.is_debug()) {
84 daeegg_cat.debug() << "Ignoring material (semantic: " << semantic << ") without assigned effect" << endl;
85 }
86 } else {
87 // Grab the common profile effect
88 const FCDEffectStandard* effect_common = (FCDEffectStandard *)effect->FindProfile(FUDaeProfileType::COMMON);
89 if (effect_common == nullptr) {
90 daeegg_cat.info() << "Ignoring effect referenced by material with semantic " << semantic
91 << " because it has no common profile" << endl;
92 } else {
93 if (daeegg_cat.is_spam()) {
94 daeegg_cat.spam() << "Processing effect, material semantic is " << semantic << endl;
95 }
96 // Set the material parameters
97 egg_material->set_amb(TO_COLOR(effect_common->GetAmbientColor()));
98 // We already process transparency using blend modes LVecBase4 diffuse =
99 // TO_COLOR(effect_common->GetDiffuseColor());
100 // diffuse.set_w(diffuse.get_w() * (1.0f -
101 // effect_common->GetOpacity())); egg_material->set_diff(diffuse);
102 egg_material->set_diff(TO_COLOR(effect_common->GetDiffuseColor()));
103 egg_material->set_emit(TO_COLOR(effect_common->GetEmissionColor()) * effect_common->GetEmissionFactor());
104 egg_material->set_shininess(effect_common->GetShininess());
105 egg_material->set_spec(TO_COLOR(effect_common->GetSpecularColor()));
106 // Now try to load in the textures
107 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::DIFFUSE, EggTexture::ET_modulate);
108 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::BUMP, EggTexture::ET_normal);
109 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::SPECULAR, EggTexture::ET_modulate_gloss);
110 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::SPECULAR_LEVEL, EggTexture::ET_gloss);
111 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::TRANSPARENT, EggTexture::ET_unspecified, EggTexture::F_alpha);
112 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::EMISSION, EggTexture::ET_add);
113#if FCOLLADA_VERSION < 0x00030005
114 process_texture_bucket(semantic, effect_common, FUDaeTextureChannel::OPACITY, EggTexture::ET_unspecified, EggTexture::F_alpha);
115#endif
116 // Now, calculate the color blend stuff.
117 _materials[semantic]->_blend = convert_blend(effect_common->GetTransparencyMode(),
118 TO_COLOR(effect_common->GetTranslucencyColor()),
119 effect_common->GetTranslucencyFactor());
120 }
121 // Find an <extra> tag to support some extra stuff from extensions
122 process_extra(semantic, effect->GetExtra());
123 }
124 if (daeegg_cat.is_spam()) {
125 daeegg_cat.spam() << "Found " << egg_textures.size() << " textures in material" << endl;
126 }
127 _materials[semantic]->_egg_material = egg_material;
128}

134void DaeMaterials::
135process_texture_bucket(const string semantic, const FCDEffectStandard* effect_common, FUDaeTextureChannel::Channel bucket, EggTexture::EnvType envtype, EggTexture::Format format) {
136 for (size_t tx = 0; tx < effect_common->GetTextureCount(bucket); ++tx) {
137 const FCDImage* image = effect_common->GetTexture(bucket, tx)->GetImage();
138 if (image == nullptr) {
139 daeegg_cat.warning() << "Texture references a nonexisting image!" << endl;
140 } else {
141 const FCDEffectParameterSampler* sampler = effect_common->GetTexture(bucket, tx)->GetSampler();
142 // FCollada only supplies absolute paths. We need to grab the document
143 // location ourselves and make the image path absolute.
144 Filename texpath;
145 if (image->GetDocument()) {
146 Filename docpath = Filename::from_os_specific(FROM_FSTRING(image->GetDocument()->GetFileUrl()));
147 docpath.make_canonical();
148 texpath = Filename::from_os_specific(FROM_FSTRING(image->GetFilename()));
149 texpath.make_canonical();
150 texpath.make_relative_to(docpath.get_dirname(), true);
151 if (daeegg_cat.is_debug()) {
152 daeegg_cat.debug() << "Found texture with path " << texpath << endl;
153 }
154 } else {
155 // Never mind.
156 texpath = Filename::from_os_specific(FROM_FSTRING(image->GetFilename()));
157 }
158 PT_EggTexture egg_texture = new EggTexture(FROM_FSTRING(image->GetDaeId()), texpath.to_os_generic());
159 // Find a set of UV coordinates
160 const FCDEffectParameterInt* uvset = effect_common->GetTexture(bucket, tx)->GetSet();
161 if (uvset != nullptr) {
162 if (daeegg_cat.is_debug()) {
163 daeegg_cat.debug() << "Texture has uv name '" << FROM_FSTRING(uvset->GetSemantic()) << "'\n";
164 }
165 string uvset_semantic (FROM_FSTRING(uvset->GetSemantic()));
166
167 // Only set the UV name if this UV set actually exists.
168 for (size_t i = 0; i < _materials[semantic]->_uvsets.size(); ++i) {
169 if (_materials[semantic]->_uvsets[i]->_semantic == uvset_semantic) {
170 egg_texture->set_uv_name(uvset_semantic);
171 break;
172 }
173 }
174 }
175 // Apply sampler stuff
176 if (sampler != nullptr) {
177 egg_texture->set_texture_type(convert_texture_type(sampler->GetSamplerType()));
178 egg_texture->set_wrap_u(convert_wrap_mode(sampler->GetWrapS()));
179 if (sampler->GetSamplerType() != FCDEffectParameterSampler::SAMPLER1D) {
180 egg_texture->set_wrap_v(convert_wrap_mode(sampler->GetWrapT()));
181 }
182 if (sampler->GetSamplerType() == FCDEffectParameterSampler::SAMPLER3D) {
183 egg_texture->set_wrap_w(convert_wrap_mode(sampler->GetWrapP()));
184 }
185 egg_texture->set_minfilter(convert_filter_type(sampler->GetMinFilter()));
186 egg_texture->set_magfilter(convert_filter_type(sampler->GetMagFilter()));
187 if (envtype != EggTexture::ET_unspecified) {
188 egg_texture->set_env_type(envtype);
189 }
190 if (format != EggTexture::F_unspecified) {
191 egg_texture->set_format(format);
192 }
193 }
194 _materials[semantic]->_egg_textures.push_back(egg_texture);
195 }
196 }
197}

203void DaeMaterials::
204process_extra(const string semantic, const FCDExtra* extra) {
205 if (extra == nullptr) return;
206 const FCDEType* etype = extra->GetDefaultType();
207 if (etype == nullptr) return;
208 for (size_t et = 0; et < etype->GetTechniqueCount(); ++et) {
209 const FCDENode* enode = ((const FCDENode*)(etype->GetTechnique(et)))->FindChildNode("double_sided");
210 if (enode != nullptr) {
211 string content = trim(enode->GetContent());
212 if (content == "1" || content == "true") {
213 _materials[semantic]->_double_sided = true;
214 } else if (content == "0" || content == "false") {
215 _materials[semantic]->_double_sided = false;
216 } else {
217 daeegg_cat.warning() << "Expected <double_sided> tag to be either 1 or 0, found '" << content << "' instead" << endl;
218 }
219 }
220 }

243void DaeMaterials::
244apply_to_group(const string semantic, const PT(EggGroup) to, bool invert_transparency) {
245 if (_materials.count(semantic) > 0) {
246 PT(DaeBlendSettings) blend = _materials[semantic]->_blend;
247 if (blend && blend->_enabled) {
248 to->set_blend_mode(EggGroup::BM_add);
249 to->set_blend_color(blend->_color);
250 if (invert_transparency) {
251 to->set_blend_operand_a(blend->_operand_b);
252 to->set_blend_operand_b(blend->_operand_a);
253 } else {
254 to->set_blend_operand_a(blend->_operand_a);
255 to->set_blend_operand_b(blend->_operand_b);
256 }
257 } else if (blend && invert_transparency) {
258 to->set_blend_mode(EggGroup::BM_add);
259 to->set_blend_color(blend->_color);
260 to->set_blend_operand_a(blend->_operand_b);
261 to->set_blend_operand_b(blend->_operand_a);
262 }
263 }

270const string DaeMaterials::
271get_uvset_name(const string semantic, FUDaeGeometryInput::Semantic input_semantic, int32 input_set) {
272 if (_materials.count(semantic) > 0) {
273 if (input_set == -1 && _materials[semantic]->_uvsets.size() == 1) {
274 return _materials[semantic]->_uvsets[0]->_semantic;
275 } else {
276 for (size_t i = 0; i < _materials[semantic]->_uvsets.size(); ++i) {
277 if (_materials[semantic]->_uvsets[i]->_input_set == input_set &&
278 _materials[semantic]->_uvsets[i]->_input_semantic == input_semantic) {
279 return _materials[semantic]->_uvsets[i]->_semantic;
280 }
281 }
282 // If we can't find it, let's look again, but don't care for the
283 // input_semantic this time. The reason for this is that some tools
284 // export textangents and texbinormals bound to a uvset with input
285 // semantic TEXCOORD.
286 for (size_t i = 0; i < _materials[semantic]->_uvsets.size(); ++i) {
287 if (_materials[semantic]->_uvsets[i]->_input_set == input_set) {
288 if (daeegg_cat.is_debug()) {
289 daeegg_cat.debug() << "Using uv set with non-matching input semantic " << _materials[semantic]->_uvsets[i]->_semantic << "\n";
290 }
291 return _materials[semantic]->_uvsets[i]->_semantic;
292 }
293 }
294 if (daeegg_cat.is_debug()) {
295 daeegg_cat.debug() << "No uv set binding found for input set " << input_set << "\n";
296 }
297 }
298 }
299 return "";
300}

381PT(DaeMaterials::DaeBlendSettings) DaeMaterials::
382convert_blend(FCDEffectStandard::TransparencyMode mode, const LColor &transparent, double transparency) {
383 // Create the DaeBlendSettings and fill it with some defaults.
384 PT(DaeBlendSettings) blend = new DaeBlendSettings();
385 blend->_enabled = true;
386 blend->_color = LColor::zero();
387 blend->_operand_a = EggGroup::BO_unspecified;
388 blend->_operand_b = EggGroup::BO_unspecified;
389
390 // First fill in the color value.
391 if (mode == FCDEffectStandard::A_ONE) {// || mode == FCDEffectStandard::A_ZERO) {
392 double value = transparent[3] * transparency;
393 blend->_color = LColor(value, value, value, value);
394 } else if (mode == FCDEffectStandard::RGB_ZERO) {//|| mode == FCDEffectStandard::RGB_ONE) {
395 blend->_color = transparent * transparency;
396 blend->_color[3] = luminance(blend->_color);
397 } else {
398 daeegg_cat.error() << "Unknown opaque type found!" << endl;
399 blend->_enabled = false;
400 return blend;
401 }
402
403 // Now figure out the operands.
404 if (mode == FCDEffectStandard::RGB_ZERO) {// || mode == FCDEffectStandard::A_ZERO) {
405 blend->_operand_a = EggGroup::BO_one_minus_constant_color;
406 blend->_operand_b = EggGroup::BO_constant_color;
407 } else if (mode == FCDEffectStandard::A_ONE) {// || mode == FCDEffectStandard::RGB_ONE) {
408 blend->_operand_a = EggGroup::BO_constant_color;
409 blend->_operand_b = EggGroup::BO_one_minus_constant_color;
410 } else {
411 daeegg_cat.error() << "Unknown opaque type found!" << endl;
412 blend->_enabled = false;
413 return blend;
414 }
415
416 // See if we can optimize out the color.
417 if (blend->_operand_a == EggGroup::BO_constant_color) {
418 if (blend->_color == LColor::zero()) {
419 blend->_operand_a = EggGroup::BO_zero;
420 } else if (blend->_color == LColor(1, 1, 1, 1)) {
421 blend->_operand_a = EggGroup::BO_one;
422 }
423 }
424 if (blend->_operand_b == EggGroup::BO_constant_color) {
425 if (blend->_color == LColor::zero()) {
426 blend->_operand_b = EggGroup::BO_zero;
427 } else if (blend->_color == LColor(1, 1, 1, 1)) {
428 blend->_operand_b = EggGroup::BO_one;
429 }
430 }
431 if (blend->_operand_a == EggGroup::BO_one_minus_constant_color) {
432 if (blend->_color == LColor::zero()) {
433 blend->_operand_a = EggGroup::BO_one;
434 } else if (blend->_color == LColor(1, 1, 1, 1)) {
435 blend->_operand_a = EggGroup::BO_zero;
436 }
437 }
438 if (blend->_operand_b == EggGroup::BO_one_minus_constant_color) {
439 if (blend->_color == LColor::zero()) {
440 blend->_operand_b = EggGroup::BO_one;
441 } else if (blend->_color == LColor(1, 1, 1, 1)) {
442 blend->_operand_b = EggGroup::BO_zero;
443 }
444 }
445
446 // See if we can entirely disable the blend.
447 if (blend->_operand_a == EggGroup::BO_one && blend->_operand_b == EggGroup::BO_zero) {
448 blend->_enabled = false;
449 }
450 return blend;
451}
