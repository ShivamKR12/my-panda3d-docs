151string TextEncoder::
152encode_wchar(char32_t ch, TextEncoder::Encoding encoding) {
153 switch (encoding) {
154 case E_iso8859:
155 if ((ch & ~0xff) == 0) {
156 return string(1, (char)ch);
157 } else {
158 // The character won't fit in the 8-bit ISO 8859. See if we can make it
159 // fit by reducing it to its ascii equivalent (essentially stripping off
160 // an unusual accent mark).
161 const UnicodeLatinMap::Entry *map_entry =
162 UnicodeLatinMap::look_up(ch);
163 if (map_entry != nullptr && map_entry->_ascii_equiv != 0) {
164 // Yes, it has an ascii equivalent.
165 if (map_entry->_ascii_additional != 0) {
166 // In fact, it has two of them.
167 return
168 string(1, map_entry->_ascii_equiv) +
169 string(1, map_entry->_ascii_additional);
170 }
171 return string(1, map_entry->_ascii_equiv);
172 }
173 // Nope; return "." for lack of anything better.
174 return ".";
175 }
176
177 case E_utf8:
178 if ((ch & ~0x7f) == 0) {
179 return string(1, (char)ch);
180 } else if ((ch & ~0x7ff) == 0) {
181 return
182 string(1, (char)((ch >> 6) | 0xc0)) +
183 string(1, (char)((ch & 0x3f) | 0x80));
184 } else if ((ch & ~0xffff) == 0) {
185 return
186 string(1, (char)((ch >> 12) | 0xe0)) +
187 string(1, (char)(((ch >> 6) & 0x3f) | 0x80)) +
188 string(1, (char)((ch & 0x3f) | 0x80));
189 } else {
190 return
191 string(1, (char)((ch >> 18) | 0xf0)) +
192 string(1, (char)(((ch >> 12) & 0x3f) | 0x80)) +
193 string(1, (char)(((ch >> 6) & 0x3f) | 0x80)) +
194 string(1, (char)((ch & 0x3f) | 0x80));
195 }
196
197 case E_utf16be:
198 if ((ch & ~0xffff) == 0) {
199 // Note that this passes through surrogates and BOMs unharmed.
200 return
201 string(1, (char)(ch >> 8)) +
202 string(1, (char)(ch & 0xff));
203 } else {
204 // Use a surrogate pair.
205 uint32_t v = (uint32_t)ch - 0x10000u;
206 uint16_t hi = (v >> 10u) | 0xd800u;
207 uint16_t lo = (v & 0x3ffu) | 0xdc00u;
208 char encoded[4] = {
209 (char)(hi >> 8),
210 (char)(hi & 0xff),
211 (char)(lo >> 8),
212 (char)(lo & 0xff),
213 };
214 return string(encoded, 4);
215 }
216
217 case E_cp437:
218 if ((ch & ~0x7f) == 0) {
219 return string(1, (char)ch);
220 }
221 else if (ch >= 0 && ch < 0x266b) {
222 // This case is not optimized, because we don't really need it right now.
223 for (int i = 0; i < 256; ++i) {
224 if (cp437_table[i] == ch) {
225 return std::string(1, (char)i);
226 }
227 }
228 }
229 return ".";
230 }
231
232 return "";
233}

239string TextEncoder::
240encode_wtext(const wstring &wtext, TextEncoder::Encoding encoding) {
241 string result;
242
243 for (size_t i = 0; i < wtext.size(); ++i) {
244 wchar_t ch = wtext[i];
245
246 // On some systems, wstring may be UTF-16, and contain surrogate pairs.
247#if WCHAR_MAX < 0x10FFFF
248 if (ch >= 0xd800 && ch < 0xdc00 && (i + 1) < wtext.size()) {
249 // This is a high surrogate. Look for a subsequent low surrogate.
250 wchar_t ch2 = wtext[i + 1];
251 if (ch2 >= 0xdc00 && ch2 < 0xe000) {
252 // Yes, this is a low surrogate.
253 char32_t code_point = 0x10000 + ((ch - 0xd800) << 10) + (ch2 - 0xdc00);
254 result += encode_wchar(code_point, encoding);
255 i++;
256 continue;
257 }
258 }
259#endif
260
261 result += encode_wchar(ch, encoding);
262 }
263
264 return result;
265}
