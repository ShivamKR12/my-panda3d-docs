164bool Character::
165cull_callback(CullTraverser *trav, CullTraverserData &data) {
166 // For now, we update the character during the cull traversal; this prevents
167 // us from needlessly updating characters that aren't in the view frustum.
168 // We may need a better way to do this optimization later, to handle
169 // characters that might animate themselves in front of the view frustum.
170
171 if (_do_lod_animation) {
172 int this_frame = ClockObject::get_global_clock()->get_frame_count();
173
174 CPT(TransformState) rel_transform = get_rel_transform(trav, data);
175 LPoint3 center = _lod_center * rel_transform->get_mat();
176 PN_stdfloat dist2 = center.dot(center);
177
178 if (this_frame != _view_frame || dist2 < _view_distance2) {
179 _view_frame = this_frame;
180 _view_distance2 = dist2;
181
182 // Now compute the lod delay.
183 PN_stdfloat dist = sqrt(dist2);
184 double delay = 0.0;
185 if (dist > _lod_near_distance) {
186 delay = _lod_delay_factor * (dist - _lod_near_distance) / (_lod_far_distance - _lod_near_distance);
187 nassertr(delay > 0.0, false);
188 }
189 set_lod_current_delay(delay);
190
191 if (char_cat.is_spam()) {
192 char_cat.spam()
193 << "Distance to " << NodePath::any_path(this) << " in frame "
194 << this_frame << " is " << dist << ", computed delay is " << delay
195 << "\n";
196 }
197 }
198 }
199
200 update();
201 return true;
202}

218CPT(TransformState) Character::
219calc_tight_bounds(LPoint3 &min_point, LPoint3 &max_point, bool &found_any,
220 const TransformState *transform, Thread *current_thread) const {
221 // This method is overridden by Character solely to provide a hook to force
222 // the joints to update before computing the bounding volume.
223 ((Character *)this)->update_to_now();
224
225 // Unfortunately, calling update_to_now() will invalidate the node's cached
226 // bounding volume, which causes a problem when this is called during the
227 // traversal, e.g. due to a ShowBoundsEffect. As a hacky fix to work
228 // around this, we will force-recompute all of the bounding volumes of our
229 // parent nodes immediately.
230 Parents parents = get_parents();
231 for (size_t i = 0; i < parents.get_num_parents(); ++i) {
232 PandaNode *parent = parents.get_parent(i);
233 parent->get_bounds();
234 }
235
236 return PandaNode::calc_tight_bounds(min_point, max_point,
237 found_any, transform, current_thread);
238}

447void Character::
448r_copy_children(const PandaNode *from, PandaNode::InstanceMap &inst_map,
449 Thread *current_thread) {
450 // We assume there will be no instancing going on below the Character node.
451 // If there is, too bad; it will get flattened out.
452
453 // We preempt the node's r_copy_children() operation with our own function
454 // that keeps track of the old vs. new nodes and also updates any Geoms we
455 // find with our new dynamic vertices.
456
457 const Character *from_char;
458 DCAST_INTO_V(from_char, from);
459 NodeMap node_map;
460 JointMap joint_map;
461
462 LightMutexHolder from_holder(from_char->_lock);
463 LightMutexHolder holder(_lock);
464
465 size_t num_bundles = _bundles.size();
466 nassertv(from_char->_bundles.size() == num_bundles);
467 for (size_t i = 0; i < num_bundles; ++i) {
468 fill_joint_map(joint_map, _bundles[i]->get_bundle(), from_char->_bundles[i]->get_bundle());
469 }
470
471 GeomVertexMap gvmap;
472 GeomJointMap gjmap;
473 GeomSliderMap gsmap;
474 r_copy_char(this, from_char, from_char, node_map, joint_map,
475 gvmap, gjmap, gsmap);
476
477 for (size_t i = 0; i < num_bundles; ++i) {
478 copy_node_pointers(node_map, _bundles[i]->get_bundle(), from_char->_bundles[i]->get_bundle());
479 }

488void Character::
489update_bundle(PartBundleHandle *old_bundle_handle, PartBundle *new_bundle) {
490 if (old_bundle_handle->get_bundle() == new_bundle) {
491 // Trivially return.
492 return;
493 }
494
495 // First, merge the bundles, to ensure we have the same set of joints in the
496 // new bundle.
497 JointMap joint_map;
498 r_merge_bundles(joint_map, old_bundle_handle->get_bundle(), new_bundle);
499
500 PartBundleNode::update_bundle(old_bundle_handle, new_bundle);
501
502 // Now convert the geometry to use the new bundle.
503 GeomVertexMap gvmap;
504 GeomJointMap gjmap;
505 GeomSliderMap gsmap;
506 r_update_geom(this, joint_map, gvmap, gjmap, gsmap);
507}

513CPT(TransformState) Character::
514get_rel_transform(CullTraverser *trav, CullTraverserData &data) {
515 // Get a pointer to the camera node.
516 Camera *camera = trav->get_scene()->get_camera_node();
517
518 // Get the camera space transform.
519 CPT(TransformState) rel_transform;
520
521 NodePath lod_center = camera->get_lod_center();
522 if (!lod_center.is_empty()) {
523 rel_transform =
524 lod_center.get_net_transform()->invert_compose(data.get_net_transform(trav));
525 } else {
526 NodePath cull_center = camera->get_cull_center();
527 if (!cull_center.is_empty()) {
528 rel_transform =
529 cull_center.get_net_transform()->invert_compose(data.get_net_transform(trav));
530 } else {
531 rel_transform = data.get_modelview_transform(trav);
532 }
533 }
534
535 return rel_transform;
536}

607void Character::
608r_merge_bundles(Character::JointMap &joint_map,
609 PartGroup *old_group, PartGroup *new_group) {
610 joint_map[old_group] = new_group;
611
612 if (new_group->is_character_joint()) {
613 CharacterJoint *new_joint;
614 DCAST_INTO_V(new_joint, new_group);
615
616 // Make sure the new_joint references this as its new character.
617 new_joint->_character = this;
618
619 if (old_group != new_group &&
620 old_group->is_character_joint()) {
621 CharacterJoint *old_joint;
622 DCAST_INTO_V(old_joint, old_group);
623
624 // Since the old_joint will be getting dropped, reset its character
625 // reference.
626 old_joint->_character = nullptr;
627
628 // Copy any _net_transform and _local_transform operations to the new
629 // joint.
630 CharacterJoint::NodeList::iterator ni;
631 for (ni = old_joint->_net_transform_nodes.begin();
632 ni != old_joint->_net_transform_nodes.end();
633 ++ni) {
634 new_joint->_net_transform_nodes.insert(*ni);
635 }
636 for (ni = old_joint->_local_transform_nodes.begin();
637 ni != old_joint->_local_transform_nodes.end();
638 ++ni) {
639 new_joint->_local_transform_nodes.insert(*ni);
640 }
641 }
642 }
643
644 if (old_group == new_group) {
645 return;
646 }
647
648 int i = 0, j = 0;
649 int old_num_children = old_group->get_num_children();
650 int new_num_children = new_group->get_num_children();
651
652 PartGroup::Children new_children(PartGroup::get_class_type());
653 new_children.reserve(std::max(old_num_children, new_num_children));
654
655 while (i < old_num_children && j < new_num_children) {
656 PartGroup *pc = old_group->get_child(i);
657 PartGroup *ac = new_group->get_child(j);
658
659 if (pc->get_name() < ac->get_name()) {
660 // Here is a group that exists in old_group, but not in new_group.
661 // Duplicate it.
662 PartGroup *new_pc = pc->make_copy();
663 new_children.push_back(new_pc);
664
665 r_merge_bundles(joint_map, pc, new_pc);
666 i++;
667
668 } else if (ac->get_name() < pc->get_name()) {
669 // Here is a group that exists in new_group, but not in old_group.
670 // Preserve it.
671 new_children.push_back(ac);
672
673 r_merge_bundles(joint_map, ac, ac);
674 j++;
675
676 } else {
677 // Here is a group that exists in both. Preserve it.
678 new_children.push_back(ac);
679
680 r_merge_bundles(joint_map, pc, ac);
681 i++;
682 j++;
683 }
684 }
685
686 while (i < old_num_children) {
687 PartGroup *pc = old_group->get_child(i);
688
689 // Here is a group that exists in old_group, but not in new_group.
690 // Duplicate it.
691 PartGroup *new_pc = pc->make_copy();
692 new_children.push_back(new_pc);
693
694 r_merge_bundles(joint_map, pc, new_pc);
695 i++;
696 }
697
698 while (j < new_num_children) {
699 PartGroup *ac = new_group->get_child(j);
700
701 // Here is a group that exists in new_group, but not in old_group.
702 // Preserve it.
703 new_children.push_back(ac);
704
705 r_merge_bundles(joint_map, ac, ac);
706 j++;
707 }
708
709 new_group->_children.swap(new_children);
710}

719void Character::
720r_copy_char(PandaNode *dest, const PandaNode *source,
721 const Character *from, Character::NodeMap &node_map,
722 const Character::JointMap &joint_map,
723 Character::GeomVertexMap &gvmap,
724 Character::GeomJointMap &gjmap, Character::GeomSliderMap &gsmap) {
725
726 if (source->is_geom_node()) {
727 const GeomNode *source_gnode;
728 GeomNode *dest_gnode;
729 DCAST_INTO_V(source_gnode, source);
730 DCAST_INTO_V(dest_gnode, dest);
731
732 dest_gnode->remove_all_geoms();
733 int num_geoms = source_gnode->get_num_geoms();
734 for (int i = 0; i < num_geoms; i++) {
735 const Geom *geom = source_gnode->get_geom(i);
736 const RenderState *state = source_gnode->get_geom_state(i);
737 dest_gnode->add_geom(copy_geom(geom, joint_map, gvmap, gjmap, gsmap), state);
738 }
739 }
740
741 int num_children = source->get_num_children();
742 for (int i = 0; i < num_children; i++) {
743 const PandaNode *source_child = source->get_child(i);
744 int source_sort = source->get_child_sort(i);
745
746 PT(PandaNode) dest_child;
747 if (source_child->is_of_type(Character::get_class_type())) {
748 // We make a special case for nodes of type Character. If we encounter
749 // one of these, we have a Character under a Character, and the nested
750 // Character's copy should be called instead of ours.
751 dest_child = source_child->copy_subgraph();
752
753 } else {
754 // Otherwise, we assume that make_copy() will make a suitable copy of
755 // the node. This does limit the sorts of things we can have parented
756 // to a Character and expect copy_subgraph() to work correctly. Too
757 // bad.
758 dest_child = source_child->make_copy();
759 r_copy_char(dest_child, source_child, from, node_map, joint_map,
760 gvmap, gjmap, gsmap);
761 }
762 dest->add_child(dest_child, source_sort);
763 node_map[source_child] = dest_child;
764 }
765}

803PT(Geom) Character::
804copy_geom(const Geom *source, const Character::JointMap &joint_map,
805 Character::GeomVertexMap &gvmap, Character::GeomJointMap &gjmap,
806 Character::GeomSliderMap &gsmap) {
807 CPT(GeomVertexFormat) format = source->get_vertex_data()->get_format();
808 if (format->get_animation().get_animation_type() == Geom::AT_none) {
809 // Not animated, so never mind.
810 return (Geom *)source;
811 }
812
813 PT(Geom) dest = source->make_copy();
814
815 CPT(GeomVertexData) orig_vdata = source->get_vertex_data();
816 PT(GeomVertexData) new_vdata;
817 GeomVertexMap::iterator gvmi = gvmap.find(orig_vdata);
818 if (gvmi != gvmap.end()) {
819 new_vdata = (*gvmi).second;
820 } else {
821 new_vdata = new GeomVertexData(*orig_vdata);
822
823 new_vdata->set_transform_table(redirect_transform_table(orig_vdata->get_transform_table(), joint_map, gjmap));
824 new_vdata->set_transform_blend_table(redirect_transform_blend_table(orig_vdata->get_transform_blend_table(), joint_map, gjmap));
825 new_vdata->set_slider_table(redirect_slider_table(orig_vdata->get_slider_table(), gsmap));
826
827 gvmap.insert(GeomVertexMap::value_type(orig_vdata, new_vdata));
828 }
829
830 dest->set_vertex_data(new_vdata);
831
832 return dest;

839void Character::
840copy_node_pointers(const Character::NodeMap &node_map,
841 PartGroup *dest, const PartGroup *source) {
842 if (dest->is_character_joint()) {
843 nassertv(dest != source);
844 const CharacterJoint *source_joint;
845 CharacterJoint *dest_joint;
846 DCAST_INTO_V(source_joint, source);
847 DCAST_INTO_V(dest_joint, dest);
848
849 CharacterJoint::NodeList::const_iterator ai;
850 for (ai = source_joint->_net_transform_nodes.begin();
851 ai != source_joint->_net_transform_nodes.end();
852 ++ai) {
853 PandaNode *source_node = (*ai);
854
855 NodeMap::const_iterator mi;
856 mi = node_map.find(source_node);
857 if (mi != node_map.end()) {
858 PandaNode *dest_node = (*mi).second;
859
860 // Here's an internal joint that the source Character was animating
861 // directly. We'll animate our corresponding joint the same way.
862 dest_joint->set_character(this);
863 dest_joint->add_net_transform(dest_node);
864 }
865 }
866
867 for (ai = source_joint->_local_transform_nodes.begin();
868 ai != source_joint->_local_transform_nodes.end();
869 ++ai) {
870 PandaNode *source_node = (*ai);
871
872 NodeMap::const_iterator mi;
873 mi = node_map.find(source_node);
874 if (mi != node_map.end()) {
875 PandaNode *dest_node = (*mi).second;
876
877 // Here's an internal joint that the source Character was animating
878 // directly. We'll animate our corresponding joint the same way.
879 dest_joint->set_character(this);
880 dest_joint->add_local_transform(dest_node);
881 }
882 }
883 }
884
885 // Now recurse over children.
886 int i = 0, j = 0;
887 int dest_num_children = dest->get_num_children();
888 int source_num_children = source->get_num_children();
889
890 while (i < dest_num_children && j < source_num_children) {
891 PartGroup *pc = dest->get_child(i);
892 PartGroup *ac = source->get_child(j);
893
894 if (pc->get_name() < ac->get_name()) {
895 i++;
896 } else if (ac->get_name() < pc->get_name()) {
897 j++;
898 } else {
899 copy_node_pointers(node_map, pc, ac);
900 i++;
901 j++;
902 }
903 }

910CPT(TransformTable) Character::
911redirect_transform_table(const TransformTable *source,
912 const Character::JointMap &joint_map,
913 Character::GeomJointMap &gjmap) {
914 if (source == nullptr) {
915 return nullptr;
916 }
917
918 PT(TransformTable) dest = new TransformTable(*source);

999 if (ji != gjmap.end()) {
