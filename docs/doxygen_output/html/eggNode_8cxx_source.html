247void EggNode::
248test_under_integrity() const {
249 if (_parent == nullptr) {
250 // If we have no parent, everything should be zero.
251 nassertv(_depth == 0);
252 nassertv(_under_flags == 0);
253 } else {
254 // Otherwise, make sure we're consistent with our parent.
255 _parent->test_ref_count_integrity();
256
257 nassertv(_depth == _parent->_depth + 1);
258
259 // We can't perform too much checking on the under_flags, since we don't
260 // know which bits should have been added for this node. We'll verify
261 // that at least we didn't accidentally take some bits away.
262 nassertv((_under_flags & _parent->_under_flags) == _parent->_under_flags);
263
264 // Make sure we're mentioned in our parent's children list.
265 EggGroupNode::iterator ci;
266 ci = find(_parent->begin(), _parent->end(), this);
267 nassertv(ci != _parent->end());
268
269 // Now recurse up our parent.
270 _parent->test_under_integrity();
271 }
272}

297void EggNode::
298update_under(int depth_offset) {
299 int depth;
300 if (_parent == nullptr) {
301 depth = 0;
302 _under_flags = 0;
303 _vertex_frame = nullptr;
304 _node_frame = nullptr;
305 _vertex_frame_inv = nullptr;
306 _node_frame_inv = nullptr;
307 _vertex_to_node = nullptr;
308 _node_to_vertex = nullptr;
309 } else {
310 _parent->test_ref_count_integrity();
311 depth = _parent->_depth + 1;
312 _under_flags = _parent->_under_flags;
313 _vertex_frame = _parent->_vertex_frame;
314 _node_frame = _parent->_node_frame;
315 _vertex_frame_inv = _parent->_vertex_frame_inv;
316 _node_frame_inv = _parent->_node_frame_inv;
317 _vertex_to_node = _parent->_vertex_to_node;
318 _node_to_vertex = _parent->_node_to_vertex;
319 }
320
321 if (depth - _depth != depth_offset) {
322 egg_cat.error() << "Cycle in egg graph or invalid egg pointer!\n";
323 return;
324 }
325 _depth = depth;
326
327 adjust_under();
328}
