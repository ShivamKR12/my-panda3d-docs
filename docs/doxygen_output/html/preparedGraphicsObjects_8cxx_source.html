905void PreparedGraphicsObjects::
906release_vertex_buffer(VertexBufferContext *vbc) {
907 ReMutexHolder holder(_lock);
908
909 vbc->get_data()->clear_prepared(this);
910
911 size_t data_size_bytes = vbc->get_data()->get_data_size_bytes();
912 GeomEnums::UsageHint usage_hint = vbc->get_data()->get_usage_hint();
913
914 // We have to set the Data pointer to NULL at this point, since the Data
915 // itself might destruct at any time after it has been released.
916 vbc->_object = nullptr;

918 bool removed = (_prepared_vertex_buffers.erase(vbc) != 0);
919 nassertv(removed);
920
921 if (_support_released_buffer_cache) {
922 cache_unprepared_buffer(vbc, data_size_bytes, usage_hint,
923 _vertex_buffer_cache,
924 _vertex_buffer_cache_lru, _vertex_buffer_cache_size,
925 released_vbuffer_cache_size,
926 _released_vertex_buffers);
927 } else {
928 _released_vertex_buffers.push_back(vbc);
929 }

938release_all_vertex_buffers() {
939 ReMutexHolder holder(_lock);
940
941 int num_vertex_buffers = (int)_prepared_vertex_buffers.size() + (int)_enqueued_vertex_buffers.size();
942
943 for (BufferContext *bc : _prepared_vertex_buffers) {
944 VertexBufferContext *vbc = (VertexBufferContext *)bc;
945 vbc->get_data()->clear_prepared(this);
946 vbc->_object = nullptr;
947
948 _released_vertex_buffers.push_back(vbc);
949 }
950
951 _prepared_vertex_buffers.clear();
952 _enqueued_vertex_buffers.clear();
953
954 // Also clear the cache of recently-unprepared vertex buffers.
955 BufferCache::iterator bci;
956 for (bci = _vertex_buffer_cache.begin();
957 bci != _vertex_buffer_cache.end();
958 ++bci) {
959 BufferList &buffer_list = (*bci).second;

961 for (BufferContext *bc : buffer_list) {
962 VertexBufferContext *vbc = (VertexBufferContext *)bc;
963 _released_vertex_buffers.push_back(vbc);

1416bool PreparedGraphicsObjects::EnqueuedObject::
1417cancel() {
1418 PreparedGraphicsObjects *pgo = _pgo;
1419 if (_object == nullptr || pgo == nullptr) {
1420 nassertr(done(), false);
1421 return false;
1422 }
1423
1424 // We don't upcall here, because the dequeue function will end up calling
1425 // notify_removed().
1426 _result = nullptr;
1427 _pgo = nullptr;
1428
1429 if (_object->is_of_type(Texture::get_class_type())) {
1430 return pgo->dequeue_texture((Texture *)_object.p());
1431
1432 } else if (_object->is_of_type(Geom::get_class_type())) {
1433 return pgo->dequeue_geom((Geom *)_object.p());
1434
1435 } else if (_object->is_of_type(Shader::get_class_type())) {
1436 return pgo->dequeue_shader((Shader *)_object.p());
1437
1438 } else if (_object->is_of_type(GeomVertexArrayData::get_class_type())) {
1439 return pgo->dequeue_vertex_buffer((GeomVertexArrayData *)_object.p());
1440
1441 } else if (_object->is_of_type(GeomPrimitive::get_class_type())) {
1442 return pgo->dequeue_index_buffer((GeomPrimitive *)_object.p());
1443
1444 } else if (_object->is_of_type(ShaderBuffer::get_class_type())) {
1445 return pgo->dequeue_shader_buffer((ShaderBuffer *)_object.p());
1446 }
1447 return false;

1458void PreparedGraphicsObjects::
1459begin_frame(GraphicsStateGuardianBase *gsg, Thread *current_thread) {
1460 ReMutexHolder holder(_lock, current_thread);
1461
1462 // First, release all the textures, geoms, and buffers awaiting release.
1463 if (!_released_textures.empty()) {
1464 gsg->release_textures(_released_textures);
1465 _released_textures.clear();
1466 }
1467
1468 if (!_released_samplers.empty()) {
1469 for (SamplerContext *sc : _released_samplers) {
1470 gsg->release_sampler(sc);
1471 }
1472 _released_samplers.clear();
1473 }
1474
1475 if (!_released_geoms.empty()) {
1476 for (GeomContext *gc : _released_geoms) {
1477 gsg->release_geom(gc);
1478 }
1479 _released_geoms.clear();
1480 }
1481
1482 if (!_released_shaders.empty()) {
1483 for (ShaderContext *sc : _released_shaders) {
1484 gsg->release_shader(sc);
1485 }
1486 _released_shaders.clear();
1487 }
1488
1489 if (!_released_vertex_buffers.empty()) {
1490 gsg->release_vertex_buffers(_released_vertex_buffers);
1491 _released_vertex_buffers.clear();
1492 }
1493
1494 if (!_released_index_buffers.empty()) {
1495 gsg->release_index_buffers(_released_index_buffers);
1496 _released_index_buffers.clear();
1497 }
1498
1499 if (!_released_shader_buffers.empty()) {
1500 gsg->release_shader_buffers(_released_shader_buffers);
1501 _released_shader_buffers.clear();
1502 }
1503
1504 // Reset the residency trackers.
1505 _texture_residency.begin_frame(current_thread);
1506 _vbuffer_residency.begin_frame(current_thread);
1507 _ibuffer_residency.begin_frame(current_thread);
1508 _sbuffer_residency.begin_frame(current_thread);
1509
1510 // Now prepare all the textures, geoms, and buffers awaiting preparation.
1511 EnqueuedTextures::iterator qti;
1512 for (qti = _enqueued_textures.begin();
1513 qti != _enqueued_textures.end();
1514 ++qti) {
1515 Texture *tex = qti->first;
1516 TextureContext *tc = tex->prepare_now(this, gsg);
1517 if (tc != nullptr) {
1518 if (tex->get_num_async_transfer_buffers() == 0) {
1519 gsg->update_texture(tc, true);
1520 if (qti->second != nullptr) {
1521 qti->second->set_result(tc);
1522 }
1523 } else {
1524 // Async update
1525 CompletionToken token;
1526 if (qti->second != nullptr) {
1527 token = [tc, fut = std::move(qti->second)] (bool success) {
1528 if (success) {
1529 fut->set_result(tc);
1530 } else {
1531 fut->notify_removed();
1532 }
1533 };
1534 }
1535 gsg->update_texture(tc, false, std::move(token));
1536 }
1537 }
1538 }
1539
1540 _enqueued_textures.clear();
1541
1542 EnqueuedSamplers::iterator qsmi;
1543 for (qsmi = _enqueued_samplers.begin();
1544 qsmi != _enqueued_samplers.end();
1545 ++qsmi) {
1546 const SamplerState &sampler = (*qsmi);
1547 sampler.prepare_now(this, gsg);
1548 }
1549
1550 _enqueued_samplers.clear();
1551
1552 EnqueuedGeoms::iterator qgi;
1553 for (qgi = _enqueued_geoms.begin();
1554 qgi != _enqueued_geoms.end();
1555 ++qgi) {
1556 Geom *geom = (*qgi);
1557 geom->prepare_now(this, gsg);
1558 }
1559
1560 _enqueued_geoms.clear();
1561
1562 EnqueuedShaders::iterator qsi;
1563 for (qsi = _enqueued_shaders.begin();
1564 qsi != _enqueued_shaders.end();
1565 ++qsi) {
1566 Shader *shader = qsi->first;
1567 ShaderContext *sc = shader->prepare_now(this, gsg);
1568 if (qsi->second != nullptr) {
1569 qsi->second->set_result(sc);
1570 }
1571 }
1572
1573 _enqueued_shaders.clear();
1574
1575 EnqueuedVertexBuffers::iterator qvbi;
1576 for (qvbi = _enqueued_vertex_buffers.begin();
1577 qvbi != _enqueued_vertex_buffers.end();
1578 ++qvbi) {
1579 GeomVertexArrayData *data = (*qvbi);
1580 data->prepare_now(this, gsg);
1581 }
1582
1583 _enqueued_vertex_buffers.clear();
1584
1585 EnqueuedIndexBuffers::iterator qibi;
1586 for (qibi = _enqueued_index_buffers.begin();
1587 qibi != _enqueued_index_buffers.end();
1588 ++qibi) {
1589 GeomPrimitive *data = (*qibi);
1590 // We need this check because the actual index data may not actually have
1591 // propagated to the draw thread yet.
1592 if (data->is_indexed()) {
1593 data->prepare_now(this, gsg);
1594 }
1595 }
1596
1597 _enqueued_index_buffers.clear();
1598
1599 for (ShaderBuffer *buffer : _enqueued_shader_buffers) {
1600 buffer->prepare_now(this, gsg);
1601 }
1602
1603 _enqueued_shader_buffers.clear();
1604}

1638void PreparedGraphicsObjects::
1639cache_unprepared_buffer(BufferContext *buffer, size_t data_size_bytes,
1640 GeomEnums::UsageHint usage_hint,
1641 PreparedGraphicsObjects::BufferCache &buffer_cache,
1642 PreparedGraphicsObjects::BufferCacheLRU &buffer_cache_lru,
1643 size_t &buffer_cache_size,
1644 int released_buffer_cache_size,
1645 pvector<BufferContext *> &released_buffers) {
1646 BufferCacheKey key;
1647 key._data_size_bytes = data_size_bytes;
1648 key._usage_hint = usage_hint;
1649
1650 buffer_cache[key].push_back(buffer);
1651 buffer_cache_size += data_size_bytes;
1652
1653 // Move the key to the head of the LRU.
1654 BufferCacheLRU::iterator li =
1655 find(buffer_cache_lru.begin(), buffer_cache_lru.end(), key);
1656 if (li != buffer_cache_lru.end()) {
1657 buffer_cache_lru.erase(li);
1658 }
1659 buffer_cache_lru.insert(buffer_cache_lru.begin(), key);
1660
1661 // Now release not-recently-used buffers until we fit within the constrained
1662 // size.
1663 while ((int)buffer_cache_size > released_buffer_cache_size) {
1664 nassertv(!buffer_cache_lru.empty());
1665 const BufferCacheKey &release_key = *buffer_cache_lru.rbegin();
1666 BufferList &buffer_list = buffer_cache[release_key];
1667 while (!buffer_list.empty() &&
1668 (int)buffer_cache_size > released_buffer_cache_size) {
1669 BufferContext *released_buffer = buffer_list.back();
1670 buffer_list.pop_back();
1671 released_buffers.push_back(released_buffer);
1672 buffer_cache_size -= release_key._data_size_bytes;
1673 }
1674
1675 if (buffer_list.empty()) {
1676 buffer_cache.erase(release_key);
1677 buffer_cache_lru.pop_back();
1678 }
1679 }
1680}

1686BufferContext *PreparedGraphicsObjects::
1687get_cached_buffer(size_t data_size_bytes, GeomEnums::UsageHint usage_hint,
1688 PreparedGraphicsObjects::BufferCache &buffer_cache,
1689 PreparedGraphicsObjects::BufferCacheLRU &buffer_cache_lru,
1690 size_t &buffer_cache_size) {

1711 }
1712 }
1713
1714 buffer_cache_size -= data_size_bytes;
1715 return buffer;
