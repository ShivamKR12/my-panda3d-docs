140PNMFileTypeIMG::Reader::
141Reader(PNMFileType *type, istream *file, bool owns_file, string magic_number) :
142 PNMReader(type, file, owns_file)
143{
144 if (img_header_type == IHT_long) {
145 if (!read_magic_number(_file, magic_number, 8)) {
146 // Although raw IMG files have no magic number, they may have a pair of
147 // ushorts or ulongs at the beginning to indicate the file size.
148 if (pnmimage_img_cat.is_debug()) {
149 pnmimage_img_cat.debug()
150 << "IMG image file appears to be empty.\n";
151 }
152 _is_valid = false;
153 return;
154 }
155
156 _x_size =
157 ((unsigned char)magic_number[0] << 24) |
158 ((unsigned char)magic_number[1] << 16) |
159 ((unsigned char)magic_number[2] << 8) |
160 ((unsigned char)magic_number[3]);
161
162 _y_size =
163 ((unsigned char)magic_number[4] << 24) |
164 ((unsigned char)magic_number[5] << 16) |
165 ((unsigned char)magic_number[6] << 8) |
166 ((unsigned char)magic_number[7]);
167
168 } else if (img_header_type == IHT_short) {
169 if (!read_magic_number(_file, magic_number, 4)) {
170 if (pnmimage_img_cat.is_debug()) {
171 pnmimage_img_cat.debug()
172 << "IMG image file appears to be empty.\n";
173 }
174 _is_valid = false;
175 return;
176 }
177
178 _x_size =
179 ((unsigned char)magic_number[0] << 8) |
180 ((unsigned char)magic_number[1]);
181
182 _y_size =
183 ((unsigned char)magic_number[2] << 8) |
184 ((unsigned char)magic_number[3]);
185
186 } else {
187 _x_size = img_size[0];
188 _y_size = img_size[1];
189 }
190
191 if (_x_size == 0 || _y_size == 0 ||
192 _x_size > INSANE_SIZE || _y_size > INSANE_SIZE) {
193 _is_valid = false;
194 if (img_header_type == IHT_none) {
195 pnmimage_img_cat.error()
196 << "Must specify img-xsize and img-ysize to load headerless raw files.\n";
197 } else {
198 pnmimage_img_cat.debug()
199 << "IMG file does not have a valid xsize,ysize header.\n";
200 }
201 return;
202 }
203
204 _maxval = 255;
205 _num_channels = 3;
206
207 if (pnmimage_img_cat.is_debug()) {
208 pnmimage_img_cat.debug()
209 << "Reading IMG " << *this << "\n";
210 }
211}

298bool PNMFileTypeIMG::Writer::
299write_row(xel *row_data, xelval *) {
300 int x;
301 for (x = 0; x < _x_size; x++) {
302 write_uchar_IMG(_file, (unsigned char)(255*PPM_GETR(row_data[x])/_maxval));
303 write_uchar_IMG(_file, (unsigned char)(255*PPM_GETG(row_data[x])/_maxval));
