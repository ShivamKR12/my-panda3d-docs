54bool CullableObject::
55munge_geom(GraphicsStateGuardianBase *gsg, GeomMunger *munger,
56 const CullTraverser *traverser, bool force) {
57 nassertr(munger != nullptr, false);
58
59 Thread *current_thread = traverser->get_current_thread();
60 PStatTimer timer(_munge_pcollector, current_thread);
61 if (_geom != nullptr) {
62 GraphicsStateGuardianBase *gsg = traverser->get_gsg();
63 int gsg_bits = gsg->get_supported_geom_rendering();
64 if (!hardware_point_sprites) {
65 // If support for hardware point sprites or perspective-scaled points is
66 // disabled, we don't allow the GSG to tell us it supports them.
67 gsg_bits &= ~(Geom::GR_point_perspective | Geom::GR_point_sprite);
68 }
69 if (!hardware_points) {
70 // If hardware-points is off, we don't allow any kind of point
71 // rendering, except plain old one-pixel points;
72 gsg_bits &= ~(Geom::GR_point_bits & ~Geom::GR_point);
73 }
74 int geom_rendering;
75 int unsupported_bits;
76
77 {
78 GeomPipelineReader geom_reader(_geom, current_thread);
79 _munged_data = geom_reader.get_vertex_data();
80
81#ifdef _DEBUG
82 {
83 GeomVertexDataPipelineReader data_reader(_munged_data, current_thread);
84 data_reader.check_array_readers();
85 nassertr(geom_reader.check_valid(&data_reader), false);
86 }
87#endif // _DEBUG
88
89 geom_rendering = geom_reader.get_geom_rendering();
90 geom_rendering = _state->get_geom_rendering(geom_rendering);
91 geom_rendering = _internal_transform->get_geom_rendering(geom_rendering);
92 unsupported_bits = geom_rendering & ~gsg_bits;
93
94 if (unsupported_bits & Geom::GR_per_point_size) {
95 // If we have a shader that processes the point size, we can assume it
96 // does the right thing.
97 const ShaderAttrib *sattr;
98 if (_state->get_attrib(sattr) && sattr->get_flag(ShaderAttrib::F_shader_point_size)) {
99 unsupported_bits &= ~Geom::GR_per_point_size;
100 }
101 }
102
103 if (geom_rendering & Geom::GR_point_bits) {
104 if (geom_reader.get_primitive_type() != Geom::PT_points) {
105 if (singular_points ||
106 (unsupported_bits & Geom::GR_render_mode_point) != 0) {
107 // Isolate the points so there's no unneeded overlap.
108 _geom = _geom->make_points();
109 }
110 }
111 }
112 if (unsupported_bits & Geom::GR_render_mode_wireframe) {
113 if (geom_reader.get_primitive_type() != Geom::PT_lines) {
114 _geom = _geom->make_lines();
115 }
116 }
117 }
118
119 if ((unsupported_bits & Geom::GR_point_bits) != 0) {
120 // The GSG doesn't support rendering these fancy points directly; we
121 // have to render them in software instead. Munge them into quads.
122 // This will replace the _geom and _munged_data, and might also replace
123 // _state.
124 if (pgraph_cat.is_spam()) {
125 pgraph_cat.spam()
126 << "munge_points_to_quads() for geometry with bits: "
127 << std::hex << geom_rendering << ", unsupported: "
128 << (unsupported_bits & Geom::GR_point_bits) << std::dec << "\n";
129 }
130 if (!munge_points_to_quads(traverser, force)) {
131 return false;
132 }
133 }
134
135 // Now invoke the munger to ensure the resulting geometry is in a GSG-
136 // friendly form.
137 {
138 PStatTimer timer(_munge_geom_pcollector, current_thread);
139 if (!munger->munge_geom(_geom, _munged_data, force, current_thread)) {
140 return false;
141 }
142 }
143
144 // If we have prepared it for skinning via the shader generator, mark a
145 // flag on the state so that the shader generator will do this. We should
146 // probably find a cleaner way to do this.
147 const ShaderAttrib *sattr;
148 if (_state->get_attrib(sattr) && sattr->auto_shader()) {
149 GeomVertexDataPipelineReader data_reader(_munged_data, current_thread);
150 if (data_reader.get_format()->get_animation().get_animation_type() == Geom::AT_hardware) {
151 static CPT(RenderState) state = RenderState::make(
152 DCAST(ShaderAttrib, ShaderAttrib::make())->set_flag(ShaderAttrib::F_hardware_skinning, true));
153 _state = _state->compose(state);
154 }
155
156 gsg->ensure_generated_shader(_state);
157 } else {
158 // We may need to munge the state for the fixed-function pipeline.
159 StateMunger *state_munger = (StateMunger *)munger;
160 if (state_munger->should_munge_state()) {
161 _state = state_munger->munge_state(_state);
162 }
163 }
164
165 // If there is any animation left in the vertex data after it has been
166 // munged--that is, we couldn't arrange to handle the animation in
167 // hardware--then we have to calculate that animation now.
168 bool cpu_animated = false;
169
170 CPT(GeomVertexData) animated_vertices =
171 _munged_data->animate_vertices(force, current_thread);
172 if (animated_vertices != _munged_data) {
173 cpu_animated = true;
174 std::swap(_munged_data, animated_vertices);
175 }
176
177 if (sattr != nullptr) {
178 if (_instances != nullptr &&
179 sattr->get_flag(ShaderAttrib::F_hardware_instancing)) {
180 // We are under an InstancedNode, and the shader implements hardware.
181 // Munge the instance list into the vertex data.
182 munge_instances(current_thread);
183 _num_instances = _instances->size();
184 _instances = nullptr;
185 } else {
186 // No, use the instance count from the ShaderAttrib.
187 int count = sattr->get_instance_count();
188 _num_instances = (count > 0) ? (size_t)count : 1;
189 }
190 } else {
191 _num_instances = 1;
192 }
193
194#ifndef NDEBUG
195 if (show_vertex_animation) {
196 GeomVertexDataPipelineReader data_reader(_munged_data, current_thread);
197 bool hardware_animated = (data_reader.get_format()->get_animation().get_animation_type() == Geom::AT_hardware);
198 if (cpu_animated || hardware_animated) {
199 // These vertices were animated, so flash them red or blue.
200 static const double flash_rate = 1.0; // 1 state change per second
201 int cycle = (int)(ClockObject::get_global_clock()->get_frame_time() * flash_rate);
202 if ((cycle & 1) == 0) {
203 _state = cpu_animated ? get_flash_cpu_state() : get_flash_hardware_state();
204 }
205 }
206 }
207#endif
208 }
209
210 return true;
211}

248munge_points_to_quads(const CullTraverser *traverser, bool force) {
249 Thread *current_thread = traverser->get_current_thread();
250
251 // Better get the animated vertices, in case we're showing sprites on an
252 // animated model for some reason.
253 CPT(GeomVertexData) source_data =
254 _munged_data->animate_vertices(force, current_thread);
255
256 if (!force && !source_data->request_resident()) {
257 return false;
258 }
259
260 GeomVertexDataPipelineReader reader(source_data, current_thread);
261 reader.check_array_readers();
262
263 PStatTimer timer(_munge_sprites_pcollector, current_thread);
264 _sw_sprites_pcollector.add_level(reader.get_num_rows());
265
266 GraphicsStateGuardianBase *gsg = traverser->get_gsg();
267
268 GeomVertexReader vertex(&reader, InternalName::get_vertex());
269 GeomVertexReader normal(&reader, InternalName::get_normal());
270 GeomVertexReader rotate(&reader, InternalName::get_rotate());
271 GeomVertexReader size(&reader, InternalName::get_size());
272 GeomVertexReader aspect_ratio(&reader, InternalName::get_aspect_ratio());
273
274 bool has_normal = (normal.has_column());
275 bool has_color = (reader.has_column(InternalName::get_color()));
276 bool has_texcoord = (reader.has_column(InternalName::get_texcoord()));
277 bool has_rotate = (rotate.has_column());
278 bool has_size = (size.has_column());
279 bool has_aspect_ratio = (aspect_ratio.has_column());

281 bool sprite_texcoord = false;
282 const TexGenAttrib *tex_gen = DCAST(TexGenAttrib, _state->get_attrib(TexGenAttrib::get_class_slot()));
283 if (tex_gen != nullptr) {
284 if (tex_gen->get_mode(TextureStage::get_default()) == TexGenAttrib::M_point_sprite) {
285 sprite_texcoord = true;
286
287 // Turn off the TexGenAttrib, since we don't want it now.
288 _state = _state->set_attrib(tex_gen->remove_stage(TextureStage::get_default()));
289 }
290 }
291
292 PN_stdfloat point_size = 1;
293 bool perspective = false;
294 const RenderModeAttrib *render_mode = DCAST(RenderModeAttrib, _state->get_attrib(RenderModeAttrib::get_class_slot()));
295 if (render_mode != nullptr) {
296 point_size = render_mode->get_thickness();
297 perspective = render_mode->get_perspective();
298
299 if (render_mode->get_mode() != RenderModeAttrib::M_filled_flat) {
300 // Render the new polygons with M_filled_flat, for a slight performance
301 // advantage when software rendering.
302 _state = _state->set_attrib(RenderModeAttrib::make(RenderModeAttrib::M_filled_flat));
303 }
304 }
305
306 // Get the vertex format of the newly created geometry.
307 CPT(GeomVertexFormat) new_format;
308
309 {
310 LightMutexHolder holder(_format_lock);
311 SourceFormat sformat(reader.get_format(), sprite_texcoord);
312 FormatMap::iterator fmi = _format_map.find(sformat);
313 if (fmi != _format_map.end()) {
314 new_format = (*fmi).second;
315
316 } else {
317 // We have to construct the format now.
318 PT(GeomVertexArrayFormat) new_array_format;
319 if (sformat._retransform_sprites) {
320 // With retransform_sprites in effect, we will be sending ordinary 3-D
321 // points to the graphics API.
322 new_array_format =
323 new GeomVertexArrayFormat(InternalName::get_vertex(), 3,
324 Geom::NT_stdfloat,
325 Geom::C_point);
326 } else {
327 // Without retransform_sprites, we will be sending 4-component clip-
328 // space points.
329 new_array_format =
330 new GeomVertexArrayFormat(InternalName::get_vertex(), 4,
331 Geom::NT_stdfloat,
332 Geom::C_clip_point);
333 }
334 if (has_normal) {
335 const GeomVertexColumn *c = reader.get_format()->get_normal_column();
336 new_array_format->add_column
337 (InternalName::get_normal(), c->get_num_components(),
338 c->get_numeric_type(), c->get_contents());
339 }
340 if (has_color) {
341 const GeomVertexColumn *c = reader.get_format()->get_color_column();
342 new_array_format->add_column
343 (InternalName::get_color(), c->get_num_components(),
344 c->get_numeric_type(), c->get_contents());
345 }
346 if (sprite_texcoord) {
347 new_array_format->add_column
348 (InternalName::get_texcoord(), 2,
349 Geom::NT_stdfloat,
350 Geom::C_texcoord);
351
352 } else if (has_texcoord) {
353 const GeomVertexColumn *c = reader.get_format()->get_column(InternalName::get_texcoord());
354 new_array_format->add_column
355 (InternalName::get_texcoord(), c->get_num_components(),
356 c->get_numeric_type(), c->get_contents());
357 }
358
359 // Go through the other columns and copy them from the original.
360 for (size_t ai = 0; ai < sformat._format->get_num_arrays(); ++ai) {
361 const GeomVertexArrayFormat *aformat = sformat._format->get_array(ai);
362
363 for (size_t ci = 0; ci < (size_t)aformat->get_num_columns(); ++ci) {
364 const GeomVertexColumn *column = aformat->get_column(ci);
365 const InternalName *name = column->get_name();
366 if (name != InternalName::get_vertex() &&
367 name != InternalName::get_normal() &&
368 name != InternalName::get_color() &&
369 name != InternalName::get_texcoord() &&
370 name != InternalName::get_rotate() &&
371 name != InternalName::get_size() &&
372 name != InternalName::get_aspect_ratio()) {
373
374 new_array_format->add_column(name,
375 column->get_num_components(), column->get_numeric_type(),
376 column->get_contents());
377 }
378 }
379 }
380
381 new_format = GeomVertexFormat::register_format(new_array_format);
382 _format_map[sformat] = new_format;
383 }
384 }
385
386 CoordinateSystem internal_cs = gsg->get_internal_coordinate_system();
387 LMatrix4 internal = _internal_transform->get_mat();
388 PN_stdfloat scale = _internal_transform->get_scale()[1];
389
390 SceneSetup *scene = traverser->get_scene();
391 const Lens *lens = scene->get_lens();
392 LMatrix4 projection =
393 LMatrix4::convert_mat(internal_cs, lens->get_coordinate_system()) *
394 lens->get_projection_mat();
395
396 int viewport_width = scene->get_viewport_width();
397 int viewport_height = scene->get_viewport_height();
398
399 // We need a standard projection matrix, in a known coordinate system, to
400 // compute the perspective height.
401 LMatrix4 height_projection;
402 if (perspective) {
403 height_projection =
404 LMatrix4::convert_mat(CS_yup_right, lens->get_coordinate_system()) *
405 lens->get_projection_mat();
406 }
407
408 LMatrix4 render_transform = internal * projection;
409 LMatrix4 inv_render_transform;
410 inv_render_transform.invert_from(render_transform);
411
412 // Now convert all of the vertices in the GeomVertexData to quads. We
413 // always convert all the vertices, assuming all the vertices are referenced
414 // by GeomPrimitives, because we want to optimize for the most common case.
415 int orig_verts = reader.get_num_rows();
416 int new_verts = 4 * orig_verts; // each vertex becomes four.
417
418 PT(GeomVertexData) new_data = new GeomVertexData
419 (source_data->get_name(), new_format, Geom::UH_stream);
420 new_data->unclean_set_num_rows(new_verts);
421
422 GeomVertexWriter new_vertex(new_data, InternalName::get_vertex());
423 GeomVertexWriter new_normal(new_data, InternalName::get_normal());
424 GeomVertexWriter new_texcoord(new_data, InternalName::get_texcoord());
425
426 nassertr(new_vertex.has_column(), false);
427 unsigned char *write_ptr = new_vertex.get_array_handle()->get_write_pointer();
428
429 // Collect all other columns that we just need to copy the data of.
430 struct CopyOp {
431 unsigned char *_to_pointer;
432 const unsigned char *_from_pointer;
433 size_t _num_bytes;
434 size_t _from_stride;
435 };
436 pvector<CopyOp> copies;
437
438 const GeomVertexArrayFormat *aformat = new_format->get_array(0);
439 for (size_t ci = 0; ci < (size_t)aformat->get_num_columns(); ++ci) {
440 const GeomVertexColumn *column = aformat->get_column(ci);
441 const InternalName *name = column->get_name();
442 if (name != InternalName::get_vertex() &&
443 (retransform_sprites || name != InternalName::get_normal()) &&
444 (!sprite_texcoord || name != InternalName::get_texcoord())) {
445
446 int source_array;
447 const GeomVertexColumn *source_column;
448 if (reader.get_format()->get_array_info(name, source_array, source_column)) {
449 CopyOp copy;
450 copy._to_pointer = write_ptr + (size_t)column->get_start();
451 copy._from_pointer = reader.get_array_reader(source_array)->get_read_pointer(true) + (size_t)source_column->get_start();
452 copy._num_bytes = (size_t)column->get_total_bytes();
453 copy._from_stride = reader.get_format()->get_array(source_array)->get_stride();
454
455 if (!copies.empty() &&
456 (copy._to_pointer == copies.back()._to_pointer + copies.back()._num_bytes) &&
457 (copy._from_pointer == copies.back()._from_pointer + copies.back()._num_bytes)) {
458 // Merge with previous.
459 copies.back()._num_bytes += copy._num_bytes;
460 } else {
461 copies.push_back(copy);
462 }
463 }
464 }
465 }
466 size_t to_stride = aformat->get_stride();
467
468 // We'll keep an array of all of the points' eye-space coordinates, and
469 // their distance from the camera, so we can sort the points for each
470 // primitive, below.
471 PointData *points;
472 {
473 PStatTimer t2(_munge_sprites_verts_pcollector, current_thread);
474 points = (PointData *)alloca(orig_verts * sizeof(PointData));
475 int vi = 0;
476 while (!vertex.is_at_end()) {
477 // Get the point in eye-space coordinates.
478 LPoint3 eye = internal.xform_point(vertex.get_data3());
479 PN_stdfloat dist = gsg->compute_distance_to(eye);
480 points[vi]._dist = dist;
481
482 // The point in clip coordinates.
483 LPoint4 p4 = LPoint4(eye[0], eye[1], eye[2], 1.0f) * projection;
484
485 if (has_size) {
486 point_size = size.get_data1();
487 }
488
489 PN_stdfloat scale_y = point_size;
490 if (perspective) {
491 // Perspective-sized points. Here point_size is the point's height in
492 // 3-d units. To arrange that, we need to figure out the appropriate
493 // scaling factor based on the current viewport and projection matrix.
494 LVector3 height(0.0f, point_size * scale, scale);
495 height = height * height_projection;
496 scale_y = height[1] * viewport_height;
497
498 // We should then divide the radius by the distance from the camera
499 // plane, to emulate the glPointParameters() behavior.
500 if (!lens->is_orthographic()) {
501 scale_y /= dist;
502 }
503 }
504
505 // Also factor in the homogeneous scale for being in clip coordinates
506 // still.
507 scale_y *= p4[3];
508
509 PN_stdfloat scale_x = scale_y;
510 if (has_aspect_ratio) {
511 scale_x *= aspect_ratio.get_data1();
512 }
513
514 // Define the first two corners based on the scales in X and Y.
515 LPoint2 c0(scale_x, scale_y);
516 LPoint2 c1(-scale_x, scale_y);
517
518 if (has_rotate) {
519 // If we have a rotate factor, apply it to those two corners.
520 PN_stdfloat r = rotate.get_data1();
521 LMatrix3 mat = LMatrix3::rotate_mat(r);
522 c0 = c0 * mat;
523 c1 = c1 * mat;
524 }
525
526 // Finally, scale the corners in their newly-rotated position, to
527 // compensate for the aspect ratio of the viewport.
528 PN_stdfloat rx = 1.0f / viewport_width;
529 PN_stdfloat ry = 1.0f / viewport_height;
530 c0.set(c0[0] * rx, c0[1] * ry);
531 c1.set(c1[0] * rx, c1[1] * ry);
532
533 if (retransform_sprites) {
534 // With retransform_sprites in effect, we must reconvert the resulting
535 // quad back into the original 3-D space.
536 new_vertex.set_data4(inv_render_transform.xform(LPoint4(p4[0] + c0[0], p4[1] + c0[1], p4[2], p4[3])));
537 new_vertex.set_data4(inv_render_transform.xform(LPoint4(p4[0] + c1[0], p4[1] + c1[1], p4[2], p4[3])));
538 new_vertex.set_data4(inv_render_transform.xform(LPoint4(p4[0] - c1[0], p4[1] - c1[1], p4[2], p4[3])));
539 new_vertex.set_data4(inv_render_transform.xform(LPoint4(p4[0] - c0[0], p4[1] - c0[1], p4[2], p4[3])));
540
541 } else {
542 // Without retransform_sprites, we can simply load the clip-space
543 // coordinates.
544 new_vertex.set_data4(p4[0] + c0[0], p4[1] + c0[1], p4[2], p4[3]);
545 new_vertex.set_data4(p4[0] + c1[0], p4[1] + c1[1], p4[2], p4[3]);
546 new_vertex.set_data4(p4[0] - c1[0], p4[1] - c1[1], p4[2], p4[3]);
547 new_vertex.set_data4(p4[0] - c0[0], p4[1] - c0[1], p4[2], p4[3]);
548
549 if (has_normal) {
550 // We need to transform the normals to clip-space too, then.
551 LNormal c = render_transform.xform_vec(normal.get_data3());
552 new_normal.set_data3(c);
553 new_normal.set_data3(c);
554 new_normal.set_data3(c);
555 new_normal.set_data3(c);
556 }
557 }
558 if (sprite_texcoord) {
559 new_texcoord.set_data2(1.0f, 0.0f);
560 new_texcoord.set_data2(0.0f, 0.0f);
561 new_texcoord.set_data2(1.0f, 1.0f);
562 new_texcoord.set_data2(0.0f, 1.0f);
563 }
564
565 // Other columns are simply duplicated for each vertex.
566 for (CopyOp &copy : copies) {
567 memcpy(copy._to_pointer, copy._from_pointer, copy._num_bytes);
568 copy._to_pointer += to_stride;
569 memcpy(copy._to_pointer, copy._from_pointer, copy._num_bytes);
570 copy._to_pointer += to_stride;
571 memcpy(copy._to_pointer, copy._from_pointer, copy._num_bytes);
572 copy._to_pointer += to_stride;
573 memcpy(copy._to_pointer, copy._from_pointer, copy._num_bytes);
574 copy._to_pointer += to_stride;
575 copy._from_pointer += copy._from_stride;
576 }
577
578 ++vi;
579 }
580
581 nassertr(vi == orig_verts, false);
582 nassertr(new_data->get_num_rows() == new_verts, false);
583 }
584
585 // Determine the format we should use to store the indices. Don't choose
586 // NT_uint8, as Direct3D 9 doesn't support it.
587 const GeomVertexArrayFormat *new_prim_format = nullptr;
588 if (new_verts < 0xffff) {
589 new_prim_format = GeomPrimitive::get_index_format(GeomEnums::NT_uint16);
590
591 } else {
592 new_prim_format = GeomPrimitive::get_index_format(GeomEnums::NT_uint32);
593 }
594
595 PT(Geom) new_geom = new Geom(new_data);
596
597 // Replace each primitive in the Geom (it's presumably a GeomPoints
598 // primitive, although it might be some other kind of primitive if we got
599 // here because RenderModeAttrib::M_point is enabled) with a new primitive
600 // that replaces each vertex with a quad of the appropriate scale and
601 // orientation.
602
603 // BUG: if we're rendering polygons in M_point mode with a CullFaceAttrib in
604 // effect, we won't actually apply the CullFaceAttrib but will always render
605 // all of the vertices of the polygons. This is certainly a bug, but a very
606 // minor one; and in order to fix it we'd have to do the face culling
607 // ourselves--not sure if it's worth it.
608
609 {
610 PStatTimer t3(_munge_sprites_prims_pcollector, current_thread);
611 GeomPipelineReader geom_reader(_geom, current_thread);
612 int num_primitives = geom_reader.get_num_primitives();
613 for (int pi = 0; pi < num_primitives; ++pi) {
614 const GeomPrimitive *primitive = geom_reader.get_primitive(pi);
615 if (primitive->get_num_vertices() != 0) {
616 // Extract out the list of vertices referenced by the primitive.
617 int num_vertices = primitive->get_num_vertices();
618 unsigned int *vertices = (unsigned int *)alloca(num_vertices * sizeof(unsigned int));
619 unsigned int *vertices_end = vertices + num_vertices;
620
621 if (primitive->is_indexed()) {
622 // Indexed case.
623 GeomVertexReader index(primitive->get_vertices(), 0, current_thread);
624 for (unsigned int *vi = vertices; vi != vertices_end; ++vi) {
625 unsigned int v = index.get_data1i();
626 nassertr(v < (unsigned int)orig_verts, false);
627 (*vi) = v;
628 }
629 } else {
630 // Nonindexed case.
631 unsigned int first_vertex = primitive->get_first_vertex();
632 for (int i = 0; i < num_vertices; ++i) {
633 unsigned int v = i + first_vertex;
634 nassertr(v < (unsigned int)orig_verts, false);
635 vertices[i] = v;
636 }
637 }
638
639 // Now sort the points in order from back-to-front so they will render
640 // properly with transparency, at least with each other.
641 std::sort(vertices, vertices_end, SortPoints(points));
642
643 // Go through the points, now in sorted order, and generate a pair of
644 // triangles for each one. We generate indexed triangles instead of
645 // two-triangle strips, since this seems to be generally faster on PC
646 // hardware (otherwise, we'd have to nearly double the vertices to
647 // stitch all the little triangle strips together).
648 PT(GeomPrimitive) new_primitive = new GeomTriangles(Geom::UH_stream);
649 int new_prim_verts = 6 * num_vertices; // two triangles per point.
650
651 PT(GeomVertexArrayData) new_index
652 = new GeomVertexArrayData(new_prim_format, GeomEnums::UH_stream);
653 new_index->unclean_set_num_rows(new_prim_verts);
654
655 GeomVertexWriter index(new_index, 0);
656 nassertr(index.has_column(), false);
657 for (unsigned int *vi = vertices; vi != vertices_end; ++vi) {
658 int new_vi = (*vi) * 4;
659 nassertr(index.get_write_row() + 6 <= new_prim_verts, false);
660 index.set_data1i(new_vi);
661 index.set_data1i(new_vi + 1);
662 index.set_data1i(new_vi + 2);
663 index.set_data1i(new_vi + 2);
664 index.set_data1i(new_vi + 1);
665 index.set_data1i(new_vi + 3);
666 }
667 new_primitive->set_vertices(new_index, new_prim_verts);
668
669 int min_vi = primitive->get_min_vertex();
670 int max_vi = primitive->get_max_vertex();
671 new_primitive->set_minmax(min_vi * 4, max_vi * 4 + 3, nullptr, nullptr);
672
673 new_geom->add_primitive(new_primitive);
674 }
675 }
676 }
677
678 _geom = new_geom.p();
679 _munged_data = std::move(new_data);
680
681 return true;
682}
