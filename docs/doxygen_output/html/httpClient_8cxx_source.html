299void HTTPClient::
300set_proxy_spec(const string &proxy_spec) {
301 clear_proxy();
302
303 string trim_proxy_spec = trim(proxy_spec);
304
305 // Tokenize the string based on the semicolons.
306 if (!trim_proxy_spec.empty()) {
307 vector_string proxies;
308 tokenize(trim_proxy_spec, proxies, ";");
309
310 for (vector_string::const_iterator pi = proxies.begin();
311 pi != proxies.end();
312 ++pi) {
313 const string &spec = (*pi);
314
315 // Divide out the scheme and the hostname.
316 string scheme;
317 string proxy;
318 size_t equals = spec.find('=');
319 if (equals == string::npos) {
320 scheme = "";
321 proxy = trim(spec);
322 } else {
323 scheme = trim(spec.substr(0, equals));
324 proxy = trim(spec.substr(equals + 1));
325 }
326
327 if (proxy == "DIRECT" || proxy.empty()) {
328 add_proxy(scheme, URLSpec());
329 } else {
330 add_proxy(scheme, URLSpec(proxy, true));
331 }
332 }
333 }

436void HTTPClient::
437add_proxy(const string &scheme, const URLSpec &proxy) {
438 URLSpec proxy_url(proxy);
439
440 // The scheme is always converted to lowercase.
441 string lc_scheme;
442 lc_scheme.reserve(scheme.length());
443 string::const_iterator si;
444 for (si = scheme.begin(); si != scheme.end(); ++si) {
445 lc_scheme += tolower(*si);
446 }
447
448 // Remove the trailing colon, if there is one.
449 if (!lc_scheme.empty() && lc_scheme[lc_scheme.length() - 1] == ':') {
450 lc_scheme = lc_scheme.substr(0, lc_scheme.length() - 1);
451 }
452
453 if (!proxy_url.empty()) {
454 // Enforce the scheme that we use to communicate to the proxy itself.
455 // This is not the same as lc_scheme, which is the scheme of the requested
456 // connection. Generally, all proxies speak HTTP, except for Socks
457 // proxies.
458
459 if (lc_scheme == "socks") {
460 // Scheme "socks" implies we talk to the proxy via the "socks" scheme,
461 // no matter what scheme the user actually specified.
462 proxy_url.set_scheme("socks");
463
464 } else if (!proxy_url.has_scheme()) {
465 // Otherwise, if the user didn't specify a scheme to talk to the proxy,
466 // the default is "http".
467 proxy_url.set_scheme("http");
468 }
469 }
470
471 _proxies_by_scheme[lc_scheme].push_back(proxy_url);
472}

512void HTTPClient::
513get_proxies_for_url(const URLSpec &url, pvector<URLSpec> &proxies) const {
514 // First, check if the hostname matches any listed in direct_hosts.
515 string hostname = url.get_server();
516
517 // If the hostname is empty, treat it as a special case: we don't match any
518 // of the hostnames listed in direct_hosts (even "*").
519 if (!hostname.empty()) {
520 DirectHosts::const_iterator si;
521 for (si = _direct_hosts.begin(); si != _direct_hosts.end(); ++si) {
522 if ((*si).matches(hostname)) {
523 // It matches, so don't use any proxies.
524 proxies.push_back(URLSpec());
525 return;
526 }
527 }
528 }
529
530 // Build our list of proxies into a temporary vector, so we can pull out
531 // duplicates later.
532 pvector<URLSpec> temp_list;
533
534 // Now choose the appropriate proxy based on the scheme.
535 string scheme = url.get_scheme();
536 bool got_any = false;
537
538 if (!scheme.empty()) {
539 // If we have a scheme, try to match it.
540 if (get_proxies_for_scheme(scheme, temp_list)) {
541 got_any = true;
542 }
543 }
544
545 if (!got_any && (scheme.empty() || url.is_ssl())) {
546 // An empty scheme (or an ssl-style scheme) implies we will need to make a
547 // direct connection, so fallback to a socks-style andor https-style
548 // scheme.
549
550 if (get_proxies_for_scheme("socks", temp_list)) {
551 got_any = true;
552 }
553 if (get_proxies_for_scheme("https", temp_list)) {
554 got_any = true;
555 }
556 }
557
558 if (!got_any) {
559 // If we didn't find our scheme of choice, fall back to the default proxy
560 // type, if we've got one.
561 if (get_proxies_for_scheme("", temp_list)) {
562 got_any = true;
563 }
564 }
565
566 if (_try_all_direct) {
567 // We may try a direct connection if all else fails.
568 temp_list.push_back(URLSpec());
569 }
570
571 // Finally, as a very last resort, fall back to the HTTP proxy.
572 if (!got_any) {
573 get_proxies_for_scheme("http", temp_list);
574 }
575
576 // Now remove duplicate entries as we copy the resulting list out.
577 pvector<URLSpec>::iterator pi;
578 pset<URLSpec> used;
579 for (pi = temp_list.begin(); pi != temp_list.end(); ++pi) {
580 if (used.insert(*pi).second) {
581 // This is a unique one.
582 proxies.push_back(*pi);
583 }
584 }
585}

754void HTTPClient::
755send_cookies(std::ostream &out, const URLSpec &url) {
756 HTTPDate now = HTTPDate::now();
757 bool any_expired = false;
758 bool first_cookie = true;
759
760 Cookies::const_iterator ci;
761 for (ci = _cookies.begin(); ci != _cookies.end(); ++ci) {
762 const HTTPCookie &cookie = (*ci);
763 if (cookie.is_expired(now)) {
764 any_expired = true;
765
766 } else if (cookie.matches_url(url)) {
767 if (first_cookie) {
768 out << "Cookie: ";
769 first_cookie = false;
770 } else {
771 out << "; ";
772 }
773 out << cookie.get_name() << "=" << cookie.get_value();
774 }
775 }
776
777 if (!first_cookie) {
778 out << "\r\n";
779 }
780
781 if (any_expired) {
782 Cookies new_cookies;
783 Cookies::const_iterator ci;
784 for (ci = _cookies.begin(); ci != _cookies.end(); ++ci) {
785 const HTTPCookie &cookie = (*ci);
786 if (!cookie.is_expired(now)) {
787 new_cookies.insert(new_cookies.end(), cookie);
788 }
789 }
790 _cookies.swap(new_cookies);
791 }

802bool HTTPClient::
803load_client_certificate() {
804 if (!_client_certificate_loaded) {
805 _client_certificate_loaded = true;
806
807 if (!_client_certificate_filename.empty()) {
808 _client_certificate_filename.set_text();
809
810 // First, read the complete file into memory.
811 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
812
813 if (!vfs->read_file(_client_certificate_filename,
814 _client_certificate_pem, true)) {
815 // Could not find or read file.
816 downloader_cat.warning()
817 << "Could not read " << _client_certificate_filename << ".\n";
818 return false;
819 }
820 }
821
822 if (!_client_certificate_pem.empty()) {
823 // Create an in-memory BIO to read the "file" from the memory buffer,
824 // and call the low-level routines to read the keys from the BIO.
825 BIO *mbio = BIO_new_mem_buf((void *)_client_certificate_pem.data(),
826 _client_certificate_pem.length());
827
828 ERR_clear_error();
829 _client_certificate_priv =
830 PEM_read_bio_PrivateKey(mbio, nullptr, nullptr,
831 (char *)_client_certificate_passphrase.c_str());
832
833 // Rewind the "file" to the beginning in order to read the public key
834 // (which might appear first in the file).
835 (void)BIO_reset(mbio);
836
837 ERR_clear_error();
838 _client_certificate_pub =
839 PEM_read_bio_X509(mbio, nullptr, nullptr, nullptr);
840
841 BIO_free(mbio);
842
843
844 NotifySeverity sev = NS_debug;
845 string source = "memory";
846 if (!_client_certificate_filename.empty()) {
847 // Only report status to "info" severity if we have read the
848 // certificate from a file. If it came from an in-memory image, a
849 // failure will presumably be handled by whoever set the image.
850 sev = NS_info;
851 source = _client_certificate_filename;
852 }
853
854 if (downloader_cat.is_on(sev)) {
855 if (_client_certificate_priv != nullptr &&
856 _client_certificate_pub != nullptr) {
857 downloader_cat.out(sev)
858 << "Read client certificate from " << source << "\n";
859
860 } else {
861 if (_client_certificate_priv == nullptr) {
862 downloader_cat.out(sev)
863 << "Could not read private key from " << source << "\n";
864 }
865
866 if (_client_certificate_pub == nullptr) {
867 downloader_cat.out(sev)
868 << "Could not read public key from " << source << "\n";
869 }
870 }
871 }
872 }
873 }
874
875 return (_client_certificate_priv != nullptr &&
876 _client_certificate_pub != nullptr);
877}

919add_preapproved_server_certificate_pem(const URLSpec &url, const string &pem) {
920 // Create an in-memory BIO to read the "file" from the memory buffer, and
921 // call the low-level routine to read the cert from the BIO.
922 BIO *mbio = BIO_new_mem_buf((void *)pem.data(), pem.length());
923
924 ERR_clear_error();
925 X509 *cert = PEM_read_bio_X509(mbio, nullptr, nullptr, nullptr);

928 if (cert == nullptr) {

934 string server_and_port = url.get_server_and_port();
935 PreapprovedServerCerts::iterator psci =

1293string HTTPClient::
1294select_username(const URLSpec &url, bool is_proxy, const string &realm) const {
1295 string username;
1296
1297 // Look in several places in order to find the matching username.
1298
1299 // Fist, if there's a username on the URL, that always wins (except when we
1300 // are looking for a proxy username).
1301 if (url.has_username() && !is_proxy) {
1302 username = url.get_username();
1303 }
1304
1305 // Otherwise, start looking on the HTTPClient.
1306 if (is_proxy) {
1307 if (username.empty()) {
1308 // Try the *proxyrealm.
1309 username = get_username("*proxy", realm);
1310 }
1311 if (username.empty()) {
1312 // Then, try *proxyany realm.
1313 username = get_username("*proxy", string());
1314 }
1315 }
1316 if (username.empty()) {
1317 // Try the specific serverrealm.
1318 username = get_username(url.get_server(), realm);
1319 }
1320 if (username.empty()) {
1321 // Then, try the specific serverany realm.
1322 username = get_username(url.get_server(), string());
1323 }
1324 if (username.empty()) {
1325 // Then, try any server with this realm.
1326 username = get_username(string(), realm);
1327 }
1328 if (username.empty()) {
1329 // Then, take the general password.
1330 username = get_username(string(), string());
1331 }
1332
1333 return username;

1340HTTPAuthorization *HTTPClient::
1341select_auth(const URLSpec &url, bool is_proxy, const string &last_realm) {
1342 Domains &domains = is_proxy ? _proxy_domains : _www_domains;
1343 string canon = HTTPAuthorization::get_canonical_url(url).get_url();
1344
1345 // Look for the longest domain string that is a prefix of our canonical URL.
1346 // We have to make a linear scan through the list.
1347 Domains::const_iterator best_di = domains.end();
1348 size_t longest_length = 0;
1349 Domains::const_iterator di;
1350 for (di = domains.begin(); di != domains.end(); ++di) {
1351 const string &domain = (*di).first;
1352 size_t length = domain.length();
1353 if (domain == canon.substr(0, length)) {
1354 // This domain string matches. Is it the longest?
1355 if (length > longest_length) {
1356 best_di = di;
1357 longest_length = length;
1358 }
1359 }
1360 }
1361
1362 if (best_di != domains.end()) {
1363 // Ok, we found a matching domain. Use it.
1364 if (downloader_cat.is_spam()) {
1365 downloader_cat.spam()
1366 << "Choosing domain " << (*best_di).first << " for " << url << "\n";
1367 }
1368 const Realms &realms = (*best_di).second._realms;
1369 // First, try our last realm.
1370 Realms::const_iterator ri;
1371 ri = realms.find(last_realm);
1372 if (ri != realms.end()) {
1373 return (*ri).second;
1374 }
1375
1376 if (!realms.empty()) {
1377 // Oh well, just return the first realm.
1378 return (*realms.begin()).second;
1379 }
1380 }
1381
1382 // No matching domains.
1383 return nullptr;
1384}

1456X509_NAME *HTTPClient::
1457parse_x509_name(const string &source) {
1458 X509_NAME *result = nullptr;
1459
1460 result = X509_NAME_new();
1461 bool added_any = false;
1462
1463 string::const_iterator si;
1464 si = source.begin();
1465 while (si != source.end()) {
1466 if ((*si) == '/') {
1467 // Skip a slash delimiter.
1468 ++si;
1469 } else {
1470 string type;
1471 while (si != source.end() && (*si) != '=' && (*si) != '/') {
1472 if ((*si) == '\\') {
1473 ++si;
1474 if (si != source.end()) {
1475 type += (*si);
1476 ++si;
1477 }
1478 } else {
1479 type += (*si);
1480 ++si;
1481 }
1482 }
1483
1484 int nid = OBJ_txt2nid((char *)type.c_str());
1485 if (nid == NID_undef) {
1486 downloader_cat.info()
1487 << "Unknown type " << type << " in X509 name: " << source
1488 << "\n";
1489 X509_NAME_free(result);
1490 return nullptr;
1491 }
1492
1493 string value;
1494
1495 if (si != source.end() && (*si) == '=') {
1496 ++si;
1497 while (si != source.end() && (*si) != '/') {
1498 if ((*si) == '\\') {
1499 ++si;
1500 if (si != source.end()) {
1501 value += (*si);
1502 ++si;
1503 }
1504 } else {
1505 value += (*si);
1506 ++si;
1507 }
1508 }
1509 }
1510
1511 if (!value.empty()) {
1512 int add_result =
1513 X509_NAME_add_entry_by_NID(result, nid, V_ASN1_APP_CHOOSE,
1514 (unsigned char *)value.c_str(), -1, -1, 0);
1515 if (!add_result) {
1516 downloader_cat.info()
1517 << "Unable to add " << type << "=" << value << " in X509 name: "
1518 << source << "\n";
1519 X509_NAME_free(result);
1520 return nullptr;
1521 }
1522 added_any = true;
1523 }
1524 }
1525 }
1526
1527 if (!added_any) {
1528 downloader_cat.info()
1529 << "Invalid empty X509 name: " << source << "\n";
1530 X509_NAME_free(result);
1531 return nullptr;
1532 }
1533
1534 return result;
1535}

1541bool HTTPClient::
1542x509_name_subset(X509_NAME *name_a, X509_NAME *name_b) {
1543 int count_a = X509_NAME_entry_count(name_a);
1544 for (int ai = 0; ai < count_a; ai++) {
1545 X509_NAME_ENTRY *na = X509_NAME_get_entry(name_a, ai);
1546
1547 int bi = X509_NAME_get_index_by_OBJ(name_b, X509_NAME_ENTRY_get_object(na), -1);
1548 if (bi < 0) {
1549 // This entry in name_a is not defined in name_b.
1550 return false;
1551 }
1552
1553 X509_NAME_ENTRY *nb = X509_NAME_get_entry(name_b, bi);
1554 ASN1_STRING *na_value = X509_NAME_ENTRY_get_data(na);
1555 ASN1_STRING *nb_value = X509_NAME_ENTRY_get_data(nb);
1556 if (na_value->length != nb_value->length ||
1557 memcmp(na_value->data, nb_value->data, na_value->length) != 0) {
1558 // This entry in name_a doesn't match that of name_b.
1559 return false;
1560 }
1561 }
