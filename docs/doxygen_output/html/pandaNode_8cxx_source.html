54/*
55 * There are two different interfaces here for making and breaking parent-
56 * child connections: the fundamental PandaNode interface, via add_child() and
57 * remove_child() (and related functions), and the NodePath support interface,
58 * via attach(), detach(), and reparent(). They both do essentially the same
59 * thing, but with slightly different inputs. The PandaNode interfaces try to
60 * guess which NodePaths should be updated as a result of the scene graph
61 * change, while the NodePath interfaces already know. The NodePath support
62 * interface functions are strictly called from within the NodePath class, and
63 * are used to implement NodePath::reparent_to() and NodePath::remove_node(),
64 * etc. The fundamental interface, on the other hand, is intended to be
65 * called directly by the user. The fundamental interface has a slightly
66 * lower overhead because it does not need to create a NodePathComponent chain
67 * where one does not already exist; however, the NodePath support interface
68 * is more useful when the NodePath already does exist, because it ensures
69 * that the particular NodePath calling it is kept appropriately up-to-date.
70 */

477add_child(PandaNode *child_node, int sort, Thread *current_thread) {
478 nassertv(child_node != nullptr);

481 // Whoops, adding this child node would introduce a cycle in the scene
482 // graph.
483 return;
484 }

587 Thread *current_thread) {
588 nassertr(orig_child != nullptr, false);
589 nassertr(new_child != nullptr, false);

828void PandaNode::
829steal_children(PandaNode *other, Thread *current_thread) {
830 if (other == this) {
831 // Trivial.
832 return;
833 }
834
835 // We do this through the high-level interface for convenience. This could
836 // begin to be a problem if we have a node with hundreds of children to
837 // copy; this could break down the ov_set.insert() method, which is an
838 // O(n^2) operation. If this happens, we should rewrite this to do a
839 // simpler add_child() operation that involves push_back() instead of
840 // insert(), and then sort the down list at the end.
841
842 int num_children = other->get_num_children();
843 int i;
844 for (i = 0; i < num_children; i++) {
845 PandaNode *child_node = other->get_child(i);
846 int sort = other->get_child_sort(i);
847 add_child(child_node, sort, current_thread);
848 }
849 int num_stashed = other->get_num_stashed();
850 for (i = 0; i < num_stashed; i++) {
851 PandaNode *child_node = other->get_stashed(i);
852 int sort = other->get_stashed_sort(i);
853 add_stashed(child_node, sort, current_thread);
854 }
855
856 other->remove_all_children(current_thread);

1253int PandaNode::
1254compare_tags(const PandaNode *other) const {
1255 CDReader cdata(_cycler);
1256 CDReader cdata_other(other->_cycler);
1257
1258 const TagData &a_data = cdata->_tag_data;
1259 const TagData &b_data = cdata_other->_tag_data;
1260
1261 size_t ai = 0;
1262 size_t bi = 0;
1263 while (ai < a_data.size() && bi < b_data.size()) {
1264 int cmp = strcmp(a_data.get_key(ai).c_str(), b_data.get_key(bi).c_str());
1265 if (cmp != 0) {
1266 return cmp;
1267 }
1268
1269 cmp = strcmp(a_data.get_data(ai).c_str(), b_data.get_data(bi).c_str());
1270 if (cmp != 0) {
1271 return cmp;
1272 }
1273
1274 ++ai;
1275 ++bi;
1276 }
1277 if (ai < a_data.size()) {
1278 // list A is longer.
1279 return 1;
1280 }
1281 if (bi < b_data.size()) {
1282 // list B is longer.
1283 return -1;
1284 }
1285
1286 // We compare these by pointer, since it's problematic to call up into
1287 // Python from arbitrary C++ code.
1288 if (_python_tag_data != other->_python_tag_data) {
1289 return (_python_tag_data < other->_python_tag_data) ? -1 : 1;
1290 }
1291
1292 return 0;
1293}

1301void PandaNode::
1302copy_all_properties(PandaNode *other) {
1303 if (other == this) {
1304 // Trivial.
1305 return;
1306 }
1307
1308 bool any_transform_changed = false;
1309 bool any_state_changed = false;
1310 bool any_draw_mask_changed = false;
1311 Thread *current_thread = Thread::get_current_thread();
1312 OPEN_ITERATE_CURRENT_AND_UPSTREAM(_cycler, current_thread) {
1313 CDStageWriter cdataw(_cycler, pipeline_stage, current_thread);
1314 CDStageReader cdatar(other->_cycler, pipeline_stage, current_thread);
1315
1316 if (cdataw->_transform != cdatar->_transform) {
1317 any_transform_changed = true;
1318 }
1319 if (cdataw->_state != cdatar->_state) {
1320 any_state_changed = true;
1321 }
1322 if (cdataw->_draw_control_mask != cdatar->_draw_control_mask ||
1323 cdataw->_draw_show_mask != cdatar->_draw_show_mask) {
1324 any_draw_mask_changed = true;
1325 }
1326
1327 cdataw->_transform = cdatar->_transform;
1328 if (other->_prev_transform_valid == _reset_prev_transform_seq) {
1329 cdataw->_prev_transform = cdatar->_prev_transform;
1330 } else {
1331 cdataw->_prev_transform = cdatar->_transform;
1332 }
1333 cdataw->_state = cdatar->_state;
1334 cdataw->_effects = cdatar->_effects;
1335 cdataw->_draw_control_mask = cdatar->_draw_control_mask;
1336 cdataw->_draw_show_mask = cdatar->_draw_show_mask;
1337
1338 // The collide mask becomes the union of the two masks. This is important
1339 // to preserve properties such as the default GeomNode bitmask.
1340 cdataw->_into_collide_mask |= cdatar->_into_collide_mask;
1341
1342 for (size_t n = 0; n < cdatar->_tag_data.size(); ++n) {
1343 cdataw->_tag_data.store(cdatar->_tag_data.get_key(n), cdatar->_tag_data.get_data(n));
1344 }
1345
1346 static const int change_bits = (FB_transform | FB_state | FB_effects |
1347 FB_tag | FB_draw_mask | FB_decal |
1348 FB_show_bounds | FB_show_tight_bounds);
1349 cdataw->_fancy_bits =
1350 (cdataw->_fancy_bits & ~change_bits) |
1351 (cdatar->_fancy_bits & change_bits);
1352
1353 if (pipeline_stage == 0) {
1354 _prev_transform_valid = _reset_prev_transform_seq;
1355 }
1356 }
1357 CLOSE_ITERATE_CURRENT_AND_UPSTREAM(_cycler);
1358
1359 // It's okay to copy the tags by pointer, because get_python_tags does a
1360 // copy-on-write.
1361 _python_tag_data = other->_python_tag_data;
1362
1363 if (any_transform_changed || any_state_changed || any_draw_mask_changed) {
1364 mark_bounds_stale(current_thread);
1365
1366 if (any_transform_changed) {
1367 transform_changed();
1368 }
1369 if (any_state_changed) {
1370 state_changed();
1371 }
1372 if (any_draw_mask_changed) {
1373 draw_mask_changed();
1374 }
1375 mark_bam_modified();
1376 }
1377}

1394void PandaNode::
1395replace_node(PandaNode *other) {
1396 // nassertv(Thread::get_current_pipeline_stage() == 0);
1397
1398 if (other == this) {
1399 // Trivial.
1400 return;
1401 }
1402
1403 // Make sure the other node is not destructed during the execution of this
1404 // method.
1405 PT(PandaNode) keep_other = other;
1406
1407 // Get all the important scene graph properties.
1408 copy_all_properties(other);
1409
1410 // Fix up the NodePaths.
1411 {
1412 LightReMutexHolder holder1(other->_paths_lock);
1413 LightReMutexHolder holder2(_paths_lock);
1414 Paths::iterator pi;
1415 for (pi = other->_paths.begin(); pi != other->_paths.end(); ++pi) {
1416 (*pi)->_node = this;
1417 _paths.insert(*pi);
1418 }
1419 other->_paths.clear();
1420 }
1421
1422 // Get the children.
1423 steal_children(other);
1424
1425 // Switch the parents.
1426 Thread *current_thread = Thread::get_current_thread();
1427 Parents other_parents = other->get_parents();
1428 for (size_t i = 0; i < other_parents.get_num_parents(); ++i) {
1429 PandaNode *parent = other_parents.get_parent(i);
1430 if (find_parent(parent) != -1) {
1431 // This node was already a child of this parent; don't change it.
1432 parent->remove_child(other);
1433 } else {
1434 // This node was not yet a child of this parent; now it is.
1435 parent->replace_child(other, this, current_thread);
1436 }
1437 }
1438}

2537bool PandaNode::
2538stage_replace_child(PandaNode *orig_child, PandaNode *new_child,
2539 int pipeline_stage, Thread *current_thread) {
2540 {
2541 CDStageWriter cdata(_cycler, pipeline_stage, current_thread);
2542 CDStageWriter cdata_orig_child(orig_child->_cycler, pipeline_stage, current_thread);
2543 CDStageWriter cdata_new_child(new_child->_cycler, pipeline_stage, current_thread);
2544
2545 // First, look for the parent in the child's up list, to ensure the child
2546 // is known.
2547 int parent_index = orig_child->do_find_parent(this, cdata_orig_child);
2548 if (parent_index < 0) {
2549 // Nope, no relation.
2550 return false;
2551 }
2552
2553 if (orig_child == new_child) {
2554 // Trivial no-op.
2555 return true;
2556 }
2557
2558 // Don't let orig_child be destructed yet.
2559 PT(PandaNode) keep_orig_child = orig_child;
2560
2561 // If we already have new_child as a child, remove it first.
2562 if (stage_remove_child(new_child, pipeline_stage, current_thread)) {
2563 sever_connection(this, new_child, pipeline_stage, current_thread);
2564 }
2565
2566 PT(Down) down = cdata->modify_down();
2567 int child_index = do_find_child(orig_child, down);
2568 if (child_index >= 0) {
2569 // The child exists; replace it.
2570 DownConnection &dc = (*down)[child_index];
2571 nassertr(dc.get_child() == orig_child, false);
2572 dc.set_child(new_child);
2573
2574 } else {
2575 PT(Down) stashed = cdata->modify_stashed();
2576 int stashed_index = do_find_child(orig_child, stashed);
2577 if (stashed_index >= 0) {
2578 // The child has been stashed; remove it.
2579 DownConnection &dc = (*stashed)[stashed_index];
2580 nassertr(dc.get_child() == orig_child, false);
2581 dc.set_child(new_child);
2582
2583 } else {
2584 // Never heard of this child. This shouldn't be possible, because the
2585 // parent was in the child's up list, above. Must be some internal
2586 // error.
2587 nassertr(false, false);
2588 return false;
2589 }
2590 }
2591
2592 // Now adjust the bookkeeping on both children.
2593 cdata_new_child->modify_up()->insert(UpConnection(this));
2594 int num_erased = cdata_orig_child->modify_up()->erase(UpConnection(this));
2595 nassertr(num_erased == 1, false);
2596 }
2597
2598 sever_connection(this, orig_child, pipeline_stage, current_thread);
2599 new_connection(this, new_child, pipeline_stage, current_thread);
2600
2601 force_bounds_stale(pipeline_stage, current_thread);
2602 orig_child->parents_changed();
2603 new_child->parents_changed();
2604 mark_bam_modified();
2605 orig_child->mark_bam_modified();
2606 new_child->mark_bam_modified();
2607
2608 return true;
2609}

3431 if (drawmask_cat.is_debug()) {
3432 drawmask_cat.debug(false)
3433 << "exception_mask = " << exception_mask << "\n";
3434 }
3435
3436 // Now compute the union, applying the above exception.
3437 net_draw_control_mask |= child_control_mask;
3438 net_draw_show_mask |= child_show_mask;
3439
3440 net_draw_control_mask &= ~exception_mask;
3441 net_draw_show_mask |= exception_mask;
3442 }
3443
3444 if (drawmask_cat.is_debug()) {
3445 drawmask_cat.debug(false)
3446 << "child_control_mask = " << child_control_mask
3447 << "\nchild_show_mask = " << child_show_mask
3448 << "\nnet_draw_control_mask = " << net_draw_control_mask
3449 << "\nnet_draw_show_mask = " << net_draw_show_mask
3450 << "\n";
3451 }
3452
3453 off_clip_planes = orig_cp->compose_off(child_cdata->_off_clip_planes);
3454
3455 if (update_bounds) {
3456 if (!child_cdata->_external_bounds->is_empty()) {
3457#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
3458 child_volumes_ref.push_back(child_cdata->_external_bounds);
3459#endif
3460 nassertr(child_volumes_i < num_children + 1, CDStageWriter(_cycler, pipeline_stage, cdata));
3461 child_volumes[child_volumes_i++] = child_cdata->_external_bounds;
3462 }
3463 child_vertices += child_cdata->_nested_vertices;
3464
3465 connection._external_bounds = child_cdata->_external_bounds->as_geometric_bounding_volume();
3466 }
3467
3468 connection._net_draw_control_mask = child_control_mask;
3469 connection._net_draw_show_mask = child_show_mask;
3470 }
3471 }
3472
3473 {
3474 // Now grab the write lock on this node.
3475 CDStageWriter cdataw(_cycler, pipeline_stage, current_thread);
3476 if (last_update == cdataw->_last_update &&
3477 next_update == cdataw->_next_update) {
3478 // Great, no one has monkeyed with these while we were computing the
3479 // cache. Safe to store the computed values and return.
3480 cdataw->_net_collide_mask = net_collide_mask;
3481
3482 if (renderable) {
3483 // Any explicit draw control mask on this node trumps anything
3484 // inherited from below, except a show-through.
3485 DrawMask draw_control_mask = cdataw->_draw_control_mask;
3486 DrawMask draw_show_mask = cdataw->_draw_show_mask;
3487
3488 DrawMask show_through_mask = net_draw_control_mask & net_draw_show_mask;
3489
3490 net_draw_control_mask |= draw_control_mask;
3491 net_draw_show_mask = (net_draw_show_mask & ~draw_control_mask) | (draw_show_mask & draw_control_mask);
3492
3493 net_draw_show_mask |= show_through_mask;
3494
3495 // There are renderable nodes below, so the implicit draw bits are
3496 // all on.
3497 cdataw->_net_draw_control_mask = net_draw_control_mask;
3498 cdataw->_net_draw_show_mask = net_draw_show_mask | ~net_draw_control_mask;
3499 if (drawmask_cat.is_debug()) {
3500 drawmask_cat.debug(false)
3501 << "renderable, set mask " << cdataw->_net_draw_show_mask << "\n";
3502 }
3503 } else {
3504 // There are no renderable nodes below, so the implicit draw bits
3505 // are all off. Also, we don't care about the draw mask on this
3506 // particular node (since nothing below it is renderable anyway).
3507 cdataw->_net_draw_control_mask = net_draw_control_mask;
3508 cdataw->_net_draw_show_mask = net_draw_show_mask;
3509 if (drawmask_cat.is_debug()) {
3510 drawmask_cat.debug(false)
3511 << "not renderable, set mask " << cdataw->_net_draw_show_mask << "\n";
3512 }
3513 }
3514
3515 cdataw->_off_clip_planes = off_clip_planes;
3516
3517 if (update_bounds) {
3518 cdataw->_nested_vertices = cdataw->_internal_vertices + child_vertices;
3519
3520 BoundingVolume::BoundsType btype = cdataw->_bounds_type;
3521 if (btype == BoundingVolume::BT_default) {
3522 btype = bounds_type;
3523 }
3524
3525 compute_external_bounds(cdataw->_external_bounds, btype,
3526 child_volumes, child_volumes_i,
3527 pipeline_stage, current_thread);
3528
3529 nassertr(cdataw->_external_bounds != nullptr, cdataw);
3530
3531 cdataw->_last_bounds_update = next_update;
3532 }
3533
3534 cdataw->_last_update = next_update;
3535
3536 if (drawmask_cat.is_debug()) {
3537 drawmask_cat.debug(false)
3538 << "} " << *this << "::update_cached();\n";
3539 }
3540
3541 nassertr(cdataw->_last_update == cdataw->_next_update, cdataw);
3542
3543 // Even though implicit bounding volume is not (yet?) part of the bam
3544 // stream.
3545 mark_bam_modified();
3546 return cdataw;
3547 }
3548
3549 if (cdataw->_last_update == cdataw->_next_update &&
3550 (!update_bounds || cdataw->_last_bounds_update == cdataw->_next_update)) {
3551 // Someone else has computed the cache for us. OK.
3552 return cdataw;
3553 }
3554 }
3555
3556 // We need to go around again. Release the write lock, and grab the read
3557 // lock back.
3558 cdata = CDLockedStageReader(_cycler, pipeline_stage, current_thread);
3559
3560 if (cdata->_last_update == cdata->_next_update &&
3561 (!update_bounds || cdata->_last_bounds_update == cdata->_next_update)) {
3562 // Someone else has computed the cache for us while we were diddling
3563 // with the locks. OK.
3564 return CDStageWriter(_cycler, pipeline_stage, cdata);
3565 }
3566
3567 } while (true);
3568}

3801int PandaNode::CData::
3802complete_pointers(TypedWritable **p_list, BamReader *manager) {
3803 int pi = CycleData::complete_pointers(p_list, manager);
3804
3805 // Get the state and transform pointers.
3806 RenderState *state;
3807 DCAST_INTO_R(state, p_list[pi++], pi);
3808 _state = state;
3809
3810 TransformState *transform;
3811 DCAST_INTO_R(transform, p_list[pi++], pi);
3812 _prev_transform = _transform = transform;
3813
3814/*
3815 * Finalize these pointers now to decrement their artificially-held reference
3816 * counts. We do this now, rather than later, in case some other object
3817 * reassigns them a little later on during initialization, before they can
3818 * finalize themselves normally (for instance, the character may change the
3819 * node's transform). If that happens, the pointer may discover that no one
3820 * else holds its reference count when it finalizes, which will constitute a
3821 * memory leak (see the comments in TransformState::finalize(), etc.).
3822 */
3823 manager->finalize_now((RenderState *)_state.p());
3824 manager->finalize_now((TransformState *)_transform.p());
3825
3826
3827
3828 // Get the effects pointer.
3829 RenderEffects *effects;
3830 DCAST_INTO_R(effects, p_list[pi++], pi);
3831 _effects = effects;
3832
3833/*
3834 * Finalize these pointers now to decrement their artificially-held reference
3835 * counts. We do this now, rather than later, in case some other object
3836 * reassigns them a little later on during initialization, before they can
3837 * finalize themselves normally (for instance, the character may change the
3838 * node's transform). If that happens, the pointer may discover that no one
3839 * else holds its reference count when it finalizes, which will constitute a
3840 * memory leak (see the comments in TransformState::finalize(), etc.).
3841 */
3842 manager->finalize_now((RenderEffects *)_effects.p());
3843
3844
3845
3846 // Get the parent and child pointers.
3847 pi += complete_up_list(*modify_up(), "up", p_list + pi, manager);
3848 pi += complete_down_list(*modify_down(), "down", p_list + pi, manager);
3849 pi += complete_down_list(*modify_stashed(), "stashed", p_list + pi, manager);
3850
3851 // Since the _effects and _states members have been finalized by now, this
3852 // should be safe.
3853 set_fancy_bit(FB_transform, !_transform->is_identity());
3854 set_fancy_bit(FB_state, !_state->is_empty());
3855 set_fancy_bit(FB_effects, !_effects->is_empty());
3856 set_fancy_bit(FB_tag, !_tag_data.is_empty());
3857 set_fancy_bit(FB_decal, _effects->has_decal());
3858 set_fancy_bit(FB_show_bounds, _effects->has_show_bounds());
3859 set_fancy_bit(FB_show_tight_bounds, _effects->has_show_tight_bounds());
3860
3861 // Mark the bounds stale.
3862 ++_next_update;
3863
3864 nassertr(!_transform->is_invalid(), pi);
3865 nassertr(!_prev_transform->is_invalid(), pi);
3866
3867 return pi;
3868}

3933void PandaNode::CData::
3934write_up_list(const PandaNode::Up &up_list,
3935 BamWriter *manager, Datagram &dg) const {
3936/*
3937 * When we write a PandaNode, we write out its complete list of child node
3938 * pointers, but we only write out the parent node pointers that have already
3939 * been added to the bam file by a previous write operation. This is a bit of
3940 * trickery that allows us to write out just a subgraph (instead of the
3941 * complete graph) when we write out an arbitrary node in the graph, yet also
3942 * allows us to keep nodes completely in sync when we use the bam format for
3943 * streaming scene graph operations over the network.
3944 */
3945
3946 int num_parents = 0;
3947 Up::const_iterator ui;
3948 for (ui = up_list.begin(); ui != up_list.end(); ++ui) {
3949 PandaNode *parent_node = (*ui).get_parent();
3950 if (manager->has_object(parent_node)) {
3951 num_parents++;
3952 }
3953 }
3954 nassertv(num_parents == (int)(uint16_t)num_parents);
3955 dg.add_uint16(num_parents);
3956 for (ui = up_list.begin(); ui != up_list.end(); ++ui) {
3957 PandaNode *parent_node = (*ui).get_parent();
3958 if (manager->has_object(parent_node)) {
3959 manager->write_pointer(dg, parent_node);
3960 }
3961 }
3962}

4111void PandaNodePipelineReader::
4112check_cached(bool update_bounds) const {
4113 UpdateSeq last_update = update_bounds
4114 ? _cdata->_last_bounds_update
4115 : _cdata->_last_update;
4116
4117 if (last_update != _cdata->_next_update) {
4118 // The cache is stale; it needs to be rebuilt.
4119
4120 // We'll need to get a fresh read pointer, since another thread might
4121 // already have modified the pointer on the object since we queried it.
4122#ifdef DO_PIPELINING
4123 node_unref_delete((CycleData *)_cdata);
4124#endif // DO_PIPELINING
4125 ((PandaNodePipelineReader *)this)->_cdata = nullptr;
4126 int pipeline_stage = _current_thread->get_pipeline_stage();
4127 PandaNode::CDLockedStageReader fresh_cdata(_node->_cycler, pipeline_stage, _current_thread);
4128 if (fresh_cdata->_last_update == fresh_cdata->_next_update &&
4129 (!update_bounds || fresh_cdata->_last_bounds_update == fresh_cdata->_next_update)) {
4130 // What luck, some other thread has already freshened the cache for us.
4131 // Save the new pointer, and let the lock release itself.
4132 if (_cdata != (const PandaNode::CData *)fresh_cdata) {
4133 ((PandaNodePipelineReader *)this)->_cdata = fresh_cdata;
