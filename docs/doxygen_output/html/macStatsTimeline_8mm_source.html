197void MacStatsTimeline::
198draw_bar(int row, int from_x, int to_x, int collector_index,
199 const std::string &collector_name) {
200 int top = row_to_pixel(row);
201 int bottom = row_to_pixel(row + 1);
202 int scale = 4;
203
204 top += 1;
205
206 MacStatsMonitor *monitor = MacStatsGraph::_monitor;
207
208 bool is_highlighted = row == _highlighted_row && _highlighted_x >= from_x && _highlighted_x < to_x;
209 CGContextSetFillColorWithColor(_ctx,
210 monitor->get_collector_color(collector_index, is_highlighted));
211
212 if (to_x < from_x + 1) {
213 // Too tiny to draw.
214 }
215 else if (to_x < from_x + scale) {
216 // It's just a tiny sliver. This is a more reliable way to draw it.
217 CGRect rect = CGRectMake(from_x, top, to_x - from_x, bottom - top);
218 CGContextFillRect(_ctx, rect);
219 }
220 else {
221 int left = std::max(from_x, -scale - 1);
222 int right = std::min(std::max(to_x, from_x + 1), get_xsize() + scale);
223
224 double radius = std::min((double)scale, (right - left) / 2.0);
225 CGContextBeginPath(_ctx);
226 CGContextAddArc(_ctx, right - radius - 0.5, top + radius, radius, -0.5 * M_PI, 0.0, NO);
227 CGContextAddArc(_ctx, right - radius - 0.5, bottom - radius, radius, 0.0, 0.5 * M_PI, NO);
228 CGContextAddArc(_ctx, left + radius, bottom - radius, radius, 0.5 * M_PI, M_PI, NO);
229 CGContextAddArc(_ctx, left + radius, top + radius, radius, M_PI, 1.5 * M_PI, NO);
230 CGContextClosePath(_ctx);
231 CGContextFillPath(_ctx);
232
233 if ((to_x - from_x) >= scale * 4) {
234 // Only bother drawing the text if we've got some space to draw on.
235 const CFStringRef keys[] = {
236 (__bridge CFStringRef)NSForegroundColorAttributeName,
237 (__bridge CFStringRef)NSFontAttributeName,
238 };
239 const void *values[] = {
240 monitor->get_collector_text_color(collector_index, is_highlighted),
241 [NSFont systemFontOfSize:0.0],
242 };
243 CFDictionaryRef attribs = CFDictionaryCreate(kCFAllocatorDefault, (const void **)keys, (const void **)values, 2, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
244
245 CFStringRef str = CFStringCreateWithCString(kCFAllocatorDefault, collector_name.c_str(), kCFStringEncodingUTF8);
246 CFAttributedStringRef astr = CFAttributedStringCreate(kCFAllocatorDefault, str, attribs);
247
248 CTLineRef line = CTLineCreateWithAttributedString(astr);
249 CGRect bounds = CTLineGetImageBounds(line, _ctx);
250 CFRelease(astr);
251 CFRelease(str);
252
253 int text_width = bounds.size.width;
254 int text_height = bounds.size.height;
255
256 double center = (from_x + to_x) / 2.0;
257 double text_left = std::max(from_x, 0) + scale / 2.0;
258 double text_right = std::min(to_x, get_xsize()) - scale / 2.0;
259 double text_top = top + (bottom - top - text_height) / 2.0 + text_height;
260
261 if (text_width >= text_right - text_left) {
262 size_t c = collector_name.rfind(':');
263 if (text_right - text_left < scale * 6) {
264 // It's a really tiny space. Draw a single letter.
265 UniChar ch = *(collector_name.data() + (c != std::string::npos ? c + 1 : 0));
266
267 CFStringRef str = CFStringCreateWithCharacters(kCFAllocatorDefault, &ch, 1);
268 CFAttributedStringRef astr = CFAttributedStringCreate(kCFAllocatorDefault, str, attribs);
269
270 CTLineRef new_line = CTLineCreateWithAttributedString((CFAttributedStringRef)astr);
271 bounds = CTLineGetImageBounds(new_line, _ctx);
272 text_width = bounds.size.width;
273
274 CFRelease(line);
275 CFRelease(astr);
276 CFRelease(str);
277 line = new_line;
278 }
279 else {
280 // Maybe just use everything after the last colon.
281 if (c != std::string::npos) {
282 const char *short_name = collector_name.data() + c + 1;
283 CFStringRef str = CFStringCreateWithCString(kCFAllocatorDefault, short_name, kCFStringEncodingUTF8);
284 CFAttributedStringRef astr = CFAttributedStringCreate(kCFAllocatorDefault, str, attribs);
285
286 CTLineRef new_line = CTLineCreateWithAttributedString((CFAttributedStringRef)astr);
287 bounds = CTLineGetImageBounds(new_line, _ctx);
288 text_width = bounds.size.width;
289
290 CFRelease(line);
291 CFRelease(astr);
292 CFRelease(str);
293 line = new_line;
294 }
295 }
296 }
297
298 if (text_width >= text_right - text_left) {
299 // Have CoreText truncate to the correct length.
300 static CFStringRef token_str = CFSTR("\u2026");
301 CFAttributedStringRef token_astr = CFAttributedStringCreate(kCFAllocatorDefault, token_str, attribs);
302 CTLineRef token_line = CTLineCreateWithAttributedString(token_astr);
303 CTLineRef trunc_line = CTLineCreateTruncatedLine(line, text_right - text_left, kCTLineTruncationEnd, token_line);
304 CFRelease(line);
305 CFRelease(token_astr);
306 CFRelease(token_line);
307 line = trunc_line;
308 CGContextSetTextPosition(_ctx, text_left, text_top);
309 }
310 else if (center - text_width / 2.0 < 0.0) {
311 // Put it against the left-most edge.
312 CGContextSetTextPosition(_ctx, scale, text_top);
313 }
314 else if (center + text_width / 2.0 >= get_xsize()) {
315 // Put it against the right-most edge.
316 CGContextSetTextPosition(_ctx, get_xsize() - scale - text_width, text_top);
317 }
318 else {
319 // It fits just fine, center it.
320 CGContextSetTextPosition(_ctx, center - text_width / 2.0, text_top);
321 }
322
323 if (line != nullptr) {
324 CTLineDraw(line, _ctx);
325 CFRelease(line);
326 }
327 CFRelease(attribs);
328 }
329 }

336void MacStatsTimeline::
337end_draw() {
338 // Recalculate the size of the graph.
339 if (!_threads.empty()) {
340 const ThreadRow &last_thread_row = _threads.back();
341 int new_height = row_to_pixel(last_thread_row._row_offset + last_thread_row._rows.size());
342 _graph_height_constraint.constant = new_height;
343 }
344
345 _graph_view.needsDisplay = YES;
346
347 // If we scroll sideways while we're also scrolling vertically such that the
348 // overhang becomes visible due to elasticity, the overhang doesn't update.
349 // I could only find this private method for fixing this problem.
350 // Not needed as of macOS 14 and up, since the normal drawRect DTRT there.
351#if __MAC_OS_X_VERSION_MAX_ALLOWED < 140000
352 NSClipView *clip_view = ((MacStatsScrollableGraphViewController *)_graph_view_controller).clipView;
353 if ([clip_view respondsToSelector:@selector(_setNeedsDisplayInOverhang:)]) {
354#pragma clang diagnostic push
355#pragma clang diagnostic ignored "-Wobjc-method-access"
356 [clip_view _setNeedsDisplayInOverhang:YES];
357#pragma clang diagnostic pop
358 }
359#endif
360
361 if (_threads_changed) {
362 while (_thread_labels.size() < _threads.size()) {
363 NSTextField *label = [NSTextField labelWithString:@"Thread"];
364 label.translatesAutoresizingMaskIntoConstraints = NO;
365 [_thread_area addSubview:label];
366
367 [label.rightAnchor constraintEqualToAnchor:_thread_area.rightAnchor constant:-8].active = YES;
368 [_thread_area.widthAnchor constraintGreaterThanOrEqualToAnchor:label.widthAnchor constant:16].active = YES;
369
370 NSLayoutConstraint *constraint;
371 if (@available(macOS 11.0, *)) {
372 constraint = [label.topAnchor constraintEqualToAnchor:_thread_area.topAnchor];
373 } else {
374 constraint = [label.topAnchor constraintEqualToAnchor:_thread_area.topAnchor];
375 }
376 constraint.active = YES;
377
378 _thread_labels.push_back(std::make_pair(label, constraint));
379 }
380
381 for (size_t i = 0; i < _threads.size(); ++i) {
382 const ThreadRow &thread_row = _threads[i];
383 NSTextField *label = _thread_labels[i].first;
384 NSLayoutConstraint *label_constraint = _thread_labels[i].second;
385
386 label.stringValue = [NSString stringWithUTF8String:thread_row._label.c_str()];
387 label_constraint.constant = row_to_pixel(thread_row._row_offset);
388 }
389
390 _thread_area.needsDisplay = YES;
391 _threads_changed = false;
392 }
393
394 if (_guide_bars_changed) {
395 _scale_area.needsDisplay = YES;
396 _guide_bars_changed = false;
397 }
398}

534bool MacStatsTimeline::
535handle_key(int graph_x, int graph_y, bool pressed, UniChar c, unsigned short key_code) {
536 // Accept WASD based on their position rather than their mapping
537 int flag = 0;
538 switch (key_code) {
539 case 13:
540 flag = F_w;
541 break;
542 case 0:
543 flag = F_a;
544 break;
545 case 1:
546 flag = F_s;
547 break;
548 case 2:
549 flag = F_d;
550 break;
551 }
552 if (flag == 0) {
553 switch (c) {
554 case 0x1c:
555 case NSLeftArrowFunctionKey:
556 flag = F_left;
557 break;
558 case 0x1d:
559 case NSRightArrowFunctionKey:
560 flag = F_right;
561 break;
562 case 'w':
563 flag = F_w;
564 break;
565 case 'a':
566 flag = F_a;
567 break;
568 case 's':
569 flag = F_s;
570 break;
571 case 'd':
572 flag = F_d;
573 break;
574 }
575 }
576 if (flag != 0) {
577 if (pressed) {
578 if (flag & (F_w | F_s)) {
579 _zoom_center = pixel_to_timestamp(graph_x);
580 }
581 if (_keys_held == 0) {
582 start_animation();
583 }
584 _keys_held |= flag;
585 }
586 else if (_keys_held != 0) {
587 _keys_held &= ~flag;
588 }
589 return true;
590 }
591 return false;
592}

618 _drag_start_x = graph_x;

620 _zoom_center = pixel_to_timestamp(graph_x);

656void MacStatsTimeline::
657handle_motion(int graph_x, int graph_y) {
658 if (_drag_mode == DM_none && _potential_drag_mode == DM_none &&
659 graph_x >= 0 && graph_y >= 0 && graph_x < get_xsize() && graph_y < get_ysize()) {
660 // When the mouse is over a color bar, highlight it.
661 int row = pixel_to_row(graph_y);
662 std::swap(_highlighted_x, graph_x);
663 std::swap(_highlighted_row, row);
664
665 if (row >= 0) {
666 PStatTimeline::force_redraw(row, graph_x, graph_x);
667 }
668 PStatTimeline::force_redraw(_highlighted_row, _highlighted_x, _highlighted_x);
669
670 if ((_keys_held & (F_w | F_s)) != 0) {
671 // Update the zoom center if we move the mouse while zooming with the
672 // keyboard.
673 _zoom_center = pixel_to_timestamp(graph_x);
674 }
675 }
676 else {
677 // If the mouse is in some drag mode, stop highlighting.
678 if (_highlighted_row != -1) {
679 int row = _highlighted_row;
680 _highlighted_row = -1;
681 PStatTimeline::force_redraw(row, _highlighted_x, _highlighted_x);
682 }
683 }
684
685 if (_drag_mode == DM_pan) {
686 int delta = _drag_start_x - graph_x;
687 _drag_start_x = graph_x;
688 set_horizontal_scroll(get_horizontal_scroll() + pixel_to_height(delta));
689 return;
690 }
691
692 return MacStatsGraph::handle_motion(graph_x, graph_y);
693}

745#if __MAC_OS_X_VERSION_MAX_ALLOWED >= 140000
