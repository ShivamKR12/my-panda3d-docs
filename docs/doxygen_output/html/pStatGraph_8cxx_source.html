152string PStatGraph::
153format_number(double value) {
154 char buffer[128];
155
156 if (value < 0.01) {
157 sprintf(buffer, "%0.4f", value);
158 } else if (value < 0.1) {
159 sprintf(buffer, "%0.3f", value);
160 } else if (value < 1.0) {
161 sprintf(buffer, "%0.2f", value);
162 } else if (value < 10.0) {
163 sprintf(buffer, "%0.1f", value);
164 } else {
165 sprintf(buffer, "%0.0f", value);
166 }
167
168 return buffer;
169}

175string PStatGraph::
176format_number(double value, int guide_bar_units, const string &unit_name) {
177 string label;
178
179 if ((guide_bar_units & GBU_named) != 0) {
180 // Units are whatever is specified by unit_name, not a time unit at all.
181 int int_value = (int)value;
182 if ((double)int_value == value) {
183 // Probably a counter or something, don't display .0 suffix.
184 label = format_string(int_value);
185 } else {
186 label = format_number(value);
187 }
188 if ((guide_bar_units & GBU_show_units) != 0 && !unit_name.empty()) {
189 label += " ";
190 label += unit_name;
191 }
192 }
193 else {
194 // Units are either milliseconds or hz, or both.
195 if ((guide_bar_units & GBU_ms) != 0) {
196 if ((guide_bar_units & GBU_show_units) != 0 &&
197 value > 0 && value < 0.000001) {
198 double ns = value * 1000000000.0;
199 label += format_number(ns);
200 label += " ns";
201 }
202 else if ((guide_bar_units & GBU_show_units) != 0 &&
203 value > 0 && value < 0.001) {
204 double us = value * 1000000.0;
205 label += format_number(us);
206#ifdef _WIN32
207 label += " \xb5s";
208#else
209 label += " \xc2\xb5s";
210#endif
211 }
212 else if ((guide_bar_units & GBU_show_units) == 0 || value < 1.0) {
213 double ms = value * 1000.0;
214 label += format_number(ms);
215 if ((guide_bar_units & GBU_show_units) != 0) {
216 label += " ms";
217 }
218 }
219 else {
220 label += format_number(value);
221 label += " s";
222 }
223 }
224
225 if ((guide_bar_units & GBU_hz) != 0) {
226 double hz = 1.0 / value;
227
228 if ((guide_bar_units & GBU_show_units) != 0 &&
229 (guide_bar_units & GBU_ms) == 0) {
230 if (hz >= 1000000000) {
231 label += format_number(hz / 1000000000);
232 label += " GHz";
233 }
234 else if (hz >= 1000000) {
235 label += format_number(hz / 1000000);
236 label += " MHz";
237 }
238 else if (hz >= 1000) {
239 label += format_number(hz / 1000);
240 label += " kHz";
241 }
242 else {
243 label += format_number(hz);
244 label += " Hz";
245 }
246 }
247 else {
248 if ((guide_bar_units & GBU_ms) != 0) {
249 label += " (";
250 }
251 label += format_number(hz);
252 if ((guide_bar_units & GBU_show_units) != 0) {
253 label += " Hz";
254 }
255 if ((guide_bar_units & GBU_ms) != 0) {
256 label += ")";
257 }
258 }
259 }
260 }
261
262 return label;
263}

305void PStatGraph::
306update_guide_bars(int num_bars, double scale) {
307 _guide_bars.clear();
308
309 // We'd like to draw about num_bars bars on the chart. But we also want the
310 // bars to be harmonics of the target frame rate, so that the bottom bar is
311 // at tfrn or n * tfr, where n is an integer, and the upper bars are even
312 // multiples of that.
313
314 // Choose a suitable harmonic of the target frame rate near the bottom part
315 // of the chart.
316
317 double bottom = (double)num_bars / scale;
318
319 double harmonic;
320 if (_target_frame_rate < bottom) {
321 // n * tfr
322 harmonic = floor(bottom / _target_frame_rate + 0.5) * _target_frame_rate;
323
324 } else {
325 // tfr n
326 harmonic = _target_frame_rate / floor(_target_frame_rate / bottom + 0.5);
327 }
328
329 // Now, make a few bars at k harmonic.
330 for (int k = 1; k / harmonic <= scale; k++) {
331 _guide_bars.push_back(make_guide_bar(k / harmonic));
332 }
333
334 _guide_bars_changed = true;
335}

340PStatGraph::GuideBar PStatGraph::
341make_guide_bar(double value, PStatGraph::GuideBarStyle style) const {
342 string label = format_number(value, _guide_bar_units, _unit_name);
343
344 if ((style == GBS_normal) &&
345 (_guide_bar_units & GBU_named) == 0) {
346 // If it's a time unit, check to see if it matches our target frame rate.
347 double hz = 1.0 / value;
348 if (IS_THRESHOLD_EQUAL(hz, _target_frame_rate, 0.001)) {
349 style = GBS_target;
350 }
351 }
352
353 return GuideBar(value, label, style);
354}
