138void wglGraphicsBuffer::
139bind_texture_to_pbuffer() {
140 wglGraphicsStateGuardian *wglgsg;
141 DCAST_INTO_V(wglgsg, _gsg);
142
143 // Find the color texture, if there is one. That one can be bound to the
144 // framebuffer. All others must be marked RTM_copy_to_texture.
145
146 int tex_index = -1;
147 CDLockedReader cdata(_cycler);
148 for (size_t i = 0; i != cdata->_textures.size(); ++i) {
149 const RenderTexture &rt = cdata->_textures[i];
150 RenderTexturePlane plane = rt._plane;
151 if (plane == RTP_color && rt._rtm_mode == RTM_bind_or_copy) {
152 tex_index = i;
153 break;
154 }
155 }
156
157 if (tex_index >= 0) {
158 const RenderTexture &rt = cdata->_textures[tex_index];
159 Texture *tex = rt._texture;
160 if (_pbuffer_bound != nullptr && _pbuffer_bound != tex) {
161 _pbuffer_bound->release(wglgsg->get_prepared_objects());
162 _pbuffer_bound = 0;
163 }
164 tex->set_size_padded(get_x_size(), get_y_size());
165
166 if (tex->get_match_framebuffer_format()) {
167 if (_fb_properties.get_alpha_bits()) {
168 tex->set_format(Texture::F_rgba);
169 } else {
170 tex->set_format(Texture::F_rgb);
171 }
172 }
173 TextureContext *tc = tex->prepare_now(_gsg->get_prepared_objects(), _gsg);
174 nassertv(tc != nullptr);
175 CLP(TextureContext) *gtc = DCAST(CLP(TextureContext), tc);
176 GLenum target = wglgsg->get_texture_target(tex->get_texture_type());
177 if (target == GL_NONE) {
178 CDWriter cdataw(_cycler, cdata, false);
179 nassertv(cdata->_textures.size() == cdataw->_textures.size());
180 cdataw->_textures[tex_index]._rtm_mode = RTM_copy_texture;
181 return;
182 }
183 GLP(BindTexture)(target, gtc->get_view_index(0));
184 if (_fb_properties.is_single_buffered()) {
185 wglgsg->_wglBindTexImageARB(_pbuffer, WGL_FRONT_LEFT_ARB);
186 } else {
187 wglgsg->_wglBindTexImageARB(_pbuffer, WGL_BACK_LEFT_ARB);
188 }
189 _pbuffer_bound = tex;
190 } else {
191 if (_pbuffer_bound != nullptr) {
192 _pbuffer_bound->release(wglgsg->get_prepared_objects());
193 _pbuffer_bound = 0;
194 }
195 }
196}

281bool wglGraphicsBuffer::
282open_buffer() {
283
284 // pbuffers don't seem to work correctly in double-buffered mode. Besides,
285 // the back buffer is a pointless waste of space. So always use a single-
286 // buffered gsg.
287
288 _fb_properties.set_back_buffers(0);
289 _draw_buffer_type = RenderBuffer::T_front;
290 _screenshot_buffer_type = RenderBuffer::T_front;
291
292 // GSG creationinitialization.
293
294 wglGraphicsStateGuardian *wglgsg;
295 if (_gsg == nullptr) {
296 // There is no old gsg. Create a new one.
297 wglgsg = new wglGraphicsStateGuardian(_engine, _pipe, nullptr);
298 wglgsg->choose_pixel_format(_fb_properties, true);
299 _gsg = wglgsg;
300 } else {
301 // If the old gsg has the wrong pixel format, create a new one that shares
302 // with the old gsg.
303 DCAST_INTO_R(wglgsg, _gsg, false);
304 if (wglgsg->get_engine() != _engine ||
305 !wglgsg->get_fb_properties().subsumes(_fb_properties) ||
306 !wglgsg->get_fb_properties().is_single_buffered() ||
307 !wglgsg->pfnum_supports_pbuffer()) {
308 wglgsg = new wglGraphicsStateGuardian(_engine, _pipe, wglgsg);
309 wglgsg->choose_pixel_format(_fb_properties, true);
310 _gsg = wglgsg;
311 }
312 }
313
314 // Use the temp window to initialize the gsg.
315
316 HDC twindow_dc = wglgsg->get_twindow_dc();
317 if (twindow_dc == 0) {
318 // If we couldn't make a window, we can't get a GL context.
319 _gsg = nullptr;
320 return false;
321 }
322 HGLRC context = wglgsg->get_context(twindow_dc);
323 if (context == 0) {
324 _gsg = nullptr;
325 return false;
326 }
327 wglGraphicsPipe::wgl_make_current(twindow_dc, context,
328 &_make_current_pcollector);
329 wglgsg->reset_if_new();
330 wglgsg->report_my_gl_errors();
331 if (!wglgsg->get_fb_properties().verify_hardware_software
332 (_fb_properties,wglgsg->get_gl_renderer())) {
333 _gsg = nullptr;
334 return false;
335 }
336 _fb_properties = wglgsg->get_fb_properties();
337
338 // Now that we have fully made a window and used that window to create a
339 // rendering context, we can attempt to create a pbuffer. This might fail
340 // if the pbuffer extensions are not supported.
341
342 if (!rebuild_bitplanes()) {
343 wglGraphicsPipe::wgl_make_current(0, 0, &_make_current_pcollector);
344 _gsg = nullptr;
345 return false;
346 }
347
348 _is_valid = true;
349
350 return true;
351}

390bool wglGraphicsBuffer::
391rebuild_bitplanes() {
392 wglGraphicsStateGuardian *wglgsg;
393 DCAST_INTO_R(wglgsg, _gsg, false);
394
395 if (!wglgsg->_supports_pbuffer) {
396 wgldisplay_cat.info()
397 << "PBuffers not supported by GL implementation.\n";
398 return false;
399 }
400
401 // Find the texture to bind to the color buffer.
402 Texture *bindtexture = nullptr;
403 for (int i=0; i<count_textures(); i++) {
404 if ((get_rtm_mode(i) == RTM_bind_or_copy)&&
405 (get_texture(i)->get_format() != Texture::F_depth_stencil)) {
406 bindtexture = get_texture(i);
407 break;
408 }
409 }
410
411 // If we already have a pbuffer, and if it's lost, then force the rebuild.
412
413 if (_pbuffer_dc) {
414 int flag = 0;
415 wglgsg->_wglQueryPbufferARB(_pbuffer, WGL_PBUFFER_LOST_ARB, &flag);
416 if (flag != 0) {
417 release_pbuffer();
418 }
419 }
420
421 // Determine what pbuffer attributes are needed for currently-applicable
422 // textures.
423
424 if (_host != nullptr && (_creation_flags & GraphicsPipe::BF_size_track_host) != 0) {
425 if (_host->get_size() != _size) {
426 set_size_and_recalc(_host->get_x_size(),
427 _host->get_y_size());
428 }
429 }
430 int desired_x = get_x_size();
431 int desired_y = get_y_size();
432 if ((bindtexture != 0)&&(Texture::get_textures_power_2() != ATS_none)) {
433 desired_x = Texture::up_to_power_2(desired_x);
434 desired_y = Texture::up_to_power_2(desired_y);
435 }
436 bool desired_mipmap = false;
437 Texture::TextureType desired_type = Texture::TT_2d_texture;
438 if (bindtexture != 0) {
439 desired_mipmap = bindtexture->uses_mipmaps();
440 desired_type = bindtexture->get_texture_type();
441 }
442
443 if ((_pbuffer != 0)&&
444 (_pbuffer_sizex == desired_x)&&
445 (_pbuffer_sizey == desired_y)&&
446 (_pbuffer_mipmap == desired_mipmap)&&
447 (_pbuffer_type == desired_type)) {
448 // the pbuffer we already have is fine. Do not rebuild.
449 return true;
450 }
451
452 // Release the old pbuffer, if there was one.
453
454 release_pbuffer();
455
456 // Allocate the new pbuffer.
457
458 int pfnum = wglgsg->get_pfnum();
459
460 static const int max_attrib_list = 64;
461 int iattrib_list[max_attrib_list];
462 int ni = 0;
463
464 if (_fb_properties.get_alpha_bits()) {
465 iattrib_list[ni++] = WGL_TEXTURE_FORMAT_ARB;
466 iattrib_list[ni++] = WGL_TEXTURE_RGBA_ARB;
467 } else {
468 iattrib_list[ni++] = WGL_TEXTURE_FORMAT_ARB;
469 iattrib_list[ni++] = WGL_TEXTURE_RGB_ARB;
470 }
471
472 if (desired_mipmap) {
473 iattrib_list[ni++] = WGL_MIPMAP_TEXTURE_ARB;
474 iattrib_list[ni++] = 1;
475 }
476
477 switch (desired_type) {
478 case Texture::TT_cube_map:
479 iattrib_list[ni++] = WGL_TEXTURE_TARGET_ARB;
480 iattrib_list[ni++] = WGL_TEXTURE_CUBE_MAP_ARB;
481 break;
482
483 case Texture::TT_1d_texture:
484 iattrib_list[ni++] = WGL_TEXTURE_TARGET_ARB;
485 iattrib_list[ni++] = WGL_TEXTURE_1D_ARB;
486 break;
487
488 default:
489 iattrib_list[ni++] = WGL_TEXTURE_TARGET_ARB;
490 iattrib_list[ni++] = WGL_TEXTURE_2D_ARB;
491 }
492
493 // Terminate the list.
494 nassertr(ni <= max_attrib_list, false);
495 iattrib_list[ni] = 0;
496
497 HDC twindow_dc = wglgsg->get_twindow_dc();
498 if (twindow_dc == 0) {
499 return false;
500 }
501
502 HGLRC context = wglgsg->get_context(twindow_dc);
503 if (context == 0) {
504 return false;
505 }
506 wglGraphicsPipe::wgl_make_current(twindow_dc, context,
507 &_make_current_pcollector);
508
509 _pbuffer = wglgsg->_wglCreatePbufferARB(twindow_dc, pfnum,
510 desired_x, desired_y, iattrib_list);
511
512 if (_pbuffer == 0) {
513 wgldisplay_cat.info()
514 << "Attempt to create pbuffer failed.\n";
515 return false;
516 }
517
518 _pbuffer_dc = wglgsg->_wglGetPbufferDCARB(_pbuffer);
519 _pbuffer_mipmap = desired_mipmap;
520 _pbuffer_type = desired_type;
521 _pbuffer_sizex = desired_x;
522 _pbuffer_sizey = desired_y;
523
524 return true;
525}
