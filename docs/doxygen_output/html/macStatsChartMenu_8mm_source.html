108void MacStatsChartMenu::
109do_update() {
110 PStatView &view = _monitor->get_view(_thread_index);
111 _last_level_index = view.get_level_index();
112
113 const PStatClientData *client_data = _monitor->get_client_data();
114 if (_thread_index != 0) {
115 std::string thread_name = client_data->get_thread_name(_thread_index);
116 _menu.title = [NSString stringWithUTF8String:thread_name.c_str()];
117 }
118
119 if (client_data->get_num_collectors() > _collector_items.size()) {
120 _collector_items.resize(client_data->get_num_collectors(), std::make_pair(nullptr, nullptr));
121 }
122
123 // The menu item(s) for the thread's frame time goes second.
124 const PStatViewLevel *view_level = view.get_top_level();
125 if (_thread_index == 0) {
126 if (add_view(_menu, view_level, false, _time_items_end)) {
127 ++_time_items_end;
128 ++_level_items_end;
129 }
130 } else {
131 for (int c = 0; c < view_level->get_num_children(); ++c) {
132 if (add_view(_menu, view_level->get_child(c), false, _time_items_end)) {
133 ++_time_items_end;
134 ++_level_items_end;
135 }
136 }
137 }
138
139 // And then the menu item(s) for each of the level values.
140 int num_toplevel_collectors = client_data->get_num_toplevel_collectors();
141 for (int tc = 0; tc < num_toplevel_collectors; tc++) {
142 int collector = client_data->get_toplevel_collector(tc);
143 if (client_data->has_collector(collector) &&
144 client_data->get_collector_has_level(collector, _thread_index)) {
145
146 PStatView &level_view = _monitor->get_level_view(collector, _thread_index);
147 add_view(_menu, level_view.get_top_level(), true, _level_items_end);
148 }
149 }
150}

156bool MacStatsChartMenu::
157add_view(NSMenu *parent_menu, const PStatViewLevel *view_level,
158 bool show_level, int insert_at) {
159 int collector = view_level->get_collector();
160
161 NSMenuItem *&menu_item = _collector_items[collector].first;
162 NSMenu *&menu = _collector_items[collector].second;
163
164 const PStatClientData *client_data = _monitor->get_client_data();
165
166 int num_children = view_level->get_num_children();
167 if (menu == nullptr && num_children == 0) {
168 // For a collector without children, no point in making a submenu. We just
169 // have the item open a strip chart directly (no point in creating a flame
170 // graph if there are no children).
171 if (menu_item != nullptr) {
172 // Already exists.
173 return false;
174 }
175
176 std::string collector_name = client_data->get_collector_name(collector);
177 if (show_level) {
178 menu_item = make_menu_item(parent_menu, insert_at,
179 collector_name.c_str(), @selector(handleOpenStripChartLevel:), collector);
180 } else {
181 menu_item = make_menu_item(parent_menu, insert_at,
182 collector_name.c_str(), @selector(handleOpenStripChart:), collector);
183 }
184 return true;
185 }
186 else if (menu_item != nullptr && menu == nullptr) {
187 // Unhook the signal handler, we are creating a submenu.
188 menu_item.action = nil;
189 }
190
191 // Create a submenu.
192 bool added_item = false;
193 if (menu_item == nullptr) {
194 std::string collector_name = client_data->get_collector_name(collector);
195 menu_item = make_menu_item(parent_menu, insert_at, collector_name.c_str(), nil);
196 added_item = true;
197 }
198
199 if (menu == nullptr) {
200 menu = [[NSMenu alloc] init];
201 [parent_menu setSubmenu:menu forItem:menu_item];
202
203 if (show_level) {
204 make_menu_item(menu, -1, "Open Strip Chart",
205 @selector(handleOpenStripChartLevel:), collector);
206 } else {
207 make_menu_item(menu, -1, "Open Strip Chart",
208 @selector(handleOpenStripChart:), collector);
209
210 if (collector == 0) {
211 collector = -1;
212 }
213
214 make_menu_item(menu, -1, "Open Flame Graph",
215 @selector(handleOpenFlameGraph:), collector);
216 }
217
218 [menu addItem:[NSMenuItem separatorItem]];
219 [menu release];
220 }
221
222 for (int c = 0; c < num_children; ++c) {
223 add_view(menu, view_level->get_child(c), show_level, 2 + !show_level);
224 }
225
226 return added_item;
227}
