84bool InternalName::
85unref() const {
86 if (_parent == nullptr) {
87 // No parent; no problem. This is the root InternalName. Actually, this
88 // probably shouldn't be destructing, but I guess it might at application
89 // shutdown.
90 return TypedWritableReferenceCount::unref();
91 }
92
93 LightMutexHolder holder(_parent->_name_table_lock);
94
95 if (ReferenceCount::unref()) {
96 return true;
97 }
98
99 // The reference count has just reached zero.
100 NameTable::iterator ni = _parent->_name_table.find(_basename);
101 nassertr(ni != _parent->_name_table.end(), false);
102 _parent->_name_table.erase(ni);
103
104 return false;
105}

112PT(InternalName) InternalName::
113append(const string &name) {
114 test_ref_count_integrity();
115
116 if (name.empty()) {
117 return this;
118 }
119
120 size_t dot = name.rfind('.');
121 if (dot != string::npos) {
122 return append(name.substr(0, dot))->append(name.substr(dot + 1));
123 }
124
125 LightMutexHolder holder(_name_table_lock);
126
127 NameTable::iterator ni = _name_table.find(name);
128 if (ni != _name_table.end()) {
129 return (*ni).second;
130 }
131
132 InternalName *internal_name = new InternalName(this, name);
133 _name_table[name] = internal_name;
134 return internal_name;
135}

281void InternalName::
282finalize(BamReader *) {
283 // Unref the pointer that we explicitly reffed in make_from_bam().
284 unref();
285
286 // We should never get back to zero after unreffing our own count, because
287 // we expect to have been stored in a pointer somewhere. If we do get to
288 // zero, it's a memory leak; the way to avoid this is to call unref_delete()
289 // above instead of unref(), but this is dangerous to do from within a
290 // virtual function.
291 nassertv(get_ref_count() != 0);
292}

309TypedWritable *InternalName::
310make_from_bam(const FactoryParams &params) {
311 // The process of making a InternalName is slightly different than making
312 // other Writable objects. That is because all creation of InternalNames
313 // should be done through the make() constructor.
314 DatagramIterator scan;
315 BamReader *manager;
316
317 parse_params(params, scan, manager);
318
319 // The name is the only thing written to the data stream.
320 string name = scan.get_string();
321
322 // Make a new InternalName with that name (or get the previous one if there
323 // is one already).
324 PT(InternalName) me = make(name);
325
326 // But now we have a problem, since we have to hold the reference count and
327 // there's no way to return a TypedWritable while still holding the
328 // reference count! We work around this by explicitly upping the count, and
329 // also setting a finalize() callback to down it later.
330 me->ref();
331 manager->register_finalize(me);
332
333 return me.p();
334}
