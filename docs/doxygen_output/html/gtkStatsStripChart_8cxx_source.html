183on_click_label(int collector_index) {
184 if (collector_index < 0) {
185 // Clicking on whitespace in the graph is the same as clicking on the top
186 // label.
187 collector_index = get_collector_index();
188 }
189
190 if (collector_index == get_collector_index() && collector_index != 0) {
191 // Clicking on the top label means to go up to the parent level.

193 GtkStatsGraph::_monitor->get_client_data();
194 if (client_data->has_collector(collector_index)) {
195 const PStatCollectorDef &def =
196 client_data->get_collector_def(collector_index);
197 if (def._parent_index == 0 && get_view().get_show_level()) {
198 // Unless the parent is "Frame", and we're not a time collector.
199 } else {
200 set_collector_index(def._parent_index);
201 }
202 }
203
204 } else {
205 // Clicking on any other label means to focus on that.
206 set_collector_index(collector_index);
207 }

351void GtkStatsStripChart::
352copy_region(int start_x, int end_x, int dest_x) {
353 // We are not allowed to copy a surface onto itself, so we have to create a
354 // temporary surface to copy to.
355 end_x = std::min(end_x, get_xsize());
356 GdkWindow *window = gtk_widget_get_window(_graph_window);
357 cairo_surface_t *temp_surface =
358 gdk_window_create_similar_image_surface(window, CAIRO_FORMAT_RGB24, end_x - start_x, get_ysize(), 1);
359 {
360 cairo_t *temp_cr = cairo_create(temp_surface);
361 cairo_set_source_surface(temp_cr, _cr_surface, -start_x, 0);
362 cairo_paint(temp_cr);
363 cairo_destroy(temp_cr);
364 }
365
366 cairo_set_source_surface(_cr, temp_surface, 0, 0);
367 cairo_rectangle(_cr, dest_x, 0, end_x - start_x, get_ysize());
368 cairo_fill(_cr);
369
370 cairo_surface_destroy(temp_surface);
371
372 gdk_window_invalidate_rect(window, nullptr, FALSE);

379void GtkStatsStripChart::
380draw_slice(int x, int w, const PStatStripChart::FrameData &fdata) {
381 // Start by clearing the band first.
382 cairo_set_source_rgb(_cr, 1.0, 1.0, 1.0);
383 cairo_rectangle(_cr, x, 0, w, get_ysize());
384 cairo_fill(_cr);
385
386 double overall_time = 0.0;
387 int y = get_ysize();
388
389 FrameData::const_iterator fi;
390 for (fi = fdata.begin(); fi != fdata.end(); ++fi) {
391 const ColorData &cd = (*fi);
392 overall_time += cd._net_value;
393 cairo_set_source(_cr, get_collector_pattern(cd._collector_index,
394 _highlighted_index == cd._collector_index));
395
396 if (overall_time > get_vertical_scale()) {
397 // Off the top. Go ahead and clamp it by hand, in case it's so far off
398 // the top we'd overflow the 16-bit pixel value.
399 cairo_rectangle(_cr, x, 0, w, y);
400 cairo_fill(_cr);
401 // And we can consider ourselves done now.
402 return;
403 }
404
405 int top_y = height_to_pixel(overall_time);
406 cairo_rectangle(_cr, x, top_y, w, y - top_y);
407 cairo_fill(_cr);
408 y = top_y;
409 }

608gboolean GtkStatsStripChart::
609handle_motion(int graph_x, int graph_y) {
610 if (_drag_mode == DM_none && _potential_drag_mode == DM_none &&
611 graph_x >= 0 && graph_y >= 0 && graph_x < get_xsize() && graph_y < get_ysize()) {
612 // When the mouse is over a color bar, highlight it.
613 int collector_index = get_collector_under_pixel(graph_x, graph_y);
614 _label_stack.highlight_label(collector_index);
615 on_enter_label(collector_index);
616 }
617 else {
618 // If the mouse is in some drag mode, stop highlighting.
619 _label_stack.highlight_label(-1);
620 on_leave_label(_highlighted_index);
621 }
622
623 if (_drag_mode == DM_scale) {
624 double ratio = 1.0 - ((double)graph_y / (double)get_ysize());
625 if (ratio > 0.0) {
626 double new_scale = _drag_scale_start / ratio;
627 if (!IS_NEARLY_EQUAL(get_vertical_scale(), new_scale)) {
628 // Disable smoothing while we do this expensive operation.
629 set_average_mode(false);
630 set_vertical_scale(_drag_scale_start / ratio);
631 }
632 }
633 return TRUE;
634 }
635 else if (_drag_mode == DM_new_guide_bar) {
636 // We haven't created the new guide bar yet; we won't until the mouse
637 // comes within the graph's region.
638 if (graph_y >= 0 && graph_y < get_ysize()) {
639 set_drag_mode(DM_guide_bar);
640 _drag_guide_bar = add_user_guide_bar(pixel_to_height(graph_y));
641 return TRUE;
642 }
643 }
644 else if (_drag_mode == DM_guide_bar) {
645 move_user_guide_bar(_drag_guide_bar, pixel_to_height(graph_y));
646 return TRUE;
647 }
648
649 return GtkStatsGraph::handle_motion(graph_x, graph_y);
650}

720int GtkStatsStripChart::
721draw_guide_label(cairo_t *cr, const PStatGraph::GuideBar &bar, int last_y) {
722 switch (bar._style) {
723 case GBS_target:
724 cairo_set_source_rgb(cr, rgb_light_gray[0], rgb_light_gray[1], rgb_light_gray[2]);
725 break;
726
727 case GBS_user:
728 cairo_set_source_rgb(cr, rgb_user_guide_bar[0], rgb_user_guide_bar[1], rgb_user_guide_bar[2]);
729 break;
730
731 default:
732 cairo_set_source_rgb(cr, rgb_dark_gray[0], rgb_dark_gray[1], rgb_dark_gray[2]);
733 break;
734 }
735
736 int y = height_to_pixel(bar._height);
737 const std::string &label = bar._label;
738
739 PangoLayout *layout = gtk_widget_create_pango_layout(_scale_area, label.c_str());
740 int width, height;
741 pango_layout_get_pixel_size(layout, &width, &height);
742
743 if (bar._style != GBS_user) {
744 double from_height = pixel_to_height(y + height * _cr_scale);
745 double to_height = pixel_to_height(y - height * _cr_scale);
746 if (find_user_guide_bar(from_height, to_height) >= 0) {
747 // Omit the label: there's a user-defined guide bar in the same space.
748 g_object_unref(layout);
749 return last_y;
750 }
751 }
752
753 if (y >= 0 && y < get_ysize()) {
754 // Now convert our y to a coordinate within our drawing area.
755 int junk_x;
756
757 y /= _cr_scale;
758
759 // The y coordinate comes from the graph_window.
760 gtk_widget_translate_coordinates(_graph_window, _scale_area,
