122int PStatClientData::
123find_collector(const std::string &fullname) const {
124 // Take the last bit, we can compare it more cheaply, only check the full
125 // name if the basename matches.
126 const char *colon = strrchr(fullname.c_str(), ':');
127 std::string name(colon != nullptr ? colon + 1 : fullname.c_str());
128
129 for (int index = 0; index < get_num_collectors(); ++index) {
130 const PStatCollectorDef *def = _collectors[index]._def;
131 if (def != nullptr && def->_name == name &&
132 get_collector_fullname(index) == fullname) {
133 return index;
134 }
135 }
136
137 return -1;
138}

188bool PStatClientData::
189set_collector_has_level(int index, int thread_index, bool flag) {
190 bool any_changed = false;
191 slot_collector(index);
192 nassertr(index >= 0 && index < (int)_collectors.size(), false);
193
194 if (_collectors[index]._is_level.get_bit(thread_index) != flag) {
195 any_changed = true;
196 _collectors[index]._is_level.set_bit_to(thread_index, flag);
197 }
198
199 // Turning this on for a given collector also implicitly turns all of its
200 // ancestors.
201 if (flag) {
202 PStatCollectorDef *def = _collectors[index]._def;
203 if (def != nullptr && def->_parent_index != 0) {
204 if (set_collector_has_level(def->_parent_index, thread_index, flag)) {
205 any_changed = true;
206 }
207 }
208 }
209
210 if (any_changed) {
211 _is_dirty = true;
212 }
213
214 return any_changed;
215}

321 if (parent == child) {

342void PStatClientData::
343add_collector(PStatCollectorDef *def) {
344 slot_collector(def->_index);
345 nassertv(def->_index >= 0 && def->_index < (int)_collectors.size());
346
347 if (_collectors[def->_index]._def != nullptr) {
348 // Free the old definition, if any.
349 delete _collectors[def->_index]._def;
350 }
351
352 _collectors[def->_index]._def = def;
353 update_toplevel_collectors();
354
355 // If we already had the _is_level flag set, it should be immediately
356 // applied to all ancestors.
357 const BitArray &is_level = _collectors[def->_index]._is_level;
358 int max_threads = is_level.get_num_bits();
359 for (int thread_index = 0; thread_index < max_threads; ++thread_index) {
360 if (is_level.get_bit(thread_index)) {
361 set_collector_has_level(def->_parent_index, thread_index, true);
362 }
363 }
364
365 _is_dirty = true;
366}

372void PStatClientData::
373define_thread(int thread_index, const string &name, bool mark_alive) {
374 // A sanity check on the index number.
375 nassertv(thread_index < 1000);
376
377 // Make sure we have enough slots allocated.
378 while ((int)_threads.size() <= thread_index) {
379 _threads.push_back(Thread());
380 }
381
382 if (mark_alive) {
383 _threads[thread_index]._is_alive = true;
384 }
385
386 if (!name.empty()) {
387 _threads[thread_index]._name = name;
388 }
389
390 if (_threads[thread_index]._data.is_null()) {
391 _threads[thread_index]._data = new PStatThreadData(this);
392 }
393
394 _is_dirty = true;
395}

442 out << "[\n";
443
444 for (int thread_index = 0; thread_index < get_num_threads(); ++thread_index) {
445 const Thread &thread = _threads[thread_index];
446
447 if (thread_index == 0) {

451 else if (!thread._name.empty()) {

454 << ",\"tid\":" << thread_index << ",\"args\":{\"name\":\""
455 << thread._name << "\"}}";
456 }
457 if (thread._data == nullptr || thread._data->is_empty()) {
458 continue;
459 }
460
461 int first_frame = thread._data->get_oldest_frame_number();
462 int last_frame = thread._data->get_latest_frame_number();
463 for (int frame_number = first_frame; frame_number <= last_frame; ++frame_number) {
464 const PStatFrameData &frame_data = thread._data->get_frame(frame_number);
465 size_t num_events = frame_data.get_num_events();
466 for (size_t i = 0; i < num_events; ++i) {
467 int collector_index = frame_data.get_time_collector(i);
468 out
469 << ",\n{\"name\":\"" << get_collector_fullname(collector_index)
470 << "\",\"ts\":" << (uint64_t)(frame_data.get_time(i) * 1000000)
471 << ",\"ph\":\"" << (frame_data.is_start(i) ? 'B' : 'E') << "\""
472 << ",\"pid\":" << pid << ",\"tid\":" << thread_index << "}";
473 }
474 }
475 }

477 out << "\n]\n";
478 out.flush();

485write_datagram(Datagram &dg) const {
486 for (const Collector &collector : _collectors) {
487 PStatCollectorDef *def = collector._def;

510void PStatClientData::
511read_datagram(DatagramIterator &scan) {
512 while (scan.peek_int16() != -1) {
513 PStatCollectorDef *def = new PStatCollectorDef;
514 def->read_datagram(scan);
515 add_collector(def);
516 _collectors[def->_index]._is_level.read_datagram(scan, nullptr);
517 }
518 scan.skip_bytes(2);
519
520 int thread_index;
521 while ((thread_index = scan.get_int16()) != -1) {
522 std::string name = scan.get_string();
523 define_thread(thread_index, name, true);
524
525 _threads[thread_index]._data->read_datagram(scan, this);
526 }
527
528 update_toplevel_collectors();
529}
