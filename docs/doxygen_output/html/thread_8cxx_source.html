93PT(Thread) Thread::
94bind_thread(const std::string &name, const std::string &sync_name) {
95 PT(Thread) thread = new ExternalThread(name, sync_name);
96#ifndef HAVE_THREADS
97 Thread *current_thread = get_main_thread();
98#else
99 Thread *current_thread = ThreadImpl::bind_thread(thread);
100#endif
101 if (current_thread != nullptr &&
102 current_thread != thread.p()) {
103 // This thread already has an associated thread.
104 nassertr(current_thread->get_name() == name &&
105 current_thread->get_sync_name() == sync_name, current_thread);
106 thread = current_thread;
107 }
108 return thread;
109}

188bool Thread::
189start(ThreadPriority priority, bool joinable) {
190 nassertr(!_started, false);
191
192 if (!support_threads) {
193 thread_cat->warning()
194 << *this << " could not be started: support-threads is false.\n";
195 return false;
196 }
197
198 _joinable = joinable;
199 _started = _impl.start(priority, joinable);
200
201 if (!_started) {
202 thread_cat->warning()
203 << *this << " could not be started!\n";
204 }
205
206 return _started;
207}
