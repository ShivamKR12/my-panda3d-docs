67void CocoaGLGraphicsStateGuardian::
68get_properties(FrameBufferProperties &properties, NSOpenGLPixelFormat* pixel_format, int screen) {
69
70 properties.clear();
71
72 // Now update our framebuffer_mode and bit depth appropriately.
73 GLint double_buffer, stereo, aux_buffers, color_float, color_size,
74 alpha_size, depth_size, stencil_size, accum_size, sample_buffers,
75 samples, renderer_id, accelerated, window, pbuffer;
76
77 [pixel_format getValues:&double_buffer forAttribute:NSOpenGLPFADoubleBuffer forVirtualScreen:screen];
78 [pixel_format getValues:&stereo forAttribute:NSOpenGLPFAStereo forVirtualScreen:screen];
79 [pixel_format getValues:&aux_buffers forAttribute:NSOpenGLPFAAuxBuffers forVirtualScreen:screen];
80 [pixel_format getValues:&color_float forAttribute:NSOpenGLPFAColorFloat forVirtualScreen:screen];
81 [pixel_format getValues:&color_size forAttribute:NSOpenGLPFAColorSize forVirtualScreen:screen];
82 [pixel_format getValues:&alpha_size forAttribute:NSOpenGLPFAAlphaSize forVirtualScreen:screen];
83 [pixel_format getValues:&depth_size forAttribute:NSOpenGLPFADepthSize forVirtualScreen:screen];
84 [pixel_format getValues:&stencil_size forAttribute:NSOpenGLPFAStencilSize forVirtualScreen:screen];
85 [pixel_format getValues:&accum_size forAttribute:NSOpenGLPFAAccumSize forVirtualScreen:screen];
86 [pixel_format getValues:&sample_buffers forAttribute:NSOpenGLPFASampleBuffers forVirtualScreen:screen];
87 [pixel_format getValues:&samples forAttribute:NSOpenGLPFASamples forVirtualScreen:screen];
88 [pixel_format getValues:&renderer_id forAttribute:NSOpenGLPFARendererID forVirtualScreen:screen];
89 [pixel_format getValues:&accelerated forAttribute:NSOpenGLPFAAccelerated forVirtualScreen:screen];
90 [pixel_format getValues:&window forAttribute:NSOpenGLPFAWindow forVirtualScreen:screen];
91 [pixel_format getValues:&pbuffer forAttribute:NSOpenGLPFAPixelBuffer forVirtualScreen:screen];
92
93 properties.set_back_buffers(double_buffer);
94 properties.set_stereo(stereo);
95 properties.set_rgb_color(1);
96 properties.set_float_color(color_float);
97 properties.set_color_bits(color_size);
98 properties.set_stencil_bits(stencil_size);
99 properties.set_depth_bits(depth_size);
100 properties.set_alpha_bits(alpha_size);
101 properties.set_accum_bits(accum_size);
102 if (sample_buffers > 0) {
103 properties.set_multisamples(samples);
104 }
105 // TODO: add aux buffers
106
107 // Cocoa doesn't provide individual RGB bits. Make assumptions.
108 // Note that color_size seems to be returning values like 32, which
109 // suggests that it contains the alpha size as well.
110
111 if (color_size == 24 || color_size == 32) {
112 properties.set_rgba_bits(8, 8, 8, alpha_size);
113
114 } else if (color_size == 64) {
115 properties.set_rgba_bits(16, 16, 16, alpha_size);
116
117 } else if (color_size == 128) {
118 properties.set_rgba_bits(32, 32, 32, alpha_size);
119
120 } else if (color_size >= 3) {
121 // Assume it's giving us at least one of each.
122 properties.set_red_bits(1);
123 properties.set_green_bits(1);
124 properties.set_blue_bits(1);
125 }
126
127 // Extract the renderer ID bits and check if our renderer matches the known
128 // software renderers.
129 renderer_id &= kCGLRendererIDMatchingMask;
130 if (renderer_id == kCGLRendererGenericID ||
131 renderer_id == kCGLRendererGenericFloatID ||
132 renderer_id == kCGLRendererAppleSWID) {
133
134 properties.set_force_software(1);
135 }
136
137 if (accelerated) {
138 properties.set_force_hardware(1);
139 }
140
141 // Cautiously setting this to true. It appears that macOS framebuffers are
142 // sRGB-capable, but I don't really know how to verify this.
143 if (color_size == 32 && !color_float) {
144 properties.set_srgb_color(true);
145 }
146}

152void CocoaGLGraphicsStateGuardian::
153choose_pixel_format(const FrameBufferProperties &properties,
154 CGDirectDisplayID display,
155 bool need_pbuffer) {
156
157 _context = nil;
158 _fbprops.clear();
159
160 // Neither Cocoa nor CGL seem to have a mechanism to query the available
161 // pixel formats, unfortunately, so the only thing we can do is ask for one
162 // with the properties we have requested.
163 pvector<NSOpenGLPixelFormatAttribute> attribs;
164 attribs.reserve(15);
165
166 // Picked this up from the pyglet source - seems to be necessary to support
167 // RAGE-II, which is not compliant.
168 attribs.push_back(NSOpenGLPFAAllRenderers);
169
170 // Don't let it fall back to a different renderer.
171 attribs.push_back(NSOpenGLPFANoRecovery);
172
173 // Consider pixel formats with properties equal to or better than we
174 // requested.
175 attribs.push_back(NSOpenGLPFAMinimumPolicy);
176
177 if (!properties.is_single_buffered()) {
178 attribs.push_back(NSOpenGLPFADoubleBuffer);
179 }
180
181 if (properties.is_stereo()) {
182 attribs.push_back(NSOpenGLPFAStereo);
183 }
184
185 int aux_buffers = properties.get_aux_rgba() + properties.get_aux_hrgba() + properties.get_aux_float();
186 attribs.push_back(NSOpenGLPFAAuxBuffers);
187 attribs.push_back(aux_buffers);
188 attribs.push_back(NSOpenGLPFAColorSize);
189 attribs.push_back(properties.get_color_bits());
190
191 if (properties.get_float_color()) {
192 attribs.push_back(NSOpenGLPFAColorFloat);
193 }
194
195 // Set the depth buffer bits to 24 manually when 1 is requested.
196 // This prevents getting a depth buffer of only 16 bits when requesting 1.
197 attribs.push_back(NSOpenGLPFADepthSize);
198 if (properties.get_depth_bits() == 1) {
199 attribs.push_back(24);
200 }
201 else {
202 attribs.push_back(properties.get_depth_bits());
203 }
204
205 attribs.push_back(NSOpenGLPFAStencilSize);
206 attribs.push_back(properties.get_stencil_bits());
207
208 // Curious case - if we request anything less than 8 alpha bits, then on
209 // some ATI cards, it will grab a pixel format with just 2 alpha bits, which
210 // just shows a white window and nothing else. Might have something to do
211 // with the compositing window manager. Omitting it altogether seems to
212 // make it grab one with 8 bits, though. Dirty hack. Needs more research.
213 if (properties.get_alpha_bits() > 0) {
214 attribs.push_back(NSOpenGLPFAAlphaSize);
215 attribs.push_back(std::max(8, properties.get_alpha_bits()));
216 }
217
218 if (properties.get_multisamples() > 0) {
219 attribs.push_back(NSOpenGLPFASampleBuffers);
220 attribs.push_back(1);
221 attribs.push_back(NSOpenGLPFASamples);
222 attribs.push_back(properties.get_multisamples());
223 attribs.push_back(NSOpenGLPFAMultisample);
224 }
225
226 if (properties.get_force_software()) {
227 // Request the generic software renderer.
228 attribs.push_back(NSOpenGLPFARendererID);
229 attribs.push_back(kCGLRendererGenericFloatID);
230 }
231
232 if (properties.get_force_hardware()) {
233 attribs.push_back(NSOpenGLPFAAccelerated);
234 }
235
236 // This seems to cause getting a 3.2+ context to fail.
237 //attribs.push_back(NSOpenGLPFAWindow);
238
239 if (need_pbuffer) {
240 attribs.push_back(NSOpenGLPFAPixelBuffer);
241 }
242
243 // Required when going fullscreen, optional when windowed
244 attribs.push_back(NSOpenGLPFAScreenMask);
245 attribs.push_back(CGDisplayIDToOpenGLDisplayMask(display));
246
247 // Set OpenGL version if a minimum was requested.
248 if (gl_version.size() >= 1 && NSAppKitVersionNumber >= NSAppKitVersionNumber10_7) {
249 //NB. There is also NSOpenGLProfileVersion4_1Core, but this seems to cause
250 // a software implementation to be selected on my mac mini running 10.11.
251 if (gl_version[0] >= 4 || (gl_version.size() >= 2 && gl_version[0] == 3 && gl_version[1] >= 2)) {
252 attribs.push_back((NSOpenGLPixelFormatAttribute)99); // NSOpenGLPFAOpenGLProfile
253 attribs.push_back((NSOpenGLPixelFormatAttribute)0x3200); // NSOpenGLProfileVersion3_2Core
254 }
255 }
256
257 // End of the array
258 attribs.push_back((NSOpenGLPixelFormatAttribute)0);
259
260 // Create the format.
261 NSOpenGLPixelFormat* format = [[NSOpenGLPixelFormat alloc] initWithAttributes:&attribs[0]];
262 if (format == nil) {
263 cocoadisplay_cat.error() <<
264 "Could not find a usable pixel format.\n";
265 return;
266 }
267
268 // For now, I'm just using the first virtual screen.
269 if (cocoadisplay_cat.is_debug()) {
270 cocoadisplay_cat.debug() <<
271 "Pixel format has " << [format numberOfVirtualScreens] << " virtual screens.\n";
272 }
273 get_properties(_fbprops, format, 0);
274
275 // Don't enable sRGB unless it was explicitly requested.
276 if (!properties.get_srgb_color()) {
277 _fbprops.set_srgb_color(false);
278 }
279
280 // TODO: print out renderer
281
282 _context = [[NSOpenGLContext alloc] initWithFormat:format shareContext:_share_context];
283 _format = format;
284 if (_context == nil) {
285 cocoadisplay_cat.error() <<
286 "Failed to create OpenGL context!\n";
287 return;
288 }
289
290 // Disable vsync via the built-in mechanism, which doesn't work on Mojave
291 GLint swap = 0;
292 [_context setValues:&swap forParameter:NSOpenGLCPSwapInterval];
293
294 if (cocoadisplay_cat.is_debug()) {
295 cocoadisplay_cat.debug()
296 << "Created context " << _context << ": " << _fbprops << "\n";
297 }
298}
