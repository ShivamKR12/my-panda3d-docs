455void MacStatsPianoRoll::
456handle_motion(int graph_x, int graph_y) {
457 if (_drag_mode == DM_none && _potential_drag_mode == DM_none) {
458 // When the mouse is over a color bar, highlight it.
459 int collector_index = get_collector_under_pixel(graph_x, graph_y);
460 _label_stack.highlight_label(collector_index);
461 on_enter_label(collector_index);
462
463 /*
464 // Now we want to get a WM_MOUSELEAVE when the mouse leaves the graph
465 // window.
466 TRACKMOUSEEVENT tme = {
467 sizeof(TRACKMOUSEEVENT),
468 TME_LEAVE,
469 _graph_window,
470 0
471 };
472 TrackMouseEvent(&tme);
473 */
474 }
475 else {
476 // If the mouse is in some drag mode, stop highlighting.
477 _label_stack.highlight_label(-1);
478 on_leave_label(_highlighted_index);
479 }
480
481 if (_drag_mode == DM_scale) {
482 double ratio = (double)graph_x / (double)get_xsize();
483 if (ratio > 0.0f) {
484 set_horizontal_scale(_drag_scale_start / ratio);
485 }
486 return;
487 }
488 else if (_drag_mode == DM_new_guide_bar) {
489 // We haven't created the new guide bar yet; we won't until the mouse
490 // comes within the graph's region.
491 if (graph_x >= 0 && graph_x < get_xsize()) {
492 set_drag_mode(DM_guide_bar);
493 _drag_guide_bar = add_user_guide_bar(pixel_to_height(graph_x));
494 return;
495 }
496
497 } else if (_drag_mode == DM_guide_bar) {
498 move_user_guide_bar(_drag_guide_bar, pixel_to_height(graph_x));
499 return;
500 }
501
502 return MacStatsGraph::handle_motion(graph_x, graph_y);
503}

701void MacStatsPianoRoll::
702draw_guide_label(CGContextRef ctx, const PStatGraph::GuideBar &bar) {
703 NSColor *color;
704 if (@available(macOS 11.0, *)) {
705 color = [NSColor tertiaryLabelColor];
706 } else {
707 // Otherwise it's hard to see on the dark titlebars
708 color = [NSColor windowFrameTextColor];
709 }
710 /*
711 switch (bar._style) {
712 case GBS_target:
713 color = [NSColor colorWithDeviceRed:rgb_light_gray[0] green:rgb_light_gray[1] blue:rgb_light_gray[2] alpha:1.0];
714 break;
715
716 case GBS_user:
717 color = [NSColor colorWithDeviceRed:rgb_user_guide_bar[0] green:rgb_user_guide_bar[1] blue:rgb_user_guide_bar[2] alpha:1.0];
718 break;
719
720 default:
721 color = [NSColor colorWithDeviceRed:rgb_dark_gray[0] green:rgb_dark_gray[1] blue:rgb_dark_gray[2] alpha:1.0];
722 break;
723 }*/
724
725 int x = height_to_pixel(bar._height);
726 const std::string &label = bar._label;
727
728 const CFStringRef keys[] = {
729 (__bridge CFStringRef)NSForegroundColorAttributeName,
730 (__bridge CFStringRef)NSFontAttributeName,
731 };
732 const void *values[] = {
733 color,
734 [NSFont systemFontOfSize:0.0],
735 };
736 CFDictionaryRef attribs = CFDictionaryCreate(kCFAllocatorDefault, (const void **)keys, (const void **)values, 2, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
737
738 CFStringRef str = CFStringCreateWithCString(kCFAllocatorDefault, label.c_str(), kCFStringEncodingUTF8);
739 CFAttributedStringRef astr = CFAttributedStringCreate(kCFAllocatorDefault, str, attribs);
740 CFRelease(attribs);
741 CFRelease(str);
742
743 CTLineRef line = CTLineCreateWithAttributedString(astr);
744 CFRelease(astr);
745 CGRect bounds = CTLineGetImageBounds(line, ctx);
746 int width = bounds.size.width;
747
748 if (bar._style != GBS_user) {
749 double from_height = pixel_to_height(x - width);
750 double to_height = pixel_to_height(x + width);
751 if (find_user_guide_bar(from_height, to_height) >= 0) {
752 // Omit the label: there's a user-defined guide bar in the same space.
753 CFRelease(line);
754 return;
755 }
756 }
757
758 if (x >= 0 && x < get_xsize()) {
759 CGContextSetTextPosition(ctx, x + 6, 6);
