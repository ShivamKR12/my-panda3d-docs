55void WebGLGraphicsStateGuardian::
56choose_pixel_format(const FrameBufferProperties &properties,
57 const char *target) {
58
59 nassertv(_context == 0);
60
61 EmscriptenWebGLContextAttributes attribs;
62 emscripten_webgl_init_context_attributes(&attribs);
63
64 attribs.alpha = (properties.get_alpha_bits() > 0);
65 attribs.depth = (properties.get_depth_bits() > 0);
66 attribs.stencil = (properties.get_stencil_bits() > 0);
67 attribs.majorVersion = 2;
68 attribs.minorVersion = 0;
69 attribs.enableExtensionsByDefault = false;
70
71 EMSCRIPTEN_WEBGL_CONTEXT_HANDLE result;
72 result = emscripten_webgl_create_context(target, &attribs);
73
74 if (result <= 0) {
75 // Fall back to WebGL 1.
76 attribs.majorVersion = 1;
77 result = emscripten_webgl_create_context(target, &attribs);
78 }
79
80 if (result > 0) {
81 _context = result;
82 _have_context = true;
83
84 // We may lose the WebGL context at any time, at which time we have to be
85 // prepared to drop all resources.
86 emscripten_set_webglcontextlost_callback(target, (void *)this, false, &on_context_event);
87 emscripten_set_webglcontextrestored_callback(target, (void *)this, false, &on_context_event);
88 } else {
89 webgldisplay_cat.error() << "Context creation failed.\n";
90 }

105EM_BOOL WebGLGraphicsStateGuardian::
106on_context_event(int type, const void *, void *user_data) {
107 WebGLGraphicsStateGuardian *gsg = (WebGLGraphicsStateGuardian *)user_data;
108 nassertr(gsg != NULL, false);
109
110 if (type == EMSCRIPTEN_EVENT_WEBGLCONTEXTLOST) {
111 webgldisplay_cat.info() << "WebGL context lost!\n";
112 gsg->context_lost();
113 return true;
114
115 } else if (type == EMSCRIPTEN_EVENT_WEBGLCONTEXTRESTORED) {
116 webgldisplay_cat.info() << "WebGL context restored!\n";
117 gsg->reset();
118 return true;
119 }
120
121 return false;
122}
