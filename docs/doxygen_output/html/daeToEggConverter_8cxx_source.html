113bool DAEToEggConverter::
114convert_file(const Filename &filename) {
115 // Reset stuff
116 clear_error();
117 _joints.clear();
118 if (_error_handler == nullptr) {
119 _error_handler = new FUErrorSimpleHandler;
120 }
121
122 // The default coordinate system is Y-up
123 if (_egg_data->get_coordinate_system() == CS_default) {
124 _egg_data->set_coordinate_system(CS_yup_right);
125 }
126
127 // Read the file
128 FCollada::Initialize();
129 _document = FCollada::LoadDocument(filename.to_os_specific().c_str());
130 if (_document == nullptr) {
131 daeegg_cat.error() << "Failed to load document: " << _error_handler->GetErrorString() << endl;
132 FCollada::Release();
133 return false;
134 }
135 // Make sure the file uses consistent coordinate system and length
136 if (_document->GetAsset() != nullptr) {
137 FCDocumentTools::StandardizeUpAxisAndLength(_document);
138 }
139
140 // Process the scene
141 process_asset();
142 PT(EggGroup) scene_group;
143 string model_name = _character_name;
144
145 FCDSceneNode* visual_scene = _document->GetVisualSceneInstance();
146 if (visual_scene != nullptr) {
147 if (model_name.empty()) {
148 // By lack of anything better...
149 model_name = FROM_FSTRING(visual_scene->GetName());
150 }
151 scene_group = new EggGroup(model_name);
152 _egg_data->add_child(scene_group);
153
154 for (size_t ch = 0; ch < visual_scene->GetChildrenCount(); ++ch) {
155 process_node(scene_group, visual_scene->GetChild(ch));
156 }
157 } else {
158 daeegg_cat.warning()
159 << "No visual scene instance found in COLLADA document.\n";
160 }
161
162 // Now process the characters. This depends on information from collected
163 // joints, which is why it's done in a second step.
164 if (get_animation_convert() != AC_none) {
165 Characters::iterator it;
166 DaeCharacter *character;
167 for (it = _characters.begin(); it != _characters.end(); ++it) {
168 character = *it;
169 if (get_animation_convert() != AC_chan) {
170 character->bind_joints(_joints);
171
172 const FCDGeometryMesh *mesh = character->_skin_mesh;
173
174 if (mesh != nullptr) {
175 PT(DaeMaterials) materials = new DaeMaterials(character->_instance);
176 if (daeegg_cat.is_spam()) {
177 daeegg_cat.spam() << "Processing mesh for controller\n";
178 }
179 process_mesh(character->_node_group, mesh, materials, character);
180 }
181 }
182 }
183
184 // Put the joints in bind pose.
185 for (size_t ch = 0; ch < visual_scene->GetChildrenCount(); ++ch) {
186 character->adjust_joints(visual_scene->GetChild(ch), _joints, LMatrix4d::ident_mat());
187 }
188
189 if (scene_group != nullptr) {
190 // Mark the scene as character.
191 if (get_animation_convert() == AC_chan) {
192 _egg_data->remove_child(scene_group);
193 } else {
194 scene_group->set_dart_type(EggGroup::DT_default);
195 }
196 }
197
198 if (get_animation_convert() != AC_model) {
199 _table = new EggTable();
200 _table->set_table_type(EggTable::TT_table);
201 _egg_data->add_child(_table);
202
203 PT(EggTable) bundle = new EggTable(model_name);
204 bundle->set_table_type(EggTable::TT_bundle);
205 _table->add_child(bundle);
206
207 PT(EggTable) skeleton = new EggTable("<skeleton>");
208 skeleton->set_table_type(EggTable::TT_table);
209 bundle->add_child(skeleton);
210
211 pset<float> keys;
212
213 Characters::iterator it;
214 DaeCharacter *character;
215 for (it = _characters.begin(); it != _characters.end(); ++it) {
216 character = *it;
217
218 // Collect key frame timings.
219 if (get_animation_convert() == AC_both ||
220 get_animation_convert() == AC_chan) {
221 character->collect_keys(keys);
222 }
223 }
224
225 if (_frame_inc != 0.0) {
226 // A frame increment was given, this means that we have to sample the
227 // animation.
228 float start, end;
229 if (_end_frame != _start_frame) {
230 start = _start_frame;
231 end = _end_frame;
232 } else {
233 // No range was given. Infer the frame range from the keys.
234 start = *keys.begin();
235 end = *keys.rbegin();
236 }
237 keys.clear();
238
239 for (float t = start; t <= end; t += _frame_inc) {
240 keys.insert(t);
241 }
242 } else {
243 // No sampling parameters given; not necessarily a failure, since the
244 // animation may already be sampled. We use the key frames as
245 // animation frames.
246 if (_end_frame != 0.0) {
247 // An end frame was given, chop off all keys after that.
248 float end = _end_frame;
249 pset<float>::iterator ki;
250 for (ki = keys.begin(); ki != keys.end(); ++ki) {
251 if (*ki > end && !IS_THRESHOLD_EQUAL(*ki, end, 0.001)) {
252 keys.erase(ki, keys.end());
253 break;
254 }
255 }
256 }
257 if (_start_frame != 0.0) {
258 // A start frame was given, chop off all keys before that.
259 float start = _start_frame;
260 pset<float>::iterator ki;
261 for (ki = keys.begin(); ki != keys.end(); ++ki) {
262 if (*ki > start && !IS_THRESHOLD_EQUAL(*ki, start, 0.001)) {
263 keys.erase(keys.begin(), ki);
264 break;
265 }
266 }
267 }
268
269 // Check that this does indeed look like a sampled animation; if not,
270 // issue an appropriate warning.
271 pset<float>::const_iterator ki = keys.begin();
272 if (ki != keys.end()) {
273 float last = *ki;
274 float diff = 0;
275
276 for (++ki; ki != keys.end(); ++ki) {
277 if (diff != 0 && !IS_THRESHOLD_EQUAL((*ki - last), diff, 0.001)) {
278 daeegg_cat.error()
279 << "This does not appear to be a sampled animation.\n"
280 << "Specify the -sf, -ef and -if options to indicate how the "
281 << "animations should be sampled.\n";
282 break;
283 }
284 diff = (*ki - last);
285 last = *ki;
286 }
287 }
288 }
289
290 // It doesn't really matter which character we grab for this as it'll
291 // iterate over the whole graph right now anyway.
292 for (size_t ch = 0; ch < visual_scene->GetChildrenCount(); ++ch) {
293 character->build_table(skeleton, visual_scene->GetChild(ch), keys);
294 }
295 }
296 }
297
298 // Clean up and return
299 SAFE_DELETE(visual_scene);
300 SAFE_DELETE(_document);
301 FCollada::Release();
302 return true;
303}

311DistanceUnit DAEToEggConverter::
312get_input_units() {
313 if (IS_NEARLY_EQUAL(_unit_meters, 0.001)) {
314 return DU_millimeters;
315 }
316 if (IS_NEARLY_EQUAL(_unit_meters, 0.01)) {
317 return DU_centimeters;

329 return DU_feet;

331 if (IS_NEARLY_EQUAL(_unit_meters, 0.0254)) {
332 return DU_inches;
333 }
334 if (IS_NEARLY_EQUAL(_unit_meters, 1852.0)) {
335 return DU_nautical_miles;
336 }
337 if (IS_NEARLY_EQUAL(_unit_meters, 5280.0 * 12.0 * 0.0254)) {
338 return DU_statute_miles;
339 }
340
341 // Whatever.
342 return DU_invalid;

345void DAEToEggConverter::
346process_asset() {
347 const FCDAsset *asset = _document->GetAsset();
348 if (_document->GetAsset() == nullptr) {
349 return;
350 }
351
352 _unit_name = FROM_FSTRING(asset->GetUnitName());
353 _unit_meters = asset->GetUnitConversionFactor();
354
355 // Read out the coordinate system
356 FMVector3 up_axis = asset->GetUpAxis();
357
358 if (up_axis == FMVector3(0, 1, 0)) {
359 _egg_data->set_coordinate_system(CS_yup_right);
360
361 } else if (up_axis == FMVector3(0, 0, 1)) {
362 _egg_data->set_coordinate_system(CS_zup_right);
363
364 } else {
365 _egg_data->set_coordinate_system(CS_invalid);
366 daeegg_cat.warning() << "Unrecognized coordinate system!\n";
367 }

370// Process the node. If forced is true, it will even process it if its known
371// to be a skeleton root.
372void DAEToEggConverter::
373process_node(EggGroupNode *parent, const FCDSceneNode* node, bool forced) {
374 nassertv(node != nullptr);
375 string node_id = FROM_FSTRING(node->GetDaeId());
376 if (daeegg_cat.is_spam()) {
377 daeegg_cat.spam() << "Processing node with ID '" << node_id << "'" << endl;
378 }
379
380 // Create an egg group for this node
381 PT(EggGroup) node_group = new EggGroup(FROM_FSTRING(node->GetDaeId()));
382 process_extra(node_group, node->GetExtra());
383 parent->add_child(node_group);
384
385 // Check if its a joint
386 if (node->IsJoint()) {
387 string sid = FROM_FSTRING(node->GetSubId());
388 node_group->set_group_type(EggGroup::GT_joint);
389
390 if (!_joints.insert(DaeCharacter::JointMap::value_type(sid,
391 DaeCharacter::Joint(node_group, node))).second) {
392 daeegg_cat.error()
393 << "Joint with sid " << sid << " occurs more than once!\n";
394 }
395 }
396
397 // Loop through the transforms and apply them (in reverse order)
398 for (size_t tr = node->GetTransformCount(); tr > 0; --tr) {
399 apply_transform(node_group, node->GetTransform(tr - 1));
400 }
401 // node_group->set_transform3d(convert_matrix(node->ToMatrix()));
402
403 // Loop through the instances and process them
404 for (size_t in = 0; in < node->GetInstanceCount(); ++in) {
405 process_instance(node_group, node->GetInstance(in));
406 }
407
408 // Loop through the children and recursively process them
409 for (size_t ch = 0; ch < node->GetChildrenCount(); ++ch) {
410 process_node(DCAST(EggGroupNode, node_group), node->GetChild(ch));
411 }
412
413 // Loop through any possible scene node instances and process those, too.
414 for (size_t in = 0; in < node->GetInstanceCount(); ++in) {
415 const FCDEntity *entity = node->GetInstance(in)->GetEntity();
416 if (entity && entity->GetType() == FCDEntity::SCENE_NODE) {
417 process_node(node_group, (const FCDSceneNode*) entity);
418 }
419 }

422void DAEToEggConverter::
423process_instance(EggGroup *parent, const FCDEntityInstance* instance) {
424 nassertv(instance != nullptr);
425 nassertv(instance->GetEntity() != nullptr);
426 // Check what kind of instance this is
427 switch (instance->GetType()) {
428 case FCDEntityInstance::GEOMETRY:
429 {
430 if (get_animation_convert() != AC_chan) {
431 const FCDGeometry* geometry = (const FCDGeometry*) instance->GetEntity();
432 assert(geometry != nullptr);
433 if (geometry->IsMesh()) {
434 // Now, handle the mesh.
435 process_mesh(parent, geometry->GetMesh(), new DaeMaterials((const FCDGeometryInstance*) instance));
436 }
437 if (geometry->IsSpline()) {
438 process_spline(parent, FROM_FSTRING(geometry->GetName()), const_cast<FCDGeometrySpline*> (geometry->GetSpline()));
439 }
440 }
441 }
442 break;
443
444 case FCDEntityInstance::CONTROLLER:
445 // Add the dart tag and process the controller instance
446 // parent->set_dart_type(EggGroup::DT_default);
447 process_controller(parent, (const FCDControllerInstance*) instance);
448 break;
449
450 case FCDEntityInstance::MATERIAL:
451 // We don't process this directly, handled per-geometry instead.
452 break;
453
454 case FCDEntityInstance::SIMPLE:
455 {
456 // Grab the entity and check its type.
457 const FCDEntity* entity = instance->GetEntity();
458 if (entity->GetType() != FCDEntity::SCENE_NODE) {
459 daeegg_cat.warning() << "Unsupported entity type found" << endl;
460 }
461 }
462 break;
463
464 default:
465 daeegg_cat.warning() << "Unsupported instance type found" << endl;
466 }
467}

469// Processes the given mesh.
470void DAEToEggConverter::
471process_mesh(EggGroup *parent, const FCDGeometryMesh* mesh,
472 DaeMaterials *materials, DaeCharacter *character) {
473
474 nassertv(mesh != nullptr);
475 if (daeegg_cat.is_debug()) {
476 daeegg_cat.debug() << "Processing mesh with id " << FROM_FSTRING(mesh->GetDaeId()) << endl;
477 }
478
479 // Create the egg stuff to hold this mesh
480 PT(EggGroup) mesh_group = new EggGroup(FROM_FSTRING(mesh->GetDaeId()));
481 parent->add_child(mesh_group);
482 PT(EggVertexPool) mesh_pool = new EggVertexPool(FROM_FSTRING(mesh->GetDaeId()));
483 mesh_group->add_child(mesh_pool);
484
485 // First retrieve the vertex source
486 if (mesh->GetSourceCount() == 0) {
487 if (daeegg_cat.is_debug()) {
488 daeegg_cat.debug() << "Mesh with id " << FROM_FSTRING(mesh->GetDaeId()) << " has no sources" << endl;
489 }
490 return;
491 }
492 const FCDGeometrySource* vsource = mesh->FindSourceByType(FUDaeGeometryInput::POSITION);
493 if (vsource == nullptr) {
494 if (daeegg_cat.is_debug()) {
495 daeegg_cat.debug() << "Mesh with id " << FROM_FSTRING(mesh->GetDaeId()) << " has no source for POSITION data" << endl;
496 }
497 return;
498 }
499
500 // Loop through the polygon groups and add them
501 if (daeegg_cat.is_spam()) {
502 daeegg_cat.spam() << "Mesh with id " << FROM_FSTRING(mesh->GetDaeId()) << " has " << mesh->GetPolygonsCount() << " polygon groups" << endl;
503 }
504 if (mesh->GetPolygonsCount() == 0) return;
505
506 // This is an array of pointers, I know. But since they are refcounted, I
507 // don't have a better idea.
508 PT(EggGroup) *primitive_holders = new PT(EggGroup) [mesh->GetPolygonsCount()];
509 for (size_t gr = 0; gr < mesh->GetPolygonsCount(); ++gr) {
510 const FCDGeometryPolygons* polygons = mesh->GetPolygons(gr);
511 string material_semantic = FROM_FSTRING(polygons->GetMaterialSemantic());
512
513 // Stores which group holds the primitives.
514 PT(EggGroup) primitiveholder;
515 // If we have materials, make a group for each material. Then, apply the
516 // material's per-group stuff.
517 if (materials != nullptr && (!polygons->GetMaterialSemantic().empty()) && mesh->GetPolygonsCount() > 1) {
518 // primitiveholder = new EggGroup(FROM_FSTRING(mesh->GetDaeId()) + "." +
519 // material_semantic);
520 primitiveholder = new EggGroup;
521 mesh_group->add_child(primitiveholder);
522 } else {
523 primitiveholder = mesh_group;
524 }
525 primitive_holders[gr] = primitiveholder;
526 // Apply the per-group data of the materials, if we have it.
527 if (materials != nullptr) {
528 materials->apply_to_group(material_semantic, primitiveholder, _invert_transparency);
529 }
530 // Find the position sources
531 const FCDGeometryPolygonsInput* pinput = polygons->FindInput(FUDaeGeometryInput::POSITION);
532 assert(pinput != nullptr);
533 const uint32* indices = pinput->GetIndices();
534 // Find the normal sources
535 const FCDGeometrySource* nsource = mesh->FindSourceByType(FUDaeGeometryInput::NORMAL);
536 const FCDGeometryPolygonsInput* ninput = polygons->FindInput(FUDaeGeometryInput::NORMAL);
537 const uint32* nindices;
538 if (ninput != nullptr) nindices = ninput->GetIndices();
539 // Find texcoord sources
540 const FCDGeometrySource* tcsource = mesh->FindSourceByType(FUDaeGeometryInput::TEXCOORD);
541 const FCDGeometryPolygonsInput* tcinput = polygons->FindInput(FUDaeGeometryInput::TEXCOORD);
542 const uint32* tcindices;
543 if (tcinput != nullptr) tcindices = tcinput->GetIndices();
544 // Find vcolor sources
545 const FCDGeometrySource* csource = mesh->FindSourceByType(FUDaeGeometryInput::COLOR);
546 const FCDGeometryPolygonsInput* cinput = polygons->FindInput(FUDaeGeometryInput::COLOR);
547 const uint32* cindices;
548 if (cinput != nullptr) cindices = cinput->GetIndices();
549 // Find binormal sources
550 const FCDGeometrySource* bsource = mesh->FindSourceByType(FUDaeGeometryInput::TEXBINORMAL);
551 const FCDGeometryPolygonsInput* binput = polygons->FindInput(FUDaeGeometryInput::TEXBINORMAL);
552 const uint32* bindices;
553 if (binput != nullptr) bindices = binput->GetIndices();
554 // Find tangent sources
555 const FCDGeometrySource* tsource = mesh->FindSourceByType(FUDaeGeometryInput::TEXTANGENT);
556 const FCDGeometryPolygonsInput* tinput = polygons->FindInput(FUDaeGeometryInput::TEXTANGENT);
557 const uint32* tindices;
558 if (tinput != nullptr) tindices = tinput->GetIndices();
559 // Get a name for potential coordinate sets
560 string tcsetname;
561 if (materials != nullptr && tcinput != nullptr) {
562 if (daeegg_cat.is_debug()) {
563 daeegg_cat.debug()
564 << "Assigning texcoord set " << tcinput->GetSet()
565 << " to semantic '" << material_semantic << "'\n";
566 }
567 tcsetname = materials->get_uvset_name(material_semantic,
568 FUDaeGeometryInput::TEXCOORD, tcinput->GetSet());
569 }
570 string tbsetname;
571 if (materials != nullptr && binput != nullptr) {
572 if (daeegg_cat.is_debug()) {
573 daeegg_cat.debug()
574 << "Assigning texbinormal set " << binput->GetSet()
575 << " to semantic '" << material_semantic << "'\n";
576 }
577 tbsetname = materials->get_uvset_name(material_semantic,
578 FUDaeGeometryInput::TEXBINORMAL, binput->GetSet());
579 }
580 string ttsetname;
581 if (materials != nullptr && tinput != nullptr) {
582 if (daeegg_cat.is_debug()) {
583 daeegg_cat.debug()
584 << "Assigning textangent set " << tinput->GetSet()
585 << " to semantic '" << material_semantic << "'\n";
586 }
587 ttsetname = materials->get_uvset_name(material_semantic,
588 FUDaeGeometryInput::TEXTANGENT, tinput->GetSet());
589 }
590 // Loop through the indices and add the vertices.
591 for (size_t ix = 0; ix < pinput->GetIndexCount(); ++ix) {
592 PT_EggVertex vertex = mesh_pool->make_new_vertex();
593 const float* data = &vsource->GetData()[indices[ix]*3];
594 vertex->set_pos(LPoint3d(data[0], data[1], data[2]));
595
596 if (character != nullptr) {
597 // If this is skinned geometry, add the vertex influences.
598 character->influence_vertex(indices[ix], vertex);
599 }
600
601 // Process the normal
602 if (nsource != nullptr && ninput != nullptr) {
603 assert(nsource->GetStride() == 3);
604 data = &nsource->GetData()[nindices[ix]*3];
605 vertex->set_normal(LVecBase3d(data[0], data[1], data[2]));
606 }
607 // Process the texcoords
608 if (tcsource != nullptr && tcinput != nullptr) {
609 assert(tcsource->GetStride() == 2 || tcsource->GetStride() == 3);
610 data = &tcsource->GetData()[tcindices[ix]*tcsource->GetStride()];
611 if (tcsource->GetStride() == 2) {
612 vertex->set_uv(tcsetname, LPoint2d(data[0], data[1]));
613 } else {
614 vertex->set_uvw(tcsetname, LPoint3d(data[0], data[1], data[2]));
615 }
616 }
617 // Process the color
618 if (csource != nullptr && cinput != nullptr) {
619 assert(csource->GetStride() == 3 || csource->GetStride() == 4);
620 if (csource->GetStride() == 3) {
621 data = &csource->GetData()[cindices[ix]*3];
622 vertex->set_color(LColor(data[0], data[1], data[2], 1.0f));
623 } else {
624 data = &csource->GetData()[cindices[ix]*4];
625 vertex->set_color(LColor(data[0], data[1], data[2], data[3]));
626 }
627 }
628 // Possibly add a UV object
629 if ((bsource != nullptr && binput != nullptr) || (tsource != nullptr && tinput != nullptr)) {
630 if (bsource != nullptr && binput != nullptr) {
631 assert(bsource->GetStride() == 3);
632 data = &bsource->GetData()[bindices[ix]*3];
633 PT(EggVertexUV) uv_obj = vertex->modify_uv_obj(tbsetname);
634 if (uv_obj == nullptr) {
635 uv_obj = new EggVertexUV(tbsetname, LTexCoordd());
636 }
637 uv_obj->set_binormal(LVecBase3d(data[0], data[1], data[2]));
638 }
639 if (tsource != nullptr && tinput != nullptr) {
640 assert(tsource->GetStride() == 3);
641 data = &tsource->GetData()[tindices[ix]*3];
642 PT(EggVertexUV) uv_obj = vertex->modify_uv_obj(ttsetname);
643 if (uv_obj == nullptr) {
644 uv_obj = new EggVertexUV(ttsetname, LTexCoordd());
645 }
646 uv_obj->set_tangent(LVecBase3d(data[0], data[1], data[2]));
647 }
648 }
649 vertex->transform(parent->get_node_to_vertex());
650 }
651 }
652 // Loop again for the polygons
653 for (size_t gr = 0; gr < mesh->GetPolygonsCount(); ++gr) {
654 const FCDGeometryPolygons* polygons = mesh->GetPolygons(gr);
655 // Now loop through the faces
656 uint32 offset = 0;
657 for (size_t fa = 0; fa < polygons->GetFaceVertexCountCount(); ++fa) {
658 PT(EggPrimitive) primitive = nullptr;
659 // Create a primitive that matches the fcollada type
660 switch (polygons->GetPrimitiveType()) {
661 case FCDGeometryPolygons::LINES:
662 primitive = new EggLine();
663 break;
664 case FCDGeometryPolygons::POLYGONS:
665 primitive = new EggPolygon();
666 break;
667 case FCDGeometryPolygons::TRIANGLE_FANS:
668 primitive = new EggTriangleFan();
669 break;
670 case FCDGeometryPolygons::TRIANGLE_STRIPS:
671 primitive = new EggTriangleStrip();
672 break;
673 case FCDGeometryPolygons::POINTS:
674 primitive = new EggPoint();
675 break;
676 case FCDGeometryPolygons::LINE_STRIPS:
677 daeegg_cat.warning() << "Linestrips not yet supported!" << endl;
678 break;
679 default:
680 daeegg_cat.warning() << "Unsupported primitive type found!" << endl;
681 }
682 if (primitive != nullptr) {
683 primitive_holders[gr]->add_child(primitive);
684 if (materials != nullptr) {
685 materials->apply_to_primitive(FROM_FSTRING(polygons->GetMaterialSemantic()), primitive);
686 }
687 for (size_t ve = 0; ve < polygons->GetFaceVertexCount(fa); ++ve) {
688 assert(mesh_pool->has_vertex(ve + polygons->GetFaceVertexOffset() + offset));
689 primitive->add_vertex(mesh_pool->get_vertex(ve + polygons->GetFaceVertexOffset() + offset));
690 }
691 }
692 offset += polygons->GetFaceVertexCount(fa);
693 }
694 }
695 delete[] primitive_holders;
696}
