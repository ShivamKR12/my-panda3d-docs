125int SpriteParticleRenderer::
126extract_textures_from_node(const NodePath &node_path, NodePathCollection &np_col, TextureCollection &tex_col) {
127 NodePath tex_node_path = node_path;
128 NodePath geom_node_path;
129
130 // Look for a sequence node first, in case they want animated texture
131 // sprites
132 if (!tex_node_path.is_empty() && tex_node_path.node()->get_type() != SequenceNode::get_class_type()) {
133 tex_node_path = node_path.find("**/+SequenceNode");
134 }
135
136 // Nodepath contains a sequence node, attempt to read its textures.
137 if (!tex_node_path.is_empty()) {
138 int frame_count = tex_node_path.get_num_children();
139 // We do it this way in order to preserve the order of the textures in the
140 // sequence. If we use a find_all_textures() that order is lost.
141 for (int i = 0; i < frame_count; ++i) {
142 geom_node_path = tex_node_path.get_child(i);
143 if (!geom_node_path.is_empty()) {
144 // Since this is a SequenceNode, there will be only one texture on
145 // this geom_node_path.
146 tex_col.add_textures_from(geom_node_path.find_all_textures());
147 np_col.add_path(geom_node_path);
148 }
149 }
150 // If unsuccessful, try again as if the node were a normal GeomNode.
151 if (tex_col.get_num_textures() == 0) {
152 geom_node_path = NodePath();
153 tex_col.clear();
154 np_col.clear();
155 }
156 }
157
158 // If a sequence node is not found, we just want to look for a regular geom
159 // node.
160 if (geom_node_path.is_empty()) {
161 // Find the first GeomNode.
162 if (!node_path.is_empty() && node_path.node()->get_type() != GeomNode::get_class_type()) {
163 geom_node_path = node_path.find("**/+GeomNode");
164 if (geom_node_path.is_empty()) {
165 particlesystem_cat.error();
166 return 0;
167 }
168 } else {
169 geom_node_path = node_path;
170 }
171
172 // Grab the first texture.
173 tex_col.add_texture(geom_node_path.find_texture("*"));
174 if (tex_col.get_num_textures() < 1) {
175 particlesystem_cat.error()
176 << geom_node_path << " does not contain a texture.\n";
177 return 0;
178 } else {
179 np_col.add_path(geom_node_path);
180 }
181 }
182 return 1;
183}

265void SpriteParticleRenderer::
266add_from_node(const NodePath &node_path, bool size_from_texels, bool resize) {
267 nassertv(!node_path.is_empty());
268
269 NodePathCollection np_col;
270 TextureCollection tex_col;
271
272 if (_anims.empty())
273 resize = true;
274
275 // Load the found textures into the renderer.
276 if (extract_textures_from_node(node_path,np_col,tex_col)) {
277 pvector< LTexCoord > ll,ur;
278 GeomNode *gnode = nullptr;
279 const Geom *geom;
280 const GeomPrimitive *primitive;
281
282 for (int i = 0; i < np_col.get_num_paths(); ++i) {
283 // Get the node from which we'll extract the geometry information.
284 gnode = DCAST(GeomNode, np_col[i].node());
285
286 // Now examine the UV's of the first Geom within the GeomNode.
287 nassertv(gnode->get_num_geoms() > 0);
288 geom = gnode->get_geom(0);
289
290 bool got_texcoord = false;
291 LTexCoord min_uv(0.0f, 0.0f);
292 LTexCoord max_uv(0.0f, 0.0f);
293
294 GeomVertexReader texcoord(geom->get_vertex_data(),
295 InternalName::get_texcoord());
296 if (texcoord.has_column()) {
297 for (size_t pi = 0; pi < geom->get_num_primitives(); ++pi) {
298 primitive = geom->get_primitive(pi);
299 for (int vi = 0; vi < primitive->get_num_vertices(); ++vi) {
300 int vert = primitive->get_vertex(vi);
301 texcoord.set_row_unsafe(vert);
302
303 if (!got_texcoord) {
304 min_uv = max_uv = texcoord.get_data2();
305 got_texcoord = true;
306
307 } else {
308 const LVecBase2 &uv = texcoord.get_data2();
309
310 min_uv[0] = min(min_uv[0], uv[0]);
311 max_uv[0] = max(max_uv[0], uv[0]);
312 min_uv[1] = min(min_uv[1], uv[1]);
313 max_uv[1] = max(max_uv[1], uv[1]);
314 }
315 }
316 }
317 }
318
319 if (got_texcoord) {
320 // We don't really pay attention to orientation of UV's here; a minor
321 // flaw. We assume the minimum is in the lower-left, and the maximum
322 // is in the upper-right.
323 ll.push_back(min_uv);
324 ur.push_back(max_uv);
325 }
326 }
327
328 _anims.push_back(new SpriteAnim(tex_col,ll,ur));
329
330 if (resize) {
331 gnode = DCAST(GeomNode, np_col[0].node());
332 geom = gnode->get_geom(0);
333
334 bool got_vertex = false;
335 LVertex min_xyz(0.0f, 0.0f, 0.0f);
336 LVertex max_xyz(0.0f, 0.0f, 0.0f);
337
338 GeomVertexReader vertex(geom->get_vertex_data(),
339 InternalName::get_vertex());
340 if (vertex.has_column()) {
341 for (size_t pi = 0; pi < geom->get_num_primitives(); ++pi) {
342 primitive = geom->get_primitive(pi);
343 for (int vi = 0; vi < primitive->get_num_vertices(); ++vi) {
344 int vert = primitive->get_vertex(vi);
345 vertex.set_row_unsafe(vert);
346
347 if (!got_vertex) {
348 min_xyz = max_xyz = vertex.get_data3();
349 got_vertex = true;
350
351 } else {
352 const LVecBase3 &xyz = vertex.get_data3();
353
354 min_xyz[0] = min(min_xyz[0], xyz[0]);
355 max_xyz[0] = max(max_xyz[0], xyz[0]);
356 min_xyz[1] = min(min_xyz[1], xyz[1]);
357 max_xyz[1] = max(max_xyz[1], xyz[1]);
358 min_xyz[2] = min(min_xyz[2], xyz[2]);
359 max_xyz[2] = max(max_xyz[2], xyz[2]);
360 }
361 }
362 }
363 }
364
365 if (got_vertex) {
366 PN_stdfloat width = max_xyz[0] - min_xyz[0];
367 PN_stdfloat height = max(max_xyz[1] - min_xyz[1],
368 max_xyz[2] - min_xyz[2]);
369
370 if (size_from_texels) {
371 // If size_from_texels is true, we get the particle size from the
372 // number of texels in the source image.
373 PN_stdfloat y_texels = _anims[0]->get_frame(0)->get_y_size() * fabs(_anims[0]->get_ur(0)[1] - _anims[0]->get_ll(0)[1]);
374 set_size(y_texels * width / height, y_texels);
375 } else {
376 // If size_from_texels is false, we get the particle size from the
377 // size of the polygon.
378 set_size(width, height);
379 }
380
381 } else {
382 // With no vertices, just punt.
383 set_size(1.0f, 1.0f);
384 }
385 }
386 init_geoms();
387 }
388}

405void SpriteParticleRenderer::
406init_geoms() {
407 CPT(RenderState) state = _render_state;
408 SpriteAnim *anim;
409 int anim_count = _anims.size();
410 int i,j;
411
412 // Setup format
413 PT(GeomVertexArrayFormat) array_format;
414
415 if (vertex_colors_prefer_packed) {
416 array_format = new GeomVertexArrayFormat
417 (InternalName::get_vertex(), 3, Geom::NT_stdfloat, Geom::C_point,
418 InternalName::get_color(), 1, Geom::NT_packed_dabc, Geom::C_color);
419 }
420 else {
421 array_format = new GeomVertexArrayFormat
422 (InternalName::get_vertex(), 3, Geom::NT_stdfloat, Geom::C_point,
423 InternalName::get_color(), 4, Geom::NT_uint8, Geom::C_color);
424 }
425
426 if (_animate_theta || _theta != 0.0f) {
427 array_format->add_column
428 (InternalName::get_rotate(), 1, Geom::NT_stdfloat, Geom::C_other);
429 }
430
431 _base_y_scale = _initial_y_scale;
432 _aspect_ratio = _width / _height;
433
434 PN_stdfloat final_x_scale = _animate_x_ratio ? _final_x_scale : _initial_x_scale;
435 PN_stdfloat final_y_scale = _animate_y_ratio ? _final_y_scale : _initial_y_scale;
436
437 if (_animate_y_ratio) {
438 _base_y_scale = max(_initial_y_scale, _final_y_scale);
439 array_format->add_column
440 (InternalName::get_size(), 1, Geom::NT_stdfloat, Geom::C_other);
441 }
442
443 if (_aspect_ratio * _initial_x_scale != _initial_y_scale ||
444 _aspect_ratio * final_x_scale != final_y_scale) {
445 array_format->add_column
446 (InternalName::get_aspect_ratio(), 1, Geom::NT_stdfloat,
447 Geom::C_other);
448 }
449
450 CPT(GeomVertexFormat) format = GeomVertexFormat::register_format
451 (new GeomVertexFormat(array_format));
452
453 // Reset render() data structures
454 for (i = 0; i < (int)_ttl_count.size(); ++i) {
455 PANDA_FREE_ARRAY(_ttl_count[i]);
456 }
457 _anim_size.resize(anim_count);
458 _ttl_count.clear();
459 _ttl_count.resize(anim_count);
460
461 // Reset sprite primitive data in order to prepare for next pass.
462 _sprite_primitive.clear();
463 _sprites.clear();
464 _vdata.clear();
465 _sprite_writer.clear();
466
467 GeomNode *render_node = get_render_node();
468 render_node->remove_all_geoms();
469
470 // For each animation...
471 for (i = 0; i < anim_count; ++i) {
472 anim = _anims[i];
473 _anim_size[i] = anim->get_num_frames();
474
475 _sprite_primitive.push_back(pvector<PT(Geom)>());
476 _sprites.push_back(pvector<PT(GeomPoints)>());
477 _vdata.push_back(pvector<PT(GeomVertexData)>());
478 _sprite_writer.push_back(pvector<SpriteWriter>());
479
480 // For each frame of the animation...
481 for (j = 0; j < _anim_size[i]; ++j) {
482 _ttl_count[i] = (int *)PANDA_MALLOC_ARRAY(_anim_size[i] * sizeof(int));
483 _vdata[i].push_back(new GeomVertexData("sprite_particles", format, Geom::UH_stream));
484 PT(Geom) geom = new Geom(_vdata[i][j]);
485 _sprite_primitive[i].push_back((Geom*)geom);
486 _sprites[i].push_back(new GeomPoints(Geom::UH_stream));
487 geom->add_primitive(_sprites[i][j]);
488
489 // This will be overwritten in render(), but we had to have some initial
490 // value
491 _sprite_writer[i].push_back(SpriteWriter());
492
493 state = state->add_attrib(RenderModeAttrib::make(RenderModeAttrib::M_unchanged, _base_y_scale * _height, true));
494 if (anim->get_frame(j) != nullptr) {
495 state = state->add_attrib(TextureAttrib::make(anim->get_frame(j)));
496 state = state->add_attrib(TexGenAttrib::make(TextureStage::get_default(), TexGenAttrib::M_point_sprite));
497
498 // Build a transform to convert the texture coordinates to the ll, ur
499 // space.
500 LPoint2 ul(anim->get_ll(j)[0], anim->get_ur(j)[1]);
501 LPoint2 lr(anim->get_ur(j)[0], anim->get_ll(j)[1]);
502 LVector2 sc = lr - ul;
503
504 CPT(TransformState) ts = TransformState::make_pos_rotate_scale2d(ul, 0.0f, sc);
505 state = state->add_attrib(TexMatrixAttrib::make(TextureStage::get_default(), ts));
506 }
507
508 render_node->add_geom(_sprite_primitive[i][j], state);
509 }
510 }
511
512 nassertv(render_node->check_valid());
513}

534void SpriteParticleRenderer::
535render(pvector< PT(PhysicsObject) >& po_vector, int ttl_particles) {
536 PStatTimer t1(_render_collector);
537 // There is no texture data available, exit.
538 if (_anims.empty()) {
539 return;
540 }
541
542 BaseParticle *cur_particle;
543 int remaining_particles = ttl_particles;
544 int i,j; // loop counters
545 int anim_count = _anims.size(); // number of animations
546 int frame; // frame index, used in indicating which frame to use when not animated
547 // First, since this is the only time we have access to the actual
548 // particles, do some delayed initialization.
549 if (_animate_frames || anim_count) {
550 if (!_birth_list.empty()) {
551 for (vector_int::iterator vIter = _birth_list.begin(); vIter != _birth_list.end(); ++vIter) {
552 cur_particle = (BaseParticle*)po_vector[*vIter].p();
553 i = int(NORMALIZED_RAND()*anim_count);
554
555 // If there are multiple animations to choose from, choose one at
556 // random for this new particle
557 cur_particle->set_index(i < anim_count?i:i-1);
558
559 // This is an experimental age offset so that the animations don't
560 // appear synchronized. If we are using animations, try to vary the
561 // frame flipping a bit for particles in the same litter. A similar
562 // effect might be a achieved by using a small lifespan spread value
563 // on the factory.
564
565 // Perhaps we should look into other methods. The age offset doesn't
566 // seem to be cutting it.
567 if (_animate_frames) {
568 cur_particle->set_age(cur_particle->get_age()+i/10.0*cur_particle->get_lifespan());
569 }
570 }
571 }
572 }
573 _birth_list.clear();
574
575 // Create vertex writers for each of the possible geoms. Could possibly be
576 // changed to only create writers for geoms that would be used according to
577 // the animation configuration.
578 for (i = 0; i < anim_count; ++i) {
579 for (j = 0; j < _anim_size[i]; ++j) {
580 // Set the particle per frame counts to 0.
581 memset(_ttl_count[i], 0, _anim_size[i]*sizeof(int));
582 _sprite_writer[i][j].vertex = GeomVertexWriter(_vdata[i][j], InternalName::get_vertex());
583 _sprite_writer[i][j].color = GeomVertexWriter(_vdata[i][j], InternalName::get_color());
584 _sprite_writer[i][j].rotate = GeomVertexWriter(_vdata[i][j], InternalName::get_rotate());
585 _sprite_writer[i][j].size = GeomVertexWriter(_vdata[i][j], InternalName::get_size());
586 _sprite_writer[i][j].aspect_ratio = GeomVertexWriter(_vdata[i][j], InternalName::get_aspect_ratio());
587 }
588 }
589
590 // init the aabb
591 _aabb_min.set(99999.0f, 99999.0f, 99999.0f);
592 _aabb_max.set(-99999.0f, -99999.0f, -99999.0f);
593
594 // run through every filled slot
595 for (i = 0; i < (int)po_vector.size(); i++) {
596 cur_particle = (BaseParticle *) po_vector[i].p();
597
598 if (!cur_particle->get_alive()) {
599 continue;
600 }
601
602 LPoint3 position = cur_particle->get_position();
603
604 // x aabb adjust
605 if (position[0] > _aabb_max[0])
606 _aabb_max[0] = position[0];
607 else if (position[0] < _aabb_min[0])
608 _aabb_min[0] = position[0];
609
610 // y aabb adjust
611 if (position[1] > _aabb_max[1])
612 _aabb_max[1] = position[1];
613 else if (position[1] < _aabb_min[1])
614 _aabb_min[1] = position[1];
615
616 // z aabb adjust
617 if (position[2] > _aabb_max[2])
618 _aabb_max[2] = position[2];
619 else if (position[2] < _aabb_min[2])
620 _aabb_min[2] = position[2];
621
622
623 PN_stdfloat t = cur_particle->get_parameterized_age();
624 int anim_index = cur_particle->get_index();
625
626 // If an animation has been removed, we need to reassign those particles
627 // assigned to the removed animation.
628 if(_animation_removed && (anim_index >= anim_count)) {
629 anim_index = int(NORMALIZED_RAND()*anim_count);
630 anim_index = anim_index<anim_count?anim_index:anim_index-1;
631 cur_particle->set_index(anim_index);
632 }
633
634 // Find the frame
635 if (_animate_frames) {
636 if (_animate_frames_rate == 0.0f) {
637 frame = (int)(t*_anim_size[anim_index]);
638 } else {
639 frame = (int)fmod(cur_particle->get_age()*_animate_frames_rate+1,_anim_size[anim_index]);
640 }
641 } else {
642 frame = _animate_frames_index;
643 }
644
645 // Quick check make sure our math above didn't result in an invalid frame.
646 frame = (frame < _anim_size[anim_index]) ? frame : (_anim_size[anim_index]-1);
647 ++_ttl_count[anim_index][frame];
648
649 // Calculate the color This is where we'll want to give the renderer the
650 // new color
651 LColor c = _color_interpolation_manager->generateColor(t);
652
653 int alphamode=get_alpha_mode();
654 if (alphamode != PR_ALPHA_NONE) {
655 if (alphamode == PR_ALPHA_OUT)
656 c[3] *= (1.0f - t) * get_user_alpha();
657 else if (alphamode == PR_ALPHA_IN)
658 c[3] *= t * get_user_alpha();
659 else if (alphamode == PR_ALPHA_IN_OUT) {
660 c[3] *= 2.0f * min(t, 1.0f - t) * get_user_alpha();
661 }
662 else {
663 assert(alphamode == PR_ALPHA_USER);
664 c[3] *= get_user_alpha();
665 }
666 }
667
668 // Send the data on its way...
669 _sprite_writer[anim_index][frame].vertex.add_data3(position);
670 _sprite_writer[anim_index][frame].color.add_data4(c);
671
672 PN_stdfloat current_x_scale = _initial_x_scale;
673 PN_stdfloat current_y_scale = _initial_y_scale;
674
675 if (_animate_x_ratio || _animate_y_ratio) {
676 if (_blend_method == PP_BLEND_CUBIC) {
677 t = CUBIC_T(t);
678 }
679
680 if (_animate_x_ratio) {
681 current_x_scale = (_initial_x_scale +
682 (t * (_final_x_scale - _initial_x_scale)));
683 }
684 if (_animate_y_ratio) {
685 current_y_scale = (_initial_y_scale +
686 (t * (_final_y_scale - _initial_y_scale)));
687 }
688 }
689
690 if (_sprite_writer[anim_index][frame].size.has_column()) {
691 _sprite_writer[anim_index][frame].size.add_data1f(current_y_scale * _height);
692 }
693 if (_sprite_writer[anim_index][frame].aspect_ratio.has_column()) {
694 _sprite_writer[anim_index][frame].aspect_ratio.add_data1f(_aspect_ratio * current_x_scale / current_y_scale);
695 }
696 if (_animate_theta) {
697 _sprite_writer[anim_index][frame].rotate.add_data1f(cur_particle->get_theta());
698 } else if (_sprite_writer[anim_index][frame].rotate.has_column()) {
699 _sprite_writer[anim_index][frame].rotate.add_data1f(_theta);
700 }
701
702 // maybe jump out early?
703 remaining_particles--;
704 if (remaining_particles == 0) {
705 break;
706 }
707 }
708 int n = 0;
709 GeomNode *render_node = get_render_node();
710
711 for (i = 0; i < anim_count; ++i) {
712 for (j = 0; j < _anim_size[i]; ++j) {
713 _sprites[i][j]->clear_vertices();
714 _sprite_writer[i][j].clear();
715

717 // and the Geom to the GeomNode, in case it got flattened away.
718 _sprite_primitive[i][j]->set_primitive(0, _sprites[i][j]);
719 _sprite_primitive[i][j]->set_vertex_data(_vdata[i][j]);
720

723 }
724 }
725

728 for (j = 0; j < _anim_size[i]; ++j) {
729 _sprites[i][j]->add_next_vertices(_ttl_count[i][j]);
730 }
731 }
732 } else {
733 for (i = 0; i < anim_count; ++i) {
734 _sprites[i][_animate_frames_index]->add_next_vertices(_ttl_count[i][_animate_frames_index]);
735 }
736 }
737
738 // done filling geompoint node, now do the bb stuff
739 LPoint3 aabb_center = _aabb_min + ((_aabb_max - _aabb_min) * 0.5f);
740 PN_stdfloat radius = (aabb_center - _aabb_min).length();
741
742 for (i = 0; i < anim_count; ++i) {
743 for (j = 0; j < _anim_size[i]; ++j) {
744 nassertv(_sprite_primitive[i][j]->check_valid());
745 BoundingSphere sphere(aabb_center, radius);
746 _sprite_primitive[i][j]->set_bounds(&sphere);
747 }
748 }
749
750 get_render_node()->mark_internal_bounds_stale();
751 nassertv(render_node->check_valid());
752 _animation_removed = false;
753}
