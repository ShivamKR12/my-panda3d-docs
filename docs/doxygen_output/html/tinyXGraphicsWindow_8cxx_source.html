139void TinyXGraphicsWindow::
140end_flip() {
141 if (_xwindow == (X11_Window)nullptr || !_flip_ready) {
142 GraphicsWindow::end_flip();
143 return;
144 }
145
146 if (_reduced_frame_buffer != nullptr) {
147 // Zoom the reduced buffer onto the full buffer.
148 ZB_zoomFrameBuffer(_full_frame_buffer, 0, 0,
149 _full_frame_buffer->xsize, _full_frame_buffer->ysize,
150 _reduced_frame_buffer, 0, 0,
151 _reduced_frame_buffer->xsize, _reduced_frame_buffer->ysize);
152 }
153
154 // We can't just point the XPutImage directly at our own framebuffer data,
155 // even if the bytes_per_pixel matches, because some X displays will respect
156 // the alpha channel and make the window transparent there. We don't want
157 // transparent windows where the alpha data happens to less than 1.0.
158 ZB_copyFrameBufferNoAlpha(_full_frame_buffer, _ximage->data, _pitch);
159
160 XPutImage(_display, _xwindow, _gc, _ximage, 0, 0, 0, 0,
161 _full_frame_buffer->xsize, _full_frame_buffer->ysize);
162 XFlush(_display);
163 GraphicsWindow::end_flip();
164}

223bool TinyXGraphicsWindow::
224open_window() {
225 TinyXGraphicsPipe *tinyx_pipe;
226 DCAST_INTO_R(tinyx_pipe, _pipe, false);
227
228 // GSG CreationInitialization
229 TinyGraphicsStateGuardian *tinygsg;
230 if (_gsg == 0) {
231 // There is no old gsg. Create a new one.
232 tinygsg = new TinyGraphicsStateGuardian(_engine, _pipe, nullptr);
233 _gsg = tinygsg;
234 } else {
235 DCAST_INTO_R(tinygsg, _gsg, false);
236 }
237
238 XVisualInfo vinfo_template;
239 vinfo_template.screen = _screen;
240 vinfo_template.depth = 32;
241 vinfo_template.c_class = TrueColor;
242
243 // Try to get each of these properties in turn.
244 int try_masks[] = {
245 VisualScreenMask | VisualDepthMask | VisualClassMask,
246 VisualScreenMask | VisualClassMask,
247 VisualScreenMask | VisualDepthMask,
248 VisualScreenMask,
249 0,
250 };
251
252 int i = 0;
253 int num_vinfos = 0;
254 XVisualInfo *vinfo_array;
255 while (try_masks[i] != 0 && num_vinfos == 0) {
256 vinfo_array =
257 XGetVisualInfo(_display, try_masks[i], &vinfo_template, &num_vinfos);
258 ++i;
259 }
260
261 if (num_vinfos == 0) {
262 // No suitable X visual.
263 tinydisplay_cat.error()
264 << "No suitable X Visual available; cannot open window.\n";
265 return false;
266 }
267 _visual_info = &vinfo_array[0];
268
269 _visual = _visual_info->visual;
270 _depth = _visual_info->depth;
271 _bytes_per_pixel = _depth / 8;
272 if (_bytes_per_pixel == 3) {
273 // Seems to be a special case.
274 _bytes_per_pixel = 4;
275 }
276 tinydisplay_cat.info()
277 << "Got X Visual with depth " << _depth << " (bpp " << _bytes_per_pixel << ") and class ";
278 switch (_visual_info->c_class) {
279 case TrueColor:
280 tinydisplay_cat.info(false) << "TrueColor\n";
281 break;
282
283 case DirectColor:
284 tinydisplay_cat.info(false) << "DirectColor\n";
285 break;
286
287 case StaticColor:
288 tinydisplay_cat.info(false) << "StaticColor\n";
289 break;
290
291 case StaticGray:
292 tinydisplay_cat.info(false) << "StaticGray\n";
293 break;
294
295 case GrayScale:
296 tinydisplay_cat.info(false) << "GrayScale\n";
297 break;
298
299 case PseudoColor:
300 tinydisplay_cat.info(false) << "PseudoColor\n";
301 break;
302 }
303
304 setup_colormap(_visual_info);
305
306 if (!x11GraphicsWindow::open_window()) {
307 return false;
308 }
309
310 _gc = XCreateGC(_display, _xwindow, 0, nullptr);
311
312 create_full_frame_buffer();
313 if (_full_frame_buffer == nullptr) {
314 tinydisplay_cat.error()
315 << "Could not create frame buffer.\n";
316 return false;
317 }
318 create_reduced_frame_buffer();
319 create_ximage();
320 nassertr(_ximage != nullptr, false);
321
322 tinygsg->_current_frame_buffer = _full_frame_buffer;
323
324 tinygsg->reset_if_new();
325 if (!tinygsg->is_valid()) {
326 close_window();
327 return false;
328 }
329
330 XMapWindow(_display, _xwindow);
331
332 if (_properties.get_raw_mice()) {
333 open_raw_mice();
334 } else {
335 if (tinydisplay_cat.is_debug()) {
336 tinydisplay_cat.debug()
337 << "Raw mice not requested.\n";
338 }
339 }
340
341 // Create a WindowHandle for ourselves
342 _window_handle = NativeWindowHandle::make_x11(_xwindow);
343
344 // And tell our parent window that we're now its child.
345 if (_parent_window_handle != nullptr) {
346 _parent_window_handle->attach_child(_window_handle);
347 }
348
349 return true;
350}

364void TinyXGraphicsWindow::
365create_full_frame_buffer() {
366 if (_full_frame_buffer != nullptr) {
367 ZB_close(_full_frame_buffer);
368 _full_frame_buffer = nullptr;
369 }
370
371 int mode;
372 switch (_bytes_per_pixel) {
373 case 1:
374 tinydisplay_cat.error()
375 << "Palette images are currently not supported.\n";
376 return;
377
378 case 2:
379 mode = ZB_MODE_5R6G5B;
380 break;
381 case 4:
382 mode = ZB_MODE_RGBA;
383 break;
384
385 default:
386 return;
387 }
388
389 _full_frame_buffer = ZB_open(_properties.get_x_size(), _properties.get_y_size(), mode, 0, 0, 0, 0);
390 _pitch = (_full_frame_buffer->xsize * _bytes_per_pixel + 3) & ~3;
