125void OpenALAudioSound::
126play() {
127 ReMutexHolder holder(OpenALAudioManager::_lock);
128
129 if (!is_valid()) return;
130
131 PN_stdfloat px, py, pz, vx, vy, vz;
132
133 if (!_active) {
134 _paused = true;
135 return;
136 }
137
138 stop();
139
140 if (!require_sound_data()) {
141 cleanup();
142 return;
143 }
144
145 _manager->starting_sound(this);
146 if (!is_playing()) {
147 return;
148 }
149
150 _manager->make_current();
151
152 alGetError(); // clear errors
153
154 // nonpositional sources are made relative to the listener so they don't
155 // move
156 alSourcei(_source,AL_SOURCE_RELATIVE,_positional?AL_FALSE:AL_TRUE);
157 al_audio_errcheck("alSourcei(_source,AL_SOURCE_RELATIVE)");
158
159 // set source properties that we have stored
160 set_volume(_volume);
161 // set_balance(_balance);
162
163 set_3d_min_distance(_min_dist);
164 set_3d_max_distance(_max_dist);
165 set_3d_drop_off_factor(_drop_off_factor);
166 get_3d_attributes(&px, &py, &pz, &vx, &vy, &vz);
167 set_3d_attributes(px, py, pz, vx, vy, vz);
168 set_3d_direction(get_3d_direction());
169
170 _playing_loops = _loop_count;
171 if (_playing_loops == 0) {
172 _playing_loops = 1000000000;
173 }
174 _loops_completed = 0;
175
176 double play_rate = _play_rate * _manager->get_play_rate();
177 audio_debug("playing. Rate=" << play_rate);
178 alSourcef(_source, AL_PITCH, play_rate);
179 _playing_rate = play_rate;
180
181 if (_sd->_sample) {
182 push_fresh_buffers();
183 // The macOS implementation of alSourcePlay resets the offset, so we call
184 // it before and avoid calling restart_stalled_audio() afterwards (#1607)
185#ifdef HAVE_OPENAL_FRAMEWORK
186 ALenum status;
187 alGetSourcei(_source, AL_SOURCE_STATE, &status);
188 if (status != AL_PLAYING) {
189 alSourcePlay(_source);
190 }
191#endif
192 alSourcef(_source, AL_SEC_OFFSET, _start_time);
193 _stream_queued[0]._time_offset = _start_time;
194#ifndef HAVE_OPENAL_FRAMEWORK
195 restart_stalled_audio();
196#endif
197 } else {
198 audio_debug("Play: stream tell = " << _sd->_stream->tell() << " seeking " << _start_time);
199 if (_sd->_stream->tell() != _start_time) {
200 _sd->_stream->seek(_start_time);
201 }
202 push_fresh_buffers();
203 restart_stalled_audio();
204 }
205 double rtc = TrueClock::get_global_ptr()->get_short_time();
206 set_calibrated_clock(rtc, _start_time, 1.0);
207 _current_time = _start_time;
208 _start_time = 0.0;
209}

214void OpenALAudioSound::
215stop() {
216 ReMutexHolder holder(OpenALAudioManager::_lock);
217
218 if (!is_valid()) return;
219
220 if (is_playing()) {
221 _manager->make_current();
222
223 nassertv(has_sound_data());
224
225 alGetError(); // clear errors
226 alSourceStop(_source);
227 al_audio_errcheck("stopping a source");
228 alSourcei(_source, AL_BUFFER, 0);
229 al_audio_errcheck("clear source buffers");
230 for (int i=0; i<((int)(_stream_queued.size())); i++) {
231 ALuint buffer = _stream_queued[i]._buffer;
232 if (buffer != _sd->_sample) {
233 _manager->delete_buffer(buffer);
234 }
235 }
236 _stream_queued.resize(0);
237 }
238
239 _paused = false;
240
241 _manager->stopping_sound(this);
242 release_sound_data(false);
243}

360void OpenALAudioSound::
361queue_buffer(ALuint buffer, int samples, int loop_index, double time_offset) {
362 ReMutexHolder holder(OpenALAudioManager::_lock);
363
364 nassertv(is_playing());
365
366 // Now push the buffer into the stream queue.
367 alGetError();
368 alSourceQueueBuffers(_source,1,&buffer);
369 ALenum err = alGetError();
370 if (err != AL_NO_ERROR) {
371 audio_error("could not load sample buffer into the queue");
372 cleanup();
373 return;
374 }
375 QueuedBuffer buf;
376 buf._buffer = buffer;
377 buf._samples = samples;
378 buf._loop_index = loop_index;
379 buf._time_offset = time_offset;
380 _stream_queued.push_back(buf);
381}

386ALuint OpenALAudioSound::
387make_buffer(int samples, int channels, int rate, unsigned char *data) {
388 ReMutexHolder holder(OpenALAudioManager::_lock);
389
390 nassertr(is_playing(), 0);
391
392 // Allocate a buffer to hold the data.
393 alGetError();
394 ALuint buffer;
395 alGenBuffers(1, &buffer);
396 if (alGetError() != AL_NO_ERROR) {
397 audio_error("could not allocate an OpenAL buffer object");
398 cleanup();
399 return 0;
400 }
401
402 // Now fill the buffer with the data provided.
403 alBufferData(buffer,
404 (channels>1) ? AL_FORMAT_STEREO16 : AL_FORMAT_MONO16,
405 data, samples * channels * 2, rate);
406 int err = alGetError();
407 if (err != AL_NO_ERROR) {
408 audio_error("could not fill OpenAL buffer object with data");
409 cleanup();
410 return 0;
411 }
412
413 return buffer;
414}

420int OpenALAudioSound::
421read_stream_data(int bytelen, unsigned char *buffer) {
422 ReMutexHolder holder(OpenALAudioManager::_lock);
423
424 nassertr(has_sound_data(), 0);
425
426 MovieAudioCursor *cursor = _sd->_stream;
427 int channels = cursor->audio_channels();
428 int rate = cursor->audio_rate();
429 int space = bytelen / (channels * 2);
430 int fill = 0;
431
432 while (space && (_loops_completed < _playing_loops)) {
433 double t = cursor->tell();
434 double remain = cursor->length() - t;
435 if (remain > 60.0) {
436 remain = 60.0;
437 }
438 int samples = (int)(remain * rate);
439 if (samples <= 0) {
440 _loops_completed += 1;
441 cursor->seek(_loop_start);
442 continue;
443 }
444 if (_sd->_stream->ready() == 0) {
445 if (_sd->_stream->aborted()) {
446 _loops_completed = _playing_loops;
447 }
448 return fill;
449 }
450 if (samples > space) {
451 samples = space;
452 }
453 if (samples > _sd->_stream->ready()) {
454 samples = _sd->_stream->ready();
455 }
456 samples = cursor->read_samples(samples, (int16_t *)buffer);
457 if (audio_cat.is_debug()) {
458 size_t hval = AddHash::add_hash(0, (uint8_t*)buffer, samples*channels*2);
459 audio_debug("Streaming " << cursor->get_source()->get_name() << " at " << t << " hash " << hval);
460 }
461 if (samples == 0) {
462 _loops_completed += 1;
463 cursor->seek(_loop_start);
464 if (_playing_loops >= 1000000000) {
465 // Prevent infinite loop if endlessly looping empty sound
466 return fill;
467 }
468 continue;
469 }
470 fill += samples;
471 space -= samples;
472 buffer += (samples * channels * 2);
473 }
474 return fill;
475}

483void OpenALAudioSound::
484correct_calibrated_clock(double rtc, double t) {
485 ReMutexHolder holder(OpenALAudioManager::_lock);
486
487 nassertv(is_playing());
488
489 double cc = (rtc - _calibrated_clock_base) * _calibrated_clock_scale;
490 double diff = cc-t;
491 _calibrated_clock_decavg = (_calibrated_clock_decavg * 0.95) + (diff * 0.05);
492 if (diff > 0.5) {
493 set_calibrated_clock(rtc, t, 1.0);
494 _calibrated_clock_decavg = 0.0;
495 } else {
496 double scale = 1.0;
497 if ((_calibrated_clock_decavg > 0.01) && (diff > 0.01)) {
498 scale = 0.98;
499 }
500 if ((_calibrated_clock_decavg < -0.01) && (diff < -0.01)) {
501 scale = 1.03;
502 }
503 if ((_calibrated_clock_decavg < -0.05) && (diff < -0.05)) {
504 scale = 1.2;
505 }
506 if ((_calibrated_clock_decavg < -0.15) && (diff < -0.15)) {
507 scale = 1.5;
508 }
509 set_calibrated_clock(rtc, cc, scale);
510 }
511 //cc = (rtc - _calibrated_clock_base) * _calibrated_clock_scale;
512}

517void OpenALAudioSound::
518pull_used_buffers() {
519 ReMutexHolder holder(OpenALAudioManager::_lock);
520
521 if (!is_valid()) return;
522 nassertv(is_playing());
523 nassertv(has_sound_data());
524
525 while (_stream_queued.size()) {
526 ALuint buffer = 0;
527 ALint num_buffers = 0;
528 alGetSourcei(_source, AL_BUFFERS_PROCESSED, &num_buffers);
529 if (num_buffers <= 0) {
530 break;
531 }
532 alSourceUnqueueBuffers(_source, 1, &buffer);
533 int err = alGetError();
534 if (err == AL_NO_ERROR) {
535 if (_stream_queued[0]._buffer != buffer) {
536 // This is certainly atypical: most implementations of OpenAL unqueue
537 // buffers in FIFO order. However, some (e.g. Apple's) can unqueue
538 // buffers out-of-order if playback is interrupted. So, we don't freak
539 // out unless `buffer` isn't in _stream_queued at all.
540 bool found_culprit = false;
541 for (auto it = _stream_queued.begin(); it != _stream_queued.end(); ++it) {
542 if (it->_buffer == buffer) {
543 // Phew. Found it. Just remove that.
544 _stream_queued.erase(it);
545 found_culprit = true;
546 break;
547 }
548 }
549 if (!found_culprit) {
550 audio_error(get_name() << ": corruption in stream queue");
551 cleanup();
552 return;
553 }
554 } else {
555 _stream_queued.pop_front();
556 if (_stream_queued.size()) {
557 double al = _stream_queued[0]._time_offset + _stream_queued[0]._loop_index * _length;
558 double rtc = TrueClock::get_global_ptr()->get_short_time();
559 correct_calibrated_clock(rtc, al);
560 }
561 if (buffer != _sd->_sample) {
562 _manager->delete_buffer(buffer);
563 }
564 }
565 } else {
566 break;
567 }
568 }
569}

575void OpenALAudioSound::
576push_fresh_buffers() {
577 ReMutexHolder holder(OpenALAudioManager::_lock);
578 static unsigned char data[65536];
579
580 if (!is_valid()) return;
581 nassertv(is_playing());
582 nassertv(has_sound_data());
583
584 if (_sd->_sample) {
585 while ((_loops_completed < _playing_loops) &&
586 (_stream_queued.size() < 100)) {
587 queue_buffer(_sd->_sample, 0,_loops_completed, _loop_start);
588 _loops_completed += 1;
589 }
590 } else {
591 MovieAudioCursor *cursor = _sd->_stream;
592 int channels = cursor->audio_channels();
593 int rate = cursor->audio_rate();
594
595 int fill = 0;
596 for (size_t i = 0; i < _stream_queued.size(); i++) {
597 fill += _stream_queued[i]._samples;
598 }
599
600 while ((_loops_completed < _playing_loops) &&
601 (fill < (int)(audio_buffering_seconds * rate * channels))) {
602 int loop_index = _loops_completed;
603 double time_offset = cursor->tell();
604 int samples = read_stream_data(65536, data);
605 if (samples == 0) {
606 break;
607 }
608 ALuint buffer = make_buffer(samples, channels, rate, data);
609 if (!is_valid() || !buffer) return;
610 queue_buffer(buffer, samples, loop_index, time_offset);
611 if (!is_valid()) return;
612 fill += samples;
613 }
614 }
