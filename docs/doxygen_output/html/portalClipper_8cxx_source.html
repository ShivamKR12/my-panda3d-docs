144void PortalClipper::
145draw_lines() {
146 if (!_list.empty()) {
147 _created_data = nullptr;
148
149 PT(GeomVertexData) vdata = new GeomVertexData
150 ("portal", GeomVertexFormat::get_v3c(), Geom::UH_static);
151 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
152 GeomVertexWriter color(vdata, InternalName::get_color());
153
154 PT(GeomLinestrips) lines = new GeomLinestrips(Geom::UH_static);
155 PT(GeomPoints) points = new GeomPoints(Geom::UH_static);
156
157 int v = 0;
158 LineList::const_iterator ll;
159 SegmentList::const_iterator sl;
160
161 for (ll = _list.begin(); ll != _list.end(); ll++) {
162 const SegmentList &segs = (*ll);
163
164 if (segs.size() < 2) {
165 // A segment of length 1 is just a point.
166 for (sl = segs.begin(); sl != segs.end(); sl++) {
167 points->add_vertex(v);
168 vertex.add_data3((*sl)._point);
169 color.add_data4((*sl)._color);
170 v++;
171 }
172 points->close_primitive();
173
174 } else {
175 // A segment of length 2 or more is a line segment or segments.
176 for (sl = segs.begin(); sl != segs.end(); sl++) {
177 lines->add_vertex(v);
178 vertex.add_data3((*sl)._point);
179 color.add_data4((*sl)._color);
180 v++;
181 }
182 lines->close_primitive();
183 }
184 }
185
186 if (lines->get_num_vertices() != 0) {
187 PT(Geom) geom = new Geom(vdata);
188 geom->add_primitive(lines);
189 _previous->add_geom(geom);
190 }
191 if (points->get_num_vertices() != 0) {
192 PT(Geom) geom = new Geom(vdata);
193 geom->add_primitive(points);
194 _previous->add_geom(geom);
195 }
196 }

202bool PortalClipper::
203prepare_portal(const NodePath &node_path)
204{
205 // Get the Portal Node from this node_path
206 PandaNode *node = node_path.node();
207 _portal_node = nullptr;
208 if (node->is_of_type(PortalNode::get_class_type())) {
209 _portal_node = DCAST(PortalNode, node);
210 }
211 nassertr(_portal_node != nullptr, false);
212
213 // Get the geometry from the portal
214 if (portal_cat.is_spam()) {
215 portal_cat.spam() << *_portal_node << endl;
216 }
217
218 // Get the camera transformation matrix
219 CPT(TransformState) ctransform = node_path.get_transform(_scene_setup->get_cull_center());
220 // CPT(TransformState) ctransform =
221 // node_path.get_transform(_scene_setup->get_camera_path());
222 LMatrix4 cmat = ctransform->get_mat();
223 if (portal_cat.is_spam()) {
224 portal_cat.spam() << cmat << endl;
225 }
226
227 LVertex temp[4];
228 temp[0] = _portal_node->get_vertex(0);
229 temp[1] = _portal_node->get_vertex(1);
230 temp[2] = _portal_node->get_vertex(2);
231 temp[3] = _portal_node->get_vertex(3);
232
233 if (portal_cat.is_spam()) {
234 portal_cat.spam() << "before transformation to camera space" << endl;
235 portal_cat.spam() << temp[0] << endl;
236 portal_cat.spam() << temp[1] << endl;
237 portal_cat.spam() << temp[2] << endl;
238 portal_cat.spam() << temp[3] << endl;
239 }
240
241 temp[0] = temp[0]*cmat;
242 temp[1] = temp[1]*cmat;
243 temp[2] = temp[2]*cmat;
244 temp[3] = temp[3]*cmat;
245
246 LPlane portal_plane(temp[0], temp[1], temp[2]);
247 if (!is_facing_view(portal_plane)) {
248 if (portal_cat.is_debug()) {
249 portal_cat.debug() << "portal failed 1st level test (isn't facing the camera)\n";
250 }
251 return false;
252 }
253
254 if (portal_cat.is_spam()) {
255 portal_cat.spam() << "after transformation to camera space" << endl;
256 portal_cat.spam() << temp[0] << endl;
257 portal_cat.spam() << temp[1] << endl;
258 portal_cat.spam() << temp[2] << endl;
259 portal_cat.spam() << temp[3] << endl;
260 }
261
262 // check if the portal intersects with the cameras 0 point (center of
263 // projection). In that case the portal will invert itself. portals
264 // intersecting the near plane or the 0 point are a weird case anyhow,
265 // therefore we don't reduce the frustum any further and just return true.
266 // In effect the portal doesn't reduce visibility but will draw everything
267 // in its out cell
268 const Lens *lens = _scene_setup->get_lens();
269 LVector3 forward = LVector3::forward(lens->get_coordinate_system());
270 int forward_axis;
271 if (forward[1]) {
272 forward_axis = 1;
273 }
274 else if (forward[2]) {
275 forward_axis = 2;
276 }
277 else {
278 forward_axis = 0;
279 }
280 if ((temp[0][forward_axis] * forward[forward_axis] <= 0) ||
281 (temp[1][forward_axis] * forward[forward_axis] <= 0) ||
282 (temp[2][forward_axis] * forward[forward_axis] <= 0) ||
283 (temp[3][forward_axis] * forward[forward_axis] <= 0)) {
284 if (portal_cat.is_debug()) {
285 portal_cat.debug() << "portal intersects with center of projection.." << endl;
286 }
287 return true;
288 }
289
290 // project portal points, so they are in the -1..1 range
291 LPoint3 projected_coords[4];
292 lens->project(temp[0], projected_coords[0]);
293 lens->project(temp[1], projected_coords[1]);
294 lens->project(temp[2], projected_coords[2]);
295 lens->project(temp[3], projected_coords[3]);
296
297 if (portal_cat.is_spam()) {
298 portal_cat.spam() << "after projection to 2d" << endl;
299 portal_cat.spam() << projected_coords[0] << endl;
300 portal_cat.spam() << projected_coords[1] << endl;
301 portal_cat.spam() << projected_coords[2] << endl;
302 portal_cat.spam() << projected_coords[3] << endl;
303 }
304
305 // calculate axis aligned bounding box of the portal
306 PN_stdfloat min_x, max_x, min_y, max_y;
307 min_x = min(min(min(projected_coords[0][0], projected_coords[1][0]), projected_coords[2][0]), projected_coords[3][0]);
308 max_x = max(max(max(projected_coords[0][0], projected_coords[1][0]), projected_coords[2][0]), projected_coords[3][0]);
309 min_y = min(min(min(projected_coords[0][1], projected_coords[1][1]), projected_coords[2][1]), projected_coords[3][1]);
310 max_y = max(max(max(projected_coords[0][1], projected_coords[1][1]), projected_coords[2][1]), projected_coords[3][1]);
311
312 if (portal_cat.is_spam()) {
313 portal_cat.spam() << "min_x " << min_x << ";max_x " << max_x << ";min_y " << min_y << ";max_y " << max_y << endl;
314 }
315
316 // clip the minima and maxima against the viewport
317 min_x = max(min_x, _reduced_viewport_min[0]);
318 min_y = max(min_y, _reduced_viewport_min[1]);
319 max_x = min(max_x, _reduced_viewport_max[0]);
320 max_y = min(max_y, _reduced_viewport_max[1]);
321
322 if (portal_cat.is_spam()) {
323 portal_cat.spam() << "after clipping: min_x " << min_x << ";max_x " << max_x << ";min_y " << min_y << ";max_y " << max_y << endl;
324 }
325
326 if ((min_x >= max_x) || (min_y >= max_y)) {
327 if (portal_cat.is_debug()) {
328 portal_cat.debug() << "portal got clipped away \n";
329 }
330 return false;
331 }
332
333 // here we know the portal is in view and we have its clipped extents
334 _reduced_viewport_min.set(min_x, min_y);
335 _reduced_viewport_max.set(max_x, max_y);
336
337 // calculate the near and far points so we can construct a frustum
338 LPoint3 near_point[4];
339 LPoint3 far_point[4];
340 lens->extrude(LPoint2(min_x, min_y), near_point[0], far_point[0]);
341 lens->extrude(LPoint2(max_x, min_y), near_point[1], far_point[1]);
342 lens->extrude(LPoint2(max_x, max_y), near_point[2], far_point[2]);
343 lens->extrude(LPoint2(min_x, max_y), near_point[3], far_point[3]);
344
345 // With these points, construct the new reduced frustum
346 _reduced_frustum = new BoundingHexahedron(far_point[0], far_point[1], far_point[2], far_point[3],
347 near_point[0], near_point[1], near_point[2], near_point[3]);
348
349 if (portal_cat.is_debug()) {
350 portal_cat.debug() << *_reduced_frustum << endl;
351 }
352
353 // do debug rendering, if requested
354 if (debug_portal_cull) {
355 // draw the reduced frustum
356 _color = LColor(0,0,1,1);
357 draw_hexahedron(DCAST(BoundingHexahedron, _reduced_frustum));
358
359 // lets first add the clipped portal (in yellow)
360 _color = LColor(1,1,0,1);
361 move_to((near_point[0]*0.99+far_point[0]*0.01)); // I choose a point in the middle between near and far.. could also be some other z value..
362 draw_to((near_point[1]*0.99+far_point[1]*0.01));
363 draw_to((near_point[2]*0.99+far_point[2]*0.01));
364 draw_to((near_point[3]*0.99+far_point[3]*0.01));
365 draw_to((near_point[0]*0.99+far_point[0]*0.01));
366
367 // ok, now lets add the original portal (in cyan)
368 _color = LColor(0,1,1,1);
369 move_to(temp[0]);
370 draw_to(temp[1]);
371 draw_to(temp[2]);
372 draw_to(temp[3]);
373 draw_to(temp[0]);
374 }
375
376 return true;
