115PT(BoundingVolume) CollisionCapsule::
116compute_internal_bounds() const {
117 PT(BoundingVolume) bound = CollisionSolid::compute_internal_bounds();
118
119 if (bound->is_of_type(GeometricBoundingVolume::get_class_type())) {
120 GeometricBoundingVolume *gbound;
121 DCAST_INTO_R(gbound, bound, bound);
122
123 LVector3 vec = (_b - _a);
124 if (vec.normalize()) {
125 // The bounding volume includes both endpoints, plus a little bit more
126 // to include the radius in both directions.
127 LPoint3 points[2];
128 points[0] = _a - vec * _radius;
129 points[1] = _b + vec * _radius;
130
131 gbound->around(points, points + 2);
132
133 } else {
134 // Both endpoints are coincident; therefore, the bounding volume is a
135 // sphere.
136 BoundingSphere sphere(_a, _radius);
137 gbound->extend_by(&sphere);
138 }
139 }
140

147PT(CollisionEntry) CollisionCapsule::
148test_intersection_from_box(const CollisionEntry &entry) const {
149 const CollisionBox *box;
150 DCAST_INTO_R(box, entry.get_from(), nullptr);
151
152 const LMatrix4 wrt_mat = entry.get_wrt_mat();
153 const LMatrix4 inv_wrt_mat = entry.get_inv_wrt_mat();
154
155 // We find the closest point on the box to the line
156 // segment defined by the endpoints of the capsule. To do
157 // this, convert the capsule into the box's coordinate space.
158 LPoint3 box_min = box->get_min();
159 LPoint3 box_max = box->get_max();
160 LPoint3 a = _a * inv_wrt_mat;
161 LPoint3 b = _b * inv_wrt_mat;
162 LVector3 delta = b - a;
163
164 double min_dist = DBL_MAX;
165 PN_stdfloat t;
166 LPoint3 closest_point;
167 // Iterate through the 6 planes of the box
168 for (int i = 0; i < 6; i++) {
169 if (!box->get_plane(i).intersects_line(t, a, delta)) {
170 continue;
171 }
172 if (t > 1.0) t = 1.0;
173 if (t < 0.0) t = 0.0;
174 if (t == min_dist) continue;
175 LPoint3 intersection_point = a + delta * t;
176 // Find the closest point on the box to the intersection point
177 LPoint3 p = intersection_point.fmax(box_min).fmin(box_max);
178 double dist = (p - intersection_point).length_squared();
179 if (dist < min_dist) {
180 min_dist = dist;
181 closest_point = p;
182 }
183 }
184 // Convert the closest point to the capsule's coordinate
185 // space where the capsule is aligned with the y axis.
186 closest_point = closest_point * wrt_mat * _inv_mat;
187 a = _a * _inv_mat;
188 b = _b * _inv_mat;
189
190 // Find the closest point on the line segment
191 LPoint3 point_on_segment;
192 point_on_segment[0] = a[0];
193 point_on_segment[2] = a[2];
194 if (closest_point[1] < std::min(a[1], b[1])) {
195 point_on_segment[1] = std::min(a[1], b[1]);
196 } else if (closest_point[1] > std::max(a[1], b[1])) {
197 point_on_segment[1] = std::max(a[1], b[1]);
198 } else {
199 point_on_segment[1] = closest_point[1];
200 }
201
202 if ((closest_point - point_on_segment).length_squared() > _radius * _radius) {
203 // No intersection.
204 return nullptr;
205 }
206
207 PT(CollisionEntry) new_entry = new CollisionEntry(entry);
208 LVector3 normal;
209 if (point_on_segment == closest_point) {
210 // If the box directly intersects the line segment, use
211 // the box's center to determine the surface normal.
212 normal = (box->get_center() * wrt_mat * _inv_mat) - point_on_segment;
213 if (closest_point != a && closest_point != b) normal[1] = 0;
214 } else {
215 normal = (closest_point - point_on_segment);
216 }
217 normal.normalize();
218 LPoint3 surface_point = point_on_segment + normal * _radius;
219 new_entry->set_interior_point(_mat.xform_point(closest_point));
220 new_entry->set_surface_point(_mat.xform_point(surface_point));
221 new_entry->set_surface_normal(_mat.xform_vec(normal));
222 return new_entry;
223}

228PT(CollisionEntry) CollisionCapsule::
229test_intersection_from_sphere(const CollisionEntry &entry) const {
230 const CollisionSphere *sphere;
231 DCAST_INTO_R(sphere, entry.get_from(), nullptr);
232
233 CPT(TransformState) wrt_space = entry.get_wrt_space();
234 CPT(TransformState) wrt_prev_space = entry.get_wrt_prev_space();
235
236 const LMatrix4 &wrt_mat = wrt_space->get_mat();
237
238 LPoint3 from_a = sphere->get_center() * wrt_mat;
239 LPoint3 from_b = from_a;
240
241 LPoint3 contact_point;
242 PN_stdfloat actual_t = 0.0f;
243
244 if (wrt_prev_space != wrt_space) {
245 // If the sphere is moving relative to the capsule, it becomes a capsule itself.
246 from_a = sphere->get_center() * wrt_prev_space->get_mat();
247 }
248
249 LVector3 from_direction = from_b - from_a;
250
251 LVector3 from_radius_v =
252 LVector3(sphere->get_radius(), 0.0f, 0.0f) * wrt_mat;
253 PN_stdfloat from_radius = length(from_radius_v);
254
255 double t1, t2;
256 if (!intersects_line(t1, t2, from_a, from_direction, from_radius)) {
257 // No intersection.
258 return nullptr;
259 }
260
261 if (t2 < 0.0 || t1 > 1.0) {
262 // Both intersection points are before the start of the segment or after
263 // the end of the segment.
264 return nullptr;
265 }
266
267 // doubles, not floats, to satisfy min and max templates.
268 actual_t = std::min(1.0, std::max(0.0, t1));
269 contact_point = from_a + actual_t * (from_b - from_a);
270
271 if (collide_cat.is_debug()) {
272 collide_cat.debug()
273 << "intersection detected from " << entry.get_from_node_path() << " into "
274 << entry.get_into_node_path() << "\n";
275 }
276 PT(CollisionEntry) new_entry = new CollisionEntry(entry);
277
278 LPoint3 into_intersection_point;
279 if (t2 > 1.0) {
280 // Point b is within the capsule. The first intersection point is point b
281 // itself.
282 into_intersection_point = from_b;
283 } else {
284 // Point b is outside the capsule, and point a is either inside the capsule or
285 // beyond it. The first intersection point is at t2.
286 into_intersection_point = from_a + t2 * from_direction;
287 }
288 set_intersection_point(new_entry, into_intersection_point, from_radius);
289
290 LPoint3 fake_contact_point;
291 LVector3 contact_normal;
292 calculate_surface_point_and_normal(contact_point,
293 from_radius,
294 fake_contact_point,
295 contact_normal);
296 new_entry->set_contact_pos(contact_point);
297 new_entry->set_contact_normal(contact_normal);
298 new_entry->set_t(actual_t);
299
300 return new_entry;
301}

479PT(CollisionEntry) CollisionCapsule::
480test_intersection_from_capsule(const CollisionEntry &entry) const {
481 const CollisionCapsule *capsule;
482 DCAST_INTO_R(capsule, entry.get_from(), nullptr);
483
484 LPoint3 into_a = _a;
485 LVector3 into_direction = _b - into_a;
486
487 const LMatrix4 &wrt_mat = entry.get_wrt_mat();
488
489 LPoint3 from_a = capsule->get_point_a() * wrt_mat;
490 LPoint3 from_b = capsule->get_point_b() * wrt_mat;
491 LVector3 from_direction = from_b - from_a;
492
493 LVector3 from_radius_v =
494 LVector3(capsule->get_radius(), 0.0f, 0.0f) * wrt_mat;
495 PN_stdfloat from_radius = length(from_radius_v);
496
497 // Determine the points on each segment with the smallest distance between.
498 double into_t, from_t;
499 calc_closest_segment_points(into_t, from_t,
500 into_a, into_direction,
501 from_a, from_direction);
502 LPoint3 into_closest = into_a + into_direction * into_t;
503 LPoint3 from_closest = from_a + from_direction * from_t;
504
505 // If the distance is greater than the sum of capsule radii, the test fails.
506 LVector3 closest_vec = from_closest - into_closest;
507 PN_stdfloat distance = closest_vec.length();
508 if (distance > _radius + from_radius) {
509 return nullptr;
510 }
511
512 if (collide_cat.is_debug()) {
513 collide_cat.debug()
514 << "intersection detected from " << entry.get_from_node_path()
515 << " into " << entry.get_into_node_path() << "\n";
516 }
517 PT(CollisionEntry) new_entry = new CollisionEntry(entry);
518
519 if (distance != 0) {
520 // This is the most common case, where the line segments don't touch
521 // exactly. We point the normal along the vector of the closest distance.
522 LVector3 surface_normal = closest_vec * (1.0 / distance);
523
524 new_entry->set_surface_point(into_closest + surface_normal * _radius);
525 new_entry->set_interior_point(from_closest - surface_normal * from_radius);
526
527 if (has_effective_normal() && capsule->get_respect_effective_normal()) {
528 new_entry->set_surface_normal(get_effective_normal());
529 } else if (distance != 0) {
530 new_entry->set_surface_normal(surface_normal);
531 }
532 } else {
533 // The rare case of the line segments touching exactly.
534 set_intersection_point(new_entry, into_closest, 0);
535 }
536
537 return new_entry;
538}

595void CollisionCapsule::
596fill_viz_geom() {
597 if (collide_cat.is_debug()) {
598 collide_cat.debug()
599 << "Recomputing viz for " << *this << "\n";
600 }
601
602 // Generate the vertices such that we draw a capsule with one endpoint at (0,
603 // 0, 0), and another at (0, length, 0). Then we'll rotate and translate it
604 // into place with the appropriate look_at matrix.
605 LVector3 direction = (_b - _a);
606 PN_stdfloat length = direction.length();
607
608 PT(GeomVertexData) vdata = new GeomVertexData
609 ("collision", GeomVertexFormat::get_v3(),
610 Geom::UH_static);
611 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
612
613 PT(GeomTristrips) strip = new GeomTristrips(Geom::UH_static);
614 // Generate the first endcap.
615 static const int num_slices = 8;
616 static const int num_rings = 4;
617 int ri, si;
618 for (ri = 0; ri < num_rings; ri++) {
619 for (si = 0; si <= num_slices; si++) {
620 vertex.add_data3(calc_sphere1_vertex(ri, si, num_rings, num_slices));
621 vertex.add_data3(calc_sphere1_vertex(ri + 1, si, num_rings, num_slices));
622 }
623 strip->add_next_vertices((num_slices + 1) * 2);
624 strip->close_primitive();
625 }
626
627 // Now the cylinder sides.
628 for (si = 0; si <= num_slices; si++) {
629 vertex.add_data3(calc_sphere1_vertex(num_rings, si, num_rings, num_slices));
630 vertex.add_data3(calc_sphere2_vertex(num_rings, si, num_rings, num_slices,
631 length));
632 }
633 strip->add_next_vertices((num_slices + 1) * 2);
634 strip->close_primitive();
635
636 // And the second endcap.
637 for (ri = num_rings - 1; ri >= 0; ri--) {
638 for (si = 0; si <= num_slices; si++) {
639 vertex.add_data3(calc_sphere2_vertex(ri + 1, si, num_rings, num_slices, length));
640 vertex.add_data3(calc_sphere2_vertex(ri, si, num_rings, num_slices, length));
641 }
642 strip->add_next_vertices((num_slices + 1) * 2);
643 strip->close_primitive();
644 }
645
646 PT(Geom) geom = new Geom(vdata);
647 geom->add_primitive(strip);
648
649 // Now transform the vertices to their actual location.
650 LMatrix4 mat;
651 look_at(mat, direction, LVector3(0.0f, 0.0f, 1.0f), CS_zup_right);
652 mat.set_row(3, _a);
653 geom->transform_vertices(mat);
654
655 _viz_geom->add_geom(geom, get_solid_viz_state());
656 _bounds_viz_geom->add_geom(geom, get_solid_bounds_viz_state());
657}

680LVertex CollisionCapsule::
681calc_sphere1_vertex(int ri, int si, int num_rings, int num_slices) {
682 PN_stdfloat r = (PN_stdfloat)ri / (PN_stdfloat)num_rings;
683 PN_stdfloat s = (PN_stdfloat)si / (PN_stdfloat)num_slices;
684
685 // Find the point on the rim, based on the slice.
686 PN_stdfloat theta = s * 2.0f * MathNumbers::pi;
687 PN_stdfloat x_rim = ccos(theta);
688 PN_stdfloat z_rim = csin(theta);
689
690 // Now pull that point in towards the pole, based on the ring.
691 PN_stdfloat phi = r * 0.5f * MathNumbers::pi;
692 PN_stdfloat to_pole = csin(phi);
693
694 PN_stdfloat x = _radius * x_rim * to_pole;
695 PN_stdfloat y = -_radius * ccos(phi);
696 PN_stdfloat z = _radius * z_rim * to_pole;
697
698 return LVertex(x, y, z);
699}

705LVertex CollisionCapsule::
706calc_sphere2_vertex(int ri, int si, int num_rings, int num_slices,
707 PN_stdfloat length) {
708 PN_stdfloat r = (PN_stdfloat)ri / (PN_stdfloat)num_rings;
709 PN_stdfloat s = (PN_stdfloat)si / (PN_stdfloat)num_slices;
710
711 // Find the point on the rim, based on the slice.
712 PN_stdfloat theta = s * 2.0f * MathNumbers::pi;
713 PN_stdfloat x_rim = ccos(theta);
714 PN_stdfloat z_rim = csin(theta);
715
716 // Now pull that point in towards the pole, based on the ring.
717 PN_stdfloat phi = r * 0.5f * MathNumbers::pi;
718 PN_stdfloat to_pole = csin(phi);
719
720 PN_stdfloat x = _radius * x_rim * to_pole;
721 PN_stdfloat y = length + _radius * ccos(phi);
722 PN_stdfloat z = _radius * z_rim * to_pole;
723
724 return LVertex(x, y, z);
725}

731void CollisionCapsule::
732calc_closest_segment_points(double &t1, double &t2,
733 const LPoint3 &from1, const LVector3 &delta1,
734 const LPoint3 &from2, const LVector3 &delta2) {
735 // Copyright 2001 softSurfer, 2012 Dan Sunday
736 // This code may be freely used, distributed and modified for any purpose
737 // providing that this copyright notice is included with it.
738 // SoftSurfer makes no warranty for this code, and cannot be held
739 // liable for any real or imagined damage resulting from its use.
740 // Users of this code must verify correctness for their application.
741 LVector3 w = from1 - from2;
742 PN_stdfloat a = delta1.dot(delta1); // always >= 0
743 PN_stdfloat b = delta1.dot(delta2);
744 PN_stdfloat c = delta2.dot(delta2); // always >= 0
745 PN_stdfloat d = delta1.dot(w);
746 PN_stdfloat e = delta2.dot(w);
747 PN_stdfloat D = a * c - b * b; // always >= 0
748 PN_stdfloat sN, sD = D;
749 PN_stdfloat tN, tD = D;
750
751 // compute the line parameters of the two closest points
752 if (IS_NEARLY_ZERO(D)) { // the lines are almost parallel
753 sN = 0.0; // force using point P0 on segment S1
754 sD = 1.0; // to prevent possible division by 0.0 later
755 tN = e;
756 tD = c;
757 } else {
758 // get the closest points on the infinite lines
759 sN = (b*e - c*d);
760 tN = (a*e - b*d);
761 if (sN < 0.0) { // sc < 0 => the s=0 edge is visible
762 sN = 0.0;
763 tN = e;
764 tD = c;
765 } else if (sN > sD) { // sc > 1 => the s=1 edge is visible
766 sN = sD;
767 tN = e + b;
768 tD = c;
769 }
770 }
771
772 if (tN < 0.0) { // tc < 0 => the t=0 edge is visible
773 tN = 0.0;
774 // recompute sc for this edge
775 if (-d < 0.0) {
776 sN = 0.0;
777 } else if (-d > a) {
778 sN = sD;
779 } else {
780 sN = -d;
781 sD = a;
782 }
783 } else if (tN > tD) { // tc > 1 => the t=1 edge is visible
784 tN = tD;
785 // recompute sc for this edge
786 if ((-d + b) < 0.0) {
787 sN = 0;
788 } else if ((-d + b) > a) {
789 sN = sD;
790 } else {
791 sN = (-d + b);
792 sD = a;
793 }
794 }
795
796 // finally do the division to get sc and tc
797 t1 = (IS_NEARLY_ZERO(sN) ? 0.0 : sN / sD);
798 t2 = (IS_NEARLY_ZERO(tN) ? 0.0 : tN / tD);
799}

809bool CollisionCapsule::
810intersects_line(double &t1, double &t2,
811 const LPoint3 &from0, const LVector3 &delta0,
812 PN_stdfloat inflate_radius) const {
813 // Convert the line into our canonical coordinate space: the capsule is aligned
814 // with the y axis.
815 LPoint3 from = from0 * _inv_mat;
816 LVector3 delta = delta0 * _inv_mat;
817
818 PN_stdfloat radius = _radius + inflate_radius;
819
820 // Now project the line into the X-Z plane to test for intersection with a
821 // 2-d circle around the origin. The equation for this is very similar to
822 // the formula for the intersection of a line with a sphere; see
823 // CollisionSphere::intersects_line() for the complete derivation. It's a
824 // little bit simpler because the circle is centered on the origin.
825 LVector2 from2(from[0], from[2]);
826 LVector2 delta2(delta[0], delta[2]);
827
828 double A = dot(delta2, delta2);
829
830 if (IS_NEARLY_ZERO(A)) {
831 // If the delta2 is 0, the line is perpendicular to the X-Z plane. The
832 // whole line intersects with the infinite cylinder if the point is within
833 // the circle.
834 if (from2.dot(from2) > radius * radius) {
835 // Nope, the 2-d point is outside the circle, so no intersection.
836 return false;
837 }
838
839 if (IS_NEARLY_ZERO(delta[1])) {
840 // Actually, the whole delta vector is 0, so the line is just a point.
841 // In this case, (since we have already shown the point is within the
842 // infinite cylinder), we intersect if and only if the three-dimensional
843 // point is between the endcaps.
844 if (from[1] < -radius || from[1] > _length + radius) {
845 // Way out.
846 return false;
847 }
848 if (from[1] < 0.0f) {
849 // Possibly within the first endcap.
850 if (from.dot(from) > radius * radius) {
851 return false;
852 }
853 } else if (from[1] > _length) {
854 // Possibly within the second endcap.
855 from[1] -= _length;
856 if (from.dot(from) > radius * radius) {
857 return false;
858 }
859 }
860
861 // The point is within the capsule!
862 t1 = t2 = 0.0;
863 return true;
864 }
865
866 // The 2-d point is within the circle, so compute our intersection points
867 // to include the entire vertical slice of the cylinder.
868 t1 = (-radius - from[1]) / delta[1];
869 t2 = (_length + radius - from[1]) / delta[1];
870
871 } else {
872 // The line is not perpendicular to the X-Z plane, so its projection into
873 // the plane is 2-d line. Test that 2-d line for intersection with the
874 // circular projection of the cylinder.
875
876 double B = 2.0f * dot(delta2, from2);
877 double fc_d2 = dot(from2, from2);
878 double C = fc_d2 - radius * radius;
879
880 double radical = B*B - 4.0*A*C;
881
882 if (IS_NEARLY_ZERO(radical)) {
883 // Tangent.
884 t1 = t2 = -B / (2.0*A);
885
886 } else if (radical < 0.0) {
887 // No real roots: no intersection with the line.
888 return false;
889
890 } else {
891 double reciprocal_2A = 1.0 / (2.0 * A);
892 double sqrt_radical = sqrtf(radical);
893 t1 = ( -B - sqrt_radical ) * reciprocal_2A;
894 t2 = ( -B + sqrt_radical ) * reciprocal_2A;
895 }
896 }
897
898 // Now we need to verify that the intersection points fall within the length
899 // of the cylinder.
900 PN_stdfloat t1_y = from[1] + t1 * delta[1];
901 PN_stdfloat t2_y = from[1] + t2 * delta[1];
902
903 if (t1_y < -radius && t2_y < -radius) {
904 // Both points are way off the bottom of the capsule; no intersection.
905 return false;
906 } else if (t1_y > _length + radius && t2_y > _length + radius) {
907 // Both points are way off the top of the capsule; no intersection.
908 return false;
909 }
910
911 if (t1_y < 0.0f) {
912 // The starting point is off the bottom of the capsule. Test the line
913 // against the first endcap.
914 double t1a, t2a;
915 if (!sphere_intersects_line(t1a, t2a, 0.0f, from, delta, radius)) {
916 // If there's no intersection with the endcap, there can't be an
917 // intersection with the cylinder.
918 return false;
919 }
920 t1 = t1a;
921
922 } else if (t1_y > _length) {
923 // The starting point is off the top of the capsule. Test the line against
924 // the second endcap.
925 double t1b, t2b;
926 if (!sphere_intersects_line(t1b, t2b, _length, from, delta, radius)) {
927 // If there's no intersection with the endcap, there can't be an
928 // intersection with the cylinder.
929 return false;
930 }
931 t1 = t1b;
932 }
933
934 if (t2_y < 0.0f) {
935 // The ending point is off the bottom of the capsule. Test the line against
936 // the first endcap.
937 double t1a, t2a;
938 if (!sphere_intersects_line(t1a, t2a, 0.0f, from, delta, radius)) {
939 // If there's no intersection with the endcap, there can't be an
940 // intersection with the cylinder.
941 return false;
942 }
943 t2 = t2a;
944
945 } else if (t2_y > _length) {
946 // The ending point is off the top of the capsule. Test the line against the
947 // second endcap.
948 double t1b, t2b;
949 if (!sphere_intersects_line(t1b, t2b, _length, from, delta, radius)) {
950 // If there's no intersection with the endcap, there can't be an
951 // intersection with the cylinder.
952 return false;
953 }
954 t2 = t2b;
955 }
956
957 return true;
958}

965bool CollisionCapsule::
966sphere_intersects_line(double &t1, double &t2, PN_stdfloat center_y,
967 const LPoint3 &from, const LVector3 &delta,
968 PN_stdfloat radius) {
969 // See CollisionSphere::intersects_line() for a derivation of the formula
970 // here.
971 double A = dot(delta, delta);
972
973 nassertr(A != 0.0, false);
974
975 LVector3 fc = from;
976 fc[1] -= center_y;
977 double B = 2.0f* dot(delta, fc);
978 double fc_d2 = dot(fc, fc);
979 double C = fc_d2 - radius * radius;
980
981 double radical = B*B - 4.0*A*C;
982
983 if (IS_NEARLY_ZERO(radical)) {
984 // Tangent.
985 t1 = t2 = -B / (2.0 * A);
986 return true;
987
988 } else if (radical < 0.0) {
989 // No real roots: no intersection with the line.
990 return false;
991 }
992
993 double reciprocal_2A = 1.0 / (2.0 * A);
994 double sqrt_radical = sqrtf(radical);
995 t1 = ( -B - sqrt_radical ) * reciprocal_2A;
996 t2 = ( -B + sqrt_radical ) * reciprocal_2A;
997
998 return true;
999}

1009bool CollisionCapsule::
1010intersects_parabola(double &t, const LParabola &parabola,
1011 double t1, double t2,
1012 const LPoint3 &p1, const LPoint3 &p2) const {
1013 // I don't even want to think about the math to do this calculation directly
1014 // --it's even worse than sphere-parabola. So I'll use the recursive
1015 // subdivision solution again, just like I did for sphere-parabola.
1016
1017 // First, see if the line segment (p1 - p2) comes sufficiently close to the
1018 // parabola. Do this by computing the parametric intervening point and
1019 // comparing its distance from the linear intervening point.
1020 double tmid = (t1 + t2) * 0.5;
1021
1022 if (tmid != t1 && tmid != t2) {
1023 LPoint3 pmid = parabola.calc_point(tmid);
1024 LPoint3 pmid2 = (p1 + p2) * 0.5f;
1025
1026 if ((pmid - pmid2).length_squared() > 0.001f) {
1027 // Subdivide.
1028 if (intersects_parabola(t, parabola, t1, tmid, p1, pmid)) {
1029 return true;
1030 }
1031 return intersects_parabola(t, parabola, tmid, t2, pmid, p2);
1032 }
1033 }
1034
1035 // The line segment is sufficiently close; compare the segment itself.
1036 double t1a, t2a;
1037 if (!intersects_line(t1a, t2a, p1, p2 - p1, 0.0f)) {
1038 return false;
1039 }
1040
1041 if (t2a < 0.0 || t1a > 1.0) {
1042 return false;
1043 }
1044
1045 t = std::max(t1a, 0.0);
1046 return true;
1047}

1054void CollisionCapsule::
1055calculate_surface_point_and_normal(const LPoint3 &surface_point,
1056 double extra_radius,
1057 LPoint3 &result_point,
1058 LVector3 &result_normal) const {
1059 // Convert the point into our canonical space for analysis.
1060 LPoint3 point = surface_point * _inv_mat;
1061 LVector3 normal;
1062
1063 if (point[1] <= 0.0) {
1064 // The point is on the first endcap.
1065 normal = point;
1066 if (!normal.normalize()) {
1067 normal.set(0.0, -1.0, 0.0);
1068 }
1069 point = normal * _radius;
1070
1071 } else if (point[1] >= _length) {
1072 // The point is on the second endcap.
1073 normal.set(point[0], point[1] - _length, point[2]);
1074 if (!normal.normalize()) {
1075 normal.set(0.0, 1.0, 0.0);
1076 }
1077 point = normal * _radius;
1078 point[1] += _length;
1079
1080 } else {
1081 // The point is within the cylinder part.
1082 LVector2d normal2d(point[0], point[2]);
1083 if (!normal2d.normalize()) {
1084 normal2d.set(0.0, 1.0);
1085 }
1086 normal.set(normal2d[0], 0.0, normal2d[1]);
1087 point.set(normal[0] * _radius, point[1], normal[2] * _radius);
1088 }
1089
1090 // Now convert the point and normal back into real space.
1091 result_point = point * _mat;
1092 result_normal = normal * _mat;
1093}
