78bool AndroidGraphicsWindow::
79begin_frame(FrameMode mode, Thread *current_thread) {
80 PStatTimer timer(_make_current_pcollector, current_thread);
81
82 begin_frame_spam(mode);
83 if (_gsg == nullptr) {
84 return false;
85 }
86
87 // XXX not open yet.
88 if (_egl_surface == EGL_NO_SURFACE) {
89 return false;
90 }
91
92 AndroidGraphicsStateGuardian *androidgsg;
93 DCAST_INTO_R(androidgsg, _gsg, false);
94 {
95 if (eglGetCurrentDisplay() == _egl_display &&
96 eglGetCurrentSurface(EGL_READ) == _egl_surface &&
97 eglGetCurrentSurface(EGL_DRAW) == _egl_surface &&
98 eglGetCurrentContext() == androidgsg->_context) {
99 // No need to make the context current again. Short-circuit this
100 // possibly-expensive call.
101 } else {
102 // Need to set the context.
103 if (!eglMakeCurrent(_egl_display, _egl_surface, _egl_surface, androidgsg->_context)) {
104 androiddisplay_cat.error() << "Failed to call eglMakeCurrent: "
105 << get_egl_error_string(eglGetError()) << "\n";
106 }
107 }
108 }
109
110 // Now that we have made the context current to a window, we can reset the
111 // GSG state if this is the first time it has been used. (We can't just
112 // call reset() when we construct the GSG, because reset() requires having a
113 // current context.)
114 androidgsg->reset_if_new();
115
116 if (mode == FM_render) {
117 // begin_render_texture();
118 clear_cube_map_selection();
119 }
120
121 _gsg->set_current_properties(&get_fb_properties());
122 return _gsg->begin_frame(current_thread);
123}

262bool AndroidGraphicsWindow::
263open_window() {
264 // GSG CreationInitialization
265 AndroidGraphicsStateGuardian *androidgsg;
266 if (_gsg == 0) {
267 // There is no old gsg. Create a new one.
268 androidgsg = new AndroidGraphicsStateGuardian(_engine, _pipe, nullptr);
269 androidgsg->choose_pixel_format(_fb_properties, false, false);
270 _gsg = androidgsg;
271 } else {
272 // If the old gsg has the wrong pixel format, create a new one that shares
273 // with the old gsg.
274 DCAST_INTO_R(androidgsg, _gsg, false);
275 if (!androidgsg->get_fb_properties().subsumes(_fb_properties)) {
276 androidgsg = new AndroidGraphicsStateGuardian(_engine, _pipe, androidgsg);
277 androidgsg->choose_pixel_format(_fb_properties, false, false);
278 _gsg = androidgsg;
279 }
280 }
281
282 // Register the callbacks
283 assert(_app != nullptr);
284 _app->userData = this;
285 _app->onAppCmd = handle_command;
286 _app->onInputEvent = handle_input_event;
287
288 // Wait until Android has opened the window.
289 while (_app->window == nullptr) {
290 process_events();
291 }
292
293 // create_surface should have been called by now.
294 if (_egl_surface == EGL_NO_SURFACE) {
295 return false;
296 }
297
298 // Set some other properties.
299 _properties.set_origin(0, 0);
300 _properties.set_cursor_hidden(true);
301 _properties.set_undecorated(true);
302
303 if (!androidgsg->get_fb_properties().verify_hardware_software
304 (_fb_properties, androidgsg->get_gl_renderer())) {
305 close_window();
306 return false;
307 }
308
309 _fb_properties = androidgsg->get_fb_properties();
310
311 return true;
312}

338bool AndroidGraphicsWindow::
339create_surface() {
340 AndroidGraphicsStateGuardian *androidgsg;
341 DCAST_INTO_R(androidgsg, _gsg, false);
342
343 // Reconfigure the window buffers to match that of our framebuffer config.
344 ANativeWindow_setBuffersGeometry(_app->window, 0, 0, androidgsg->_format);
345
346 // Set any window flags
347 uint32_t add_flags = 0;
348 uint32_t del_flags = 0;
349 if (_properties.get_fullscreen()) {
350 add_flags |= AWINDOW_FLAG_FULLSCREEN;
351 } else {
352 del_flags |= AWINDOW_FLAG_FULLSCREEN;
353 }
354 ANativeActivity_setWindowFlags(_app->activity, add_flags, del_flags);
355
356 // Create the EGL surface.
357 _egl_surface = eglCreateWindowSurface(_egl_display, androidgsg->_fbconfig, _app->window, nullptr);
358 if (eglGetError() != EGL_SUCCESS) {
359 androiddisplay_cat.error()
360 << "Failed to create window surface.\n";
361 return false;
362 }
363
364 // Create a context.
365 if (androidgsg->_context == EGL_NO_CONTEXT) {
366 if (!androidgsg->create_context()) {
367 return false;
368 }
369 }
370
371 // Switch to our newly created context.
372 if (!eglMakeCurrent(_egl_display, _egl_surface, _egl_surface, androidgsg->_context)) {
373 androiddisplay_cat.error() << "Failed to call eglMakeCurrent: "
374 << get_egl_error_string(eglGetError()) << "\n";
375 }
376
377 // Query the size of the surface. EGLint width, height;
378 // eglQuerySurface(_egl_display, _egl_surface, EGL_WIDTH, &width);
379 // eglQuerySurface(_egl_display, _egl_surface, EGL_HEIGHT, &height);
380
381 androidgsg->reset_if_new();
382 if (!androidgsg->is_valid()) {
383 close_window();
384 return false;
385 }
386
387 return true;
388}

404void AndroidGraphicsWindow::
405ns_handle_command(int32_t command) {
406 WindowProperties properties;
407
408 switch (command) {
409 case APP_CMD_SAVE_STATE:
410 // The system has asked us to save our current state. Do so.
411 // engine->app->savedState = malloc(sizeof(struct saved_state));
412 // *((struct saved_state*)engine->app->savedState) = engine->state;
413 // engine->app->savedStateSize = sizeof(struct saved_state);
414 break;
415 case APP_CMD_INIT_WINDOW:
416 // The window is being shown, get it ready.
417 if (_app->window != nullptr) {
418 create_surface();
419 properties.set_size(ANativeWindow_getWidth(_app->window),
420 ANativeWindow_getHeight(_app->window));
421 properties.set_minimized(false);
422 system_changed_properties(properties);
423 }
424 break;
425 case APP_CMD_CONFIG_CHANGED:
426 properties.set_size(ANativeWindow_getWidth(_app->window),
427 ANativeWindow_getHeight(_app->window));
428 system_changed_properties(properties);
429 break;
430 case APP_CMD_TERM_WINDOW:
431 destroy_surface();
432 properties.set_minimized(true);
433 system_changed_properties(properties);
434 break;
435 case APP_CMD_WINDOW_RESIZED:
436 properties.set_size(ANativeWindow_getWidth(_app->window),
437 ANativeWindow_getHeight(_app->window));
438 system_changed_properties(properties);
439 break;
440 case APP_CMD_WINDOW_REDRAW_NEEDED:
441 break;
442 case APP_CMD_CONTENT_RECT_CHANGED:
443 /*properties.set_origin(_app->contentRect.left, _app->contentRect.top);
444 properties.set_size(_app->contentRect.right - _app->contentRect.left,
445 _app->contentRect.bottom - _app->contentRect.top);
446 system_changed_properties(properties);*/
447 break;
448 case APP_CMD_GAINED_FOCUS:
449 properties.set_foreground(true);
450 system_changed_properties(properties);
451 break;
452 case APP_CMD_LOST_FOCUS:
453 properties.set_foreground(false);

532int32_t AndroidGraphicsWindow::
533handle_motion_event(const AInputEvent *event) {
534 int32_t action = AMotionEvent_getAction(event);
535 int32_t pointer_index = (action >> AMOTION_EVENT_ACTION_POINTER_INDEX_SHIFT);
536 action &= AMOTION_EVENT_ACTION_MASK;
537
538 if (action == AMOTION_EVENT_ACTION_DOWN) {
539 _primary_pointer_down = true;
540 }
541 else if (action == AMOTION_EVENT_ACTION_UP
542 || action == AMOTION_EVENT_ACTION_CANCEL) {
543 _primary_pointer_down = false;
544 }
545
546 int32_t button_state = AMotionEvent_getButtonState(event);
547
548 // Emulate mouse click; as long as the primary pointer is held down,
549 // and no other mouse button is causing it, the primary mouse button
550 // is considered depressed.
551 if (button_state == 0 && _primary_pointer_down &&
552 action != AMOTION_EVENT_ACTION_BUTTON_RELEASE) {
553 button_state |= AMOTION_EVENT_BUTTON_PRIMARY;
554 }
555
556 int32_t changed = _mouse_button_state ^ button_state;
557 if (changed != 0) {
558 if (changed & AMOTION_EVENT_BUTTON_PRIMARY) {
559 if (button_state & AMOTION_EVENT_BUTTON_PRIMARY) {
560 _input->button_down(MouseButton::one());
561 } else {
562 _input->button_up(MouseButton::one());
563 }
564 }
565 if (changed & AMOTION_EVENT_BUTTON_SECONDARY) {
566 if (button_state & AMOTION_EVENT_BUTTON_SECONDARY) {
567 _input->button_down(MouseButton::three());
568 } else {
569 _input->button_up(MouseButton::three());
570 }
571 }
572 if (changed & AMOTION_EVENT_BUTTON_TERTIARY) {
573 if (button_state & AMOTION_EVENT_BUTTON_TERTIARY) {
574 _input->button_down(MouseButton::two());
575 } else {
576 _input->button_up(MouseButton::two());
577 }
578 }
579 if (changed & AMOTION_EVENT_BUTTON_BACK) {
580 if (button_state & AMOTION_EVENT_BUTTON_BACK) {
581 _input->button_down(MouseButton::four());
582 } else {
583 _input->button_up(MouseButton::four());
584 }
585 }
586 if (changed & AMOTION_EVENT_BUTTON_FORWARD) {
587 if (button_state & AMOTION_EVENT_BUTTON_FORWARD) {
588 _input->button_down(MouseButton::five());
589 } else {
590 _input->button_up(MouseButton::five());
591 }
592 }
593 _mouse_button_state = button_state;
594 }
595
596 if (action == AMOTION_EVENT_ACTION_SCROLL) {
597 float v = AMotionEvent_getAxisValue(event, AMOTION_EVENT_AXIS_VSCROLL, pointer_index);
598 float h = AMotionEvent_getAxisValue(event, AMOTION_EVENT_AXIS_HSCROLL, pointer_index);
599
600 if (v > 0.5) {
601 while (v > 0) {
602 _input->button_down(MouseButton::wheel_up());
603 _input->button_up(MouseButton::wheel_up());
604 v -= 1;
605 }
606 }
607 else if (v < 0.5) {
608 while (v < 0) {
609 _input->button_down(MouseButton::wheel_down());
610 _input->button_up(MouseButton::wheel_down());
611 v += 1;
612 }
613 }
614
615 // In my tests the scroll direction is opposite of what the docs indicate,
616 // ie. a positive value is left, not right.
617 if (h > 0.5) {
618 while (h > 0) {
619 _input->button_down(MouseButton::wheel_left());
620 _input->button_up(MouseButton::wheel_left());
621 h -= 1;
622 }
623 }
624 else if (h < 0.5) {
625 while (h < 0) {
626 _input->button_down(MouseButton::wheel_right());
627 _input->button_up(MouseButton::wheel_right());
628 h += 1;
629 }
630 }
631 }
632
633 float x = AMotionEvent_getX(event, 0) - _app->contentRect.left;
634 float y = AMotionEvent_getY(event, 0) - _app->contentRect.top;
635
636 _input->set_pointer_in_window(x, y);
637
638 return 1;
639}
