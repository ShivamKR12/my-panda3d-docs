161bool
162encrypt_stream(istream &source, ostream &dest, const string &password,
163 const string &algorithm, int key_length, int iteration_count) {
164 OEncryptStream encrypt;
165 if (!algorithm.empty()) {
166 encrypt.set_algorithm(algorithm);
167 }
168 if (key_length > 0) {
169 encrypt.set_key_length(key_length);
170 }
171 if (iteration_count >= 0) {
172 encrypt.set_iteration_count(iteration_count);
173 }
174 encrypt.open(&dest, false, password);
175
176 static const size_t buffer_size = 4096;
177 char buffer[buffer_size];
178
179 source.read(buffer, buffer_size);
180 size_t count = source.gcount();
181 while (count != 0) {
182 encrypt.write(buffer, count);
183 source.read(buffer, buffer_size);
184 count = source.gcount();
185 }
186 encrypt.close();
187
188 return (!source.fail() || source.eof()) && (!encrypt.fail());
189}
