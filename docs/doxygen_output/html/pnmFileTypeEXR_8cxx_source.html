185bool PNMFileTypeEXR::
186matches_magic_number(const string &magic_number) const {
187 nassertr(magic_number.size() >= 2, false);
188
189 if (magic_number.size() >= 4) {
190 // If we have already read all four bytes, use the built-in
191 // function to check them.
192 return IMF::isImfMagic(magic_number.data());
193 } else {
194 // Otherwise, check only the first two bytes and call it good enough.
195 return magic_number[0] == ((IMF::MAGIC >> 0) & 0x00ff) &&
196 magic_number[1] == ((IMF::MAGIC >> 8) & 0x00ff);
197 }
198}

223PNMFileTypeEXR::Reader::
224Reader(PNMFileType *type, istream *file, bool owns_file, string magic_number) :
225 PNMReader(type, file, owns_file),
226 _strm(new ImfStdIstream(*_file, magic_number)),
227 _imf_file(*_strm)
228{
229 const IMF::Header &header = _imf_file.header();
230
231 IMATH_NAMESPACE::Box2i dw = header.dataWindow();
232 _x_size = dw.max.x - dw.min.x + 1;
233 _y_size = dw.max.y - dw.min.y + 1;
234
235 // Find the channels we care about, and ensure they're placed in the
236 // correct order.
237 _channel_names.clear();
238
239 const IMF::ChannelList &channels = header.channels();
240
241 // Note: including Y in this list allows us to handle grayscale or
242 // grayscale/alpha images correctly, but also incorrectly detects
243 // luminance/chroma images as grayscale only. However, these kind
244 // of images are a pain to handle anyway, so maybe that's OK.
245 const char *possible_channel_names[] = { "R", "G", "B", "Y", "A", nullptr };
246 for (const char **pni = possible_channel_names; *pni != nullptr; ++pni) {
247 std::string name = *pni;
248 IMF::ChannelList::ConstIterator ci = channels.find(name.c_str());
249 if (ci != channels.end()) {
250 // Found a match.
251 if (name == "Y" && !_channel_names.empty()) {
252 // Y is luminance or grayscale. Ignore Y if there are
253 // already any RGB channels.
254 } else {
255 _channel_names.push_back(name);
256 }
257 }
258 }
259
260 if (_channel_names.empty()) {
261 // Didn't find any channel names that match R, G, B, A, so just
262 // ask for RGB anyway and trust the OpenEXR library to do the
263 // right thing. Actually, it just fills them with black, but
264 // whatever.
265 _channel_names.push_back("R");
266 _channel_names.push_back("G");
267 _channel_names.push_back("B");
268 }
269
270 _num_channels = (int)_channel_names.size();
271 if (_num_channels == 0 || _num_channels > 4) {
272 _is_valid = false;
273 return;
274 }
275 // We read all OpenEXR files to floating-point, even UINT type, so
276 // _maxval doesn't matter. But we set it anyway.
277 _maxval = 65535;
278
279 _is_valid = true;
280}

307bool PNMFileTypeEXR::Reader::
308read_pfm(PfmFile &pfm) {
309 pfm.clear(_x_size, _y_size, _num_channels);
310 vector_float table;
311 pfm.swap_table(table);
312
313 PN_float32 *table_data = table.data();
314 size_t x_stride = sizeof(PN_float32) * pfm.get_num_channels();
315 size_t y_stride = x_stride * pfm.get_x_size();
316 nassertr(y_stride * pfm.get_y_size() <= table.size() * sizeof(PN_float32), false);
317
318 const IMF::Header &header = _imf_file.header();
319 IMATH_NAMESPACE::Box2i dw = header.dataWindow();
320
321 IMF::FrameBuffer frameBuffer;
322 for (int ci = 0; ci < pfm.get_num_channels(); ++ci) {
323 char *base = (char *)(table_data - (dw.min.x + dw.min.y * pfm.get_x_size()) * pfm.get_num_channels() + ci);
324 frameBuffer.insert(_channel_names[ci].c_str(),
325 IMF::Slice(IMF::FLOAT, base, x_stride, y_stride,
326 1, 1, 0.0));
327 }
328
329 _imf_file.setFrameBuffer(frameBuffer);
330
331 try {
332 _imf_file.readPixels(dw.min.y, dw.max.y);
333 } catch (const std::exception &exc) {
334 pnmimage_exr_cat.error()
335 << exc.what() << "\n";
336 return false;
337 }
338
339 pfm.swap_table(table);
340 return true;
341}

393bool PNMFileTypeEXR::Writer::
394write_pfm(const PfmFile &pfm) {
395 const vector_float &table = pfm.get_table();
396 const PN_float32 *table_data = table.data();
397 size_t x_stride = sizeof(PN_float32) * pfm.get_num_channels();
398 size_t y_stride = x_stride * pfm.get_x_size();
399 nassertr(y_stride * pfm.get_y_size() <= table.size() * sizeof(PN_float32), false);
400
401 const char *channel_names_1[] = { "G" };
402 const char *channel_names_2[] = { "G", "A" };
403 const char *channel_names_3[] = { "R", "G", "B" };
404 const char *channel_names_4[] = { "R", "G", "B", "A" };
405 const char **channel_names = nullptr;
406
407 switch (pfm.get_num_channels()) {
408 case 1:
409 channel_names = channel_names_1;
410 break;
411
412 case 2:
413 channel_names = channel_names_2;
414 break;
415
416 case 3:
417 channel_names = channel_names_3;
418 break;
419
420 case 4:
421 channel_names = channel_names_4;
422 break;
423
424 default:
425 return false;
426 };
427
428 IMF::Header header(pfm.get_x_size(), pfm.get_y_size());
429 for (int ci = 0; ci < pfm.get_num_channels(); ++ci) {
430 header.channels().insert(channel_names[ci], IMF::Channel(IMF::FLOAT));
431 }
432
433 IMF::FrameBuffer frameBuffer;
434 for (int ci = 0; ci < pfm.get_num_channels(); ++ci) {
435 const char *base = (const char *)(table_data + ci);
436 frameBuffer.insert(channel_names[ci],
437 IMF::Slice(IMF::FLOAT, (char *)base, x_stride, y_stride));
438 }
439
440 ImfStdOstream strm(*_file);
441 IMF::OutputFile file(strm, header);
442 file.setFrameBuffer(frameBuffer);
443
444 try {
445 file.writePixels(pfm.get_y_size());
446 } catch (const std::exception &exc) {
447 pnmimage_exr_cat.error()
448 << exc.what() << "\n";
449 return false;
450 }
451
452 return true;
453}
