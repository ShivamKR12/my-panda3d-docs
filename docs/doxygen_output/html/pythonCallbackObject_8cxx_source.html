117void PythonCallbackObject::
118do_python_callback(CallbackData *cbdata) {
119 nassertv(cbdata != nullptr);
120
121 // Wrap the cbdata up in a Python object, then put it in a tuple, for the
122 // argument list.
123 PyObject *pycbdata =
124 DTool_CreatePyInstanceTyped(cbdata, Dtool_TypedObject,
125 false, false, cbdata->get_type_index());
126 PyObject *args = Py_BuildValue("(O)", pycbdata);
127 Py_DECREF(pycbdata);
128
129 PyObject *result = PythonThread::call_python_func(_function, args);
130 Py_DECREF(args);
131
132 if (result == nullptr) {
133 if (PyErr_Occurred() != PyExc_SystemExit) {
134 util_cat.error()
135 << "Exception occurred in " << *this << "\n";
136 }
137 } else {
138 Py_DECREF(result);
139 }
