320void WinStatsStripChart::
321draw_slice(int x, int w, const PStatStripChart::FrameData &fdata) {
322 // Start by clearing the band first.
323 RECT rect = { x, 0, x + w, get_ysize() };
324 FillRect(_bitmap_dc, &rect, (HBRUSH)GetStockObject(WHITE_BRUSH));
325
326 double overall_time = 0.0;
327 int y = get_ysize();
328
329 FrameData::const_iterator fi;
330 for (fi = fdata.begin(); fi != fdata.end(); ++fi) {
331 const ColorData &cd = (*fi);
332 overall_time += cd._net_value;
333 HBRUSH brush = get_collector_brush(cd._collector_index, cd._collector_index == _highlighted_index);
334
335 if (overall_time > get_vertical_scale()) {
336 // Off the top. Go ahead and clamp it by hand, in case it's so far off
337 // the top we'd overflow the 16-bit pixel value.
338 rect.top = 0;
339 rect.bottom = y;
340 FillRect(_bitmap_dc, &rect, brush);
341 // And we can consider ourselves done now.
342 return;
343 }
344
345 int top_y = height_to_pixel(overall_time);
346 rect.top = top_y;
347 rect.bottom = y;
348 FillRect(_bitmap_dc, &rect, brush);
349 y = top_y;
350 }

469LONG WinStatsStripChart::
470graph_window_proc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam) {
471 switch (msg) {
472 case WM_LBUTTONDOWN:
473 if (_potential_drag_mode == DM_none) {
474 set_drag_mode(DM_scale);
475 int16_t y = HIWORD(lparam);
476 _drag_scale_start = pixel_to_height(y);
477 SetCapture(_graph_window);
478 return 0;
479
480 } else if (_potential_drag_mode == DM_guide_bar && _drag_guide_bar >= 0) {
481 set_drag_mode(DM_guide_bar);
482 int16_t y = HIWORD(lparam);
483 _drag_start_y = y;
484 SetCapture(_graph_window);
485 return 0;
486 }
487 break;
488
489 case WM_MOUSEMOVE:
490 if (_drag_mode == DM_none && _potential_drag_mode == DM_none) {
491 // When the mouse is over a color bar, highlight it.
492 int16_t x = LOWORD(lparam);
493 int16_t y = HIWORD(lparam);
494
495 int collector_index = get_collector_under_pixel(x, y);
496 _label_stack.highlight_label(collector_index);
497 on_enter_label(collector_index);
498
499 // Now we want to get a WM_MOUSELEAVE when the mouse leaves the graph
500 // window.
501 TRACKMOUSEEVENT tme = {
502 sizeof(TRACKMOUSEEVENT),
503 TME_LEAVE,
504 _graph_window,
505 0
506 };
507 TrackMouseEvent(&tme);
508 }
509 else {
510 // If the mouse is in some drag mode, stop highlighting.
511 _label_stack.highlight_label(-1);
512 on_leave_label(_highlighted_index);
513 }
514
515 if (_drag_mode == DM_scale) {
516 int16_t y = HIWORD(lparam);
517 double ratio = 1.0 - ((double)y / (double)get_ysize());
518 if (ratio > 0.0) {
519 double new_scale = _drag_scale_start / ratio;
520 if (!IS_NEARLY_EQUAL(get_vertical_scale(), new_scale)) {
521 // Disable smoothing while we do this expensive operation.
522 set_average_mode(false);
523 set_vertical_scale(_drag_scale_start / ratio);
524 }
525 }
526 return 0;
527
528 } else if (_drag_mode == DM_new_guide_bar) {
529 // We haven't created the new guide bar yet; we won't until the mouse
530 // comes within the graph's region.
531 int16_t y = HIWORD(lparam);
532 if (y >= 0 && y < get_ysize()) {
533 set_drag_mode(DM_guide_bar);
534 _drag_guide_bar = add_user_guide_bar(pixel_to_height(y));
535 return 0;
536 }
537
538 } else if (_drag_mode == DM_guide_bar) {
539 int16_t y = HIWORD(lparam);
540 move_user_guide_bar(_drag_guide_bar, pixel_to_height(y));
541 return 0;
542 }
543 break;
544
545 case WM_MOUSELEAVE:
546 // When the mouse leaves the graph, stop highlighting.
547 _label_stack.highlight_label(-1);
548 on_leave_label(_highlighted_index);
549 break;
550
551 case WM_LBUTTONUP:
552 if (_drag_mode == DM_scale) {
553 set_drag_mode(DM_none);
554 ReleaseCapture();
555 return 0;
556
557 } else if (_drag_mode == DM_guide_bar) {
558 int16_t y = HIWORD(lparam);
559 if (y < 0 || y >= get_ysize()) {
560 remove_user_guide_bar(_drag_guide_bar);
561 } else {
562 move_user_guide_bar(_drag_guide_bar, pixel_to_height(y));
563 }
564 set_drag_mode(DM_none);
565 ReleaseCapture();
566 return 0;
567 }
568 break;
569
570 case WM_LBUTTONDBLCLK:
571 {
572 // Double-clicking on a color bar in the graph is the same as double-
573 // clicking on the corresponding label.
574 int16_t x = LOWORD(lparam);
575 int16_t y = HIWORD(lparam);
576 on_click_label(get_collector_under_pixel(x, y));
577 return 0;
578 }
579 break;
580
581 case WM_CONTEXTMENU:
582 {
583 POINT point;
584 if (GetCursorPos(&point) && ScreenToClient(_graph_window, &point)) {
585 int collector_index = get_collector_under_pixel(point.x, point.y);
586 if (collector_index >= 0) {
587 on_popup_label(collector_index);
588 }
589 }
590 return 0;
591 }
592 break;
593
594 default:
595 break;
596 }
597
598 return WinStatsGraph::graph_window_proc(hwnd, msg, wparam, lparam);
599}

730draw_guide_bar(HDC hdc, int from_x, int to_x,
731 const PStatGraph::GuideBar &bar) {
732 int y = height_to_pixel(bar._height);
733
734 if (y > 0) {
735 // Only draw it if it's not too close to the top.
736 switch (bar._style) {

739 break;

741 case GBS_user:
742 SelectObject(hdc, _user_guide_bar_pen);
743 break;
744
745 case GBS_normal:
746 SelectObject(hdc, _dark_pen);
747 break;
748 }
749 MoveToEx(hdc, from_x, y, nullptr);

759int WinStatsStripChart::
760draw_guide_label(HDC hdc, int x, const PStatGraph::GuideBar &bar, int last_y) {
761 switch (bar._style) {
762 case GBS_target:
763 SetTextColor(hdc, _light_color);
764 break;
765
766 case GBS_user:
767 SetTextColor(hdc, _user_guide_bar_color);
768 break;
769
770 case GBS_normal:
771 SetTextColor(hdc, _dark_color);
772 break;
773 }
774
775 int y = height_to_pixel(bar._height);
776 const std::string &label = bar._label;
777 SIZE size;
778 GetTextExtentPoint32(hdc, label.data(), label.length(), &size);
779
780 if (bar._style != GBS_user) {
781 double from_height = pixel_to_height(y + size.cy);
782 double to_height = pixel_to_height(y - size.cy);
783 if (find_user_guide_bar(from_height, to_height) >= 0) {
784 // Omit the label: there's a user-defined guide bar in the same space.
785 return last_y;
786 }
787 }
788
789 int this_y = _graph_top + y - size.cy / 2;
790 if (y >= 0 && y < get_ysize() &&
791 (last_y < this_y || last_y > this_y + size.cy)) {
792 TextOut(hdc, x, this_y,
793 label.data(), label.length());
794 last_y = this_y;
795 }
796
797 return last_y;
798}
