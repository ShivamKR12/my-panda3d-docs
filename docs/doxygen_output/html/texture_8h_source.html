100 enum Format {
101 F_depth_stencil = 1,
102 F_color_index,
103 F_red,
104 F_green,
105 F_blue,
106 F_alpha,
107 F_rgb, // any suitable RGB mode, whatever the hardware prefers
108
109 // The following request a particular number of bits for the GSG's
110 // internal_format (as stored in the framebuffer), but this request is not
111 // related to the pixel storage within the Texture object itself, which is
112 // always get_num_components() * get_component_width().
113 F_rgb5, // 5 bits per R,G,B channel
114 F_rgb8, // 8 bits per R,G,B channel
115 F_rgb12, // 12 bits per R,G,B channel
116 F_rgb332, // 3 bits per R & G, 2 bits for B
117
118 F_rgba, // any suitable RGBA mode, whatever the hardware prefers
119
120 // Again, the following bitdepth requests are only for the GSG; within the
121 // Texture object itself, these are all equivalent.
122 F_rgbm, // as above, but only requires 1 bit for alpha (i.e. mask)
123 F_rgba4, // 4 bits per R,G,B,A channel
124 F_rgba5, // 5 bits per R,G,B channel, 1 bit alpha
125 F_rgba8, // 8 bits per R,G,B,A channel
126 F_rgba12, // 12 bits per R,G,B,A channel
127
128 F_luminance,
129 F_luminance_alpha, // 8 bits luminance, 8 bits alpha
130 F_luminance_alphamask, // 8 bits luminance, only needs 1 bit of alpha
131
132 F_rgba16, // 16 bits per R,G,B,A channel
133 F_rgba32, // 32 bits per R,G,B,A channel
134
135 F_depth_component,
136 F_depth_component16,
137 F_depth_component24,
138 F_depth_component32,
139
140 F_r16,
141 F_rg16,
142 F_rgb16,
143
144 // These formats are in the sRGB color space. RGB is 2.2 gamma corrected,

146 F_srgb,
147 F_srgb_alpha,
148 F_sluminance,

151 F_r32i, // 32-bit integer, used for atomic access
152 F_r32,

156 F_r8i, // 8 integer bits per R channel

202 enum CompressionMode {
203 // Generic compression modes. Usually, you should choose one of these.
204 CM_default, // on or off, according to compressed-textures
205 CM_off, // uncompressed image
206 CM_on, // whatever compression the driver supports
207
208 // Specific compression modes. Use only when you really want to use a
209 // particular compression algorithm. Use with caution; not all drivers
210 // support all compression modes. You can use
211 // GSG::get_supports_compressed_texture_format() to query the available
212 // compression modes for a particular GSG.
213 CM_fxt1,
214 CM_dxt1, // BC1: RGB with optional binary alpha.
215 CM_dxt2, // Like DXT3, but assumes premultiplied alpha
216 CM_dxt3, // BC2: RGB with uncompressed 4-bit alpha.
217 CM_dxt4, // Like DXT5, but assumes premultiplied alpha
218 CM_dxt5, // BC3: RGB with separately compressed 8-bit alpha.
219 CM_pvr1_2bpp,
220 CM_pvr1_4bpp,
221 CM_rgtc, // BC4/BC5: 1 or 2 channels, individually compressed.
222 CM_etc1,
223 CM_etc2,
224 CM_eac, // EAC: 1 or 2 channels.
225 };
