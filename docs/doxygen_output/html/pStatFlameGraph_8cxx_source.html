66void PStatFlameGraph::
67update() {
68 const PStatClientData *client_data = _monitor->get_client_data();
69
70 // Don't bother to update the thread data until we know at least something
71 // about the collectors and threads.
72 if (client_data->get_num_collectors() != 0 &&
73 client_data->get_num_threads() != 0) {
74 const PStatThreadData *thread_data =
75 client_data->get_thread_data(_thread_index);
76 if (!thread_data->is_empty()) {
77 if (_frame_number >= 0) {
78 if (thread_data->has_frame(_frame_number)) {
79 if (_current_frame != _frame_number) {
80 _current_frame = _frame_number;
81 update_data();
82 }
83 }
84 } else {
85 int frame_number = thread_data->get_latest_frame_number();
86 if (frame_number != _current_frame) {
87 _current_frame = frame_number;
88
89 update_data();
90 }
91 }
92 }
93 }
94
95 idle();
96}

408void PStatFlameGraph::
409update_data() {
410 const PStatClientData *client_data = _monitor->get_client_data();
411 if (client_data == nullptr) {
412 return;
413 }
414
415 const PStatThreadData *thread_data = client_data->get_thread_data(_thread_index);
416 if (thread_data == nullptr || thread_data->is_empty()) {
417 return;
418 }
419
420 const PStatFrameData &frame_data = thread_data->get_frame(_current_frame);
421
422 bool first_time = _stack._children.empty();
423
424 StackLevel *top = &_stack;
425 top->reset();
426
427 size_t num_events = frame_data.get_num_events();
428 for (size_t ei = 0; ei < num_events; ++ei) {
429 int collector_index = frame_data.get_time_collector(ei);
430 double time = frame_data.get_time(ei);
431
432 if (frame_data.is_start(ei)) {
433 // If we have a collector index, use it to determine which bottom-level
434 // stack frames we are interested in.
435 if (_collector_index < 0 ||
436 _collector_index == collector_index ||
437 top != &_stack) {
438 top = top->start(collector_index, time);
439 }
440 else {
441 // Check whether one of the parents matches, perhaps.
442 int parent_index = collector_index;
443 do {
444 const PStatCollectorDef &def = client_data->get_collector_def(parent_index);
445 if (parent_index == def._parent_index) {
446 break;
447 }
448 parent_index = def._parent_index;
449 if (parent_index == _collector_index) {
450 // Yes, let it through.
451 top = top->start(collector_index, time);
452 break;
453 }
454 }
455 while (parent_index >= 0 && client_data->has_collector(parent_index));
456 }
457 }
458 else {
459 top = top->stop(collector_index, time);
460 }
461 }
462 top = top->stop_all(frame_data.get_end());
463 nassertv(top == &_stack);
464
465 if (first_time) {
466 _stack.reset_averages();
467 }
468
469 if (!_average_mode) {
470 // Redraw right away, except in average mode, where it's done in animate().
471 _time_width = _stack.get_net_value(false);
472 if (_time_width == 0.0) {
473 _time_width = 1.0 / pstats_target_frame_rate;
474 }
475 normal_guide_bars();
476 force_redraw();
477 }
478}

562bool PStatFlameGraph::
563animate(double time, double dt) {
564 if (!_average_mode) {
565 return false;
566 }
567
568 if (_stack.update_averages(_average_cursor)) {
569 _time_width = _stack.get_net_value(true);
570 if (_time_width == 0.0) {
571 _time_width = 1.0 / pstats_target_frame_rate;
572 }
573 normal_guide_bars();
574 }
575
576 // Always use force_redraw, since the mouse position may have changed.
577 force_redraw();
578
579 // Cycle through the ring buffers.
580 _average_cursor = (_average_cursor + 1) % _num_average_frames;

668bool PStatFlameGraph::StackLevel::
669update_averages(size_t cursor) {
670 _values[cursor] = get_net_value(false);
671
672 bool changed = false;
673
674 double sum = 0;
675 for (double value : _values) {
676 sum += value;
677 }
678 double avg = sum / _num_average_frames;
679 if (avg != _avg_net_value) {
680 _avg_net_value = avg;
681 changed = true;
682 }
683
684 for (auto &item : _children) {
685 if (item.second.update_averages(cursor)) {
686 changed = true;
687 }
688 }

696const PStatFlameGraph::StackLevel *PStatFlameGraph::StackLevel::
697locate(int depth, double time, bool average) const {
698 if (time < 0.0) {
699 return nullptr;
700 }
701 for (const auto &item : _children) {
702 double value = item.second.get_net_value(average);
703 if (time < value) {
704 if (depth == 0) {
705 // This is it.
706 return &item.second;
707 } else {
708 // Recurse.
709 return item.second.locate(depth - 1, time, average);
710 }
711 }
712 time -= value;
713 }
714 return nullptr;
715}
