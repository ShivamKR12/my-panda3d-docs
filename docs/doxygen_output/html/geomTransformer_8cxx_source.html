169bool GeomTransformer::
170transform_texcoords(Geom *geom, const InternalName *from_name,
171 InternalName *to_name, const LMatrix4 &mat) {
172 PStatTimer timer(_apply_texcoord_collector);
173
174 nassertr(geom != nullptr, false);
175
176 SourceTexCoords st;
177 st._mat = mat;
178 st._from = from_name;
179 st._to = to_name;
180 st._vertex_data = geom->get_vertex_data();
181
182 NewVertexData &new_data = _texcoords[st];
183 if (new_data._vdata.is_null()) {
184 if (!st._vertex_data->has_column(from_name)) {
185 // No from_name column; no change.
186 return false;
187 }
188
189 PT(GeomVertexData) new_vdata;
190
191 // We have not yet converted these texcoords. Do so now.
192 if (st._vertex_data->has_column(to_name)) {
193 new_vdata = new GeomVertexData(*st._vertex_data);
194 } else {
195 const GeomVertexColumn *old_column =
196 st._vertex_data->get_format()->get_column(from_name);
197 new_vdata = st._vertex_data->replace_column
198 (to_name, old_column->get_num_components(),
199 old_column->get_numeric_type(),
200 old_column->get_contents());
201 }
202
203 CPT(GeomVertexFormat) format = new_vdata->get_format();
204
205 GeomVertexWriter tdata(new_vdata, to_name);
206 GeomVertexReader fdata(new_vdata, from_name);
207
208 while (!fdata.is_at_end()) {
209 const LPoint4 &coord = fdata.get_data4();
210 tdata.set_data4(coord * mat);
211 }
212 new_data._vdata = new_vdata;
213 }
214
215 geom->set_vertex_data(new_data._vdata);
216 if (st._vertex_data->get_ref_count() > 1) {
217 _vdata_assoc[new_data._vdata]._might_have_unused = true;
218 _vdata_assoc[st._vertex_data]._might_have_unused = true;
219 }
220
221 return true;
222}

387bool GeomTransformer::
388apply_texture_colors(Geom *geom, TextureStage *ts, Texture *tex,
389 const TexMatrixAttrib *tma, const LColor &base_color,
390 bool keep_vertex_color) {
391 PStatTimer timer(_apply_texture_color_collector);
392
393 nassertr(geom != nullptr, false);
394
395 PT(TexturePeeker) peeker = tex->peek();
396 if (peeker == nullptr) {
397 return false;
398 }
399
400 if (peeker->get_x_size() == 1 &&
401 peeker->get_y_size() == 1 &&
402 peeker->get_z_size() == 1) {
403 // If it's just a one-pixel texture (e.g. a simple ram image), don't
404 // bother scanning the UV's. Just extract the color and apply it.
405 LColor color;
406 peeker->lookup(color, 0.0f, 0.0f);
407 color.set(color[0] * base_color[0],
408 color[1] * base_color[1],
409 color[2] * base_color[2],
410 color[3] * base_color[3]);
411 if (keep_vertex_color) {
412 return transform_colors(geom, color);
413 } else {
414 return set_color(geom, color);
415 }
416 }
417
418 bool got_mat = false;
419 LMatrix4 mat = LMatrix4::ident_mat();
420 if (tma != nullptr && tma->has_stage(ts)) {
421 mat = tma->get_mat(ts);
422 got_mat = !mat.almost_equal(LMatrix4::ident_mat());
423 }
424
425 // This version of the code just applied one overall flat color to the
426 // entire mesh. Turned out not to be good enough. Instead, we'll look up
427 // each vertex in the texture map and apply the nearest color to the vertex.
428 /*
429 // Scan the UV's to get the used range. This is particularly necessary for
430 // palettized textures.
431
432 LPoint3 min_point, max_point;
433 bool found_any = false;
434 geom->calc_tight_bounds(min_point, max_point, found_any,
435 geom->get_vertex_data(),
436 got_mat, mat,
437 ts->get_texcoord_name(),
438 Thread::get_current_thread());
439 if (found_any) {
440 // Now use that UV range to determine the overall color of the geom's
441 // texture.
442 LColor color;
443 peeker->filter_rect(color,
444 min_point[0], min_point[1], min_point[2],
445 max_point[0], max_point[1], max_point[2]);
446 color.set(color[0] * base_color[0],
447 color[1] * base_color[1],
448 color[2] * base_color[2],
449 color[3] * base_color[3]);
450 if (keep_vertex_color) {
451 return transform_colors(geom, color);
452 } else {
453 return set_color(geom, color);
454 }
455 }
456
457 return false;
458 */
459
460 SourceTextureColors stc;
461 stc._ts = ts;
462 stc._tex = tex;
463 stc._tma = tma;
464 stc._base_color = base_color;
465 stc._keep_vertex_color = keep_vertex_color;
466 stc._vertex_data = geom->get_vertex_data();
467
468 NewVertexData &new_data = _tex_colors[stc];
469 if (new_data._vdata.is_null()) {
470 // We have not yet applied these texture colors. Do so now.
471
472 PT(GeomVertexData) vdata;
473
474 // Make sure the vdata has a color column.
475 if (stc._vertex_data->has_column(InternalName::get_color())) {
476 vdata = new GeomVertexData(*stc._vertex_data);
477 }
478 else if (vertex_colors_prefer_packed) {
479 // Create a color column where there wasn't one before.
480 vdata = new GeomVertexData(*stc._vertex_data->set_color
481 (LColor(1.0f, 1.0f, 1.0f, 1.0f), 1, Geom::NT_packed_dabc, Geom::C_color));
482 keep_vertex_color = false;
483 }
484 else {
485 vdata = new GeomVertexData(*stc._vertex_data->set_color
486 (LColor(1.0f, 1.0f, 1.0f, 1.0f), 4, Geom::NT_uint8, Geom::C_color));
487 keep_vertex_color = false;
488 }
489
490 // Check whether it has 2-d or 3-d texture coordinates.
491 bool tex3d = false;
492 const GeomVertexColumn *column = vdata->get_format()->get_column(ts->get_texcoord_name());
493 if (column == nullptr) {
494 return false;
495 }
496 if (column->get_num_components() >= 3) {
497 tex3d = true;
498 }
499
500 // Now walk through the vertices and apply each color from the texture as
501 // we go.
502 if (keep_vertex_color) {
503 // We want to modulate the existing vertex color.
504 GeomVertexReader gtexcoord(vdata, ts->get_texcoord_name());
505 GeomVertexRewriter gcolor(vdata, InternalName::get_color());
506
507 if (got_mat || tex3d) {
508 while (!gtexcoord.is_at_end()) {
509 LTexCoord3 p = gtexcoord.get_data3();
510 LColor c = gcolor.get_data4();
511 p = p * mat;
512 LColor color;
513 peeker->lookup(color, p[0], p[1], p[2]);
514 color.set(color[0] * base_color[0] * c[0],
515 color[1] * base_color[1] * c[1],
516 color[2] * base_color[2] * c[2],
517 color[3] * base_color[3] * c[3]);
518 gcolor.set_data4(color);
519 }
520 } else {
521 while (!gtexcoord.is_at_end()) {
522 LTexCoord p = gtexcoord.get_data2();
523 LColor c = gcolor.get_data4();
524 LColor color;
525 peeker->lookup(color, p[0], p[1]);
526 color.set(color[0] * base_color[0] * c[0],
527 color[1] * base_color[1] * c[1],
528 color[2] * base_color[2] * c[2],
529 color[3] * base_color[3] * c[3]);
530 gcolor.set_data4(color);
531 }
532 }
533 } else {
534 // We want to replace any existing vertex color.
535 GeomVertexReader gtexcoord(vdata, ts->get_texcoord_name());
536 GeomVertexWriter gcolor(vdata, InternalName::get_color());
537
538 if (got_mat || tex3d) {
539 while (!gtexcoord.is_at_end()) {
540 LTexCoord3 p = gtexcoord.get_data3();
541 p = p * mat;
542 LColor color;
543 peeker->lookup(color, p[0], p[1], p[2]);
544 color.set(color[0] * base_color[0],
545 color[1] * base_color[1],
546 color[2] * base_color[2],
547 color[3] * base_color[3]);
548 gcolor.set_data4(color);
549 }
550 } else {
551 while (!gtexcoord.is_at_end()) {
552 LTexCoord p = gtexcoord.get_data2();
553 LColor color;
554 peeker->lookup(color, p[0], p[1]);
555 color.set(color[0] * base_color[0],
556 color[1] * base_color[1],
557 color[2] * base_color[2],
558 color[3] * base_color[3]);
559 gcolor.set_data4(color);
560 }
561 }
562 }
563
564 new_data._vdata = vdata;
565 }
566
567 geom->set_vertex_data(new_data._vdata);
568 if (stc._vertex_data->get_ref_count() > 1) {
569 _vdata_assoc[new_data._vdata]._might_have_unused = true;
570 _vdata_assoc[stc._vertex_data]._might_have_unused = true;
571 }
572
573 return true;
574}

588bool GeomTransformer::
589apply_texture_colors(GeomNode *node, const RenderState *state) {
590 bool any_changed = false;
591
592 GeomNode::CDWriter cdata(node->_cycler);
593 GeomNode::GeomList::iterator gi;
594 PT(GeomNode::GeomList) geoms = cdata->modify_geoms();
595 for (gi = geoms->begin(); gi != geoms->end(); ++gi) {
596 GeomNode::GeomEntry &entry = (*gi);
597 CPT(RenderState) geom_state = state->compose(entry._state);
598
599 const TextureAttrib *ta = DCAST(TextureAttrib, geom_state->get_attrib(TextureAttrib::get_class_slot()));
600 if (ta != nullptr) {
601 CPT(TextureAttrib) ta2 = ta->filter_to_max(1);
602 if (ta2->get_num_on_stages() > 0) {
603 TextureStage *ts = ta2->get_on_stage(0);
604 Texture *tex = ta2->get_on_texture(ts);
605 const TexMatrixAttrib *tma = DCAST(TexMatrixAttrib, geom_state->get_attrib(TexMatrixAttrib::get_class_slot()));
606
607 const ColorAttrib *ca = DCAST(ColorAttrib, geom_state->get_attrib(ColorAttrib::get_class_slot()));
608 LColor base_color(1.0f, 1.0f, 1.0f, 1.0f);
609 bool keep_vertex_color = true;
610 if (ca != nullptr && ca->get_color_type() == ColorAttrib::T_flat) {
611 base_color = ca->get_color();
612 keep_vertex_color = false;
613 }
614
615 PT(Geom) new_geom = entry._geom.get_read_pointer()->make_copy();
616 if (apply_texture_colors(new_geom, ts, tex, tma, base_color, keep_vertex_color)) {
617 entry._geom = new_geom;
618 any_changed = true;
619
620 if (new_geom->get_vertex_data()->has_column(InternalName::get_color())) {
621 // Ensure we have a ColorAttrib::make_vertex() attrib.
622 CPT(RenderState) color_state = entry._state->set_attrib(ColorAttrib::make_vertex());
623 if (entry._state != color_state) {
624 entry._state = color_state;
625 any_changed = true;
626 }
627 }
628 }
629
630 // Also remove any texture references from the GeomState.
631 CPT(RenderState) no_tex_state = entry._state->remove_attrib(TextureAttrib::get_class_slot());
632 if (entry._state != no_tex_state) {
633 entry._state = no_tex_state;
634 any_changed = true;
635 }
636 }
637 }
638 }
639
640 return any_changed;
641}

752 if (node->get_num_geoms() < 2) {
753 return false;
754 }
755
756 GeomNode::CDWriter cdata(node->_cycler);
757 PT(GeomNode::GeomList) geoms = cdata->modify_geoms();
758
759 // For each geom, calculate a canonicalized RenderState, and classify all
760 // the geoms according to that. By "canonicalize" here, we simply mean
761 // removing the ColorAttrib.
762
763 typedef pmap <CPT(RenderState), pvector<int> > StateTable;
764 StateTable state_table;
765
766 for (int i = 0; i < (int)geoms->size(); i++) {
767 GeomNode::GeomEntry &entry = (*geoms)[i];
768 CPT(RenderState) canon = entry._state->remove_attrib(ColorAttrib::get_class_slot());
769 state_table[canon].push_back(i);
770 }
771
772 // For each group of geoms, check for mismatch.
773
774 bool any_changed = false;
775 StateTable::iterator si;
776 for (si = state_table.begin(); si != state_table.end(); si++) {
777
778 // If the geoms in the group already have the same RenderStates, then
779 // nothing needs to be done to this group.
780
781 const pvector<int> &indices = (*si).second;
782 bool mismatch = false;
783 for (int i = 1; i < (int)indices.size(); i++) {
784 if ((*geoms)[indices[i]]._state != (*geoms)[indices[0]]._state) {
785 mismatch = true;
786 break;
787 }
788 }
789 if (!mismatch) {
790 continue;
791 }
792
793 // The geoms do not have the same RenderState, but they could, since their
794 // canonicalized states are the same. Canonicalize them, by applying the
795 // colors to the vertices.
796
797 const RenderState *canon_state = (*si).first;
798 for (int i = 0; i < (int)indices.size(); i++) {
799 GeomNode::GeomEntry &entry = (*geoms)[indices[i]];
800 const RenderAttrib *ra = entry._state->get_attrib_def(ColorAttrib::get_class_slot());
801 const ColorAttrib *ca = DCAST(ColorAttrib, ra);
802 if (ca->get_color_type() == ColorAttrib::T_vertex) {
803 // All we need to do is ensure that the geom has a color column.
804 if (!entry._geom.get_read_pointer()->get_vertex_data()->has_column(InternalName::get_color())) {
805 PT(Geom) new_geom = entry._geom.get_read_pointer()->make_copy();
806 if (set_color(new_geom, LColor(1,1,1,1))) {
807 entry._geom = new_geom;
808 }
809 }
810 } else {
811 // A flat color (or "off", which is white). Set the vertices to the
812 // indicated flat color.
813 LColor c = ca->get_color();
814 PT(Geom) new_geom = entry._geom.get_read_pointer()->make_copy();
815 if (set_color(new_geom, c)) {
816 entry._geom = new_geom;
817 }
818 }
819 entry._state = canon_state->add_attrib(ColorAttrib::make_vertex());
820 any_changed = true;
821 }
822 }
823
824 return any_changed;
825}

832reverse_normals(Geom *geom) {
833 nassertr(geom != nullptr, false);
834 CPT(GeomVertexData) orig_data = geom->get_vertex_data();

868bool GeomTransformer::
869doubleside(GeomNode *node) {
870 int num_geoms = node->get_num_geoms();
871 for (int i = 0; i < num_geoms; ++i) {
872 CPT(Geom) orig_geom = node->get_geom(i);
873 bool has_normals = (orig_geom->get_vertex_data()->has_column(InternalName::get_normal()));
874 if (has_normals) {
875 // If the geometry has normals, we have to duplicate it to reverse the
876 // normals on the duplicate copy.
877 PT(Geom) new_geom = orig_geom->reverse();
878 reverse_normals(new_geom);
879 node->add_geom(new_geom, node->get_geom_state(i));

881 } else {
882 // If there are no normals, we can just doubleside it in place. This is
883 // preferable because we can share vertices.
884 orig_geom.clear();
885 node->modify_geom(i)->doubleside_in_place();
886 }
887 }
888
889 return (num_geoms != 0);

955int GeomTransformer::
956collect_vertex_data(Geom *geom, int collect_bits, bool format_only) {
957 CPT(GeomVertexData) vdata = geom->get_vertex_data();
958 if (vdata->get_num_rows() > _max_collect_vertices) {
959 // Don't even bother.
960 return 0;
961 }
962
963 CPT(GeomVertexFormat) format = vdata->get_format();
964
965 NewCollectedKey key;
966 if ((collect_bits & SceneGraphReducer::CVD_name) != 0) {
967 key._name = vdata->get_name();
968 }
969 if ((collect_bits & SceneGraphReducer::CVD_format) != 0) {
970 key._format = format;
971 }
972 if ((collect_bits & SceneGraphReducer::CVD_usage_hint) != 0) {
973 key._usage_hint = vdata->get_usage_hint();
974 } else {
975 key._usage_hint = Geom::UH_unspecified;
976 }
977 if ((collect_bits & SceneGraphReducer::CVD_animation_type) != 0) {
978 key._animation_type = format->get_animation().get_animation_type();
979 } else {
980 key._animation_type = Geom::AT_none;
981 }
982
983 AlreadyCollectedMap::const_iterator ai;
984 ai = _already_collected_map.find(vdata);
985 if (ai != _already_collected_map.end()) {
986 // We've previously collected this vertex data; reuse it.
987 const AlreadyCollectedData &acd = (*ai).second;
988 SourceGeom source_geom;
989 source_geom._geom = geom;
990 source_geom._vertex_offset = acd._vertex_offset;
991 acd._ncd->_source_geoms.push_back(source_geom);
992 return 0;
993 }
994
995 // We haven't collected this vertex data yet; associate it with a new data.
996 NewCollectedMap::iterator ni = _new_collected_map.find(key);
997 NewCollectedData *ncd;
998 if (ni != _new_collected_map.end()) {
999 ncd = (*ni).second;
1000
1001 } else {
1002 // We haven't encountered a compatible GeomVertexData before. Create a
1003 // new entry.
1004 ncd = new NewCollectedData(vdata);
1005 _new_collected_list.push_back(ncd);
1006 _new_collected_map[key] = ncd;
1007 }
1008
1009 if (ncd->_new_format != format) {
1010 ncd->_new_format = format->get_union_format(ncd->_new_format);
1011 }
1012
1013 int this_num_vertices = vdata->get_num_rows();
1014
1015 if (!format_only &&
1016 ncd->_num_vertices + this_num_vertices > _max_collect_vertices) {
1017 // Whoa, hold the phone! Too many vertices going into this one
1018 // GeomVertexData object; we'd better start over.
1019 ncd = new NewCollectedData(vdata);
1020 _new_collected_list.push_back(ncd);
1021 _new_collected_map[key] = ncd;
1022 }
1023
1024 int vertex_offset = ncd->_num_vertices;
1025
1026 AlreadyCollectedData &acd = _already_collected_map[vdata];
1027 acd._ncd = ncd;
1028 acd._vertex_offset = vertex_offset;
1029
1030 SourceGeom source_geom;
1031 source_geom._geom = geom;
1032 source_geom._vertex_offset = vertex_offset;
1033 ncd->_source_geoms.push_back(source_geom);
1034
1035 SourceData source_data;
1036 source_data._vdata = vdata;
1037 source_data._num_vertices = this_num_vertices;
1038
1039 ncd->_source_datas.push_back(source_data);
1040 ncd->_num_vertices += this_num_vertices;
1041
1042 return 0;
1043}

1060 int num_adjusted = 0;

1062
1063 GeomNode::CDWriter cdata(node->_cycler);

1065 PT(GeomNode::GeomList) geoms = cdata->modify_geoms();

1166int GeomTransformer::NewCollectedData::
1167apply_format_only_changes() {
1168 int num_modified = 0;
1169
1170 // We probably don't need to use a map, since GeomVertexData::convert_to()
1171 // already caches its result, but we do it anyway just in case there's
1172 // danger of overflowing the cache. What the heck, it's easy to do.
1173 typedef pmap< CPT(GeomVertexData), CPT(GeomVertexData) > VDataMap;
1174 VDataMap vdata_map;
1175
1176 SourceGeoms::iterator sgi;
1177 for (sgi = _source_geoms.begin(); sgi != _source_geoms.end(); ++sgi) {
1178 SourceGeom &sg = (*sgi);
1179 CPT(GeomVertexData) orig_data = sg._geom->get_vertex_data();
1180
1181 if (orig_data->get_format() != _new_format) {
1182 VDataMap::iterator mi = vdata_map.find(orig_data);
1183 if (mi != vdata_map.end()) {
1184 // Already modified this vdata.
1185 sg._geom->set_vertex_data((*mi).second);
1186
1187 } else {
1188 // Modify this vdata to the new format.
1189 CPT(GeomVertexData) new_data = orig_data->convert_to(_new_format);
1190 vdata_map[orig_data] = new_data;
1191 ++num_modified;
1192
1193 sg._geom->set_vertex_data(new_data);
1194 }
1195 }
1196 }
1197
1198 return num_modified;
1199}

1205int GeomTransformer::NewCollectedData::
1206apply_collect_changes() {
1207 if (_num_vertices == 0) {
1208 return 0;
1209 }
1210
1211 _new_data =
1212 new GeomVertexData(_vdata_name, _new_format, _usage_hint);
1213
1214 _new_data->unclean_set_num_rows(_num_vertices);
1215
1216 // Copy each source data into the new GeomVertexData, one at a time.
1217 int vertex_offset = 0;
1218 SourceDatas::iterator sdi;
1219 for (sdi = _source_datas.begin(); sdi != _source_datas.end(); ++sdi) {
1220 SourceData &sd = (*sdi);
1221 CPT(GeomVertexData) vdata = sd._vdata;
1222
1223 if (_new_format != vdata->get_format()) {
1224 // Convert (non-destructively) the current Geom's vertex data to the new
1225 // format, so we can just blindly append the vertices to _new_data,
1226 // within append_vdata().
1227 vdata = vdata->convert_to(_new_format);
1228 }
1229
1230 append_vdata(vdata, vertex_offset);
1231 vertex_offset += sd._num_vertices;
1232 }
1233
1234 nassertr(vertex_offset == _num_vertices, 0);
1235
1236 if (_new_btable != nullptr) {
1237 _new_btable->set_rows(_new_btable_rows);
1238 _new_data->set_transform_blend_table(_new_btable);
1239 }
1240
1241 update_geoms();
1242
1243 _new_data.clear();
1244 _new_btable.clear();
1245 _new_btable_rows.clear();
1246
1247 return 1;
1248}
1249
1254void GeomTransformer::NewCollectedData::
1255append_vdata(const GeomVertexData *vdata, int vertex_offset) {
1256 for (size_t i = 0; i < vdata->get_num_arrays(); ++i) {
1257 PT(GeomVertexArrayDataHandle) new_handle = _new_data->modify_array_handle(i);
1258 CPT(GeomVertexArrayDataHandle) old_handle = vdata->get_array_handle(i);
1259 size_t stride = (size_t)_new_format->get_array(i)->get_stride();
1260 size_t start_byte = (size_t)vertex_offset * stride;
1261 size_t copy_bytes = old_handle->get_data_size_bytes();
1262 nassertv(start_byte + copy_bytes <= new_handle->get_data_size_bytes());
1263
1264 new_handle->copy_subdata_from(start_byte, copy_bytes, old_handle, 0, copy_bytes);
1265 }
1266
1267 // Also, copy the animation data (if any). This means combining transform
1268 // andor slider tables, and might therefore mean remapping transform indices
1269 // in the vertices. Each of these has a slightly different way to handle
1270 // the remapping, because they have slightly different kinds of data.
1271
1272 if (vdata->get_transform_table() != nullptr ||
1273 _new_data->get_transform_table() != nullptr) {
1274 // The TransformTable.
1275 CPT(TransformTable) old_table;
1276 if (vdata->get_transform_table() != nullptr) {
1277 old_table = vdata->get_transform_table();
1278 } else {
1279 PT(TransformTable) temp_table = new TransformTable;
1280 // There's an implicit identity transform for all nodes.
1281 PT(VertexTransform) identity_transform = new UserVertexTransform("identity");
1282 temp_table->add_transform(identity_transform);
1283 old_table = TransformTable::register_table(temp_table);
1284 }
1285
1286 // First, build a mapping of the transforms we already have in the current
1287 // table. We must do this because the TransformTable doesn't
1288 // automatically unquify index numbers for us (it doesn't store an index).
1289 typedef pmap<const VertexTransform *, int> AddedTransforms;
1290 AddedTransforms added_transforms;
1291
1292 int num_old_transforms = old_table->get_num_transforms();
1293 for (int i = 0; i < num_old_transforms; i++) {
1294 added_transforms[old_table->get_transform(i)] = i;
1295 }
1296
1297 // Now create a new table. We have to create a new table instead of
1298 // modifying the existing one, since a registered TransformTable cannot be
1299 // modified.
1300 PT(TransformTable) new_table;
1301 if (_new_data->get_transform_table() != nullptr) {
1302 new_table = new TransformTable(*_new_data->get_transform_table());
1303 } else {
1304 new_table = new TransformTable;
1305 }
1306
1307 // Now walk through the old table and copy over its transforms. We will
1308 // build up an IndexMap of old index numbers to new index numbers while we
1309 // go, which we can use to modify the vertices.
1310 IndexMap transform_map;
1311
1312 int num_transforms = old_table->get_num_transforms();
1313 transform_map.reserve(num_transforms);
1314 for (int ti = 0; ti < num_transforms; ++ti) {
1315 const VertexTransform *transform = old_table->get_transform(ti);
1316 AddedTransforms::iterator ai = added_transforms.find(transform);
1317 if (ai != added_transforms.end()) {
1318 // Already got this one in the table.
1319 transform_map.push_back((*ai).second);
1320 } else {
1321 // This is a new one.
1322 int tj = new_table->add_transform(transform);
1323 transform_map.push_back(tj);
1324 added_transforms[transform] = tj;
1325 }
1326 }
1327 _new_data->set_transform_table(TransformTable::register_table(new_table));
1328
1329 // And now modify the vertices to update the indices to their new values
1330 // in the new table. This requires a nested loop, since each column of
1331 // transform_index might define multiple index values.
1332 GeomVertexRewriter index(_new_data, InternalName::get_transform_index());
1333 if (index.has_column()) {
1334 int num_values = index.get_column()->get_num_values();
1335 int num_rows = vdata->get_num_rows();
1336
1337 index.set_row_unsafe(vertex_offset);
1338 for (int ci = 0; ci < num_rows; ++ci) {
1339 LVecBase4i indices = index.get_data4i();
1340 for (int i = 0; i < num_values; i++) {
1341 nassertv(indices[i] >= 0 && indices[i] < (int)transform_map.size());
1342 indices[i] = transform_map[indices[i]];
1343 }
1344 index.set_data4i(indices);
1345 }
1346 }
1347 }
1348
1349 if (vdata->get_transform_blend_table() != nullptr) {
1350 // The TransformBlendTable. This one is the easiest, because we can
1351 // modify it directly, and it will uniquify blend objects for us.
1352
1353 // We have a few special optimizations to handle the TransformBlendTable,
1354 // since it's a very common case and therefore worth spending a bit of
1355 // effort to optimize deeply.
1356
1357 CPT(TransformBlendTable) old_btable = vdata->get_transform_blend_table();
1358
1359 if (_new_btable == nullptr) {
1360 _new_btable = new TransformBlendTable;
1361 _new_btable->add_blend(TransformBlend());
1362 }
1363
1364 SparseArray new_rows = old_btable->get_rows();
1365 new_rows <<= vertex_offset;
1366 _new_btable_rows |= new_rows;
1367
1368 // We still need to build up the IndexMap.
1369 IndexMap blend_map;
1370
1371 int num_blends = old_btable->get_num_blends();
1372 blend_map.reserve(num_blends);
1373 for (int bi = 0; bi < num_blends; ++bi) {
1374 int bj = _new_btable->add_blend(old_btable->get_blend(bi));
1375 blend_map.push_back(bj);
1376 }
1377
1378 // Modify the indices. This is simpler than the transform_index, above,
1379 // because each column of transform_blend may only define one index value.
1380 GeomVertexRewriter index(_new_data, InternalName::get_transform_blend());
1381 if (index.has_column()) {
1382 int num_rows = vdata->get_num_rows();
1383 index.set_row_unsafe(vertex_offset);
1384
1385 for (int ci = 0; ci < num_rows; ++ci) {
1386 int orig_index = index.get_data1i();
1387 nassertv(orig_index >= 0 && orig_index < (int)blend_map.size());
1388 int new_index = blend_map[orig_index];
1389 index.set_data1i(new_index);
1390 }
1391 }
1392 }
1393
1394 if (vdata->get_slider_table() != nullptr) {
1395 // The SliderTable. This one requires making a copy, like the
1396 // TransformTable (since it can't be modified once registered either), but
1397 // at least it uniquifies sliders added to it. Also, it doesn't require
1398 // indexing into it, so we don't have to build an IndexMap to modify the
1399 // vertices with.
1400 const SliderTable *old_sliders = vdata->get_slider_table();
1401 PT(SliderTable) new_sliders;
1402 if (_new_data->get_slider_table() != nullptr) {
1403 new_sliders = new SliderTable(*_new_data->get_slider_table());
1404 } else {
1405 new_sliders = new SliderTable;
1406 }
1407 int num_sliders = old_sliders->get_num_sliders();
1408 for (int si = 0; si < num_sliders; ++si) {
1409 SparseArray new_rows = old_sliders->get_slider_rows(si);
1410 new_rows <<= vertex_offset;
1411 new_sliders->add_slider(old_sliders->get_slider(si), new_rows);
1412 }
1413 _new_data->set_slider_table(SliderTable::register_table(new_sliders));
1414 }
1415}

1428
1432void GeomTransformer::VertexDataAssoc::
1433remove_unused_vertices(const GeomVertexData *vdata) {
1434 if (_geoms.empty()) {
1435 // Trivial case.
1436 return;
1437 }
1438
1439 PT(Thread) current_thread = Thread::get_current_thread();
1440
1441 BitArray referenced_vertices;
1442 bool any_referenced = false;
1443 GeomList::iterator gi;
1444 for (gi = _geoms.begin(); gi != _geoms.end(); ++gi) {
1445 Geom *geom = (*gi);
1446 if (geom->get_vertex_data() != vdata) {
1447 continue;
1448 }
1449
1450 any_referenced = true;
1451 int num_primitives = geom->get_num_primitives();
1452 for (int i = 0; i < num_primitives; ++i) {
1453 GeomPrimitivePipelineReader reader(geom->get_primitive(i), current_thread);
1454 reader.get_referenced_vertices(referenced_vertices);
1455 }
1456 }
1457
1458 if (!any_referenced) {
1459 return;
1460 }
1461
