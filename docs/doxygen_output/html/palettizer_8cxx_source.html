353void Palettizer::
354read_txa_file(std::istream &txa_file, const string &txa_filename) {
355 // Clear out the group dependencies, in preparation for reading them again
356 // from the .txa file.
357 Groups::iterator gi;
358 for (gi = _groups.begin(); gi != _groups.end(); ++gi) {
359 PaletteGroup *group = (*gi).second;
360 group->clear_depends();
361 group->set_dirname("");
362 }
363
364 // Also reset _shadow_color_type.
365 _shadow_color_type = nullptr;
366 _shadow_alpha_type = nullptr;
367
368 if (!_txa_file.read(txa_file, txa_filename)) {
369 exit(1);
370 }
371
372 if (_color_type == nullptr) {
373 nout << "No valid output image file type available; cannot run.\n"
374 << "Use :imagetype command in .txa file.\n";
375 exit(1);
376 }
377
378 // Compute the correct dependency level and order for each group. This will
379 // help us when we assign the textures to their groups.
380 for (gi = _groups.begin(); gi != _groups.end(); ++gi) {
381 PaletteGroup *group = (*gi).second;
382 group->reset_dependency_level();
383 }
384
385 for (gi = _groups.begin(); gi != _groups.end(); ++gi) {
386 PaletteGroup *group = (*gi).second;
387 group->set_dependency_level(1);
388 }
389
390 bool any_changed;
391 do {
392 any_changed = false;
393 for (gi = _groups.begin(); gi != _groups.end(); ++gi) {
394 PaletteGroup *group = (*gi).second;
395 if (group->set_dependency_order()) {
396 any_changed = true;
397 }
398 }
399 } while (any_changed);
400}

423void Palettizer::
424process_command_line_eggs(bool force_texture_read, const Filename &state_filename) {
425 _command_line_textures.clear();
426
427 // Start by scanning all the egg files we read up on the command line.
428 CommandLineEggs::const_iterator ei;
429 for (ei = _command_line_eggs.begin();
430 ei != _command_line_eggs.end();
431 ++ei) {
432 EggFile *egg_file = (*ei);
433
434 egg_file->scan_textures();
435 egg_file->get_textures(_command_line_textures);
436
437 egg_file->pre_txa_file();
438 _txa_file.match_egg(egg_file);
439 egg_file->post_txa_file();
440 }
441
442 // Now that all of our egg files are read in, build in all the cross links
443 // and back pointers and stuff.
444 EggFiles::const_iterator efi;
445 for (efi = _egg_files.begin(); efi != _egg_files.end(); ++efi) {
446 (*efi).second->build_cross_links();
447 }
448
449 // Now match each of the textures mentioned in those egg files against a
450 // line in the .txa file.
451 CommandLineTextures::iterator ti;
452 for (ti = _command_line_textures.begin();
453 ti != _command_line_textures.end();
454 ++ti) {
455 TextureImage *texture = *ti;
456
457 if (force_texture_read || texture->is_newer_than(state_filename)) {
458 // If we're forcing a redo, or the texture image has changed, re-read
459 // the complete image.
460 texture->read_source_image();
461 } else {
462 // Otherwise, just the header is sufficient.
463 texture->read_header();
464 }
465
466 texture->mark_texture_named();
467 texture->pre_txa_file();
468 _txa_file.match_texture(texture);
469 texture->post_txa_file();
470 }
471
472 // And now, assign each of the current set of textures to an appropriate
473 // group or groups.
474 for (ti = _command_line_textures.begin();
475 ti != _command_line_textures.end();
476 ++ti) {
477 TextureImage *texture = *ti;
478 texture->assign_groups();
479 }
480
481 // And then the egg files need to sign up for a particular TexturePlacement,
482 // so we can determine some more properties about how the textures are
483 // placed (for instance, how big the UV range is for a particular
484 // TexturePlacement).
485 for (efi = _egg_files.begin(); efi != _egg_files.end(); ++efi) {
486 (*efi).second->choose_placements();
487 }
488
489 // Now that *that's* done, we need to make sure the various
490 // TexturePlacements require the right size for their textures.
491 for (ti = _command_line_textures.begin();
492 ti != _command_line_textures.end();
493 ++ti) {
494 TextureImage *texture = *ti;
495 texture->determine_placement_size();
496 }
497
498 // Now that each texture has been assigned to a suitable group, make sure
499 // the textures are placed on specific PaletteImages.
500 Groups::iterator gi;
501 for (gi = _groups.begin(); gi != _groups.end(); ++gi) {
502 PaletteGroup *group = (*gi).second;
503 group->update_unknown_textures(_txa_file);
504 group->place_all();
505 }
506}

514void Palettizer::
515process_all(bool force_texture_read, const Filename &state_filename) {
516 // First, clear all the basic properties on the source texture images, so we
517 // can reapply them from the complete set of egg files and thereby ensure
518 // they are up-to-date.
519 Textures::iterator ti;
520 for (ti = _textures.begin(); ti != _textures.end(); ++ti) {
521 TextureImage *texture = (*ti).second;
522 texture->clear_source_basic_properties();
523 }
524
525 // If there *were* any egg files on the command line, deal with them.
526 CommandLineEggs::const_iterator ei;
527 for (ei = _command_line_eggs.begin();
528 ei != _command_line_eggs.end();
529 ++ei) {
530 EggFile *egg_file = (*ei);
531
532 egg_file->scan_textures();
533 egg_file->get_textures(_command_line_textures);
534 }
535
536 // Then match up all the egg files we know about with the .txa file.
537 EggFiles::const_iterator efi;
538 for (efi = _egg_files.begin(); efi != _egg_files.end(); ++efi) {
539 EggFile *egg_file = (*efi).second;
540 egg_file->pre_txa_file();
541 _txa_file.match_egg(egg_file);
542 egg_file->post_txa_file();
543 }
544
545 // Now that all of our egg files are read in, build in all the cross links
546 // and back pointers and stuff.
547 for (efi = _egg_files.begin(); efi != _egg_files.end(); ++efi) {
548 (*efi).second->build_cross_links();
549
550 // Also make sure each egg file's properties are applied to the source
551 // image (since we reset all the source image properties, above).
552 (*efi).second->apply_properties_to_source();
553 }
554
555 // Now match each of the textures in the world against a line in the .txa
556 // file.
557 for (ti = _textures.begin(); ti != _textures.end(); ++ti) {
558 TextureImage *texture = (*ti).second;
559 if (force_texture_read || texture->is_newer_than(state_filename)) {
560 texture->read_source_image();
561 }
562
563 texture->mark_texture_named();
564 texture->pre_txa_file();
565 _txa_file.match_texture(texture);
566 texture->post_txa_file();
567
568 // We need to do this to avoid bloating memory.
569 texture->release_source_image();
570 }
571
572 // And now, assign each texture to an appropriate group or groups.
573 for (ti = _textures.begin(); ti != _textures.end(); ++ti) {
574 TextureImage *texture = (*ti).second;
575 texture->assign_groups();
576 }
577
578 // And then the egg files need to sign up for a particular TexturePlacement,
579 // so we can determine some more properties about how the textures are
580 // placed (for instance, how big the UV range is for a particular
581 // TexturePlacement).
582 for (efi = _egg_files.begin(); efi != _egg_files.end(); ++efi) {
583 (*efi).second->choose_placements();
584 }
585
586 // Now that *that's* done, we need to make sure the various
587 // TexturePlacements require the right size for their textures.
588 for (ti = _textures.begin(); ti != _textures.end(); ++ti) {
589 TextureImage *texture = (*ti).second;
590 texture->determine_placement_size();
591 }
592
593 // Now that each texture has been assigned to a suitable group, make sure
594 // the textures are placed on specific PaletteImages.
595 Groups::iterator gi;
596 for (gi = _groups.begin(); gi != _groups.end(); ++gi) {
597 PaletteGroup *group = (*gi).second;
598 group->update_unknown_textures(_txa_file);
599 group->place_all();
600 }
601}

656bool Palettizer::
657read_stale_eggs(bool redo_all) {
658 bool okflag = true;
659
660 pvector<EggFiles::iterator> invalid_eggs;
661
662 EggFiles::iterator ei;
663 for (ei = _egg_files.begin(); ei != _egg_files.end(); ++ei) {
664 EggFile *egg_file = (*ei).second;
665 if (!egg_file->had_data() &&
666 (egg_file->is_stale() || redo_all)) {
667 if (!egg_file->read_egg(_noabs)) {
668 invalid_eggs.push_back(ei);
669
670 } else {
671 egg_file->scan_textures();
672 egg_file->choose_placements();
673 egg_file->release_egg_data();
674 }
675 }
676 }
677
678 // Now eliminate all the invalid egg files.
679 pvector<EggFiles::iterator>::iterator ii;
680 for (ii = invalid_eggs.begin(); ii != invalid_eggs.end(); ++ii) {
681 EggFiles::iterator ei = (*ii);
682 EggFile *egg_file = (*ei).second;
683 if (egg_file->get_source_filename().exists()) {
684 // If there is an invalid egg file, remove it; hopefully it will get
685 // rebuilt properly next time.
686 nout << "Removing invalid egg file: "
687 << FilenameUnifier::make_user_filename(egg_file->get_source_filename())
688 << "\n";
689
690 egg_file->get_source_filename().unlink();
691 okflag = false;
692
693 } else {
694 // If the egg file is simply missing, quietly remove any record of it
695 // from the database.
696 egg_file->remove_egg();
697 _egg_files.erase(ei);
698 }
699 }
700
701 if (!okflag) {
702 nout << "\n"
703 << "Some errors in egg files encountered.\n"
704 << "Re-run make install or make opt-pal to try to regenerate these.\n\n";
705 }
706
707 return okflag;
708}

839get_texture(const string &name) {
840 // Look first in the same-case name, just in case it happens to be there
841 // (from an older version of egg-palettize that did this).
842 Textures::iterator ti = _textures.find(name);
843 if (ti != _textures.end()) {
844 return (*ti).second;
845 }
846

848 // their downcase names (to implement case insensitivity).

856 image->set_name(name);
857 // image->set_filename(name);
858 _textures.insert(Textures::value_type(downcase_name, image));
859

984int Palettizer::
985complete_pointers(TypedWritable **p_list, BamReader *manager) {
986 int index = TypedWritable::complete_pointers(p_list, manager);
987
988 if (p_list[index] != nullptr) {
989 DCAST_INTO_R(_color_type, p_list[index], index);
990 }
991 index++;
992
993 if (p_list[index] != nullptr) {
994 DCAST_INTO_R(_alpha_type, p_list[index], index);
995 }
996 index++;
997
998 if (p_list[index] != nullptr) {
999 DCAST_INTO_R(_shadow_color_type, p_list[index], index);
1000 }
1001 index++;
1002
1003 if (p_list[index] != nullptr) {
1004 DCAST_INTO_R(_shadow_alpha_type, p_list[index], index);
1005 }
1006 index++;
1007
1008 int i;
1009 for (i = 0; i < _num_egg_files; i++) {
1010 EggFile *egg_file;
1011 DCAST_INTO_R(egg_file, p_list[index], index);
1012 _egg_files.insert(EggFiles::value_type(egg_file->get_name(), egg_file));
1013 index++;
1014 }
1015
1016 for (i = 0; i < _num_groups; i++) {
1017 PaletteGroup *group;
1018 DCAST_INTO_R(group, p_list[index], index);
1019 _groups.insert(Groups::value_type(group->get_name(), group));
1020 index++;
1021 }
1022
1023 for (i = 0; i < _num_textures; i++) {
1024 TextureImage *texture;
1025 DCAST_INTO_R(texture, p_list[index], index);
1026
1027 string name = downcase(texture->get_name());
1028 std::pair<Textures::iterator, bool> result = _textures.insert(Textures::value_type(name, texture));
1029 if (!result.second) {
1030 // Two textures mapped to the same slot--probably a case error (since we
1031 // just changed this rule).
1032 _texture_conflicts.push_back(texture);
1033 }
1034 index++;
1035 }
1036
1037 return index;
1038}

1045void Palettizer::
1046finalize(BamReader *manager) {
1047 // Walk through the list of texture names that were in conflict. These can
1048 // only happen if there were two different names that different only in
1049 // case, which means the textures.boo file was created before we introduced
1050 // the rule that case is insignificant.
1051 TextureConflicts::iterator ci;
1052 for (ci = _texture_conflicts.begin();
1053 ci != _texture_conflicts.end();
1054 ++ci) {
1055 TextureImage *texture_b = (*ci);
1056 string downcase_name = downcase(texture_b->get_name());
1057
1058 Textures::iterator ti = _textures.find(downcase_name);
1059 nassertv(ti != _textures.end());
1060 TextureImage *texture_a = (*ti).second;
1061 _textures.erase(ti);
1062
1063 if (!texture_b->is_used() || !texture_a->is_used()) {
1064 // If either texture is not used, there's not really a conflict--the
1065 // other one wins.
1066 if (texture_a->is_used()) {
1067 bool inserted1 = _textures.insert(Textures::value_type(downcase_name, texture_a)).second;
1068 nassertd(inserted1) { }
1069
1070 } else if (texture_b->is_used()) {
1071 bool inserted2 = _textures.insert(Textures::value_type(downcase_name, texture_b)).second;
1072 nassertd(inserted2) { }
1073 }
1074
1075 } else {
1076 // If both textures are used, there *is* a conflict.
1077 nout << "Texture name conflict: \"" << texture_a->get_name()
1078 << "\" vs. \"" << texture_b->get_name() << "\"\n";
1079 if (texture_a->get_name() != downcase_name &&
1080 texture_b->get_name() != downcase_name) {
1081 // Arbitrarily pick texture_a to get the right case.
1082 bool inserted1 = _textures.insert(Textures::value_type(downcase_name, texture_a)).second;
1083 bool inserted2 = _textures.insert(Textures::value_type(texture_b->get_name(), texture_b)).second;
1084 nassertd(inserted1 && inserted2) { }
1085
1086 } else {
1087 // One of them is already the right case.
1088 bool inserted1 = _textures.insert(Textures::value_type(texture_a->get_name(), texture_a)).second;
1089 bool inserted2 = _textures.insert(Textures::value_type(texture_b->get_name(), texture_b)).second;
1090 nassertd(inserted1 && inserted2) { }
1091 }
1092 }
1093 }
1094}
