151istream &
152operator >> (istream &in, FmodSpeakerMode &sm) {
153 string word;
154 in >> word;
155
156 if (word.size() == 0) {
157 sm = FSM_unspecified;
158 } else if (cmp_nocase(word, "raw") == 0) {
159 sm = FSM_raw;
160 } else if (cmp_nocase(word, "mono") == 0) {
161 sm = FSM_mono;
162 } else if (cmp_nocase(word, "stereo") == 0) {
163 sm = FSM_stereo;
164 } else if (cmp_nocase(word, "quad") == 0) {
165 sm = FSM_quad;
166 } else if (cmp_nocase(word, "surround") == 0) {
167 sm = FSM_surround;
168 } else if (cmp_nocase(word, "5point1") == 0 ||
169 cmp_nocase(word, "5.1") == 0) {
170 sm = FSM_5point1;
171 } else if (cmp_nocase(word, "7point1") == 0 ||
172 cmp_nocase(word, "7.1") == 0) {
173 sm = FSM_7point1;
174
175 } else {
176 audio_cat->error() << "Invalid FmodSpeakerMode value: " << word << "\n";
177 sm = FSM_unspecified;
178 }
179
180 return in;
