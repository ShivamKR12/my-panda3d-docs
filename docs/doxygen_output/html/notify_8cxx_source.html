184NotifyCategory *Notify::
185get_category(const string &basename, NotifyCategory *parent_category) {
186 // The string should not contain colons.
187 nassertr(basename.find(':') == string::npos, nullptr);
188
189 string fullname;
190 if (parent_category != nullptr) {
191 fullname = parent_category->get_fullname() + ":" + basename;
192 } else {
193 // The parent_category is NULL. If basename is empty, that means we refer
194 // to the very top-level category (with an empty fullname); otherwise,
195 // it's a new category just below that top level.
196 if (!basename.empty()) {
197 parent_category = get_top_category();
198 fullname = ":" + basename;
199 }
200 }
201
202 std::pair<Categories::iterator, bool> result =
203 _categories.insert(Categories::value_type(fullname, nullptr));
204
205 bool inserted = result.second;
206 NotifyCategory *&category = (*result.first).second;
207
208 if (inserted) {
209 // If we just inserted a new record, then we have to create a new Category
210 // pointer. Otherwise, there was already one created from before.
211 category = new NotifyCategory(fullname, basename, parent_category);
212 }
213
214 return category;
215}

235NotifyCategory *Notify::
236get_category(const string &fullname) {
237 Categories::const_iterator ci;
238 ci = _categories.find(fullname);
239 if (ci != _categories.end()) {
240 return (*ci).second;
241 }
242
243 // No such Category; create one. First identify the parent name, based on
244 // the rightmost colon.
245 NotifyCategory *parent_category = nullptr;
246 string basename = fullname;
247
248 size_t colon = fullname.rfind(':');
249 if (colon != string::npos) {
250 parent_category = get_category(fullname.substr(0, colon));
251 basename = fullname.substr(colon + 1);
252
253 } else if (!fullname.empty()) {
254 // The fullname didn't begin with a colon. Infer one.
255 parent_category = get_top_category();
256 }
257
258 return get_category(basename, parent_category);
259}

343bool Notify::
344assert_failure(const char *expression, int line,
345 const char *source_file) {
346 ostringstream message_str;
347 message_str
348 << expression << " at line " << line << " of " << source_file;
349 string message = message_str.str();
350
351 if (!_assert_failed) {
352 // We only save the first assertion failure message, as this is usually
353 // the most meaningful when several occur in a row.
354 _assert_failed = true;
355 _assert_error_message = message;
356 }
357
358 if (has_assert_handler()) {
359 return (*_assert_handler)(expression, line, source_file);
360 }
361
362#ifdef ANDROID
363 // Write to Android log system.
364 __android_log_assert("assert", "Panda3D", "Assertion failed: %s", message.c_str());
365#endif
366
367#ifdef __EMSCRIPTEN__
368 // Write to JavaScript console.
369 emscripten_log(EM_LOG_CONSOLE | EM_LOG_ERROR | EM_LOG_C_STACK,
370 "Assertion failed: %s", message.c_str());
371
372#else
373 nout << "Assertion failed: " << message << "\n";
374#endif
375
376 // This is redefined here, shadowing the defining in config_prc.h, so we can
377 // guarantee it has already been constructed.
378 ALIGN_16BYTE ConfigVariableBool assert_abort("assert-abort", false);
379 if (assert_abort) {
380 // Make sure the error message has been flushed to the output.
381 nout.flush();
382
383#ifdef _MSC_VER
384 // How to trigger an exception in VC++ that offers to take us into the
385 // debugger? abort() doesn't do it. We used to be able to assert(false),
386 // but in VC++ 7 that just throws an exception, and an uncaught exception
387 // just exits, without offering to open the debugger.
388
389 // DebugBreak() seems to be provided for this purpose, but it doesn't seem
390 // to work properly either, since we don't seem to get a reliable stack
391 // trace.
392
393 // The old reliable int 3 works (at least on an Intel platform) if you are
394 // already running within a debugger. But it doesn't offer to bring up a
395 // debugger otherwise.
396
397 // So we'll force a segfault, which works every time.
398 int *ptr = nullptr;
399 *ptr = 1;
400
401#elif defined(__EMSCRIPTEN__)
402 // This should drop us into the browser's JavaScript debugger.
403 //emscripten_debugger();
404 EM_ASM(debugger;);
405
406#else // _MSC_VER
407 abort();
408#endif // _MSC_VER
409 }
410
411 return true;
412}

419NotifySeverity Notify::
420string_severity(const string &str) {
421 // Convert the string to lowercase for a case-insensitive comparison.
422 string lstring;
423 for (string::const_iterator si = str.begin();
424 si != str.end();
425 ++si) {
426 lstring += (char)tolower(*si);
427 }
428
429 if (lstring == "spam") {
430 return NS_spam;
431
432 } else if (lstring == "debug") {
433 return NS_debug;
434
435 } else if (lstring == "info") {
436 return NS_info;
437
438 } else if (lstring == "warning") {
439 return NS_warning;
440
441 } else if (lstring == "error") {

443
444 } else if (lstring == "fatal") {

446

457void Notify::
458config_initialized() {
459 // We allow this to be called more than once to allow the user to specify a
460 // notify-output even after the initial import of Panda3D modules. However,
461 // it cannot be changed after the first time it is set.
462
463#if defined(ANDROID)
464 // Android redirects stdio and stderr to /dev/null,
465 // but does provide its own logging system. We use a special
466 // type of stream that redirects it to Android's log system.
467
468 Notify *ptr = Notify::ptr();
469
470 for (int severity = 0; severity <= NS_fatal; ++severity) {
471 int priority = ANDROID_LOG_UNKNOWN;
472 if (severity != NS_unspecified) {
473 priority = severity + 1;
474 }
475 ptr->_log_streams[severity] = new AndroidLogStream(priority);
476 }
477
478#elif defined(__EMSCRIPTEN__)
479 // We have no writable filesystem in JavaScript. Instead, we set up a
480 // special stream that logs straight into the Javascript console.
481
482 EmscriptenLogStream *error_stream = new EmscriptenLogStream(EM_LOG_CONSOLE | EM_LOG_ERROR);
483 EmscriptenLogStream *warn_stream = new EmscriptenLogStream(EM_LOG_CONSOLE | EM_LOG_WARN);
484 EmscriptenLogStream *info_stream = new EmscriptenLogStream(EM_LOG_CONSOLE);
485
486 Notify *ptr = Notify::ptr();
487 ptr->_log_streams[NS_unspecified] = info_stream;
488 ptr->_log_streams[NS_spam] = info_stream;
489 ptr->_log_streams[NS_debug] = info_stream;
490 ptr->_log_streams[NS_info] = info_stream;
491 ptr->_log_streams[NS_warning] = warn_stream;
492 ptr->_log_streams[NS_error] = error_stream;
493 ptr->_log_streams[NS_fatal] = error_stream;
494
495#else
496 if (_global_ptr == nullptr || _global_ptr->_ostream_ptr == &cerr) {
497 static ConfigVariableFilename notify_output
498 ("notify-output", "",
499 "The filename to which to write all the output of notify");
500
501 // We use this to ensure that only one thread can initialize the output.
502 static patomic_flag initialized = ATOMIC_FLAG_INIT;
503
504 std::string value = notify_output.get_value();
505 if (!value.empty() && !initialized.test_and_set()) {
506 Notify *ptr = Notify::ptr();
507
508 if (value == "stdout") {
509 cout.setf(std::ios::unitbuf);
510 ptr->set_ostream_ptr(&cout, false);
511
512 } else if (value == "stderr") {
513 ptr->set_ostream_ptr(&cerr, false);
514
515 } else {
516 Filename filename = value;
517 filename.set_text();
518#ifdef BUILD_IPHONE
519 // On the iPhone, route everything through cerr, and then send cerr to
520 // the log file, since we can't get the cerr output otherwise.
521 string os_specific = filename.to_os_specific();
522 int logfile_fd = open(os_specific.c_str(), O_WRONLY | O_CREAT | O_TRUNC, 0666);
523 if (logfile_fd < 0) {
524 nout << "Unable to open file " << filename << " for output.\n";
525 } else {
526 dup2(logfile_fd, STDOUT_FILENO);
527 dup2(logfile_fd, STDERR_FILENO);
528 close(logfile_fd);
529
530 ptr->set_ostream_ptr(&cerr, false);
531 }
532#else
533 pofstream *out = new pofstream;
534 if (!filename.open_write(*out)) {
535 nout << "Unable to open file " << filename << " for output.\n";
536 delete out;
537 } else {
538 out->setf(std::ios::unitbuf);
539 ptr->set_ostream_ptr(out, true);
540 }
541#endif // BUILD_IPHONE
542 }
543#ifdef ANDROID
544 } else {
545 // By default, we always redirect the notify stream to the Android log.
546 Notify *ptr = Notify::ptr();
547 ptr->set_ostream_ptr(new AndroidLogStream(ANDROID_LOG_INFO), true);
548#endif
549 }
550 }
551#endif
552}
