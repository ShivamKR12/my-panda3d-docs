172void AsyncTask::
173set_name(const string &name) {
174 if (_manager != nullptr) {
175 MutexHolder holder(_manager->_lock);
176 if (Namable::get_name() != name) {
177 // Changing an active task's name requires moving it around on its name
178 // index.
179
180 _manager->remove_task_by_name(this);
181 Namable::set_name(name);
182 _manager->add_task_by_name(this);
183 }
184 } else {
185 // If it hasn't been started anywhere, we can just change the name.
186 Namable::set_name(name);
187 }
188
189#ifdef DO_PSTATS
190 // Update the PStatCollector with the new name. If the name includes a
191 // colon, we stop the collector name there, and don't go further.
192 size_t end = name.size();
193 size_t colon = name.find(':');
194 if (colon != string::npos) {
195 end = std::min(end, colon);
196 }
197
198 // If the name ends with a hyphen followed by a string of digits, we strip
199 // all that off, for the parent collector, to group related tasks together
200 // in the pstats graph. We still create a child collector that contains the
201 // full name, however.
202 size_t trimmed = end;
203 size_t p = trimmed;
204 while (true) {
205 while (p > 0 && isdigit(name[p - 1])) {
206 --p;
207 }
208 if (p > 0 && (name[p - 1] == '-' || name[p - 1] == '_')) {
209 --p;
210 trimmed = p;
211 } else {
212 //p = trimmed;
213 break;
214 }
215 }
216 PStatCollector parent(_tasks_pcollector, name.substr(0, trimmed));
217 // prevent memory leak _task_pcollector = PStatCollector(parent,
218 // name.substr(0, end));
219 _task_pcollector = parent;
220#endif // DO_PSTATS
221}

254 if (chain_name != _chain_name) {
255 if (_manager != nullptr) {
256 MutexHolder holder(_manager->_lock);
257 if (_state == S_active) {
258 // Changing chains on an "active" (i.e. enqueued) task means removing
259 // it and re-inserting it into the queue.
260 PT(AsyncTask) hold_task = this;
261 PT(AsyncTaskManager) manager = _manager;
262
263 AsyncTaskChain *chain_a = manager->do_find_task_chain(_chain_name);
264 nassertv(chain_a != nullptr);
265 chain_a->do_remove(this);
266 _chain_name = chain_name;
267
268 jump_to_task_chain(manager);
269
270 } else {
271 // If it's sleeping, currently being serviced, or something else, we

400AsyncTask::DoneStatus AsyncTask::
401unlock_and_do_task() {
402 nassertr(_manager != nullptr, DS_done);
403 PT(ClockObject) clock = _manager->get_clock();
404
405 // Indicate that this task is now the current task running on the thread.
406 Thread *current_thread = Thread::get_current_thread();
407 nassertr(current_thread->_current_task == nullptr, DS_interrupt);
408
409#ifdef __GNUC__
410 __attribute__((unused))
411#endif
412 void *ptr = AtomicAdjust::compare_and_exchange_ptr
413 (current_thread->_current_task, nullptr, (TypedReferenceCount *)this);
414
415 // If the return value is other than nullptr, someone else must have
416 // assigned the task first, in another thread. That shouldn't be possible.
417
418 // But different versions of gcc appear to have problems compiling these
419 // assertions correctly.
420#ifndef __GNUC__
421 nassertr(ptr == nullptr, DS_interrupt);
422 nassertr(current_thread->_current_task == this, DS_interrupt);
423#endif // __GNUC__
424
425 // It's important to release the lock while the task is being serviced.
426 _manager->_lock.unlock();
427
428 double start = clock->get_real_time();
429 _task_pcollector.start();
430 DoneStatus status = do_task();
431 _task_pcollector.stop();
432 double end = clock->get_real_time();
433
434 // Now reacquire the lock (so we can return with the lock held).
435 _manager->_lock.lock();
436
437 _dt = end - start;
438 _max_dt = std::max(_dt, _max_dt);
439 _total_dt += _dt;
440
441 _chain->_time_in_frame += _dt;
442
443 // Now indicate that this is no longer the current task.
444 nassertr(current_thread->_current_task == this, status);
445
446 ptr = AtomicAdjust::compare_and_exchange_ptr
447 (current_thread->_current_task, (TypedReferenceCount *)this, nullptr);
448
449 // If the return value is other than this, someone else must have assigned
450 // the task first, in another thread. That shouldn't be possible.
451
452 // But different versions of gcc appear to have problems compiling these
453 // assertions correctly.
454#ifndef __GNUC__
455 nassertr(ptr == this, DS_interrupt);
456 nassertr(current_thread->_current_task == nullptr, DS_interrupt);
457#endif // __GNUC__
458
459 return status;
460}
