77const PStatFrameData &PStatThreadData::
78get_frame(int frame_number) const {
79 int rel_frame = frame_number - _first_frame_number;
80 int num_frames = _frames.size();
81 if (rel_frame >= num_frames) {
82 rel_frame = num_frames - 1;
83 }
84
85 while (rel_frame >= 0 && _frames[rel_frame] == nullptr) {
86 rel_frame--;
87 }
88 if (rel_frame < 0) {
89 // No frame data that old. Return the oldest frame we've got.
90 rel_frame = 0;
91 while (rel_frame < num_frames &&
92 _frames[rel_frame] == nullptr) {
93 rel_frame++;
94 }
95 }
96
97 if (rel_frame >= 0 && rel_frame < num_frames) {
98 PStatFrameData *frame = _frames[rel_frame];
99 nassertr(frame != nullptr, _null_frame);
100 nassertr(frame->get_start() >= 0.0, _null_frame);
101 return *frame;
102 }
103
104 nassertr(_null_frame.get_start() >= 0.0, _null_frame);
105 return _null_frame;
106}

145int PStatThreadData::
146get_frame_number_at_time(double time, int hint) const {
147 hint -= _first_frame_number;
148 if (hint >= 0 && hint < (int)_frames.size()) {
149 if (_frames[hint] != nullptr &&
150 _frames[hint]->get_start() <= time) {
151 // The hint might be right. Scan forward from there.
152 int i = hint + 1;
153 while (i < (int)_frames.size() &&
154 (_frames[i] == nullptr ||
155 _frames[i]->get_start() <= time)) {
156 if (_frames[i] != nullptr) {
157 hint = i;
158 }
159 ++i;
160 }
161 return _first_frame_number + hint;
162 }
163 }
164
165 // The hint is wrong, we have to search the entire set.
166 int first_i = 0;
167 int last_i = _frames.size() - 1;
168 while (first_i < last_i && _frames[first_i] == nullptr) {
169 ++first_i;
170 }
171 while (first_i < last_i && _frames[last_i] == nullptr) {
172 --last_i;
173 }
174
175 if (first_i >= last_i) {
176 // There are no frames.
177 return _first_frame_number - 1;
178 }
179
180 // Take a guess by interpolating based on the first and last frame times.
181 double first = _frames[first_i]->get_start();
182 if (time < first) {
183 return _first_frame_number - 1;
184 }
185
186 double last = _frames[last_i]->get_start();
187 double t = (time - first) / (last - first);
188 hint = std::max(0, (int)(t * (last_i - first_i))) + first_i;
189
190 // Find a frame around the guess that has data.
191 if (_frames[hint] == nullptr) {
192 if (hint <= first_i) {
193 hint = first_i;
194 }
195 else do {
196 // Skip backward until we find a frame with data.
197 --hint;
198 } while (_frames[hint] == nullptr);
199 }
200
201 if (_frames[hint]->get_start() <= time) {
202 // Search forward.
203 for (int i = hint + 1; i < (int)_frames.size(); ++i) {
204 const PStatFrameData *frame = _frames[i];
205 if (frame != nullptr) {
206 if (frame->get_start() <= time) {
207 hint = i;
208 } else {
209 break;
210 }
211 }
212 }
213 return _first_frame_number + hint;
214 }
215
216 // Search backward.
217 for (int i = hint - 1; i >= 0; --i) {
218 const PStatFrameData *frame = _frames[i];
219 if (frame != nullptr && frame->get_start() <= time) {
220 return _first_frame_number + i;
221 }
222 }
223
224 return _first_frame_number - 1;
225}

318bool PStatThreadData::
319prune_history(double time) {
320 double oldest_allowable_time = time - _history;
321 while (!_frames.empty() &&
322 (_frames.front() == nullptr ||
323 _frames.front()->is_time_empty() ||
324 _frames.front()->get_start() < oldest_allowable_time)) {
325 delete _frames.front();

341void PStatThreadData::
342record_new_frame(int frame_number, PStatFrameData *frame_data) {
343 nassertv(frame_data != nullptr);
344 nassertv(!frame_data->is_empty());
345
346 // First, remove all the old frames that fall outside of our history window.
347 // Then, add enough empty frame definitions to account for the latest frame
348 // number. This might involve some skips, since we don't guarantee that we
349 // get all the frames in order or even at all.
350 if (prune_history(frame_data->get_start())) {
351 _first_frame_number = frame_number;
352 _frames.push_back(nullptr);
353 }
354 else {
355 while (_first_frame_number + (int)_frames.size() <= frame_number) {
356 _frames.push_back(nullptr);
357 }
358 }
359
360 int index = frame_number - _first_frame_number;
361
362 // It's possible to receive frames out of order.
363 while (index < 0) {
364 _frames.push_front(nullptr);
365 ++index;
366 --_first_frame_number;
367 }
368
369 nassertv(index >= 0 && index < (int)_frames.size());
370
371 if (_frames[index] != nullptr) {
372 nout << "Got repeated frame data for frame " << frame_number << "\n";
373 delete _frames[index];
374 }
375
376 _frames[index] = frame_data;
377 _computed_elapsed_frames = false;
378}

416void PStatThreadData::
417compute_elapsed_frames() const {
418 if (_frames.empty()) {
419 // No frames in the data at all.
420 _got_elapsed_frames = false;
421 }
422 else {
423 _now_i = _frames.size() - 1;
424 while (_now_i > 0 && _frames[_now_i] == nullptr) {
425 _now_i--;
426 }
427 if (_now_i < 0) {
428 // No frames have any real data.
429 _got_elapsed_frames = false;
430 }
431 else {
432 nassertv(_frames[_now_i] != nullptr);
433
434 double now = _frames[_now_i]->get_end();
435 double then = now - pstats_average_time;
436
437 int old_i = _now_i;
438 _then_i = _now_i;
439
440 while (old_i >= 0) {
441 const PStatFrameData *frame = _frames[old_i];
442 if (frame != nullptr) {
443 if (frame->get_start() > then) {
444 _then_i = old_i;
445 } else {
446 break;
447 }
448 }
449 old_i--;
450 }
451
452 nassertv(_then_i >= 0);
453 nassertv(_frames[_then_i] != nullptr);
454 _got_elapsed_frames = true;
455
456 _now_i += _first_frame_number;
457 _then_i += _first_frame_number;
458 }
459 }
460
461 _computed_elapsed_frames = true;
