120bool TinyGraphicsBuffer::
121open_buffer() {
122 // GSG CreationInitialization
123 TinyGraphicsStateGuardian *tinygsg;
124 if (_gsg == 0) {
125 // There is no old gsg. Create a new one.
126 tinygsg = new TinyGraphicsStateGuardian(_engine, _pipe, nullptr);
127 _gsg = tinygsg;
128 } else {
129 DCAST_INTO_R(tinygsg, _gsg, false);
130 }
131
132 create_frame_buffer();
133 if (_frame_buffer == nullptr) {
134 tinydisplay_cat.error()
135 << "Could not create frame buffer.\n";
136 return false;
137 }
138
139 _fb_properties.set_depth_bits(ZB_Z_BITS);
140
141 tinygsg->_current_frame_buffer = _frame_buffer;
142
143 tinygsg->reset_if_new();
144 if (!tinygsg->is_valid()) {
145 close_buffer();
146 return false;
147 }
148
149 _is_valid = true;
150 return true;
