68void DaeCharacter::
69bind_joints(JointMap &joint_map) {
70 _joints.clear();
71
72 size_t num_joints = _skin_controller->GetJointCount();
73 _joints.reserve(num_joints);
74
75 // Record the bind pose for each joint.
76 for (size_t j = 0; j < num_joints; ++j) {
77 const FCDSkinControllerJoint *skin_joint = _skin_controller->GetJoint(j);
78 std::string sid = FROM_FSTRING(skin_joint->GetId());
79 LMatrix4d bind_pose;
80 bind_pose.invert_from(DAEToEggConverter::convert_matrix(
81 skin_joint->GetBindPoseInverse()));
82
83 // Check that we already encountered this joint during traversal.
84 JointMap::iterator ji = joint_map.find(sid);
85 if (ji != joint_map.end()) {
86 Joint &joint = ji->second;
87
88 if (joint._character != nullptr) {
89 // In some cases, though, multiple controllers share the same joints.
90 // We can't support this without duplicating the joint structure, so
91 // we check if the bind poses are the same.
92 if (!joint._bind_pose.almost_equal(bind_pose, 0.0001)) {
93 // Ugh. What else could we do?
94 daeegg_cat.error()
95 << "Multiple controllers share joint with sid " << sid
96 << ", with different bind poses.\n";
97 }
98 } else {
99 // Mark the joint as being controlled by this character.
100 joint._bind_pose = bind_pose;
101 joint._character = this;
102 }
103
104 _joints.push_back(joint);
105 } else {
106 daeegg_cat.warning()
107 << "Unknown joint sid being referenced: '" << sid << "'\n";
108
109 // We still have to add a dummy joint or the index will be off.
110 _joints.push_back(Joint(nullptr, nullptr));
111 }
112 }

174 const FCDSkinControllerVertex *influence = _skin_controller->GetVertexInfluence(index);
175
176 for (size_t pa = 0; pa < influence->GetPairCount(); ++pa) {

213 FCDAnimatedList animateds;
214
215 // Collect all the animation curves
216 for (size_t t = 0; t < node->GetTransformCount(); ++t) {
217 FCDTransform *transform = node->GetTransform(t);
218 FCDAnimated *animated = transform->GetAnimated();
219
220 if (animated != nullptr) {
221 const FCDAnimationCurveListList &all_curves = animated->GetCurves();
222
223 for (size_t ci = 0; ci < all_curves.size(); ++ci) {
224 const FCDAnimationCurveTrackList &curves = all_curves[ci];
225 if (curves.empty()) {
