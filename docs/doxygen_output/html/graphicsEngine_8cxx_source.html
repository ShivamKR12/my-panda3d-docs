261GraphicsOutput *GraphicsEngine::
262make_output(GraphicsPipe *pipe,
263 const string &name, int sort,
264 const FrameBufferProperties &fb_prop,
265 const WindowProperties &win_prop,
266 int flags,
267 GraphicsStateGuardian *gsg,
268 GraphicsOutput *host) {
269
270/*
271 * The code here is tricky because the gsg that is passed in might be in the
272 * uninitialized state. As a result, pipe::make_output may not be able to
273 * tell which DirectX capabilities or OpenGL extensions are supported and
274 * which are not. Worse yet, it can't query the API, because that can only be
275 * done from the draw thread, and this is the app thread. So here's the
276 * workaround: this routine calls pipe::make_output, which returns a "non-
277 * certified" window. That means that the pipe doesn't promise that the draw
278 * thread will actually succeed in initializing the window. This routine then
279 * calls open_windows, which attempts to initialize the window. If
280 * open_windows fails to initialize the window, then this routine will ask
281 * pipe::make_output to try again, this time using a different set of OpenGL
282 * extensions or DirectX capabilities. This is what the "retry" parameter to
283 * pipe::make_output is for - it specifies, in an abstract manner, which set
284 * of capabiltiesextensions to try. The only problem with this design is that
285 * it requires the engine to call open_windows, which is slow. To make things
286 * faster, the pipe can choose to "precertify" its creations. If it chooses
287 * to do so, this is a guarantee that the windows it returns will not fail in
288 * open_windows. However, most graphics pipes will only precertify if you
289 * pass them an already-initialized gsg. Long story short, if you want
290 * make_output to be fast, use an already-initialized gsg.
291 */
292
293 // Simplify the input parameters.
294
295 int x_size = 0, y_size = 0;
296 if (win_prop.has_size()) {
297 x_size = win_prop.get_x_size();
298 y_size = win_prop.get_y_size();
299 }
300 if ((x_size == 0)&&(y_size == 0)) {
301 flags |= GraphicsPipe::BF_size_track_host;
302 }
303 if (host != nullptr) {
304 host = host->get_host();
305 }
306
307 // If a gsg or host was supplied, and either is not yet initialized, then
308 // call open_windows to get both ready. If that fails, give up on using the
309 // supplied gsg and host.
310
311 if (host == nullptr) {
312 if (gsg != nullptr) {
313 if ((!gsg->is_valid())||(gsg->needs_reset())) {
314 open_windows();
315 }
316 if ((!gsg->is_valid())||(gsg->needs_reset())) {
317 gsg = nullptr;
318 }
319 }
320 } else {
321 if ((host->get_gsg()==nullptr)||
322 (!host->is_valid())||
323 (!host->get_gsg()->is_valid())||
324 (host->get_gsg()->needs_reset())) {
325 open_windows();
326 }
327 if ((host->get_gsg()==nullptr)||
328 (!host->is_valid())||
329 (!host->get_gsg()->is_valid())||
330 (host->get_gsg()->needs_reset())) {
331 host = nullptr;
332 gsg = nullptr;
333 } else {
334 gsg = host->get_gsg();
335 }
336 }
337
338 // Sanity check everything.
339
340 nassertr(pipe != nullptr, nullptr);
341 if (gsg != nullptr) {
342 nassertr(pipe == gsg->get_pipe(), nullptr);
343 //nassertr(this == gsg->get_engine(), nullptr);
344 }
345
346 // Are we really asking for a callback window?
347 if ((flags & GraphicsPipe::BF_require_callback_window)!=0) {
348 PT(GraphicsStateGuardian) this_gsg = gsg;
349 if (this_gsg == nullptr) {
350 // If we don't already have a GSG, we have to ask the pipe to make a new
351 // one, unencumbered by window dressing.
352 this_gsg = pipe->make_callback_gsg(this);
353 }
354 if (this_gsg != nullptr) {
355 CallbackGraphicsWindow *window = new CallbackGraphicsWindow(this, pipe, name, fb_prop, win_prop, flags, this_gsg);
356 window->_sort = sort;
357 do_add_window(window);
358 do_add_gsg(window->get_gsg(), pipe);
359 display_cat.info() << "Created output of type CallbackGraphicsWindow\n";
360 return window;
361 }
362
363 // Couldn't make a callback window, because the pipe wouldn't make an
364 // unencumbered GSG.
365 return nullptr;
366 }
367
368 // Determine if a parasite buffer meets the user's specs.
369
370 bool can_use_parasite = false;
371 if ((host != nullptr)&&
372 ((flags&GraphicsPipe::BF_require_window)==0)&&
373 ((flags&GraphicsPipe::BF_require_callback_window)==0)&&
374 ((flags&GraphicsPipe::BF_refuse_parasite)==0)&&
375 ((flags&GraphicsPipe::BF_can_bind_color)==0)&&
376 ((flags&GraphicsPipe::BF_can_bind_every)==0)&&
377 ((flags&GraphicsPipe::BF_rtt_cumulative)==0)&&
378 ((flags&GraphicsPipe::BF_can_bind_layered)==0)) {
379 if ((flags&GraphicsPipe::BF_fb_props_optional) ||
380 (host->get_fb_properties().subsumes(fb_prop))) {
381 can_use_parasite = true;
382 }
383 }
384
385 // If parasite buffers are preferred, then try a parasite first. Even if
386 // prefer-parasite-buffer is set, parasites are not preferred if the host
387 // window is too small, or if the host window does not have the requested
388 // properties.
389
390 if ((prefer_parasite_buffer) &&
391 (can_use_parasite) &&
392 (x_size <= host->get_x_size())&&
393 (y_size <= host->get_y_size())&&
394 (host->get_fb_properties().subsumes(fb_prop))) {
395 ParasiteBuffer *buffer = new ParasiteBuffer(host, name, x_size, y_size, flags);
396 buffer->_sort = sort;
397 do_add_window(buffer);
398 do_add_gsg(host->get_gsg(), pipe);
399 display_cat.info() << "Created output of type ParasiteBuffer\n";
400 return buffer;
401 }
402
403 // If force-parasite-buffer is set, we create a parasite buffer even if it's
404 // less than ideal. You might set this if you really don't trust your
405 // graphics driver's support for offscreen buffers.
406 if (force_parasite_buffer && can_use_parasite) {
407 ParasiteBuffer *buffer = new ParasiteBuffer(host, name, x_size, y_size, flags);
408 buffer->_sort = sort;
409 do_add_window(buffer);
410 do_add_gsg(host->get_gsg(), pipe);
411 display_cat.info() << "Created output of type ParasiteBuffer\n";
412 return buffer;
413 }
414
415 // Ask the pipe to create a window.
416
417 for (int retry=0; retry<10; retry++) {
418 bool precertify = false;
419 PT(GraphicsOutput) window =
420 pipe->make_output(name, fb_prop, win_prop, flags, this, gsg, host, retry, precertify);
421 if (window != nullptr) {
422 window->_sort = sort;
423 if (precertify && gsg != nullptr && window->get_gsg() == gsg) {
424 do_add_window(window);
425 display_cat.info()
426 << "Created output of type " << window->get_type() << "\n";
427 return window;
428 }
429 do_add_window(window);
430 open_windows();
431 if (window->is_valid()) {
432 display_cat.info()
433 << "Created output of type " << window->get_type() << "\n";
434
435 if (window->get_fb_properties().subsumes(fb_prop)) {
436 return window;
437 } else {
438 if (flags & GraphicsPipe::BF_fb_props_optional) {
439 display_cat.warning()
440 << "FrameBufferProperties available less than requested.\n";
441 display_cat.warning(false)
442 << " requested: " << fb_prop << "\n"
443 << " got: " << window->get_fb_properties() << "\n";
444 return window;
445 }
446 display_cat.error()
447 << "Could not get requested FrameBufferProperties; abandoning window.\n";
448 display_cat.error(false)
449 << " requested: " << fb_prop << "\n"
450 << " got: " << window->get_fb_properties() << "\n";
451 }
452 } else {
453 display_cat.info()
454 << window->get_type() << " wouldn't open; abandoning.\n";
455 display_cat.debug(false)
456 << " requested: " << fb_prop << "\n";
457 }
458
459 // No good; delete the window and keep trying.
460 remove_window(window);
461 }
462 }
463
464 // Parasite buffers were not preferred, but the pipe could not create a
465 // window to the user's specs. Try a parasite as a last hope.
466
467 if (can_use_parasite) {
468 ParasiteBuffer *buffer = new ParasiteBuffer(host, name, x_size, y_size, flags);
469 buffer->_sort = sort;
470 do_add_window(buffer);
471 do_add_gsg(host->get_gsg(), pipe);
472 display_cat.info() << "Created output of type ParasiteBuffer\n";
473 return buffer;
474 }
475
476 // Could not create a window to the user's specs.
477
478 return nullptr;
479}

524bool GraphicsEngine::
525remove_window(GraphicsOutput *window) {
526 nassertr(window != nullptr, false);
527 Thread *current_thread = Thread::get_current_thread();
528
529 // First, make sure we know what this window is.
530 PT(GraphicsOutput) ptwin = window;
531 size_t count;
532 {
533 ReMutexHolder holder(_lock, current_thread);
534 if (!_windows_sorted) {
535 do_resort_windows();
536 }
537 count = _windows.erase(ptwin);
538 }
539
540 // Also check whether it is in _new_windows.
541 {
542 MutexHolder new_windows_holder(_new_windows_lock, current_thread);
543 size_t old_size = _new_windows.size();
544 _new_windows.erase(std::remove(_new_windows.begin(), _new_windows.end(), ptwin), _new_windows.end());
545 if (count == 0 && _new_windows.size() < old_size) {
546 count = 1;
547 }
548 }
549
550 if (count == 0) {
551 // Never heard of this window. Do nothing.
552 return false;
553 }
554
555 do_remove_window(window, current_thread);
556
557 GraphicsStateGuardian *gsg = window->get_gsg();
558 if (gsg != nullptr) {
559 PreparedGraphicsObjects *pgo = gsg->get_prepared_objects();
560 if (pgo != nullptr) {
561 // Check to see if any other still-active windows share this context.
562 bool any_common = false;
563 {
564 ReMutexHolder holder(_lock, current_thread);
565 Windows::iterator wi;
566 for (wi = _windows.begin(); wi != _windows.end() && !any_common; ++wi) {
567 GraphicsStateGuardian *gsg2 = (*wi)->get_gsg();
568 if (gsg2 != nullptr &&
569 gsg2->get_prepared_objects() == pgo) {
570 any_common = true;
571 }
572 }
573 }
574 if (!any_common) {
575 // If no windows still use this context, release all textures, etc.
576 // We do this in case there is a floating pointer somewhere keeping
577 // the GSG from destructing when its window goes away. A leaked GSG
578 // pointer is bad enough, but there's no reason we also need to keep
579 // around all of the objects allocated on graphics memory.
580 pgo->release_all();
581 }
582 }
583 }
584
585 nassertr(count == 1, true);
586 return true;
587}

593void GraphicsEngine::
594remove_all_windows() {
595 Thread *current_thread = Thread::get_current_thread();
596
597 ReMutexHolder holder(_lock, current_thread);
598
599 // Let's move the _windows vector into a local copy first, and walk through
600 // that local copy, just in case someone we call during the loop attempts to
601 // modify _windows. I don't know what code would be doing this, but it
602 // appeared to be happening, and this worked around it.
603 Windows old_windows;
604 old_windows.swap(_windows);
605 for (GraphicsOutput *win : old_windows) {
606 nassertv(win != nullptr);
607 do_remove_window(win, current_thread);
608 GraphicsStateGuardian *gsg = win->get_gsg();
609 if (gsg != nullptr) {
610 gsg->release_all();
611 }
612 }
613
614 {
615 MutexHolder new_windows_holder(_new_windows_lock, current_thread);
616 for (GraphicsOutput *win : _new_windows) {
617 nassertv(win != nullptr);
618 do_remove_window(win, current_thread);
619 GraphicsStateGuardian *gsg = win->get_gsg();
620 if (gsg != nullptr) {
621 gsg->release_all();
622 }
623 }
624 _new_windows.clear();
625 }
626
627 _app.do_close(this, current_thread);
628 _app.do_pending(this, current_thread);
629 terminate_threads(current_thread);
630
631 // It seems a safe assumption that we're about to exit the application or
632 // otherwise shut down Panda. Although it's a bit of a hack, since it's not
633 // really related to removing windows, this would nevertheless be a fine
634 // time to ensure the model cache (if any) has been flushed to disk.
635 BamCache::flush_global_index();
636
637 // And, hey, let's stop the vertex paging threads, if any.
638 VertexDataPage::stop_threads();
639
640 // Stopping the tasks means we have to release the Python GIL while
641 // this method runs (hence it is marked BLOCKING), so that any
642 // Python tasks on other threads won't deadlock grabbing the GIL.
643 AsyncTaskManager::get_global_ptr()->stop_threads();
644
645#ifdef DO_PSTATS
646 PStatClient::get_global_pstats()->disconnect();
647#endif
648
649#if defined(_WIN32) && defined(HAVE_THREADS) && defined(SIMPLE_THREADS)
650 // Send a message to the input message pump asking it to shut itself down.
651 // If we don't do that, the next call will deadlock.
652 WinInputDeviceManager::stop_thread();
653#endif
654
655 // Well, and why not clean up all threads here?
656 Thread::prepare_for_exit();
657}

708void GraphicsEngine::
709render_frame() {
710 Thread *current_thread = Thread::get_current_thread();
711 ReMutexHolder public_holder(_public_lock);
712
713 // Since this gets called every frame, we should take advantage of the
714 // opportunity to flush the cache if necessary.
715 BamCache::consider_flush_global_index();
716
717 // Anything that happens outside of GraphicsEngine::render_frame() is deemed
718 // to be App.
719#ifdef DO_PSTATS
720 _render_frame_pcollector.start();
721 //if (_app_pcollector.is_started()) {
722 // _app_pcollector.stop();
723 //}
724#endif
725
726 // Make sure our buffers and windows are fully realized before we render a
727 // frame. We do this particularly to realize our offscreen buffers, so
728 // that we don't render a frame before the offscreen buffers are ready
729 // (which might result in a frame going by without some textures having
730 // been rendered).
731 open_windows();
732
733 if (display_cat.is_spam()) {
734 display_cat.spam()
735 << "render_frame() - frame " << _clock->get_frame_count() << "\n";
736 }
737
738 {
739 ReMutexHolder holder(_lock, current_thread);
740
741 if (!_windows_sorted) {
742 do_resort_windows();
743 }
744
745 if (sync_flip && _flip_state != FS_flip) {
746 do_flip_frame(current_thread);
747 }
748
749 // Are any of the windows ready to be deleted?
750 Windows new_windows;
751 new_windows.reserve(_windows.size());
752 Windows::iterator wi;
753 for (wi = _windows.begin(); wi != _windows.end(); ++wi) {
754 GraphicsOutput *win = (*wi);
755 nassertv(win != nullptr);
756 if (win->get_delete_flag()) {
757 do_remove_window(win, current_thread);
758
759 } else {
760 new_windows.push_back(win);
761
762 // Let's calculate each scene's bounding volume here in App, before we
763 // cycle the pipeline. The cull traversal will calculate it anyway,
764 // but if we calculate it in App first before it gets calculated in
765 // the Cull thread, it will be more likely to stick for subsequent
766 // frames, so we won't have to recompute it each frame.
767 int num_drs = win->get_num_active_display_regions();
768 for (int i = 0; i < num_drs; ++i) {
769 PT(DisplayRegion) dr = win->get_active_display_region(i);
770 if (dr != nullptr) {
771 NodePath camera_np = dr->get_camera(current_thread);
772 if (!camera_np.is_empty()) {
773 Camera *camera = DCAST(Camera, camera_np.node());
774 NodePath scene = camera->get_scene();
775 if (scene.is_empty()) {
776 scene = camera_np.get_top(current_thread);
777 }
778 if (!scene.is_empty()) {
779 scene.get_bounds(current_thread);
780 }
781 }
782 }
783 }
784 }
785 }
786 _windows = std::move(new_windows);
787
788 // Go ahead and release any textures' ram images for textures that were
789 // drawn in the previous frame.
790 {
791 MutexHolder holder2(_loaded_textures_lock);
792 LoadedTextures::iterator lti;
793 for (lti = _loaded_textures.begin(); lti != _loaded_textures.end(); ++lti) {
794 LoadedTexture &lt = (*lti);
795 if (lt._tex->get_image_modified() == lt._image_modified) {
796 lt._tex->texture_uploaded();
797 }
798 }
799 _loaded_textures.clear();
800 }
801
802 // Now it's time to do any drawing from the main frame--after all of the
803 // App code has executed, but before we begin the next frame.
804 _app.do_frame(this, current_thread);
805
806 // Grab each thread's mutex again after all windows have flipped, and wait
807 // for the thread to finish.
808 {
809 PStatTimer timer(_wait_pcollector, current_thread);
810 Threads::const_iterator ti;
811 for (ti = _threads.begin(); ti != _threads.end(); ++ti) {
812 RenderThread *thread = (*ti).second;
813 thread->_cv_mutex.acquire();
814
815 while (thread->_thread_state != TS_wait) {
816 thread->_cv_done.wait();
817 }
818 }
819 }
820
821#if defined(THREADED_PIPELINE) && defined(DO_PSTATS)
822 _cyclers_pcollector.set_level(_pipeline->get_num_cyclers());
823 _dirty_cyclers_pcollector.set_level(_pipeline->get_num_dirty_cyclers());

825#ifdef DEBUG_THREADS
826 if (PStatClient::is_connected()) {
827 _pipeline->iterate_all_cycler_types(pstats_count_cycler_type, this);
828 _pipeline->iterate_dirty_cycler_types(pstats_count_dirty_cycler_type, this);
829 }
830#endif // DEBUG_THREADS
831
832#endif // THREADED_PIPELINE && DO_PSTATS

834 GeomCacheManager::flush_level();
835 CullTraverser::flush_level();
836 RenderState::flush_level();
837 TransformState::flush_level();
838 CullableObject::flush_level();

846#endif // THREADED_PIPELINE

932 // and get blocked on the mutex straight away.
933 thread->_thread_state = TS_do_frame;

937 thread->_cv_mutex.release();
938 }

942 // those threads before we can flip.

944 }

947

950 // applications.
951 PStatTimer timer(_yield_pcollector, current_thread);
952 Thread::force_yield();
953 } else if (!Thread::is_true_threads()) {
954 PStatTimer timer(_yield_pcollector, current_thread);
955 Thread::consider_yield();
956 }
957
958 // Anything that happens outside of GraphicsEngine::render_frame() is deemed
959 // to be App.
960 //_app_pcollector.start();
961 _render_frame_pcollector.stop();
962}

972void GraphicsEngine::
973open_windows() {
974 Thread *current_thread = Thread::get_current_thread();
975
976 ReMutexHolder holder(_lock, current_thread);
977
978 NewWindows new_windows;
979 {
980 MutexHolder new_windows_holder(_new_windows_lock, current_thread);
981 if (_new_windows.empty()) {
982 return;
983 }
984
985 for (auto it = _new_windows.begin(); it != _new_windows.end(); ++it) {
986 GraphicsOutput *window = *it;
987
988 WindowRenderer *cull =
989 get_window_renderer(_threading_model.get_cull_name(),
990 _threading_model.get_cull_stage());
991 WindowRenderer *draw =
992 get_window_renderer(_threading_model.get_draw_name(),
993 _threading_model.get_draw_stage());
994
995 if (_threading_model.get_cull_sorting()) {
996 cull->add_window(cull->_cull, window);
997 draw->add_window(draw->_draw, window);
998 } else {
999 cull->add_window(cull->_cdraw, window);
1000 }
1001
1002 // Ask the pipe which thread it prefers to run its windowing commands in
1003 // (the "window thread"). This is the thread that handles the commands
1004 // to open, resize, etc. the window. X requires this to be done in the
1005 // app thread (along with all the other windows, since X is strictly
1006 // single-threaded), but Windows requires this to be done in draw
1007 // (because once an OpenGL context has been bound in a given thread, it
1008 // cannot subsequently be bound in any other thread, and we have to bind
1009 // a context in open_window()).
1010
1011 switch (window->get_pipe()->get_preferred_window_thread()) {
1012 case GraphicsPipe::PWT_app:
1013 _app.add_window(_app._window, window);
1014 break;
1015
1016 case GraphicsPipe::PWT_draw:
1017 draw->add_window(draw->_window, window);
1018 break;
1019 }
1020
1021 _windows.push_back(window);
1022 }
1023
1024 // Steal the list, since remove_window() may remove from _new_windows.
1025 new_windows = std::move(_new_windows);
1026 }
1027
1028 do_resort_windows();
1029
1030 // We do it twice, to allow both cull and draw to process the window.
1031 for (int i = 0; i < 2; ++i) {
1032 _app.do_windows(this, current_thread);
1033 _app.do_pending(this, current_thread);
1034
1035 PStatTimer timer(_wait_pcollector, current_thread);
1036 Threads::const_iterator ti;
1037 for (ti = _threads.begin(); ti != _threads.end(); ++ti) {
1038 RenderThread *thread = (*ti).second;
1039 thread->_cv_mutex.acquire();
1040
1041 while (thread->_thread_state != TS_wait) {
1042 thread->_cv_done.wait();
1043 }
1044
1045 thread->_thread_state = TS_do_windows;
1046 thread->_cv_mutex.release();
1047 thread->_cv_start.notify();
1048 }
1049 }
1050
1051 // Now go through the list again to check whether they opened successfully.
1052 for (auto it = new_windows.begin(); it != new_windows.end(); ++it) {
1053 GraphicsOutput *window = *it;
1054 if (window->is_valid()) {
1055 do_add_gsg(window->get_gsg(), window->get_pipe());
1056 } else {
1057 remove_window(window);
1058 }
1059 }
1060}

1442void GraphicsEngine::
1443cull_to_bins(GraphicsEngine::Windows wlist, Thread *current_thread) {
1444 PStatTimer timer(_cull_pcollector, current_thread);
1445
1446 _singular_warning_last_frame = _singular_warning_this_frame;
1447 _singular_warning_this_frame = false;
1448
1449 // Keep track of the cameras we have already used in this thread to render
1450 // DisplayRegions.
1451 typedef pmap<CullKey, DisplayRegion *> AlreadyCulled;
1452 AlreadyCulled already_culled;
1453
1454 // We cull shadow passes last; whether we cull them depends on whether their
1455 // respective frusta are in view of a "normal" camera.
1456 pvector<PT(SceneSetup)> shadow_passes;
1457 pmap<NodePath, UnionBoundingVolume> non_shadow_bounds;
1458
1459 size_t wlist_size = wlist.size();
1460 for (size_t wi = 0; wi < wlist_size; ++wi) {
1461 GraphicsOutput *win = wlist[wi];
1462 if (win->is_active() && win->get_gsg()->is_active()) {
1463 if (display_cat.is_spam()) {
1464 display_cat.spam()
1465 << "Culling window " << win->get_name() << "\n";
1466 }
1467
1468 GraphicsStateGuardian *gsg = win->get_gsg();
1469 PStatTimer timer(win->get_cull_window_pcollector(), current_thread);
1470 int num_display_regions = win->get_num_active_display_regions();
1471 for (int i = 0; i < num_display_regions; ++i) {
1472 PT(DisplayRegion) dr = win->get_active_display_region(i);
1473 if (dr != nullptr) {
1474 PT(SceneSetup) scene_setup;
1475 PT(CullResult) cull_result;
1476 CullKey key;
1477 {
1478 PStatTimer timer(_cull_setup_pcollector, current_thread);
1479 DisplayRegionPipelineReader dr_reader(dr, current_thread);
1480 scene_setup = setup_scene(gsg, &dr_reader);
1481 if (scene_setup == nullptr) {
1482 continue;
1483 }
1484
1485 key._gsg = gsg;
1486 key._camera = dr_reader.get_camera();
1487 key._lens_index = dr_reader.get_lens_index();
1488 }
1489
1490 // If this is a shadow pass, postpone culling it until we've culled
1491 // all the other passes, and collected their bounding volumes.
1492 Light *light = nullptr;
1493 if (!key._camera.is_empty()) {
1494 light = key._camera.node()->as_light();
1495 }
1496 if (light != nullptr) {
1497 shadow_passes.push_back(std::move(scene_setup));
1498 continue;
1499 }
1500 else if (!shadow_passes.empty()) {
1501 NodePath scene_root = scene_setup->get_scene_root();
1502 const GeometricBoundingVolume *gbv = scene_setup->get_view_frustum();
1503 UnionBoundingVolume &bounds = non_shadow_bounds[scene_root];
1504 if (gbv == nullptr || gbv->is_infinite()) {
1505 bounds.set_infinite();
1506 } else {
1507 bounds.add_component(gbv);
1508 }
1509 }
1510
1511 AlreadyCulled::iterator aci = already_culled.insert(AlreadyCulled::value_type(std::move(key), nullptr)).first;
1512 if ((*aci).second == nullptr) {
1513 // We have not used this camera already in this thread. Perform
1514 // the cull operation.
1515 cull_result = dr->get_cull_result(current_thread);
1516 if (cull_result != nullptr) {
1517 cull_result = cull_result->make_next();
1518 } else {
1519 // This DisplayRegion has no cull results; draw it.
1520 cull_result = new CullResult(gsg, dr->get_draw_region_pcollector());
1521 }
1522 (*aci).second = dr;
1523 cull_to_bins(win, gsg, dr, scene_setup, cull_result, current_thread);
1524
1525 } else {
1526 // We have already culled a scene using this camera in this
1527 // thread, and now we're being asked to cull another scene using
1528 // the same camera. (Maybe this represents two different
1529 // DisplayRegions for the left and right channels of a stereo
1530 // image.) Of course, the cull result will be the same, so just
1531 // use the result from the other DisplayRegion.
1532 DisplayRegion *other_dr = (*aci).second;
1533 cull_result = other_dr->get_cull_result(current_thread);
1534 }
1535
1536 // Save the results for next frame.
1537 dr->set_cull_result(std::move(cull_result), std::move(scene_setup), current_thread);
1538 }
1539 }
1540 }
1541 }
1542
1543 // Now cull all the shadow passes if their cull bounds are in view.
1544 // We don't bother checking the AlreadyCulled list, because we know there is
1545 // only one output per GSG+light combination.
1546 for (PT(SceneSetup) &scene_setup : shadow_passes) {
1547 DisplayRegion *dr = scene_setup->get_display_region();
1548 GraphicsOutput *win = dr->get_window();
1549 PStatTimer timer(win->get_cull_window_pcollector(), current_thread);
1550
1551 PT(CullResult) cull_result;
1552
1553 // Are the cull bounds in view of another camera?
1554 GeometricBoundingVolume *frustum = scene_setup->get_view_frustum();
1555 if (frustum == nullptr ||
1556 non_shadow_bounds[scene_setup->get_scene_root()].contains(frustum)) {
1557 GraphicsStateGuardian *gsg = win->get_gsg();
1558 cull_result = dr->get_cull_result(current_thread);
1559 if (cull_result != nullptr) {
1560 cull_result = cull_result->make_next();
1561 } else {
1562 // This DisplayRegion has no cull results; draw it.
1563 cull_result = new CullResult(gsg, dr->get_draw_region_pcollector());
1564 }
1565 cull_to_bins(win, gsg, dr, scene_setup, cull_result, current_thread);
1566 }
1567 else if (display_cat.is_spam()) {
1568 display_cat.spam()
1569 << *scene_setup->get_camera_node()
1570 << " frustum is not in view, skipping shadow pass\n";
1571 }
1572
1573 // Even save the results if null, to tell the draw pass that we don't want
1574 // to draw this at all.
1575 dr->set_cull_result(std::move(cull_result), std::move(scene_setup), current_thread);
1576 }

1877PT(SceneSetup) GraphicsEngine::
1878setup_scene(GraphicsStateGuardian *gsg, DisplayRegionPipelineReader *dr) {
1879 Thread *current_thread = dr->get_current_thread();
1880 PStatTimer timer(_cull_setup_pcollector, current_thread);
1881
1882 GraphicsOutput *window = dr->get_window();
1883 // The window pointer shouldn't be NULL, since we presumably got to this
1884 // particular DisplayRegion by walking through a list on a window.
1885 nassertr(window != nullptr, nullptr);
1886
1887 NodePath camera = dr->get_camera();
1888 if (camera.is_empty()) {
1889 // No camera, no draw.
1890 return nullptr;
1891 }
1892
1893 Camera *camera_node;
1894 DCAST_INTO_R(camera_node, camera.node(), nullptr);
1895
1896 if (!camera_node->is_active()) {
1897 // Camera inactive, no draw.
1898 return nullptr;
1899 }
1900 camera_node->cleanup_aux_scene_data(current_thread);
1901
1902 int lens_index = dr->get_lens_index();
1903 Lens *lens = camera_node->get_lens(lens_index);
1904 if (lens == nullptr || !camera_node->get_lens_active(lens_index)) {
1905 // No lens, no draw.
1906 return nullptr;
1907 }
1908
1909 NodePath scene_root = camera_node->get_scene();
1910 if (scene_root.is_empty()) {
1911 // If there's no explicit scene specified, use whatever scene the camera
1912 // is parented within. This is the normal and preferred case; the use of
1913 // an explicit scene is now deprecated.
1914 scene_root = camera.get_top(current_thread);
1915 }
1916
1917 PT(SceneSetup) scene_setup = new SceneSetup;
1918
1919 // We will need both the camera transform (the net transform to the camera
1920 // from the scene) and the world transform (the camera transform inverse, or
1921 // the net transform to the scene from the camera). These are actually
1922 // defined from the parent of the scene_root, because the scene_root's own
1923 // transform is immediately applied to these during rendering. (Normally,
1924 // the parent of the scene_root is the empty NodePath, although it need not
1925 // be.)
1926 NodePath scene_parent = scene_root.get_parent(current_thread);
1927 CPT(TransformState) camera_transform = camera.get_transform(scene_parent, current_thread);
1928 CPT(TransformState) world_transform = scene_parent.get_transform(camera, current_thread);
1929
1930 if (camera_transform->is_invalid()) {
1931 // There must be a singular transform over the scene.
1932 if (!_singular_warning_last_frame) {
1933 display_cat.warning()
1934 << "Scene " << scene_root << " has net scale ("
1935 << scene_root.get_scale(NodePath()) << "); cannot render.\n";
1936 _singular_warning_this_frame = true;
1937 }
1938 return nullptr;
1939 }
1940
1941 if (world_transform->is_invalid()) {
1942 // There must be a singular transform over the camera.
1943 if (!_singular_warning_last_frame) {
1944 display_cat.warning()
1945 << "Camera " << camera << " has net scale ("
1946 << camera.get_scale(NodePath()) << "); cannot render.\n";
1947 }
1948 _singular_warning_this_frame = true;
1949 return nullptr;
1950 }
1951
1952 CPT(RenderState) initial_state = camera_node->get_initial_state();
1953
1954 if (window->get_inverted()) {
1955 // If the window is to be inverted, we must set the inverted flag on the
1956 // SceneSetup object, so that the GSG will be able to invert the
1957 // projection matrix at the last minute.
1958 scene_setup->set_inverted(true);
1959
1960 // This also means we need to globally invert the sense of polygon vertex
1961 // ordering.
1962 initial_state = initial_state->compose(get_invert_polygon_state());
1963 }
1964
1965 scene_setup->set_display_region(dr->get_object());
1966 scene_setup->set_viewport_size(dr->get_pixel_width(), dr->get_pixel_height());
1967 scene_setup->set_scene_root(scene_root);
1968 scene_setup->set_camera_path(camera);
1969 scene_setup->set_camera_node(camera_node);
1970 scene_setup->set_lens(lens);
1971 scene_setup->set_initial_state(initial_state);
1972 scene_setup->set_camera_transform(camera_transform);
1973 scene_setup->set_world_transform(world_transform);
1974
1975 CPT(TransformState) cs_transform = gsg->get_cs_transform_for(lens->get_coordinate_system());
1976 scene_setup->set_cs_transform(cs_transform);
1977
1978 CPT(TransformState) cs_world_transform = cs_transform->compose(world_transform);
1979 scene_setup->set_cs_world_transform(cs_world_transform);
1980
1981 if (view_frustum_cull) {
1982 // If we're to be performing view-frustum culling, determine the bounding
1983 // volume associated with the current viewing frustum.
1984
1985 // First, we have to get the current viewing frustum, which comes from the
1986 // lens.
1987 PT(BoundingVolume) bv = scene_setup->get_cull_bounds();
1988
1989 if (bv != nullptr && !bv->is_infinite() &&
1990 bv->as_geometric_bounding_volume() != nullptr) {
1991 // Transform it into the appropriate coordinate space.
1992 PT(GeometricBoundingVolume) local_frustum;
1993 local_frustum = bv->make_copy()->as_geometric_bounding_volume();
1994 nassertr(!local_frustum.is_null(), nullptr);
1995
1996 CPT(TransformState) cull_center_transform =
1997 scene_setup->get_cull_center().get_transform(scene_parent, current_thread);
1998 local_frustum->xform(cull_center_transform->get_mat());
1999
2000 scene_setup->set_view_frustum(local_frustum);
2001 }
2002 }
2003
2004 return scene_setup;
2005}

2011do_draw(GraphicsOutput *win, GraphicsStateGuardian *gsg, DisplayRegion *dr, Thread *current_thread) {
2012 // Statistics
2013 PStatGPUTimer timer(gsg, dr->get_draw_region_pcollector(), current_thread);
2014
2015 gsg->push_group_marker(dr->get_debug_name());
2016
2017 PT(CullResult) cull_result;

2019 {
2020 DisplayRegion::CDCullReader cdata(dr->_cycler_cull, current_thread);
2021 cull_result = cdata->_cull_result;
2022 scene_setup = cdata->_scene_setup;
2023 }

2026
2027 {
2028 DisplayRegionPipelineReader dr_reader(dr, current_thread);

2037 }

2047 gsg->set_state_and_transform(state, TransformState::make_identity());

2050 cbobj->do_callback(&cbdata);
2051
2052 // We don't trust the state the callback may have left us in.
2053 gsg->clear_state_and_transform();
2054
2055 } else if (cull_result == nullptr || scene_setup == nullptr) {
2056 // Nothing to see here.
2057
2058 } else if (dr->is_stereo()) {
2059 // We don't actually draw the stereo DisplayRegions. These are just
2060 // placeholders; we draw the individual left and right eyes instead. (We
2061 // might still clear the stereo DisplayRegions, though, since it's
2062 // probably faster to clear right and left channels in one pass, than to
2063 // clear them in two separate passes.)
2064
2065 } else if (!gsg->set_scene(scene_setup)) {
2066 // The scene or lens is inappropriate somehow.
2067 display_cat.error()
2068 << gsg->get_type() << " cannot render scene with specified lens.\n";
2069
2070 } else {
2071 if (gsg->begin_scene()) {
2072 cull_result->draw(current_thread);
2073 gsg->end_scene();
2074 }
2075 }
2076
2077 gsg->pop_group_marker();
2078}

2207void GraphicsEngine::
2208auto_adjust_capabilities(GraphicsStateGuardian *gsg) {
2209
2210/*
2211 * The rule we use when auto-reconfiguring is as follows. The global
2212 * capabilities must initially be set to conservative values. When the first
2213 * GSG comes into existence, its capabilities will be checked, and the global
2214 * capabilities may be elevated to more aggressive values. At first glance,
2215 * this might seem backward, and it might seem better to do it the other way:
2216 * start with all global capabilities aggressively set, and then disable
2217 * capabilities when you discover a gsg that doesn't support them. However,
2218 * that approach doesn't work, because once a global capability is enabled,
2219 * there is no going back. If textures_power_2 has ever been set to 'none',
