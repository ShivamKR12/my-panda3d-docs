132bool CocoaGraphicsWindow::
133move_pointer(int device, int x, int y) {
134 // Hack! Will go away when we have floating-point mouse pos.
135 PointerData md = get_pointer(device);
136 if (md.get_x() == x && md.get_y() == y) {
137 return true;
138 }
139
140 // Mouse position is expressed in screen points and not pixels, but in Panda3D
141 // we are using pixel coordinates.
142 // Instead of using convertPointFromBacking and have complex logic to cope with
143 // the change of coordinate system, we cheat and directly use the contents scale
144 // of the view layer to convert pixel coordinates into screen point coordinates.
145 CGFloat contents_scale = _view.layer.contentsScale;
146 if (device == 0) {
147 CGPoint point;
148 if (_properties.get_fullscreen()) {
149 point = CGPointMake(float(x) / contents_scale,
150 float(y) / contents_scale);
151 } else {
152 point = CGPointMake((float(x) + _properties.get_x_origin()) / contents_scale,
153 (float(y) + _properties.get_y_origin()) / contents_scale);
154 }
155
156 if (CGWarpMouseCursorPosition(point) == kCGErrorSuccess) {
157 //After moving (or warping) the mouse position, CG starts an event
158 // suppression interval during which no more mouse events can occur
159 // This interval can be interupted by the following call :
160 CGAssociateMouseAndMouseCursorPosition(YES);
161 // Generate a mouse event.
162 NSPoint pos = [_window mouseLocationOutsideOfEventStream];
163 NSPoint loc = [_view convertPoint:pos fromView:nil];
164 BOOL inside = [_view mouse:loc inRect:[_view bounds]];
165 handle_mouse_moved_event(inside, loc.x, loc.y, true);
166 return true;
167 }
168 } else {
169 // No support for raw mice at the moment.
170 }
171 return false;
172}

279 // application, it could be either a HIViewRef or an NSView or NSWindow.
280 // Currently, we only support a Cocoa NSView, but we could in the future
281 // add support for Carbon parents using HICocoaView.

284 NativeWindowHandle::IntHandle *int_handle = DCAST(NativeWindowHandle::IntHandle, os_handle);
285 ptr_handle = (void*) int_handle->get_handle();

290 if ([nsobj isKindOfClass:[NSView class]]) {

301 }

305 NSScreen *screen;
306 NSEnumerator *e = [[NSScreen screens] objectEnumerator];
307 while (screen = (NSScreen *) [e nextObject]) {
308 NSNumber *num = [[screen deviceDescription] objectForKey: @"NSScreenNumber"];
309 if (_display == (CGDirectDisplayID) [num longValue]) {
310 break;
311 }

314 // Configure the origin and the size of the window.
315 // On macOS, screen coordinates are expressed in "points" which are independant
316 // of the pixel density of the screen. Panda3D however, expresses the size and
317 // origin of a window in pixels.
318 // So, when opening a window, we need to convert the origin and size from pixel
319 // units into point units. However, this conversion depends on the pixel density
320 // of the screen, the backing scale factor.
321 // As the origin and size of a window depends on the size of the screen of the
322 // parent view, their size must be converted first from points to pixels.
323 // If a window (or a view) is not configured to support high-dpi screen, macOS
324 // will upscale the window (or view) when displayed on a high-dpi screen.
325 // Therefore its backing scale factor will always be 1.0
326 // In a Panda3D application, windows are always configured as high resolution
327 // capable, but the view is only configured as high resolution if the dpi-aware
328 // configuration flag is set.
329 // If the app is not dpi-aware, we must upscale its size and origin from points
330 // into pixels as the window is always high resolution capable.
331
332 // Center the window if coordinates were set to -1 or -2 TODO: perhaps in
333 // future, in the case of -1, it should use the origin used in a previous
334 // run of Panda

336 // Size of the requested window
337 NSSize size = NSMakeSize(_properties.get_x_size(), _properties.get_y_size());
338 NSRect container;
339 CGFloat backing_scale_factor = screen.backingScaleFactor;
340 if (parent_nsview != NULL) {
341 // Convert parent view bounds into pixel units.
342 container = [parent_nsview convertRectToBacking:[parent_nsview bounds]];
343 // If the app is not dpi-aware, we must convert its size from points into
344 // pixels as the window is always high resolution capable
345 if (!dpi_aware) {
346 size = [parent_nsview convertSizeToBacking:size];
347 }
348 } else {
349 container = [screen frame];
350 container.origin = NSMakePoint(0, 0);
351 container = [screen convertRectToBacking:container];
352 if (!dpi_aware) {
353 // Weirdly NSScreen does not respond to convertSizeToBacking, so we have to
354 // create a dummy rect just for converting the window size.
355 NSRect rect;
356 rect.origin = NSMakePoint(0, 0);
357 rect.size = size;
358 rect = [screen convertRectToBacking:rect];
359 size = rect.size;
360 }
361 }

363 int y = _properties.get_y_origin();

366 // easier to simply use the backing scale factor and don't bother with
367 // coordinate system transformations.
368 if (x < 0) {
369 x = floor(container.size.width / 2 - size.width / 2);
370 } else if (!dpi_aware) {
371 x *= backing_scale_factor;
372 }
373 if (y < 0) {
374 y = floor(container.size.height / 2 - size.height / 2);
375 } else if (!dpi_aware) {
376 y *= backing_scale_factor;
377 }
378 if (dpi_aware) {
379 _properties.set_origin(x, y);
380 } else {
381 _properties.set_origin(x / backing_scale_factor, y / backing_scale_factor);
382 }

384 if (_parent_window_handle == (WindowHandle *)NULL) {
385 // Content rectangle
386 NSRect rect;
387 if (_properties.get_fullscreen()) {
388 rect = [screen convertRectFromBacking:container];
389 } else {

391 size.width, size.height);

393 rect = [parent_nsview convertRectFromBacking:rect];
394 } else {
395 rect = [screen convertRectFromBacking:rect];
396 }
397 }

399 // Configure the window decorations
400 NSUInteger windowStyle;

402 windowStyle = NSBorderlessWindowMask;
403 } else if (_properties.get_fixed_size()) {
404 // Fixed size windows should not show the resize button.
405 windowStyle = NSTitledWindowMask | NSClosableWindowMask |
406 NSMiniaturizableWindowMask;
407 } else {
408 windowStyle = NSTitledWindowMask | NSClosableWindowMask |
409 NSMiniaturizableWindowMask | NSResizableWindowMask;
410 }

412 // Create the window.
413 if (cocoadisplay_cat.is_debug()) {
414 NSString *str = NSStringFromRect(rect);
415 cocoadisplay_cat.debug()
416 << "Creating NSWindow with content rect " << [str UTF8String] << "\n";

420 initWithContentRect: rect
421 styleMask:windowStyle
422 screen:screen
423 window:this];

426 cocoadisplay_cat.error()
427 << "Failed to create Cocoa window.\n";
428 return false;
429 }
430 }

432 // Create the NSView to render to.
433 NSRect rect = NSMakeRect(0, 0, _properties.get_x_size(), _properties.get_y_size());
434 _view = [[CocoaPandaView alloc] initWithFrame:rect window:this];
435 if (_parent_window_handle == nullptr) {
436 [_window setContentView:_view];
437 [_window makeFirstResponder:_view];

446 // pointer approach, so let's do the same here.

449 // And tell our parent window that we're now its child.

455 [_view respondsToSelector:@selector(setWantsExtendedDynamicRangeOpenGLSurface:)]) {

462 [_view setWantsBestResolutionOpenGLSurface:dpi_aware];
463 if (_properties.has_icon_filename()) {
464 NSImage *image = load_image(_properties.get_icon_filename());

466 // We're technically changing the application icon, but this is most
467 // likely what the developer means. There isn't really a "window icon"

496 }

504 [_window miniaturize:nil];

781 [_view setFrameSize:size];

783 if (cocoadisplay_cat.is_debug()) {
784 cocoadisplay_cat.debug()
785 << "Setting size to " << width << ", " << height << "\n";
786 }
787
788 // Cocoa doesn't send an event, and the other resize-window handlers
789 // will do nothing once the properties have been changed, so do this now
790 handle_resize_event();
791 properties.clear_size();
792
793 } else {
794 CFMutableArrayRef modes = find_display_modes(width, height);
795
796 if (CFArrayGetCount(modes) > 0) {
797 bool switched = false;
798 for (CFIndex i = 0; i < CFArrayGetCount(modes); i++) {
799 CGDisplayModeRef mode = (CGDisplayModeRef) CFArrayGetValueAtIndex(modes, i);
800 if (do_switch_fullscreen(mode)) {
801 switched = true;
802 break;
803 }
804 }
805
806 if (switched) {
807 // Yay! Our resolution has changed.
808 _properties.set_size(width, height);
809 properties.clear_size();
810 } else {
811 cocoadisplay_cat.error()
812 << "Failed to change display mode.\n";
813 }
814
815 } else {
816 cocoadisplay_cat.error()
817 << "Could not find a suitable display mode with size " << width
818 << "x" << height << "!\n";
819 }
820 CFRelease(modes);
821 }
822 }
823
824 if (properties.has_origin() && !_properties.get_fullscreen()) {
825 int x = properties.get_x_origin();
826 int y = properties.get_y_origin();
827
828 // Get the frame for the screen
829 NSRect frame;
830 NSRect container;
831 // Note again that we are using the view to convert the frame and container
832 // size from points into pixels.
833 if (_window != nil) {
834 NSRect window_frame = [_window frame];
835 frame = [_view convertRectToBacking:[_window contentRectForFrameRect:window_frame]];
836 NSScreen *screen = [_window screen];
837 nassertv(screen != nil);
838 container = [_view convertRectToBacking:[screen frame]];
839
840 // Prevent the centering from overlapping the Dock
841 if (y < 0) {
842 NSRect visible_frame = [screen visibleFrame];
843 if (window_frame.size.height == visible_frame.size.height) {
844 y = 0;
845 }
846 }
847 } else {
848 frame = [_view convertRectToBacking:[_view frame]];
849 container = [[_view superview] convertRectToBacking:[[_view superview] frame]];
850 }
851
852 if (x < 0) {
853 x = floor(container.size.width / 2 - frame.size.width / 2);
854 }
855 if (y < 0) {
856 y = floor(container.size.height / 2 - frame.size.height / 2);
857 }
858 _properties.set_origin(x, y);
859
860 frame.origin.x = x;
861 // Y coordinate in backing store is not flipped, but origin is still at the bottom left
862 frame.origin.y = y - container.size.height;
863
864 if (cocoadisplay_cat.is_debug()) {
865 cocoadisplay_cat.debug()
866 << "Setting window content origin to "
867 << frame.origin.x << ", " << frame.origin.y << "\n";
868 }
869
870 if (_window != nil) {
871 frame = [_view convertRectFromBacking:frame];
872 [_window setFrame:[_window frameRectForContentRect:frame] display:NO];
873 } else {
874 frame = [_view convertRectFromBacking:frame];
875 [_view setFrame:frame];
876 }
877 properties.clear_origin();
878 }
879
880 if (properties.has_maximized() && _window != nil) {
881 _properties.set_maximized(properties.get_maximized());
882 if (properties.get_maximized() != !![_window isZoomed]) {
883 [_window zoom:nil];
884 }
885 properties.clear_maximized();
886 }
887
888 if (properties.has_foreground() && !_properties.get_fullscreen() && _window != nil) {
889 _properties.set_foreground(properties.get_foreground());
890 if (!_properties.get_minimized()) {
891 if (properties.get_foreground()) {
892 [_window makeKeyAndOrderFront: nil];
893 } else {
894 [_window orderBack: nil];
895 }
896 }
897 properties.clear_foreground();
898 }
899
900 // TODO: support raw mice.
901
902 if (properties.has_cursor_hidden()) {
903 if (properties.get_cursor_hidden() != _properties.get_cursor_hidden()) {
904 if (properties.get_cursor_hidden() && _input->get_pointer().get_in_window()) {
905 [NSCursor hide];
906 _mouse_hidden = true;
907 } else if (_mouse_hidden) {
908 [NSCursor unhide];
909 _mouse_hidden = false;
910 }
911 _properties.set_cursor_hidden(properties.get_cursor_hidden());
912 }
913 properties.clear_cursor_hidden();
914 }
915
916 if (properties.has_icon_filename()) {
917 Filename icon_filename = properties.get_icon_filename();
918 NSImage *image = load_image(icon_filename);
919
920 if (image != nil || icon_filename.empty()) {
921 // We're technically changing the application icon, but this is most
922 // likely what the developer means. There isn't really a "window icon"
923 // in Mac OS X.
924 [NSApp setApplicationIconImage:image];
925 _properties.set_icon_filename(icon_filename);
926 properties.clear_icon_filename();
927 }
928 }
929
930 if (properties.has_cursor_filename()) {
931 Filename cursor_filename = properties.get_cursor_filename();
932
933 if (cursor_filename.empty()) {
934 // Release the existing cursor.
935 if (_cursor != nil) {
936 [_cursor release];
937 _cursor = nil;
938 }
939 properties.set_cursor_filename(cursor_filename);
940 properties.clear_cursor_filename();
941 } else {
942 NSCursor *cursor = load_cursor(cursor_filename);
943 if (cursor != nil) {
944 // Replace the existing cursor.
945 if (_cursor != nil) {
946 [_cursor release];
947 }
948 _cursor = cursor;
949 _properties.set_cursor_filename(cursor_filename);
950 properties.clear_cursor_filename();
951 }
952 }
953 // This will ensure that NSView's resetCursorRects gets called, which sets
954 // the appropriate cursor rects.
955 [[_view window] invalidateCursorRectsForView:_view];
956 }
957
958 if (properties.has_z_order() && _window != nil) {
959 _properties.set_z_order(properties.get_z_order());
960
961 if (!_properties.get_fullscreen()) {
962 switch (properties.get_z_order()) {
963 case WindowProperties::Z_bottom:
964 [_window setLevel: NSNormalWindowLevel - 1];
965 break;
966
967 case WindowProperties::Z_normal:
968 [_window setLevel: NSNormalWindowLevel];
969 break;
970
971 case WindowProperties::Z_top:
972 [_window setLevel: NSPopUpMenuWindowLevel];
973 break;
974 }
975 }
976 properties.clear_z_order();
977 }
978
979 if (properties.has_mouse_mode()) {
980 switch (properties.get_mouse_mode()) {
981 case WindowProperties::M_absolute:
982 case WindowProperties::M_confined: // confined is maintained in mouse move event
983 CGAssociateMouseAndMouseCursorPosition(true);
984 _properties.set_mouse_mode(properties.get_mouse_mode());
985 properties.clear_mouse_mode();
986 break;
987
988 case WindowProperties::M_relative:
989 CGAssociateMouseAndMouseCursorPosition(false);
990 _properties.set_mouse_mode(properties.get_mouse_mode());
991 properties.clear_mouse_mode();
992 break;
993 }
994 }
995
996 if (_context_needs_update && _gsg != nullptr) {
997 update_context();
998 }

1019CFMutableArrayRef CocoaGraphicsWindow::
1020find_display_modes(int width, int height) {
1021 CFDictionaryRef options = NULL;
1022 // We want to select the display mode with the same scaling factor as the
1023 // current view to avoid cropping or scaling issues.
1024 // CGDisplayCopyAllDisplayModes() does not return upscaled display modes
1025 // nor the current mode, unless explicitly asked with
1026 // kCGDisplayShowDuplicateLowResolutionModes
1027 // (which is undocumented...).
1028 const CFStringRef dictkeys[] = {kCGDisplayShowDuplicateLowResolutionModes};
1029 const CFBooleanRef dictvalues[] = {kCFBooleanTrue};
1030 options = CFDictionaryCreate(NULL,
1031 (const void **)dictkeys,
1032 (const void **)dictvalues,
1033 1,
1034 &kCFCopyStringDictionaryKeyCallBacks,
1035 &kCFTypeDictionaryValueCallBacks);
1036 CFMutableArrayRef valid_modes;
1037 valid_modes = CFArrayCreateMutable(NULL, 0, &kCFTypeArrayCallBacks);
1038
1039 CFArrayRef modes = CGDisplayCopyAllDisplayModes(_display, options);
1040 if (options != NULL) {
1041 CFRelease(options);
1042 }
1043
1044 size_t num_modes = CFArrayGetCount(modes);
1045 CGDisplayModeRef mode;
1046 mode = CGDisplayCopyDisplayMode(_display);
1047
1048 // Calculate requested display size and pixel size
1049 CGSize display_size;
1050 CGSize pixel_size;
1051 if (dpi_aware) {
1052 pixel_size = NSMakeSize(width, height);
1053 display_size = [_view convertSizeFromBacking:pixel_size];
1054 } else {
1055 display_size = NSMakeSize(width, height);
1056 // Calculate the pixel width and height of the fullscreen mode we want using
1057 // the current display mode dimensions and pixel dimensions.
1058 size_t pixel_width = (size_t(width) * CGDisplayModeGetPixelWidth(mode)) / CGDisplayModeGetWidth(mode);
1059 size_t pixel_height = (size_t(height) * CGDisplayModeGetPixelHeight(mode)) / CGDisplayModeGetHeight(mode);
1060 pixel_size = NSMakeSize(pixel_width, pixel_height);
1061 }
1062
1063 // First check if the current mode is adequate.
1064 if (CGDisplayModeGetWidth(mode) == display_size.width &&
1065 CGDisplayModeGetHeight(mode) == display_size.height &&
1066 CGDisplayModeGetPixelWidth(mode) == pixel_size.width &&
1067 CGDisplayModeGetPixelHeight(mode) == pixel_size.height) {
1068 CFArrayAppendValue(valid_modes, mode);
1069 CGDisplayModeRelease(mode);
1070 return valid_modes;
1071 }
1072
1073 // Get the current refresh rate and pixel encoding.
1074 CFStringRef current_pixel_encoding;
1075 double refresh_rate;
1076
1077 current_pixel_encoding = CGDisplayModeCopyPixelEncoding(mode);
1078 refresh_rate = CGDisplayModeGetRefreshRate(mode);
1079 CGDisplayModeRelease(mode);
1080
1081 for (size_t i = 0; i < num_modes; ++i) {
1082 mode = (CGDisplayModeRef) CFArrayGetValueAtIndex(modes, i);
1083
1084 CFStringRef pixel_encoding = CGDisplayModeCopyPixelEncoding(mode);
1085
1086 // We select the fullscreen display mode using he same scaling factor
1087 // To do this we check the mode width and height but also actual pixel widh
1088 // and height.
1089 if (CGDisplayModeGetWidth(mode) == display_size.width &&
1090 CGDisplayModeGetHeight(mode) == display_size.height &&
1091 (int)(CGDisplayModeGetRefreshRate(mode) + 0.5) == (int)(refresh_rate + 0.5) &&
1092 CGDisplayModeGetPixelWidth(mode) == pixel_size.width &&
1093 CGDisplayModeGetPixelHeight(mode) == pixel_size.height &&
1094 CFStringCompare(pixel_encoding, current_pixel_encoding, 0) == kCFCompareEqualTo) {
1095 if (CGDisplayModeGetRefreshRate(mode) == refresh_rate) {
1096 // Exact match for refresh rate, prioritize this.
1097 CFArrayInsertValueAtIndex(valid_modes, 0, mode);
1098 } else {
1099 CFArrayAppendValue(valid_modes, mode);
1100 }
1101 }
1102 CFRelease(pixel_encoding);
1103 }
1104
1105 CFRelease(current_pixel_encoding);
1106 CFRelease(modes);
1107 return valid_modes;
1108}

1114bool CocoaGraphicsWindow::
1115do_switch_fullscreen(CGDisplayModeRef mode) {
1116 if (mode == NULL) {
1117 if (_windowed_mode == NULL) {
1118 // Already windowed.
1119 return true;
1120 }
1121
1122 // Switch back to the mode we were in when we were still windowed.
1123 CGDisplaySetDisplayMode(_display, _windowed_mode, NULL);
1124 CGDisplayModeRelease(_windowed_mode);
1125 if (CGDisplayIsMain(_display)) {
1126 CGReleaseAllDisplays();
1127 } else {
1128 CGDisplayRelease(_display);
1129 }
1130 _windowed_mode = NULL;
1131 _context_needs_update = true;
1132
1133 } else {
1134 if (_windowed_mode != NULL && _fullscreen_mode == mode) {
1135 // Already fullscreen in that size.
1136 return true;
1137 }
1138
1139 // Store the existing mode under _windowed_mode.
1140 _windowed_mode = CGDisplayCopyDisplayMode(_display);
1141 _fullscreen_mode = mode;
1142 _context_needs_update = true;
1143
1144 // Display must be captured by the application before switching mode.
1145 // If not, the change of mode and resolution will be applied on all the other applications,
1146 // although they are no longer visible.
1147 // This also leads to weird bugs when switching back to the desktop mode.
1148 CGError err;
1149 if (CGDisplayIsMain(_display)) {
1150 // In multidisplay setup, all the displays must be captured or the switch will be notified anyway.
1151 err = CGCaptureAllDisplays();
1152 } else {
1153 err = CGDisplayCapture(_display);
1154 }
1155 if (err != kCGErrorSuccess) {
1156 return false;
1157 }
1158
1159 err = CGDisplaySetDisplayMode(_display, _fullscreen_mode, NULL);
1160
1161 if (err != kCGErrorSuccess) {
1162 if (CGDisplayIsMain(_display)) {
1163 CGReleaseAllDisplays();
1164 } else {
1165 CGDisplayRelease(_display);
1166 }
1167 return false;
1168 }
1169
1170 NSRect frame = [[[_view window] screen] frame];
1171 if (cocoadisplay_cat.is_debug()) {
1172 NSString *str = NSStringFromSize([_view convertSizeToBacking:frame.size]);
1173 cocoadisplay_cat.debug()
1174 << "Switched to fullscreen, screen size is now " << [str UTF8String] << "\n";
1175 }
1176
1177 if (_window != nil) {
1178 // Exit macOS' own fullscreen mode, since our own fullscreen mode
1179 // doesn't work properly with it.
1180 if ([_window styleMask] & NSFullScreenWindowMask) {
1181 [_window toggleFullScreen:nil];
1182 }
1183
1184 // For some reason, setting the style mask makes it give up its
1185 // first-responder status.
1186 if ([_window respondsToSelector:@selector(setStyleMask:)]) {
1187 [_window setStyleMask:([_window styleMask] & NSFullScreenWindowMask)];
1188 }
1189 [_window makeFirstResponder:_view];
1190 [_window setLevel:CGShieldingWindowLevel()];
1191 [_window makeKeyAndOrderFront:nil];
1192
1193 // Window and view frame must be updated *after* the window reconfiguration
1194 // or the size is not set properly !
1195 [_window setFrame:frame display:YES];
1196 [_view setFrame:frame];
1197 [_window update];
1198 }
1199 }
1200
1201 return true;
1202}

1209NSData *CocoaGraphicsWindow::
1210load_image_data(const Filename &filename) {
1211 if (filename.empty()) {
1212 return nil;
1213 }
1214
1215 // Note: perhaps eventually we will need to create an NSImageRep
1216 // implementation, but for now, Apple seems to support the major image
1217 // formats.
1218
1219 // Resolve the filename on the model path.
1220 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
1221 Filename resolved (filename);
1222 if (!vfs->resolve_filename(resolved, get_model_path())) {
1223 // The filename doesn't exist.
1224 cocoadisplay_cat.warning()
1225 << "Could not find filename " << filename << "\n";
1226 return 0;
1227 }
1228
1229 // Look in our index.
1230 IconImages::const_iterator it = _images.find(resolved);
1231 if (it != _images.end()) {
1232 // Found it.
1233 return (*it).second;
1234 }
1235
1236 cocoadisplay_cat.info()
1237 << "Loading NSImage from file " << resolved << "\n";
1238
1239 PT(VirtualFile) vfile = vfs->get_file(resolved);
1240 if (vfile == NULL) {
1241 return nil;
1242 }
1243 std::istream *str = vfile->open_read_file(true);
1244 if (str == NULL) {
1245 cocoadisplay_cat.error()
1246 << "Could not open file " << filename << " for reading\n";
1247 return nil;
1248 }
1249
1250 size_t size = vfile->get_file_size(str);
1251 char* buffer = (char*) malloc(size);
1252 str->read(buffer, size);
1253 vfile->close_read_file(str);
1254
1255 NSData *data = [NSData dataWithBytesNoCopy:buffer length:size];
1256 if (data == nil) {
1257 cocoadisplay_cat.error()
1258 << "Could not load image data from file " << filename << "\n";
1259 return nil;
1260 }
1261
1262 _images[resolved] = data;
1263 return data;
1264}

1286NSCursor *CocoaGraphicsWindow::
1287load_cursor(const Filename &filename) {
1288 NSData *image_data = load_image_data(cursor_filename);
1289 if (image_data == nil) {
1290 return nil;
1291 }
1292
1293 // Read the metadata from the image, which should contain hotspotX and
1294 // hotspotY properties.
1295 CGImageSourceRef cg_image = CGImageSourceCreateWithData((CFDataRef)image_data, nullptr);
1296 if (cg_image == NULL) {
1297 return nil;
1298 }
1299
1300 NSDictionary *image_props = (NSDictionary *)CGImageSourceCopyPropertiesAtIndex(cg_image, 0, nil);
1301 CFRelease(cg_image);
1302
1303 if (image_props == nil) {
1304 return nil;
1305 }
1306
1307 CGFloat hotspot_x = 0.0f;
1308 CGFloat hotspot_y = 0.0f;
1309 if (NSNumber *number = image_props[@"hotspotX"]) {
1310 hotspot_x = [number floatValue];
1311 }
1312 if (NSNumber *number = image_props[@"hotspotY"]) {
1313 hotspot_y = [number floatValue];
1314 }
1315 [image_props release];
1316
1317 NSImage *image = [[NSImage alloc] initWithData:image_data];
1318
1319 NSCursor *cursor = nil;
1320 if (image != nil) {
1321 // Apple recognizes that hotspots are usually specified from a .cur
1322 // file, whose origin is in the top-left, so there's no need to flip
1323 // it like most other Cocoa coordinates.
1324 cursor = [[NSCursor alloc] initWithImage:image
1325 hotSpot:NSMakePoint(hotspot_x, hotspot_y)];
1326 [image release];
1327 }
1328
1329 return cursor;
1330}

1593void CocoaGraphicsWindow::
1594handle_key_event(NSEvent *event) {
1595 NSUInteger modifierFlags = [event modifierFlags];
1596
1597 // NB. This is actually a on-off toggle, not up-down. Should we instead
1598 // rapidly fire two successive up-down events?
1599 handle_modifier(modifierFlags, NSAlphaShiftKeyMask, KeyboardButton::caps_lock());
1600
1601 // Check if any of the modifier keys have changed.
1602 handle_modifier(modifierFlags, NSShiftKeyMask, KeyboardButton::shift());
1603 handle_modifier(modifierFlags, NSControlKeyMask, KeyboardButton::control());
1604 handle_modifier(modifierFlags, NSAlternateKeyMask, KeyboardButton::alt());
1605 handle_modifier(modifierFlags, NSCommandKeyMask, KeyboardButton::meta());
1606
1607 // These are not documented, but they seem to be a reliable indicator of the
1608 // status of the leftright modifier keys.
1609 handle_modifier(modifierFlags, 0x0002, KeyboardButton::lshift());
1610 handle_modifier(modifierFlags, 0x0004, KeyboardButton::rshift());
1611 handle_modifier(modifierFlags, 0x0001, KeyboardButton::lcontrol());
1612 handle_modifier(modifierFlags, 0x2000, KeyboardButton::rcontrol());
1613 handle_modifier(modifierFlags, 0x0020, KeyboardButton::lalt());
1614 handle_modifier(modifierFlags, 0x0040, KeyboardButton::ralt());
1615 handle_modifier(modifierFlags, 0x0008, KeyboardButton::lmeta());
1616 handle_modifier(modifierFlags, 0x0010, KeyboardButton::rmeta());
1617
1618 _modifier_keys = modifierFlags;
1619
1620 // Get the raw button and send it.
1621 ButtonHandle raw_button = map_raw_key([event keyCode]);
1622 if (raw_button != ButtonHandle::none()) {
1623 // This is not perfect. Eventually, this whole thing should probably be
1624 // replaced with something that uses IOKit or so. In particular, the
1625 // flaws are: - OS eats unmodified F11, F12, scroll lock, pause - no up
1626 // events for caps lock - no robust way to distinguish updown for modkeys
1627 if ([event type] == NSKeyUp) {
1628 _input->raw_button_up(raw_button);
1629
1630 } else if ([event type] == NSFlagsChanged) {
1631 bool down = false;
1632 if (raw_button == KeyboardButton::lshift()) {
1633 down = (modifierFlags & 0x0002);
1634 } else if (raw_button == KeyboardButton::rshift()) {
1635 down = (modifierFlags & 0x0004);
1636 } else if (raw_button == KeyboardButton::lcontrol()) {
1637 down = (modifierFlags & 0x0001);
1638 } else if (raw_button == KeyboardButton::rcontrol()) {
1639 down = (modifierFlags & 0x2000);
1640 } else if (raw_button == KeyboardButton::lalt()) {
1641 down = (modifierFlags & 0x0020);
1642 } else if (raw_button == KeyboardButton::ralt()) {
1643 down = (modifierFlags & 0x0040);
1644 } else if (raw_button == KeyboardButton::lmeta()) {
1645 down = (modifierFlags & 0x0008);
1646 } else if (raw_button == KeyboardButton::rmeta()) {
1647 down = (modifierFlags & 0x0010);
1648 } else if (raw_button == KeyboardButton::caps_lock()) {
1649 // Emulate down-up, annoying hack!
1650 _input->raw_button_down(raw_button);
1651 }
1652 if (down) {
1653 _input->raw_button_down(raw_button);
1654 } else {
1655 _input->raw_button_up(raw_button);
1656 }
1657 } else if (![event isARepeat]) {
1658 _input->raw_button_down(raw_button);
1659 }
1660 }
1661
1662 // FlagsChanged events only carry modifier key information.
1663 if ([event type] == NSFlagsChanged) {
1664 return;
1665 }
1666
1667 TISInputSourceRef input_source = TISCopyCurrentKeyboardLayoutInputSource();
1668 CFDataRef layout_data = (CFDataRef)TISGetInputSourceProperty(input_source, kTISPropertyUnicodeKeyLayoutData);
1669 const UCKeyboardLayout *layout = (const UCKeyboardLayout *)CFDataGetBytePtr(layout_data);
1670
1671 if ([event type] == NSKeyDown) {
1672 // Translate it to a unicode character for keystrokes. I would use
1673 // interpretKeyEvents and insertText, but that doesn't handle dead keys.
1674 UInt32 modifier_state = (modifierFlags >> 16) & 0xFF;
1675 UniChar ustr[8];
1676 UniCharCount length;
1677
1678 UCKeyTranslate(layout, [event keyCode], kUCKeyActionDown, modifier_state,
1679 LMGetKbdType(), 0, &_dead_key_state, sizeof(ustr), &length, ustr);
1680
1681 for (int i = 0; i < length; ++i) {
1682 UniChar c = ustr[i];
1683 if (cocoadisplay_cat.is_spam()) {
1684 cocoadisplay_cat.spam()
1685 << "Handling keystroke, character " << (int)c;
1686 if (c < 128 && isprint(c)) {
1687 cocoadisplay_cat.spam(false) << " '" << (char)c << "'";
1688 }
1689 cocoadisplay_cat.spam(false) << "\n";
1690 }
1691 _input->keystroke(c);
1692 }
1693 }
1694
1695 // [NSEvent charactersIgnoringModifiers] doesn't ignore the shift key, so we
1696 // need to do what that method is doing manually.
1697 _dead_key_state = 0;
1698 UniChar c;
1699 UniCharCount length;
1700 UCKeyTranslate(layout, [event keyCode], kUCKeyActionDisplay,
1701 0, LMGetKbdType(), kUCKeyTranslateNoDeadKeysMask, &_dead_key_state,
1702 sizeof(c), &length, &c);
1703 CFRelease(input_source);
1704
1705 if (length != 1) {
1706 return;
1707 }
1708
1709 // If UCKeyTranslate could not map the key into a valid unicode character or
1710 // reserved symbol (See NSEvent.h), it returns 0x10 as translated character.
1711 // This happens e.g.e with the combination Fn+F1.. keys.
1712 // In that case, as fallback, we use charactersIgnoringModifiers to retrieve
1713 // the character without modifiers.
1714 if (c == 0x10) {
1715 NSString *str = [event charactersIgnoringModifiers];
1716 if (str == nil || [str length] != 1) {
1717 return;
1718 }
1719 c = [str characterAtIndex: 0];
1720 }
1721
1722 ButtonHandle button = map_key(c);
1723
1724 if (button == ButtonHandle::none()) {
1725 // That done, continue trying to find out the button handle.
1726 NSString *str = [[NSString alloc] initWithCharacters:&c length:length];
1727 if ([str canBeConvertedToEncoding: NSASCIIStringEncoding]) {
1728 // Nhm, ascii character perhaps?
1729 str = [str lowercaseString];
1730 const char *c_str = [str cStringUsingEncoding: NSASCIIStringEncoding];
1731 button = KeyboardButton::ascii_key(c_str[0]);
1732 }
1733 }
1734
1735 if (button == ButtonHandle::none()) {
1736 cocoadisplay_cat.warning()
1737 << "Unhandled keypress, character " << (int) c << ", keyCode " << [event keyCode]
1738 << ", type " << [event type] << ", flags " << [event modifierFlags] << "\n";
1739 return;
1740 }
1741
1742 if (cocoadisplay_cat.is_spam()) {
1743 cocoadisplay_cat.spam()
1744 << "Handled keypress, character " << (int) c << ", keyCode " << [event keyCode]
1745 << ", type " << [event type] << ", flags " << [event modifierFlags] << "\n";
1746 }
1747
1748 // Let's get it off our chest.
1749 if ([event type] == NSKeyUp) {
1750 _input->button_up(button);
1751 } else {
1752 _input->button_down(button);
1753 }
1754}

1797void CocoaGraphicsWindow::
1798handle_mouse_moved_event(bool in_window, double x, double y, bool absolute) {
1799 double nx, ny;
1800
1801 // Mouse position is received in screen points and not pixels, but in Panda3D
1802 // we want to have the coordinates expressed in pixels.
1803 // Instead of using convertPointFrom/toBackingStore and have complex logic to
1804 // cope with the change of coordinate system, we cheat and directly use the
1805 // contents scale of the view layer to convert screen point into pixels and
1806 // vice-versa.
1807 CGFloat contents_scale = _view.layer.contentsScale;
1808 if (absolute) {
1809 if (cocoadisplay_cat.is_spam()) {
1810 if (in_window != _input->get_pointer().get_in_window()) {
1811 if (in_window) {
1812 cocoadisplay_cat.spam() << "Mouse pointer entered window\n";
1813 } else {
1814 cocoadisplay_cat.spam() << "Mouse pointer exited window\n";
1815 }
1816 }
1817 }
1818
1819 nx = x * contents_scale;
1820 ny = y * contents_scale;
1821
1822 } else {
1823 // We received deltas, so add it to the current mouse position.
1824 PointerData md = _input->get_pointer();
1825 nx = md.get_x() + x * contents_scale;
1826 ny = md.get_y() + y * contents_scale;
1827 }
1828
1829 if (_properties.get_mouse_mode() == WindowProperties::M_confined
1830 && !in_window) {
1831 CGPoint point;
1832
1833 nx = std::max(0., std::min((double) get_x_size() - 1, nx));
1834 ny = std::max(0., std::min((double) get_y_size() - 1, ny));
1835
1836 // Convert back mouse position to screen space using point units
1837 if (_properties.get_fullscreen()) {
1838 point = CGPointMake(nx / contents_scale,
1839 ny / contents_scale);
1840 } else {
1841 point = CGPointMake((nx + _properties.get_x_origin()) / contents_scale,
1842 (ny + _properties.get_y_origin()) / contents_scale);
1843 }
1844
1845 if (CGWarpMouseCursorPosition(point) == kCGErrorSuccess) {
1846 //After moving (or warping) the mouse position, CG starts an event
1847 // suppression interval during which no more mouse events can occur
1848 // This interval can be interupted by the following call :
1849 CGAssociateMouseAndMouseCursorPosition(YES);
1850 in_window = true;
1851 } else {
1852 cocoadisplay_cat.warning() << "Failed to return mouse pointer to window\n";
1853 }
1854 }
1855
1856 if (in_window) {
1857 _input->set_pointer_in_window(nx, ny);
1858 } else {
1859 _input->set_pointer_out_of_window();
1860 }
1861
1862 if (in_window != _mouse_hidden && _properties.get_cursor_hidden()) {
1863 // Hide the cursor if the mouse enters the window, and unhide it when the
1864 // mouse leaves the window.
1865 if (in_window) {
1866 [NSCursor hide];
1867 } else {
1868 [NSCursor unhide];
1869 }
1870 _mouse_hidden = in_window;
1871 }

1909ButtonMap *CocoaGraphicsWindow::
1910get_keyboard_map() const {
1911 TISInputSourceRef input_source;
1912 CFDataRef layout_data;
1913 const UCKeyboardLayout *layout;
1914
1915 // Get the current keyboard layout data.
1916 input_source = TISCopyCurrentKeyboardLayoutInputSource();
1917 layout_data = (CFDataRef) TISGetInputSourceProperty(input_source, kTISPropertyUnicodeKeyLayoutData);
1918 layout = (const UCKeyboardLayout *)CFDataGetBytePtr(layout_data);
1919
1920 ButtonMap *map = new ButtonMap;
1921
1922 UniChar chars[4];
1923 UniCharCount num_chars;
1924
1925 // Iterate through the known scancode range and see what every scan code is
1926 // mapped to.
1927 for (int k = 0; k <= 0x7E; ++k) {
1928 ButtonHandle raw_button = map_raw_key(k);
1929 if (raw_button == ButtonHandle::none()) {
1930 continue;
1931 }
1932
1933 UInt32 dead_keys = 0;
1934 if (UCKeyTranslate(layout, k, kUCKeyActionDisplay, 0, LMGetKbdType(),
1935 kUCKeyTranslateNoDeadKeysMask, &dead_keys, 4,
1936 &num_chars, chars) == noErr) {
1937 if (num_chars > 0 && chars[0] != 0x10) {
1938 ButtonHandle button = ButtonHandle::none();
1939
1940 if (chars[0] > 0 && chars[0] <= 0x7f) {
1941 button = KeyboardButton::ascii_key(chars[0]);
1942 }
1943 if (button == ButtonHandle::none()) {
1944 button = map_key(chars[0]);
1945 }
1946 if (button != ButtonHandle::none()) {
1947 map->map_button(raw_button, button);
1948 }
1949 } else {
1950 // A special function key or modifier key, which isn't remapped by the
1951 // OS.
1952 map->map_button(raw_button, raw_button);
1953 }
1954 }
1955 }
1956
1957 CFRelease(input_source);
1958 return map;
1959}
