268on_click_label(int collector_index) {
269 if (collector_index < 0) {
270 // Clicking on whitespace in the graph is the same as clicking on the top
271 // label.
272 collector_index = get_collector_index();
273 }
274
275 if (collector_index == get_collector_index()) {
276 // Clicking on the top label means to go up to the parent level.
277 if (collector_index != 0) {
278 const PStatClientData *client_data =
279 MacStatsGraph::_monitor->get_client_data();
280 if (client_data->has_collector(collector_index)) {
281 const PStatCollectorDef &def =
282 client_data->get_collector_def(collector_index);
283 if (def._parent_index == 0 && get_view().get_show_level()) {
284 // Unless the parent is "Frame", and we're not a time collector.
285 }
286 else if (def._parent_index != get_collector_index()) {
287 // If we were previously at the parent, pop it from the stack.
288 if (!_back_stack.empty() && _back_stack.back() == def._parent_index) {
289 _back_stack.pop_back();
290 if (_back_stack.empty()) {
291 _graph_view_controller.backToolbarItemVisible = NO;
292 }
293 } else {
294 if (_back_stack.empty()) {
295 _graph_view_controller.backToolbarItemVisible = YES;
296 }
297 _back_stack.push_back(get_collector_index());
298 }
299
300 set_collector_index(def._parent_index);
301 }
302 }
303 }
304 }
305 else {
306 if (_back_stack.empty()) {
307 _graph_view_controller.backToolbarItemVisible = YES;
308 }
309 _back_stack.push_back(get_collector_index());

451void MacStatsStripChart::
452draw_slice(int x, int w, const PStatStripChart::FrameData &fdata) {
453 if (!_ctx) {
454 return;
455 }
456
457 // Start by clearing the band first.
458 CGContextSetFillColorWithColor(_ctx, _background_color);
459 CGContextFillRect(_ctx, CGRectMake(x, 0, w, get_ysize()));
460
461 double overall_time = 0.0;
462 int y = get_ysize();
463
464 FrameData::const_iterator fi;
465 for (fi = fdata.begin(); fi != fdata.end(); ++fi) {
466 const ColorData &cd = (*fi);
467 overall_time += cd._net_value;
468
469 bool is_highlighted = cd._collector_index == _highlighted_index;
470 CGContextSetFillColorWithColor(_ctx,
471 MacStatsGraph::_monitor->get_collector_color(cd._collector_index, is_highlighted));
472
473 if (overall_time > get_vertical_scale()) {
474 // Off the top. Go ahead and clamp it by hand, in case it's so far off
475 // the top we'd overflow the 16-bit pixel value.
476 CGContextFillRect(_ctx, CGRectMake(x, 0, w, y));
477 // And we can consider ourselves done now.
478 return;
479 }
480
481 int top_y = height_to_pixel(overall_time);
482 CGContextFillRect(_ctx, CGRectMake(x, top_y, w, y - top_y));
483 y = top_y;
484 }
485}

668void MacStatsStripChart::
669handle_motion(int graph_x, int graph_y) {
670 if (_drag_mode == DM_none && _potential_drag_mode == DM_none &&
671 graph_x >= 0 && graph_y >= 0 && graph_x < get_xsize() && graph_y < get_ysize()) {
672 // When the mouse is over a color bar, highlight it.
673 int collector_index = get_collector_under_pixel(graph_x, graph_y);
674 _label_stack.highlight_label(collector_index);
675 on_enter_label(collector_index);
676 }
677 else {
678 // If the mouse is in some drag mode, stop highlighting.
679 _label_stack.highlight_label(-1);
680 on_leave_label(_highlighted_index);
681 }
682
683 if (_drag_mode == DM_scale) {
684 double ratio = 1.0 - ((double)graph_y / (double)get_ysize());
685 if (ratio > 0.0) {
686 double new_scale = _drag_scale_start / ratio;
687 if (!IS_NEARLY_EQUAL(get_vertical_scale(), new_scale)) {
688 // Disable smoothing while we do this expensive operation.
689 set_vertical_scale(_drag_scale_start / ratio);
690 }
691 }
692 return;
693 }
694 else if (_drag_mode == DM_new_guide_bar) {
695 // We haven't created the new guide bar yet; we won't until the mouse
696 // comes within the graph's region.
697 if (graph_y >= 0 && graph_y < get_ysize()) {
698 set_drag_mode(DM_guide_bar);
699 _drag_guide_bar = add_user_guide_bar(pixel_to_height(graph_y));
700 return;
701 }
702 }
703 else if (_drag_mode == DM_guide_bar) {
704 move_user_guide_bar(_drag_guide_bar, pixel_to_height(graph_y));
705 return;
706 }
707
708 MacStatsGraph::handle_motion(graph_x, graph_y);
709}

839int MacStatsStripChart::
840draw_guide_label(CGContextRef ctx, const PStatGraph::GuideBar &bar, int last_y) {
841 NSColor *color;
842 switch (bar._style) {
843 case GBS_target:
844 color = [NSColor secondaryLabelColor];
845 break;
846
847 case GBS_user:
848 color = [NSColor tertiaryLabelColor];
849 break;
850
851 default:
852 color = [NSColor labelColor];
853 break;
854 }
855
856 int y = height_to_pixel(bar._height);
857 const std::string &label = bar._label;
858
859 const CFStringRef keys[] = {
860 (__bridge CFStringRef)NSForegroundColorAttributeName,
861 (__bridge CFStringRef)NSFontAttributeName,
862 };
863 const void *values[] = {
864 color,
865 [NSFont systemFontOfSize:0.0],
866 };
867 CFDictionaryRef attribs = CFDictionaryCreate(kCFAllocatorDefault, (const void **)keys, (const void **)values, 2, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
868
869 CFStringRef str = CFStringCreateWithCString(kCFAllocatorDefault, label.c_str(), kCFStringEncodingUTF8);
870 CFAttributedStringRef astr = CFAttributedStringCreate(kCFAllocatorDefault, str, attribs);
871 CFRelease(attribs);
872 CFRelease(str);
873
874 CTLineRef line = CTLineCreateWithAttributedString(astr);
875 CFRelease(astr);
876 CGRect bounds = CTLineGetImageBounds(line, ctx);
877 int height = bounds.size.height;
878
879 if (bar._style != GBS_user) {
880 double from_height = pixel_to_height(y + height);
881 double to_height = pixel_to_height(y - height);
882 if (find_user_guide_bar(from_height, to_height) >= 0) {
883 // Omit the label: there's a user-defined guide bar in the same space.
884 CFRelease(line);
885 return last_y;
886 }
887 }
888
889 if (y > height && y < get_ysize() - height) {
890 // Now convert our y to a coordinate within our drawing area.
891
892 int this_y = y - height / 2;
893 if (last_y < this_y || last_y > this_y + height) {
894 CGContextSetTextPosition(ctx, 4, get_ysize() - this_y - height);
895 CTLineDraw(line, ctx);
896
897 last_y = this_y;
898 }
899 }
900
901 CFRelease(line);
902 return last_y;
903}
