67void EventHandler::
68dispatch_event(const Event *event) {
69 nassertv(event != nullptr);
70
71 // Is the event name defined in the hook table? It will be if anyone has
72 // ever assigned a hook to this particular event name.
73 Hooks::const_iterator hi;
74 hi = _hooks.find(event->get_name());
75
76 if (hi != _hooks.end()) {
77 // Yes, it is! Now walk through all the functions assigned to that event
78 // name.
79 Functions copy_functions = (*hi).second;
80
81 Functions::const_iterator fi;
82 for (fi = copy_functions.begin(); fi != copy_functions.end(); ++fi) {
83 if (event_cat.is_spam()) {
84 event_cat->spam()
85 << "calling callback 0x" << (void*)(*fi)
86 << " for event '" << event->get_name() << "'"
87 << std::endl;
88 }
89 (*fi)(event);
90 }
91 }
92
93 // now for callback hooks
94 CallbackHooks::const_iterator chi;
95 chi = _cbhooks.find(event->get_name());
96
97 if (chi != _cbhooks.end()) {
98 // found one
99 CallbackFunctions copy_functions = (*chi).second;
100
101 CallbackFunctions::const_iterator cfi;
102 for (cfi = copy_functions.begin(); cfi != copy_functions.end(); ++cfi) {
103 ((*cfi).first)(event, (*cfi).second);
104 }
105 }
106
107 // now for lambda hooks
108 LambdaHooks::const_iterator lhi;
109 lhi = _lambdahooks.find(event->get_name());
110
111 if (lhi != _lambdahooks.end()) {
112 // found one
113 const LambdaFunctions &functions = (*lhi).second;
114
115 size_t num_functions = functions.size();
116 for (size_t i = 0; i < num_functions; ++i) {
117 functions[i](event);
118 }
119 }
120
121 // Finally, check for futures that need to be triggered.
122 Futures::iterator fi;
123 fi = _futures.find(event->get_name());
124
125 if (fi != _futures.end()) {
126 AsyncFuture *fut = (*fi).second;
127 if (!fut->done()) {
128 fut->set_result((TypedReferenceCount *)event);
129 }
130 _futures.erase(fi);
131 }
132}

138void EventHandler::
139write(std::ostream &out) const {
140 Hooks::const_iterator hi;
141 hi = _hooks.begin();
142
143 CallbackHooks::const_iterator chi;
144 chi = _cbhooks.begin();
145
146 while (hi != _hooks.end() && chi != _cbhooks.end()) {
147 if ((*hi).first < (*chi).first) {

359bool EventHandler::
360remove_hooks_with(void *data) {
361 bool any_removed = false;
362
363 CallbackHooks::iterator chi;
364 for (chi = _cbhooks.begin(); chi != _cbhooks.end(); ++chi) {
365 CallbackFunctions &funcs = (*chi).second;
366 CallbackFunctions::iterator cfi;
367
368 CallbackFunctions new_funcs;
369 for (cfi = funcs.begin(); cfi != funcs.end(); ++cfi) {
370 if ((*cfi).second == data) {
371 any_removed = true;
372 } else {
373 new_funcs.insert(*cfi);
374 }
375 }
376 funcs.swap(new_funcs);
377 }
378
379 return any_removed;
380}
