84bool BamReader::
85init() {
86 nassertr(_source != nullptr, false);
87 nassertr(_needs_init, false);
88 _needs_init = false;
89 Datagram header;
90
91 if (_source->is_error()) {
92 return false;
93 }
94
95 if (!get_datagram(header)) {
96 bam_cat.error()
97 << "Unable to read Bam header.\n";
98 return false;
99 }
100
101 DatagramIterator scan(header);
102
103 _file_major = scan.get_uint16();
104 _file_minor = scan.get_uint16();
105
106 // If the major version is different, or the minor version is *newer*, we
107 // can't safely load the file.
108 if (_file_major != _bam_major_ver ||
109 _file_minor < _bam_first_minor_ver ||
110 _file_minor > _bam_last_minor_ver) {
111 bam_cat.error()
112 << "Bam file is version " << _file_major << "." << _file_minor
113 << ".\n";
114
115 if (_bam_minor_ver == _bam_first_minor_ver) {
116 bam_cat.error()
117 << "This program can only load version "
118 << _bam_major_ver << "." << _bam_first_minor_ver << " bams.\n";
119 } else {
120 bam_cat.error()
121 << "This program can only load version "
122 << _bam_major_ver << "." << _bam_first_minor_ver << " through "
123 << _bam_major_ver << "." << _bam_last_minor_ver << " bams.\n";
124 }
125
126 return false;
127 }
128
129 _file_endian = (BamEndian)scan.get_uint8();
130
131 _file_stdfloat_double = false;
132 if (_file_minor >= 27) {
133 _file_stdfloat_double = scan.get_bool();
134 }
135
136 if (scan.get_current_index() > header.get_length()) {
137 bam_cat.error()
138 << "Bam header is too short.\n";
139 return false;
140 }
141

244bool BamReader::
245read_object(TypedWritable *&ptr, ReferenceCount *&ref_ptr) {
246 ptr = nullptr;
247 ref_ptr = nullptr;
248 nassertr(_num_extra_objects == 0, false);
249
250 int start_level = _nesting_level;
251
252 // First, read the base object.
253 int object_id = p_read_object();
254
255 // Now that object might have included some pointers to other objects, which
256 // may still need to be read. And those objects might in turn require
257 // reading additional objects. Read all the remaining objects.
258
259 // Prior to 6.21, we kept track of _num_extra_objects to know when we're
260 // done.
261 while (_num_extra_objects > 0) {
262 p_read_object();
263 _num_extra_objects--;
264 }
265
266 // Beginning with 6.21, we use explicit nesting commands to know when we're
267 // done.
268 while (_nesting_level > start_level) {
269 p_read_object();
270 }
271
272 // Now look up the pointer of the object we read first. It should be
273 // available now.
274 if (object_id == 0) {
275 if (bam_cat.is_spam()) {
276 bam_cat.spam()
277 << "Returning false\n";
278 }
279 return false;
280 }
281
282 CreatedObjs::iterator oi = _created_objs.find(object_id);
283
284 if (oi == _created_objs.end()) {
285 bam_cat.error()
286 << "Undefined object encountered!\n";
287 return false;
288
289 } else {
290 CreatedObj &created_obj = (*oi).second;
291 ptr = created_obj._ptr;
292 ref_ptr = created_obj._ref_ptr;
293
294 if (bam_cat.is_spam()) {
295 if (ptr != nullptr) {
296 bam_cat.spam()
297 << "Returning object of type " << ptr->get_type() << "\n";
298 }
299 }
300 if (created_obj._change_this != nullptr ||
301 created_obj._change_this_ref != nullptr) {
302 bam_cat.warning()
303 << "Returning pointer to " << ptr->get_type()
304 << " that might change.\n";
305 }
306
307 return true;
308 }

324bool BamReader::
325resolve() {
326 bool all_completed;
327 bool any_completed_this_pass;
328
329 do {
330 if (bam_cat.is_spam()) {
331 bam_cat.spam()
332 << "resolve pass begin\n";
333 }
334 all_completed = true;
335 any_completed_this_pass = false;
336
337 ObjectPointers::iterator oi;
338 oi = _object_pointers.begin();
339 while (oi != _object_pointers.end()) {
340 int object_id = (*oi).first;
341 PointerReference &pref = (*oi).second;
342
343 CreatedObjs::iterator ci = _created_objs.find(object_id);
344 nassertr(ci != _created_objs.end(), false);
345
346 CreatedObj &created_obj = (*ci).second;
347
348 TypedWritable *object_ptr = created_obj._ptr;
349
350 // Update _now_creating, so a call to get_int_tag() from within
351 // complete_pointers() will come to the right place.
352 CreatedObjs::iterator was_creating = _now_creating;
353 _now_creating = ci;
354
355 if (resolve_object_pointers(object_ptr, pref)) {
356 // Now remove this object from the list of things that need
357 // completion. We have to be a bit careful when deleting things from
358 // the STL container while we are traversing it.
359 ObjectPointers::iterator old = oi;
360 ++oi;
361 _object_pointers.erase(old);
362 any_completed_this_pass = true;
363
364 // Does the pointer need to change?
365 if (created_obj._change_this_ref != nullptr) {
366 // Reference-counting variant.
367 TypedWritableReferenceCount *object_ref_ptr = (TypedWritableReferenceCount *)object_ptr;
368 nassertr(created_obj._ref_ptr == nullptr || created_obj._ref_ptr == object_ref_ptr, false);
369 PT(TypedWritableReferenceCount) new_ptr = created_obj._change_this_ref(object_ref_ptr, this);
370 if (new_ptr != object_ref_ptr) {
371 // Also update the reverse
372 vector_int &old_refs = _created_objs_by_pointer[object_ptr];
373 vector_int &new_refs = _created_objs_by_pointer[new_ptr];
374 for (vector_int::const_iterator oi = old_refs.begin();
375 oi != old_refs.end();
376 ++oi) {
377 new_refs.push_back(*oi);
378 }
379 _created_objs_by_pointer.erase(object_ptr);
380
381 // Remove the pointer from the finalize list (the new pointer
382 // presumably doesn't require finalizing).
383 _finalize_list.erase(object_ptr);
384 }
385 created_obj.set_ptr(new_ptr, new_ptr);
386 created_obj._change_this = nullptr;
387 created_obj._change_this_ref = nullptr;
388
389 } else if (created_obj._change_this != nullptr) {
390 // Non-reference-counting variant.
391 TypedWritable *new_ptr = created_obj._change_this(object_ptr, this);
392 if (new_ptr != object_ptr) {
393 // Also update the reverse
394 vector_int &old_refs = _created_objs_by_pointer[object_ptr];
395 vector_int &new_refs = _created_objs_by_pointer[new_ptr];
396 for (vector_int::const_iterator oi = old_refs.begin();
397 oi != old_refs.end();
398 ++oi) {
399 new_refs.push_back(*oi);
400 }
401 _created_objs_by_pointer.erase(object_ptr);
402
403 // Remove the pointer from the finalize list (the new pointer
404 // presumably doesn't require finalizing).
405 _finalize_list.erase(object_ptr);
406 }
407 created_obj.set_ptr(new_ptr, new_ptr->as_reference_count());
408 created_obj._change_this = nullptr;
409 created_obj._change_this_ref = nullptr;
410 }
411
412 } else {
413 // Couldn't complete this object yet; it'll wait for next time.
414 ++oi;
415 all_completed = false;
416 }
417
418 _now_creating = was_creating;
419 }
420
421 if (bam_cat.is_spam()) {
422 bam_cat.spam()
423 << "resolve pass end: all_completed = " << all_completed
424 << " any_completed_this_pass = " << any_completed_this_pass
425 << "\n";
426 }
427 } while (!all_completed && any_completed_this_pass);
428
429 if (all_completed) {
430 finalize();
431 } else {
432 // Report all the uncompleted objects for no good reason. This will
433 // probably have to come out later when we have cases in which some
434 // objects might legitimately be uncompleted after calling resolve(), but
435 // for now we expect resolve() to always succeed.
436 ObjectPointers::const_iterator oi;
437 for (oi = _object_pointers.begin();
438 oi != _object_pointers.end();
439 ++oi) {
440 int object_id = (*oi).first;
441 CreatedObjs::iterator ci = _created_objs.find(object_id);
442 nassertr(ci != _created_objs.end(), false);
443 CreatedObj &created_obj = (*ci).second;
444 TypedWritable *object_ptr = created_obj._ptr;
445
446 bam_cat.warning()
447 << "Unable to complete " << object_ptr->get_type() << "\n";
448 }
449 }
450
451 return all_completed;
452}

463bool BamReader::
464change_pointer(const TypedWritable *orig_pointer, const TypedWritable *new_pointer) {
465 if (orig_pointer == new_pointer) {
466 return false;
467 }
468 CreatedObjsByPointer::iterator ci = _created_objs_by_pointer.find(orig_pointer);
469 if (ci == _created_objs_by_pointer.end()) {
470 // No record of this object.
471 return false;
472 }
473
474 if (bam_cat.is_spam()) {
475 bam_cat.spam()
476 << "change_pointer(" << (void *)orig_pointer << ", "
477 << (void *)new_pointer << ") (" << new_pointer->get_type() << ")\n";
478 }
479
480 const vector_int &old_refs = (*ci).second;
481 vector_int &new_refs = _created_objs_by_pointer[new_pointer];
482
483 for (vector_int::const_iterator oi = old_refs.begin();
484 oi != old_refs.end();
485 ++oi) {
486 int object_id = (*oi);
487
488 CreatedObjs::iterator ci = _created_objs.find(object_id);
489 nassertr(ci != _created_objs.end(), false);
490 nassertr((*ci).second._ptr == orig_pointer, false);
491
492 TypedWritable *ptr = (TypedWritable *)new_pointer;
493 (*ci).second.set_ptr(ptr, ptr->as_reference_count());
494 new_refs.push_back(object_id);
495 }
496
497 _created_objs_by_pointer.erase(ci);
498
499 // Also change the pointer on the finalize_list.
500 Finalize::iterator fi = _finalize_list.find((TypedWritable *)orig_pointer);
501 if (fi != _finalize_list.end()) {
502 _finalize_list.insert((TypedWritable *)new_pointer);
503 _finalize_list.erase(fi);
504 }
505
506 return true;
507}

513TypeHandle BamReader::
514read_handle(DatagramIterator &scan) {
515 // We encode TypeHandles within the Bam file by writing a unique index
516 // number for each one to the file. When we write a particular TypeHandle
517 // for the first time, we assign it a new index number and then immediately
518 // follow it by its definition; when we write the same TypeHandle on
519 // subsequent times we only write the index number.
520
521 // Thus, to read a TypeHandle, we first read the index number. If it is a
522 // number we have not yet encountered, we must then read the definition.
523
524 // Here's the index number.
525 int id = scan.get_uint16();
526
527 if (id == 0) {
528 // Index number 0 is always, by convention, TypeHandle::none().
529 return TypeHandle::none();
530 }
531
532 IndexMap::const_iterator mi = _index_map.find(id);
533 if (mi != _index_map.end()) {
534 // We've encountered this index number before, so there should be no type
535 // definition following the id. Simply return the TypeHandle we
536 // previously associated with the id.
537 TypeHandle type = (*mi).second;
538 return type;
539 }
540
541 // We haven't encountered this index number before. This means it will be
542 // immediately followed by the type definition. This consists of the string
543 // name, followed by the list of parent TypeHandles for this type.
544
545 string name = scan.get_string();
546 bool new_type = false;
547
548 TypeRegistry *type_registry = TypeRegistry::ptr();
549 TypeHandle type = type_registry->find_type(name);
550 if (type == TypeHandle::none()) {
551 // We've never heard of this type before! This is really an error
552 // condition, but we'll do the best we can and declare it on-the-fly.
553
554 type = type_registry->register_dynamic_type(name);
555 bam_cat.warning()
556 << "Bam file '" << get_filename() << "' contains objects of unknown type: "
557 << type << "\n";
558 new_type = true;
559 _new_types.insert(type);
560 }
561
562 // Now pick up the derivation information.
563 int num_parent_classes = scan.get_uint8();
564 for (int i = 0; i < num_parent_classes; i++) {
565 TypeHandle parent_type = read_handle(scan);
566 if (new_type) {
567 type_registry->record_derivation(type, parent_type);
568 } else {
569 if (type.get_parent_towards(parent_type) != parent_type) {
570 if (bam_cat.is_debug()) {
571 bam_cat.debug()
572 << "Bam file indicates a derivation of " << type
573 << " from " << parent_type << " which is no longer true.\n";
574 }
575 }
576 }
577 }
578
579 bool inserted = _index_map.insert(IndexMap::value_type(id, type)).second;
580 nassertr(inserted, type);
581
582 if (bam_cat.is_spam()) {
583 bam_cat.spam()
584 << "Read TypeHandle for " << type << ".\n";
585 }
586
587 return type;
588}

610bool BamReader::
611read_pointer(DatagramIterator &scan) {
612 Thread::consider_yield();
613
614 nassertr(_now_creating != _created_objs.end(), false);
615 int requestor_id = (*_now_creating).first;
616
617 // Read the object ID, and associate it with the requesting object.
618 int object_id = read_object_id(scan);
619
620 PointerReference &pref = _object_pointers[requestor_id];
621 if (_reading_cycler == nullptr) {
622 // This is not being read within a read_cdata() call.
623 pref._objects.push_back(object_id);
624 } else {
625 // This *is* being read within a read_cdata() call.
626 pref._cycler_pointers[_reading_cycler].push_back(object_id);
627 }
628
629 // If the object ID is zero (which indicates a NULL pointer), we don't have
630 // to do anything else.
631 if (object_id != 0) {
632 /*
633 CreatedObj new_created_obj;
634 _created_objs.insert(CreatedObjs::value_type(object_id, new_created_obj)).second;
635 */
636
637 if (get_file_minor_ver() < 21) {
638 // Prior to bam version 6.21, we expect to read an adjunct object for
639 // each non-NULL pointer we read.
640 _num_extra_objects++;
641 }
642
643 return true;
644 } else {
645 return false;
646 }
647}

931void *BamReader::
932get_pta(DatagramIterator &scan) {
933 nassertr(_pta_id == -1, nullptr);
934 int id = read_pta_id(scan);
935
936 if (id == 0) {
937 // As always, a 0 ID indicates a NULL pointer. The caller will not be
938 // able to differentiate this case from that of a previously-read pointer,
939 // but that's OK because the next data in the Bam file is the length of
940 // the array, which will be zero--indicating an empty or NULL array.
941 return nullptr;
942 }
943
944 PTAMap::iterator pi = _pta_map.find(id);
945 if (pi == _pta_map.end()) {
946 // This is the first time we've encountered this particular ID, meaning we
947 // need to read the data now and register it.
948 _pta_id = id;
949 return nullptr;
950 }
951
952 return (*pi).second;
953}

964void BamReader::
965register_pta(void *ptr) {
966 if (_pta_id != -1) {

1053int BamReader::
1054p_read_object() {
1055 Datagram dg;
1056
1057 // First, read a datagram for the object.
1058 if (!get_datagram(dg)) {
1059 // When we run out of datagrams, we're at the end of the file.
1060 if (bam_cat.is_debug()) {
1061 bam_cat.debug()
1062 << "Reached end of bam source.\n";
1063 }
1064 return 0;
1065 }
1066
1067 // Now extract the object definition from the datagram.
1068 DatagramIterator scan(dg);
1069
1070 // First, read the BamObjectCode. In bam versions prior to 6.21, there was
1071 // no BamObjectCode in the stream.
1072 BamObjectCode boc = BOC_adjunct;
1073 if (get_file_minor_ver() >= 21) {
1074 boc = (BamObjectCode)scan.get_uint8();
1075 }
1076 switch (boc) {
1077 case BOC_push:
1078 ++_nesting_level;
1079 break;
1080
1081 case BOC_pop:
1082 --_nesting_level;
1083 return 0;
1084
1085 case BOC_adjunct:
1086 break;
1087
1088 case BOC_remove:
1089 // The BOC_remove code is a special case; it begins a record that simply
1090 // lists all of the object ID's that are no longer important to the file
1091 // and may be released.
1092 free_object_ids(scan);
1093
1094 // Now that we've freed all of the object id's indicate, read the next
1095 // object id in the stream. It's easiest to do this by calling
1096 // recursively.
1097 return p_read_object();
1098
1099 case BOC_file_data:
1100 // Another special case. This marks an auxiliary file data record that we
1101 // skip over for now, but we note its position within the stream, so that
1102 // we can hand it to a future object who may request it.
1103 {
1104 SubfileInfo info;
1105 if (!_source->save_datagram(info)) {
1106 bam_cat.error()
1107 << "Failed to read file data.\n";
1108 return 0;
1109 }
1110 _file_data_records.push_back(info);
1111 }
1112
1113 return p_read_object();
1114
1115 default:
1116 bam_cat.error()
1117 << "Encountered invalid BamObjectCode 0x" << std::hex << (int)boc << std::dec << ".\n";
1118 return 0;
1119 }
1120
1121 // An object definition in a Bam file consists of a TypeHandle definition,
1122 // defining the object's type, followed by an object ID index, defining the
1123 // particular instance (e.g. pointer) of this object.
1124
1125 TypeHandle type = read_handle(scan);
1126
1127 int object_id = read_object_id(scan);
1128
1129 if (scan.get_current_index() > dg.get_length()) {
1130 bam_cat.error()
1131 << "Found truncated datagram in bam stream\n";
1132 return 0;
1133 }
1134
1135 // There are two cases (not counting the special _remove_flag case, above).
1136 // Either this is a new object definition, or this is a reference to an
1137 // object that was previously defined.
1138
1139 // We use the TypeHandle to differentiate these two cases. By convention,
1140 // we write a TypeHandle::none() to the Bam file when we are writing a
1141 // reference to a previously-defined object, but we write the object's
1142 // actual type when we are writing its definition right now.
1143
1144 // Therefore, if the type is TypeHandle::none(), then we must have already
1145 // read in and created the object (although its pointers may not be fully
1146 // instantiated yet). On the other hand, if the type is anything else, then
1147 // we must read the definition to follow.
1148
1149 if (type != TypeHandle::none()) {
1150 // Now we are going to read and create a new object.
1151
1152 // First, we must add an entry into the map for this object ID, so that in
1153 // case this function is called recursively during the object's factory
1154 // constructor, we will have some definition for the object. For now, we
1155 // give it a NULL pointer.
1156 CreatedObj new_created_obj;
1157 CreatedObjs::iterator oi =
1158 _created_objs.insert(CreatedObjs::value_type(object_id, new_created_obj)).first;
1159 CreatedObj &created_obj = (*oi).second;
1160
1161 if (created_obj._ptr != nullptr) {
1162 // This object had already existed; thus, we are just receiving an
1163 // update for it.
1164
1165 if (_object_pointers.find(object_id) != _object_pointers.end()) {
1166 // Aieee! This object isn't even complete from the last time we
1167 // encountered it in the stream! This should never happen. Something's
1168 // corrupt or the stream was maliciously crafted.
1169 bam_cat.error()
1170 << "Found object " << object_id << " in bam stream again while "
1171 << "trying to resolve its own pointers.\n";
1172 return 0;
1173 }
1174
1175 // Update _now_creating during this call so if this function calls
1176 // read_pointer() or register_change_this() we'll match it up properly.
1177 // This might recursively call back into this p_read_object(), so be
1178 // sure to save and restore the original value of _now_creating.
1179 CreatedObjs::iterator was_creating = _now_creating;
1180 _now_creating = oi;
1181 created_obj._ptr->fillin(scan, this);
1182 _now_creating = was_creating;
1183
1184 if (scan.get_remaining_size() > 0) {
1185 bam_cat.warning()
1186 << "Skipping " << scan.get_remaining_size() << " remaining bytes "
1187 << "in datagram containing type " << type << "\n";
1188 }
1189
1190 } else {
1191 // We are receiving a new object. Now we can call the factory to create
1192 // the object.
1193
1194 // Define the parameters for passing to the object factory.
1195 FactoryParams fparams;
1196 fparams.add_param(new BamReaderParam(scan, this));
1197
1198 // As above, we update and preserve _now_creating during this call.
1199 CreatedObjs::iterator was_creating = _now_creating;
1200 _now_creating = oi;
1201 TypedWritable *object =
1202 _factory->make_instance_more_general(type, fparams);
1203 _now_creating = was_creating;
1204
1205 // And now we can store the new object pointer in the map.
1206 nassertr(created_obj._ptr == object || created_obj._ptr == nullptr, object_id);
1207 if (object == nullptr) {
1208 created_obj.set_ptr(nullptr, nullptr);
1209 } else {
1210 created_obj.set_ptr(object, object->as_reference_count());
1211 }
1212 created_obj._created = true;
1213
1214 if (created_obj._change_this_ref != nullptr) {
1215 // If the pointer is scheduled to change after complete_pointers(),
1216 // but we have no entry in _object_pointers for this object (and hence
1217 // no plan to call complete_pointers()), then just change the pointer
1218 // immediately.
1219 ObjectPointers::const_iterator ri = _object_pointers.find(object_id);
1220 if (ri == _object_pointers.end()) {
1221 PT(TypedWritableReferenceCount) object_ref = (*created_obj._change_this_ref)((TypedWritableReferenceCount *)object, this);
1222 TypedWritable *new_ptr = object_ref;
1223 created_obj.set_ptr(object_ref, object_ref);
1224 created_obj._change_this = nullptr;
1225 created_obj._change_this_ref = nullptr;
1226
1227 // Remove the pointer from the finalize list (the new pointer
1228 // presumably doesn't require finalizing).
1229 if (new_ptr != object) {
1230 _finalize_list.erase(object);
1231 }
1232 object = new_ptr;
1233 }
1234
1235 } else if (created_obj._change_this != nullptr) {
1236 // Non-reference-counting variant.
1237 ObjectPointers::const_iterator ri = _object_pointers.find(object_id);
1238 if (ri == _object_pointers.end()) {
1239 TypedWritable *new_ptr = (*created_obj._change_this)(object, this);
1240 created_obj.set_ptr(new_ptr, new_ptr->as_reference_count());
1241 created_obj._change_this = nullptr;
1242 created_obj._change_this_ref = nullptr;
1243
1244 if (new_ptr != object) {
1245 _finalize_list.erase(object);
1246 }
1247 object = new_ptr;
1248 }
1249 }
1250
1251 _created_objs_by_pointer[created_obj._ptr].push_back(object_id);
1252
1253 // Just some sanity checks
1254 if (object == nullptr) {
1255 if (bam_cat.is_debug()) {
1256 bam_cat.debug()
1257 << "Unable to create an object of type " << type << std::endl;
1258 }
1259
1260 } else if (object->get_type() != type) {
1261 if (_new_types.find(type) != _new_types.end()) {
1262 // This was a type we hadn't heard of before, so it's not really
1263 // surprising we didn't know how to create it. Suppress the warning
1264 // (make it a debug statement instead).
1265 if (bam_cat.is_debug()) {
1266 bam_cat.warning()
1267 << "Attempted to create a " << type.get_name() \
1268 << " but a " << object->get_type() \
1269 << " was created instead." << std::endl;
1270 }
1271
1272 } else {
1273 // This was a normal type that we should have known how to create.
1274 // Report the error.
1275 bam_cat.warning()
1276 << "Attempted to create a " << type.get_name() \
1277 << " but a " << object->get_type() \
1278 << " was created instead." << std::endl;
1279 }
1280
1281 } else {
1282 if (bam_cat.is_spam()) {
1283 bam_cat.spam()
1284 << "Read a " << object->get_type() << ": " << (void *)object
1285 << " (id=" << object_id << ")\n";
1286 }
1287 }
1288 }
1289
1290 // Sanity check that we read the expected number of bytes.
1291 if (scan.get_current_index() > dg.get_length()) {
1292 bam_cat.error()
1293 << "End of datagram reached while reading bam object "
1294 << type << ": " << (void *)created_obj._ptr << "\n";
1295 }
1296 }
1297
1298 return object_id;
1299}

1306bool BamReader::
1307resolve_object_pointers(TypedWritable *object,
1308 BamReader::PointerReference &pref) {
1309 // Some objects further require all of their nested objects to have been
1310 // completed (i.e. complete_pointers has been called on each nested object)
1311 // before they can themselves be completed.
1312 bool require_fully_complete = object->require_fully_complete();
1313
1314 // First do the PipelineCycler objects.
1315 CyclerPointers::iterator ci;
1316 ci = pref._cycler_pointers.begin();
1317 while (ci != pref._cycler_pointers.end()) {
1318 PipelineCyclerBase *cycler = (*ci).first;
1319 const vector_int &pointer_ids = (*ci).second;
1320
1321 if (resolve_cycler_pointers(cycler, pointer_ids, require_fully_complete)) {
1322 // Now remove this cycler from the list of things that need completion.
1323 // We have to be a bit careful when deleting things from the STL
1324 // container while we are traversing it.
1325 CyclerPointers::iterator old = ci;
1326 ++ci;
1327 pref._cycler_pointers.erase(old);
1328
1329 } else {
1330 // Couldn't complete this cycler yet; it'll wait for next time.
1331 ++ci;
1332 }
1333 }
1334
1335 if (!pref._cycler_pointers.empty()) {
1336 // If we didn't get all the cyclers, we have to wait.
1337 if (bam_cat.is_spam()) {
1338 bam_cat.spam()
1339 << "some cyclers pending: complete_pointers for " << (void *)object
1340 << " (" << object->get_type() << ")\n";
1341 }
1342 return false;
1343 }
1344
1345 // Now make sure we have all of the pointers this object is waiting for. If
1346 // any of the pointers has not yet been read in, we can't resolve this
1347 // object--we can't do anything for a given object until we have *all*
1348 // outstanding pointers for that object.
1349 bool is_complete = true;
1350
1351 vector_typedWritable references;
1352 references.reserve(pref._objects.size());
1353
1354 vector_int::const_iterator pi;
1355 for (pi = pref._objects.begin();
1356 pi != pref._objects.end() && is_complete;
1357 ++pi) {
1358 int child_id = (*pi);
1359 if (child_id == 0) {
1360 // A NULL pointer is a NULL pointer.
1361 references.push_back(nullptr);
1362 continue;
1363 }
1364
1365 // See if we have the pointer available now.
1366 CreatedObjs::const_iterator oi = _created_objs.find(child_id);
1367 if (oi == _created_objs.end()) {
1368 // No, too bad.
1369 is_complete = false;
1370 break;
1371 }
1372
1373 const CreatedObj &child_obj = (*oi).second;
1374 if (!child_obj._created) {
1375 // The child object hasn't yet been created.
1376 is_complete = false;
1377 break;
1378 }
1379
1380 if (child_obj._change_this != nullptr || child_obj._change_this_ref != nullptr) {
1381 // It's been created, but the pointer might still change.
1382 is_complete = false;
1383 break;
1384 }
1385
1386 if (require_fully_complete &&
1387 _object_pointers.find(child_id) != _object_pointers.end()) {
1388 // It's not yet complete itself.
1389 is_complete = false;
1390 break;
1391 }
1392
1393 // Yes, it's ready.
1394 references.push_back(child_obj._ptr);
1395 }
1396
1397 if (is_complete) {
1398 // Okay, here's the complete list of pointers for you!
1399 nassertr(references.size() == pref._objects.size(), false);
1400
1401 if (bam_cat.is_spam()) {
1402 bam_cat.spam()
1403 << "complete_pointers for " << (void *)object
1404 << " (" << object->get_type() << "), " << references.size()
1405 << " pointers.\n";
1406 }
1407 int num_completed = 0;
1408 if (!references.empty()) {
1409 num_completed = object->complete_pointers(&references[0], this);
1410 }
1411 if (num_completed != (int)references.size()) {
1412 bam_cat.warning()
1413 << object->get_type() << " completed " << num_completed
1414 << " of " << references.size() << " pointers.\n";
1415 nassertr(num_completed < (int)references.size(), true);
1416 }
1417 return true;
1418
1419 } else {
1420 if (bam_cat.is_spam()) {
1421 bam_cat.spam()
1422 << "not ready: complete_pointers for " << (void *)object
1423 << " (" << object->get_type() << ")\n";
1424 }
1425 }
1426
1427 return false;
1428}
