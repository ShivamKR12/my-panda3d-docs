160apply_properties(EggPrimitive *egg_prim, vector_PT_EggVertex &egg_vertices,
161 PN_stdfloat &smooth_angle) {
162 if (!_surface->_source.empty()) {
163 // This surface is derived from another surface; apply that one first.
164 CLwoSurface *parent = _converter->get_surface(_surface->_source);
165 if (parent != nullptr && parent != this) {
166 parent->apply_properties(egg_prim, egg_vertices, smooth_angle);
167 }
168 }
169
170 bool has_texture = check_texture();
171 bool has_material = check_material();
172
173 egg_prim->set_color(_diffuse_color);
174
175 if (has_material) {
176 egg_prim->set_material(_egg_material);
177 }
178
179 if (has_texture) {
180 egg_prim->set_texture(_egg_texture);
181
182 // Assign UV's to the vertices.
183 generate_uvs(egg_vertices);
184 }

187 egg_prim->set_bface_flag(_backface);
188 }
189

191 smooth_angle = std::max(smooth_angle, _smooth_angle);

202bool CLwoSurface::
203check_texture() {
204 if (_checked_texture) {
205 return (_egg_texture != nullptr);
206 }
207 _checked_texture = true;
208 _egg_texture = nullptr;
209 _map_uvs = nullptr;
210
211 if (_block == nullptr) {
212 // No texture. Not even a shader block.
213 return false;
214 }
215
216 int clip_index = _block->_clip_index;
217 if (clip_index < 0) {
218 // No image file associated with the texture.
219 return false;
220 }
221
222 CLwoClip *clip = _converter->get_clip(clip_index);
223 if (clip == nullptr) {
224 nout << "No clip image with index " << clip_index << "\n";
225 return false;
226 }
227
228 if (!clip->is_still_image()) {
229 // Can't do anything with an animated image right now.
230 return false;
231 }
232
233 Filename pathname = _converter->convert_model_path(clip->_filename);
234
235 _egg_texture = new EggTexture("clip" + format_string(clip_index), pathname);
236
237 // Do we need to generate UV's?
238 switch (_block->_projection_mode) {
239 case LwoSurfaceBlockProjection::M_planar:
240 _map_uvs = &CLwoSurface::map_planar;
241 break;
242
243 case LwoSurfaceBlockProjection::M_cylindrical:
244 _map_uvs = &CLwoSurface::map_cylindrical;
245 break;
246
247 case LwoSurfaceBlockProjection::M_spherical:
248 _map_uvs = &CLwoSurface::map_spherical;
249 break;
250
251 case LwoSurfaceBlockProjection::M_cubic:
252 _map_uvs = &CLwoSurface::map_cubic;
253 break;
254
255 case LwoSurfaceBlockProjection::M_front:
256 // Cannot generate "front" UV's, since this depends on a camera. Is it
257 // supposed to be updated in real time, like a projected texture?
258 break;
259
260 case LwoSurfaceBlockProjection::M_uv:
261 // "uv" projection means to use the existing UV's already defined for the
262 // vertex. This case was already handled in the code that created the
263 // EggVertex pointers.
264 break;
265 };
266
267 // Texture overrides the primitive's natural color.
268 _color[0] = 1.0;
269 _color[1] = 1.0;
270 _color[2] = 1.0;
271
272 return true;
273}

335void CLwoSurface::
336generate_uvs(vector_PT_EggVertex &egg_vertices) {
337 if (_map_uvs == nullptr) {
338 return;
339 }
340
341 // To do this properly near seams and singularities (for instance, the back
342 // seam and the poles of the spherical map), we will need to know the
343 // polygon's centroid.
344 LPoint3d centroid(0.0, 0.0, 0.0);
345
346 vector_PT_EggVertex::const_iterator vi;
347 for (vi = egg_vertices.begin(); vi != egg_vertices.end(); ++vi) {
348 EggVertex *egg_vertex = (*vi);
349 centroid += egg_vertex->get_pos3();
350 }
351
352 centroid /= (double)egg_vertices.size();
353 centroid = centroid * _block->_inv_transform;
354
355 // Now go back through and actually compute the UV's.
356 for (vi = egg_vertices.begin(); vi != egg_vertices.end(); ++vi) {
357 EggVertex *egg_vertex = (*vi);
358 LPoint3d pos = egg_vertex->get_pos3() * _block->_inv_transform;
359 LPoint2d uv = (this->*_map_uvs)(pos, centroid);
360 egg_vertex->set_uv(uv);
361 }
362}

367LPoint2d CLwoSurface::
368map_planar(const LPoint3d &pos, const LPoint3d &) const {
369 // A planar projection is about as easy as can be. We ignore the Y axis,
370 // and project the point into the XZ plane. Done.
371 double u = (pos[0] + 0.5);
372 double v = (pos[2] + 0.5);
373
374 return LPoint2d(u, v);
375}

381LPoint2d CLwoSurface::
382map_spherical(const LPoint3d &pos, const LPoint3d &centroid) const {
383 // To compute the x position on the frame, we only need to consider the
384 // angle of the vector about the Y axis. Project the vector into the XZ
385 // plane to do this.
386
387 LVector2d xz_orig(pos[0], pos[2]);
388 LVector2d xz = xz_orig;
389 double u_offset = 0.0;
390
391 if (xz == LVector2d::zero()) {
392 // If we have a point on either pole, we've got problems. This point maps
393 // to the entire bottom edge of the image, so which U value should we
394 // choose? It does make a difference, especially if we have a number of
395 // polygons around the south pole that all share the common vertex.
396
397 // We choose the U value based on the polygon's centroid.
398 xz.set(centroid[0], centroid[2]);
399
400 } else if (xz[1] >= 0.0 && ((xz[0] < 0.0) != (centroid[0] < 0.))) {
401 // Now, if our polygon crosses the seam along the back of the sphere--that
402 // is, the point is on the back of the sphere (xz[1] >= 0.0) and not on
403 // the same side of the XZ plane as the centroid, we've got problems too.
404 // We need to add an offset to the computed U value, either 1 or -1, to
405 // keep all the vertices of the polygon on the same side of the seam.
406
407 u_offset = (xz[0] < 0.0) ? 1.0 : -1.0;
408 }
409
410 // The U value is based on the longitude: the angle about the Y axis.
411 double u =
412 (atan2(xz[0], -xz[1]) / (2.0 * MathNumbers::pi) + 0.5 + u_offset) * _block->_w_repeat;
413
414 // Now rotate the vector into the YZ plane, and the V value is based on the
415 // latitude: the angle about the X axis.
416 LVector2d yz(pos[1], xz_orig.length());
417 double v =
418 (atan2(yz[0], yz[1]) / MathNumbers::pi + 0.5) * _block->_h_repeat;
419
420 return LPoint2d(u, v);
421}

427LPoint2d CLwoSurface::
428map_cylindrical(const LPoint3d &pos, const LPoint3d &centroid) const {
429 // This is almost identical to the spherical projection, except for the
430 // computation of V.
431
432 LVector2d xz(pos[0], pos[2]);
433 double u_offset = 0.0;
434
435 if (xz == LVector2d::zero()) {
436 // Although a cylindrical mapping does not really have a singularity at
437 // the pole, it's still possible to put a point there, and we'd like to do
438 // the right thing with the polygon that shares that point. So the
439 // singularity logic remains.
440 xz.set(centroid[0], centroid[2]);
441
442 } else if (xz[1] >= 0.0 && ((xz[0] < 0.0) != (centroid[0] < 0.))) {
443 // And cylinders do still have a seam at the back.
444 u_offset = (xz[0] < 0.0) ? 1.0 : -1.0;
445 }
446
447 double u =
448 (atan2(xz[0], -xz[1]) / (2.0 * MathNumbers::pi) + 0.5 + u_offset) * _block->_w_repeat;
449
450 // For a cylindrical mapping, the V value comes almost directly from Y.
451 // Easy.
452 double v = (pos[1] + 0.5);
453
454 return LPoint2d(u, v);
455}

460LPoint2d CLwoSurface::
461map_cubic(const LPoint3d &pos, const LPoint3d &centroid) const {
462 // A cubic projection is a planar projection, but we eliminate the dominant
463 // axis (based on the polygon's centroid) instead of arbitrarily eliminating
464 // Y.
465
466 double x = fabs(centroid[0]);
467 double y = fabs(centroid[1]);
468 double z = fabs(centroid[2]);
469
470 double u, v;
471
472 if (x > y) {
473 if (x > z) {
474 // X is dominant.
475 u = (pos[2] + 0.5);
476 v = (pos[1] + 0.5);
477 } else {
478 // Z is dominant.
479 u = (pos[0] + 0.5);
480 v = (pos[1] + 0.5);
481 }
482 } else {
483 if (y > z) {
484 // Y is dominant.
485 u = (pos[0] + 0.5);
486 v = (pos[2] + 0.5);
487 } else {
488 // Z is dominant.
489 u = (pos[0] + 0.5);
490 v = (pos[1] + 0.5);
491 }
492 }
493
494 return LPoint2d(u, v);
495}
