65CPT(InstanceList) InstanceList::
66without(const BitArray &mask) const {
67 size_t num_instances = size();
68 size_t num_culled = (size_t)mask.get_num_on_bits();
69 if (num_culled == 0) {
70 return this;
71 }
72 else if (num_culled >= num_instances) {
73 static CPT(InstanceList) empty_list;
74 if (empty_list == nullptr) {
75 empty_list = new InstanceList;
76 }
77
78 nassertr(num_culled <= num_instances, empty_list);
79 return empty_list;
80 }
81
82 InstanceList *new_list = new InstanceList;
83 new_list->_instances.reserve(num_instances - num_culled);
84
85 for (size_t i = (size_t)mask.get_lowest_off_bit(); i < num_instances; ++i) {
86 if (!mask.get_bit(i)) {
87 new_list->_instances.push_back(_instances[i]);
88 }
89 }
90
91 return new_list;
92}
