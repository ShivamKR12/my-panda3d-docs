218bool ZipArchive::
219open_read_write(std::iostream *stream, bool owns_pointer) {
220 close();
221
222 // We don't support locking when opening a file in read-write mode, because
223 // we don't bother with locking on write. But we need to have an
224 // IStreamWrapper to assign to the _read member, so we create one on-the-fly
225 // here.
226 _read = new StreamWrapper(stream, owns_pointer);
227 _write = stream;
228 _owns_stream = true; // Because we own the StreamWrapper, above.
229 _write->seekp(0, std::ios::beg);
230
231 // Check whether the read stream is empty.
232 stream->seekg(0, std::ios::end);
233 if (stream->tellg() == (streampos)0) {
234 // The read stream is empty, which is always valid.
235 _index_changed = true;
236 return true;
237 }
238
239 // The read stream is not empty, so we'd better have a valid ZipArchive.
240 return read_index();

353add_subfile(const std::string &subfile_name, std::istream *subfile_data,
354 int compression_level) {
355 nassertr(is_write_valid(), string());
356
357#ifndef HAVE_ZLIB
358 express_cat.warning()

360 return std::string();
361#endif

363 std::string name = standardize_subfile_name(subfile_name);
364 if (!name.empty()) {
365 Subfile *subfile = new Subfile(subfile_name, compression_level);
366

370 _write->seekp(fpos);

372 if (!subfile->write_header(*_write, fpos)) {
373 delete subfile;
374 return "";

378 // Failed to write the data.
379 delete subfile;
380 return "";
381 }
382
383 if (fpos > _index_start) {
384 _index_start = fpos;
385 }
386 add_new_subfile(subfile, compression_level);
387 }

525bool ZipArchive::
526add_jar_signature(X509 *cert, EVP_PKEY *pkey, const std::string &alias) {
527 nassertr(is_write_valid() && is_read_valid(), false);
528 nassertr(cert != nullptr, false);
529 nassertr(pkey != nullptr, false);
530
531 if (!X509_check_private_key(cert, pkey)) {
532 express_cat.error()
533 << "Private key does not match certificate.\n";
534 return false;
535 }
536
537 const char *ext;
538 int algo = EVP_PKEY_base_id(pkey);
539 switch (algo) {
540 case EVP_PKEY_RSA:
541 ext = ".RSA";
542 break;
543 case EVP_PKEY_DSA:
544 ext = ".DSA";
545 break;
546 case EVP_PKEY_EC:
547 ext = ".EC";
548 break;
549 default:
550 express_cat.error()
551 << "Private key has unsupported algorithm.\n";
552 return false;
553 }
554
555 // Sanitize alias to be used in a filename.
556 std::string basename;
557 for (char c : alias.substr(0, 8)) {
558 if ((c >= '0' && c <= '9') || (c >= 'A' && c <= 'Z') || c == '-' || c == '_') {
559 basename += c;
560 }
561 else if (c >= 'a' && c <= 'z') {
562 basename += (c - 0x20);
563 }
564 else if (((uint8_t)c & 0xc0) != 0x80) {
565 basename += '_';
566 }
567 }
568
569 // Generate a MANIFEST.MF file.
570 const std::string header = "Manifest-Version: 1.0\r\n\r\n";
571 const std::string header_digest = "VmrRqAIgAm0FCZViZFzpaP8OfDbN4iY0MyYFuzTMPv8=";
572
573 std::stringstream manifest;
574 SHA256_CTX manifest_ctx;
575 SHA256_Init(&manifest_ctx);
576
577 manifest << header;
578 SHA256_Update(&manifest_ctx, header.data(), header.size());
579
580 std::ostringstream sigfile_body;
581
582 for (Subfile *subfile : _subfiles) {
583 nassertr(subfile != nullptr, false);
584
585 if (subfile->_name.compare(0, 9, "META-INF/") == 0) {
586 continue;
587 }
588
589 std::string section = "Name: " + subfile->_name + "\r\n";
590 sigfile_body << section;
591
592 // Hash the subfile.
593 unsigned char digest[SHA256_DIGEST_LENGTH];
594 {
595 std::istream *stream = open_read_subfile(subfile);
596
597 SHA256_CTX subfile_ctx;
598 SHA256_Init(&subfile_ctx);
599
600 char buffer[4096];
601 stream->read(buffer, sizeof(buffer));
602 size_t count = stream->gcount();
603 while (count > 0) {
604 SHA256_Update(&subfile_ctx, buffer, count);
605 stream->read(buffer, sizeof(buffer));
606 count = stream->gcount();
607 }
608 delete stream;
609
610 SHA256_Final(digest, &subfile_ctx);
611 }
612
613 // Encode to base64.
614 section += "SHA-256-Digest: " + base64_encode(digest, SHA256_DIGEST_LENGTH) + "\r\n\r\n";
615
616 // Encode what we just wrote to the manifest file as well.
617 {
618 unsigned char digest[SHA256_DIGEST_LENGTH];
619
620 SHA256_CTX section_ctx;
621 SHA256_Init(&section_ctx);
622 SHA256_Update(&section_ctx, section.data(), section.size());
623 SHA256_Final(digest, &section_ctx);
624
625 sigfile_body << "SHA-256-Digest: " << base64_encode(digest, SHA256_DIGEST_LENGTH) << "\r\n\r\n";
626 }
627
628 manifest << section;
629 SHA256_Update(&manifest_ctx, section.data(), section.size());
630 }
631
632 // The hash for the whole manifest file goes at the beginning of the .SF file.
633 std::stringstream sigfile;
634 {
635 unsigned char digest[SHA256_DIGEST_LENGTH];
636 SHA256_Final(digest, &manifest_ctx);
637 sigfile << "Signature-Version: 1.0\r\n";
638 sigfile << "SHA-256-Digest-Manifest-Main-Attributes: " << header_digest << "\r\n";
639 sigfile << "SHA-256-Digest-Manifest: " << base64_encode(digest, SHA256_DIGEST_LENGTH) << "\r\n\r\n";
640 sigfile << sigfile_body.str();
641 }
642
643 // Sign and convert to to DER format
644 std::string sigfile_data = sigfile.str();
645 BIO *sigfile_mbio = BIO_new_mem_buf((void *)sigfile_data.data(), sigfile_data.size());
646 PKCS7 *p7 = PKCS7_sign(cert, pkey, nullptr, sigfile_mbio, PKCS7_DETACHED | PKCS7_NOATTR);
647 int der_len = i2d_PKCS7(p7, nullptr);
648 std::string signature_str(der_len, '\0');
649 unsigned char *p = (unsigned char *)signature_str.data();
650 i2d_PKCS7(p7, &p);
651 std::istringstream signature(std::move(signature_str));
652 PKCS7_free(p7);
653
654 add_subfile("META-INF/MANIFEST.MF", &manifest, 9);
655 add_subfile("META-INF/" + basename + ".SF", &sigfile, 9);
656 add_subfile("META-INF/" + basename + ext, &signature, 9);
657
658 return true;
659}
660#endif // HAVE_OPENSSL

677flush() {
678 if (!is_write_valid()) {
679 return false;
680 }
681
682 nassertr(_write != nullptr, false);
683
684 // First, mark out all of the removed subfiles.
685 for (Subfile *subfile : _removed_subfiles) {
686 delete subfile;
687 }
688 _removed_subfiles.clear();
689
690 if (_index_changed) {
691 std::streampos fpos = _index_start;

693 if (!write_index(*_write, fpos)) {

696 close();
697 return false;
698 }
699 _index_changed = false;
700 }
701
702 _write->flush();
703 return true;

717bool ZipArchive::
718repack() {
719 nassertr(is_write_valid() && is_read_valid(), false);
720 nassertr(!_filename.empty(), false);
721
722 // First, we open a temporary filename to copy the ZipArchive to.
723 Filename dirname = _filename.get_dirname();
724 if (dirname.empty()) {
725 dirname = ".";
726 }
727 Filename temp_filename = Filename::temporary(dirname, "ziptemp");
728 temp_filename.set_binary();
729 pofstream temp;
730 if (!temp_filename.open_write(temp)) {
731 express_cat.info()
732 << "Unable to open temporary file " << temp_filename << "\n";
733 return false;
734 }
735
736 // Now we scrub our internal structures so it looks like we're a brand new
737 // ZipArchive.
738 for (Subfile *subfile : _removed_subfiles) {
739 delete subfile;
740 }
741 _removed_subfiles.clear();
742
743 // And we write our contents to our new temporary file.
744 //_write = &temp;
745
746 bool success = true;
747
748 _read->acquire();
749 std::istream &read = *_read->get_istream();
750
751 // Copy over all of the subfiles.
752 std::streampos fpos = 0;
753
754 for (Subfile *subfile : _subfiles) {
755 if (!subfile->read_header(read)) {
756 success = false;
757 continue;
758 }
759
760 // We don't need to write a data descriptor, since at this point we know
761 // the checksum and sizes.
762 subfile->_flags &= ~SF_data_descriptor;
763
764 if (!subfile->write_header(temp, fpos)) {
765 success = false;
766 continue;
767 }
768
769 static const size_t buffer_size = 4096;
770 char buffer[buffer_size];
771 uint64_t num_bytes = subfile->_data_length;
772 fpos += num_bytes;
773
774 while (num_bytes >= buffer_size) {
775 read.read(buffer, buffer_size);
776 temp.write(buffer, buffer_size);
777 num_bytes -= buffer_size;
778 }
779 if (num_bytes > 0) {
780 read.read(buffer, num_bytes);
781 temp.write(buffer, num_bytes);
782 }
783 }
784 _read->release();
785
786 // Write the central directory at the end of the file.
787 success = success && write_index(temp, fpos);
788
789 if (!success) {
790 express_cat.error()
791 << "Failed to write repacked archive to " << temp_filename << ".\n";
792 temp.close();
793 temp_filename.unlink();
794 return false;
795 }
796
797 // Now close everything, and move the temporary file back over our original
798 // file.
799 Filename orig_name = _filename;
800 temp.close();
801 close();
802 orig_name.unlink();
803 if (!temp_filename.rename_to(orig_name)) {
804 express_cat.info()
805 << "Unable to rename temporary file " << temp_filename << " to "
806 << orig_name << ".\n";
807 return false;
808 }
809
810 if (!open_read_write(orig_name)) {
811 express_cat.info()
812 << "Unable to read newly repacked " << _filename
813 << ".\n";
814 return false;
815 }
816
817 return true;
818}

908 contents.push_back(basename);
909 previous = basename;
910 }
911 ++fi;
912 }

1185 }

1191

1255bool ZipArchive::
1256read_subfile(int index, vector_uchar &result) {
1257 nassertr(is_read_valid(), false);
1258 nassertr(index >= 0 && index < (int)_subfiles.size(), false);
1259 result.clear();
1260
1261 // Now look up the particular Subfile we are reading.
1262 nassertr(is_read_valid(), false);
1263 nassertr(index >= 0 && index < (int)_subfiles.size(), false);
1264 Subfile *subfile = _subfiles[index];
1265
1266 result.reserve(subfile->_uncompressed_length);
1267
1268 bool success = true;
1269 if (subfile->is_compressed() || subfile->is_encrypted()) {
1270 // If the subfile is encrypted or compressed, we can't read it directly.
1271 // Fall back to the generic implementation.
1272 std::istream *in = open_read_subfile(index);
1273 if (in == nullptr) {
1274 return false;
1275 }
1276
1277 success = VirtualFile::simple_read_file(in, result);
1278 close_read_subfile(in);
1279
1280 } else {
1281 // But if the subfile is just a plain file, we can just read the data
1282 // directly from the ZipArchive, without paying the cost of an ISubStream.
1283 static const size_t buffer_size = 4096;
1284 char buffer[buffer_size];
1285
1286 _read->acquire();
1287 if (!subfile->read_header(*_read->get_istream())) {
1288 _read->release();
1289 express_cat.error()
1290 << "Failed to read local header of "
1291 << _filename << "/" << subfile->_name << "\n";
1292 return false;
1293 }
1294 std::istream &read = *_read->get_istream();
1295 /*std::streampos data_start =*/ read.tellg();
1296
1297 size_t bytes_to_go = subfile->_uncompressed_length;
1298 read.read(buffer, std::min(bytes_to_go, buffer_size));
1299 size_t read_bytes = read.gcount();
1300
1301 while (read_bytes > 0) {
1302 result.insert(result.end(), buffer, buffer + read_bytes);
1303
1304 bytes_to_go -= read_bytes;
1305 if (bytes_to_go == 0) {
1306 break;
1307 }
1308
1309 read.read(buffer, std::min(bytes_to_go, buffer_size));
1310 read_bytes = read.gcount();
1311 }
1312
1313 _read->release();
1314 success = (bytes_to_go == 0);
1315 }
1316
1317 if (!success) {
1318 std::ostringstream message;
1319 message << "I/O error reading from " << get_filename() << " at "
1320 << get_subfile_name(index);
1321 nassert_raise(message.str());
1322 return false;
1323 }
1324
1325 return true;
1326}

1355open_read_subfile(Subfile *subfile) {
1356 // Read the header first.
1357 _read->acquire();

1359 _read->release();

1361 << "Failed to read local header of "

1364 }
1365 std::streampos data_start = _read->get_istream()->tellg();
1366 _read->release();
1367
1368 // Return an ISubStream object that references into the open ZipArchive
1369 // istream.
1370 nassertr(data_start != (streampos)0, nullptr);
1371 std::istream *stream =
1372 new ISubStream(_read, data_start,
1373 data_start + (streampos)subfile->_data_length);

1375 if (subfile->is_compressed()) {
1376#ifndef HAVE_ZLIB

1382 // IDecompressStream that wraps around the ISubStream.

1384 stream = wrapper;
1385#endif // HAVE_ZLIB
1386 }
1387
1388 if (stream->fail()) {
1389 // Hmm, some inexplicable problem.
1390 delete stream;
1391 return nullptr;
1392 }
1393

1441bool ZipArchive::
1442read_index() {
1443 nassertr(_read != nullptr, false);
1444
1445 // We acquire the IStreamWrapper lock for the duration of this method.
1446 _read->acquire();
1447 std::istream *read = _read->get_istream();
1448
1449 // ZIP files need to be read from the end.
1450 read->seekg(-2, std::ios::end);
1451 if (read->fail()) {
1452 express_cat.info()
1453 << "Unable to seek ZIP archive " << _filename << ".\n";
1454 _read->release();
1455 close();
1456 return false;
1457 }
1458
1459 std::streampos fpos = read->tellg();
1460 _file_end = fpos + (std::streamoff)2;
1461
1462 uint64_t cdir_entries = 0;
1463 uint64_t cdir_offset = 0;
1464 uint32_t comment_length = 0;
1465 std::streampos eocd_offset = 0;
1466 bool found = false;
1467
1468 // Seek backwards until we have found the the end-of-directory record.
1469 StreamReader reader(read, false);
1470 while (comment_length <= 0xffff && fpos >= 20) {
1471 //nassertr(fpos == read->tellg(), false);
1472
1473 if (reader.get_uint16() == comment_length) {
1474 // This field references the distance to the end of the .zip file, so it
1475 // could be the comment length field at the end of the record. Skip to the
1476 // beginning of the record to see if the signature matches.
1477 read->seekg(-22, std::ios::cur);
1478
1479 if (reader.get_uint32() == 0x06054b50) {
1480 // Yes, got it.
1481 eocd_offset = read->tellg() - (std::streamoff)4;
1482 reader.skip_bytes(6);
1483 cdir_entries = reader.get_uint16();
1484 /*cdir_size = */reader.get_uint32();
1485 cdir_offset = reader.get_uint32();
1486 if (comment_length > 0) {
1487 _comment = reader.get_fixed_string(comment_length);
1488 } else {
1489 _comment.clear();
1490 }
1491 found = true;
1492 break;
1493 }
1494
1495 // No, skip back to where we were and continue scanning for comments.
1496 comment_length += 2;
1497 read->seekg(22 - 4 - 4, std::ios::cur);
1498 fpos -= 2;
1499 continue;
1500 }
1501
1502 comment_length += 2;
1503 read->seekg(-4, std::ios::cur);
1504 fpos -= 2;
1505 }
1506
1507 if (!found) {
1508 express_cat.info()
1509 << "Unable to find end-of-directory record in ZIP archive " << _filename << ".\n";
1510 _read->release();
1511 close();
1512 return false;
1513 }
1514
1515 // Now look for a ZIP64 end-of-central-directory locator.
1516 if (eocd_offset >= 20) {
1517 uint64_t eocd64_offset = 0;
1518 read->seekg(eocd_offset - (std::streamoff)20);
1519 if (reader.get_uint32() == 0x07064b50) {
1520 reader.skip_bytes(4); // disk no
1521 eocd64_offset = reader.get_uint64();
1522 reader.skip_bytes(4); // disk count
1523
1524 read->seekg(eocd64_offset);
1525 if (reader.get_uint32() == 0x06064b50) {
1526 reader.skip_bytes(20);
1527 cdir_entries = reader.get_uint64();
1528 /*cdir_size = */reader.get_uint64();
1529 cdir_offset = reader.get_uint64();
1530 } else {
1531 express_cat.info()
1532 << "Unable to read ZIP64 end-of-directory record in ZIP archive "
1533 << _filename << ".\n";
1534 }
1535 }
1536 }
1537
1538 _index_start = cdir_offset;
1539
1540 // Find the central directory.
1541 read->seekg((std::streampos)cdir_offset);
1542 if (read->fail()) {
1543 express_cat.info()
1544 << "Unable to locate central directory in ZIP archive " << _filename << ".\n";
1545 _read->release();
1546 close();
1547 return false;
1548 }
1549
1550 _record_timestamp = false;
1551
1552 for (size_t i = 0; i < cdir_entries; ++i) {
1553 Subfile *subfile = new Subfile;
1554 if (!subfile->read_index(*read)) {
1555 express_cat.info()
1556 << "Failed to read central directory for " << _filename << ".\n";
1557 _read->release();
1558 close();
1559 return false;
1560 }
1561
1562 if (subfile->_timestamp != dos_epoch) {
1563 // If all subfiles have the timestamp set to the DOS epoch, we apparently
1564 // don't care about preserving timestamps.
1565 _record_timestamp = true;
1566 }
1567 _subfiles.push_back(subfile);
1568 }
1569
1570 // Sort the subfiles.
1571 size_t before_size = _subfiles.size();
1572 _subfiles.sort();
1573 size_t after_size = _subfiles.size();
1574
1575 // If these don't match, the same filename appeared twice in the index,
1576 // which shouldn't be possible.
1577 nassertr(before_size == after_size, false);
1578
1579 _read->release();
1580 return true;
1581}

1586bool ZipArchive::
1587write_index(std::ostream &write, std::streampos &fpos) {
1588 nassertr(write.tellp() == fpos, false);
1589
1590 _index_start = fpos;
1591
1592 for (Subfile *subfile : _subfiles) {
1593 if (!subfile->write_index(write, fpos)) {
1594 express_cat.info()
1595 << "Failed to write central directory entry for "
1596 << _filename << "/" << subfile->_name << ".\n";
1597 _read->release();
1598 close();
1599 return false;
1600 }
1601 }
1602
1603 size_t cdir_entries = _subfiles.size();
1604 std::streamoff cdir_size = fpos - _index_start;
1605
1606 StreamWriter writer(write);
1607
1608 if (_index_start >= 0xffffffff ||
1609 cdir_size >= 0xffffffff ||
1610 cdir_entries >= 0xffff) {
1611 // Write a ZIP64 end-of-central-directory record.
1612 writer.add_uint32(0x06064b50);
1613 writer.add_uint64(44); // size of the rest of the record (w/o first 12 bytes)
1614 writer.add_uint16(45); // version number that produced this file
1615 writer.add_uint16(45); // version number needed to read this file
1616 writer.add_uint32(0);
1617 writer.add_uint32(0);
1618 writer.add_uint64(cdir_entries);
1619 writer.add_uint64(cdir_entries);
1620 writer.add_uint64(cdir_size);
1621 writer.add_uint64(_index_start);
1622 nassertr(write.tellp() == fpos + std::streamoff(12 + 44), false);
1623
1624 // And write the ZIP64 end-of-central-directory-record locator.
1625 writer.add_uint32(0x07064b50);
1626 writer.add_uint32(0);
1627 writer.add_uint64(fpos);
1628 writer.add_uint32(1); // number of disks
1629 }
1630
1631 // Write the end of central directory record.
1632 writer.add_uint32(0x06054b50);
1633 writer.add_uint16(0);
1634 writer.add_uint16(0);
1635 writer.add_uint16(std::min((size_t)0xffffu, cdir_entries));
1636 writer.add_uint16(std::min((size_t)0xffffu, cdir_entries));
1637 writer.add_uint32(std::min((std::streamoff)0xffffffffu, cdir_size));
1638 writer.add_uint32(std::min((std::streampos)0xffffffffu, _index_start));
1639 writer.add_uint16(_comment.size());
1640 writer.append_data(_comment);
1641
1642 if (write.fail()) {
1643 express_cat.warning()
1644 << "Unable to write central directory for " << _filename << ".\n";
1645 close();
1646 return false;
1647 }
1648
1649 fpos = write.tellp();
1650 if (fpos < _file_end) {
1651 // We didn't hit the end of the file writing the index. This is a problem
1652 // because readers start looking for the EOCD record at the end of the file.
1653 // We'll have to shift the whole index forwards. Unfortunately it's hard to
1654 // anticipate having to do this ahead of time.
1655 fpos = _index_start + (_file_end - fpos);
1656 _needs_repack = true;
1657 write.seekp(fpos);
1658 return write_index(write, fpos);
1659 }
1660
1661 _file_end = fpos;
1662 return true;
1663}

1696bool ZipArchive::Subfile::
1697read_index(std::istream &read) {
1698 StreamReader reader(read);
1699
1700 if (reader.get_uint32() != 0x02014b50) {
1701 return false;
1702 }
1703
1704 /*uint16_t version = */reader.get_uint8();
1705 _system = reader.get_uint8();
1706 /*uint16_t min_version = */reader.get_uint16();
1707 _flags = reader.get_uint16();
1708 _compression_method = (CompressionMethod)reader.get_uint16();
1709 {
1710 // Convert from DOS/FAT timestamp to UNIX timestamp.
1711 uint16_t mtime = reader.get_uint16();
1712 uint16_t mdate = reader.get_uint16();
1713
1714 struct tm time = {};
1715 time.tm_sec = (mtime & 0b0000000000011111u) << 1;
1716 time.tm_min = (mtime & 0b0000011111100000u) >> 5;
1717 time.tm_hour = (mtime & 0b1111100000000000u) >> 11;
1718 time.tm_mday = (mdate & 0b0000000000011111u);
1719 time.tm_mon = ((mdate & 0b0000000111100000u) >> 5) - 1;
1720 time.tm_year = ((mdate & 0b1111111000000000u) >> 9) + 80;
1721 time.tm_isdst = -1;
1722 _timestamp = mktime(&time);
1723 }
1724 _checksum = reader.get_uint32();
1725 _data_length = reader.get_uint32();
1726 _uncompressed_length = reader.get_uint32();
1727 size_t name_length = reader.get_uint16();
1728 size_t extra_length = reader.get_uint16();
1729 size_t comment_length = reader.get_uint16();
1730 /*size_t disk_number =*/ reader.get_uint16();
1731 _internal_attribs = reader.get_uint16();
1732 _external_attribs = reader.get_uint32();
1733 _header_start = (std::streampos)reader.get_uint32();
1734
1735 std::string name = reader.get_fixed_string(name_length);
1736
1737 // Read the extra fields, which may include a UNIX timestamp, which can be
1738 // specified with greater precision than a DOS timestamp.
1739 while (extra_length >= 4) {
1740 uint16_t const tag = reader.get_uint16();
1741 uint16_t const size = reader.get_uint16();
1742 if (tag == 0x0001) {
1743 // ZIP64 extended info.
1744 int size_left = size;
1745 if (_uncompressed_length == 0xffffffffu && size_left >= 8) {
1746 _uncompressed_length = reader.get_uint64();
1747 size_left -= 8;
1748 }
1749 if (_data_length == 0xffffffffu && size_left >= 8) {
1750 _data_length = reader.get_uint64();
1751 size_left -= 8;
1752 }
1753 if ((uint64_t)_header_start == 0xffffffffu && size_left >= 8) {
1754 _header_start = reader.get_uint64();
1755 size_left -= 8;
1756 }
1757 reader.skip_bytes(size_left);
1758 } else if (tag == 0x5455 && size == 5) {
1759 reader.skip_bytes(1);
1760 _timestamp = reader.get_uint32();
1761 } else {
1762 reader.skip_bytes(size);
1763 }
1764 extra_length -= 4 + size;
1765 }
1766 // Skip leftover bytes in the extra field not large enough to contain a proper
1767 // extra tag. This may be the case for Android .apk files processed with
1768 // zipalign, which uses this for alignment.
1769 reader.skip_bytes(extra_length);
1770
1771 std::string comment = reader.get_fixed_string(comment_length);
1772
1773 if (_flags & SF_utf8_encoding) {
1774 _name = std::move(name);
