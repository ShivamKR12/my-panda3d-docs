85bool PGTop::
86cull_callback(CullTraverser *trav, CullTraverserData &data) {
87 // We create a new MouseWatcherGroup for the purposes of collecting a new
88 // set of regions visible onscreen.
89 PT(PGMouseWatcherGroup) old_watcher_group;
90 if (_watcher_group != nullptr) {
91 _watcher_group->clear_top(this);
92 old_watcher_group = _watcher_group;
93 _watcher_group = new PGMouseWatcherGroup(this);
94 }
95
96 // Now subsitute for the normal CullTraverser a special one of our own
97 // choosing. This just carries around a pointer back to the PGTop node, for
98 // the convenience of PGItems to register themselves as they are drawn.
99 PGCullTraverser pg_trav(this, trav);
100 pg_trav.local_object();
101 pg_trav._sort_index = _start_sort;
102 pg_trav.traverse_below(data);
103 pg_trav.end_traverse();
104
105 // Now tell the watcher about the new set of regions. Strictly speaking, we
106 // shouldn't do this until the frame that we're about to render has been
107 // presented; otherwise, we may make regions active before they are actually
108 // visible. But no one has complained about this so far.
109 if (_watcher_group != nullptr) {
110 nassertr(_watcher != nullptr, false);
111 _watcher->replace_group(old_watcher_group, _watcher_group);
112 }
113
114 // We've taken care of the traversal, thank you.
115 return false;
116}
