45int cb_seek(void *stream, opus_int64 offset, int whence) {
46 if (!opus_enable_seek) {
47 return -1;
48 }
49
50 istream *in = (istream *)stream;
51 nassertr(in != nullptr, -1);
52
53 switch (whence) {
54 case SEEK_SET:
55 in->seekg(offset, std::ios::beg);
56 break;
57
58 case SEEK_CUR:
59 // opusfile uses a seek with offset 0 to determine whether seeking is
60 // supported, but this is not good enough. We seek to the end and back.
61 if (offset == 0) {
62 std::streambuf *buf = in->rdbuf();
63 std::streampos pos = buf->pubseekoff(0, std::ios::cur, std::ios::in);
64 if (pos < 0) {
65 return -1;
66 }
67 if (buf->pubseekoff(0, std::ios::end, std::ios::in) >= 0) {
68 // It worked; seek back to the previous location.
69 buf->pubseekpos(pos, std::ios::in);
70 return 0;
71 } else {
72 return -1;
73 }
74 }
75 in->seekg(offset, std::ios::cur);
76 break;
77
78 case SEEK_END:
79 in->seekg(offset, std::ios::end);
80 break;
81
82 default:
83 movies_cat.error()
84 << "Illegal parameter to seek in cb_seek\n";
85 return -1;
86 }
87
88 if (in->fail()) {
89 movies_cat.error()
90 << "Failure to seek to byte " << offset;
91
92 switch (whence) {
93 case SEEK_CUR:
94 movies_cat.error(false)
95 << " from current location!\n";
96 break;
97
98 case SEEK_END:
99 movies_cat.error(false)
100 << " from end of file!\n";
101 break;
102
103 default:
104 movies_cat.error(false) << "!\n";
105 }
106
107 return -1;
108 }
109
110 return 0;
111}

181void OpusAudioCursor::
182seek(double t) {
183 if (!opus_enable_seek) {
184 return;
185 }
186
187 t = std::max(t, 0.0);
188
189 // Use op_time_seek_lap if cross-lapping is enabled.
190 ogg_int64_t sample = (ogg_int64_t)(t * 48000.0);
191 int error = op_pcm_seek(_op, sample);
192
193 // Special case for seeking to the beginning; if normal seek fails, we may
194 // be able to explicitly seek to the beginning of the file and call op_open
195 // again. This allows looping compressed .opus files.
196 if (error == OP_ENOSEEK && sample == 0) {
197 if (_stream->rdbuf()->pubseekpos(0, std::ios::in) == (std::streampos)0) {
198 OggOpusFile *op = op_open_callbacks((void *)_stream, &callbacks, nullptr, 0, nullptr);
199 if (op != nullptr) {
200 op_free(_op);
201 _op = op;
202 } else {
203 movies_cat.error()
204 << "Failed to reopen Opus file to seek to beginning.\n";
205 return;
206 }
207
208 // Reset this field for good measure, just in case this changed.
209 _audio_channels = op_channel_count(_op, -1);
210
211 _last_seek = 0.0;
212 _samples_read = 0;
213 return;
214 }
215 }
216 if (error != 0) {
217 movies_cat.error()
218 << "Seek failed (error " << error << "). Opus stream may not be seekable.\n";
219 return;
220 }
221
222 _last_seek = op_pcm_tell(_op) / 48000.0;
223 _samples_read = 0;
224}

231int OpusAudioCursor::
232read_samples(int n, int16_t *data) {
233 int16_t *end = data + (n * _audio_channels);
234
235 while (data < end) {
236 // op_read gives it to us in the exact format we need. Nifty!
237 int link;
238 int read_samples = op_read(_op, data, end - data, &link);
239 if (read_samples > 0) {
240 data += read_samples * _audio_channels;
241 _samples_read += read_samples;
242 } else {
243 if (read_samples == 0 && _length == 1.0E10) {
244 _length = op_pcm_tell(_op) / 48000.0;
245 }
246 break;
247 }
248
249 if (_link != link) {
250 // It is technically possible for it to change parameters from one link
251 // to the next. However, we don't offer this flexibility.
252 int channels = op_channel_count(_op, link);
253 if (channels != _audio_channels) {
254 movies_cat.error()
255 << "Opus file has inconsistent channel count!\n";
256
257 // We'll change it anyway. Not sure what happens next.
258 _audio_channels = channels;
259 }
260
261 _link = link;
262 }
263 }
264
265 // Fill the rest of the buffer with silence.
266 if (data < end) {
267 memset(data, 0, (unsigned char *)end - (unsigned char *)data);
268 n -= (end - data) / _audio_channels;
269 }
270 return n;
271}
