95PGItem::
96PGItem(const PGItem &copy) :
97 PandaNode(copy),
98 _notify(nullptr),
99 _has_frame(copy._has_frame),
100 _frame(copy._frame),
101 _state(copy._state),
102 _flags(copy._flags),
103 _region(new PGMouseWatcherRegion(this))
104#ifdef HAVE_AUDIO
105 , _sounds(copy._sounds)
106#endif
107{
108 // We give our region the same name as the region for the PGItem we're
109 // copying--so that this PGItem will generate the same event names when the
110 // user interacts with it.
111 _region->set_name(copy._region->get_name());
112
113 // Make a deep copy of all of the original PGItem's StateDefs.
114 size_t num_state_defs = copy._state_defs.size();
115 _state_defs.reserve(num_state_defs);
116 for (size_t i = 0; i < num_state_defs; ++i) {
117 // We cheat and cast away the const, because the frame is just a cache.
118 // But we have to get the frame out of the source before we can safely
119 // copy it.
120 StateDef &old_sd = (StateDef &)(copy._state_defs[i]);
121 old_sd._frame.remove_node();
122 old_sd._frame_stale = true;
123
124 StateDef new_sd;
125 new_sd._root = old_sd._root.copy_to(NodePath());
126 new_sd._frame_style = old_sd._frame_style;
127
128 _state_defs.push_back(new_sd);
129
130#ifdef THREADED_PIPELINE
131 if (Pipeline::get_render_pipeline()->get_num_stages() > 1) {
132 ((PGItem &)copy).update_frame((int)i);
133 update_frame((int)i);
134 }
135#endif
136 }
137}

196bool PGItem::
197cull_callback(CullTraverser *trav, CullTraverserData &data) {
198 CullTraverser::_pgui_nodes_pcollector.add_level(1);
199
200 // We try not to hold the lock for longer than necessary.
201 PT(PandaNode) state_def_root;
202 bool has_frame;
203 PGMouseWatcherRegion *region;
204 {
205 LightReMutexHolder holder(_lock);
206 has_frame = _has_frame && ((_flags & F_active) != 0);
207 region = _region;
208
209 int state = _state;
210 if (state >= 0 && (size_t)state < _state_defs.size()) {
211 StateDef &state_def = _state_defs[state];
212 if (!state_def._root.is_empty()) {
213 if (Thread::get_current_pipeline_stage() == 0) {
214 if (state_def._frame_stale) {
215 update_frame(state);
216 }
217 }
218
219 state_def_root = state_def._root.node();
220 }
221 }
222 }
223
224 if (has_frame && !data.is_this_node_hidden(trav->get_camera_mask())) {
225 // The item has a frame, so we want to generate a region for it and update
226 // the MouseWatcher.
227
228 // We can only do this if our traverser is a PGCullTraverser (which will
229 // be the case if this node was parented somewhere under a PGTop node).
230 if (trav->is_exact_type(PGCullTraverser::get_class_type())) {
231 PGCullTraverser *pg_trav;
232 DCAST_INTO_R(pg_trav, trav, true);
233
234 const LMatrix4 &transform = data.get_net_transform(trav)->get_mat();
235
236 // Consider the cull bin this object is in. Since the binning affects
237 // the render order, we want bins that render later to get higher sort
238 // values.
239 int bin_index = data._state->get_bin_index();
240 int sort;
241
242 CullBinManager *bin_manager = CullBinManager::get_global_ptr();
243 CullBinManager::BinType bin_type = bin_manager->get_bin_type(bin_index);
244 if (bin_type == CullBinManager::BT_fixed) {
245 // If the bin is a "fixed" type bin, our local sort is based on the
246 // fixed order.
247 sort = data._state->get_draw_order();
248
249 } else if (bin_type == CullBinManager::BT_unsorted) {
250 // If the bin is an "unsorted" type bin, we base the local sort on the
251 // scene graph order.
252 sort = pg_trav->_sort_index;
253 pg_trav->_sort_index++;
254
255 } else {
256 // Otherwise, the local sort is irrelevant.
257 sort = 0;
258 }
259
260 // Now what order does this bin sort relative to the other bins? This
261 // becomes the high-order part of the final sort count.
262 int bin_sort = bin_manager->get_bin_sort(data._state->get_bin_index());
263
264 // Combine the two sorts into a single int. This assumes we only need
265 // 16 bits for each sort number, possibly an erroneous assumption. We
266 // should really provide two separate sort values, both ints, in the
267 // MouseWatcherRegion; but in the interest of expediency we work within
268 // the existing interface which only provides one.
269 sort = (bin_sort << 16) | ((sort + 0x8000) & 0xffff);
270
271 const ClipPlaneAttrib *clip = nullptr;
272 const ScissorAttrib *scissor = nullptr;
273 data._state->get_attrib(clip);
274 data._state->get_attrib(scissor);
275 if (activate_region(transform, sort, clip, scissor)) {
276 pg_trav->_top->add_region(region);
277 }
278 }
279 }
280
281 if (state_def_root != nullptr) {
282 // This item has a current state definition that we should use to render
283 // the item.
284 trav->traverse_down(data, state_def_root);
285 }
286
287 // Now continue to render everything else below this node.
288 return true;
289}

296void PGItem::
297compute_internal_bounds(CPT(BoundingVolume) &internal_bounds,
298 int &internal_vertices,
299 int pipeline_stage,
300 Thread *current_thread) const {
301 LightReMutexHolder holder(_lock, current_thread);
302 int num_vertices = 0;
303
304 // First, get ourselves a fresh, empty bounding volume.
305 PT(BoundingVolume) bound;
306
307 BoundingVolume::BoundsType btype = get_bounds_type();
308 if (btype == BoundingVolume::BT_default) {
309 btype = bounds_type;
310 }
311
312 if (btype == BoundingVolume::BT_sphere) {
313 bound = new BoundingSphere;
314 } else {
315 bound = new BoundingBox;
316 }
317
318 // Now actually compute the bounding volume by putting it around all of our
319 // states' bounding volumes.
320 pvector<const BoundingVolume *> child_volumes;
321
322 // We walk through the list of state defs indirectly, calling
323 // get_state_def() on each one, to ensure that the frames are updated
324 // correctly before we measure their bounding volumes.
325 for (int i = 0; i < (int)_state_defs.size(); i++) {
326 NodePath &root = ((PGItem *)this)->do_get_state_def(i);
327 if (!root.is_empty()) {
328 PandaNode *node = root.node();
329 child_volumes.push_back(node->get_bounds(current_thread));
330 num_vertices += node->get_nested_vertices(current_thread);
331 }
332 }
333
334 const BoundingVolume **child_begin = &child_volumes[0];
335 const BoundingVolume **child_end = child_begin + child_volumes.size();
336
337 bound->around(child_begin, child_end);
338
339 internal_bounds = bound;
340 internal_vertices = num_vertices;
341}

367void PGItem::
368xform(const LMatrix4 &mat) {
369 LightReMutexHolder holder(_lock);
370 // Transform the frame.
371 LPoint3 ll(_frame[0], 0.0f, _frame[2]);
372 LPoint3 ur(_frame[1], 0.0f, _frame[3]);
373 ll = ll * mat;
374 ur = ur * mat;
375 _frame.set(ll[0], ur[0], ll[2], ur[2]);
376
377 // Transform the individual states and their frame styles.
378 for (size_t state = 0; state < _state_defs.size(); ++state) {
379 StateDef &def = _state_defs[state];
380 NodePath &root = def._root;
381 // Apply the matrix to the previous transform.
382 root.set_transform(root.get_transform()->compose(TransformState::make_mat(mat)));
383
384 // Now flatten the transform into the subgraph.
385 SceneGraphReducer gr;
386 gr.apply_attribs(root.node());
387
388 // Transform the frame style too.
389 if (def._frame_style.xform(mat)) {
390#ifdef THREADED_PIPELINE
391 if (Pipeline::get_render_pipeline()->get_num_stages() > 1) {
392 update_frame((int)state);
393 }
394 else
395#endif
396 {
397 def._frame_stale = true;
398 }
399 }
400 }
401 mark_internal_bounds_stale();
402}

411bool PGItem::
412activate_region(const LMatrix4 &transform, int sort,
413 const ClipPlaneAttrib *cpa,
414 const ScissorAttrib *sa) {
415 using std::min;
416 using std::max;
417
418 LightReMutexHolder holder(_lock);
419 // Transform all four vertices, and get the new bounding box. This way the
420 // region works (mostly) even if has been rotated.
421 LPoint3 ll = LPoint3::rfu(_frame[0], 0.0f, _frame[2]) * transform;
422 LPoint3 lr = LPoint3::rfu(_frame[1], 0.0f, _frame[2]) * transform;
423 LPoint3 ul = LPoint3::rfu(_frame[0], 0.0f, _frame[3]) * transform;
424 LPoint3 ur = LPoint3::rfu(_frame[1], 0.0f, _frame[3]) * transform;
425 LVector3 up = LVector3::up();
426 int up_axis;
427 if (up[1]) {
428 up_axis = 1;
429 }
430 else if (up[2]) {
431 up_axis = 2;
432 }
433 else {
434 up_axis = 0;
435 }
436 LVector3 right = LVector3::right();
437 int right_axis;
438 if (right[0]) {
439 right_axis = 0;
440 }
441 else if (right[2]) {
442 right_axis = 2;
443 }
444 else {
445 right_axis = 1;
446 }
447
448 LVecBase4 frame;
449 if (cpa != nullptr && cpa->get_num_on_planes() != 0) {
450 // Apply the clip plane(s) andor scissor region now that we are here in
451 // world space.
452
453 ClipPoints points;
454 points.reserve(4);
455 points.push_back(LPoint2(ll[right_axis], ll[up_axis]));
456 points.push_back(LPoint2(lr[right_axis], lr[up_axis]));
457 points.push_back(LPoint2(ur[right_axis], ur[up_axis]));
458 points.push_back(LPoint2(ul[right_axis], ul[up_axis]));
459
460 int num_on_planes = cpa->get_num_on_planes();
461 for (int i = 0; i < num_on_planes; ++i) {
462 NodePath plane_path = cpa->get_on_plane(i);
463 LPlane plane = DCAST(PlaneNode, plane_path.node())->get_plane();
464 plane.xform(plane_path.get_net_transform()->get_mat());
465
466 // We ignore the forward axis, assuming the frame is still in the right-
467 // up plane after being transformed. Not sure if we really need to
468 // support general 3-D transforms on 2-D objects.
469 clip_frame(points, plane);
470 }
471
472 if (points.empty()) {
473 // Turns out it's completely clipped after all.
474 return false;
475 }
476
477 ClipPoints::iterator pi;
478 pi = points.begin();
479 frame.set((*pi)[0], (*pi)[0], (*pi)[1], (*pi)[1]);
480 ++pi;
481 while (pi != points.end()) {
482 frame[0] = min(frame[0], (*pi)[0]);
483 frame[1] = max(frame[1], (*pi)[0]);
484 frame[2] = min(frame[2], (*pi)[1]);
485 frame[3] = max(frame[3], (*pi)[1]);
486 ++pi;
487 }
488 } else {
489 // Since there are no clip planes involved, just set the frame.
490 frame.set(min(min(ll[right_axis], lr[right_axis]), min(ul[right_axis], ur[right_axis])),
491 max(max(ll[right_axis], lr[right_axis]), max(ul[right_axis], ur[right_axis])),
492 min(min(ll[up_axis], lr[up_axis]), min(ul[up_axis], ur[up_axis])),
493 max(max(ll[up_axis], lr[up_axis]), max(ul[up_axis], ur[up_axis])));
494 }
495
496 if (sa != nullptr) {
497 // Also restrict it to within the scissor region.
498 const LVecBase4 &sf = sa->get_frame();
499 // Expand sf from 0..1 to -1..1.
500 frame.set(max(frame[0], sf[0] * 2.0f - 1.0f),
501 min(frame[1], sf[1] * 2.0f - 1.0f),
502 max(frame[2], sf[2] * 2.0f - 1.0f),
503 min(frame[3], sf[3] * 2.0f - 1.0f));
504 if (frame[1] <= frame[0] || frame[3] <= frame[2]) {
505 // Completely outside the scissor region.
506 return false;
507 }
508 }
509
510 _region->set_frame(frame);
511
512 _region->set_sort(sort);
513 _region->set_active(true);
514
515 // calculate the inverse of this transform, which is needed to go back to
516 // the frame space.
517 _frame_inv_xform.invert_from(transform);
518
519 return true;
520}

864void PGItem::
865set_focus(bool focus) {
866 LightReMutexHolder holder(_lock);
867 if (focus) {
868 if (!get_active()) {
869 // Cannot set focus on an inactive item.
870 return;
871 }
872
873 // Set the keyboard focus to this item.
874 if (_focus_item != this) {
875 if (_focus_item != nullptr) {
876 // Clear the focus from whatever item currently has it.
877 _focus_item->set_focus(false);
878 }
879 _focus_item = this;
880 }
881 if (!get_focus()) {
882 focus_in();
883 _flags |= F_focus;
884 }
885

1117void PGItem::
1118reduce_region(LVecBase4 &frame, PGItem *obscurer) const {
1119 using std::min;
1120 using std::max;
1121
1122 if (obscurer != nullptr && !obscurer->is_overall_hidden()) {
1123 LVecBase4 oframe = get_relative_frame(obscurer);
1124
1125 // Determine the four rectangular regions on the four sides of the
1126 // obscuring region.
1127 LVecBase4 right(max(frame[0], oframe[1]), frame[1], frame[2], frame[3]);
1128 LVecBase4 left(frame[0], min(frame[1], oframe[0]), frame[2], frame[3]);
1129 LVecBase4 above(frame[0], frame[1], max(frame[2], oframe[3]), frame[3]);
1130 LVecBase4 below(frame[0], frame[1], frame[2], min(frame[3], oframe[2]));
1131
1132 // Now choose the largest of those four.
1133 const LVecBase4 *largest = &right;
1134 PN_stdfloat largest_area = compute_area(*largest);
1135 compare_largest(largest, largest_area, &left);
1136 compare_largest(largest, largest_area, &above);
1137 compare_largest(largest, largest_area, &below);
1138
1139 frame = *largest;
1140 }
1141}

1147LVecBase4 PGItem::
1148get_relative_frame(PGItem *item) const {
1149 using std::min;
1150 using std::max;
1151
1152 NodePath this_np = NodePath::any_path((PGItem *)this);
1153 NodePath item_np = this_np.find_path_to(item);
1154 if (item_np.is_empty()) {
1155 item_np = NodePath::any_path(item);
1156 }
1157 const LVecBase4 &orig_frame = item->get_frame();
1158 LMatrix4 transform = item_np.get_mat(this_np);
1159
1160 // Transform the item's frame into the PGScrollFrame's coordinate space.
1161 // Transform all four vertices, and get the new bounding box. This way the
1162 // region works (mostly) even if has been rotated.
1163 LPoint3 ll(orig_frame[0], 0.0f, orig_frame[2]);
1164 LPoint3 lr(orig_frame[1], 0.0f, orig_frame[2]);
1165 LPoint3 ul(orig_frame[0], 0.0f, orig_frame[3]);
1166 LPoint3 ur(orig_frame[1], 0.0f, orig_frame[3]);
1167 ll = ll * transform;
1168 lr = lr * transform;
1169 ul = ul * transform;
1170 ur = ur * transform;
1171
1172 return LVecBase4(min(min(ll[0], lr[0]), min(ul[0], ur[0])),
1173 max(max(ll[0], lr[0]), max(ul[0], ur[0])),
1174 min(min(ll[2], lr[2]), min(ul[2], ur[2])),
1175 max(max(ll[2], lr[2]), max(ul[2], ur[2])));
1176}

1293bool PGItem::
1294clip_frame(ClipPoints &source_points, const LPlane &plane) const {
1295 if (source_points.empty()) {
1296 return true;
1297 }
1298
1299 LPoint3 from3d;
1300 LVector3 delta3d;
1301 if (!plane.intersects_plane(from3d, delta3d, LPlane(LVector3(0, 1, 0), LPoint3::zero()))) {
1302 // The clipping plane is parallel to the polygon. The polygon is either
1303 // all in or all out.
1304 if (plane.dist_to_plane(LPoint3::zero()) < 0.0) {
1305 // A point within the polygon is behind the clipping plane: the polygon
1306 // is all in.
1307 return true;
1308 }
1309 return false;
1310 }
1311
1312 // Project the line of intersection into the X-Z plane. Now we have a 2-d
1313 // clipping line.
1314 LPoint2 from2d(from3d[0], from3d[2]);
1315 LVector2 delta2d(delta3d[0], delta3d[2]);
1316
1317 PN_stdfloat a = -delta2d[1];
1318 PN_stdfloat b = delta2d[0];
1319 PN_stdfloat c = from2d[0] * delta2d[1] - from2d[1] * delta2d[0];
1320
1321 // Now walk through the points. Any point on the left of our line gets
1322 // removed, and the line segment clipped at the point of intersection.
1323
1324 // We might increase the number of vertices by as many as 1, if the plane
1325 // clips off exactly one corner. (We might also decrease the number of
1326 // vertices, or keep them the same number.)
1327 ClipPoints new_points;
1328 new_points.reserve(source_points.size() + 1);
1329
1330 LPoint2 last_point(source_points.back());
1331 bool last_is_in = is_right(last_point - from2d, delta2d);
1332 bool all_in = last_is_in;
1333
1334 for (LPoint2 this_point : source_points) {
1335 bool this_is_in = is_right(this_point - from2d, delta2d);
1336
1337 // There appears to be a compiler bug in gcc 4.0: we need to extract this
1338 // comparison outside of the if statement.
1339 bool crossed_over = (this_is_in != last_is_in);
1340 if (crossed_over) {
1341 // We have just crossed over the clipping line. Find the point of
1342 // intersection.
1343 LVector2 d = this_point - last_point;
1344 PN_stdfloat denom = (a * d[0] + b * d[1]);
1345 if (denom != 0.0) {
1346 PN_stdfloat t = -(a * last_point[0] + b * last_point[1] + c) / denom;
1347 LPoint2 p = last_point + t * d;
1348
1349 new_points.push_back(p);
1350 last_is_in = this_is_in;
1351 }
1352 }
1353
1354 if (this_is_in) {
1355 // We are behind the clipping line. Keep the point.
1356 new_points.push_back(this_point);
1357 } else {
1358 all_in = false;
1359 }
1360
1361 last_point = this_point;
1362 }
1363
1364 source_points.swap(new_points);
1365 return all_in;
1366}
