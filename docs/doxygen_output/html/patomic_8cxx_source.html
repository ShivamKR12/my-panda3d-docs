36// Randomly pick an entry into the wait table based on the hash of the address.
37// It's possible to get hash collision, but that's not so bad, it just means
38// that the other thread will get a spurious wakeup.
39struct alignas(64) WaitTableEntry {
40 SRWLOCK _lock = SRWLOCK_INIT;
41 CONDITION_VARIABLE _cvar = CONDITION_VARIABLE_INIT;
42 DWORD _waiters = 0;
43};
44static WaitTableEntry _wait_table[64] = {};
45static const size_t _wait_hash_mask = 63;

50static void __stdcall
51emulated_wake(PVOID addr) {
52 size_t i = std::hash<volatile void *>{}(addr) & (sizeof(_wait_table) / sizeof(WaitTableEntry) - 1);
53 WaitTableEntry &entry = _wait_table[i];
54 AcquireSRWLockExclusive(&entry._lock);
55 DWORD num_waiters = entry._waiters;
56 ReleaseSRWLockExclusive(&entry._lock);
57 if (num_waiters > 0) {
58 // We have to wake up all the threads, even if only one of them is for this
59 // address. Some of them will get a spurious wakeup, but that's OK.
60 WakeAllConditionVariable(&entry._cvar);
61 }
62}

68static BOOL __stdcall
69emulated_wait(volatile VOID *addr, PVOID cmp, SIZE_T size, DWORD timeout) {
70 assert(size == sizeof(LONG));
71
72 LONG cmpval = *(LONG *)cmp;
73 if (*(LONG *)addr != cmpval) {
74 return TRUE;
75 }
76
77 size_t i = std::hash<volatile void *>{}(addr) & _wait_hash_mask;
78 WaitTableEntry &entry = _wait_table[i];
79 AcquireSRWLockExclusive(&entry._lock);
80 ++entry._waiters;
81 while (*(LONG *)addr == cmpval) {
82 if (SleepConditionVariableSRW(&entry._cvar, &entry._lock, timeout, 0) != 0) {
83 // Timeout.
84 --entry._waiters;
85 ReleaseSRWLockExclusive(&entry._lock);
86 return FALSE;
87 }
88 }
89 --entry._waiters;
90 ReleaseSRWLockExclusive(&entry._lock);
91 return TRUE;
92}

98static BOOL __stdcall
99initialize_wait(volatile VOID *addr, PVOID cmp, SIZE_T size, DWORD timeout) {
100 // There's a chance of a race here, with two threads trying to initialize the
101 // functions at the same time. That's OK, because they should all produce
102 // the same results, and the stores to the function pointers are atomic.
103 HMODULE lib = GetModuleHandleW(L"api-ms-win-core-synch-l1-2-0.dll");
104 if (lib) {
105 auto wait_func = (decltype(_patomic_wait_func))GetProcAddress(lib, "WaitOnAddress");
106 auto wake_one_func = (decltype(_patomic_wake_one_func))GetProcAddress(lib, "WakeByAddressSingle");
107 auto wake_all_func = (decltype(_patomic_wake_all_func))GetProcAddress(lib, "WakeByAddressAll");
108 if (wait_func && wake_one_func && wake_all_func) {
109 // Make sure that the wake function is guaranteed to be visible to other
110 // threads by the time we assign the wait function.
111 _patomic_wake_one_func = wake_one_func;
112 _patomic_wake_all_func = wake_all_func;
113 patomic_thread_fence(std::memory_order_release);
114 _patomic_wait_func = wait_func;
115 return wait_func(addr, cmp, size, timeout);
116 }
117 }
118
119 // We don't have Windows 8's functions, use the emulated wait and wake funcs.
120 _patomic_wake_one_func = &emulated_wake;
121 _patomic_wake_all_func = &emulated_wake;
122 patomic_thread_fence(std::memory_order_release);
123 _patomic_wait_func = &emulated_wait;
124
125 return emulated_wait(addr, cmp, size, timeout);
126}

131struct alignas(64) WaitTableEntry {
132 pthread_mutex_t _lock = PTHREAD_MUTEX_INITIALIZER;
133 pthread_cond_t _cvar = PTHREAD_COND_INITIALIZER;
134 unsigned int _waiters = 0;
135};

142void
143_patomic_wait(const volatile uint32_t *value, uint32_t old) {
144 WaitTableEntry &entry = _wait_table[std::hash<const volatile void *>{}(value) & _wait_hash_mask];
145 pthread_mutex_lock(&entry._lock);
146 ++entry._waiters;
147 while (__atomic_load_n(value, __ATOMIC_SEQ_CST) == old) {
148 pthread_cond_wait(&entry._cvar, &entry._lock);
149 }
150 --entry._waiters;
151 pthread_mutex_unlock(&entry._lock);
152}

157void
158_patomic_notify_all(volatile uint32_t *value) {
159 WaitTableEntry &entry = _wait_table[std::hash<const volatile void *>{}(value) & _wait_hash_mask];
160 pthread_mutex_lock(&entry._lock);
161 unsigned int num_waiters = entry._waiters;
162 pthread_mutex_unlock(&entry._lock);
163 if (num_waiters > 0) {
164 pthread_cond_broadcast(&entry._cvar);
165 }
