104void CLP(TextureContext)::
105set_num_views(int num_views) {
106 if (_num_views > num_views) {
107 glDeleteTextures(_num_views - num_views, _indices + _num_views);
108
109 if (_buffers != nullptr) {
110 _glgsg->_glDeleteBuffers(_num_views - num_views, _buffers + num_views);
111 }
112
113 if (num_views <= 1) {
114 _index = _indices[0];
115 if (_indices != &_index) {
116 delete[] _indices;
117 _indices = &_index;
118 }
119
120#ifndef OPENGLES_1
121 if (_buffers != nullptr) {
122 _buffer = _buffers[0];
123 if (_buffers != &_buffer) {
124 delete[] _buffers;
125 _buffers = &_buffer;
126 }
127 if (num_views == 0) {
128 _buffers = nullptr;
129 }
130 }
131#endif
132 }
133 }
134 else if (_num_views == 0 && num_views == 1) {
135 glGenTextures(1, &_index);
136 _indices = &_index;
137
138#ifndef OPENGLES_1
139 if (_target == GL_TEXTURE_BUFFER) {
140 _glgsg->_glGenBuffers(1, &_buffer);
141 _buffers = &_buffer;
142 }
143#endif
144 }
145 else if (_num_views < num_views) {
146 GLuint *new_indices = new GLuint[num_views];
147 memcpy(new_indices, _indices, sizeof(GLuint) * _num_views);
148 glGenTextures(num_views - _num_views, new_indices + _num_views);
149 if (_indices != &_index) {
150 delete[] _indices;
151 }
152 _indices = new_indices;
153
154#ifndef OPENGLES_1
155 if (_target == GL_TEXTURE_BUFFER) {
156 GLuint *new_buffers = new GLuint[num_views];
157 if (_buffers != nullptr) {
158 memcpy(new_buffers, _buffers, sizeof(GLuint) * _num_views);
159 _glgsg->_glGenBuffers(num_views - _num_views, new_buffers + _num_views);
160 if (_buffers != &_buffer) {
161 delete[] _buffers;
162 }
163 } else {
164 _glgsg->_glGenBuffers(num_views, new_buffers);
165 }
166 _buffers = new_buffers;
167 }
168#endif
169 }
170
171 _num_views = num_views;
172}

180bool CLP(TextureContext)::
181needs_barrier(GLbitfield barrier, bool writing) {
182 if (!gl_enable_memory_barriers) {
183 return false;
184 }
185
186 if (barrier & GL_TEXTURE_FETCH_BARRIER_BIT) {
187 // This is always a read, so only sync RAW.
188 if (_glgsg->_texture_fetch_barrier_counter == _texture_fetch_barrier_counter) {
189 return true;
190 }
191 }
192
193 if (barrier & GL_SHADER_IMAGE_ACCESS_BARRIER_BIT) {
194 // Sync WAR, WAW and RAW, but not RAR.
195 if ((writing && _glgsg->_shader_image_access_barrier_counter == _shader_image_read_barrier_counter) ||
196 (_glgsg->_shader_image_access_barrier_counter == _shader_image_write_barrier_counter)) {
197 return true;
198 }
199 }
200
201 if (barrier & GL_TEXTURE_UPDATE_BARRIER_BIT) {
202 if ((writing && _glgsg->_texture_update_barrier_counter == _texture_read_barrier_counter) ||
203 (_glgsg->_texture_update_barrier_counter == _texture_write_barrier_counter)) {
204 return true;
205 }
206 }
207
208 if (barrier & GL_FRAMEBUFFER_BARRIER_BIT) {
209 if ((writing && _glgsg->_framebuffer_barrier_counter == _framebuffer_read_barrier_counter) ||
210 (_glgsg->_framebuffer_barrier_counter == _framebuffer_write_barrier_counter)) {
211 return true;
212 }
213 }
214
215 return false;
216}

222void CLP(TextureContext)::
223mark_incoherent(bool wrote) {
224 if (!gl_enable_memory_barriers) {
225 return;
226 }
227
228 // If we only read from it, the next read operation won't need another
229 // barrier, since it'll be reading the same data.
230 if (wrote) {
231 _texture_fetch_barrier_counter = _glgsg->_texture_fetch_barrier_counter;
232 _shader_image_write_barrier_counter = _glgsg->_shader_image_access_barrier_counter;
233 _texture_write_barrier_counter = _glgsg->_shader_image_access_barrier_counter;
234 _framebuffer_write_barrier_counter = _glgsg->_framebuffer_barrier_counter;
235 }
236
237 // We could still write to it before we read from it, so we have to always
238 // insert these barriers.
239 _shader_image_read_barrier_counter = _glgsg->_shader_image_access_barrier_counter;
240 _texture_read_barrier_counter = _glgsg->_texture_update_barrier_counter;
241 _framebuffer_read_barrier_counter = _glgsg->_framebuffer_barrier_counter;
242}

249void CLP(TextureContext)::
250return_pbo(GLuint pbo, size_t size) {
251 // Also triggers when the number of buffers is -1 (which effectively means
252 // to always delete the buffers after use).
253 if (_num_pbos > get_texture()->get_num_async_transfer_buffers() ||
254 size < _pbo_size) {
255 // We have too many PBOs, or this PBO is no longer of the proper
256 // size, so delete it rather than returning it to the pool.
257 _num_pbos--;
258 _glgsg->_glDeleteBuffers(1, &pbo);
259 } else {
260 _unused_pbos.push_front(pbo);
261 }
