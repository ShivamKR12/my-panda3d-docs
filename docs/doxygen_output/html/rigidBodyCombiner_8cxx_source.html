135bool RigidBodyCombiner::
136cull_callback(CullTraverser *trav, CullTraverserData &data) {
137 // Pretend that all of our transforms have been modified (since we don't
138 // really know which ones have).
139 Thread *current_thread = Thread::get_current_thread();
140 Transforms::iterator ti;
141 for (ti = _internal_transforms.begin();
142 ti != _internal_transforms.end();
143 ++ti) {
144 (*ti)->mark_modified(current_thread);
145 }
146
147 // Render the internal scene only--this is the optimized scene.
148 trav->traverse_down(data, _internal_root);
149
150 // Do not directly render the nodes beneath this node.
151 return false;
152}

159void RigidBodyCombiner::
160r_collect(PandaNode *node, const RenderState *state,
161 const VertexTransform *transform) {
162 CPT(RenderState) next_state = state->compose(node->get_state());
163 CPT(VertexTransform) next_transform = transform;
164 if (!node->get_transform()->is_identity() ||
165 (node->is_of_type(ModelNode::get_class_type()) &&
166 DCAST(ModelNode, node)->get_preserve_transform() != ModelNode::PT_none)) {
167 // This node has a transform we need to keep.
168 PT(NodeVertexTransform) new_transform = new NodeVertexTransform(node, transform);
169 _internal_transforms.push_back(new_transform);
170 next_transform = new_transform.p();
171
172 }
173
174 if (node->is_geom_node()) {
175 GeomNode *gnode = DCAST(GeomNode, node);
176 GeomNode *root_gnode = DCAST(GeomNode, _internal_root);
177
178 int num_geoms = gnode->get_num_geoms();
179 for (int i = 0; i < num_geoms; ++i) {
180 PT(Geom) geom = gnode->get_geom(i)->make_copy();
181 if (next_transform != nullptr) {
182 geom->set_vertex_data(convert_vd(next_transform, geom->get_vertex_data()));
183 }
184 CPT(RenderState) gstate = next_state->compose(gnode->get_geom_state(i));
185 root_gnode->add_geom(geom, gstate);
186 }
187 }
188
189 Children cr = node->get_children();
190 int num_children = cr.get_num_children();
191 for (int i = 0; i < num_children; i++) {
192 r_collect(cr.get_child(i), next_state, next_transform);
193 }
194}

200PT(GeomVertexData) RigidBodyCombiner::
201convert_vd(const VertexTransform *transform, const GeomVertexData *orig) {
202 // First, unify this operation for unique transformdata combinations. If we
203 // encounter a given GeomVertexData more than once under the same transform,
204 // we should return exactly the same GeomVertexData.
205 VDTable::iterator vdti = _vd_table.find(VDUnifier(transform, orig));
206 if (vdti != _vd_table.end()) {
207 return (*vdti).second;
208 }
209
210 PT(GeomVertexFormat) format = new GeomVertexFormat(*orig->get_format());
211 if (!orig->get_format()->has_column(InternalName::get_transform_blend())) {
212 PT(GeomVertexArrayFormat) af = new GeomVertexArrayFormat();
213 af->add_column(InternalName::get_transform_blend(), 1,
214 Geom::NT_uint16, Geom::C_index, 0, 2);
215 format->add_array(af);
216 }
217
218 GeomVertexAnimationSpec spec;
219 spec.set_panda();
220 format->set_animation(spec);
221 format->maybe_align_columns_for_animation();
222
223 CPT(GeomVertexFormat) new_format = GeomVertexFormat::register_format(format);
224 CPT(GeomVertexData) converted = orig->convert_to(new_format);
225 PT(GeomVertexData) new_data = new GeomVertexData(*converted);
226
227 if (new_data->get_transform_blend_table() == nullptr) {
228 // Create a new table that has just the one blend: all vertices hard-
229 // assigned to the indicated transform.
230 PT(TransformBlendTable) new_table = new TransformBlendTable;
231 new_table->add_blend(TransformBlend(transform, 1.0f));
232 new_table->set_rows(SparseArray::range(0, new_data->get_num_rows()));
233 new_data->set_transform_blend_table(new_table);
234 } else {
235 // The GeomVertexData already has a TransformBlendTable. In this case,
236 // we'll have to adjust it. TODO.
237 }
238
239 // Store the result for the next time.
240 _vd_table[VDUnifier(transform, orig)] = new_data;
241
242 return new_data;
243}
