189void DXShaderContext9::
190issue_parameters(GSG *gsg, int altered) {
191#ifdef HAVE_CG
192 if (_cg_program) {
193 if (altered & _shader->_mat_deps) {
194 if (altered & _shader->_mat_cache_deps) {
195 gsg->update_shader_matrix_cache(_shader, _mat_part_cache, altered);
196 }
197
198 LMatrix4f scratch;
199
200 for (Shader::ShaderMatSpec &spec : _shader->_mat_spec) {
201 if ((altered & spec._dep) == 0) {
202 continue;
203 }
204
205 CGparameter p = _cg_parameter_map[spec._id._seqno];
206 if (p == nullptr) {
207 continue;
208 }
209
210 const LVecBase4f *val = gsg->fetch_specified_value(spec, _mat_part_cache, _mat_scratch_space);
211 if (val) {
212 const float *data = (const float *)val + spec._offset;
213 LVecBase4f v;
214 LMatrix4f temp_matrix;
215 LMatrix3f temp_matrix3;
216
217 switch (spec._piece) {
218 case Shader::SMP_mat4_whole:
219 // TRANSPOSE REQUIRED
220 temp_matrix.transpose_from(*(const LMatrix4f *)data);
221 data = temp_matrix.get_data();
222 break;
223
224 case Shader::SMP_mat4_column:
225 v.set(data[0], data[4], data[8], data[12]);
226 data = v.get_data();
227 break;
228
229 case Shader::SMP_mat4_upper3x3:
230 // TRANSPOSE REQUIRED
231 temp_matrix3.set(data[0], data[4], data[8], data[1], data[5], data[9], data[2], data[6], data[10]);
232 data = temp_matrix3.get_data();
233 break;
234
235 case Shader::SMP_mat4_transpose3x3:
236 // NO TRANSPOSE REQUIRED
237 temp_matrix3.set(data[0], data[1], data[2], data[4], data[5], data[6], data[8], data[9], data[10]);
238 data = temp_matrix3.get_data();
239 break;
240
241 default:
242 break;
243 }
244
245 HRESULT hr = cgD3D9SetUniform(p, data);
246 if (FAILED(hr)) {
247 std::string name = "unnamed";
248
249 if (spec._arg[0]) {
250 name = spec._arg[0]->get_basename();
251 }
252
253 dxgsg9_cat.error()
254 << "NAME " << name << "\n" << "MAT TYPE " << spec._piece
255 << " cgD3D9SetUniform failed " << D3DERRORSTRING(hr);
256
257 CGerror error = cgGetError();
258 if (error != CG_NO_ERROR) {
259 dxgsg9_cat.error() << " CG ERROR: " << cgGetErrorString(error) << "\n";
260 }
261 }
262 }
263 }
264 }
265 }
266#endif

289bool DXShaderContext9::
290update_shader_vertex_arrays(DXShaderContext9 *prev, GSG *gsg, bool force) {
291 if (prev) prev->disable_shader_vertex_arrays(gsg);
292#ifdef HAVE_CG
293 if (!_cg_program) {
294 return true;
295 }
296
297#ifdef SUPPORT_IMMEDIATE_MODE
298/*
299 if (gsg->_use_sender) {
300 dxgsg9_cat.error() << "immediate mode shaders not implemented yet\n";
301 } else
302*/
303#endif // SUPPORT_IMMEDIATE_MODE
304 {
305 int nvarying = _shader->_var_spec.size();
306 LPDIRECT3DDEVICE9 device = gsg->_screen->_d3d_device;
307 HRESULT hr;
308
309 // Discard and recreate the VertexElementArray. This thrashes pretty
310 // bad....
311 if (_vertex_element_array != nullptr) {
312 delete _vertex_element_array;
313 }
314 _vertex_element_array = new VertexElementArray(nvarying + 2);
315 VertexElementArray* vertex_element_array = _vertex_element_array;
316
317 // Experimentally determined that DX doesn't like us crossing the streams!
318 // It seems to be okay with out-of-order offsets in both source and
319 // destination, but it wants all stream X entries grouped together, then
320 // all stream Y entries, etc. To accomplish this out outer loop processes
321 // arrays ("streams"), and we repeatedly iterate the parameters to pull
322 // out only those for a single stream.
323
324 bool apply_white_color = false;
325
326 int number_of_arrays = gsg->_data_reader->get_num_arrays();
327 for (int array_index = 0; array_index < number_of_arrays; ++array_index) {
328 const GeomVertexArrayDataHandle* array_reader =
329 gsg->_data_reader->get_array_reader(array_index);
330 if (array_reader == nullptr) {
331 dxgsg9_cat.error() << "Unable to get reader for array " << array_index << "\n";
332 continue;
333 }
334
335 for (int var_index = 0; var_index < nvarying; ++var_index) {
336 CGparameter p = _cg_parameter_map[_shader->_var_spec[var_index]._id._seqno];
337 if (p == nullptr) {
338 dxgsg9_cat.info() <<
339 "No parameter in map for parameter " << var_index <<
340 " (probably optimized away)\n";
341 continue;
342 }
343
344 InternalName *name = _shader->_var_spec[var_index]._name;
345
346 // This is copied from the GL version of this function, and I've yet
347 // to 100% convince myself that it works properly....
348 int texslot = _shader->_var_spec[var_index]._append_uv;
349 if (texslot >= 0 && texslot < gsg->_state_texture->get_num_on_stages()) {
350 TextureStage *stage = gsg->_state_texture->get_on_stage(texslot);
351 InternalName *texname = stage->get_texcoord_name();
352 if (name == InternalName::get_texcoord()) {
353 name = texname;
354 } else if (texname != InternalName::get_texcoord()) {
355 name = name->append(texname->get_basename());
356 }
357 }
358
359 if (name == InternalName::get_color() && !gsg->_vertex_colors_enabled) {
360 apply_white_color = true;
361 continue;
362 }
363
364 const GeomVertexArrayDataHandle *param_array_reader;
365 Geom::NumericType numeric_type;
366 int num_values, start, stride;
367 if (!gsg->_data_reader->get_array_info(name, param_array_reader,
368 num_values, numeric_type,
369 start, stride)) {
370 // This is apparently not an error (actually I think it is, just not
371 // a fatal one). The GL implementation fails silently in this case,
372 // but the net result is that we end up not supplying input for a
373 // shader parameter, which can cause Bad Things to happen so I'd
374 // like to at least get a hint as to what's gone wrong.
375 dxgsg9_cat.info() << "Geometry contains no data for shader parameter " << *name << "\n";
376 if (name == InternalName::get_color()) {
377 apply_white_color = true;
378 }
379 continue;
380 }
381
382 // If not associated with the array we're working on, move on.
383 if (param_array_reader != array_reader) {
384 continue;
385 }
386
387 const char *semantic = cgGetParameterSemantic(p);
388 if (semantic == nullptr) {
389 dxgsg9_cat.error() << "Unable to retrieve semantic for parameter " << var_index << "\n";
390 continue;
391 }
392
393 if (strncmp(semantic, "POSITION", strlen("POSITION")) == 0) {
394 if (numeric_type == Geom::NT_float32) {
395 switch (num_values) {
396 case 3:
397 vertex_element_array->add_position_xyz_vertex_element(array_index, start);
398 break;
399 case 4:
400 vertex_element_array->add_position_xyzw_vertex_element(array_index, start);
401 break;
402 default:
403 dxgsg9_cat.error() << "VE ERROR: invalid number of vertex coordinate elements " << num_values << "\n";
404 break;
405 }
406 } else {
407 dxgsg9_cat.error() << "VE ERROR: invalid vertex type " << numeric_type << "\n";
408 }
409 } else if (strncmp(semantic, "TEXCOORD", strlen("TEXCOORD")) == 0) {
410 int slot = atoi(semantic + strlen("TEXCOORD"));
411 if (numeric_type == Geom::NT_float32) {
412 switch (num_values) {
413 case 1:
414 vertex_element_array->add_u_vertex_element(array_index, start, slot);
415 break;
416 case 2:
417 vertex_element_array->add_uv_vertex_element(array_index, start, slot);
418 break;
419 case 3:
420 vertex_element_array->add_uvw_vertex_element(array_index, start, slot);
421 break;
422 case 4:
423 vertex_element_array->add_xyzw_vertex_element(array_index, start, slot);
424 break;
425 default:
426 dxgsg9_cat.error() << "VE ERROR: invalid number of vertex texture coordinate elements " << num_values << "\n";
427 break;
428 }
429 } else {
430 dxgsg9_cat.error() << "VE ERROR: invalid texture coordinate type " << numeric_type << "\n";
431 }
432 } else if (strncmp(semantic, "COLOR", strlen("COLOR")) == 0) {
433 if (numeric_type == Geom::NT_packed_dcba ||
434 numeric_type == Geom::NT_packed_dabc ||
435 numeric_type == Geom::NT_uint8) {
436 switch (num_values) {
437 case 4:
438 vertex_element_array->add_diffuse_color_vertex_element(array_index, start);
439 break;
440 default:
441 dxgsg9_cat.error() << "VE ERROR: invalid color coordinates " << num_values << "\n";
442 break;
443 }
444 } else {
445 dxgsg9_cat.error() << "VE ERROR: invalid color type " << numeric_type << "\n";
446 }
447 } else if (strncmp(semantic, "NORMAL", strlen("NORMAL")) == 0) {
448 if (numeric_type == Geom::NT_float32) {
449 switch (num_values) {
450 case 3:
451 vertex_element_array->add_normal_vertex_element(array_index, start);
452 break;
453 default:
454 dxgsg9_cat.error() << "VE ERROR: invalid number of normal coordinate elements " << num_values << "\n";
455 break;
456 }
457 } else {
458 dxgsg9_cat.error() << "VE ERROR: invalid normal type " << numeric_type << "\n";
459 }
460 } else if (strncmp(semantic, "BINORMAL", strlen("BINORMAL")) == 0) {
461 if (numeric_type == Geom::NT_float32) {
462 switch (num_values) {
463 case 3:
464 vertex_element_array->add_binormal_vertex_element(array_index, start);
465 break;
466 default:
467 dxgsg9_cat.error() << "VE ERROR: invalid number of binormal coordinate elements " << num_values << "\n";
468 break;
469 }
470 } else {
471 dxgsg9_cat.error() << "VE ERROR: invalid binormal type " << numeric_type << "\n";
472 }
473 } else if (strncmp(semantic, "TANGENT", strlen("TANGENT")) == 0) {
474 if (numeric_type == Geom::NT_float32) {
475 switch (num_values) {
476 case 3:
477 vertex_element_array->add_tangent_vertex_element(array_index, start);
478 break;
479 default:
480 dxgsg9_cat.error() << "VE ERROR: invalid number of tangent coordinate elements " << num_values << "\n";
481 break;
482 }
483 } else {
484 dxgsg9_cat.error() << "VE ERROR: invalid tangent type " << numeric_type << "\n";
485 }
486 } else {
487 dxgsg9_cat.error() << "Unsupported semantic " << semantic << " for parameter " << var_index << "\n";
488 }
489 }
490
491 // Get the vertex buffer for this array.
492 DXVertexBufferContext9 *dvbc;
493 if (!gsg->setup_array_data(dvbc, array_reader, force)) {
494 dxgsg9_cat.error() << "Unable to setup vertex buffer for array " << array_index << "\n";
495 continue;
496 }
497
498 // Bind this array as the data source for the corresponding stream.
499 const GeomVertexArrayFormat *array_format = array_reader->get_array_format();
500 hr = device->SetStreamSource(array_index, dvbc->_vbuffer, 0, array_format->get_stride());
501 if (FAILED(hr)) {
502 dxgsg9_cat.error() << "SetStreamSource failed" << D3DERRORSTRING(hr);
503 }
504 }
505
506 _num_bound_streams = number_of_arrays;
507
508 if (apply_white_color) {
509 // The shader needs a vertex color, but vertex colors are disabled.
510 // Bind a vertex buffer containing only one white colour.
511 int array_index = number_of_arrays;
512 LPDIRECT3DVERTEXBUFFER9 vbuffer = gsg->get_white_vbuffer();
513 hr = device->SetStreamSource(array_index, vbuffer, 0, 0);
514 if (FAILED(hr)) {
515 dxgsg9_cat.error() << "SetStreamSource failed" << D3DERRORSTRING(hr);
516 }
517 vertex_element_array->add_diffuse_color_vertex_element(array_index, 0);
518 ++_num_bound_streams;
519 }
520
521 if (_vertex_element_array != nullptr &&
522 _vertex_element_array->add_end_vertex_element()) {
523 if (dxgsg9_cat.is_debug()) {
524 // Note that the currently generated vertex declaration works but
525 // never validates. My theory is that this is due to the shader
526 // programs always using float4 whereas the vertex declaration
527 // correctly sets the number of inputs (float2, float3, etc.).
528 if (cgD3D9ValidateVertexDeclaration(_cg_program,
529 _vertex_element_array->_vertex_element_array) == CG_TRUE) {
530 dxgsg9_cat.debug() << "cgD3D9ValidateVertexDeclaration succeeded\n";
531 } else {
532 dxgsg9_cat.debug() << "cgD3D9ValidateVertexDeclaration failed\n";
533 }
534 }
535
536 // Discard the old VertexDeclaration. This thrashes pretty bad....
537 if (_vertex_declaration != nullptr) {
538 _vertex_declaration->Release();
539 _vertex_declaration = nullptr;
540 }
541
542 hr = device->CreateVertexDeclaration(_vertex_element_array->_vertex_element_array,
543 &_vertex_declaration);
544 if (FAILED(hr)) {
545 dxgsg9_cat.error() << "CreateVertexDeclaration failed" << D3DERRORSTRING(hr);
546 } else {
547 hr = device->SetVertexDeclaration(_vertex_declaration);
548 if (FAILED(hr)) {
549 dxgsg9_cat.error() << "SetVertexDeclaration failed" << D3DERRORSTRING(hr);
550 }
551 }
552 } else {
553 dxgsg9_cat.error() << "VertexElementArray creation failed\n";
554 }
555 }
556#endif // HAVE_CG
557
558 return true;
559}
