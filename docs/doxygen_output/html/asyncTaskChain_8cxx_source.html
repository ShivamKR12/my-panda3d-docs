436void AsyncTaskChain::
437do_add(AsyncTask *task) {
438 nassertv(task->_chain == nullptr &&
439 task->_manager == nullptr &&
440 task->_chain_name == get_name() &&
441 task->_state == AsyncTask::S_inactive);
442 nassertv(!do_has_task(task));
443
444 do_start_threads();
445
446 task->_chain = this;
447 task->_manager = _manager;
448
449 double now = _manager->_clock->get_frame_time();
450 task->_start_time = now;
451 task->_start_frame = _manager->_clock->get_frame_count();
452
453 // Remember the order in which tasks were added to the chain.
454 task->_implicit_sort = _next_implicit_sort++;
455
456 _manager->add_task_by_name(task);
457
458 if (task->has_delay()) {
459 // This is a deferred task. Add it to the sleeping queue.
460 task->_wake_time = now + task->get_delay();
461 task->_start_time = task->_wake_time;
462 task->_state = AsyncTask::S_sleeping;
463 _sleeping.push_back(task);
464 push_heap(_sleeping.begin(), _sleeping.end(), AsyncTaskSortWakeTime());
465
466 } else {
467 // This is an active task. Add it to the active set.
468 task->_state = AsyncTask::S_active;
469 if (task_cat.is_spam()) {
470 task_cat.spam()
471 << "Adding " << *task << " with sort " << task->get_sort()
472 << " to chain " << get_name() << " with current_sort "
473 << _current_sort << "\n";
474 }
475 if (task->get_sort() >= _current_sort) {
476 // It will run this frame.
477 _active.push_back(task);
478 push_heap(_active.begin(), _active.end(), AsyncTaskSortPriority());
479 } else {
480 // It will run next frame.
481 _next_active.push_back(task);
482 }
483 }
484 ++_num_tasks;
485 ++(_manager->_num_tasks);
486 _needs_cleanup = true;
487
488 _cvar.notify_all();
489}

496bool AsyncTaskChain::
497do_remove(AsyncTask *task, bool upon_death) {
498 nassertr(task->_chain == this, false);
499
500 switch (task->_state) {
501 case AsyncTask::S_servicing:
502 // This task is being serviced. upon_death will be called afterwards.
503 task->_state = AsyncTask::S_servicing_removed;
504 return true;
505
506 case AsyncTask::S_servicing_removed:
507 // Being serviced, though it is already marked to be removed afterwards.
508 return false;
509
510 case AsyncTask::S_sleeping:
511 // Sleeping, easy.
512 {
513 int index = find_task_on_heap(_sleeping, task);
514 nassertr(index != -1, false);
515 PT(AsyncTask) hold_task = task;
516 _sleeping.erase(_sleeping.begin() + index);
517 make_heap(_sleeping.begin(), _sleeping.end(), AsyncTaskSortWakeTime());
518 cleanup_task(task, upon_death, false);
519 }
520 return true;
521
522 case AsyncTask::S_active:
523 {
524 // Active, but not being serviced, easy.
525 PT(AsyncTask) hold_task = task;
526 int index = find_task_on_heap(_active, task);
527 if (index != -1) {
528 _active.erase(_active.begin() + index);
529 make_heap(_active.begin(), _active.end(), AsyncTaskSortPriority());
530 } else {
531 index = find_task_on_heap(_next_active, task);
532 if (index != -1) {
533 _next_active.erase(_next_active.begin() + index);
534 } else {
535 index = find_task_on_heap(_this_active, task);
536 nassertr(index != -1, false);
537 }
538 }
539 cleanup_task(task, upon_death, false);
540 return true;
541 }
542
543 default:
544 break;
545 }
546
547 return false;
548}

585void AsyncTaskChain::
586do_cleanup() {
587 if (task_cat.is_spam()) {
588 do_output(task_cat.spam());
589 task_cat.spam(false)
590 << ": do_cleanup()\n";
591 }
592 do_stop_threads();
593 _num_threads = 0;
594
595 // Don't call the upon_death functions while we clean up the tasks.
596 // Instead, store all the tasks in a list as we clean them up, and then call
597 // the upon_death functions all at once. We do this because calling
598 // upon_death wil release the lock, allowing the iterators to become
599 // invalid.
600
601 TaskHeap dead;
602 dead.reserve(_num_tasks);
603
604 _needs_cleanup = false;
605
606 TaskHeap::const_iterator ti;
607 for (ti = _active.begin(); ti != _active.end(); ++ti) {
608 AsyncTask *task = (*ti);
609 dead.push_back(task);
610 cleanup_task(task, false, false);
611 }
612 for (ti = _this_active.begin(); ti != _this_active.end(); ++ti) {
613 AsyncTask *task = (*ti);
614 dead.push_back(task);
615 cleanup_task(task, false, false);
616 }
617 for (ti = _next_active.begin(); ti != _next_active.end(); ++ti) {
618 AsyncTask *task = (*ti);
619 dead.push_back(task);
620 cleanup_task(task, false, false);
621 }
622 for (ti = _sleeping.begin(); ti != _sleeping.end(); ++ti) {
623 AsyncTask *task = (*ti);
624 dead.push_back(task);
625 cleanup_task(task, false, false);
626 }
627
628 // There might still be one task remaining: the currently-executing task.
629 nassertv(_num_tasks == 0 || _num_tasks == 1);
630
631 // Now go back and call the upon_death functions.
632 _manager->_lock.unlock();
633 for (ti = dead.begin(); ti != dead.end(); ++ti) {
634 (*ti)->upon_death(_manager, false);
635 }
636 _manager->_lock.lock();
637
638 if (task_cat.is_spam()) {
639 do_output(task_cat.spam());
640 task_cat.spam(false)
641 << ": done do_cleanup()\n";
642 }
643}

681void AsyncTaskChain::
682service_one_task(AsyncTaskChain::AsyncTaskChainThread *thread) {
683 if (!_active.empty()) {
684 PT(AsyncTask) task = _active.front();
685 pop_heap(_active.begin(), _active.end(), AsyncTaskSortPriority());
686 _active.pop_back();
687
688 if (thread != nullptr) {
689 thread->_servicing = task;
690 }
691
692 if (task_cat.is_spam()) {
693 task_cat.spam()
694 << "Servicing " << *task << " in "
695 << *Thread::get_current_thread() << "\n";
696 }
697
698 nassertv(task->get_sort() == _current_sort);
699 nassertv(task->_state == AsyncTask::S_active);
700 task->_state = AsyncTask::S_servicing;
701 task->_servicing_thread = thread;
702
703 AsyncTask::DoneStatus ds = task->unlock_and_do_task();
704
705 if (thread != nullptr) {
706 thread->_servicing = nullptr;
707 }
708 task->_servicing_thread = nullptr;
709
710 if (task->_chain == this) {
711 if (task->_state == AsyncTask::S_servicing_removed) {
712 // This task wants to kill itself.
713 cleanup_task(task, true, false);
714
715 } else if (task->_chain_name != get_name()) {
716 // The task wants to jump to a different chain.
717 PT(AsyncTask) hold_task = task;
718 cleanup_task(task, false, false);
719 task->jump_to_task_chain(_manager);
720
721 } else {
722 switch (ds) {
723 case AsyncTask::DS_cont:
724 // The task is still alive; put it on the next frame's active queue.
725 task->_state = AsyncTask::S_active;
726 _next_active.push_back(task);
727 _cvar.notify_all();
728 break;
729
730 case AsyncTask::DS_again:
731 // The task wants to sleep again.
732 {
733 double now = _manager->_clock->get_frame_time();
734 task->_wake_time = now + task->get_delay();
735 task->_start_time = task->_wake_time;
736 task->_state = AsyncTask::S_sleeping;
737 _sleeping.push_back(task);
738 push_heap(_sleeping.begin(), _sleeping.end(), AsyncTaskSortWakeTime());
739 if (task_cat.is_spam()) {
740 task_cat.spam()
741 << "Sleeping " << *task << ", wake time at "
742 << task->_wake_time - now << "\n";
743 }
744 _cvar.notify_all();
745 }
746 break;
747
748 case AsyncTask::DS_pickup:
749 // The task wants to run again this frame if possible.
750 task->_state = AsyncTask::S_active;
751 _this_active.push_back(task);
752 _cvar.notify_all();
753 break;
754
755 case AsyncTask::DS_interrupt:
756 // The task had an exception and wants to raise a big flag.
757 task->_state = AsyncTask::S_active;
758 _next_active.push_back(task);
759 if (_state == S_started) {
760 _state = S_interrupted;
761 _cvar.notify_all();
762 }
763 break;
764
765 case AsyncTask::DS_await:
766 // The task wants to wait for another one to finish.
767 task->_state = AsyncTask::S_awaiting;
768 _cvar.notify_all();
769 ++_num_awaiting_tasks;
770 break;
771
772 default:
773 // The task has finished.
774 cleanup_task(task, true, true);
775 }
776 }
777 } else {
778 task_cat.error()
779 << "Task is no longer on chain " << get_name()
780 << ": " << *task << "\n";
781 }
782
783 if (task_cat.is_spam()) {
784 task_cat.spam()
785 << "Done servicing " << *task << " in "
786 << *Thread::get_current_thread() << "\n";
787 }
788 }
789 thread_consider_yield();
790}

801void AsyncTaskChain::
802cleanup_task(AsyncTask *task, bool upon_death, bool clean_exit) {
803 if (task_cat.is_spam()) {
804 do_output(task_cat.spam());

806 << ": cleanup_task(" << *task << ", " << upon_death << ", " << clean_exit
807 << ")\n";
808 }
809
810 nassertv(task->_chain == this);
811
812 task->_state = AsyncTask::S_inactive;

840bool AsyncTaskChain::
841finish_sort_group() {
842 nassertr(_num_busy_threads == 0, true);
843
844#ifdef HAVE_THREADS
845 if (!_threads.empty()) {
846 PStatClient::thread_tick(get_name());
847 }
848#endif
849
850 if (!_active.empty()) {
851 // There are more tasks; just set the next sort value.
852 nassertr(_current_sort < _active.front()->get_sort(), true);
853 _current_sort = _active.front()->get_sort();
854 _cvar.notify_all();
855 return true;
856 }
857
858 // There are no more tasks in this epoch; advance to the next epoch.
859
860 if (!_this_active.empty() && _frame_budget >= 0.0) {
861 // Enter pickup mode. This is a special mode at the end of the epoch in
862 // which we are just re-running the tasks that think they can still run
863 // within the frame, in an attempt to use up our frame budget.
864
865 if (task_cat.is_spam()) {
866 do_output(task_cat.spam());
867 task_cat.spam(false)
868 << ": next epoch (pickup mode)\n";
869 }
870
871 _pickup_mode = true;
872 _active.swap(_this_active);
873
874 } else {
875 // Not in pickup mode.
876
877 if (task_cat.is_spam()) {
878 do_output(task_cat.spam());
879 task_cat.spam(false)
880 << ": next epoch\n";
881 }
882
883 _pickup_mode = false;
884
885 // Here, there's no difference between _this_active and _next_active.
886 // Combine them.
887 _next_active.insert(_next_active.end(), _this_active.begin(), _this_active.end());
888 _this_active.clear();
889
890 _active.swap(_next_active);
891
892 // We only tick the clock and wake sleepers in normal mode, the first time
893 // through the task list; not in pickup mode when we are re-running the
894 // stragglers just to use up our frame budget.
895
896 if (_tick_clock) {
897 if (task_cat.is_spam()) {
898 do_output(task_cat.spam());
899 task_cat.spam(false)
900 << ": tick clock\n";
901 }
902 _manager->_clock->tick();
903 _manager->_frame_cvar.notify_all();
904
905 } else if (_frame_sync) {
906 // If we're a synced chain, we have to wait at the end of the epoch for
907 // someone else to tick the clock.
908 _block_till_next_frame = true;
909 }
910
911 // Check for any sleeping tasks that need to be woken.
912 double now = _manager->_clock->get_frame_time();
913 while (!_sleeping.empty() && _sleeping.front()->_wake_time <= now) {
914 PT(AsyncTask) task = _sleeping.front();
915 if (task_cat.is_spam()) {
916 task_cat.spam()
917 << "Waking " << *task << ", wake time at "
918 << task->_wake_time - now << "\n";
919 }
920 pop_heap(_sleeping.begin(), _sleeping.end(), AsyncTaskSortWakeTime());
921 _sleeping.pop_back();
922 task->_state = AsyncTask::S_active;
923 task->_start_frame = _manager->_clock->get_frame_count();
924 _active.push_back(task);
925 }
926
927 if (task_cat.is_spam()) {
928 if (_sleeping.empty()) {
929 task_cat.spam()
930 << "No more tasks on sleeping queue.\n";
931 } else {
932 task_cat.spam()
933 << "Next sleeper: " << *_sleeping.front() << ", wake time at "
934 << _sleeping.front()->_wake_time - now << "\n";
935 }
936 }
937
938 // Any tasks that are on the active queue at the beginning of the epoch
939 // are deemed to have run one frame (or to be about to).
940 TaskHeap::const_iterator ti;
941 for (ti = _active.begin(); ti != _active.end(); ++ti) {
942 AsyncTask *task = (*ti);
943 ++task->_num_frames;
944 }
945 }
946
947 if (_timeslice_priority) {
948 filter_timeslice_priority();
949 }
950
951 nassertr((size_t)_num_tasks == _active.size() + _this_active.size() + _next_active.size() + _sleeping.size() + (size_t)_num_awaiting_tasks, true);
952 make_heap(_active.begin(), _active.end(), AsyncTaskSortPriority());
953
954 _current_sort = -INT_MAX;
955
956 if (!_active.empty()) {
957 // Signal the threads to start executing the first task again.
958 _cvar.notify_all();
959 return true;
960 }
961
962 // There are no tasks to be had anywhere. Chill.
963 _pickup_mode = false;
964 nassertr(_this_active.empty(), false);
965 return false;
966}

975void AsyncTaskChain::
976filter_timeslice_priority() {
977 if (_active.empty()) {
978 return;
979 }
980 nassertv(_timeslice_priority);
981
982 // We must first sum up the average per-epoch runtime of each task.
983 double net_runtime = 0.0;
984 int net_priority = 0;
985
986 TaskHeap::iterator ti;
987 for (ti = _active.begin(); ti != _active.end(); ++ti) {
988 AsyncTask *task = (*ti);
989 double runtime = max(task->get_average_dt(), 0.0);
990 int priority = max(task->_priority, 1);
991 net_runtime += runtime;
992 net_priority += priority;
993 }
994
995 // That gives us a timeslice budget per priority value.
996 double average_budget = net_runtime / (double)net_priority;
997
998 TaskHeap keep, postpone;
999 for (ti = _active.begin(); ti != _active.end(); ++ti) {
1000 AsyncTask *task = (*ti);
1001 double runtime = max(task->get_average_dt(), 0.0);
1002 int priority = max(task->_priority, 1);
1003 double consumed = runtime / (double)priority;
1004 // cerr << *task << " consumed " << consumed << " vs. " << average_budget
1005 // << "\n";
1006 if (consumed > average_budget) {
1007 // Postpone. Run this task next epoch.
1008 postpone.push_back(task);
1009 } else {
1010 // Keep, and run this task this epoch.
1011 keep.push_back(task);
1012 }
1013 }
1014
1015 if (keep.empty()) {
1016 // Hmm, nothing to keep. Grab the postponed task with the highest
1017 // priority and keep that instead.
1018 nassertv(!postpone.empty());
1019 ti = postpone.begin();
1020 TaskHeap::iterator max_ti = ti;
1021 ++ti;
1022 while (ti != postpone.end()) {
1023 if ((*ti)->_priority > (*max_ti)->_priority) {
1024 max_ti = ti;
1025 }
1026 }
1027
1028 // cerr << "Nothing to keep, keeping " << *(*max_ti) << " instead\n";
1029
1030 keep.push_back(*max_ti);
1031 postpone.erase(max_ti);
1032 }
1033
1034 _active.swap(keep);
1035 if (_pickup_mode) {
1036 _this_active.insert(_this_active.end(), postpone.begin(), postpone.end());
1037 } else {
1038 _next_active.insert(_next_active.end(), postpone.begin(), postpone.end());
1039 }

1041 nassertv(!_active.empty());

1185void AsyncTaskChain::
1186do_poll() {
1187 thread_consider_yield();
1188 if (_num_tasks == 0) {
1189 return;
1190 }
1191
1192 do_start_threads();
1193
1194#ifdef HAVE_THREADS
1195 if (!_threads.empty()) {
1196 return;
1197 }
1198#endif
1199
1200 if (_num_busy_threads != 0) {
1201 // We are recursively nested within another task. Return, with a warning.
1202 task_cat.warning()
1203 << "Ignoring recursive poll() within another task.\n";
1204 return;
1205 }
1206
1207 nassertv(!_pickup_mode);
1208
1209 do {
1210 while (!_active.empty()) {
1211 if (_state == S_shutdown || _state == S_interrupted) {
1212 return;
1213 }
1214 int frame = _manager->_clock->get_frame_count();
1215 if (_current_frame != frame) {
1216 _current_frame = frame;
1217 _time_in_frame = 0.0;
1218 _block_till_next_frame = false;
1219 }
1220 if (_block_till_next_frame ||
1221 (_frame_budget >= 0.0 && _time_in_frame >= _frame_budget)) {
1222 // If we've exceeded our budget, stop here. We'll resume from this
1223 // point at the next call to poll().
1224 cleanup_pickup_mode();
1225 return;
1226 }
1227
1228 _current_sort = _active.front()->get_sort();
1229
1230 // Normally, there won't be any threads running at the same time we're
1231 // in poll(). But it's possible, if someone calls set_num_threads()
1232 // while we're processing.
1233 _num_busy_threads++;
1234 service_one_task(nullptr);
1235 _num_busy_threads--;
1236 _cvar.notify_all();
1237
1238#ifdef HAVE_THREADS
1239 if (!_threads.empty()) {
1240 return;
1241 }
1242#endif
1243 }
1244
1245 finish_sort_group();
1246 } while (_pickup_mode);
1247}

1446 thread_consider_yield();

1462 _chain->_state != S_shutdown && _chain->_state != S_interrupted) {

1468 _chain->_time_in_frame = 0.0;
1469 _chain->_block_till_next_frame = false;

1473 continue;

1484 // can't pick up a new task until all of the threads finish the tasks

1502 } else {

1504 // continue.
