118bool CollisionVisualizer::
119cull_callback(CullTraverser *trav, CullTraverserData &data) {
120 // Now we go through and actually draw our visualized collision solids.
121
122 LightMutexHolder holder(_lock);
123
124 Data::const_iterator di;
125 for (di = _data.begin(); di != _data.end(); ++di) {
126 const TransformState *net_transform = (*di).first;
127 const VizInfo &viz_info = (*di).second;
128
129 CullTraverserData xform_data(data);
130
131 // We don't want to inherit the transform from above! We ignore whatever
132 // transforms were above the CollisionVisualizer node; it always renders
133 // its objects according to their appropriate net transform.
134 xform_data._net_transform = TransformState::make_identity();
135 xform_data._view_frustum = trav->get_view_frustum();
136 xform_data.apply_transform(net_transform);
137
138 // Draw all the collision solids.
139 Solids::const_iterator si;
140 for (si = viz_info._solids.begin(); si != viz_info._solids.end(); ++si) {
141 // Note that we don't preserve the clip plane attribute from the
142 // collision solid. We always draw the whole polygon (or whatever) in
143 // the CollisionVisualizer. This is a deliberate decision; clipping the
144 // polygons may obscure many collision tests that are being made.
145 const CollisionSolid *solid = (*si).first;
146 const SolidInfo &solid_info = (*si).second;
147 bool was_detected = (solid_info._detected_count > 0);
148 PT(PandaNode) node = solid->get_viz(trav, xform_data, !was_detected);
149 if (node != nullptr) {
150 // We don't want to inherit the render state from above for these
151 // guys.
152 trav->traverse_down(xform_data, node, xform_data._net_transform, get_viz_state());
153 }
154 }
155
156 // Now draw all of the detected points.
157 if (!viz_info._points.empty()) {
158 CPT(RenderState) empty_state = RenderState::make_empty();
159 CPT(RenderState) point_state = RenderState::make(RenderModeAttrib::make(RenderModeAttrib::M_unchanged, 1.0f, false));
160
161 PT(GeomVertexArrayFormat) point_array_format;
162 if (vertex_colors_prefer_packed) {
163 point_array_format = new GeomVertexArrayFormat(
164 InternalName::get_vertex(), 3, Geom::NT_stdfloat, Geom::C_point,
165 InternalName::get_color(), 1, Geom::NT_packed_dabc, Geom::C_color,
166 InternalName::get_size(), 1, Geom::NT_stdfloat, Geom::C_other);
167 }
168 else {
169 point_array_format = new GeomVertexArrayFormat(
170 InternalName::get_vertex(), 3, Geom::NT_stdfloat, Geom::C_point,
171 InternalName::get_color(), 4, Geom::NT_uint8, Geom::C_color,
172 InternalName::get_size(), 1, Geom::NT_stdfloat, Geom::C_other);
173 }
174 CPT(GeomVertexFormat) point_format =
175 GeomVertexFormat::register_format(point_array_format);
176
177 Points::const_iterator pi;
178 for (pi = viz_info._points.begin(); pi != viz_info._points.end(); ++pi) {
179 const CollisionPoint &point = (*pi);
180
181 // Draw a small red point at the surface point, and a smaller white
182 // point at the interior point.
183 {
184 PT(GeomVertexData) point_vdata =
185 new GeomVertexData("viz", point_format, Geom::UH_stream);
186
187 PT(GeomPoints) points = new GeomPoints(Geom::UH_stream);
188
189 GeomVertexWriter vertex(point_vdata, InternalName::get_vertex());
190 GeomVertexWriter color(point_vdata, InternalName::get_color());
191 GeomVertexWriter size(point_vdata, InternalName::get_size());
192
193 vertex.add_data3(point._surface_point);
194 color.add_data4(1.0f, 0.0f, 0.0f, 1.0f);
195 size.add_data1f(16.0f * _point_scale);
196 points->add_next_vertices(1);
197 points->close_primitive();
198
199 if (point._interior_point != point._surface_point) {
200 vertex.add_data3(point._interior_point);
201 color.add_data4(1.0f, 1.0f, 1.0f, 1.0f);
202 size.add_data1f(8.0f * _point_scale);
203 points->add_next_vertices(1);
204 points->close_primitive();
205 }
206
207 PT(Geom) geom = new Geom(point_vdata);
208 geom->add_primitive(points);
209
210 trav->get_cull_handler()->record_object(CullableObject(
211 geom, point_state, xform_data.get_internal_transform(trav)), trav);
212 }
213
214 // Draw the normal vector at the surface point.
215 if (!point._surface_normal.almost_equal(LVector3::zero())) {
216 PT(GeomVertexData) line_vdata =
217 new GeomVertexData("viz", GeomVertexFormat::get_v3c(),
218 Geom::UH_stream);
219
220 PT(GeomLines) lines = new GeomLines(Geom::UH_stream);
221
222 GeomVertexWriter vertex(line_vdata, InternalName::get_vertex());
223 GeomVertexWriter color(line_vdata, InternalName::get_color());
224
225 vertex.add_data3(point._surface_point);
226 vertex.add_data3(point._surface_point +
227 point._surface_normal * _normal_scale);
228 color.add_data4(1.0f, 0.0f, 0.0f, 1.0f);
229 color.add_data4(1.0f, 1.0f, 1.0f, 1.0f);
230 lines->add_next_vertices(2);
231 lines->close_primitive();
232
233 PT(Geom) geom = new Geom(line_vdata);
234 geom->add_primitive(lines);
235
236 trav->get_cull_handler()->record_object(CullableObject(
237 geom, empty_state, xform_data.get_internal_transform(trav)), trav);
238 }
239 }
240 }
241 }
242
243 // Now carry on to render our child nodes.
244 return true;
245}

276void CollisionVisualizer::
277collision_tested(const CollisionEntry &entry, bool detected) {
278 CollisionRecorder::collision_tested(entry, detected);
279
280 NodePath node_path = entry.get_into_node_path();
281 CPT(TransformState) net_transform = node_path.get_net_transform();
282 CPT(CollisionSolid) solid = entry.get_into();
283 nassertv(!solid.is_null());
284
285 LightMutexHolder holder(_lock);
286 VizInfo &viz_info = _data[std::move(net_transform)];
287 if (detected) {
288 viz_info._solids[std::move(solid)]._detected_count++;
289
290 if (entry.has_surface_point()) {
291 CollisionPoint p;
292 entry.get_all(entry.get_into_node_path(),
293 p._surface_point, p._surface_normal, p._interior_point);
294 viz_info._points.push_back(p);
295 }
296
297 } else {
298 viz_info._solids[std::move(solid)]._missed_count++;
299 }
300}
