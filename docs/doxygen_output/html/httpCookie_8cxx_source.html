85 while (start < format.length() && isspace(format[start])) {

114bool HTTPCookie::
115matches_url(const URLSpec &url) const {
116 if (_domain.empty()) {
117 return false;
118 }
119 string server = url.get_server();
120 if (server == _domain ||
121 (string(".") + server) == _domain ||
122 (server.length() > _domain.length() &&
123 server.substr(server.length() - _domain.length()) == _domain &&
124 (_domain[0] == '.' || server[server.length() - _domain.length() - 1] == '.'))) {
125 // The domain matches.
126
127 string path = url.get_path();
128 if (path.length() >= _path.length() &&
129 path.substr(0, _path.length()) == _path) {
130
131 // The path matches too.
132 if (_secure && !url.is_ssl()) {
133 // Oops, can't send a secure cookie over a non-secure connection.
134 return false;
135 }
136
137 return true;
138 }
139 }
140
141 return false;
142}

171bool HTTPCookie::
172parse_cookie_param(const string &param, bool first_param) {
173 size_t equals = param.find('=');
174
175 string key, value;
176 if (equals == string::npos) {
177 key = param;
178 } else {
179 key = param.substr(0, equals);
180 value = param.substr(equals + 1);
181 }
182
183 if (first_param) {
184 _name = key;
185 _value = value;
186
187 } else {
188 key = downcase(key);
189 if (key == "expires") {
190 _expires = HTTPDate(value);
191 if (!_expires.is_valid()) {
192 return false;
193 }
194
195 } else if (key == "path") {
196 _path = value;
197
198 } else if (key == "domain") {
199 _domain = downcase(value);
200
201 // From RFC 2965: If an explicitly specified value does not start with a
202 // dot, the user agent supplies a leading dot.
203 if (!_domain.empty() && _domain[0] != '.') {
204 _domain = string(".") + _domain;
205 }
206
207 } else if (key == "secure") {
208 _secure = true;
209
210 } else if (key == "samesite") {
211 value = downcase(value);
212 if (value == "lax") {
213 _samesite = SS_lax;
214 }
215 else if (value == "strict") {
216 _samesite = SS_strict;
217 }
218 else if (value == "none") {
219 _samesite = SS_none;
220 }
221
222 } else {
223 return false;
224 }
225 }
226
227 return true;
228}
