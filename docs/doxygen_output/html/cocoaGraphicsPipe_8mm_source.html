33CocoaGraphicsPipe::
34CocoaGraphicsPipe(CGDirectDisplayID display) : _display(display) {
35 [[NSAutoreleasePool alloc] init];
36
37 // Put Cocoa into thread-safe mode by spawning a thread which immediately
38 // exits.
39 NSThread* thread = [[NSThread alloc] init];
40 [thread start];
41 [thread autorelease];
42
43 // If the application is dpi-aware, iterate over all the screens to find the
44 // one with our display ID and get the backing scale factor to configure the
45 // detected display zoom. Otherwise the detected display zoom keeps its
46 // default value of 1.0
47
48 if (dpi_aware) {
49 NSScreen *screen;
50 NSEnumerator *e = [[NSScreen screens] objectEnumerator];
51 while (screen = (NSScreen *) [e nextObject]) {
52 NSNumber *num = [[screen deviceDescription] objectForKey: @"NSScreenNumber"];
53 if (_display == (CGDirectDisplayID) [num longValue]) {
54 set_detected_display_zoom([screen backingScaleFactor]);
55 if (cocoadisplay_cat.is_debug()) {
56 cocoadisplay_cat.debug()
57 << "Display zoom is " << [screen backingScaleFactor] << "\n";
58 }
59 break;
60 }
61 }
62 }
63
64 // We used to also obtain the corresponding NSScreen here, but this causes
65 // the application icon to start bouncing, which may be undesirable for
66 // apps that will never open a window.
67
68 // Although the name of these functions mention pixels, they actually return
69 // display points, we use the detected display zoom to transform the values
70 // into pixels.
71 _display_width = CGDisplayPixelsWide(_display) * _detected_display_zoom;
72 _display_height = CGDisplayPixelsHigh(_display) * _detected_display_zoom;
73 load_display_information();
74
75 if (cocoadisplay_cat.is_debug()) {
76 cocoadisplay_cat.debug()
77 << "Creating CocoaGraphicsPipe for display ID " << _display << "\n";
78 }
79
80 // It takes a while to fire up the display link, so let's fire it up now if
81 // we expect to need VSync.
82 if (sync_video) {
83 uint32_t counter;
84 init_vsync(counter);
85 }

91void CocoaGraphicsPipe::
92load_display_information() {
93 _display_information->_vendor_id = CGDisplayVendorNumber(_display);
94 // _display_information->_device_id = CGDisplayUnitNumber(_display);
95 // _display_information->_device_id = CGDisplaySerialNumber(_display);
96
97 // Display modes
98 CFDictionaryRef options = NULL;
99 const CFStringRef dictkeys[] = {kCGDisplayShowDuplicateLowResolutionModes};
100 const CFBooleanRef dictvalues[] = {kCFBooleanTrue};
101 options = CFDictionaryCreate(NULL,
102 (const void **)dictkeys,
103 (const void **)dictvalues,
104 1,
105 &kCFCopyStringDictionaryKeyCallBacks,
106 &kCFTypeDictionaryValueCallBacks);
107 size_t num_modes = 0;
108 int32_t current_mode_id = -1;
109 CFArrayRef modes = CGDisplayCopyAllDisplayModes(_display, options);
110 if (modes != NULL) {
111 num_modes = CFArrayGetCount(modes);
112 _display_information->_total_display_modes = num_modes;
113 _display_information->_display_mode_array = new DisplayMode[num_modes];
114
115 // Get information about the current mode.
116 CGDisplayModeRef mode = CGDisplayCopyDisplayMode(_display);
117 if (mode) {
118 current_mode_id = CGDisplayModeGetIODisplayModeID(mode);
119 CGDisplayModeRelease(mode);
120 }
121 }
122 if (options != NULL) {
123 CFRelease(options);
124 }
125
126 for (size_t i = 0; i < num_modes; ++i) {
127 CGDisplayModeRef mode = (CGDisplayModeRef) CFArrayGetValueAtIndex(modes, i);
128
129 if (dpi_aware) {
130 _display_information->_display_mode_array[i].width = CGDisplayModeGetPixelWidth(mode);
131 _display_information->_display_mode_array[i].height = CGDisplayModeGetPixelHeight(mode);
132 } else {
133 _display_information->_display_mode_array[i].width = CGDisplayModeGetWidth(mode);
134 _display_information->_display_mode_array[i].height = CGDisplayModeGetHeight(mode);
135 }
136 _display_information->_display_mode_array[i].refresh_rate = CGDisplayModeGetRefreshRate(mode);
137 _display_information->_display_mode_array[i].fullscreen_only = false;
138
139 // Read number of bits per pixels from the pixel encoding
140 CFStringRef encoding = CGDisplayModeCopyPixelEncoding(mode);
141 if (CFStringCompare(encoding, CFSTR(kIO64BitDirectPixels),
142 kCFCompareCaseInsensitive) == kCFCompareEqualTo) {
143 _display_information->_display_mode_array[i].bits_per_pixel = 64;
144
145 } else if (CFStringCompare(encoding, CFSTR(kIO32BitFloatPixels),
146 kCFCompareCaseInsensitive) == kCFCompareEqualTo) {
147 _display_information->_display_mode_array[i].bits_per_pixel = 32;
148
149 } else if (CFStringCompare(encoding, CFSTR(kIO16BitFloatPixels),
150 kCFCompareCaseInsensitive) == kCFCompareEqualTo) {
151 _display_information->_display_mode_array[i].bits_per_pixel = 16;
152
153 } else if (CFStringCompare(encoding, CFSTR(IOYUV422Pixels),
154 kCFCompareCaseInsensitive) == kCFCompareEqualTo ||
155 CFStringCompare(encoding, CFSTR(IO8BitOverlayPixels),
156 kCFCompareCaseInsensitive) == kCFCompareEqualTo) {
157 _display_information->_display_mode_array[i].bits_per_pixel = 8;
158 } else {
159 // The other possible pixel formats in IOKitIOGraphicsTypes.h have
160 // strings like "PPPP" or "-RRRRRGGGGGBBBBB", so the number of bits per
161 // pixel can be deduced from the string length. Nifty!
162 _display_information->_display_mode_array[i].bits_per_pixel = CFStringGetLength(encoding);
163 }
164
165 if (current_mode_id >= 0 && current_mode_id == CGDisplayModeGetIODisplayModeID(mode)) {
166 _display_information->_current_display_mode_index = i;
167
168 // Stop checking
169 current_mode_id = -1;
170 }
171
172 CFRelease(encoding);
173 }
174 if (modes != nullptr) {
175 CFRelease(modes);
176 }
177
178 // Get processor information
179 const NXArchInfo *ainfo = NXGetLocalArchInfo();
180 _display_information->_cpu_brand_string.assign(ainfo->description);
181
182 // Get version of Mac OS X
183 SInt32 major, minor, bugfix;
184 Gestalt(gestaltSystemVersionMajor, &major);
185 Gestalt(gestaltSystemVersionMinor, &minor);
186 Gestalt(gestaltSystemVersionBugFix, &bugfix);
187 _display_information->_os_version_major = major;
188 _display_information->_os_version_minor = minor;
189 _display_information->_os_version_build = bugfix;
190}

225bool CocoaGraphicsPipe::
226init_vsync(uint32_t &counter) {
227 if (_display_link != nil) {
228 // Already set up.
229 __atomic_load(&_vsync_counter, &counter, __ATOMIC_SEQ_CST);
230 return true;
231 }
232
233 counter = 0;
234 _vsync_counter = 0;
235
236 CVDisplayLinkRef display_link;
237 CVReturn result = CVDisplayLinkCreateWithActiveCGDisplays(&display_link);
238 if (result != kCVReturnSuccess) {
239 cocoadisplay_cat.error() << "Failed to create CVDisplayLink.\n";
240 display_link = nil;
241 return false;
242 }
243
244 result = CVDisplayLinkSetCurrentCGDisplay(display_link, _display);
245 if (result != kCVReturnSuccess) {
246 cocoadisplay_cat.error() << "Failed to set CVDisplayLink's current display.\n";
247 CVDisplayLinkRelease(display_link);
248 display_link = nil;
249 return false;
250 }
251
252 result = CVDisplayLinkSetOutputCallback(display_link, &display_link_cb, this);
253 if (result != kCVReturnSuccess) {
254 cocoadisplay_cat.error() << "Failed to set CVDisplayLink output callback.\n";
255 CVDisplayLinkRelease(display_link);
256 display_link = nil;
257 return false;
258 }
259
260 result = CVDisplayLinkStart(display_link);
261 if (result != kCVReturnSuccess) {
262 cocoadisplay_cat.error() << "Failed to start the CVDisplayLink.\n";
263 CVDisplayLinkRelease(display_link);
264 display_link = nil;
265 return false;
266 }
267
268 _display_link = display_link;
269 return true;
270}

279void CocoaGraphicsPipe::
280wait_vsync(uint32_t &counter, bool adaptive) {
281 if (_display_link == nil) {
282 return;
283 }
284
285 // Use direct atomic operations since we need this to be thread-safe even
286 // when compiling without thread support.
287 uint32_t current_count = __atomic_load_n(&_vsync_counter, __ATOMIC_SEQ_CST);
288 uint32_t diff = current_count - counter;
289 if (diff > 0) {
290 if (cocoadisplay_cat.is_spam()) {
291 cocoadisplay_cat.spam()
292 << "Missed vertical blanking interval by " << diff << " frames.\n";
293 }
294 if (adaptive) {
295 counter = current_count;
296 return;
297 }
298 }
299
300 // We only wait for the first window that gets flipped in a single frame,
301 // otherwise we end up halving our FPS when we have multiple windows!
302 int cur_frame = ClockObject::get_global_clock()->get_frame_count();
303 if (_last_wait_frame.exchange(cur_frame) == cur_frame) {
304 counter = current_count;
305 return;
306 }
307
308 patomic_wait(&_vsync_counter, current_count);
309 __atomic_load(&_vsync_counter, &counter, __ATOMIC_SEQ_CST);
310}
