124bool SheetNode::
125cull_callback(CullTraverser *trav, CullTraverserData &data) {
126 // Statistics
127 PStatTimer timer(_sheet_node_pcollector);
128
129 // Create some geometry on-the-fly to render the sheet.
130 if (get_num_u_subdiv() > 0 && get_num_v_subdiv() > 0) {
131 NurbsSurfaceEvaluator *surface = get_surface();
132 if (surface != nullptr) {
133 PT(NurbsSurfaceResult) result = surface->evaluate(data.get_node_path());
134
135 if (result->get_num_u_segments() > 0 && result->get_num_v_segments() > 0) {
136 render_sheet(trav, data, result);
137 }
138 }
139 }
140
141 return true;
142}

206PT(BoundingVolume) SheetNode::
207do_recompute_bounds(const NodePath &rel_to, int pipeline_stage,
208 Thread *current_thread) const {
209 // TODO: fix the bounds so that it properly reflects the indicated pipeline
210 // stage. At the moment, we cheat and get some of the properties from the
211 // current pipeline stage, the lazy way.
212
213 // First, get ourselves a fresh, empty bounding volume.
214 PT(BoundingVolume) bound = new BoundingSphere;
215
216 NurbsSurfaceEvaluator *surface = get_surface();
217 if (surface != nullptr) {
218 NurbsSurfaceEvaluator::Vert3Array verts;
219 get_surface()->get_vertices(verts, rel_to);
220
221 GeometricBoundingVolume *gbv;
222 DCAST_INTO_R(gbv, bound, bound);
223 gbv->around(&verts[0], &verts[0] + verts.size());
224 }
225 return bound;

231void SheetNode::
232render_sheet(CullTraverser *trav, CullTraverserData &data,
233 NurbsSurfaceResult *result) {
234 bool use_vertex_color = get_use_vertex_color();
235
236 int num_u_segments = result->get_num_u_segments();
237 int num_v_segments = result->get_num_v_segments();
238 int num_u_verts = get_num_u_subdiv() + 1;
239 int num_v_verts = get_num_v_subdiv() + 1;
240
241 CPT(GeomVertexFormat) format;
242 if (use_vertex_color) {
243 format = GeomVertexFormat::get_v3n3ct2();
244 } else {
245 format = GeomVertexFormat::get_v3n3t2();
246 }
247 PT(GeomVertexData) vdata = new GeomVertexData
248 ("sheet", format, Geom::UH_stream);
249 int expected_num_vertices = num_u_segments * (num_u_verts + 1) * num_v_segments * num_v_verts;
250 vdata->reserve_num_rows(expected_num_vertices);
251
252 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
253 GeomVertexWriter normal(vdata, InternalName::get_normal());
254 GeomVertexWriter color(vdata, InternalName::get_color());
255 GeomVertexWriter texcoord(vdata, InternalName::get_texcoord());
256
257 for (int ui = 0; ui < num_u_segments; ui++) {
258 for (int uni = 0; uni <= num_u_verts; uni++) {
259 PN_stdfloat u0 = (PN_stdfloat)uni / (PN_stdfloat)num_u_verts;
260 PN_stdfloat u0_tc = result->get_segment_u(ui, u0);
261
262 for (int vi = 0; vi < num_v_segments; vi++) {
263 for (int vni = 0; vni < num_v_verts; vni++) {
264 PN_stdfloat v = (PN_stdfloat)vni / (PN_stdfloat)(num_v_verts - 1);
265 PN_stdfloat v_tc = result->get_segment_v(vi, v);
266
267 LPoint3 point;
268 LVector3 norm;
269 result->eval_segment_point(ui, vi, u0, v, point);
270 result->eval_segment_normal(ui, vi, u0, v, norm);
271 vertex.add_data3(point);
272 normal.add_data3(norm);
273 texcoord.add_data2(u0_tc, v_tc);
274
275 if (use_vertex_color) {
276 LColor c0;
277 result->eval_segment_extended_points(ui, vi, u0, v, 0, &c0[0], 4);
278 color.add_data4(c0);
279 }
280 }
281 }
282 }
283 }
284 nassertv(vdata->get_num_rows() == expected_num_vertices);
285
286 PT(GeomTristrips) strip = new GeomTristrips(Geom::UH_stream);
287
288 int expected_num_tristrips = num_u_segments * num_u_verts * num_v_segments;
289 int expected_verts_per_tristrip = num_v_verts * 2;
290
291 int expected_prim_vertices = (expected_num_tristrips - 1) * (expected_verts_per_tristrip + strip->get_num_unused_vertices_per_primitive()) + expected_verts_per_tristrip;
292
293 strip->reserve_num_vertices(expected_prim_vertices);
294
295 int verts_per_row = num_v_segments * num_v_verts;
296
297 for (int ui = 0; ui < num_u_segments; ui++) {
298 for (int uni = 0; uni < num_u_verts; uni++) {
299 int row_start_index = ((ui * (num_u_verts + 1)) + uni) * verts_per_row;
300
301 for (int vi = 0; vi < num_v_segments; vi++) {
302 for (int vni = 0; vni < num_v_verts; vni++) {
303 int vert_index_0 = row_start_index + (vi * num_v_verts) + vni;
304 int vert_index_1 = vert_index_0 + verts_per_row;
305 strip->add_vertex(vert_index_0);
306 strip->add_vertex(vert_index_1);
307 }
308 strip->close_primitive();
309 }
310 }
311 }
312 nassertv(strip->get_num_vertices() == expected_prim_vertices);
313
314 PT(Geom) geom = new Geom(vdata);
315 geom->add_primitive(strip);
316
317 CPT(RenderState) state = data._state;
318 if (use_vertex_color) {
319 state = state->add_attrib(ColorAttrib::make_vertex());
320 }
321
322 trav->get_cull_handler()->record_object(CullableObject(
323 std::move(geom), std::move(state), data.get_internal_transform(trav)), trav);
324}
