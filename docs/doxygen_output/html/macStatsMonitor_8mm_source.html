346CGColorRef MacStatsMonitor::
347get_collector_color(int collector_index, bool highlight) {
348 Colors::iterator bi;
349 bi = _colors.find(collector_index);
350 if (bi != _colors.end()) {
351 return (*bi).second._bg[highlight];
352 }
353
354 LRGBColor rgb = PStatMonitor::get_collector_color(collector_index);
355 rgb[0] = encode_sRGB_float((float)rgb[0]);
356 rgb[1] = encode_sRGB_float((float)rgb[1]);
357 rgb[2] = encode_sRGB_float((float)rgb[2]);
358
359 PN_stdfloat bright = rgb.dot(LRGBColor(0.2126, 0.7152, 0.0722));
360 CGColorRef color = CGColorCreateGenericRGB(rgb[0], rgb[1], rgb[2], 1.0);
361
362 rgb *= 0.75;
363 CGColorRef hcolor = CGColorCreateGenericRGB(rgb[0], rgb[1], rgb[2], 1.0);
364
365 ColorSet &set = _colors[collector_index];
366 set._bg[0] = color;
367 set._bg[1] = hcolor;
368
369 if (bright >= 0.5) {
370 set._fg[0] = [NSColor blackColor];
371 } else {
372 set._fg[0] = [NSColor whiteColor];
373 }
374
375 if (bright * 0.75 >= 0.5) {
376 set._fg[1] = [NSColor blackColor];
377 } else {
378 set._fg[1] = [NSColor whiteColor];
379 }
380
381 return highlight ? hcolor : color;
382}

448set_show_status_item(bool show) {
449 if (show && _frame_rate_status_item == nil) {
450 _frame_rate_status_item = [[NSStatusBar systemStatusBar] statusItemWithLength:NSVariableStatusItemLength];
451 _frame_rate_status_item.button.action = @selector(handleClickStatusItem:);
452 [_frame_rate_status_item retain];
453
454 const PStatClientData *client_data = get_client_data();
455 if (client_data != nullptr) {
456 const PStatThreadData *thread_data = client_data->get_thread_data(0);
457 if (thread_data != nullptr) {

459 if (frame_rate != 0.0f) {

467 // Careful: for some reason, this can get called recursively.

471 [status_item release];
