166bool EggReader::
167handle_args(ProgramBase::Args &args) {
168 if (args.empty()) {
169 nout << "You must specify the egg file(s) to read on the command line.\n";
170 return false;
171 }
172
173 // Any separate egg files that are listed on the command line will get
174 // implicitly loaded up into one big egg file.
175
176 if (!args.empty()) {
177 _data->set_egg_filename(Filename::from_os_specific(args[0]));
178 }
179 Args::const_iterator ai;
180 for (ai = args.begin(); ai != args.end(); ++ai) {
181 Filename filename = Filename::from_os_specific(*ai);
182
183 EggData file_data;
184 if (filename != "-") {
185 if (!file_data.read(filename)) {
186 // Rather than returning false, we simply exit here, so the ProgramBase
187 // won't try to tell the user how to run the program just because we got
188 // a bad egg file.
189 if (_exit_on_failure) {
190 exit(1);
191 }
192 return false;
193 }
194 } else {
195 if (!file_data.read(std::cin)) {
196 if (_exit_on_failure) {
197 exit(1);
198 }
199 return false;
200 }
201 }
202
203 if (_noabs && file_data.original_had_absolute_pathnames()) {
204 nout << filename.get_basename()
205 << " includes absolute pathnames!\n";
206 if (_exit_on_failure) {
207 exit(1);
208 }
209 return false;
210 }
211
212 DSearchPath file_path;
213 file_path.append_directory(filename.get_dirname());
214
215 if (_force_complete) {
216 if (!file_data.load_externals(file_path)) {
217 if (_exit_on_failure) {
218 exit(1);
219 }
220 return false;
221 }
222 }
223
224 // Now resolve the filenames again according to the user's specified
225 // _path_replace.
226 convert_paths(&file_data, _path_replace, file_path);
227
228 _data->merge(file_data);
229 }
230
231 pre_process_egg_file();
232
233 return true;
234}

274bool EggReader::
275copy_textures() {
276 bool success = true;
277 EggTextureCollection textures;
278 textures.find_used_textures(_data);
279
280 EggTextureCollection::const_iterator ti;
281 for (ti = textures.begin(); ti != textures.end(); ++ti) {
282 EggTexture *tex = (*ti);
283 Filename orig_filename = tex->get_filename();
284 if (!orig_filename.exists()) {
285 bool found = orig_filename.resolve_filename(get_model_path());
286 if (!found) {
287 nout << "Cannot find " << orig_filename << "\n";
288 success = false;
289 continue;
290 }
291 }
292
293 Filename new_filename = orig_filename;
294 if (_got_tex_dirname) {
295 new_filename.set_dirname(_tex_dirname);
296 }
297 if (_got_tex_extension) {
298 new_filename.set_extension(_tex_extension);
299 }
300
301 if (orig_filename != new_filename) {
302 tex->set_filename(new_filename);
303
304 // The new filename is different; does it need copying?
305 int compare =
306 orig_filename.compare_timestamps(new_filename, true, true);
307 if (compare > 0) {
308 // Yes, it does. Copy it!
309 nout << "Reading " << orig_filename << "\n";
310 PNMImage image;
311 if (!image.read(orig_filename)) {
312 nout << " unable to read!\n";
313 success = false;
314 } else {
315 nout << "Writing " << new_filename << "\n";
316 if (!image.write(new_filename, _tex_type)) {
317 nout << " unable to write!\n";
318 success = false;
319 }
320 }
321 }
322 }
323 }
324
325 return success;
326}

333bool EggReader::
334do_delod(EggNode *node) {
335 if (node->is_of_type(EggGroup::get_class_type())) {
336 EggGroup *group = DCAST(EggGroup, node);
337 if (group->has_lod()) {
338 const EggSwitchCondition &cond = group->get_lod();
339 if (cond.is_of_type(EggSwitchConditionDistance::get_class_type())) {
340 const EggSwitchConditionDistance *dist =
341 DCAST(EggSwitchConditionDistance, &cond);
342 if (_delod >= dist->_switch_out && _delod < dist->_switch_in) {
343 // Preserve this group node, but not the LOD information itself.
344 nout << "Preserving LOD " << node->get_name()
345 << " (" << dist->_switch_out << " to " << dist->_switch_in
346 << ")\n";
347 group->clear_lod();
348 } else {
349 // Remove this group node.
350 nout << "Eliminating LOD " << node->get_name()
351 << " (" << dist->_switch_out << " to " << dist->_switch_in
352 << ")\n";
353 return false;
354 }
355 }
356 }
357 }
358
359 // Now process all the children.
360 if (node->is_of_type(EggGroupNode::get_class_type())) {
361 EggGroupNode *group = DCAST(EggGroupNode, node);
362 EggGroupNode::iterator ci;
363 ci = group->begin();
364 while (ci != group->end()) {
365 EggNode *child = *ci;
366 ++ci;
367
368 if (!do_delod(child)) {
369 group->remove_child(child);
370 }
371 }
372 }
373
374 return true;
