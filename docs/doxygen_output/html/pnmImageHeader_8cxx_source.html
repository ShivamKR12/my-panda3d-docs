105PNMReader *PNMImageHeader::
106make_reader(const Filename &filename, PNMFileType *type,
107 bool report_unknown_type) const {
108 if (pnmimage_cat.is_debug()) {
109 pnmimage_cat.debug()
110 << "Reading image from " << filename << "\n";
111 }
112 bool owns_file = false;
113 istream *in = nullptr;
114
115 if (filename == "-") {
116 owns_file = false;
117 in = &std::cin;
118
119 if (pnmimage_cat.is_debug()) {
120 pnmimage_cat.debug()
121 << "(reading standard input)\n";
122 }
123 } else {
124 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
125 owns_file = true;
126 PT(VirtualFile) file = vfs->get_file(filename, false);
127 if (file != nullptr) {
128#ifdef __EMSCRIPTEN__
129 // Is this a real file, and do we have that in the Emscripten cache?
130 SubfileInfo info;
131 if (file->get_system_info(info) && info.get_start() == 0) {
132 Filename fn = info.get_filename();
133 if (!fn.make_canonical()) {
134 fn.make_absolute();
135 }
136 int width, height;
137 if (query_preload(fn.c_str(), &width, &height)) {
138 if (pnmimage_cat.is_debug()) {
139 pnmimage_cat.debug()
140 << "(found in emscripten cache)\n";
141 }
142 return new PNMReaderEmscripten(type, fn, width, height);
143 }
144 }
145#endif
146 in = vfs->open_read_file(filename, true);
147 if (in != nullptr && in->fail()) {
148 file->close_read_file(in);
149 in = nullptr;
150 }
151 }
152 }
153
154 if (in == nullptr) {
155 if (pnmimage_cat.is_debug()) {
156 pnmimage_cat.debug()
157 << "Unable to open file.\n";
158 }
159 return nullptr;
160 }
161
162 return make_reader(in, owns_file, filename, string(), type,
163 report_unknown_type);
164}

189PNMReader *PNMImageHeader::
190make_reader(istream *file, bool owns_file, const Filename &filename,
191 string magic_number, PNMFileType *type,
192 bool report_unknown_type) const {
193 if (type == nullptr) {
194 if (!read_magic_number(file, magic_number, 2)) {
195 // No magic number. No image.
196 if (pnmimage_cat.is_debug()) {
197 pnmimage_cat.debug()
198 << "Image file appears to be empty.\n";
199 }
200 if (owns_file) {
201 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
202
203 // We're assuming here that the file was opened via VFS. That may not
204 // necessarily be the case, but we don't make that distinction.
205 // However, at the moment at least, that distinction doesn't matter,
206 // since vfs->close_read_file() just deletes the file pointer anyway.
207 vfs->close_read_file(file);
208 }
209 return nullptr;
210 }
211
212 type = PNMFileTypeRegistry::get_global_ptr()->
213 get_type_from_magic_number(magic_number);
214
215 if (pnmimage_cat.is_debug()) {
216 if (type != nullptr) {
217 pnmimage_cat.debug()
218 << "By magic number, image file appears to be type "
219 << type->get_name() << ".\n";
220 } else {
221 pnmimage_cat.debug()
222 << "Unable to determine image file type from magic number.\n";
223 }
224 }
225 }
226
227 if (type == nullptr && !filename.empty()) {
228 // We still don't know the type; attempt to guess it from the filename
229 // extension.
230 type = PNMFileTypeRegistry::get_global_ptr()->get_type_from_extension(filename);
231
232 if (pnmimage_cat.is_debug()) {
233 if (type != nullptr) {
234 pnmimage_cat.debug()
235 << "From its extension, image file is probably type "
236 << type->get_name() << ".\n";
237 } else {
238 pnmimage_cat.debug()
239 << "Unable to guess image file type from its extension ("
240 << filename.get_extension() << ").\n";
241 }
242 }
243 }
244
245 if (type == nullptr) {
246 // No? How about the default type associated with this image header.
247 type = _type;
248
249 if (pnmimage_cat.is_debug() && type != nullptr) {
250 pnmimage_cat.debug()
251 << "Assuming image file type is " << type->get_name() << ".\n";
252 }
253 }
254
255 if (type == nullptr) {
256 // We can't figure out what type the file is; give up.
257 if (report_unknown_type && pnmimage_cat.is_error()) {
258 pnmimage_cat.error()
259 << "Cannot determine type of image file " << filename << ".\n"
260 << "Currently supported image types:\n";
261 PNMFileTypeRegistry::get_global_ptr()->
262 write(pnmimage_cat.error(false), 2);
263 }
264 if (owns_file) {
265 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
266
267 // We're assuming here that the file was opened via VFS. That may not
268 // necessarily be the case, but we don't make that distinction.
269 // However, at the moment at least, that distinction doesn't matter,
270 // since vfs->close_read_file() just deletes the file pointer anyway.
271 vfs->close_read_file(file);
272 }
273 return nullptr;
274 }
275
276 PNMReader *reader = type->make_reader(file, owns_file, magic_number);
277 if (reader == nullptr && owns_file) {
278 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
279 vfs->close_read_file(file);
280 }
281
282 if (!reader->is_valid()) {
283 delete reader;
284 reader = nullptr;
285 }
286
287 return reader;
288}

415bool PNMImageHeader::
416read_magic_number(istream *file, string &magic_number, int num_bytes) {
417 while ((int)magic_number.size() < num_bytes) {
418 int ch = file->get();
419 if (file->eof() || file->fail()) {
420 return false;
421 }

423 }

450bool PNMImageHeader::
451compute_histogram(PNMImageHeader::HistMap &hist,
452 xel *array, xelval *alpha, int max_colors) {
453 int num_pixels = _x_size * _y_size;
454 int pi;
455
456 switch (get_color_type()) {
457 case CT_invalid:
458 return false;
459
460 case CT_grayscale:
461 for (pi = 0; pi < num_pixels; pi++) {
462 record_color(hist, PixelSpec(PPM_GETB(array[pi])));
463 if (max_colors > 0 && (int)hist.size() > max_colors) {
464 return false;
465 }
466 }
467 return true;
468
469 case CT_two_channel:
470 for (pi = 0; pi < num_pixels; pi++) {
471 record_color(hist, PixelSpec(PPM_GETB(array[pi]), alpha[pi]));
472 if (max_colors > 0 && (int)hist.size() > max_colors) {
473 return false;
474 }
475 }
476 return true;
477
478 case CT_color:
479 for (pi = 0; pi < num_pixels; pi++) {
480 record_color(hist, PixelSpec(PPM_GETR(array[pi]), PPM_GETG(array[pi]), PPM_GETB(array[pi])));
481 if (max_colors > 0 && (int)hist.size() > max_colors) {
482 return false;
483 }
484 }
485 return true;
486
487 case CT_four_channel:
488 for (pi = 0; pi < num_pixels; pi++) {
489 record_color(hist, PixelSpec(PPM_GETR(array[pi]), PPM_GETG(array[pi]), PPM_GETB(array[pi]), alpha[pi]));
490 if (max_colors > 0 && (int)hist.size() > max_colors) {
491 return false;
492 }
493 }
494 return true;
495 }
496
497 return false;
498}

504bool PNMImageHeader::
505compute_palette(PNMImageHeader::Palette &palette,
506 xel *array, xelval *alpha, int max_colors) {
507 HistMap hist;
508
509 int num_pixels = _x_size * _y_size;
510
511 // In case there are already entries in the palette, preserve them.
512 Palette::const_iterator pi;
513 for (pi = palette.begin(); pi != palette.end(); ++pi) {
514 hist.insert(HistMap::value_type(*pi, num_pixels + 1));
515 }
516
517 if (!compute_histogram(hist, array, alpha, max_colors)) {
518 return false;
519 }
520
521 // Now append the new entries discovered in the histogram onto the end of
522 // the palette.
523 palette.reserve(hist.size());
524 HistMap::const_iterator hi;
525 for (hi = hist.begin(); hi != hist.end(); ++hi) {
526 if ((*hi).second <= num_pixels) {
527 palette.push_back((*hi).first);
528 }
529 }
530
531 return true;
532}
