88update() {
89 const PStatClientData *client_data = get_monitor()->get_client_data();
90
91 // Don't bother to update the thread data until we know at least something
92 // about the collectors and threads.
93 if (client_data->get_num_collectors() != 0 &&
94 client_data->get_num_threads() != 0) {
95 const PStatThreadData *thread_data = _view.get_thread_data();
96 if (!thread_data->is_empty()) {
97 int latest = thread_data->get_latest_frame_number();
98
99 if (latest > _next_frame) {
100 draw_frames(_next_frame, latest);
101 }
102 _next_frame = latest;
103

108 Data::iterator di;

174void PStatStripChart::
175set_auto_vertical_scale() {
176 const PStatThreadData *thread_data = _view.get_thread_data();
177
178 // Calculate the median value.
179 std::vector<double> values;
180
181 if (thread_data != nullptr && !thread_data->is_empty()) {
182 // Find the oldest visible frame.
183 double start_time = pixel_to_timestamp(0);
184 int oldest_frame = std::max(
185 thread_data->get_frame_number_at_time(start_time),
186 thread_data->get_oldest_frame_number());
187 int latest_frame = thread_data->get_latest_frame_number();
188
189 for (int frame_number = oldest_frame; frame_number <= latest_frame; ++frame_number) {
190 if (thread_data->has_frame(frame_number)) {
191 const FrameData &frame = get_frame_data(frame_number);
192 values.push_back(frame._net_value);
193 }
194 }
195 }
196
197 if (values.empty()) {
198 set_default_vertical_scale();
199 return;
200 }
201
202 double median;
203 size_t half = values.size() / 2;
204 if (values.size() % 2 == 0) {
205 std::sort(values.begin(), values.end());
206 median = (values[half] + values[half + 1]) / 2.0;
207 } else {
208 std::nth_element(values.begin(), values.begin() + half, values.end());
209 median = values[half];
210 }
211
212 if (median > 0.0) {
213 // Take 1.5 times the median value as the vertical scale.
214 set_vertical_scale(median * 1.5);
215 } else {
216 set_default_vertical_scale();
217 }

224int PStatStripChart::
225get_collector_under_pixel(int xpoint, int ypoint) {
226 // First, we need to know what frame it was; to know that, we need to
227 // determine the time corresponding to the x pixel.
228 double time = pixel_to_timestamp(xpoint);
229
230 // Now use that time to determine the frame.
231 const PStatThreadData *thread_data = _view.get_thread_data();
232
233 if (time < thread_data->get_oldest_time()) {
234 return -1;
235 }
236
237 // And now we can determine which collector within the frame, based on the
238 // value height.
239 if (_average_mode) {
240 double start_time = pixel_to_timestamp(xpoint);
241 int then_i = thread_data->get_frame_number_at_time(start_time - pstats_average_time);
242 int now_i = thread_data->get_frame_number_at_time(start_time, then_i);
243
244 FrameData fdata;
245 compute_average_pixel_data(fdata, then_i, now_i, start_time);
246 double overall_value = 0.0;
247 int y = get_ysize();
248
249 FrameData::const_iterator fi;
250 for (fi = fdata.begin(); fi != fdata.end(); ++fi) {
251 const ColorData &cd = (*fi);
252 overall_value += cd._net_value;
253 y = height_to_pixel(overall_value);
254 if (y <= ypoint) {
255 return cd._collector_index;
256 }
257 }
258
259 } else {
260 int frame_number = thread_data->get_frame_number_at_time(time);
261 const FrameData &fdata = get_frame_data(frame_number);
262 double overall_value = 0.0;
263 int y = get_ysize();
264
265 FrameData::const_iterator fi;
266 for (fi = fdata.begin(); fi != fdata.end(); ++fi) {
267 const ColorData &cd = (*fi);
268 overall_value += cd._net_value;
269 y = height_to_pixel(overall_value);
270 if (y <= ypoint) {
271 return cd._collector_index;
272 }
273 }
274 }
275
276 return -1;
277}

335 const PStatClientData *client_data = _monitor->get_client_data();
336 if (!client_data->has_collector(collector_index)) {
337 return std::string();
338 }
339
340 const PStatThreadData *thread_data = _view.get_thread_data();
341
342 std::ostringstream text;
343 text << client_data->get_collector_fullname(collector_index);
344
345 double value;
346 if (collector_index == _collector_index) {
347 value = get_average_net_value();
348 }
349 else {
350 int now_i, then_i;
351 if (!thread_data->get_elapsed_frames(then_i, now_i)) {
352 return text.str();

356

363 if (frame_data.get_end() > then) {
364 double this_time = (frame_data.get_end() - then);
365 for (const ColorData &cd : get_frame_data(then_i)) {
366 if (cd._collector_index == collector_index) {
367 net_value += cd._net_value * this_time;
368 net_time += this_time;
369 break;
370 }
371 }
372 }
373 // Then we get all of each of the remaining frames.
374 for (int frame_number = then_i + 1;
375 frame_number <= now_i;
376 frame_number++) {
377 const PStatFrameData &frame_data = thread_data->get_frame(frame_number);
378 for (const ColorData &cd : get_frame_data(frame_number)) {
379 if (cd._collector_index == collector_index) {
380 double this_time = frame_data.get_net_time();
381 net_value += cd._net_value * this_time;
382 net_time += this_time;
383 break;
384 }
385 }
386 }
387
388 if (net_time == 0) {
389 return text.str();

392 }
393

396 if (collector_index != 0) {
397 const FrameData &frame = get_frame_data(thread_data->get_latest_frame_number());
398
399 for (const ColorData &cd : frame) {
400 if (cd._collector_index == collector_index) {
401 if (cd._count > 0) {
402 text << " / " << cd._count << "x";
403 }
404 break;
405 }
406 }
407 }

476void PStatStripChart::
477accumulate_frame_data(FrameData &fdata, const FrameData &additional,
478 double weight) {
479 FrameData::iterator ai;
480 FrameData::const_iterator bi;
481
482 ai = fdata.begin();
483 bi = additional.begin();
484
485 FrameData result;
486
487 if (fdata.size() == additional.size()) {
488 // Start out assuming that fdata and additional contain exactly the same
489 // set of collectors. If we discover otherwise, we'll have to bail at
490 // that point.
491 while (ai != fdata.end() &&
492 (*ai)._collector_index == (*bi)._collector_index) {
493 (*ai)._net_value += ((*bi)._net_value * weight);
494 ++ai;
495 ++bi;
496 }
497
498 if (ai == fdata.end()) {
499 // If we successfully reached the end of the list, great! We're done
500 // without any merging.
501 return;
502 }
503
504 // Otherwise, the two lists weren't identical. In that case, copy the
505 // accumulated data so far and continue from this point with the full-
506 // blown merge.
507 result.reserve(max(fdata.size(), additional.size()));
508 FrameData::const_iterator ci;
509 for (ci = fdata.begin(); ci != ai; ++ci) {
510 result.push_back(*ci);
511 }
512
513 } else {
514 // If the two lists had different lengths, clearly they aren't identical.
515 result.reserve(max(fdata.size(), additional.size()));
516 }
517
518 while (ai != fdata.end() && bi != additional.end()) {
519 if ((*ai)._i < (*bi)._i) {
520 // Here's a data value that's in data, but not in additional.
521 result.push_back(*ai);
522 ++ai;
523
524 } else if ((*bi)._i < (*ai)._i) {
525 // Here's a data value that's in additional, but not in data.
526 ColorData scaled;
527 scaled._collector_index = (*bi)._collector_index;
528 scaled._i = (*bi)._i;
529 scaled._count = 0;
530 scaled._net_value = (*bi)._net_value * weight;
531 result.push_back(scaled);
532 ++bi;
533
534 } else {
535 // Here's a data value that's in both.
536 ColorData combined;
537 combined._collector_index = (*ai)._collector_index;
538 combined._i = (*bi)._i;
539 combined._count = 0;
540 combined._net_value = (*ai)._net_value + (*bi)._net_value * weight;
541 result.push_back(combined);
542 ++ai;
543 ++bi;
544 }
545 }
546
547 while (ai != fdata.end()) {
548 // Here's a data value that's in data, but not in additional.
549 result.push_back(*ai);
550 ++ai;
551 }
552
553 while (bi != additional.end()) {
554 // Here's a data value that's in additional, but not in data.
555 ColorData scaled;
556 scaled._collector_index = (*bi)._collector_index;
557 scaled._i = (*bi)._i;
558 scaled._count = 0;
559 scaled._net_value = (*bi)._net_value * weight;
560 result.push_back(scaled);
561 ++bi;
562 }
563
564 fdata.swap(result);
565}

584const PStatStripChart::FrameData &PStatStripChart::
585get_frame_data(int frame_number) const {
586 Data::const_iterator di;
587 di = _data.find(frame_number);
588 if (di != _data.end()) {
589 return (*di).second;
590 }
591
592 const PStatThreadData *thread_data = _view.get_thread_data();
593 //assert(thread_data->has_frame(frame_number));
594 const PStatFrameData &frame_data = thread_data->get_frame(frame_number);
595 _view.set_to_frame(frame_data);
596
597 FrameData &fdata = _data[frame_number];
598 fdata._start = frame_data.get_start();
599 fdata._end = frame_data.get_end();
600 fdata._net_time = frame_data.get_net_time();
601
602 double net_value = 0.0;
603
604 const PStatViewLevel *level = _view.get_level(_collector_index);
605 int num_children = level->get_num_children();
606 for (int i = 0; i < num_children; i++) {
607 const PStatViewLevel *child = level->get_child(i);
608 ColorData cd;
609 cd._collector_index = (unsigned short)child->get_collector();
610 cd._i = (unsigned short)i;
611 cd._count = child->get_count();
612 cd._net_value = child->get_net_value();
613 if (cd._net_value != 0.0) {
614 fdata.push_back(cd);
615 net_value += cd._net_value;
616 }
617 }
618
619 // Also, there might be some value in the overall Collector that wasn't
620 // included in all of the children.
621 ColorData cd;
622 cd._collector_index = (unsigned short)level->get_collector();
623 cd._i = (unsigned short)num_children;
624 cd._count = level->get_count();
625 cd._net_value = level->get_value_alone();
626 if (cd._net_value > 0.0) {
627 fdata.push_back(cd);
628 net_value += cd._net_value;
629 }
630
631 fdata._net_value = net_value;
632
633 ((PStatStripChart *)this)->inc_label_usage(fdata);
634
635 return fdata;
636}

648void PStatStripChart::
649compute_average_pixel_data(PStatStripChart::FrameData &result,
650 int &then_i, int &now_i, double now) {
651 result.clear();
652
653 const PStatThreadData *thread_data = _view.get_thread_data();
654 if (thread_data->is_empty() || thread_data->get_oldest_time() > now) {
655 // No data.
656 return;
657 }
658
659 double then = now - pstats_average_time;
660 then_i = thread_data->get_frame_number_after(then, then_i);
661 now_i = thread_data->get_frame_number_after(now, now_i);
662
663 const FrameData *fdata = &get_frame_data(then_i);
664 then = max(then, fdata->_start);
665 double then_end = fdata->_end;
666
667 // Sum up a weighted average of all of the individual frames we pass.
668
669 // We start with just the portion of frame then_i that actually does fall
670 // within our "then to now" window.
671 accumulate_frame_data(result, *fdata, then_end - then);
672 double last = then_end;
673
674 // Then we get all of each of the middle frames.
675 double weight = 0.0;
676 for (int frame_number = then_i + 1;
677 frame_number < now_i;
678 frame_number++) {
679 if (thread_data->has_frame(frame_number)) {
680 if (weight > 0.0) {
681 accumulate_frame_data(result, *fdata, weight);
682 weight = 0.0;
683 }
684 fdata = &get_frame_data(frame_number);
685 }
686 weight += fdata->_end - last;
687 last = fdata->_end;
688 }
689
690 if (weight > 0.0) {
691 accumulate_frame_data(result, *fdata, weight);
692 }
693
694 // And finally, we get the remainder as now_i.
695 if (last <= now) {
696 accumulate_frame_data(result, get_frame_data(now_i), now - last);
697 }
698
699 scale_frame_data(result, 1.0f / (now - then));
700}

706double PStatStripChart::
707get_average_net_value() const {
708 const PStatThreadData *thread_data = _view.get_thread_data();
709 int now_i, then_i;
710 if (!thread_data->get_elapsed_frames(then_i, now_i)) {
711 return 0.0;
712 }
713 double now = _time_width + _start_time;
714 double then = now - pstats_average_time;
715
716 int num_frames = now_i - then_i + 1;
717
718 if (_collector_index == 0 && !_view.get_show_level()) {
719 // If we're showing the time for the whole frame, compute this from the
720 // total elapsed time, rather than summing up individual frames. This is
721 // more accurate and exactly matches what is reported by
722 // thread_data->get_frame_rate().
723
724 const PStatFrameData &now_frame_data = thread_data->get_frame(now_i);
725 const PStatFrameData &then_frame_data = thread_data->get_frame(then_i);
726 double now = now_frame_data.get_end();
727 double elapsed_time = (now - then_frame_data.get_start());
728 return elapsed_time / (double)num_frames;
729
730 } else {
731 // On the other hand, if we're showing the time for some sub-frame, we
732 // have to do it the less-accurate way of summing up individual frames,
733 // which might introduce errors if we are missing data for some frames,
734 // but what can you do?
735
736 const PStatThreadData *thread_data = _view.get_thread_data();
737
738 double net_value = 0.0;
739 double net_time = 0.0;
740
741 // We start with just the portion of frame then_i that actually does fall
742 // within our "then to now" window (usually some portion of it will).
743 const PStatFrameData &frame_data = thread_data->get_frame(then_i);
744 const FrameData *frame = &get_frame_data(then_i);
745 if (frame_data.get_end() > then) {
746 double this_time = (frame->_end - then);
747 net_value += frame->_net_value * this_time;
748 net_time += this_time;
749 }
750 // Then we get all of each of the remaining frames.
751 for (int frame_number = then_i + 1;
752 frame_number <= now_i;
753 frame_number++) {
754 if (thread_data->has_frame(frame_number)) {
755 frame = &get_frame_data(frame_number);
756 }
757 double this_time = frame->_net_time;
758 net_value += frame->_net_value * this_time;
759 net_time += this_time;
760 }
761
762 return net_value / net_time;
763 }

946void PStatStripChart::
947draw_frames(int first_frame, int last_frame) {
948 const PStatThreadData *thread_data = _view.get_thread_data();
949
950 last_frame = min(last_frame, thread_data->get_latest_frame_number());
951
952 if (_first_data) {
953 if (_scroll_mode) {
954 _start_time =
955 thread_data->get_frame(last_frame).get_start() - _time_width;
956 } else {
957 _start_time = thread_data->get_frame(first_frame).get_start();
958 _cursor_pixel = 0;
959 }
960 }
961
962 int first_pixel;
963 if (thread_data->has_frame(first_frame)) {
964 first_pixel =
965 timestamp_to_pixel(thread_data->get_frame(first_frame).get_start());
966 } else {
967 first_pixel = 0;
968 }
969
970 int last_pixel =
971 timestamp_to_pixel(thread_data->get_frame(last_frame).get_start());
972
973 if (_first_data && !_scroll_mode) {
974 first_pixel = min(_cursor_pixel, first_pixel);
975 }
976 _first_data = false;
977
978 if (last_pixel - first_pixel >= _xsize) {
979 // If we're drawing the whole thing all in this one swoop, just start
980 // over.
981 _start_time = thread_data->get_frame(last_frame).get_start() - _time_width;
982 first_pixel = 0;
983 last_pixel = _xsize;
984 }
985
986 if (last_pixel <= _xsize) {
987 // It all fits in one block.
988 _cursor_pixel = last_pixel;
989 draw_pixels(first_pixel, last_pixel);
990
991 } else {
992 if (_scroll_mode) {
993 // In scrolling mode, slide the world back.
994 int slide_pixels = last_pixel - _xsize;
995 // This is really slow on macOS, just redraw instead
996#ifdef __APPLE__
997 draw_pixels(0, first_pixel - slide_pixels);
998#else
999 copy_region(slide_pixels, first_pixel, 0);
1000#endif
1001 first_pixel -= slide_pixels;
1002 last_pixel -= slide_pixels;
1003 _start_time += (double)slide_pixels / (double)_xsize * _time_width;
1004 draw_pixels(first_pixel, last_pixel);
1005
1006 } else {
1007 // In wrapping mode, do it in two blocks.
1008 _cursor_pixel = -1;
1009 draw_pixels(first_pixel, _xsize);
1010 _start_time = pixel_to_timestamp(_xsize);
1011 last_pixel -= _xsize;
1012 _cursor_pixel = last_pixel;
1013 draw_pixels(0, last_pixel);
1014 }
1015 }

1021void PStatStripChart::
1022draw_pixels(int first_pixel, int last_pixel) {
1023 begin_draw(first_pixel, last_pixel);
1024 const PStatThreadData *thread_data = _view.get_thread_data();
1025
1026 if (_average_mode && !thread_data->is_empty()) {
1027 // In average mode, we have to calculate the average value for each pixel.
1028 double start_time = pixel_to_timestamp(first_pixel);
1029 int then_i = thread_data->get_frame_number_at_time(start_time - pstats_average_time);
1030 int now_i = thread_data->get_frame_number_at_time(start_time, then_i);
1031 for (int x = first_pixel; x <= last_pixel; x++) {
1032 if (x == _cursor_pixel && !_scroll_mode) {
1033 draw_cursor(x);
1034 } else {
1035 FrameData fdata;
1036 compute_average_pixel_data(fdata, then_i, now_i, pixel_to_timestamp(x));
1037 draw_slice(x, 1, fdata);
1038 }
1039 }
1040
1041 } else {
1042 // When average mode is false, we are in frame mode; just show the actual
1043 // frame data.
1044 int frame_number = -1;
1045 int x = first_pixel;
1046 while (x <= last_pixel) {
1047 if (x == _cursor_pixel && !_scroll_mode) {
1048 draw_cursor(x);
1049 x++;
1050
1051 } else {
1052 double time = pixel_to_timestamp(x);
1053 frame_number = thread_data->get_frame_number_at_time(time, frame_number);
1054 int w = 1;
1055 int stop_pixel = last_pixel;
1056 if (!_scroll_mode) {
1057 stop_pixel = min(stop_pixel, _cursor_pixel);
1058 }
1059 while (x + w < stop_pixel &&
1060 thread_data->get_frame_number_at_time(pixel_to_timestamp(x + w), frame_number) == frame_number) {
1061 w++;
1062 }
1063 if (thread_data->has_frame(frame_number)) {
1064 draw_slice(x, w, get_frame_data(frame_number));
1065 } else {
1066 draw_empty(x, w);
1067 }
1068 x += w;
1069 }
1070 }
1071 }
1072
1073 end_draw(first_pixel, last_pixel);
1074}
