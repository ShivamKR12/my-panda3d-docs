415void MacStatsGraph::
416handle_draw_graph(CGContextRef ctx, NSRect rect) {
417 CGContextSetBlendMode(ctx, kCGBlendModeCopy);
418 CGContextSetInterpolationQuality(ctx, kCGInterpolationNone);
419
420 // Quantize this so that changed_graph_size will always call force_redraw()
421 NSRect full_rect = _graph_view.bounds;
422 full_rect.size.width = (int)full_rect.size.width;
423 full_rect.size.height = (int)full_rect.size.height;
424
425 CGSize size = CGContextConvertSizeToDeviceSpace(ctx, full_rect.size);
426 int width = abs((int)size.width);
427 int height = abs((int)size.height);
428
429 if (_ctx == nullptr || _bitmap_xsize != width || _bitmap_ysize != height) {
430 if (_ctx == nullptr && _scale_area != nullptr) {
431 _scale_area.needsDisplay = YES;
432 }
433 setup_bitmap(width, height, _window.backingScaleFactor);
434
435 changed_graph_size(full_rect.size.width, full_rect.size.height);
436 }
437
438 CGImageRef image = CGBitmapContextCreateImage(_ctx);
439 CGContextDrawImage(ctx, CGRectMake(0, 0, full_rect.size.width, full_rect.size.height), image);
440 CGImageRelease(image);
441}

469setup_bitmap(int xsize, int ysize, double scale) {
470 release_bitmap();
471

475 // Ostensibly, a layer context is more efficient, but I tried it and the

479 CGContextScaleCTM(_ctx, scale, scale);
