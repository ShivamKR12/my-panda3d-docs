160void BulletDebugNode::
161add_for_draw(CullTraverser *trav, CullTraverserData &data) {
162 PT(Geom) debug_lines;
163 PT(Geom) debug_triangles;
164
165 {
166 LightMutexHolder holder(BulletWorld::get_global_lock());
167 if (_debug_world == nullptr) {
168 return;
169 }
170 if (_debug_stale) {
171 nassertv(_debug_world != nullptr);
172 PStatTimer timer(_pstat_debug);
173
174 // Collect debug geometry data
175 _drawer._lines.clear();
176 _drawer._triangles.clear();
177
178 _debug_world->debugDrawWorld();
179
180 // Render lines
181 {
182 PT(GeomVertexData) vdata =
183 new GeomVertexData("", GeomVertexFormat::get_v3c4(), Geom::UH_stream);
184 vdata->unclean_set_num_rows(_drawer._lines.size() * 2);
185
186 GeomVertexWriter vwriter(vdata, InternalName::get_vertex());
187 GeomVertexWriter cwriter(vdata, InternalName::get_color());
188
189 pvector<Line>::const_iterator lit;
190 for (lit = _drawer._lines.begin(); lit != _drawer._lines.end(); lit++) {
191 const Line &line = *lit;
192
193 vwriter.set_data3(line._p0);
194 vwriter.set_data3(line._p1);
195 cwriter.set_data4(LVecBase4(line._color));
196 cwriter.set_data4(LVecBase4(line._color));
197 }
198
199 PT(GeomPrimitive) prim = new GeomLines(Geom::UH_stream);
200 prim->set_shade_model(Geom::SM_uniform);
201 prim->add_next_vertices(_drawer._lines.size() * 2);
202
203 debug_lines = new Geom(vdata);
204 debug_lines->add_primitive(prim);
205 _debug_lines = debug_lines;
206 }
207
208 // Render triangles
209 {
210 PT(GeomVertexData) vdata =
211 new GeomVertexData("", GeomVertexFormat::get_v3c4(), Geom::UH_stream);
212 vdata->unclean_set_num_rows(_drawer._triangles.size() * 3);
213
214 GeomVertexWriter vwriter(vdata, InternalName::get_vertex());
215 GeomVertexWriter cwriter(vdata, InternalName::get_color());
216
217 pvector<Triangle>::const_iterator tit;
218 for (tit = _drawer._triangles.begin(); tit != _drawer._triangles.end(); tit++) {
219 const Triangle &tri = *tit;
220
221 vwriter.set_data3(tri._p0);
222 vwriter.set_data3(tri._p1);
223 vwriter.set_data3(tri._p2);
224 cwriter.set_data4(LVecBase4(tri._color));
225 cwriter.set_data4(LVecBase4(tri._color));
226 cwriter.set_data4(LVecBase4(tri._color));
227 }
228
229 PT(GeomPrimitive) prim = new GeomTriangles(Geom::UH_stream);
230 prim->set_shade_model(Geom::SM_uniform);
231 prim->add_next_vertices(_drawer._triangles.size() * 3);
232
233 debug_triangles = new Geom(vdata);
234 debug_triangles->add_primitive(prim);
235 _debug_triangles = debug_triangles;
236 }
237
238 // Clear collected data.
239 _drawer._lines.clear();
240 _drawer._triangles.clear();
241
242 _debug_stale = false;
243 } else {
244 debug_lines = _debug_lines;
245 debug_triangles = _debug_triangles;
246 }
247 }
248
249 // Record them without any state or transform.
250 trav->_geoms_pcollector.add_level(2);
251 trav->get_cull_handler()->record_object(CullableObject(std::move(debug_lines), RenderState::make_empty(), trav->get_scene()->get_cs_world_transform()), trav);
252 trav->get_cull_handler()->record_object(CullableObject(std::move(debug_triangles), RenderState::make_empty(), trav->get_scene()->get_cs_world_transform()), trav);
253}

295void BulletDebugNode::DebugDraw::
296drawLine(const btVector3 &from, const btVector3 &to, const btVector3 &color) {
297
298 PN_stdfloat r = color.getX();
299 PN_stdfloat g = color.getY();
300 PN_stdfloat b = color.getZ();
301
302 // Hack to get rid of triangle normals. The hack is based on the assumption
303 // that only normals are drawn in yellow.
304 if (_normals==false && r==1.0f && g==1.0f && b==0.0f) return;
305
306 Line line;
307
308 line._p0 = LVecBase3((PN_stdfloat)from.getX(),
309 (PN_stdfloat)from.getY(),
310 (PN_stdfloat)from.getZ());
311 line._p1 = LVecBase3((PN_stdfloat)to.getX(),
312 (PN_stdfloat)to.getY(),
313 (PN_stdfloat)to.getZ());
314 line._color = UnalignedLVecBase4((PN_stdfloat)r,
315 (PN_stdfloat)g,
316 (PN_stdfloat)b, 1.0f);
317
318 _lines.push_back(line);
319}
