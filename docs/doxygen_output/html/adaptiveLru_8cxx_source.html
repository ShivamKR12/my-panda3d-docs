75void AdaptiveLru::
76do_partial_lru_update(int num_updates) {
77 // Iterate sequentially through the static list of pages. As we process
78 // each page, pop it and push it back on the tail. Stop when we have
79 // processed num_updates, or come back to the starting one.
80
81 AdaptiveLruPageStaticList *start_node = (AdaptiveLruPageStaticList *)_static_list._next;
82 if (start_node == &_static_list) {
83 // List is empty.
84 return;
85 }
86
87 AdaptiveLruPageStaticList *node = start_node;
88 do {
89 AdaptiveLruPageStaticList *next = (AdaptiveLruPageStaticList *)node->_next;
90 if (--num_updates <= 0) {
91 return;
92 }
93
94 update_page((AdaptiveLruPage *)node);
95 node->remove_from_list();
96 node->insert_before(&_static_list);
97 node = next;
98 } while (node != start_node && node != &_static_list);
99}

110void AdaptiveLru::
111update_page(AdaptiveLruPage *page) {
112 int target_priority = page->_priority;
113 unsigned int lifetime_frames = _current_frame_identifier - page->_first_frame_identifier;
114 if (lifetime_frames > 0) {
115 if (page->_update_frame_identifier) {
116 unsigned int update_frames;
117
118 update_frames = (_current_frame_identifier - page->_update_frame_identifier);
119 if (update_frames > 0) {
120 if (page->_update_total_usage > 0) {
121 PN_stdfloat update_average_frame_utilization =
122 (PN_stdfloat) (page->_update_total_usage) / (PN_stdfloat)update_frames;
123
124 page->_average_frame_utilization =
125 calculate_exponential_moving_average(update_average_frame_utilization,
126 page->_average_frame_utilization);
127 } else {
128 page->_average_frame_utilization *= 1.0f - _weight;
129 }
130
131 if (page->_average_frame_utilization >= 1.0f) {
132 int integer_average_frame_utilization;
133
134 integer_average_frame_utilization =
135 (int) ((page->_average_frame_utilization - 1.0f) *
136 (PN_stdfloat) HIGH_PRIORITY_SCALE);
137 if (integer_average_frame_utilization >= LPP_New) {
138 integer_average_frame_utilization = LPP_New;
139 }
140 integer_average_frame_utilization = LPP_New -
141 integer_average_frame_utilization;
142 target_priority = integer_average_frame_utilization;
143 } else {
144 int integer_average_frame_utilization;
145
146 integer_average_frame_utilization = (int)
147 (page->_average_frame_utilization *
148 (PN_stdfloat) LOW_PRIORITY_RANGE);
149 integer_average_frame_utilization = LOW_PRIORITY_RANGE -
150 integer_average_frame_utilization;
151 target_priority = LPP_New + integer_average_frame_utilization;
152 }
153 }
154 }
155
156 page->_update_frame_identifier = _current_frame_identifier;
157 page->_update_total_usage = 0;
158 }
159
160 if (target_priority != page->_priority) {
161 page->_priority = std::min(std::max(target_priority, 0), LPP_TotalPriorities - 1);
162 ((AdaptiveLruPageDynamicList *)page)->remove_from_list();
163 ((AdaptiveLruPageDynamicList *)page)->insert_before(&_page_array[page->_priority]);
164 }
165}

173void AdaptiveLruPage::
174enqueue_lru(AdaptiveLru *lru) {
175 if (lru != _lru && _lru != nullptr) {
176 // It was previously on a different LRU. Remove it first.
177 _lru->do_remove_page(this);
178 _lru = nullptr;
179 }
180
181 if (lru == _lru) {
182 if (_lru != nullptr) {
183 // It's already on this LRU. Access it.
184 _lru->do_access_page(this);
185 }
186 } else {
187 nassertv(lru != nullptr);
188 // Add it to a new LRU.
189 _lru = lru;
190
191 _priority = AdaptiveLru::LPP_New;
192 _first_frame_identifier = _lru->_current_frame_identifier;
193 _current_frame_identifier = _lru->_current_frame_identifier;
194 _lru->do_add_page(this);
195 }

310void AdaptiveLru::
311do_access_page(AdaptiveLruPage *page) {
312 nassertv(page != nullptr && page->_lru == this);
313 LightMutexHolder holder(_lock);
314
315 if (page->_current_frame_identifier == _current_frame_identifier) {
316 // This is the second or more time this page is accessed this frame.
317 ++(page->_current_frame_usage);
318
319 } else {
320 // This page has not yet been accessed this frame. Update it.
321 page->_current_frame_identifier = _current_frame_identifier;
322 page->_last_frame_usage = page->_current_frame_usage;
323 page->_current_frame_usage = 1;
324 }
325
326 // Move it to the tail of its priority list.
327 ((AdaptiveLruPageDynamicList *)page)->remove_from_list();
328 ((AdaptiveLruPageDynamicList *)page)->insert_before(&_page_array[page->_priority]);
329
330 ++(page->_update_total_usage);

338void AdaptiveLru::
339do_evict_to(size_t target_size, bool hard_evict) {
340 int attempts;
341
342 attempts = 0;
343 do {
344 // page out lower priority pages first
345 int index;
346 for (index = LPP_TotalPriorities - 1; index >= 0; index--) {
347
348 // Store the current end of the list. If pages re-enqueue themselves
349 // during this traversal, we don't want to visit them twice.
350 AdaptiveLruPageDynamicList *end = (AdaptiveLruPageDynamicList *)_page_array[index]._prev;
351
352 AdaptiveLruPageDynamicList *node = (AdaptiveLruPageDynamicList *)_page_array[index]._next;
353
354 while (node != &_page_array[index]) {
355 AdaptiveLruPageDynamicList *next = (AdaptiveLruPageDynamicList *)node->_next;
356 AdaptiveLruPage *page = (AdaptiveLruPage *)node;
357
358 if (attempts == 0 &&
359 (page->_current_frame_identifier + 1 >= _current_frame_identifier)) {
360 // avoid swapping out pages used in the current and last frame on
361 // the first attempt
362
363 } else {
364 // We must release the lock while we call evict_lru().
365 _lock.unlock();
366 page->evict_lru();
367 _lock.lock();
368
369 if (_total_size <= target_size) {
370 // We've evicted enough to satisfy our target.
371 return;
372 }
373 }
374 if (node == end) {
375 // We've reached the former end of the list. Stop here; everything
376 // after has been re-queued.
377 break;
378 }
379 node = next;
380 }
381 }
382 attempts++;
383 } while (hard_evict && attempts < 2);
384}

390bool AdaptiveLru::
391do_validate() {
392 bool okflag = true;
393 pset<AdaptiveLruPage *> pages;
394
395 // First, walk through the dynamic pages.
396 size_t counted_size = 0;
397 int index;
398 for (index = 0; index < LPP_TotalPriorities; ++index) {
399 AdaptiveLruPageDynamicList *node = (AdaptiveLruPageDynamicList *)_page_array[index]._next;
400 while (node != &_page_array[index]) {
401 AdaptiveLruPage *page = (AdaptiveLruPage *)node;
402 counted_size += page->_lru_size;
403 if (page->_priority != index) {
404 nout << "page " << page << " has priority " << page->_priority
405 << " but is in queue " << index << "\n";
406 okflag = false;
407 }
408
409 bool inserted_ok = pages.insert(page).second;
410 if (!inserted_ok) {
411 nout << "page " << page << " appears more than once in the dynamic index\n";
412 okflag = false;
413 }
414 node = (AdaptiveLruPageDynamicList *)node->_next;
415 }
416 }
417
418 if (counted_size != _total_size) {
419 nout << "count " << counted_size << " bytes in dynamic index, but have " << _total_size << " on record\n";
420 okflag = false;
421 }
422
423 // Now, walk through the static pages.
424 counted_size = 0;
425 AdaptiveLruPageStaticList *node = (AdaptiveLruPageStaticList *)_static_list._next;
426 while (node != &_static_list) {
427 AdaptiveLruPage *page = (AdaptiveLruPage *)node;
428 counted_size += page->_lru_size;
429
430 if (pages.find(page) == pages.end()) {
431 nout << "page " << page << " appears in dynamic index, but not in static index (or multiple times in static index)\n";
432 okflag = false;
433 } else {
434 pages.erase(page);
435 }
436 node = (AdaptiveLruPageStaticList *)node->_next;
437 }
438
439 if (counted_size != _total_size) {
440 nout << "count " << counted_size << " bytes in static index, but have " << _total_size << " on record\n";
441 okflag = false;
442 }
443
444 return okflag;
445}
