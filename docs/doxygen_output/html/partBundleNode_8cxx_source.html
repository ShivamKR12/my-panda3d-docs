63void PartBundleNode::
64xform(const LMatrix4 &mat) {
65 // With plain xform(), we can't attempt to share bundles across different
66 // nodes. Better to use apply_attribs_to_vertices(), instead.
67
68 if (mat.almost_equal(LMatrix4::ident_mat())) {
69 // Don't bother.
70 return;
71 }
72
73 LightMutexHolder holder(_lock);
74 for (PT(PartBundleHandle) handle : _bundles) {
75 PartBundle *bundle = handle->get_bundle();
76 if (bundle->get_num_nodes() > 1) {
77 PT(PartBundle) new_bundle = DCAST(PartBundle, bundle->copy_subgraph());
78 update_bundle(handle, new_bundle);
79 bundle = new_bundle;
80 }
81 bundle->xform(mat);
82 }
83}

113void PartBundleNode::
114steal_bundles(PartBundleNode *other) {
115 if (other == this) {
116 return;
117 }
118
119 LightMutexHolder other_holder(other->_lock);
120 if (other->_bundles.empty()) {
121 return;
