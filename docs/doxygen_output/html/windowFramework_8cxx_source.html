496void WindowFramework::
497center_trackball(const NodePath &object) {
498 if (_trackball == nullptr) {
499 return;
500 }
501
502 PT(BoundingVolume) volume = object.get_bounds();
503 // We expect at least a geometric bounding volume around the world.
504 nassertv(volume != nullptr);
505 nassertv(volume->is_of_type(GeometricBoundingVolume::get_class_type()));
506 CPT(GeometricBoundingVolume) gbv = DCAST(GeometricBoundingVolume, volume);
507
508 if (object.has_parent()) {
509 CPT(TransformState) net_transform = object.get_parent().get_net_transform();
510 PT(GeometricBoundingVolume) new_gbv = DCAST(GeometricBoundingVolume, gbv->make_copy());
511 new_gbv->xform(net_transform->get_mat());
512 gbv = new_gbv;
513 }
514
515 // Determine the bounding sphere around the object.
516 if (gbv->is_infinite()) {
517 framework_cat.warning()
518 << "Infinite bounding volume for " << object << "\n";
519 return;
520 }
521
522 if (gbv->is_empty()) {
523 framework_cat.warning()
524 << "Empty bounding volume for " << object << "\n";
525 return;
526 }
527
528 // The BoundingVolume might be a sphere (it's likely), but since it might
529 // not, we'll take no chances and make our own sphere.
530 PT(BoundingSphere) sphere = new BoundingSphere(gbv->get_approx_center(), 0.0f);
531 if (!sphere->extend_by(gbv)) {
532 framework_cat.warning()
533 << "Cannot determine bounding volume of " << object << "\n";
534 return;
535 }
536
537 LPoint3 center = sphere->get_center();
538 PN_stdfloat radius = sphere->get_radius();
539
540 PN_stdfloat distance = 50.0f;
541
542 // Choose a suitable distance to view the whole volume in our frame. This
543 // is based on the camera lens in use. Determine the lens based on the
544 // first camera; this will be the default camera.
545 Lens *lens = nullptr;
546 if (!_cameras.empty()) {
547 Cameras::const_iterator ci;
548 for (ci = _cameras.begin();
549 ci != _cameras.end() && lens == nullptr;
550 ++ci) {
551 lens = (*ci)->get_lens();
552 }
553 }
554
555 if (lens != nullptr) {
556 LVecBase2 fov = lens->get_fov();
557 distance = radius / ctan(deg_2_rad(std::min(fov[0], fov[1]) / 2.0f));
558
559 // Ensure the far plane is far enough back to see the entire object.
560 PN_stdfloat ideal_far_plane = distance + radius * 1.5;
561 lens->set_far(std::max(lens->get_default_far(), ideal_far_plane));
562
563 // And that the near plane is far enough forward.
564 PN_stdfloat ideal_near_plane = distance - radius;
565 lens->set_near(std::min(lens->get_default_near(), ideal_near_plane));
566 }
567
568 _trackball->set_origin(center);
569 _trackball->set_pos(LVector3::forward() * distance);
570
571 // Also set the movement scale on the trackball to be consistent with the
572 // size of the model and the lens field-of-view.
573 _trackball->set_forward_scale(distance * 0.006);
574}

620NodePath WindowFramework::
621load_model(const NodePath &parent, Filename filename) {
622 framework_cat.info() << "Loading " << filename << "\n";
623
624 // If the filename already exists where it is, or if it is fully qualified,
625 // don't search along the model path for it.
626 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
627 bool search = !(filename.is_fully_qualified() || vfs->exists(filename));
628
629 // We allow loading image files here. Check to see if it might be an image
630 // file, based on the filename extension.
631 bool is_image = false;
632 string extension = filename.get_extension();
633#ifdef HAVE_ZLIB
634 if (extension == "pz" || extension == "gz") {
635 extension = Filename(filename.get_basename_wo_extension()).get_extension();
636 }
637#endif // HAVE_ZLIB
638 TexturePool *texture_pool = TexturePool::get_global_ptr();
639 LoaderFileType *model_type = nullptr;
640
641 if (!extension.empty()) {
642 LoaderFileTypeRegistry *reg = LoaderFileTypeRegistry::get_global_ptr();
643 model_type = reg->get_type_from_extension(extension);
644
645 if (model_type == nullptr) {
646 // The extension isn't a known model file type; is it a known image file
647 // extension?
648 TexturePool *texture_pool = TexturePool::get_global_ptr();
649 if (texture_pool->get_texture_type(extension) != nullptr) {
650 // It is a known image file extension.
651 is_image = true;
652 }
653 }
654 }
655
656 LoaderOptions options = PandaFramework::_loader_options;
657 if (search) {
658 options.set_flags(options.get_flags() | LoaderOptions::LF_search);
659 } else {
660 options.set_flags(options.get_flags() & ~LoaderOptions::LF_search);
661 }
662
663 Loader loader;
664 PT(PandaNode) node;
665 if (is_image) {
666 node = load_image_as_model(filename);
667 } else {
668 node = loader.load_sync(filename, options);
669
670 // It failed to load. Is it because the extension isn't recognised? If
671 // so, then we just got done printing out the known scene types, and we
672 // should also print out the supported texture types.
673 if (node == nullptr && !is_image && model_type == nullptr) {
674 texture_pool->write_texture_types(nout, 2);
675 }
676 }
677
678 if (node == nullptr) {
679 nout << "Unable to load " << filename << "\n";
680 return NodePath::not_found();
681 }
682
683 return parent.attach_new_node(node);
684}

691NodePath WindowFramework::
692load_default_model(const NodePath &parent) {
693 CPT(RenderState) state = RenderState::make_empty();
694
695 state = state->add_attrib(ColorAttrib::make_flat(LColor(0.5, 0.5, 1.0, 1.0)));
696
697 // Get the default texture to apply to the triangle; it's compiled into the
698 // code these days.
699 string rock_floor_string((const char *)rock_floor, rock_floor_len);
700 istringstream rock_floor_strm(rock_floor_string);
701 PNMImage rock_floor_pnm;
702 if (rock_floor_pnm.read(rock_floor_strm, "rock-floor.rgb")) {
703 PT(Texture) tex = new Texture;
704 tex->set_name("rock-floor.rgb");
705 tex->load(rock_floor_pnm);
706 tex->set_minfilter(SamplerState::FT_linear);
707 tex->set_magfilter(SamplerState::FT_linear);
708 state = state->add_attrib(TextureAttrib::make(tex));
709 }
710
711 GeomNode *geomnode = new GeomNode("tri");
712
713 PT(GeomVertexData) vdata = new GeomVertexData
714 ("tri", GeomVertexFormat::get_v3n3t2(),
715 Geom::UH_static);
716 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
717 GeomVertexWriter normal(vdata, InternalName::get_normal());
718 GeomVertexWriter texcoord(vdata, InternalName::get_texcoord());
719
720 vertex.add_data3(LVertex::rfu(0.0, 0.0, 0.0));
721 vertex.add_data3(LVertex::rfu(1.0, 0.0, 0.0));
722 vertex.add_data3(LVertex::rfu(0.0, 0.0, 1.0));
723
724 normal.add_data3(LNormal::back());
725 normal.add_data3(LNormal::back());
726 normal.add_data3(LNormal::back());
727
728 texcoord.add_data2(0.0, 0.0);
729 texcoord.add_data2(1.0, 0.0);
730 texcoord.add_data2(0.0, 1.0);
731
732 PT(GeomTriangles) tri = new GeomTriangles(Geom::UH_static);
733 tri->add_consecutive_vertices(0, 3);
734 tri->close_primitive();
735
736 PT(Geom) geom = new Geom(vdata);
737 geom->add_primitive(tri);
738
739 geomnode->add_geom(geom, state);
740
741 return parent.attach_new_node(geomnode);
742}

829void WindowFramework::
830adjust_dimensions() {
831 PN_stdfloat this_aspect_ratio = aspect_ratio;
832
833 int x_size = 0, y_size = 0;
834 if (_window->has_size()) {
835 x_size = _window->get_sbs_left_x_size();
836 y_size = _window->get_sbs_left_y_size();
837 }
838
839 if (this_aspect_ratio == 0.0f) {
840 // An aspect ratio of 0.0 means to try to infer it.
841 this_aspect_ratio = 1.0f;
842 if (y_size != 0) {
843 this_aspect_ratio = (float)x_size / (float)y_size;
844 }
845 }
846
847 if (!_aspect_2d.is_empty()) {
848 PN_stdfloat a2d_top;
849 PN_stdfloat a2d_bottom;
850 PN_stdfloat a2d_left;
851 PN_stdfloat a2d_right;
852
853 if (this_aspect_ratio < 1) {
854 // If the window is TALL, lets expand the top and bottom
855 _aspect_2d.set_scale(1.0f, this_aspect_ratio, this_aspect_ratio);
856
857 a2d_top = 1.0f / this_aspect_ratio;
858 a2d_bottom = -1.0f / this_aspect_ratio;
859 a2d_left = -1.0f;
860 a2d_right = 1.0f;
861 } else {
862 // If the window is WIDE, lets expand the left and right
863 _aspect_2d.set_scale(1.0f / this_aspect_ratio, 1.0f, 1.0f);
864
865 a2d_top = 1.0f;
866 a2d_bottom = -1.0f;
867 a2d_left = this_aspect_ratio * -1.0f;
868 a2d_right = this_aspect_ratio;
869 }
870
871 // Adjust aspect_2d child nodes to new aspect ratio
872 _a2d_top_center.set_pos(0.0f, 0.0f, a2d_top);
873 _a2d_bottom_center.set_pos(0.0f, 0.0f, a2d_bottom);
874 _a2d_left_center.set_pos(a2d_left, 0.0f, 0.0f);
875 _a2d_right_center.set_pos(a2d_right, 0.0f, 0.0f);
876 _a2d_top_left.set_pos(a2d_left, 0.0f, a2d_top);
877 _a2d_top_right.set_pos(a2d_right, 0.0f, a2d_top);
878 _a2d_bottom_left.set_pos(a2d_left, 0.0f, a2d_bottom);
879 _a2d_bottom_right.set_pos(a2d_right, 0.0f, a2d_bottom);
880 }
881
882 if (!_pixel_2d.is_empty()) {
883 // Adjust the pixel 2d scale
884 if (x_size > 0){
885 _pixel_2d.set_sx(2.0f / (float)x_size);
886 }
887 _pixel_2d.set_sy(1.0f);
888 if (y_size > 0){
889 _pixel_2d.set_sz(2.0f / (float)y_size);
890 }
891 }
892
893 Cameras::iterator ci;
894 for (ci = _cameras.begin(); ci != _cameras.end(); ++ci) {
895 Lens *lens = (*ci)->get_lens();
896 if (lens != nullptr) {
897 if (y_size != 0) {
898 lens->set_film_size(x_size, y_size);
899 } else {
900 lens->set_aspect_ratio(this_aspect_ratio);
901 }
902 }
903 }

1253PT(PandaNode) WindowFramework::
1254load_image_as_model(const Filename &filename) {
1255 PT(Texture) tex = TexturePool::load_texture(filename);
1256 if (tex == nullptr) {
1257 return nullptr;
1258 }
1259
1260 // Yes, it is an image file; make a texture out of it.
1261 tex->set_minfilter(SamplerState::FT_linear_mipmap_linear);
1262 tex->set_magfilter(SamplerState::FT_linear);
1263 tex->set_wrap_u(SamplerState::WM_clamp);
1264 tex->set_wrap_v(SamplerState::WM_clamp);
1265 tex->set_wrap_w(SamplerState::WM_clamp);
1266
1267 // Ok, now make a polygon to show the texture.
1268 bool has_alpha = true;
1269 LVecBase2 tex_scale = tex->get_tex_scale();
1270
1271 // Get the size from the original image (the texture may have scaled it to
1272 // make a power of 2).
1273 int x_size = tex->get_orig_file_x_size();
1274 int y_size = tex->get_orig_file_y_size();
1275
1276 // Choose the dimensions of the polygon appropriately.
1277 PN_stdfloat left,right,top,bottom;
1278 static const PN_stdfloat scale = 10.0;
1279 if (x_size > y_size) {
1280 left = -scale;
1281 right = scale;
1282 top = (scale * y_size) / x_size;
1283 bottom = -(scale * y_size) / x_size;
1284 } else if (y_size != 0) {
1285 left = -(scale * x_size) / y_size;
1286 right = (scale * x_size) / y_size;
1287 top = scale;
1288 bottom = -scale;
1289 } else {
1290 framework_cat.warning()
1291 << "Texture size is 0 0: " << *tex << "\n";
1292
1293 left = -scale;
1294 right = scale;
1295 top = scale;
1296 bottom = -scale;
1297 }
1298
1299 PT(GeomNode) card_node = new GeomNode("card");
1300 card_node->set_attrib(TextureAttrib::make(tex));
1301 if (has_alpha) {
1302 card_node->set_attrib(TransparencyAttrib::make(TransparencyAttrib::M_alpha));
1303 }
1304
1305 bool is_3d = false;
1306 if (tex->get_texture_type() == Texture::TT_3d_texture ||
1307 tex->get_texture_type() == Texture::TT_cube_map) {
1308 // For a 3-d texture, generate a cube, instead of a plain card.
1309 is_3d = true;
1310 }
1311
1312 CPT(GeomVertexFormat) vformat;
1313 if (!is_3d) {
1314 // Vertices and 2-d texture coordinates, all we need.
1315 vformat = GeomVertexFormat::get_v3t2();
1316
1317 } else {
1318 // Vertices and 3-d texture coordinates.
1319 vformat = GeomVertexFormat::register_format
1320 (new GeomVertexArrayFormat
1321 (InternalName::get_vertex(), 3,
1322 GeomEnums::NT_stdfloat, GeomEnums::C_point,
1323 InternalName::get_texcoord(), 3,
1324 GeomEnums::NT_stdfloat, GeomEnums::C_texcoord));
1325 }
1326
1327 PT(GeomVertexData) vdata = new GeomVertexData
1328 ("card", vformat, Geom::UH_static);
1329 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
1330 GeomVertexWriter texcoord(vdata, InternalName::get_texcoord());
1331
1332 if (!is_3d) {
1333 // A normal 2-d card.
1334 vertex.add_data3(LVertex::rfu(left, 0.02, top));
1335 vertex.add_data3(LVertex::rfu(left, 0.02, bottom));
1336 vertex.add_data3(LVertex::rfu(right, 0.02, top));
1337 vertex.add_data3(LVertex::rfu(right, 0.02, bottom));
1338
1339 texcoord.add_data2(0.0f, tex_scale[1]);
1340 texcoord.add_data2(0.0f, 0.0f);
1341 texcoord.add_data2(tex_scale[0], tex_scale[1]);
1342 texcoord.add_data2(tex_scale[0], 0.0f);
1343
1344 } else {
1345 // The eight vertices of a 3-d cube.
1346 vertex.add_data3(-1.0f, -1.0f, 1.0f); // 0
1347 vertex.add_data3(-1.0f, -1.0f, -1.0f); // 1
1348 vertex.add_data3(1.0f, -1.0f, -1.0f); // 2
1349 vertex.add_data3(1.0f, -1.0f, 1.0f); // 3
1350 vertex.add_data3(1.0f, 1.0f, 1.0f); // 4
1351 vertex.add_data3(1.0f, 1.0f, -1.0f); // 5
1352 vertex.add_data3(-1.0f, 1.0f, -1.0f); // 6
1353 vertex.add_data3(-1.0f, 1.0f, 1.0f); // 7
1354
1355 texcoord.add_data3(-1.0f, -1.0f, 1.0f); // 0
1356 texcoord.add_data3(-1.0f, -1.0f, -1.0f); // 1
1357 texcoord.add_data3(1.0f, -1.0f, -1.0f); // 2
1358 texcoord.add_data3(1.0f, -1.0f, 1.0f); // 3
1359 texcoord.add_data3(1.0f, 1.0f, 1.0f); // 4
1360 texcoord.add_data3(1.0f, 1.0f, -1.0f); // 5
1361 texcoord.add_data3(-1.0f, 1.0f, -1.0f); // 6
1362 texcoord.add_data3(-1.0f, 1.0f, 1.0f); // 7
1363 }
1364
1365 PT(GeomTristrips) strip = new GeomTristrips(Geom::UH_static);
1366
1367 if (!is_3d) {
1368 // The two triangles that make up a quad.
1369 strip->add_consecutive_vertices(0, 4);
1370 strip->close_primitive();
1371
1372 } else {
1373 // The twelve triangles (six quads) that make up a cube.
1374 strip->add_vertex(7);
1375 strip->add_vertex(0);
1376 strip->add_vertex(4);
1377 strip->add_vertex(3);
1378 strip->close_primitive();
1379
1380 strip->add_vertex(1);
1381 strip->add_vertex(6);
1382 strip->add_vertex(2);
1383 strip->add_vertex(5);
1384 strip->close_primitive();
1385
1386 strip->add_vertex(5);
1387 strip->add_vertex(4);
1388 strip->add_vertex(2);
1389 strip->add_vertex(3);
1390 strip->add_vertex(1);
1391 strip->add_vertex(0);
1392 strip->add_vertex(6);
1393 strip->add_vertex(7);
1394 strip->add_vertex(5);
1395 strip->add_vertex(4);
1396 strip->close_primitive();
1397 }
1398
1399 PT(Geom) geom = new Geom(vdata);
1400 geom->add_primitive(strip);
1401
1402 card_node->add_geom(geom);
1403
1404 return card_node;
1405}
