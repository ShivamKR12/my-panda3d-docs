58void AsyncTaskManager::
59cleanup() {
60 MutexHolder holder(_lock);
61
62 if (task_cat.is_debug()) {
63 do_output(task_cat.debug());
64 task_cat.debug(false)
65 << ": cleanup()\n";
66 }
67
68 // Iterate carefully in case the tasks adjust the chain list within
69 // cleanup().
70 while (!_task_chains.empty()) {
71 PT(AsyncTaskChain) chain = _task_chains[_task_chains.size() - 1];
72 _task_chains.pop_back();
73 chain->do_cleanup();
74 }
75
76 // There might be one remaining task, the current task. Especially if it
77 // wasn't running on a thread.
78 if (_num_tasks == 1) {
79 nassertv(_tasks_by_name.size() == 1);
80 TasksByName::const_iterator tbni = _tasks_by_name.begin();
81 AsyncTask *task = (*tbni);
82 nassertv(task->_state == AsyncTask::S_servicing ||
83 task->_state == AsyncTask::S_servicing_removed);
84 task->_state = AsyncTask::S_servicing_removed;
85
86 } else {
87 // If there isn't exactly one remaining task, there should be none.
88#ifndef NDEBUG
89 nassertd(_num_tasks == 0 && _tasks_by_name.empty()) {
90 task_cat.error()
91 << "_num_tasks = " << _num_tasks << " _tasks_by_name = " << _tasks_by_name.size() << "\n";
92 TasksByName::const_iterator tbni;
93 for (tbni = _tasks_by_name.begin();
94 tbni != _tasks_by_name.end();
95 ++tbni) {
96 task_cat.error()
97 << " " << *(*tbni) << "\n";
98 }
99 }
100#endif // NDEBUG
101 }
102}

161bool AsyncTaskManager::
162remove_task_chain(const string &name) {
163 MutexHolder holder(_lock);
164
165 PT(AsyncTaskChain) chain = new AsyncTaskChain(this, name);
166 TaskChains::iterator tci = _task_chains.find(chain);
167 if (tci == _task_chains.end()) {
168 // No chain.
169 return false;
170 }
171
172 chain = (*tci);
173
174 while (chain->_num_tasks != 0) {
175 // Still has tasks.
176 task_cat.info()
177 << "Waiting for tasks on chain " << name << " to finish.\n";
178 chain->do_wait_for_tasks();
179 }
180
181 // Safe to remove.
182 chain->do_cleanup();
183 _task_chains.erase(tci);
184 return true;
185}

191void AsyncTaskManager::
192add(AsyncTask *task) {
193 nassertv(task->is_runnable());
194
195 {
196 MutexHolder holder(_lock);
197
198 if (task_cat.is_debug()) {
199 task_cat.debug()
200 << "Adding " << *task << "\n";
201 }
202
203 if (task->_state == AsyncTask::S_servicing_removed) {
204 if (task->_manager == this) {
205 // Re-adding a self-removed task; this just means clearing the removed
206 // flag.
207 task->_state = AsyncTask::S_servicing;
208 return;
209 }
210 }
211
212 nassertv(task->_manager == nullptr &&
213 task->_state == AsyncTask::S_inactive);
214 nassertv(!do_has_task(task));
215
216 _lock.unlock();
217 task->upon_birth(this);
218 _lock.lock();
219 nassertv(task->_manager == nullptr &&
220 task->_state == AsyncTask::S_inactive);
221 nassertv(!do_has_task(task));
222
223 AsyncTaskChain *chain = do_find_task_chain(task->_chain_name);
224 if (chain == nullptr) {
225 task_cat.warning()
226 << "Creating implicit AsyncTaskChain " << task->_chain_name
227 << " for " << get_type() << " " << get_name() << "\n";
228 chain = do_make_task_chain(task->_chain_name);
229 }
230 chain->do_add(task);
231 }
232}

240 MutexHolder holder(_lock);
241

244 return false;

247 if (task->_state == AsyncTask::S_servicing_removed) {
248 return false;

330size_t AsyncTaskManager::
331remove(const AsyncTaskCollection &tasks) {
332 MutexHolder holder(_lock);
333 size_t num_removed = 0;
334
335 size_t num_tasks = tasks.get_num_tasks();
336 for (size_t i = 0; i < num_tasks; ++i) {
337 PT(AsyncTask) task = tasks.get_task(i);
338
339 if (task->_manager != this) {
340 // Not a member of this manager, or already removed.
341 nassertr(!do_has_task(task), num_removed);
342 } else {
343 nassertr(task->_chain->_manager == this, num_removed);
344 if (task_cat.is_debug()) {
345 task_cat.debug()
346 << "Removing " << *task << "\n";
347 }
348 if (task->_chain->do_remove(task, true)) {
349 ++num_removed;
350 } else {
351 if (task_cat.is_debug()) {
352 task_cat.debug()
353 << " (unable to remove " << *task << ")\n";
354 }
355 }
356 }
357 }
358
359 return num_removed;
360}

484void AsyncTaskManager::
485poll() {
486 MutexHolder holder(_lock);
487
488 // We iterate through with an index, rather than with an iterator, because
489 // it's possible for a task to adjust the task_chain list during its
490 // execution.
491 for (unsigned int i = 0; i < _task_chains.size(); ++i) {
492 AsyncTaskChain *chain = _task_chains[i];
493 chain->do_poll();
494
495 if (chain->_state == AsyncTaskChain::S_interrupted) {
496 // If a task returned DS_interrupt, we need to interrupt the entire
497 // manager, since an exception state may have been set.
498 break;
499 }
500 }
501
502 // Just in case the clock was ticked explicitly by one of our polling
503 // chains.
504 _frame_cvar.notify_all();
505}

512double AsyncTaskManager::
513get_next_wake_time() const {
514 MutexHolder holder(_lock);
515
516 bool got_any = false;
517 double next_wake_time = -1.0;
518
519 TaskChains::const_iterator tci;
520 for (tci = _task_chains.begin();
521 tci != _task_chains.end();
522 ++tci) {
523 AsyncTaskChain *chain = (*tci);
524 double time = chain->do_get_next_wake_time();
525 if (time >= 0.0) {
526 if (!got_any) {
527 got_any = true;
528 next_wake_time = time;
529 } else {
530 next_wake_time = std::min(time, next_wake_time);

532 }
533 }

604void AsyncTaskManager::
605remove_task_by_name(AsyncTask *task) {
606 if (!task->get_name().empty()) {
607 // We have to scan linearly through all of the tasks with the same name.
608 TasksByName::iterator tbni = _tasks_by_name.lower_bound(task);
609 while (tbni != _tasks_by_name.end()) {
610 if ((*tbni) == task) {
611 _tasks_by_name.erase(tbni);
612 return;
613 }
614 if ((*tbni)->get_name() != task->get_name()) {
615 // Too far.
616 break;
617 }
618
619 ++tbni;
620 }
621
622 // For some reason, the task wasn't on the index.
623 nassertv(false);
624 }
