55bool glxGraphicsWindow::
56begin_frame(FrameMode mode, Thread *current_thread) {
57 PStatTimer timer(_make_current_pcollector, current_thread);
58
59 begin_frame_spam(mode);
60 if (_gsg == nullptr) {
61 return false;
62 }
63 if (_awaiting_configure_since != -1) {
64 // Don't attempt to draw while we have just reconfigured the window and we
65 // haven't got the notification back yet.
66 return false;
67 }
68
69 glxGraphicsStateGuardian *glxgsg;
70 DCAST_INTO_R(glxgsg, _gsg, false);
71 {
72 LightReMutexHolder holder(glxGraphicsPipe::_x_mutex);
73
74 if (glXGetCurrentDisplay() == _display &&
75 glXGetCurrentDrawable() == _xwindow &&
76 glXGetCurrentContext() == glxgsg->_context) {
77 // No need to make the context current again. Short-circuit this
78 // possibly-expensive call.
79 } else {
80 // Need to set the context.
81 glXMakeCurrent(_display, _xwindow, glxgsg->_context);
82 }
83 }
84
85 // Now that we have made the context current to a window, we can reset the
86 // GSG state if this is the first time it has been used. (We can't just
87 // call reset() when we construct the GSG, because reset() requires having a
88 // current context.)
89 glxgsg->reset_if_new();
90
91 if (mode == FM_render) {
92 glxgsg->push_group_marker(std::string("glxGraphicsWindow ") + get_name());
93 // begin_render_texture();
94 clear_cube_map_selection();
95 }
96
97 _gsg->set_current_properties(&get_fb_properties());
98 return _gsg->begin_frame(current_thread);
99}

171bool glxGraphicsWindow::
172open_window() {
173 glxGraphicsPipe *glx_pipe;
174 DCAST_INTO_R(glx_pipe, _pipe, false);
175
176 // GSG CreationInitialization
177 glxGraphicsStateGuardian *glxgsg;
178 if (_gsg == nullptr) {
179 // There is no old gsg. Create a new one.
180 glxgsg = new glxGraphicsStateGuardian(_engine, _pipe, nullptr);
181 glxgsg->choose_pixel_format(_fb_properties, glx_pipe->get_display(), glx_pipe->get_screen(), false, false);
182 _gsg = glxgsg;
183 } else {
184 // If the old gsg has the wrong pixel format, create a new one that shares
185 // with the old gsg.
186 DCAST_INTO_R(glxgsg, _gsg, false);
187 if (glxgsg->get_engine() != _engine ||
188 !glxgsg->get_fb_properties().subsumes(_fb_properties)) {
189 glxgsg = new glxGraphicsStateGuardian(_engine, _pipe, glxgsg);
190 glxgsg->choose_pixel_format(_fb_properties, glx_pipe->get_display(), glx_pipe->get_screen(), false, false);
191 _gsg = glxgsg;
192 }
193 }
194
195 if (glxgsg->_context == nullptr) {
196 // We're supposed to have a context at this point.
197 glxdisplay_cat.error()
198 << "No GLX context: cannot open window.\n";
199 return false;
200 }
201
202 _visual_info = glxgsg->_visual;
203 if (_visual_info == nullptr) {
204 // No X visual for this fbconfig; how can we open the window?
205 glxdisplay_cat.error()
206 << "No X visual: cannot open window.\n";
207 return false;
208 }
209
210 LightReMutexHolder holder(glxGraphicsPipe::_x_mutex);
211
212 if (glxgsg->_fbconfig != None) {
213 setup_colormap(glxgsg->_fbconfig);
214 } else {
215 setup_colormap(_visual_info);
216 }
217
218 if (!x11GraphicsWindow::open_window()) {
219 return false;
220 }
221
222 glXMakeCurrent(_display, _xwindow, glxgsg->_context);
223 glxgsg->reset_if_new();
224 if (!glxgsg->is_valid()) {
225 close_window();
226 return false;
227 }
228 if (!glxgsg->get_fb_properties().verify_hardware_software
229 (_fb_properties, glxgsg->get_gl_renderer())) {
230 close_window();
231 return false;
232 }
233 _fb_properties = glxgsg->get_fb_properties();
234
235 return true;
236}

242void glxGraphicsWindow::
243setup_colormap(GLXFBConfig fbconfig) {
244 glxGraphicsStateGuardian *glxgsg;
245 DCAST_INTO_V(glxgsg, _gsg);
246 nassertv(glxgsg->_supports_fbconfig);
247
248 XVisualInfo *visual_info = glxgsg->_glXGetVisualFromFBConfig(_display, fbconfig);
249 if (visual_info == nullptr) {
250 // No X visual; no need to set up a colormap.
251 return;
252 }
253 int visual_class = visual_info->c_class;
254 Visual *visual = visual_info->visual;
255 XFree(visual_info);
256
257 glxGraphicsPipe *glx_pipe;
258 DCAST_INTO_V(glx_pipe, _pipe);
259 X11_Window root_window = glx_pipe->get_root();
260
261 int rc, is_rgb;
262
263 switch (visual_class) {
264 case PseudoColor:
265 rc = glxgsg->_glXGetFBConfigAttrib(_display, fbconfig, GLX_RGBA, &is_rgb);
266 if (rc == 0 && is_rgb) {
267 glxdisplay_cat.warning()
268 << "mesa pseudocolor not supported.\n";
269 // this is a terrible terrible hack, but it seems to work
270 _colormap = (Colormap)0;
271
272 } else {
273 _colormap = XCreateColormap(_display, root_window,
274 visual, AllocAll);
275 }
276 break;
277 case TrueColor:
278 case DirectColor:
279 _colormap = XCreateColormap(_display, root_window,
280 visual, AllocNone);
281 break;
282 case StaticColor:
283 case StaticGray:
284 case GrayScale:
285 _colormap = XCreateColormap(_display, root_window,
286 visual, AllocNone);
287 break;

289 glxdisplay_cat.error()
