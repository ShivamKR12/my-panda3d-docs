35 // First, skip whitespace.
36 const char *p = nptr;
37 while (isspace(*p)) {
38 ++p;
39 }
40
41 // Skip an optional leading sign.
42 char sign = '+';
43 if (*p == '+' || *p == '-') {
44 sign = *p;
45 ++p;
46 }
47
48 double value = 0.0;
49
50 if (isalpha(*p)) {
51 // Windows' implementation of strtod doesn't support "inf" or "nan", so
52 // check for those here.
53 if (strncasecmp(p, "inf", 3) == 0) {
54 p += 3;
55 if (strncasecmp(p, "inity", 5) == 0) {
56 p += 5;
57 }
58 value = std::numeric_limits<double>::infinity();
59
60 } else if (strncasecmp(p, "nan", 3) == 0) {
61 p += 3;
62
63 if (*p == 's' || *p == 'S') {
64 value = std::numeric_limits<double>::signaling_NaN();
65 ++p;
66 } else {
67 if (*p == 'q' || *p == 'Q') {
68 ++p;
69 }
70 value = std::numeric_limits<double>::quiet_NaN();
71 }
72
73 // It is optionally possible to include a character sequence between
74 // parentheses after "nan", to be passed to the new nan() function.
75 // Since it isn't supported universally, we will only accept a pair of
76 // empty parentheses.
77 if (strncmp(p, "()", 2) == 0) {
78 p += 2;
79 }
80
81 } else {
82 // Pass it up to the system implementation of strtod; perhaps it knows
83 // how to deal with this string.
84 return strtod(nptr, endptr);
85 }
86
87 } else {
88 // Start reading decimal digits to the left of the decimal point.
89 bool found_digits = false;
90 while (isdigit(*p)) {
91 value = (value * 10.0) + (*p - '0');
92 found_digits = true;
93 ++p;
94 }
95
96 if (*p == '.') {
97 ++p;
98 // Read decimal digits to the right of the decimal point.
99 double multiplicand = 0.1;
100 while (isdigit(*p)) {
101 value += (*p - '0') * multiplicand;
102 ++p;
103 found_digits = true;
104 multiplicand *= 0.1;
105 }
106 }
107
108 if (!found_digits) {
109 // Not a valid float.
110 if (endptr != nullptr) {
111 *endptr = (char *)nptr;
112 }
113 return 0.0;
114 }
115

117 // There's an exponent.
118 ++p;
119
120 char esign = '+';
121 if (*p == '+' || *p == '-') {
122 esign = *p;
123 ++p;
124 }
125
126 // Start reading decimal digits to the left of the decimal point.
127 double evalue = 0.0;
128 while (isdigit(*p)) {
129 evalue = (evalue * 10.0) + (*p - '0');
130 ++p;
131 }
132
133 if (esign == '-') {
134 value /= pow(10.0, evalue);
135 } else {

137 }
138 }
139 }

142 value = -value;
143 }
144

147 }
