149void GeomPrimitive::
150add_vertex(int vertex) {
151 CDWriter cdata(_cycler, true);
152
153 if (gobj_cat.is_spam()) {
154 gobj_cat.spam()
155 << this << ".add_vertex(" << vertex << ")\n";
156 }
157
158 consider_elevate_index_type(cdata, vertex);
159
160 if (requires_unused_vertices()) {
161 int num_primitives = get_num_primitives();
162 if (num_primitives > 0 &&
163 get_num_vertices() == get_primitive_end(num_primitives - 1)) {
164 // If we are beginning a new primitive, give the derived class a chance to
165 // insert some degenerate vertices.
166 if (cdata->_vertices.is_null()) {
167 do_make_indexed(cdata);
168 }
169 append_unused_vertices(cdata->_vertices.get_write_pointer(), vertex);
170 }
171 }
172
173 if (cdata->_vertices.is_null()) {
174 // The nonindexed case. We can keep the primitive nonindexed only if the
175 // vertex number happens to be the next available vertex.
176 nassertv(cdata->_num_vertices != -1);
177 if (cdata->_num_vertices == 0) {
178 cdata->_first_vertex = vertex;
179 cdata->_num_vertices = 1;
180 cdata->_modified = Geom::get_next_modified();
181 cdata->_got_minmax = false;
182 return;
183
184 } else if (vertex == cdata->_first_vertex + cdata->_num_vertices) {
185 ++cdata->_num_vertices;
186 cdata->_modified = Geom::get_next_modified();
187 cdata->_got_minmax = false;
188 return;
189 }
190
191 // Otherwise, we need to suddenly become an indexed primitive.
192 do_make_indexed(cdata);
193 }
194
195 {
196 GeomVertexArrayDataHandle handle(cdata->_vertices.get_write_pointer(),
197 Thread::get_current_thread());
198 int num_rows = handle.get_num_rows();
199 handle.set_num_rows(num_rows + 1);
200
201 unsigned char *ptr = handle.get_write_pointer();
202 switch (cdata->_index_type) {
203 case GeomEnums::NT_uint8:
204 ((uint8_t *)ptr)[num_rows] = vertex;
205 break;
206 case GeomEnums::NT_uint16:
207 ((uint16_t *)ptr)[num_rows] = vertex;
208 break;
209 case GeomEnums::NT_uint32:
210 ((uint32_t *)ptr)[num_rows] = vertex;
211 break;
212 default:
213 nassert_raise("unsupported index type");
214 break;
215 }
216 }
217
218 cdata->_modified = Geom::get_next_modified();
219 cdata->_got_minmax = false;

343bool GeomPrimitive::
344close_primitive() {
345 int num_vertices_per_primitive = get_num_vertices_per_primitive();
346
347 CDWriter cdata(_cycler, true);
348 if (num_vertices_per_primitive == 0) {
349 // This is a complex primitive type like a triangle strip: each primitive
350 // uses a different number of vertices.
351#ifndef NDEBUG
352 int num_added;
353 if (cdata->_ends.empty()) {
354 num_added = get_num_vertices();
355 } else {
356 num_added = get_num_vertices() - cdata->_ends.back();
357 num_added -= get_num_unused_vertices_per_primitive();
358 }
359 nassertr(num_added >= get_min_num_vertices_per_primitive(), false);
360#endif
361 if (cdata->_ends.get_ref_count() > 1) {
362 PTA_int new_ends;
363 new_ends.v() = cdata->_ends.v();
364 cdata->_ends = new_ends;
365 }
366 cdata->_ends.push_back(get_num_vertices());
367
368 } else {
369#ifndef NDEBUG
370 // This is a simple primitive type like a triangle: each primitive uses
371 // the same number of vertices. Assert that we added the correct number
372 // of vertices.
373 int num_vertices_per_primitive = get_num_vertices_per_primitive();
374 int num_unused_vertices_per_primitive = get_num_unused_vertices_per_primitive();
375
376 int num_vertices = get_num_vertices();
377 nassertr((num_vertices + num_unused_vertices_per_primitive) % (num_vertices_per_primitive + num_unused_vertices_per_primitive) == 0, false)
378#endif
379 }
380
381 cdata->_modified = Geom::get_next_modified();
382
383 return true;

500void GeomPrimitive::
501offset_vertices(int offset, int begin_row, int end_row) {
502 if (offset == 0 || end_row <= begin_row) {
503 return;
504 }
505
506 nassertv(begin_row >= 0 && end_row >= 0);
507 nassertv(end_row <= get_num_vertices());
508
509 if (!is_indexed() && (begin_row > 0 || end_row < get_num_vertices())) {
510 // Make it indexed unless the whole array was specified.
511 make_indexed();
512 }
513
514 if (is_indexed()) {
515 CDWriter cdata(_cycler, true);
516
517 int strip_cut_index = get_strip_cut_index(cdata->_index_type);
518
519 // Calculate the maximum vertex over our range.
520 int max_vertex = 0;
521 {
522 GeomVertexReader index_r(cdata->_vertices.get_read_pointer(), 0);
523 index_r.set_row_unsafe(begin_row);
524 for (int j = begin_row; j < end_row; ++j) {
525 int vertex = index_r.get_data1i();
526 if (vertex != strip_cut_index) {
527 max_vertex = max(max_vertex, vertex);
528 }
529 }
530 }
531
532 consider_elevate_index_type(cdata, max_vertex + offset);
533
534 {
535 GeomVertexArrayDataHandle handle(cdata->_vertices.get_write_pointer(),
536 Thread::get_current_thread());
537
538 unsigned char *ptr = handle.get_write_pointer();
539 switch (cdata->_index_type) {
540 case GeomEnums::NT_uint8:
541 for (int i = begin_row; i < end_row; ++i) {
542 uint8_t &v = ((uint8_t *)ptr)[i];
543 if (v != 0xff) {
544 v += offset;
545 }
546 }
547 break;
548
549 case GeomEnums::NT_uint16:
550 for (int i = begin_row; i < end_row; ++i) {
551 uint16_t &v = ((uint16_t *)ptr)[i];
552 if (v != 0xffff) {
553 v += offset;
554 }
555 }
556 break;
557
558 case GeomEnums::NT_uint32:
559 for (int i = begin_row; i < end_row; ++i) {
560 uint32_t &v = ((uint32_t *)ptr)[i];
561 if (v != 0xffffffff) {
562 v += offset;
563 }
564 }
565 break;
566
567 default:
568 nassert_raise("unsupported index type");
569 break;
570 }
571 }
572
573 cdata->_modified = Geom::get_next_modified();
574 cdata->_got_minmax = false;
575 }
576 else {
577 // The supplied values cover all vertices, so we don't need to make it
578 // indexed.
579 CDWriter cdata(_cycler, true);
580
581 cdata->_first_vertex += offset;
582 cdata->_modified = Geom::get_next_modified();
583 cdata->_got_minmax = false;

586 cdata->_first_vertex + cdata->_num_vertices - 1);

595void GeomPrimitive::
596make_nonindexed(GeomVertexData *dest, const GeomVertexData *source) {
597 Thread *current_thread = Thread::get_current_thread();
598
599 int num_vertices, dest_start;
600 {
601 GeomPrimitivePipelineReader reader(this, current_thread);
602 num_vertices = reader.get_num_vertices();
603 int strip_cut_index = reader.get_strip_cut_index();
604
605 GeomVertexDataPipelineWriter data_writer(dest, false, current_thread);
606 data_writer.check_array_writers();
607 dest_start = data_writer.get_num_rows();
608 data_writer.set_num_rows(dest_start + num_vertices);
609
610 GeomVertexDataPipelineReader data_reader(source, current_thread);
611 data_reader.check_array_readers();
612
613 for (int i = 0; i < num_vertices; ++i) {
614 int v = reader.get_vertex(i);
615 nassertd(v != strip_cut_index) continue;
616 data_writer.copy_row_from(dest_start + i, data_reader, v);
617 }
618 }
619
620 set_nonindexed_vertices(dest_start, num_vertices);
621}

627void GeomPrimitive::
628pack_vertices(GeomVertexData *dest, const GeomVertexData *source) {
629 Thread *current_thread = Thread::get_current_thread();
630 if (!is_indexed()) {
631 // If the primitive is nonindexed, packing is the same as converting
632 // (again) to nonindexed.
633 make_nonindexed(dest, source);
634
635 } else {
636 // The indexed case: build up a new index as we go.
637 CPT(GeomVertexArrayData) orig_vertices = get_vertices();
638 PT(GeomVertexArrayData) new_vertices = make_index_data();
639 GeomVertexWriter index(new_vertices, 0);
640 typedef pmap<int, int> CopiedIndices;
641 CopiedIndices copied_indices;
642
643 int num_vertices = get_num_vertices();
644 int dest_start = dest->get_num_rows();
645 int strip_cut_index = get_strip_cut_index();
646
647 for (int i = 0; i < num_vertices; ++i) {
648 int v = get_vertex(i);
649 if (v == strip_cut_index) {
650 continue;
651 }
652
653 // Try to add the relation { v : size() }. If that succeeds, great; if
654 // it doesn't, look up whatever we previously added for v.
655 std::pair<CopiedIndices::iterator, bool> result =
656 copied_indices.insert(CopiedIndices::value_type(v, (int)copied_indices.size()));
657 int v2 = (*result.first).second + dest_start;
658 index.add_data1i(v2);
659
660 if (result.second) {
661 // This is the first time we've seen vertex v.
662 dest->copy_row_from(v2, source, v, current_thread);
663 }
664 }
665
666 set_vertices(new_vertices);
667 }

697get_primitive_start(int n) const {
698 int num_vertices_per_primitive = get_num_vertices_per_primitive();
699 int num_unused_vertices_per_primitive = get_num_unused_vertices_per_primitive();
700
701 if (num_vertices_per_primitive == 0) {
702 // This is a complex primitive type like a triangle strip: each primitive
703 // uses a different number of vertices.
704 CDReader cdata(_cycler);
705 nassertr(n >= 0 && n <= (int)cdata->_ends.size(), -1);
706 if (n == 0) {
707 return 0;
708 } else {
709 return cdata->_ends[n - 1] + num_unused_vertices_per_primitive;
710 }
711
712 } else {
713 // This is a simple primitive type like a triangle: each primitive uses
714 // the same number of vertices.
715 return n * (num_vertices_per_primitive + num_unused_vertices_per_primitive);
716 }

971CPT(GeomPrimitive) GeomPrimitive::
972make_points() const {
973 if (is_exact_type(GeomPoints::get_class_type())) {
974 return this;
975 }
976
977 // First, get a list of all of the vertices referenced by the original
978 // primitive.
979 BitArray bits;
980 {
981 GeomPrimitivePipelineReader reader(this, Thread::get_current_thread());
982 reader.get_referenced_vertices(bits);
983 }
984
985 // Now construct a new index array with just those bits.
986 PT(GeomVertexArrayData) new_vertices = make_index_data();
987 new_vertices->unclean_set_num_rows(bits.get_num_on_bits());
988
989 GeomVertexWriter new_index(new_vertices, 0);
990 int p = bits.get_lowest_on_bit();
991 while (p != -1) {
992 while (bits.get_bit(p)) {
993 new_index.set_data1i(p);
994 ++p;
995 }
996 int q = bits.get_next_higher_different_bit(p);
997 if (q == p) {
998 break;
999 }
1000 p = q;
1001 }
1002
1003 PT(GeomPrimitive) points = new GeomPoints(UH_dynamic);
1004 points->set_vertices(new_vertices);
1005
1006 return points;
1007}

1014CPT(GeomPrimitive) GeomPrimitive::
1015make_lines() const {
1016 if (is_exact_type(GeomLines::get_class_type())) {
1017 return this;
1018 }
1019
1020 PrimitiveType prim_type = get_primitive_type();
1021 if (prim_type == PT_lines) {
1022 // It's a line strip, just decompose it.
1023 return decompose();
1024
1025 } else if (prim_type != PT_polygons && prim_type != PT_patches) {
1026 // Don't know how to represent this in wireframe.
1027 return this;
1028 }
1029
1030 if (prim_type == PT_polygons && !is_exact_type(GeomTriangles::get_class_type())) {
1031 // Decompose tristrips. We could probably make this more efficient by
1032 // making a specific implementation of make_lines for GeomTristrips.
1033 return decompose()->make_lines();
1034 }
1035
1036 // Iterate through the primitives.
1037 int num_primitives = get_num_primitives();
1038 int verts_per_prim = get_num_vertices_per_primitive();
1039
1040 PT(GeomVertexArrayData) new_vertices = make_index_data();
1041 new_vertices->unclean_set_num_rows(num_primitives * verts_per_prim * 2);
1042
1043 GeomVertexWriter new_index(new_vertices, 0);
1044
1045 for (int i = 0; i < num_primitives; ++i) {
1046 int begin = get_primitive_start(i);
1047 int end = get_primitive_end(i);
1048 if (begin == end) {
1049 continue;
1050 }
1051 for (int vi = begin; vi < end - 1; vi++) {
1052 new_index.set_data1i(get_vertex(vi));
1053 new_index.set_data1i(get_vertex(vi + 1));
1054 }
1055 new_index.set_data1i(get_vertex(end - 1));
1056 new_index.set_data1i(get_vertex(begin));
1057 }
1058
1059 PT(GeomPrimitive) lines = new GeomLines(UH_dynamic);
1060 lines->set_vertices(new_vertices);
1061
1062 return lines;
1063}

1504int GeomPrimitive::
1505release_all() {
1506 // We have to traverse a copy of the _contexts list, because the
1507 // PreparedGraphicsObjects object will call clear_prepared() in response to
1508 // each release_index_buffer(), and we don't want to be modifying the
1509 // _contexts list while we're traversing it.
1510 Contexts temp = _contexts;
1511 int num_freed = (int)_contexts.size();
1512
1513 Contexts::const_iterator ci;
1514 for (ci = temp.begin(); ci != temp.end(); ++ci) {
1515 PreparedGraphicsObjects *prepared_objects = (*ci).first;
1516 IndexBufferContext *ibc = (*ci).second;
1517 prepared_objects->release_index_buffer(ibc);
1518 }
1519
1520 // Now that we've called release_index_buffer() on every known context, the
1521 // _contexts list should have completely emptied itself.
1522 nassertr(_contexts.empty(), num_freed);
1523
1524 return num_freed;

1645void GeomPrimitive::
1646calc_tight_bounds(LPoint3 &min_point, LPoint3 &max_point,
1647 PN_stdfloat &sq_center_dist, bool &found_any,
1648 const GeomVertexData *vertex_data,
1649 bool got_mat, const LMatrix4 &mat,
1650 const InternalName *column_name,
1651 Thread *current_thread) const {
1652 GeomVertexReader reader(vertex_data, column_name, current_thread);
1653 if (!reader.has_column()) {
1654 // No vertex data.
1655 return;
1656 }
1657
1658 CDReader cdata(_cycler, current_thread);
1659 int i = 0;
1660
1661 if (cdata->_vertices.is_null()) {
1662 // Nonindexed case.
1663 nassertv(cdata->_num_vertices != -1);
1664 if (cdata->_num_vertices == 0) {
1665 return;
1666 }
1667
1668 if (got_mat) {
1669 // Find the first non-NaN vertex.
1670 while (!found_any && i < cdata->_num_vertices) {
1671 reader.set_row(cdata->_first_vertex + i);
1672 LPoint3 first_vertex = mat.xform_point_general(reader.get_data3());
1673 if (!first_vertex.is_nan()) {
1674 min_point = first_vertex;
1675 max_point = first_vertex;
1676 sq_center_dist = first_vertex.length_squared();
1677 found_any = true;
1678 }
1679 ++i;
1680 }
1681
1682 for (; i < cdata->_num_vertices; ++i) {
1683 reader.set_row_unsafe(cdata->_first_vertex + i);
1684 nassertv(!reader.is_at_end());
1685
1686 LPoint3 vertex = mat.xform_point_general(reader.get_data3());
1687
1688 min_point.set(min(min_point[0], vertex[0]),
1689 min(min_point[1], vertex[1]),
1690 min(min_point[2], vertex[2]));
1691 max_point.set(max(max_point[0], vertex[0]),
1692 max(max_point[1], vertex[1]),
1693 max(max_point[2], vertex[2]));

1695 }
1696 } else {
1697 // Find the first non-NaN vertex.
1698 while (!found_any && i < cdata->_num_vertices) {
1699 reader.set_row(cdata->_first_vertex + i);
1700 LPoint3 first_vertex = reader.get_data3();
1701 if (!first_vertex.is_nan()) {
1702 min_point = first_vertex;
1703 max_point = first_vertex;
1704 sq_center_dist = first_vertex.length_squared();
1705 found_any = true;
1706 }
1707 ++i;
1708 }
1709

1711 reader.set_row_unsafe(cdata->_first_vertex + i);
1712 nassertv(!reader.is_at_end());
1713
1714 const LVecBase3 &vertex = reader.get_data3();
1715
1716 min_point.set(min(min_point[0], vertex[0]),
1717 min(min_point[1], vertex[1]),
1718 min(min_point[2], vertex[2]));
1719 max_point.set(max(max_point[0], vertex[0]),
1720 max(max_point[1], vertex[1]),

1724 }

1731 }

1742 if (!first_vertex.is_nan()) {
1743 min_point = first_vertex;
1744 max_point = first_vertex;
1745 sq_center_dist = first_vertex.length_squared();
1746 found_any = true;
1747 }
1748 }
1749 }
1750
1751 while (!index.is_at_end()) {
1752 int ii = index.get_data1i();
1753 if (ii == strip_cut_index) {
1754 continue;
1755 }
1756 reader.set_row_unsafe(ii);
1757 nassertv(!reader.is_at_end());
1758
1759 LPoint3 vertex = mat.xform_point_general(reader.get_data3());
1760
1761 min_point.set(min(min_point[0], vertex[0]),
1762 min(min_point[1], vertex[1]),
1763 min(min_point[2], vertex[2]));
1764 max_point.set(max(max_point[0], vertex[0]),
1765 max(max_point[1], vertex[1]),
1766 max(max_point[2], vertex[2]));
1767 sq_center_dist = max(sq_center_dist, vertex.length_squared());
1768 }
1769 } else {
1770 // Find the first non-NaN vertex.
1771 while (!found_any && !index.is_at_end()) {
1772 int ii = index.get_data1i();
1773 if (ii != strip_cut_index) {
1774 reader.set_row(ii);

1776 if (!first_vertex.is_nan()) {
