148PT(PartBundle) PartBundle::
149apply_transform(const TransformState *transform) {
150 if (transform->is_identity()) {
151 // Trivial no-op.
152 return this;
153 }
154
155 AppliedTransforms::iterator ati = _applied_transforms.find(transform);
156 if (ati != _applied_transforms.end()) {
157 if ((*ati).first.is_valid_pointer()) {
158 if (auto new_bundle = (*ati).second.lock()) {
159 // Here's our cached result.
160 return new_bundle;
161 }
162 }
163 }
164
165 PT(PartBundle) new_bundle = DCAST(PartBundle, copy_subgraph());
166 new_bundle->xform(transform->get_mat());
167
168 if (ati != _applied_transforms.end()) {
169 // A stale pointer to a deleted result. Update it.
170 (*ati).first.refresh();
171 (*ati).second = new_bundle;
172 } else {
173 // No such result yet. Store it.
174 bool inserted = _applied_transforms.insert(AppliedTransforms::value_type(transform, new_bundle)).second;
175 nassertr(inserted, new_bundle);
176 }
177
178 // Make sure the new transform gets immediately applied to all of the
179 // joints.
180 new_bundle->force_update();
181
182 return new_bundle;
183}

278PT(AnimControl) PartBundle::
279load_bind_anim(Loader *loader, const Filename &filename,
280 int hierarchy_match_flags, const PartSubset &subset,
281 bool allow_async) {
282 nassertr(loader != nullptr, nullptr);
283
284 LoaderOptions anim_options(LoaderOptions::LF_search |
285 LoaderOptions::LF_report_errors |
286 LoaderOptions::LF_convert_anim);
287 string basename = filename.get_basename_wo_extension();
288
289 int anim_index = -1;
290 CPT(AnimPreloadTable) anim_preload = _anim_preload.get_read_pointer();
291 if (anim_preload != nullptr) {
292 anim_index = anim_preload->find_anim(basename);
293 }
294
295 if (anim_index < 0 || !allow_async || !Thread::is_threading_supported()) {
296 // The animation is not present in the table, or allow_async is false.
297 // Therefore, perform an ordinary synchronous load-and-bind.
298
299 PT(PandaNode) model = loader->load_sync(filename, anim_options);
300 if (model == nullptr) {
301 // Couldn't load the file.
302 return nullptr;
303 }
304 AnimBundle *anim = AnimBundleNode::find_anim_bundle(model);
305 if (anim == nullptr) {
306 // No anim bundle.
307 return nullptr;
308 }
309 PT(AnimControl) control = bind_anim(anim, hierarchy_match_flags, subset);
310 if (control == nullptr) {
311 // Couldn't bind.
312 return nullptr;
313 }
314 control->set_anim_model(model);
315 return control;
316 }
317
318 // The animation is present in the table, so we can perform an asynchronous
319 // load-and-bind.
320 PN_stdfloat frame_rate = anim_preload->get_base_frame_rate(anim_index);
321 int num_frames = anim_preload->get_num_frames(anim_index);
322 PT(AnimControl) control =
323 new AnimControl(basename, this, frame_rate, num_frames);
324
325 if (!subset.is_include_empty()) {
326 // Figure out the actual subset of joints to be bound.
327 int joint_index = 0;
328 BitArray bound_joints;
329 find_bound_joints(joint_index, false, bound_joints, subset);
330 control->set_bound_joints(bound_joints);
331 }
332
333 PT(BindAnimRequest) request =
334 new BindAnimRequest(string("bind:") + filename.get_basename(),
335 filename, anim_options, loader, control,
336 hierarchy_match_flags, subset);
337 request->set_priority(async_bind_priority);
338 loader->load_async(request);
339
340 return control;
341}

475bool PartBundle::
476update() {
477 Thread *current_thread = Thread::get_current_thread();
478 CDWriter cdata(_cycler, false, current_thread);
479 bool any_changed = false;
480
481 double now = ClockObject::get_global_clock()->get_frame_time(current_thread);
482 if (now > cdata->_last_update + _update_delay || cdata->_anim_changed) {
483 bool anim_changed = cdata->_anim_changed;
484 bool frame_blend_flag = cdata->_frame_blend_flag;
485
486 any_changed = do_update(this, cdata, nullptr, false, anim_changed,
487 current_thread);
488
489 // Now update all the controls for next time.
490 ChannelBlend::const_iterator cbi;
491 for (cbi = cdata->_blend.begin(); cbi != cdata->_blend.end(); ++cbi) {
492 AnimControl *control = (*cbi).first;
493 control->mark_channels(frame_blend_flag);
494 }
495
496 cdata->_anim_changed = false;
497 cdata->_last_update = now;
498 }
499
500 return any_changed;
501}

576bool PartBundle::
577do_bind_anim(AnimControl *control, AnimBundle *anim,
578 int hierarchy_match_flags, const PartSubset &subset) {
579 nassertr(Thread::get_current_pipeline_stage() == 0, false);
580
581 // Make sure this pointer doesn't destruct during the lifetime of this
582 // method.
583 PT(AnimBundle) ptanim = anim;
584
585 if ((hierarchy_match_flags & HMF_ok_wrong_root_name) == 0) {
586 // Make sure the root names match.
587 if (get_name() != ptanim->get_name()) {
588 if (chan_cat.is_error()) {
589 chan_cat.error()
590 << "Root name of part (" << get_name()
591 << ") does not match that of anim (" << ptanim->get_name()
592 << ")\n";
593 }
594 return false;
595 }
596 }
597
598 if (!check_hierarchy(anim, nullptr, hierarchy_match_flags)) {
599 return false;
600 }
601
602 // Grabbing the lock early prevents any other thread in stage 0 from also
603 // trying to modify the channel list at the same time.
604 CDLockedReader cdata(_cycler);
605
606 plist<int> holes;
607 int channel_index = 0;
608 pick_channel_index(holes, channel_index);
609
610 if (!holes.empty()) {
611 channel_index = holes.front();
612 }
613
614 int joint_index = 0;
615 BitArray bound_joints;
616 if (subset.is_include_empty()) {
617 bound_joints = BitArray::all_on();
618 }
619 bind_hierarchy(ptanim, channel_index, joint_index,
620 subset.is_include_empty(), bound_joints, subset);
621 control->setup_anim(this, anim, channel_index, bound_joints);
622
623 determine_effective_channels(cdata);
624
625 return true;
626}

654void PartBundle::
655do_set_control_effect(AnimControl *control, PN_stdfloat effect, CData *cdata) {
656 nassertv(control->get_part() == this);
657
658 if (effect == 0.0f) {
659 // An effect of zero means to eliminate the control.
660 ChannelBlend::iterator cbi = cdata->_blend.find(control);
661 if (cbi != cdata->_blend.end()) {
662 cdata->_blend.erase(cbi);
663 cdata->_anim_changed = true;
664 }
665
666 } else {
667 // Otherwise we define it.
668
669 // If anim_blend_flag is false, we only allow one AnimControl at a time.
670 // Stop all of the other AnimControls.
671 if (!cdata->_anim_blend_flag) {
672 clear_and_stop_intersecting(control, cdata);
673 }
674
675 if (do_get_control_effect(control, cdata) != effect) {
676 cdata->_blend[control] = effect;
677 cdata->_anim_changed = true;
678 }
679 cdata->_last_control_set = control;
680 }
681
682 determine_effective_channels(cdata);
683}

707void PartBundle::
708clear_and_stop_intersecting(AnimControl *control, CData *cdata) {
709 ChannelBlend new_blend;
710 bool any_changed = false;
711
712 ChannelBlend::iterator cbi;
713 for (cbi = cdata->_blend.begin(); cbi != cdata->_blend.end(); ++cbi) {
714 AnimControl *ac = (*cbi).first;
715 if (ac == control ||
716 !ac->get_bound_joints().has_bits_in_common(control->get_bound_joints())) {
717 // Save this control--it's either the target control, or it has no
718 // joints in common with the target control.
719 new_blend.insert(new_blend.end(), (*cbi));
720 } else {
721 // Remove and stop this control.
722 ac->stop();
723 any_changed = true;
724 }
725 }
726
727 if (any_changed) {
728 cdata->_blend.swap(new_blend);
729 cdata->_anim_changed = true;
730 determine_effective_channels(cdata);
731 }
732}
