76bool PythonLoaderFileType::
77init(PyObject *loader) {
78 nassertr(loader != nullptr, false);
79 nassertr(_load_func == nullptr, false);
80 nassertr(_save_func == nullptr, false);
81
82 // Check the extensions member. If we already have a registered extension,
83 // it must occur in the list.
84 PyObject *extensions = PyObject_GetAttrString(loader, "extensions");
85 if (extensions != nullptr) {
86 if (PyUnicode_Check(extensions)) {
87 Dtool_Raise_TypeError("extensions list should be a list or tuple");
88 Py_DECREF(extensions);
89 return false;
90 }
91
92 PyObject *tuple = PySequence_Tuple(extensions);
93 Py_ssize_t num_items = PyTuple_GET_SIZE(tuple);
94 Py_DECREF(extensions);
95
96 if (num_items == 0) {
97 PyErr_SetString(PyExc_ValueError, "extensions list may not be empty");
98 Py_DECREF(tuple);
99 return false;
100 }
101
102 bool found_extension = false;
103
104 for (Py_ssize_t i = 0; i < num_items; ++i) {
105 PyObject *extension = PyTuple_GET_ITEM(tuple, i);
106 const char *extension_str;
107 Py_ssize_t extension_len;
108 extension_str = PyUnicode_AsUTF8AndSize(extension, &extension_len);
109 if (extension_str == nullptr) {
110 Py_DECREF(tuple);
111 return false;
112 }
113
114 if (_extension.empty()) {
115 _extension.assign(extension_str, extension_len);
116 found_extension = true;
117 } else {
118 std::string new_extension(extension_str, extension_len);
119 if (_extension == new_extension) {
120 found_extension = true;
121 } else {
122 if (!_additional_extensions.empty()) {
123 _additional_extensions += ' ';
124 }
125 _additional_extensions += new_extension;
126 }
127 }
128 }
129 Py_DECREF(tuple);
130
131 if (!found_extension) {
132 PyObject *repr = PyObject_Repr(loader);
133 loader_cat.error()
134 << "Registered extension '" << _extension
135 << "' does not occur in extensions list of "
136 << PyUnicode_AsUTF8(repr) << "\n";
137 Py_DECREF(repr);
138 return false;
139 }
140 } else {
141 return false;
142 }
143
144 PyObject *supports_compressed = PyObject_GetAttrString(loader, "supports_compressed");
145 if (supports_compressed != nullptr) {
146 if (supports_compressed == Py_True) {
147 _supports_compressed = true;
148 }
149 else if (supports_compressed == Py_False) {
150 _supports_compressed = false;
151 }
152 else {
153 Dtool_Raise_TypeError("supports_compressed must be a bool");
154 Py_DECREF(supports_compressed);
155 return false;
156 }
157 Py_DECREF(supports_compressed);
158 }
159 else {
160 PyErr_Clear();
161 }
162
163 _load_func = PyObject_GetAttrString(loader, "load_file");
164 PyErr_Clear();
165 _save_func = PyObject_GetAttrString(loader, "save_file");
166 PyErr_Clear();
167
168 if (_load_func == nullptr && _save_func == nullptr) {
169 PyErr_Format(PyExc_TypeError,
170 "loader plug-in %R does not define load_file or save_file function",
171 loader);
172 return false;
173 }
174
175 // We don't need this any more.
176 Py_CLEAR(_entry_point);
177
178 return true;
179}

184bool PythonLoaderFileType::
185ensure_loaded() const {
186 if (_load_func != nullptr || _save_func != nullptr) {
187 return true;
188 }
189 nassertr_always(_entry_point != nullptr, false);
190
191#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
192 PyGILState_STATE gstate;
193 gstate = PyGILState_Ensure();
194#endif
195
196 if (loader_cat.is_info()) {
197 PyObject *repr = PyObject_Repr(_entry_point);
198
199 loader_cat.info()
200 << "loading file type module: "
201 << PyUnicode_AsUTF8(repr) << "\n";
202 Py_DECREF(repr);
203 }
204
205 PyObject *result = PyObject_CallMethod(_entry_point, (char *)"load", nullptr);
206
207 bool success = false;
208 if (result != nullptr) {
209 success = ((PythonLoaderFileType *)this)->init(result);
210 } else {
211 PyErr_Clear();
212 PyObject *repr = PyObject_Repr(_entry_point);
213
214 loader_cat.error()
215 << "unable to load "
216 << PyUnicode_AsUTF8(repr) << "\n";
217 Py_DECREF(repr);
218 }
219
220#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
221 PyGILState_Release(gstate);
222#endif
223
224 return success;
225}

284PT(PandaNode) PythonLoaderFileType::
285load_file(const Filename &path, const LoaderOptions &options,
286 BamCacheRecord *record) const {
287 // Let's check whether the file even exists before calling Python.
288 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
289 PT(VirtualFile) vfile = vfs->get_file(path);
290 if (vfile == nullptr) {
291 return nullptr;
292 }
293
294 if (!supports_load()) {
295 return nullptr;
296 }
297
298 if (record != nullptr) {
299 record->add_dependent_file(vfile);
300 }
301
302#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
303 PyGILState_STATE gstate;
304 gstate = PyGILState_Ensure();
305#endif
306
307 // Wrap the arguments.
308 PyObject *args = PyTuple_New(3);
309 PyTuple_SET_ITEM(args, 0, DTool_CreatePyInstance((void *)&path, Dtool_Filename, false, true));
310 PyTuple_SET_ITEM(args, 1, DTool_CreatePyInstance((void *)&options, Dtool_LoaderOptions, false, true));
311 if (record != nullptr) {
312 record->ref();
313 PyTuple_SET_ITEM(args, 2, DTool_CreatePyInstanceTyped((void *)record, Dtool_BamCacheRecord, true, false, record->get_type_index()));
314 } else {
315 PyTuple_SET_ITEM(args, 2, Py_NewRef(Py_None));
316 }
317
318 PT(PandaNode) node;
319
320 PyObject *result = PythonThread::call_python_func(_load_func, args);
321 if (result != nullptr) {
322 if (DtoolInstance_Check(result)) {
323 node = (PandaNode *)DtoolInstance_UPCAST(result, Dtool_PandaNode);
324 }
325 Py_DECREF(result);
326 }
327
328 Py_DECREF(args);
329
330 if (node == nullptr) {
331 PyObject *exc_type = PyErr_Occurred();
332 if (!exc_type) {
333 loader_cat.error()
334 << "load_file must return valid PandaNode or raise exception\n";
335 } else {
336 loader_cat.error()
337 << "Loading " << path.get_basename()
338 << " failed with " << ((PyTypeObject *)exc_type)->tp_name << " exception.\n";
339 PyErr_Clear();
340 }
341 }
342
343#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
344 PyGILState_Release(gstate);
345#endif
346
347 if (node != nullptr && node->is_of_type(ModelRoot::get_class_type())) {
348 ModelRoot *model_root = DCAST(ModelRoot, node.p());
349 model_root->set_fullpath(path);
350 model_root->set_timestamp(vfile->get_timestamp());
351 }
352
353 return node;
354}

359bool PythonLoaderFileType::
360save_file(const Filename &path, const LoaderOptions &options,
361 PandaNode *node) const {
362 if (!supports_save()) {
363 return false;
364 }
365
366 nassertr(node != nullptr, false);
367 node->ref();
368
369#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
370 PyGILState_STATE gstate;
371 gstate = PyGILState_Ensure();
372#endif
373
374 // Wrap the arguments.
375 PyObject *args = PyTuple_New(3);
376 PyTuple_SET_ITEM(args, 0, DTool_CreatePyInstance((void *)&path, Dtool_Filename, false, true));
377 PyTuple_SET_ITEM(args, 1, DTool_CreatePyInstance((void *)&options, Dtool_LoaderOptions, false, true));
378 PyTuple_SET_ITEM(args, 2, DTool_CreatePyInstanceTyped((void *)node, Dtool_PandaNode, true, false, node->get_type_index()));
379
380 PyObject *result = PythonThread::call_python_func(_load_func, args);
381 Py_DECREF(args);
382 if (result != nullptr) {
383 Py_DECREF(result);
384 } else {
385 PyErr_Clear();
386 loader_cat.error()
387 << "save_file failed with an exception.\n";
388 }
389
390#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
391 PyGILState_Release(gstate);
392#endif
393
394 return (result != nullptr);
