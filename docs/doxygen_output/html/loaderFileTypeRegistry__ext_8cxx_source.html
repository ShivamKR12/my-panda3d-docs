67void Extension<LoaderFileTypeRegistry>::
68unregister_type(PyObject *type) {
69 // Are we passing in a C++ file type object?
70 LoaderFileType *extracted_type;
71 if (DtoolInstance_GetPointer(type, extracted_type, Dtool_LoaderFileType)) {
72 _this->unregister_type(extracted_type);

76 // If not, we may be passing in a Python file type.
77 PyObject *load_func = PyObject_GetAttrString(type, "load_file");
78 PyObject *save_func = PyObject_GetAttrString(type, "save_file");
79 PyErr_Clear();

81 if (load_func == nullptr && save_func == nullptr) {
82 Dtool_Raise_TypeError("expected loader type");
83 return;
84 }
85
86 // Keep looping until we've removed all instances of it.
87 bool found_any;
88 do {
89 found_any = false;
90 size_t num_types = _this->get_num_types();
91 for (size_t i = 0; i < num_types; ++i) {
92 LoaderFileType *type = _this->get_type(i);
93 if (type->is_of_type(PythonLoaderFileType::get_class_type())) {
94 PythonLoaderFileType *python_type = (PythonLoaderFileType *)type;
95 if (python_type->_load_func == load_func &&
96 python_type->_save_func == save_func) {
97 _this->unregister_type(python_type);
98 delete python_type;
99 found_any = true;
100 break;
