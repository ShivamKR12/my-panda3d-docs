115void TinyWinGraphicsWindow::
116end_flip() {
117 if (!_flip_ready) {
118 GraphicsWindow::end_flip();
119 return;
120 }
121
122 HBITMAP bm = CreateCompatibleBitmap(_hdc, _frame_buffer->xsize, _frame_buffer->ysize);
123 HDC bmdc = CreateCompatibleDC(_hdc);
124 SelectObject(bmdc, bm);
125
126 int fb_xsize = get_fb_x_size();
127 int fb_ysize = get_fb_y_size();
128 int fb_ytop = _frame_buffer->ysize - fb_ysize;
129
130 SetDIBits(_hdc, bm, fb_ytop, fb_ysize, _frame_buffer->pbuf,
131 &_bitmap_info, DIB_RGB_COLORS);
132
133 if (fb_xsize == _frame_buffer->xsize) {
134 BitBlt(_hdc, 0, 0, fb_xsize, fb_ysize,
135 bmdc, 0, 0, SRCCOPY);
136 } else {
137 // SetStretchBltMode(_hdc, HALFTONE);
138 StretchBlt(_hdc, 0, 0, _frame_buffer->xsize, _frame_buffer->ysize,
139 bmdc, 0, 0,fb_xsize, fb_ysize,
140 SRCCOPY);
141 }
142
143 DeleteDC(bmdc);
144 DeleteObject(bm);
145 GdiFlush();
146 GraphicsWindow::end_flip();
147}

185bool TinyWinGraphicsWindow::
186open_window() {
187 if (!WinGraphicsWindow::open_window()) {
188 return false;
189 }
190
191 // GSG CreationInitialization
192 TinyGraphicsStateGuardian *tinygsg;
193 if (_gsg == 0) {
194 // There is no old gsg. Create a new one.
195 tinygsg = new TinyGraphicsStateGuardian(_engine, _pipe, nullptr);
196 _gsg = tinygsg;
197 } else {
198 DCAST_INTO_R(tinygsg, _gsg, false);
199 }
200
201 _hdc = GetDC(_hWnd);
202
203 create_frame_buffer();
204 if (_frame_buffer == nullptr) {
205 tinydisplay_cat.error()
206 << "Could not create frame buffer.\n";
207 return false;
208 }
209
210 tinygsg->_current_frame_buffer = _frame_buffer;
211
212 tinygsg->reset_if_new();
213 if (!tinygsg->is_valid()) {
214 close_window();
215 return false;
216 }
217
218 return true;
