33inline void NormalizeTime(timeval &in)
34{
35 while (in.tv_usec >= USEC)
36 {
37 in.tv_usec -= USEC;
38 in.tv_sec++;
39 }
40
41 while (in.tv_usec < 0)
42 {
43 in.tv_usec += USEC;
44 in.tv_sec--;
45 }
46}

51inline void TimeDif(const struct timeval &start, const struct timeval &fin, struct timeval &answer)
