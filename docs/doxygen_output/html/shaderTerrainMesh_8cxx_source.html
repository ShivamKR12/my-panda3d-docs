125 MutexHolder holder(_lock);
126 if (!do_check_heightfield())
127 return false;

251void ShaderTerrainMesh::do_init_chunk(Chunk* chunk) {
252 if (chunk->size > _chunk_size) {
253
254 // Compute children chunk size
255 size_t child_chunk_size = chunk->size / 2;
256
257 // Subdivide chunk into 4 children
258 for (size_t y = 0; y < 2; ++y) {
259 for (size_t x = 0; x < 2; ++x) {
260 Chunk* child = new Chunk();
261 child->size = child_chunk_size;
262 child->depth = chunk->depth + 1;
263 child->x = chunk->x + x * child_chunk_size;
264 child->y = chunk->y + y * child_chunk_size;
265 do_init_chunk(child);
266 chunk->children[x + 2*y] = child;
267 }
268 }
269 } else {
270 // Final chunk, initialize all children to zero
271 for (size_t i = 0; i < 4; ++i) {
272 chunk->children[i] = nullptr;
273 }
274 }

293void ShaderTerrainMesh::do_compute_bounds(Chunk* chunk) {
294
295 // Final chunk (Leaf)
296 if (chunk->size == _chunk_size) {
297
298 // Get a pointer to the PNMImage data, this is faster than using get_xel()
299 // for all pixels, since get_xel() also includes bounds checks and so on.
300 xel* data = _heightfield.get_array();
301
302 // Pixel getter function. Note that we have to flip the Y-component, since
303 // panda itself also flips it
304 // auto get_xel = [&](size_t x, size_t y){ return data[x + (_size - 1 - y) * _size].b / (PN_stdfloat)PGM_MAXMAXVAL; };
305 #define get_xel(x, y) (data[(x) + (_size - 1 - (y)) * _size].b / (PN_stdfloat)PGM_MAXMAXVAL)
306
307 // Iterate over all pixels
308 PN_stdfloat avg_height = 0.0, min_height = 1.0, max_height = 0.0;
309 for (size_t x = 0; x < _chunk_size; ++x) {
310 for (size_t y = 0; y < _chunk_size; ++y) {
311
312 // Access data directly, to improve performance
313 PN_stdfloat height = get_xel(chunk->x + x, chunk->y + y);
314 avg_height += height;
315 min_height = min(min_height, height);
316 max_height = max(max_height, height);
317 }
318 }
319
320 // Normalize average height
321 avg_height /= _chunk_size * _chunk_size;
322
323 // Store values
324 chunk->min_height = min_height;
325 chunk->max_height = max_height;
326 chunk->avg_height = avg_height;
327
328 // Get edges in the order (0, 0) (1, 0) (0, 1) (1, 1)
329 for (size_t y = 0; y < 2; ++y) {
330 for (size_t x = 0; x < 2; ++x) {
331 chunk->edges.set_cell(x + 2 * y, get_xel(
332 chunk->x + x * (_chunk_size - 1),
333 chunk->y + y * (_chunk_size - 1)
334 ));
335 }
336 }
337
338 #undef get_xel
339
340 } else {
341
342 // Reset heights
343 chunk->avg_height = 0.0;
344 chunk->min_height = 1.0;
345 chunk->max_height = 0.0;
346
347 // Perform bounds computation for every children and merge the children values
348 for (size_t i = 0; i < 4; ++i) {
349 do_compute_bounds(chunk->children[i]);
350 chunk->avg_height += chunk->children[i]->avg_height / 4.0;
351 chunk->min_height = min(chunk->min_height, chunk->children[i]->min_height);
352 chunk->max_height = max(chunk->max_height, chunk->children[i]->max_height);
353 }
354
355 // Also take the edge points from the children
356 chunk->edges.set_x(chunk->children[0]->edges.get_x());
357 chunk->edges.set_y(chunk->children[1]->edges.get_y());
358 chunk->edges.set_z(chunk->children[2]->edges.get_z());
359 chunk->edges.set_w(chunk->children[3]->edges.get_w());
360 }
361}

374void ShaderTerrainMesh::do_create_chunk_geom() {
375
376 // Convert chunk size to an integer, because we operate on integers and get
377 // signed/unsigned mismatches otherwise
378 int size = (int)_chunk_size;
379
380 // Create vertex data
381 PT(GeomVertexData) gvd = new GeomVertexData("vertices", GeomVertexFormat::get_v3(), Geom::UH_static);
382 gvd->reserve_num_rows( (size + 3) * (size + 3) );
383 GeomVertexWriter vertex_writer(gvd, "vertex");
384
385 // Create primitive
386 PT(GeomPrimitive) triangles = nullptr;
387 if (_generate_patches) {
388 triangles = new GeomPatches(3, Geom::UH_static);
389 } else {
390 triangles = new GeomTriangles(Geom::UH_static);
391 }
392
393 // Insert chunk vertices
394 for (int y = -1; y <= size + 1; ++y) {
395 for (int x = -1; x <= size + 1; ++x) {
396 LVector3 vtx_pos(x / (PN_stdfloat)size, y / (PN_stdfloat)size, 0.0f);
397 // Stitched vertices at the cornders
398 if (x == -1 || y == -1 || x == size + 1 || y == size + 1) {
399 vtx_pos.set_z(-1.0f / (PN_stdfloat)size);
400 vtx_pos.set_x(max((PN_stdfloat)0, min((PN_stdfloat)1, vtx_pos.get_x())));
401 vtx_pos.set_y(max((PN_stdfloat)0, min((PN_stdfloat)1, vtx_pos.get_y())));
402 }
403 vertex_writer.add_data3(vtx_pos);
404 }
405 }
406
407 // Its important to use int and not size_t here, since we do store negative values
408 // auto get_point_index = [&size](int x, int y){ return (x + 1) + (size + 3) * (y + 1); };
409 #define get_point_index(x, y) (((x) + 1) + (size + 3) * ((y) + 1))
410
411 // Create triangles
412 for (int y = -1; y <= size; ++y) {
413 for (int x = -1; x <= size; ++x) {
414 // Get point indices of the quad vertices
415 int tl = get_point_index(x, y);
416 int tr = get_point_index(x + 1, y);
417 int bl = get_point_index(x, y + 1);
418 int br = get_point_index(x + 1, y + 1);
419
420 // Vary triangle scheme on each uneven quad
421 if (stm_use_hexagonal_layout && (x + y) % 2 == 0 ) {
422 triangles->add_vertices(tl, tr, br);
423 triangles->add_vertices(tl, br, bl);
424 } else {
425 triangles->add_vertices(tl, tr, bl);
426 triangles->add_vertices(bl, tr, br);
427 }
428 }
429 }
430
431 #undef get_point_index
432
433 // Construct geom
434 PT(Geom) geom = new Geom(gvd);
435 geom->add_primitive(triangles);
436
437 // Do not set any bounds, we do culling ourself
438 geom->clear_bounds();
439 geom->set_bounds(new OmniBoundingVolume());
440 _chunk_geom = geom;
441}

460void ShaderTerrainMesh::add_for_draw(CullTraverser *trav, CullTraverserData &data) {
461 MutexHolder holder(_lock);
462
463 // Make sure the terrain was properly initialized, and the geom was created
464 // successfully
465 nassertv(_data_texture != nullptr);
466 nassertv(_chunk_geom != nullptr);
467
468 _basic_collector.start();
469
470 // Get current frame count
471 int frame_count = ClockObject::get_global_clock()->get_frame_count();
472
473 if (_last_frame_count != frame_count) {
474 // Frame count changed, this means we are at the beginning of a new frame.
475 // In this case, update the frame count and reset the view index.
476 _last_frame_count = frame_count;
477 _current_view_index = 0;
478 }
479
480 // Get transform and render state for this render pass
481 CPT(TransformState) modelview_transform = data.get_internal_transform(trav);
482 CPT(RenderState) state = data._state->compose(get_state());
483
484 // Store a handle to the scene setup
485 const SceneSetup* scene = trav->get_scene();
486
487 // Get the MVP matrix, this is required for the LOD
488 const Lens* current_lens = scene->get_lens();
489 const LMatrix4& projection_mat = current_lens->get_projection_mat();
490
491 // Get the current lens bounds
492 PT(BoundingVolume) cam_bounds = scene->get_cull_bounds();
493
494 // Transform the camera bounds with the main camera transform
495 DCAST(GeometricBoundingVolume, cam_bounds)->xform(scene->get_camera_transform()->get_mat());
496
497 TraversalData traversal_data;
498 traversal_data.cam_bounds = cam_bounds;
499 traversal_data.model_mat = get_transform()->get_mat();
500 traversal_data.mvp_mat = modelview_transform->get_mat() * projection_mat;
501 traversal_data.emitted_chunks = 0;
502 traversal_data.storage_ptr = (ChunkDataEntry*)_data_texture->modify_ram_image().p();
503 traversal_data.screen_size.set(scene->get_viewport_width(), scene->get_viewport_height());
504
505 // Move write pointer so it points to the beginning of the current view
506 traversal_data.storage_ptr += _data_texture->get_x_size() * _current_view_index;
507
508 if (_update_enabled) {
509 // Traverse recursively
510 _lod_collector.start();
511 do_traverse(&_base_chunk, &traversal_data);
512 _lod_collector.stop();
513 } else {
514 // Do a rough guess of the emitted chunks, we don't know the actual count
515 // (we would have to store it). This is only for debugging anyways, so
516 // its not important we get an accurate count here.
517 traversal_data.emitted_chunks = _data_texture->get_x_size();
518 }
519
520 // Set shader inputs
521 CPT(RenderAttrib) current_shader_attrib = state->get_attrib_def(ShaderAttrib::get_class_slot());
522
523 // Make sure the user didn't forget to set a shader
524 if (!DCAST(ShaderAttrib, current_shader_attrib)->has_shader()) {
525 shader_terrain_cat.warning() << "No shader set on the terrain! You need to set the appropriate shader!" << endl;
526 }
527
528 // Should never happen
529 nassertv(current_shader_attrib != nullptr);
530
531 current_shader_attrib = DCAST(ShaderAttrib, current_shader_attrib)->set_shader_input(
532 ShaderInput("ShaderTerrainMesh.terrain_size", LVecBase2i(_size)));
533 current_shader_attrib = DCAST(ShaderAttrib, current_shader_attrib)->set_shader_input(
534 ShaderInput("ShaderTerrainMesh.chunk_size", LVecBase2i(_chunk_size)));
535 current_shader_attrib = DCAST(ShaderAttrib, current_shader_attrib)->set_shader_input(
536 ShaderInput("ShaderTerrainMesh.view_index", LVecBase2i(_current_view_index)));
537 current_shader_attrib = DCAST(ShaderAttrib, current_shader_attrib)->set_shader_input(
538 ShaderInput("ShaderTerrainMesh.data_texture", _data_texture));
539 current_shader_attrib = DCAST(ShaderAttrib, current_shader_attrib)->set_shader_input(
540 ShaderInput("ShaderTerrainMesh.heightfield", _heightfield_tex));
541 current_shader_attrib = DCAST(ShaderAttrib, current_shader_attrib)->set_instance_count(
542 traversal_data.emitted_chunks);
543
544 state = state->set_attrib(current_shader_attrib, 10000);
545
546 // Emit chunk
547 trav->get_cull_handler()->record_object(CullableObject(
548 _chunk_geom, std::move(state), std::move(modelview_transform)), trav);
549
550 // After rendering, increment the view index
551 ++_current_view_index;
552
553 if (_current_view_index > (size_t)stm_max_views) {
554 shader_terrain_cat.error() << "More views than supported! Increase the stm-max-views config variable!" << endl;
555 }
556
557 _basic_collector.stop();
558}

571CPT(TransformState) ShaderTerrainMesh::
572calc_tight_bounds(LPoint3 &min_point, LPoint3 &max_point, bool &found_any,
573 const TransformState *transform, Thread *current_thread) const {
574 CPT(TransformState) next_transform =
575 PandaNode::calc_tight_bounds(min_point, max_point, found_any, transform,
576 current_thread);
577
578 const LMatrix4 &mat = next_transform->get_mat();
579 LPoint3 terrain_min_point = LPoint3(0, 0, 0) * mat;
580 LPoint3 terrain_max_point = LPoint3(1, 1, 1) * mat;

582 min_point = terrain_min_point;
583 max_point = terrain_max_point;

586 min_point = min_point.fmin(terrain_min_point);
587 max_point = max_point.fmax(terrain_max_point);
588 }
589
590 return next_transform;
591}

632void ShaderTerrainMesh::do_traverse(Chunk* chunk, TraversalData* data, bool fully_visible) {
633
634 // Don't check bounds if we are fully visible
635 if (!fully_visible) {
636
637 // Construct chunk bounding volume
638 PN_stdfloat scale = 1.0 / (PN_stdfloat)_size;
639 LPoint3 bb_min(chunk->x * scale, chunk->y * scale, chunk->min_height);
640 LPoint3 bb_max((chunk->x + chunk->size) * scale, (chunk->y + chunk->size) * scale, chunk->max_height);
641
642 BoundingBox bbox = BoundingBox(bb_min, bb_max);
643 DCAST(GeometricBoundingVolume, &bbox)->xform(data->model_mat);
644 int intersection = data->cam_bounds->contains(&bbox);
645
646 if (intersection == BoundingVolume::IF_no_intersection) {
647 // No intersection with frustum
648 return;
649 }
650
651 // If the bounds are fully visible, there is no reason to perform culling
652 // on the children, so we set this flag to prevent any bounding computation
653 // on the child nodes.
654 fully_visible = (intersection & BoundingVolume::IF_all) != 0;
655 }
656
657 // Check if the chunk should be subdivided. In case the chunk is a leaf node,
658 // the chunk will never get subdivided.
659 // NOTE: We still always perform the LOD check. This is for the reason that
660 // the lod check also computes the CLOD factor, which is useful.
661 if (do_check_lod_matches(chunk, data) || chunk->size == _chunk_size) {
662 do_emit_chunk(chunk, data);
663 } else {
664 // Traverse children
665 for (size_t i = 0; i < 4; ++i) {
666 do_traverse(chunk->children[i], data, fully_visible);
667 }
668 }
669}

684bool ShaderTerrainMesh::do_check_lod_matches(Chunk* chunk, TraversalData* data) {
685
686 // Project all points to world space
687 LVector2 projected_points[4];
688 for (size_t y = 0; y < 2; ++y) {
689 for (size_t x = 0; x < 2; ++x) {
690
691 // Compute point in model space (0,0,0 to 1,1,1)
692 LVector3 edge_pos = LVector3(
693 (PN_stdfloat)(chunk->x + x * (chunk->size - 1)) / (PN_stdfloat)_size,
694 (PN_stdfloat)(chunk->y + y * (chunk->size - 1)) / (PN_stdfloat)_size,
695 chunk->edges.get_cell(x + 2 * y)
696 );
697 LVector4 projected = data->mvp_mat.xform(LVector4(edge_pos, 1.0));
698 if (projected.get_w() == 0.0) {
699 projected.set(0.0, 0.0, -1.0, 1.0f);
700 }
701 projected *= 1.0 / projected.get_w();
702 projected_points[x + 2 * y].set(
703 projected.get_x() * data->screen_size.get_x(),
704 projected.get_y() * data->screen_size.get_y());
705 }
706 }
707
708 // Compute the length of the edges in screen space
709 PN_stdfloat edge_top = (projected_points[1] - projected_points[3]).length_squared();
710 PN_stdfloat edge_right = (projected_points[0] - projected_points[2]).length_squared();
711 PN_stdfloat edge_bottom = (projected_points[2] - projected_points[3]).length_squared();
712 PN_stdfloat edge_left = (projected_points[0] - projected_points[1]).length_squared();
713
714 // CLOD factor
715 PN_stdfloat max_edge = max(edge_top, max(edge_right, max(edge_bottom, edge_left)));
716
717 // Micro-Optimization: We use length_squared() instead of length() to compute the
718 // maximum edge length. This reduces it to one csqrt instead of four.
719 max_edge = csqrt(max_edge);
720
721 PN_stdfloat tesselation_factor = (max_edge / _target_triangle_width) / (PN_stdfloat)_chunk_size;
722 PN_stdfloat clod_factor = max(0.0, min(1.0, 2.0 - tesselation_factor));
723
724 // Store the clod factor
725 chunk->last_clod = clod_factor;
726
727 return tesselation_factor <= 2.0;
728}

739void ShaderTerrainMesh::do_emit_chunk(Chunk* chunk, TraversalData* data) {
740 if (data->emitted_chunks >= _data_texture->get_x_size()) {
741
742 // Only print warning once
743 if (data->emitted_chunks == _data_texture->get_x_size()) {
744 shader_terrain_cat.error() << "Too many chunks in the terrain! Consider lowering the desired LOD, or increase the stm-max-chunk-count variable." << endl;
745 data->emitted_chunks++;
746 }
747 return;
748 }
749
750 ChunkDataEntry& data_entry = *data->storage_ptr;
751 data_entry.x = chunk->x;
752 data_entry.y = chunk->y;
753 data_entry.size = chunk->size / _chunk_size;
754 data_entry.clod = chunk->last_clod;
755
756 data->emitted_chunks++;
757 data->storage_ptr++;
758}

769LPoint3 ShaderTerrainMesh::uv_to_world(const LTexCoord& coord) const {
