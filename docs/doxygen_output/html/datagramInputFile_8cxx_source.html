124bool DatagramInputFile::
125get_datagram(Datagram &data) {
126 nassertr(_in != nullptr, false);
127 _read_first_datagram = true;
128
129 // First, get the size of the upcoming datagram.
130 StreamReader reader(_in, false);
131 uint32_t num_bytes_32 = reader.get_uint32();
132 if (_in->fail()) {
133 return false;
134 }
135
136 if (num_bytes_32 == 0) {
137 // A special case for a zero-length datagram: no need to try to read any
138 // data.
139 data.clear();
140 return true;
141 }
142
143 if (_in->eof()) {
144 return false;
145 }
146
147 size_t num_bytes = (size_t)num_bytes_32;
148 if (num_bytes_32 == (uint32_t)-1) {
149 // Another special case for a value larger than 32 bits.
150 uint64_t num_bytes_64 = reader.get_uint64();
151
152 if (_in->fail()) {
153 _error = true;
154 return false;
155 }
156
157 if (num_bytes_64 == 0) {
158 data.clear();
159 return false;
160 }
161
162 if (_in->eof()) {
163 return false;
164 }
165
166 num_bytes = (size_t)num_bytes_64;
167
168 // Make sure we have a reasonable datagram size for putting into memory.
169 if (num_bytes_64 != (uint64_t)num_bytes) {
170 _error = true;
171 return false;
172 }
173 }
174
175 // Now, read the datagram itself. We construct an empty datagram, use
176 // pad_bytes to make it big enough, and read *directly* into the datagram's
177 // internal buffer. Doing this saves us a copy operation.
178 data = Datagram();
179
180 size_t bytes_read = 0;
181 while (bytes_read < num_bytes) {
182 size_t bytes_left = num_bytes - bytes_read;
183
184 if (_in->eof()) {
185 _error = true;
186 return false;
187 }
188
189 // Hold up a second - datagrams >4MB are pretty large by bam/network
190 // standards. Let's take it 4MB at a time just in case the length is
191 // corrupt, so we don't allocate potentially a few GBs of RAM only to
192 // find a truncated file.
193 bytes_left = std::min(bytes_left, (size_t)4*1024*1024);
194
195 PTA_uchar buffer = data.modify_array();
196 buffer.resize(buffer.size() + bytes_left);
197 unsigned char *ptr = &buffer.p()[bytes_read];
198
199 _in->read((char *)ptr, (streamsize)bytes_left);
200 if (_in->fail()) {
201 _error = true;
202 return false;
203 }
204
205 bytes_read += bytes_left;
206 }
207
208 Thread::consider_yield();
209
210 return true;
211}

222bool DatagramInputFile::
223save_datagram(SubfileInfo &info) {
224 nassertr(_in != nullptr, false);
225 _read_first_datagram = true;
226
227 // First, get the size of the upcoming datagram.
228 StreamReader reader(_in, false);
229 size_t num_bytes_32 = reader.get_uint32();
230 if (_in->fail() || (_in->eof() && num_bytes_32 > 0)) {
231 return false;
232 }
233
234 streamsize num_bytes = (streamsize)num_bytes_32;
235 if (num_bytes_32 == (uint32_t)-1) {
236 // Another special case for a value larger than 32 bits.
237 num_bytes = reader.get_uint64();
238 }
239
240 // If this stream is file-based, we can just point the SubfileInfo directly
241 // into this file.
242 if (_file != nullptr) {
243 info = SubfileInfo(_file, _in->tellg(), num_bytes);
244 _in->seekg(num_bytes, std::ios::cur);
245 return true;
246 }
247
248 // Otherwise, we have to dump the data into a temporary file.
249 PT(TemporaryFile) tfile = new TemporaryFile(Filename::temporary("", ""));
250 pofstream out;
251 Filename filename = tfile->get_filename();
252 filename.set_binary();
253 if (!filename.open_write(out)) {
254 util_cat.error()
255 << "Couldn't write to " << tfile->get_filename() << "\n";
256 return false;
257 }
258
259 if (util_cat.is_debug()) {
260 util_cat.debug()
261 << "Copying " << num_bytes << " bytes to " << tfile->get_filename() << "\n";
262 }
263
264 streamsize num_remaining = num_bytes;
265 static const size_t buffer_size = 4096;
266 char buffer[buffer_size];
267
268 _in->read(buffer, std::min((streamsize)buffer_size, num_remaining));
269 streamsize count = _in->gcount();
270 while (count != 0) {
271 out.write(buffer, count);
272 if (out.fail()) {
273 util_cat.error()
274 << "Couldn't write " << num_bytes << " bytes to "
275 << tfile->get_filename() << "\n";
276 return false;
277 }
278 num_remaining -= count;
279 if (num_remaining == 0) {
280 break;
281 }
282 _in->read(buffer, std::min((streamsize)buffer_size, num_remaining));
283 count = _in->gcount();
284 }
285
286 if (num_remaining != 0) {
287 util_cat.error()
288 << "Truncated data stream.\n";
289 return false;
290 }
291
292 info = SubfileInfo(tfile, 0, num_bytes);
293 return true;
294}
