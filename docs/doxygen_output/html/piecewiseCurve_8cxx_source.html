227bool PiecewiseCurve::
228insert_curveseg(int ti, ParametricCurve *seg, PN_stdfloat tlength) {
229 if (ti < 0 || ti > (int)_segs.size()) {
230 return false;
231 }
232
233 if (ti == (int)_segs.size()) {
234 _segs.push_back(Curveseg(seg, get_max_t() + tlength));
235
236 } else if (ti==0) {
237 _segs.insert(_segs.begin(),
238 Curveseg(seg, tlength));
239
240 } else {
241 _segs.insert(_segs.begin() + ti,
242 Curveseg(seg, _segs[ti-1]._tend + tlength));
243 }
244
245 return true;
246}

253bool PiecewiseCurve::
254remove_curveseg(int ti) {
255 if (ti < 0 || ti >= (int)_segs.size()) {
256 return false;
257 }
258
259 PN_stdfloat tlength = get_tlength(ti);
260 _segs.erase(_segs.begin() + ti);
261
262 // Now update the _tend figures for everything after the one we removed.
263 while (ti < (int)_segs.size()) {
264 _segs[ti]._tend -= tlength;
265 ti++;
266 }
267
268 _last_ti = 0;
269 return true;
270}

335void PiecewiseCurve::
336make_nurbs(int order, int num_cvs,
337 const PN_stdfloat knots[], const LVecBase4 cvs[]) {
338 remove_all_curvesegs();
339
340 for (int i=0; i<num_cvs - order + 1; i++) {

395bool PiecewiseCurve::
396find_curve(const ParametricCurve *&curve, PN_stdfloat &t) const {
397 // Check the index computed by the last call to find_curve(). If it's still
398 // a reasonable starting value, start searching from there. This way, we
399 // take advantage of locality of reference: the search is trivial it is the
400 // same segment as last time, or the next segment after the last one.
401 if (_last_ti>0 && _segs[_last_ti-1]._tend>=t) {
402 // However, if the new t value precedes that of last time, we'll have to
403 // start over.
404
405 // We do some messy casting so we can get away with assigning a value to a
406 // member within a const function. This assignment doesn't really count
407 // as a const violation since we're just updating a cached value, not
408 // changing any real data of the class.
409 ((PiecewiseCurve *)this)->_last_ti = 0;
410 }
411
412 int ti;
413 for (ti = _last_ti; ti < (int)_segs.size(); ti++) {
414 if (_segs[ti]._tend+0.00001f > t) {
415 break;
416 }
417 }
418
419 if (ti < (int)_segs.size()) {
420 // Adjust t to the range [0,1).
421 if (ti > 0) {
422 t = (t - _segs[ti-1]._tend) / (_segs[ti]._tend - _segs[ti-1]._tend);
423 } else {
424 t /= _segs[0]._tend;
425 }
426 }
427
428 if (t < 0) {
429 // Oops.
430 curve = _segs[0]._curve;
431 t = 0.0f;
432 return false;
433 }
434
435 if (ti >= (int)_segs.size() || !_segs[ti]._curve->is_valid()) {
436 assert(ti <= (int)_segs.size());
437
438 // If we're out of bounds, or the curve is undefined, we're probably
439 // screwed. There's one exception: if we were right on a border between
440 // curves, try the curve before.
441
442 if (ti > 0 && t < _segs[ti-1]._tend+0.0001f) {
443 ti--;
444 t = 1.0f;
445 }
446
447 if (ti >= (int)_segs.size()) {
448 if (_segs.empty()) {
449 curve = nullptr;
450 t = 0.0f;
451 return false;
452 } else {
453 curve = _segs.back()._curve;
454 t = 1.0f;
455 return false;
456 }
457 } else if (!_segs[ti]._curve->is_valid()) {
458 curve = _segs[ti]._curve;
459 return false;
460 }
461 }
462
463 // Again, some messy casting so we can get away with updating the cached
464 // index value for next time.
465 ((PiecewiseCurve *)this)->_last_ti = ti;
466
467 // Now scale t back into the curve's own valid range.
468 t *= _segs[ti]._curve->get_max_t();
469 curve = _segs[ti]._curve;
470 return true;
471}
