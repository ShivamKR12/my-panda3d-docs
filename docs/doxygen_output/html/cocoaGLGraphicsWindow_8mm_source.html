51begin_frame(FrameMode mode, Thread *current_thread) {
52 PStatTimer timer(_make_current_pcollector, current_thread);
53
54 begin_frame_spam(mode);
55 if (_gsg == nullptr) {
56 return false;
57 }
58
59 CocoaGLGraphicsStateGuardian *cocoagsg;
60 DCAST_INTO_R(cocoagsg, _gsg, false);
61 nassertr(cocoagsg->_context != nil, false);
62 nassertr(_view != nil, false);
63
64 // Place a lock on the context.
65 cocoagsg->lock_context();
66
67 // Set the drawable.
68 // Although not recommended, it is technically possible to use the same
69 // context with multiple different-sized windows. If that happens, the
70 // context needs to be updated accordingly.
71 if ([cocoagsg->_context view] != _view) {
72 // XXX I'm not 100% sure that changing the view requires it to update.
73 _context_needs_update = true;
74 [cocoagsg->_context setView:_view];
75
76 if (cocoadisplay_cat.is_spam()) {
77 cocoadisplay_cat.spam()
78 << "Switching context to view " << _view << "\n";
79 }
80 }
81
82 // Update the context if necessary, to make it reallocate buffers etc.
83 if (_context_needs_update) {
84 if ([NSThread isMainThread]) {
85 [cocoagsg->_context update];
86 _context_needs_update = false;
87 } else {
88 cocoagsg->unlock_context();
89 return false;
90 }
91 }
92
93 // Lock the view for drawing.
94 if (!_properties.get_fullscreen()) {
95 nassertr_always([_view lockFocusIfCanDraw], false);
96 }
97
98 // Make the context current.
99 [cocoagsg->_context makeCurrentContext];
100
101 // Now that we have made the context current to a window, we can reset the
102 // GSG state if this is the first time it has been used. (We can't just
103 // call reset() when we construct the GSG, because reset() requires having a
104 // current context.)
105 cocoagsg->reset_if_new();
106
107 if (mode == FM_render) {
108 // begin_render_texture();
109 clear_cube_map_selection();
110 }
111
112 _gsg->set_current_properties(&get_fb_properties());
113 return _gsg->begin_frame(current_thread);

190bool CocoaGLGraphicsWindow::
191open_window() {
192 // GSG CreationInitialization
193 CocoaGLGraphicsStateGuardian *cocoagsg;
194 if (_gsg == nullptr) {
195 // There is no old gsg. Create a new one.
196 cocoagsg = new CocoaGLGraphicsStateGuardian(_engine, _pipe, nullptr);
197 cocoagsg->choose_pixel_format(_fb_properties, _display, false);
198 _gsg = cocoagsg;
199 } else {
200 // If the old gsg has the wrong pixel format, create a new one that shares
201 // with the old gsg.
202 DCAST_INTO_R(cocoagsg, _gsg, false);
203 if (cocoagsg->get_engine() != _engine ||
204 !cocoagsg->get_fb_properties().subsumes(_fb_properties)) {
205 cocoagsg = new CocoaGLGraphicsStateGuardian(_engine, _pipe, cocoagsg);
206 cocoagsg->choose_pixel_format(_fb_properties, _display, false);
207 _gsg = cocoagsg;
208 }
209 }
210
211 if (cocoagsg->_context == nil) {
212 // Could not obtain a proper context.
213 _gsg.clear();
214 close_window();
215 return false;
216 }
217
218 if (!CocoaGraphicsWindow::open_window()) {
219 return false;
220 }
221
222 // Make the context current.
223 cocoagsg->lock_context();
224 _context_needs_update = false;
225 [cocoagsg->_context makeCurrentContext];
226 [cocoagsg->_context setView:_view];
227 [cocoagsg->_context update];
228
229 cocoagsg->reset_if_new();
230 cocoagsg->unlock_context();
231
232 if (!cocoagsg->is_valid()) {
233 close_window();
234 return false;
235 }
236
237 if (!cocoagsg->get_fb_properties().verify_hardware_software
238 (_fb_properties, cocoagsg->get_gl_renderer())) {
239 close_window();
240 return false;
241 }
242 _fb_properties = cocoagsg->get_fb_properties();
243
244 return true;
245}
