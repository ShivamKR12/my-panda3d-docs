227void GeomVertexData::
228set_format(const GeomVertexFormat *format) {
229 Thread *current_thread = Thread::get_current_thread();
230 nassertv(format->is_registered());
231
232 CDLockedReader cdata(_cycler, current_thread);
233
234 if (format == cdata->_format) {
235 // Trivially no-op.
236 return;
237 }
238
239 CDWriter cdataw(_cycler, cdata, true);
240
241 // Put the current data aside, so we can copy it back in below.
242 CPT(GeomVertexData) orig_data = new GeomVertexData(*this);
243
244 // Assign the new format. This means clearing out all of our current arrays
245 // and replacing them with new, empty arrays.
246 cdataw->_format = format;
247
248 UsageHint usage_hint = cdataw->_usage_hint;
249 cdataw->_arrays.clear();
250 int num_arrays = cdataw->_format->get_num_arrays();
251 for (int i = 0; i < num_arrays; i++) {
252 PT(GeomVertexArrayData) array = new GeomVertexArrayData
253 (cdataw->_format->get_array(i), usage_hint);
254 cdataw->_arrays.push_back(array.p());
255 }
256
257 // Now copy the original data back in. This will automatically convert it
258 // to the new format.

261 clear_cache_stage();
262 cdataw->_modified = Geom::get_next_modified();
263 cdataw->_animated_vertices.clear();

449void GeomVertexData::
450copy_from(const GeomVertexData *source, bool keep_data_objects,
451 Thread *current_thread) {
452 const GeomVertexFormat *source_format = source->get_format();
453 const GeomVertexFormat *dest_format = get_format();
454
455 int num_rows = source->get_num_rows();
456 int num_arrays = source_format->get_num_arrays();
457 int source_i;
458
459 // First, check to see if any arrays can be simply appropriated for the new
460 // format, without changing the data.
461 pset<int> done_arrays;
462
463 for (source_i = 0; source_i < num_arrays; ++source_i) {
464 const GeomVertexArrayFormat *source_array_format =
465 source_format->get_array(source_i);
466
467 bool array_done = false;
468
469 int dest_num_arrays = dest_format->get_num_arrays();
470 for (int dest_i = 0;
471 dest_i < dest_num_arrays && !array_done;
472 ++dest_i) {
473 const GeomVertexArrayFormat *dest_array_format =
474 dest_format->get_array(dest_i);
475 if (dest_array_format->is_data_subset_of(*source_array_format)) {
476 // Great! Just use the same data for this one.
477 if (keep_data_objects) {
478 // Copy the data, but keep the same GeomVertexArrayData object.
479
480 modify_array_handle(dest_i)->copy_data_from(source->get_array_handle(source_i));
481 } else {
482 // Copy the GeomVertexArrayData object.
483 if (get_array(dest_i) != source->get_array(source_i)) {
484 set_array(dest_i, source->get_array(source_i));
485 }
486 }
487
488 array_done = true;
489 done_arrays.insert(dest_i);
490 }
491 }
492 }
493
494 // Now make sure the arrays we didn't share are all filled in.
495 {
496 GeomVertexDataPipelineWriter writer(this, true, Thread::get_current_thread());
497 writer.check_array_writers();
498 writer.reserve_num_rows(num_rows);
499 writer.set_num_rows(num_rows);
500 }
501
502 // Now go back through and copy any data that's left over.
503 for (source_i = 0; source_i < num_arrays; ++source_i) {
504 CPT(GeomVertexArrayDataHandle) array_handle = source->get_array_handle(source_i);
505 const unsigned char *array_data = array_handle->get_read_pointer(true);
506 const GeomVertexArrayFormat *source_array_format = source_format->get_array(source_i);
507 int num_columns = source_array_format->get_num_columns();
508 for (int di = 0; di < num_columns; ++di) {
509 const GeomVertexColumn *source_column = source_array_format->get_column(di);
510
511 int dest_i = dest_format->get_array_with(source_column->get_name());
512 if (dest_i >= 0 && done_arrays.count(dest_i) == 0) {
513 // The data type exists in the new format; we have to copy it.
514 const GeomVertexArrayFormat *dest_array_format =
515 dest_format->get_array(dest_i);
516 const GeomVertexColumn *dest_column =
517 dest_array_format->get_column(source_column->get_name());
518 nassertv(dest_column != nullptr);
519
520 if (dest_column->is_bytewise_equivalent(*source_column)) {
521 // We can do a quick bytewise copy.
522 PT(GeomVertexArrayDataHandle) dest_handle = modify_array_handle(dest_i);
523 unsigned char *dest_array_data = dest_handle->get_write_pointer();
524
525 bytewise_copy(dest_array_data + dest_column->get_start(),
526 dest_array_format->get_stride(),
527 array_data + source_column->get_start(), source_array_format->get_stride(),
528 source_column, num_rows);
529
530 } else if (dest_column->is_packed_argb() &&
531 source_column->is_uint8_rgba()) {
532 // A common special case: OpenGL color to DirectX color.
533 PT(GeomVertexArrayDataHandle) dest_handle = modify_array_handle(dest_i);
534 unsigned char *dest_array_data = dest_handle->get_write_pointer();
535
536 uint8_rgba_to_packed_argb
537 (dest_array_data + dest_column->get_start(),
538 dest_array_format->get_stride(),
539 array_data + source_column->get_start(), source_array_format->get_stride(),
540 num_rows);
541
542 } else if (dest_column->is_uint8_rgba() &&
543 source_column->is_packed_argb()) {
544 // Another common special case: DirectX color to OpenGL color.
545 PT(GeomVertexArrayDataHandle) dest_handle = modify_array_handle(dest_i);
546 unsigned char *dest_array_data = dest_handle->get_write_pointer();
547
548 packed_argb_to_uint8_rgba
549 (dest_array_data + dest_column->get_start(),
550 dest_array_format->get_stride(),
551 array_data + source_column->get_start(), source_array_format->get_stride(),
552 num_rows);
553
554 } else {
555 // A generic copy.
556 if (gobj_cat.is_debug()) {
557 gobj_cat.debug()
558 << "generic copy " << *dest_column << " from "
559 << *source_column << "\n";
560 }
561 GeomVertexWriter to(this);
562 to.set_column(dest_i, dest_column);
563 GeomVertexReader from(source);
564 from.set_column(source_i, source_column);
565
566 while (!from.is_at_end()) {
567 to.set_data4(from.get_data4());
568 }
569 }
570 }
571 }
572 }
573
574 // Also convert the animation tables as necessary.
575 const GeomVertexAnimationSpec &source_animation = source_format->get_animation();
576 const GeomVertexAnimationSpec &dest_animation = dest_format->get_animation();
577 if (source_animation != dest_animation) {
578 if (dest_animation.get_animation_type() == AT_hardware) {
579 // Convert Panda-style animation tables to hardware-style animation
580 // tables.
581 CPT(TransformBlendTable) blend_table = source->get_transform_blend_table();
582 if (blend_table != nullptr) {
583 PT(TransformTable) transform_table = new TransformTable;
584 TransformMap already_added;
585
586 if (dest_animation.get_indexed_transforms()) {
587 // Build an indexed transform array. This is easier; this means we
588 // can put the blends in any order.
589 GeomVertexWriter weight(this, InternalName::get_transform_weight());
590 GeomVertexWriter index(this, InternalName::get_transform_index());
591 GeomVertexReader from(source, InternalName::get_transform_blend());
592
593 while (!from.is_at_end()) {
594 const TransformBlend &blend = blend_table->get_blend(from.get_data1i());
595 LVecBase4 weights = LVecBase4::zero();
596 LVecBase4i indices(0, 0, 0, 0);
597
598 if (blend.get_num_transforms() <= 4) {
599 for (size_t i = 0; i < blend.get_num_transforms(); i++) {
600 weights[i] = blend.get_weight(i);
601 indices[i] = add_transform(transform_table, blend.get_transform(i),
602 already_added);
603 }
604 } else {
605 // Limit the number of blends to the four with highest weights.
606 TransformBlend blend2(blend);
607 blend2.limit_transforms(4);
608 blend2.normalize_weights();
609
610 for (size_t i = 0; i < 4; i++) {
611 weights[i] = blend2.get_weight(i);
612 indices[i] = add_transform(transform_table, blend2.get_transform(i),
613 already_added);
614 }
615 }
616
617 if (weight.has_column()) {
618 weight.set_data4(weights);
619 }
620 index.set_data4i(indices);
621 }
622 } else {
623 // Build a nonindexed transform array. This means we have to use
624 // the same n transforms, in the same order, for each vertex.
625 GeomVertexWriter weight(this, InternalName::get_transform_weight());
626 GeomVertexReader from(source, InternalName::get_transform_blend());
627
628 while (!from.is_at_end()) {
629 const TransformBlend &blend = blend_table->get_blend(from.get_data1i());
630 LVecBase4 weights = LVecBase4::zero();
631
632 for (size_t i = 0; i < blend.get_num_transforms(); i++) {
633 int index = add_transform(transform_table, blend.get_transform(i),
634 already_added);
635 nassertv(index < 4);
636 weights[index] = blend.get_weight(i);
637 }
638 if (weight.has_column()) {
639 weight.set_data4(weights);
640 }
641 }
642 }
643
644 clear_transform_blend_table();
645 set_transform_table(TransformTable::register_table(transform_table));
646 }
647 }
648 }

675CPT(GeomVertexData) GeomVertexData::
676convert_to(const GeomVertexFormat *new_format) const {
677 Thread *current_thread = Thread::get_current_thread();
678
679 if (new_format == get_format()) {
680 // Trivial case: no change is needed.
681 return this;
682 }
683
684 // Look up the new format in our cache--maybe we've recently applied it.
685 PT(CacheEntry) entry;
686
687 CacheKey key(new_format);
688
689 _cache_lock.acquire();
690 Cache::const_iterator ci = _cache.find(&key);
691 if (ci == _cache.end()) {
692 _cache_lock.release();
693
694 } else {
695 entry = (*ci).second;
696 _cache_lock.release();
697 nassertr(entry->_source == this, nullptr);
698
699 // Here's an element in the cache for this computation. Record a cache
700 // hit, so this element will stay in the cache a while longer.
701 entry->refresh(current_thread);
702
703 CDCacheReader cdata(entry->_cycler);
704 if (cdata->_result != nullptr) {
705 return cdata->_result;
706 }
707
708 // The cache entry is stale, but we'll recompute it below. Note that
709 // there's a small race condition here; another thread might recompute the
710 // cache at the same time. No big deal, since it'll compute the same
711 // result.
712 }
713
714 // Okay, convert the data to the new format.
715 if (gobj_cat.is_debug()) {
716 gobj_cat.debug()
717 << "Converting " << get_num_rows() << " rows from " << *get_format()
718 << " to " << *new_format << "\n";
719 }
720 PStatTimer timer(_convert_pcollector);
721
722 PT(GeomVertexData) new_data =
723 new GeomVertexData(get_name(), new_format, get_usage_hint());
724 new_data->set_transform_blend_table(get_transform_blend_table());
725 new_data->set_slider_table(get_slider_table());
726
727 new_data->copy_from(this, false);
728
729 // Record the new result in the cache.
730 if (entry == nullptr) {
731 // Create a new entry for the result.
732 // We don't need the key anymore, move the pointers into the CacheEntry.
733 entry = new CacheEntry((GeomVertexData *)this, std::move(key));
734
735 {
736 LightMutexHolder holder(_cache_lock);
737 bool inserted = ((GeomVertexData *)this)->_cache.insert(Cache::value_type(&entry->_key, entry)).second;
738 if (!inserted) {
739 // Some other thread must have beat us to the punch. Never mind.
740 return new_data;
741 }
742 }
743
744 // And tell the cache manager about the new entry. (It might immediately
745 // request a delete from the cache of the thing we just added.)
746 entry->record(current_thread);
747 }
748
749 // Finally, store the cached result on the entry.
750 CDCacheWriter cdata(entry->_cycler, true, current_thread);
751 cdata->_result = new_data;
752
753 return new_data;
754}

912CPT(GeomVertexData) GeomVertexData::
913animate_vertices(bool force, Thread *current_thread) const {
914#ifdef DO_PIPELINING
915 {
916 // In the pipelining case, we take a simple short-route optimization: if
917 // the vdata isn't animated, we don't need to grab any mutex first.
918 CDReader cdata(_cycler, current_thread);
919 if (cdata->_format->get_animation().get_animation_type() != AT_panda) {
920 return this;
921 }
922 }
923#endif // DO_PIPELINING
924
925 PStatTimer timer(((GeomVertexData *)this)->_char_pcollector, current_thread);
926
927 // Now that we've short-circuited the short route, we reasonably believe the
928 // vdata is animated. Grab the mutex and make sure it's still animated
929 // after we've acquired it.
930 CDLockedReader cdata(_cycler, current_thread);
931 if (cdata->_format->get_animation().get_animation_type() != AT_panda) {
932 return this;
933 }
934
935 UpdateSeq modified;
936 {
937 PStatTimer timer2(((GeomVertexData *)this)->_blends_pcollector, current_thread);
938 if (!cdata->_transform_blend_table.is_null()) {
939 if (cdata->_slider_table != nullptr) {
940 modified =
941 std::max(cdata->_transform_blend_table.get_read_pointer()->get_modified(current_thread),
942 cdata->_slider_table->get_modified(current_thread));
943 } else {
944 modified = cdata->_transform_blend_table.get_read_pointer()->get_modified(current_thread);
945 }
946
947 } else if (cdata->_slider_table != nullptr) {
948 modified = cdata->_slider_table->get_modified(current_thread);
949
950 } else {
951 // No transform blend table or slider table--ergo, no vertex animation.
952 return this;
953 }
954 }
955
956 if (cdata->_animated_vertices_modified == modified &&
957 cdata->_animated_vertices != nullptr) {
958 // No changes.
959 return cdata->_animated_vertices;
960 }
961
962 if (!force && !request_resident()) {
963 // The vertex data isn't resident. Return the best information we've got.
964 if (cdata->_animated_vertices != nullptr) {
965 return cdata->_animated_vertices;
966 }
967 return this;
968 }
969
970 CDWriter cdataw(((GeomVertexData *)this)->_cycler, cdata, false);
971 cdataw->_animated_vertices_modified = modified;
972 ((GeomVertexData *)this)->update_animated_vertices(cdataw, current_thread);
973
974 return cdataw->_animated_vertices;

1067void GeomVertexData::
1068do_set_color(GeomVertexData *vdata, const LColor &color) {
1069 // This function is used relatively often (by the SceneGraphReducer, when
1070 // flattening colors, and by the munger, when munging colors), so I've
1071 // written out a version that avoids the performance overhead of the packer
1072 // and GeomVertexWriter.
1073
1074 const GeomVertexFormat *format = vdata->get_format();
1075 const GeomVertexColumn *column;
1076 int array_index;
1077 if (!format->get_array_info(InternalName::get_color(), array_index, column)) {
1078 nassert_raise("no color column");
1079 return;
1080 }
1081
1082 size_t stride = format->get_array(array_index)->get_stride();
1083
1084 GeomVertexColumn::Packer *packer = column->_packer;
1085 nassertv(packer != nullptr);
1086
1087 // Pack into a buffer, which we will then copy.
1088 unsigned char buffer[32];
1089 size_t bufsize = column->get_total_bytes();
1090#ifdef STDFLOAT_DOUBLE
1091 packer->set_data4d(buffer, color);
1092#else
1093 packer->set_data4f(buffer, color);
1094#endif
1095
1096 PT(GeomVertexArrayDataHandle) handle = vdata->modify_array_handle(array_index);
1097 unsigned char *write_ptr = handle->get_write_pointer();
1098 unsigned char *end_ptr = write_ptr + handle->get_data_size_bytes();
1099 write_ptr += column->get_start();
1100
1101 if (bufsize == 4) {
1102 // Most common case.
1103 while (write_ptr < end_ptr) {
1104 write_ptr[0] = buffer[0];
1105 write_ptr[1] = buffer[1];
1106 write_ptr[2] = buffer[2];
1107 write_ptr[3] = buffer[3];
1108 write_ptr += stride;
1109 }
1110 } else {
1111 while (write_ptr < end_ptr) {
1112 memcpy(write_ptr, buffer, bufsize);
1113 write_ptr += stride;
1114 }
1115 }
1116}

1121void GeomVertexData::
1122bytewise_copy(unsigned char *to, int to_stride,
1123 const unsigned char *from, int from_stride,
1124 const GeomVertexColumn *from_type,
1125 int num_records) {
1126 if (gobj_cat.is_debug()) {
1127 gobj_cat.debug()
1128 << "bytewise_copy(" << (void *)to << ", " << to_stride
1129 << ", " << (const void *)from << ", " << from_stride
1130 << ", " << *from_type << ", " << num_records << ")\n";
1131 }
1132 if (to_stride == from_type->get_total_bytes() &&
1133 from_stride == from_type->get_total_bytes()) {
1134 // Fantastic! It's just a linear array of this one data type. Copy the
1135 // whole thing all at once.
1136 memcpy(to, from, num_records * from_type->get_total_bytes());
1137
1138 } else {
1139 // Ok, it's interleaved in with other data. Copy them one record at a
1140 // time.
1141 while (num_records > 0) {
1142 memcpy(to, from, from_type->get_total_bytes());
1143 to += to_stride;
1144 from += from_stride;
1145 num_records--;
1146 }
1147 }

1159PT(GeomVertexData) GeomVertexData::
1160replace_column(InternalName *name, int num_components,
1161 GeomVertexData::NumericType numeric_type,
1162 GeomVertexData::Contents contents) const {
1163 CDReader cdata(_cycler);
1164 PT(GeomVertexFormat) new_format = new GeomVertexFormat(*cdata->_format);
1165
1166 // Remove the old description of the type from the format.
1167 bool removed_type_array = false;
1168 int old_type_array = cdata->_format->get_array_with(name);
1169 if (old_type_array != -1) {
1170 GeomVertexArrayFormat *array_format = new_format->modify_array(old_type_array);
1171 if (array_format->get_num_columns() == 1) {
1172 // Actually, this array didn't have any other data types, so just drop
1173 // the whole array.
1174 new_format->remove_array(old_type_array);
1175 removed_type_array = true;
1176
1177 } else {
1178 // Remove the description for the type, but don't bother to repack the
1179 // array.
1180 array_format->remove_column(name);
1181 }
1182 }
1183
1184 // Now define a new array to contain just the type.
1185 int new_type_array = -1;
1186 if (num_components != 0) {
1187 PT(GeomVertexArrayFormat) type_array_format =
1188 new GeomVertexArrayFormat(name, num_components, numeric_type, contents);
1189 new_type_array = new_format->add_array(type_array_format);
1190 }
1191
1192 CPT(GeomVertexFormat) format =
1193 GeomVertexFormat::register_format(new_format);
1194
1195 if (gobj_cat.is_debug()) {
1196 gobj_cat.debug()
1197 << "Replacing data type " << *name << "; converting "
1198 << get_num_rows() << " rows from "
1199 << *cdata->_format << " to " << *format << "\n";
1200 }
1201
1202 PT(GeomVertexData) new_data = new GeomVertexData(*this, format);
1203
1204 int j = 0;
1205 int num_arrays = get_num_arrays();
1206 for (int i = 0; i < num_arrays; ++i) {
1207 if (i == old_type_array) {
1208 if (!removed_type_array) {
1209 // Pointer-copy the original array that includes the type (since it
1210 // also includes other data).
1211 new_data->set_array(j, get_array(i));
1212 ++j;
1213 }
1214
1215 } else {
1216 // Just pointer-copy any arrays other than type.
1217 new_data->set_array(j, get_array(i));
1218 ++j;
1219 }
1220 }
1221
1222 if (new_type_array != -1) {
1223 nassertr(j == new_type_array, new_data);
1224
1225 // For the new type array, we set up a temporary array that has room for
1226 // the right number of rows.
1227 PT(GeomVertexArrayData) new_array = new GeomVertexArrayData
1228 (format->get_array(j), get_usage_hint());
1229 new_array->set_num_rows(get_num_rows());
1230 new_data->set_array(j, new_array);
1231 }
1232
1233 return new_data;
1234}

1433void GeomVertexData::
1434update_animated_vertices(GeomVertexData::CData *cdata, Thread *current_thread) {
1435 PStatTimer timer(_char_pcollector, current_thread);
1436
1437 int num_rows = get_num_rows();
1438
1439 if (gobj_cat.is_debug()) {
1440 gobj_cat.debug()
1441 << "Animating " << num_rows << " vertices for " << get_name()
1442 << "\n";
1443 }
1444
1445 const GeomVertexFormat *orig_format = cdata->_format;
1446 CPT(GeomVertexFormat) new_format = orig_format;
1447
1448 if (cdata->_animated_vertices == nullptr) {
1449 new_format = orig_format->get_post_animated_format();
1450 cdata->_animated_vertices =
1451 new GeomVertexData(get_name(), new_format,
1452 std::min(get_usage_hint(), UH_dynamic));
1453 }
1454 PT(GeomVertexData) new_data = cdata->_animated_vertices;
1455
1456 // We have to make a complete copy of the data first so we can modify it.
1457 // If we were clever, we could maybe just figure out the subset of the data
1458 // that might have changed since last frame, but that's too much trouble
1459 // (and isn't obviously faster than just copying the whole thing).
1460 new_data->copy_from(this, true);
1461
1462 // First, apply all of the morphs.
1463 CPT(SliderTable) slider_table = cdata->_slider_table;
1464 if (slider_table != nullptr) {
1465 PStatTimer timer2(_morphs_pcollector);
1466 int num_morphs = orig_format->get_num_morphs();
1467 for (int mi = 0; mi < num_morphs; mi++) {
1468 CPT(InternalName) slider_name = orig_format->get_morph_slider(mi);
1469
1470 const SparseArray &sliders = slider_table->find_sliders(slider_name);
1471 if (!sliders.is_zero()) {
1472 nassertv(!sliders.is_inverse());
1473 int num_slider_subranges = sliders.get_num_subranges();
1474 for (int sni = 0; sni < num_slider_subranges; ++sni) {
1475 int slider_begin = sliders.get_subrange_begin(sni);
1476 int slider_end = sliders.get_subrange_end(sni);
1477 for (int sn = slider_begin; sn < slider_end; ++sn) {
1478 const VertexSlider *slider = slider_table->get_slider(sn);
1479 const SparseArray &rows = slider_table->get_slider_rows(sn);
1480 nassertv(!rows.is_inverse());
1481
1482 PN_stdfloat slider_value = slider->get_slider();
1483 if (slider_value != 0.0f) {
1484 CPT(InternalName) base_name = orig_format->get_morph_base(mi);
1485 CPT(InternalName) delta_name = orig_format->get_morph_delta(mi);
1486
1487 GeomVertexRewriter data(new_data, base_name);
1488 GeomVertexReader delta(this, delta_name);
1489 int num_subranges = rows.get_num_subranges();
1490
1491 if (data.get_column()->get_num_values() == 4) {
1492 if (data.get_column()->has_homogeneous_coord()) {
1493 // Scale the delta by the homogeneous coordinate.
1494 for (int i = 0; i < num_subranges; ++i) {
1495 int begin = rows.get_subrange_begin(i);
1496 int end = rows.get_subrange_end(i);
1497 data.set_row_unsafe(begin);
1498 delta.set_row_unsafe(begin);
1499 for (int j = begin; j < end; ++j) {
1500 LPoint4 vertex = data.get_data4();
1501 LPoint3 d = delta.get_data3();
1502 d *= slider_value * vertex[3];
1503 data.set_data4(vertex[0] + d[0],
1504 vertex[1] + d[1],
1505 vertex[2] + d[2],
1506 vertex[3]);
1507 }
1508 }
1509 } else {
1510 // Just apply the four-component delta.
1511 for (int i = 0; i < num_subranges; ++i) {
1512 int begin = rows.get_subrange_begin(i);
1513 int end = rows.get_subrange_end(i);
1514 data.set_row_unsafe(begin);
1515 delta.set_row_unsafe(begin);
1516 for (int j = begin; j < end; ++j) {
1517 const LPoint4 &vertex = data.get_data4();
1518 LPoint4 d = delta.get_data4();
1519 data.set_data4(vertex + d * slider_value);
1520 }
1521 }
1522 }
1523 } else {
1524 // 3-component or smaller values; don't worry about a
1525 // homogeneous coordinate.
1526 for (int i = 0; i < num_subranges; ++i) {
1527 int begin = rows.get_subrange_begin(i);
1528 int end = rows.get_subrange_end(i);
1529 data.set_row_unsafe(begin);
1530 delta.set_row_unsafe(begin);
1531 for (int j = begin; j < end; ++j) {
1532 const LPoint3 &vertex = data.get_data3();
1533 LPoint3 d = delta.get_data3();
1534 data.set_data3(vertex + d * slider_value);
1535 }
1536 }
1537 }
1538 }
1539 }

1545 // Then apply the transforms.
1546 CPT(TransformBlendTable) tb_table = cdata->_transform_blend_table.get_read_pointer(current_thread);
1547 if (tb_table != nullptr) {
1548 // Recompute all the blends up front, so we don't have to test each one
1549 // for staleness at each vertex.
1550 {
1551 PStatTimer timer4(_blends_pcollector);
1552 int num_blends = tb_table->get_num_blends();
1553 for (int bi = 0; bi < num_blends; bi++) {
1554 tb_table->get_blend(bi).update_blend(current_thread);
1555 }
1556 }

1559 PStatTimer timer3(_skinning_pcollector);

1562 int num_subranges = rows.get_num_subranges();

1575 blend_array_format->get_column(0)->get_component_bytes() == 2) {

1578 new GeomVertexArrayDataHandle(cdata->_arrays[blend_array_index].get_read_pointer(current_thread), current_thread);
1579 const unsigned short *blendt = (const unsigned short *)blend_array_handle->get_read_pointer(true);

1582 for (ci = 0; ci < new_format->get_num_points(); ci++) {

1584

1587 int end = rows.get_subrange_end(i);

1595 // vertices that shares the blend index first_bi.

1598 // the next vertex with a different blend index.

1608 }

1611 // blend index. Transform all those vertices as a block.

1758void GeomVertexData::
1759do_transform_point_column(const GeomVertexFormat *format, GeomVertexRewriter &data,
1760 const LMatrix4 &mat, int begin_row, int end_row) {
1761 const GeomVertexColumn *data_column = data.get_column();
1762 int num_values = data_column->get_num_values();
1763
1764 if ((num_values == 3 || num_values == 4) &&
1765 data_column->get_numeric_type() == NT_float32) {
1766 // The table of points is a table of LPoint3f's or LPoint4f's. Optimize
1767 // this common case.
1768 GeomVertexArrayDataHandle *data_handle = data.get_array_handle();
1769
1770 size_t stride = data.get_stride();
1771 size_t num_rows = end_row - begin_row;
1772 unsigned char *datat = data_handle->get_write_pointer();
1773 datat += data_column->get_start() + begin_row * stride;
1774 LMatrix4f matf = LCAST(float, mat);
1775
1776 if (num_values == 3) {
1777 table_xform_point3f(datat, num_rows, stride, matf);
1778 } else {
1779 table_xform_vecbase4f(datat, num_rows, stride, matf);
1780 }
1781
1782 } else if (num_values == 4) {
1783 // Use the GeomVertexRewriter to adjust the 4-component points.
1784
1785 data.set_row_unsafe(begin_row);
1786 for (int j = begin_row; j < end_row; ++j) {
1787 LPoint4 vertex = data.get_data4();
1788 data.set_data4(vertex * mat);
1789 }
1790
1791 } else {
1792 // Use the GeomVertexRewriter to adjust the 3-component points.
1793
1794 data.set_row_unsafe(begin_row);
1795 for (int j = begin_row; j < end_row; ++j) {
1796 LPoint3 vertex = data.get_data3();
1797 data.set_data3(vertex * mat);
1798 }
1799 }

1805void GeomVertexData::
1806do_transform_vector_column(const GeomVertexFormat *format, GeomVertexRewriter &data,
1807 const LMatrix4 &mat, int begin_row, int end_row) {
1808 const GeomVertexColumn *data_column = data.get_column();
1809 int num_values = data_column->get_num_values();
1810
1811 LMatrix4 xform;
1812 bool normalize = false;
1813 if (data_column->get_contents() == C_normal) {
1814 // This is to preserve perpendicularity to the surface.
1815 LVecBase3 scale_sq(mat.get_row3(0).length_squared(),
1816 mat.get_row3(1).length_squared(),
1817 mat.get_row3(2).length_squared());
1818 if (IS_THRESHOLD_EQUAL(scale_sq[0], scale_sq[1], 2.0e-3f) &&
1819 IS_THRESHOLD_EQUAL(scale_sq[0], scale_sq[2], 2.0e-3f)) {
1820 // There is a uniform scale.
1821 LVecBase3 scale, shear, hpr;
1822 if (IS_THRESHOLD_EQUAL(scale_sq[0], 1, 2.0e-3f)) {
1823 // No scale to worry about.
1824 xform = mat;
1825 } else if (decompose_matrix(mat.get_upper_3(), scale, shear, hpr)) {
1826 // Make a new matrix with scale/translate taken out of the equation.
1827 compose_matrix(xform, LVecBase3(1, 1, 1), shear, hpr, LVecBase3::zero());
1828 } else {
1829 normalize = true;
1830 }
1831 } else {
1832 // There is a non-uniform scale, so we need to do all this to preserve
1833 // orthogonality to the surface.
1834 xform.invert_from(mat);
1835 xform.transpose_in_place();
1836 normalize = true;
1837 }
1838 } else {
1839 xform = mat;
1840 }
1841
1842 if ((num_values == 3 || num_values == 4) &&
1843 data_column->get_numeric_type() == NT_float32) {
1844 // The table of vectors is a table of LVector3f's or LVector4f's.
1845 // Optimize this common case.
1846 GeomVertexArrayDataHandle *data_handle = data.get_array_handle();
1847
1848 size_t stride = data.get_stride();
1849 size_t num_rows = end_row - begin_row;
1850 unsigned char *datat = data_handle->get_write_pointer();
1851 datat += data_column->get_start() + begin_row * stride;
1852 LMatrix4f matf = LCAST(float, xform);
1853
1854 if (normalize) {
1855 table_xform_normal3f(datat, num_rows, stride, matf);
1856 } else if (num_values == 3) {
1857 table_xform_vector3f(datat, num_rows, stride, matf);
1858 } else {
1859 table_xform_vecbase4f(datat, num_rows, stride, matf);
1860 }
1861
1862 } else {
1863 // Use the GeomVertexRewriter to transform the vectors.
1864 data.set_row_unsafe(begin_row);
1865
1866 if (normalize) {
1867 for (int j = begin_row; j < end_row; ++j) {
1868 LVector3 vector = data.get_data3();
1869 vector *= xform;
1870 vector.normalize();
1871 data.set_data3(vector);
1872 }
1873 } else {
1874 for (int j = begin_row; j < end_row; ++j) {
1875 LVector3 vector = data.get_data3();
1876 data.set_data3(vector * xform);
1877 }
1878 }
1879 }
1880}

1933table_xform_vecbase4f(unsigned char *datat, size_t num_rows, size_t stride,

1935#if defined(HAVE_EIGEN) && defined(LINMATH_ALIGN)

1942 for (size_t i = 0; i < num_rows; ++i) {
1943 table.row(i) *= matf._m;
1944 }
1945 return;
1946 }
1947#endif // HAVE_EIGEN

1949 // If the table is properly aligned (or we don't require alignment), we can
1950 // directly use the high-level LVecBase4f object, which will do the right
1951 // thing.
1952 for (size_t i = 0; i < num_rows; ++i) {
1953 LVecBase4f &vertex = *(LVecBase4f *)(&datat[i * stride]);
1954 vertex *= matf;
