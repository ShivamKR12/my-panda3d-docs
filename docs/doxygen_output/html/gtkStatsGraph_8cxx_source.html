342cairo_pattern_t *GtkStatsGraph::
343get_collector_pattern(int collector_index, bool highlight) {
344 Brushes::iterator bi;
345 bi = _brushes.find(collector_index);
346 if (bi != _brushes.end()) {
347 return highlight ? (*bi).second.second : (*bi).second.first;
348 }
349
350 // Ask the monitor what color this guy should be.
351 LRGBColor rgb = _monitor->get_collector_color(collector_index);
352 cairo_pattern_t *pattern = cairo_pattern_create_rgb(
353 encode_sRGB_float((float)rgb[0]),
354 encode_sRGB_float((float)rgb[1]),
355 encode_sRGB_float((float)rgb[2]));
356 cairo_pattern_t *hpattern = cairo_pattern_create_rgb(
357 encode_sRGB_float((float)rgb[0] * 0.75f),
358 encode_sRGB_float((float)rgb[1] * 0.75f),
359 encode_sRGB_float((float)rgb[2] * 0.75f));
360
361 _brushes[collector_index] = std::make_pair(pattern, hpattern);
362 return highlight ? hpattern : pattern;
363}

368LRGBColor GtkStatsGraph::
369get_collector_text_color(int collector_index, bool highlight) {
370 TextColors::iterator tci;
371 tci = _text_colors.find(collector_index);
372 if (tci != _text_colors.end()) {
373 return highlight ? (*tci).second.second : (*tci).second.first;
374 }
375
376 LRGBColor rgb = _monitor->get_collector_color(collector_index);
377 double bright =
378 rgb[0] * 0.2126 +
379 rgb[1] * 0.7152 +
380 rgb[2] * 0.0722;
381 LRGBColor color = bright >= 0.5 ? LRGBColor(0) : LRGBColor(1);
382 LRGBColor hcolor = bright * 0.75 >= 0.5 ? LRGBColor(0) : LRGBColor(1);
383
384 _text_colors[collector_index] = std::make_pair(color, hcolor);
385 return highlight ? hcolor : color;
386}

493void GtkStatsGraph::
494setup_surface(int xsize, int ysize, int scale) {
495 release_surface();
496
497 _surface_xsize = xsize;
498 _surface_ysize = ysize;
499 _pixel_scale = scale * _monitor->get_resolution() * 4 / 96;
500
501 GdkWindow *window = gtk_widget_get_window(_graph_window);
502 _cr_surface = gdk_window_create_similar_image_surface(window, CAIRO_FORMAT_RGB24, _surface_xsize, _surface_ysize, 1);
503 _cr = cairo_create(_cr_surface);
504 _cr_scale = scale;
505
506 cairo_set_source_rgb(_cr, 1.0, 1.0, 1.0);
507 cairo_paint(_cr);
508
509 // Cache the font scale attribute.
510 _pango_attrs = pango_attr_list_new();
511 PangoAttribute *attr = pango_attr_scale_new(scale * 0.9);
512 attr->start_index = 0;
513 attr->end_index = -1;
514 pango_attr_list_insert(_pango_attrs, attr);

682gboolean GtkStatsGraph::
683tick_callback(GtkWidget *widget, GdkFrameClock *clock, gpointer data) {
684 GtkStatsGraph *graph = (GtkStatsGraph *)data;
685 gint64 new_time = gdk_frame_clock_get_frame_time(clock);
686 if (graph->_time == 0) {
687 // First frame, so we don't have a dt yet.
688 graph->_time = new_time;
689 return TRUE;
690 }
691 gint64 delta = new_time - graph->_time;
692 if (delta == 0) {
693 return TRUE;
694 }
695 if (graph->animate(new_time / 1000000.0, delta / 1000000.0)) {
696 graph->_time = new_time;
697 return TRUE;
698 } else {
699 graph->_timer_id = 0;
700 return FALSE;
701 }
