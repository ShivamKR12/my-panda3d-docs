346void MacStatsFlameGraph::
347draw_bar(int depth, int from_x, int to_x, int collector_index, int parent_index) {
348 double bottom = get_ysize() - depth * 4.000 * 5;
349 double top = bottom - 4.000 * 5;
350
351 top += 1;
352
353 MacStatsMonitor *monitor = MacStatsGraph::_monitor;
354
355 bool is_highlighted = collector_index == _highlighted_index;
356 CGContextSetFillColorWithColor(_ctx,
357 monitor->get_collector_color(collector_index, is_highlighted));
358
359 if (to_x < from_x + 3) {
360 // It's just a tiny sliver. This is a more reliable way to draw it.
361 CGRect rect = CGRectMake(from_x, top, to_x - from_x, bottom - top);
362 CGContextFillRect(_ctx, rect);
363 }
364 else {
365 double radius = std::min((double)4.000, (to_x - from_x) / 2.0);
366 CGContextBeginPath(_ctx);
367 CGContextAddArc(_ctx, to_x - radius, top + radius, radius, -0.5 * M_PI, 0.0, NO);
368 CGContextAddArc(_ctx, to_x - radius, bottom - radius, radius, 0.0, 0.5 * M_PI, NO);
369 CGContextAddArc(_ctx, from_x + radius, bottom - radius, radius, 0.5 * M_PI, M_PI, NO);
370 CGContextAddArc(_ctx, from_x + radius, top + radius, radius, M_PI, 1.5 * M_PI, NO);
371 CGContextClosePath(_ctx);
372 CGContextFillPath(_ctx);
373
374 if ((to_x - from_x) >= 4.000 * 4) {
375 // Only bother drawing the text if we've got some space to draw on.
376 int left = std::max(from_x, 0) + 4.000 / 2;
377 int right = std::min(to_x, get_xsize()) - 4.000 / 2;
378
379 const PStatClientData *client_data = monitor->get_client_data();
380 const PStatCollectorDef &def = client_data->get_collector_def(collector_index);
381
382 const CFStringRef keys[] = {
383 (__bridge CFStringRef)NSForegroundColorAttributeName,
384 (__bridge CFStringRef)NSFontAttributeName,
385 };
386 const void *values[] = {
387 monitor->get_collector_text_color(collector_index, is_highlighted),
388 [NSFont systemFontOfSize:0.0],
389 };
390 CFDictionaryRef attribs = CFDictionaryCreate(kCFAllocatorDefault, (const void **)keys, (const void **)values, 2, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
391
392 CFStringRef str = CFStringCreateWithCString(kCFAllocatorDefault, def._name.c_str(), kCFStringEncodingUTF8);
393 CFAttributedStringRef astr = CFAttributedStringCreate(kCFAllocatorDefault, str, attribs);
394
395 CTLineRef line = CTLineCreateWithAttributedString(astr);
396 CGRect bounds = CTLineGetImageBounds(line, _ctx);
397 CFRelease(astr);
398 CFRelease(str);
399
400 if (bounds.size.width < right - left) {
401 // We have room for more. Show the collector's actual parent, if it's
402 // different than the block it's shown above.
403 if (def._parent_index > 0 && def._parent_index != parent_index) {
404 const PStatCollectorDef &parent_def = client_data->get_collector_def(def._parent_index);
405 std::string long_name = parent_def._name + ":" + def._name;
406
407 CFStringRef long_str = CFStringCreateWithCString(kCFAllocatorDefault, long_name.c_str(), kCFStringEncodingUTF8);
408 CFAttributedStringRef long_astr = CFAttributedStringCreate(kCFAllocatorDefault, long_str, attribs);
409
410 CTLineRef long_line = CTLineCreateWithAttributedString((CFAttributedStringRef)long_astr);
411 CGRect long_bounds = CTLineGetImageBounds(long_line, _ctx);
412
413 if (long_bounds.size.width < right - left) {
414 CFRelease(line);
415 line = long_line;
416 bounds = long_bounds;
417 } else {
418 CFRelease(long_line);
419 }
420 CFRelease(long_astr);
421 CFRelease(long_str);
422 }
423 }
424 else {
425 static CFStringRef token_str = CFSTR("\u2026");
426 CFAttributedStringRef token_astr = CFAttributedStringCreate(kCFAllocatorDefault, token_str, attribs);
427 CTLineRef token_line = CTLineCreateWithAttributedString(token_astr);
428 CTLineRef trunc_line = CTLineCreateTruncatedLine(line, right - left, kCTLineTruncationEnd, token_line);
429 CFRelease(line);
430 CFRelease(token_astr);
431 CFRelease(token_line);
432 line = trunc_line;
433 }
434
435 // Center the text vertically in the bar.
436 if (line != nullptr) {
437 CGContextSetTextPosition(_ctx, left, top + (bottom - top + bounds.size.height) / 2);
438 CTLineDraw(line, _ctx);
439 CFRelease(line);
440 }
441 CFRelease(attribs);
442 }
443 }

650void MacStatsFlameGraph::
651handle_motion(int graph_x, int graph_y) {
652 if (_drag_mode == DM_none && _potential_drag_mode == DM_none &&
653 graph_x >= 0 && graph_y >= 0 && graph_x < get_xsize() && graph_y < get_ysize()) {
654 // When the mouse is over a color bar, highlight it.
655 int depth = pixel_to_depth(graph_y);
656 int collector_index = get_bar_collector(depth, graph_x);
657 on_enter_label(collector_index);
658 }
659 else {
660 // If the mouse is in some drag mode, stop highlighting.
661 _label_stack.highlight_label(-1);
662 on_leave_label(_highlighted_index);
663 }
664
665 if (_drag_mode == DM_new_guide_bar) {
666 // We haven't created the new guide bar yet; we won't until the mouse
667 // comes within the graph's region.
668 if (graph_x >= 0 && graph_x < get_xsize()) {
669 set_drag_mode(DM_guide_bar);
670 _drag_guide_bar = add_user_guide_bar(pixel_to_height(graph_x));
671 return;
672 }
673 }
674 else if (_drag_mode == DM_guide_bar) {
675 move_user_guide_bar(_drag_guide_bar, pixel_to_height(graph_x));
676 return;
677 }
678
679 return MacStatsGraph::handle_motion(graph_x, graph_y);
680}
