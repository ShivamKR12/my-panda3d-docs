60int UserDataAudio::
61read_samples(int n, int16_t *data) {
62 int ready = (_data.size() / _desired_channels);
63 int desired = n * _desired_channels;
64 if (n > ready) {
65 n = ready;
66 }
67 int avail = n * _desired_channels;
68 for (int i=0; i<avail; i++) {
69 data[i] = _data[i];
70 }
71 for (int i=avail; i<desired; i++) {
72 data[i] = 0;
73 }
74 for (int i=0; i<avail; i++) {
75 _data.pop_front();
76 }
77 return n;
78}

112void UserDataAudio::
113append(const vector_uchar &str) {
114 nassertv(!_aborted);
115 int samples = str.size() / (2 * _desired_channels);
116 int words = samples * _desired_channels;
117 for (int i = 0; i < words; ++i) {
