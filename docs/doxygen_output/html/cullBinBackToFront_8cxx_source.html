39add_object(CullableObject *object, Thread *current_thread) {
40 // Determine the center of the bounding volume.
41 CPT(BoundingVolume) volume = object->_geom->get_bounds(current_thread);
42 if (volume->is_empty()) {
43 // No point in culling objects with no volume.
44 return;
45 }
46
47 const GeometricBoundingVolume *gbv = volume->as_geometric_bounding_volume();
48 nassertv(gbv != nullptr);
49
50 LPoint3 center = gbv->get_approx_center();
51 nassertv(object->_internal_transform != nullptr);
52 center = center * object->_internal_transform->get_mat();
53
54 PN_stdfloat distance = _gsg->compute_distance_to(center);
55 _objects.push_back(ObjectData(object, distance));
56}
