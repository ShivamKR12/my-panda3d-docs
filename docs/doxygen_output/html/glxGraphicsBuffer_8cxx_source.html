152bool glxGraphicsBuffer::
153open_buffer() {
154 glxGraphicsPipe *glx_pipe;
155 DCAST_INTO_R(glx_pipe, _pipe, false);
156
157 // GSG CreationInitialization
158 glxGraphicsStateGuardian *glxgsg;
159 if (_gsg == nullptr) {
160 // There is no old gsg. Create a new one.
161 glxgsg = new glxGraphicsStateGuardian(_engine, _pipe, nullptr);
162 glxgsg->choose_pixel_format(_fb_properties, glx_pipe->get_display(), glx_pipe->get_screen(), true, false);
163 _gsg = glxgsg;
164 } else {
165 // If the old gsg has the wrong pixel format, create a new one that shares
166 // with the old gsg.
167 DCAST_INTO_R(glxgsg, _gsg, false);
168
169 if (glxgsg->get_engine() != _engine ||
170 !glxgsg->_context_has_pbuffer ||
171 !glxgsg->get_fb_properties().subsumes(_fb_properties)) {
172 // We need a new pixel format, and hence a new GSG.
173 glxgsg = new glxGraphicsStateGuardian(_engine, _pipe, glxgsg);
174 glxgsg->choose_pixel_format(_fb_properties, glx_pipe->get_display(), glx_pipe->get_screen(), true, false);
175 _gsg = glxgsg;
176 }
177 }
178
179 if (glxgsg->_fbconfig == None || !glxgsg->_context_has_pbuffer) {
180 // If we didn't use an fbconfig to create the GSG, or it doesn't support
181 // buffers, we can't create a PBuffer.
182 return false;
183 }
184
185 nassertr(glxgsg->_supports_pbuffer, false);
186
187 LightReMutexHolder holder(glxGraphicsPipe::_x_mutex);
188
189 static const int max_attrib_list = 32;
190 int attrib_list[max_attrib_list];
191 int n = 0;
192
193 if (glxgsg->_uses_sgix_pbuffer) {
194 // The SGI version passed in the size in the parameter list.
195 nassertr(n < max_attrib_list, false);
196 attrib_list[n] = (int)None;
197 _pbuffer = glxgsg->_glXCreateGLXPbufferSGIX(glxgsg->_display, glxgsg->_fbconfig,
198 get_x_size(), get_y_size(), attrib_list);
199 } else {
200 // The official GLX 1.3 version passes in the size in the attrib list.
201 attrib_list[n++] = GLX_PBUFFER_WIDTH;
202 attrib_list[n++] = get_x_size();
203 attrib_list[n++] = GLX_PBUFFER_HEIGHT;
204 attrib_list[n++] = get_y_size();
205
206 nassertr(n < max_attrib_list, false);
207 attrib_list[n] = (int)None;
208 _pbuffer = glxgsg->_glXCreatePbuffer(glxgsg->_display, glxgsg->_fbconfig,
209 attrib_list);
210 }
211
212 if (_pbuffer == None) {
213 glxdisplay_cat.error()
214 << "failed to create GLX pbuffer.\n";
215 return false;
216 }
217
218 glXMakeCurrent(_display, _pbuffer, glxgsg->_context);
219 glxgsg->reset_if_new();
220 if (!glxgsg->is_valid()) {
221 close_buffer();
222 return false;
223 }
224 if (!glxgsg->get_fb_properties().verify_hardware_software
225 (_fb_properties, glxgsg->get_gl_renderer())) {
226 close_buffer();
227 return false;
228 }
229 _fb_properties = glxgsg->get_fb_properties();
230
231 _is_valid = true;
232 return true;
233}
