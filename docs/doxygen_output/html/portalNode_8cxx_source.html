174PandaNode *PortalNode::
175combine_with(PandaNode *other) {
176 if (is_exact_type(get_class_type()) &&
177 other->is_exact_type(get_class_type())) {
178 // Two PortalNodes can combine, but only if they have the same name,
179 // because the name is often meaningful.
180 PortalNode *cother = DCAST(PortalNode, other);
181 if (get_name() == cother->get_name()) {
182 return this;
183 }
184
185 // Two PortalNodes with different names can't combine.
186 return nullptr;
187 }
188
189 return PandaNode::combine_with(other);
190}

201bool PortalNode::
202cull_callback(CullTraverser *trav, CullTraverserData &data) {
203 Thread *current_thread = trav->get_current_thread();
204
205 PortalClipper *portal_viewer = trav->get_portal_clipper();
206 set_visible(false);
207 if (is_open() && !_cell_out.is_empty() && portal_viewer && data._portal_depth <= _max_depth) {
208 portal_cat.debug() << "checking portal node " << *this << endl;
209 portal_cat.debug() << "portal_depth is " << data._portal_depth << endl;
210 PT(GeometricBoundingVolume) vf = trav->get_view_frustum();
211 PT(BoundingVolume) reduced_frustum;
212
213 // remember old viewport and frustum, so we can restore them for the
214 // siblings. (it gets changed by the prepare_portal call)
215 LPoint2 old_reduced_viewport_min, old_reduced_viewport_max;
216 portal_viewer->get_reduced_viewport(old_reduced_viewport_min, old_reduced_viewport_max);
217 PT(BoundingHexahedron) old_bh = portal_viewer->get_reduced_frustum();
218
219 if (portal_viewer->prepare_portal(data.get_node_path())) {
220 if ((reduced_frustum = portal_viewer->get_reduced_frustum())) {
221 // remember current clip state, we might change it
222 CPT(RenderState) old_clip_state = portal_viewer->get_clip_state();
223
224 set_visible(true);
225 // The frustum is in camera space
226 vf = DCAST(GeometricBoundingVolume, reduced_frustum);
227
228 // create a copy of this reduced frustum, we'll transform it from
229 // camera space to the cell_out space
230 PT(BoundingHexahedron) new_bh = DCAST(BoundingHexahedron, vf->make_copy());
231
232 // Get the net trasform of the _cell_out as seen from the camera.
233 CPT(TransformState) cell_transform = _cell_out.get_net_transform();
234 CPT(TransformState) frustum_transform = cell_transform ->invert_compose(portal_viewer->_scene_setup->get_cull_center().get_net_transform());
235
236 // transform to _cell_out space
237 new_bh->xform(frustum_transform->get_mat());
238
239 CPT(RenderState) next_state = data._state;
240
241 // set clipping planes, if desired..
242 if (_clip_plane) {
243 // create a copy of this reduced frustum, we'll transform it from
244 // camera space to this portal node's space (because the clip planes
245 // are attached to this node)
246 PT(BoundingHexahedron) temp_bh = DCAST(BoundingHexahedron, vf->make_copy());
247 CPT(TransformState) temp_frustum_transform = data.get_node_path().get_net_transform()->invert_compose(portal_viewer->_scene_setup->get_cull_center().get_net_transform());
248
249 portal_cat.spam() << "clipping plane frustum transform " << *temp_frustum_transform << endl;
250 portal_cat.spam() << "frustum before transform " << *temp_bh << endl;
251 // transform to portalNode space
252 temp_bh->xform(temp_frustum_transform->get_mat());
253
254 portal_cat.spam() << "frustum after transform " << *temp_bh << endl;
255
256 _left_plane_node->set_plane(-temp_bh->get_plane(4)); // left plane of bh
257 _right_plane_node->set_plane(-temp_bh->get_plane(2));// right plane of bh
258 _top_plane_node->set_plane(-temp_bh->get_plane(3)); // top plane of bh
259 _bottom_plane_node->set_plane(-temp_bh->get_plane(1));// bottom plane of bh
260
261 portal_cat.spam() << "left plane " << *_left_plane_node << endl;
262 portal_cat.spam() << "right plane " << *_right_plane_node << endl;
263 portal_cat.spam() << "top plane " << *_top_plane_node << endl;
264 portal_cat.spam() << "bottom plane " << *_bottom_plane_node << endl;
265
266 // remember the clip state we just generated
267 portal_viewer->set_clip_state(_clip_state);
268
269 if (old_clip_state) {
270 portal_cat.spam() << "parent clip state " << *old_clip_state << endl;
271 } else {
272 portal_cat.spam() << "parent clip state None" << endl;
273 }
274 portal_cat.spam() << "own clip state " << *_clip_state << endl;
275 portal_cat.spam() << "next state " << *next_state << endl;
276
277 // undo parent clip state and compose our new clip state ito the new
278 // state
279 if (old_clip_state != nullptr) {
280 next_state = old_clip_state->invert_compose(next_state);
281 portal_cat.spam() << "next state after removing parent state " << *next_state << endl;
282 }
283 next_state = next_state->compose(_clip_state);
284 portal_cat.spam() << "next state after composition " << *next_state << endl;
285 }
286
287 CullTraverserData next_data(_cell_out,
288 cell_transform,
289 next_state, new_bh,
290 current_thread);
291 next_data._portal_depth = data._portal_depth + 1;
292
293 portal_viewer->set_reduced_frustum(new_bh);
294 portal_cat.spam() << "cull_callback: before traversing " << _cell_out.get_name() << endl;
295 trav->traverse_below(next_data);
296 portal_cat.spam() << "cull_callback: after traversing " << _cell_out.get_name() << endl;
297
298 // restore clip state
299 portal_viewer->set_clip_state(old_clip_state);
300 }
301 }
302 // reset portal viewer frustum for the siblings;
303 portal_viewer->set_reduced_frustum(old_bh);
304 // reset portal viewer viewport for the siblings;
305 portal_viewer->set_reduced_viewport(old_reduced_viewport_min, old_reduced_viewport_max);
306 }
307 // Now carry on to render our child nodes.
308 return true;
309}

339void PortalNode::
340compute_internal_bounds(CPT(BoundingVolume) &internal_bounds,
341 int &internal_vertices,
342 int pipeline_stage,
343 Thread *current_thread) const {
344 // First, get ourselves a fresh, empty bounding volume.
345 PT(BoundingVolume) bound = new BoundingSphere;
346 GeometricBoundingVolume *gbv = DCAST(GeometricBoundingVolume, bound);
347
348 // Now actually compute the bounding volume by putting it around all of our
349 // vertices.
350
351 const LPoint3 *vertices_begin = &_vertices[0];
352 const LPoint3 *vertices_end = vertices_begin + _vertices.size();
353
354 // Now actually compute the bounding volume by putting it around all
355 gbv->around(vertices_begin, vertices_end);
356
357 internal_bounds = bound;
358 internal_vertices = 0;
359}
