125 pieces[0] = "tpose";

224 GLCAT.error() << basename << " should be mat4 or mat3\n";

269 objShader->cp_add_mat_spec(bind);
270 }
271 } else {
272 GLCAT.warning() << basename << "[" << param_size << "] should not be an array, only the first element will be defined\n";
273 }
274 }

276 return true;
277 }

434void CLP(ShaderContext)::
435reflect_attribute(int i, char *name_buffer, GLsizei name_buflen) {
436 GLint param_size;
437 GLenum param_type;
438
439 // Get the name, size, and type of this attribute.
440 name_buffer[0] = 0;
441 _glgsg->_glGetActiveAttrib(_glsl_program, i, name_buflen, nullptr,
442 &param_size, &param_type, name_buffer);
443
444 // Get the attrib location.
445 GLint p = _glgsg->_glGetAttribLocation(_glsl_program, name_buffer);
446
447 if (GLCAT.is_debug()) {
448 GLCAT.debug()
449 << "Active attribute " << name_buffer << " with size " << param_size
450 << " and type 0x" << hex << param_type << dec
451 << " is bound to location " << p << "\n";
452 }
453
454 if (p == -1 || strncmp(name_buffer, "gl_", 3) == 0) {
455 // A gl_ attribute such as gl_Vertex requires us to pass the standard
456 // vertex arrays as we would do without shader. Not all drivers return -1
457 // in glGetAttribLocation for gl_ prefixed attributes, so we check the
458 // prefix of the input ourselves, just to be sure.
459 _uses_standard_vertex_arrays = true;
460 return;
461 }
462
463 Shader::ShaderArgId arg_id;
464 arg_id._name = name_buffer;
465 arg_id._seqno = p;
466
467 Shader::ShaderVarSpec bind;
468 bind._id = arg_id;
469 bind._name = nullptr;
470 bind._append_uv = -1;
471 bind._elements = 1;
472
473 // Check if this is an integer input- if so, we have to bind it differently.
474 switch (param_type) {
475 case GL_INT:
476 case GL_INT_VEC2:
477 case GL_INT_VEC3:
478 case GL_INT_VEC4:
479 bind._numeric_type = Shader::SPT_int;
480 break;
481 case GL_BOOL:
482 case GL_BOOL_VEC2:
483 case GL_BOOL_VEC3:
484 case GL_BOOL_VEC4:
485 bind._numeric_type = Shader::SPT_bool;
486 break;
487 case GL_UNSIGNED_INT:
488 case GL_UNSIGNED_INT_VEC2:
489 case GL_UNSIGNED_INT_VEC3:
490 case GL_UNSIGNED_INT_VEC4:
491 bind._numeric_type = Shader::SPT_uint;
492 break;
493#ifndef OPENGLES
494 case GL_DOUBLE:
495 case GL_DOUBLE_VEC2:
496 case GL_DOUBLE_VEC3:
497 case GL_DOUBLE_VEC4:
498 bind._numeric_type = Shader::SPT_double;
499 break;
500#endif
501 default:
502 bind._numeric_type = Shader::SPT_float;
503 }

510 bind._name = InternalName::get_vertex();
511
512 } else if (noprefix == "Normal") {
513 bind._name = InternalName::get_normal();
514

526
527 } else if (noprefix.substr(0, 8) == "Binormal") {

530 bind._append_uv = atoi(noprefix.substr(8).c_str());
531 }
532
533 } else if (noprefix.substr(0, 13) == "MultiTexCoord") {
534 bind._name = InternalName::get_texcoord();
535 bind._append_uv = atoi(noprefix.substr(13).c_str());
536
537 } else if (noprefix == "InstanceMatrix") {
538 bind._name = InternalName::get_instance_matrix();
539
540 if (param_type != GL_FLOAT_MAT4x3) {
541 GLCAT.error() << "p3d_InstanceMatrix should be mat4x3!\n";
542 }
543
544 } else {
545 GLCAT.error() << "Unrecognized vertex attrib '" << name_buffer << "'!\n";
546 return;
547 }
548 } else {
549 // Arbitrarily named attribute.
550 bind._name = InternalName::make(name_buffer);
551 }
552
553 // Get the number of bind points for arrays and matrices.
554 switch (param_type) {
555 case GL_FLOAT_MAT3x2:
556 case GL_FLOAT_MAT3:
557 case GL_FLOAT_MAT3x4:
558#ifndef OPENGLES
559 case GL_DOUBLE_MAT3x2:
560 case GL_DOUBLE_MAT3:
561 case GL_DOUBLE_MAT3x4:
562#endif
563 bind._elements = 3 * param_size;
564 break;
565
566 case GL_FLOAT_MAT4x2:
567 case GL_FLOAT_MAT4x3:
568 case GL_FLOAT_MAT4:
569#ifndef OPENGLES
570 case GL_DOUBLE_MAT4x2:
571 case GL_DOUBLE_MAT4x3:
572 case GL_DOUBLE_MAT4:
573#endif
574 bind._elements = 4 * param_size;
575 break;
576
577 default:
578 bind._elements = param_size;
579 break;
580 }
581
582 // Experimental feature.
583 if (gl_fixed_vertex_attrib_locations) {
584 GLint loc;
585 if (bind._name == InternalName::get_vertex()) {
586 loc = 0;
587 } else if (bind._name == InternalName::get_transform_weight()) {
588 loc = 1;
589 } else if (bind._name == InternalName::get_normal()) {
590 loc = 2;
591 } else if (bind._name == InternalName::get_color()) {
592 loc = 3;
593 } else if (bind._name == InternalName::get_transform_index()) {
594 loc = 7;
595 } else if (bind._name == InternalName::get_texcoord() &&
596 bind._append_uv >= 0 && bind._append_uv < 8) {

598 } else {
599 GLCAT.error()
600 << "Vertex attrib '" << name_buffer << "' not yet supported with "
601 "gl-fixed-vertex-attrib-locations!\n";
602 return;
603 }
604 if (loc != p) {
605 GLCAT.error()
606 << "Vertex attrib '" << name_buffer << "' was bound to the wrong slot!\n";
607 return;
608 }
609 // _glgsg->_glBindAttribLocation(_glsl_program, loc, name_buffer);
610 _enabled_attribs.set_range(loc, bind._elements);
611 }
612
613 _shader->_var_spec.push_back(bind);
614}

619void CLP(ShaderContext)::
620reflect_uniform_block(int i, const char *name, char *name_buffer, GLsizei name_buflen) {
621 // GLint offset = 0;
622
623 GLint data_size = 0;
624 GLint param_count = 0;
625 _glgsg->_glGetActiveUniformBlockiv(_glsl_program, i, GL_UNIFORM_BLOCK_DATA_SIZE, &data_size);
626 _glgsg->_glGetActiveUniformBlockiv(_glsl_program, i, GL_UNIFORM_BLOCK_ACTIVE_UNIFORMS, &param_count);
627
628 if (param_count <= 0) {
629 return;
630 }
631
632 // We use a GeomVertexArrayFormat to describe the uniform buffer layout.
633 // GeomVertexArrayFormat block_format; block_format.set_pad_to(data_size);
634
635 // Get an array containing the indices of all the uniforms in this block.
636 GLuint *indices = (GLuint *)alloca(param_count * sizeof(GLint));
637 _glgsg->_glGetActiveUniformBlockiv(_glsl_program, i, GL_UNIFORM_BLOCK_ACTIVE_UNIFORM_INDICES, (GLint *)indices);
638
639 // Acquire information about the way the uniforms in this block are packed.
640 GLint *offsets = (GLint *)alloca(param_count * sizeof(GLint));
641 GLint *mstrides = (GLint *)alloca(param_count * sizeof(GLint));
642 GLint *astrides = (GLint *)alloca(param_count * sizeof(GLint));
643 _glgsg->_glGetActiveUniformsiv(_glsl_program, param_count, indices, GL_UNIFORM_OFFSET, offsets);
644 _glgsg->_glGetActiveUniformsiv(_glsl_program, param_count, indices, GL_UNIFORM_MATRIX_STRIDE, mstrides);
645 _glgsg->_glGetActiveUniformsiv(_glsl_program, param_count, indices, GL_UNIFORM_ARRAY_STRIDE, astrides);
646
647 for (int ui = 0; ui < param_count; ++ui) {
648 name_buffer[0] = 0;
649 GLint param_size;
650 GLenum param_type;
651 _glgsg->_glGetActiveUniform(_glsl_program, indices[ui], name_buflen, nullptr, &param_size, &param_type, name_buffer);
652
653 // Strip off [0] suffix that some drivers append to arrays.
654 size_t size = strlen(name_buffer);
655 if (size > 3 && strncmp(name_buffer + (size - 3), "[0]", 3) == 0) {
656 name_buffer[size - 3] = 0;
657 }
658
659 GeomEnums::NumericType numeric_type;
660 GeomEnums::Contents contents = GeomEnums::C_other;
661 int num_components = 1;
662
663 switch (param_type) {
664 case GL_INT:
665 case GL_INT_VEC2:
666 case GL_INT_VEC3:
667 case GL_INT_VEC4:
668 numeric_type = GeomEnums::NT_int32;
669 break;
670
671 case GL_BOOL:
672 case GL_BOOL_VEC2:
673 case GL_BOOL_VEC3:

802 // uniforms in the list starting with "_main_" (for example,
803 // "_main_0_gp5fp[0]") we need to skip those, because we don't know anything
804 // about them

806 if (GLCAT.is_debug()) {
807 GLCAT.debug() << "Ignoring uniform " << name_buffer << " which may be generated by buggy Nvidia driver.\n";
808 }

812 if (p < 0) {
813 // Special meaning, or it's in a uniform block. Let it go.
814 return;

817 // Strip off [0] suffix that some drivers append to arrays.
818 size_t size = strlen(name_buffer);

820 size -= 3;
821 name_buffer[size] = 0;

841 if (size > 15 && matrix_name.compare(size - 9, 9, "Transpose") == 0) {
842 transpose = true;

847 inverse = true;
848 matrix_name = matrix_name.substr(0, size - 7);
849 }
