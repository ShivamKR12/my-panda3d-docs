62LVector3 LinearSourceForce::
63get_child_vector(const PhysicsObject *po) {
64 LVector3 distance_vector = po->get_position() - get_force_center();
65 PN_stdfloat distance_squared = distance_vector.length_squared();
66
67 if (distance_squared == 0) {
68 return distance_vector;
69 }
70
71 PN_stdfloat scalar = get_scalar_term();
72
73 switch (get_falloff_type()) {
74 case FT_ONE_OVER_R_OVER_DISTANCE:
75 return (distance_vector / sqrt(distance_squared)) * scalar;
76 case FT_ONE_OVER_R_OVER_DISTANCE_SQUARED:
77 return (distance_vector / distance_squared) * scalar;
78 case FT_ONE_OVER_R_OVER_DISTANCE_CUBED:
79 return (distance_vector / (distance_squared * sqrt(distance_squared))) * scalar;
80 default:
81 return distance_vector * scalar;
82 }
83}
