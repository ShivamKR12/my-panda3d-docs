96void CullTraverserData::
97apply_transform(const TransformState *node_transform) {
98 if (!node_transform->is_identity()) {
99 if (_instances != nullptr) {
100 InstanceList *instances = new InstanceList(*_instances);
101 for (InstanceList::Instance &instance : *instances) {
102 instance.set_transform(instance.get_transform()->compose(node_transform));
103 }
104 _instances = std::move(instances);
105 return;
106 }
107
108 _net_transform = _net_transform->compose(node_transform);
109
110 if (_view_frustum != nullptr || _cull_planes != nullptr) {
111 // We need to move the viewing frustums into the node's coordinate space
112 // by applying the node's inverse transform.
113 const LMatrix4 *inverse_mat = node_transform->get_inverse_mat();
114 if (inverse_mat != nullptr) {
115 // Copy the bounding volumes for the frustums so we can transform
116 // them.
117 if (_view_frustum != nullptr) {
118 _view_frustum = _view_frustum->make_copy()->as_geometric_bounding_volume();
119 nassertv(_view_frustum != nullptr);
120
121 _view_frustum->xform(*inverse_mat);
122 }
123
124 if (_cull_planes != nullptr) {
125 _cull_planes = _cull_planes->xform(*inverse_mat);
126 }
127 }
128 else {
129 // But we can't invert a singular transform! Instead of trying, we'll
130 // just give up on frustum culling from this point down.
131 pgraph_cat.warning()
132 << "Singular transformation detected on node: " << get_node_path() << "\n";
133 _view_frustum = nullptr;
134 _cull_planes = nullptr;
135 }
136 }
137 }

144bool CullTraverserData::
145is_instance_in_view(const TransformState *instance_transform, const DrawMask &camera_mask) const {
146 PT(GeometricBoundingVolume) view_frustum_p;
147 const GeometricBoundingVolume *view_frustum = nullptr;
148
149 if (_view_frustum != nullptr) {
150 if (!instance_transform->is_identity()) {
151 // We need to move the viewing frustums into the node's coordinate space
152 // by applying the node's inverse transform.
153 const LMatrix4 *inverse_mat = instance_transform->get_inverse_mat();
154 if (inverse_mat != nullptr) {
155 // Copy the bounding volumes for the frustums so we can transform them.
156 view_frustum_p = _view_frustum->make_copy()->as_geometric_bounding_volume();
157 nassertr(view_frustum_p != nullptr, false);
158
159 view_frustum_p->xform(*inverse_mat);
160 view_frustum = view_frustum_p;
161 } else {
162 // Don't render instances with a singular transformation.
163 return false;
164 }
165 } else {
166 view_frustum = _view_frustum;
167 }
168 }
169
170 PandaNode::Children children = _node_reader.get_children();
171 int num_children = children.get_num_children();
172 for (int i = 0; i < num_children; ++i) {
173 const PandaNode::DownConnection &child = children.get_child_connection(i);
174
175 if (!child.compare_draw_mask(_draw_mask, camera_mask)) {
176 // If there are no draw bits in common with the camera, the node is out.
177 continue;
178 }
179
180 if (view_frustum == nullptr) {
181 return true;
182 }
183
184 const GeometricBoundingVolume *node_gbv = child.get_bounds();
185 nassertd(node_gbv != nullptr) continue;
186
187 if (view_frustum->contains(node_gbv)) {
188 return true;
189 }
190 }
191 return false;

232bool CullTraverserData::
233apply_cull_planes(const CullPlanes *planes, const GeometricBoundingVolume *node_gbv) {
234 if (!planes->is_empty()) {
235 // Also cull against the current clip planes.
236 int result;
237 CPT(CullPlanes) new_planes = planes->do_cull(result, _state, node_gbv);
238
239 if (pgraph_cat.is_spam()) {
240 pgraph_cat.spam()
241 << get_node_path() << " cull planes cull result = " << std::hex
242 << result << std::dec << "\n";
243 new_planes->write(pgraph_cat.spam(false));
244 }
245
246 if (result == BoundingVolume::IF_no_intersection) {
247 // No intersection at all. Cull.
248 return false;
249 }
250 else if ((result & BoundingVolume::IF_all) != 0) {
251 // The node and its descendants are completely in front of all of the
252 // clip planes and occluders. The do_cull() call should therefore have
253 // removed all of the clip planes and occluders.
254 nassertr(new_planes->is_empty(), true);
255 }
256 else if (!_node_reader.is_final() && !new_planes->is_empty()) {
257 _cull_planes = std::move(new_planes);
258 }
259 }
260
261 return true;
