108bool MovingPartBase::
109do_update(PartBundle *root, const CycleData *root_cdata, PartGroup *parent,
110 bool parent_changed, bool anim_changed,
111 Thread *current_thread) {
112 bool any_changed = false;
113 bool needs_update = anim_changed;
114
115 // See if any of the channel values have changed since last time.
116
117 if (!needs_update) {
118 if (_forced_channel != nullptr) {
119 needs_update = _forced_channel->has_changed(0, 0.0, 0, 0.0);
120
121 } else if (_effective_control != nullptr) {
122 const PartBundle::CData *cdata = (const PartBundle::CData *)root_cdata;
123 needs_update = _effective_control->channel_has_changed(_effective_channel, cdata->_frame_blend_flag);
124
125 } else {
126 const PartBundle::CData *cdata = (const PartBundle::CData *)root_cdata;
127 PartBundle::ChannelBlend::const_iterator bci;
128 for (bci = cdata->_blend.begin();
129 !needs_update && bci != cdata->_blend.end();
130 ++bci) {
131 AnimControl *control = (*bci).first;
132
133 AnimChannelBase *channel = nullptr;
134 int channel_index = control->get_channel_index();
135 if (channel_index >= 0 && channel_index < (int)_channels.size()) {
136 channel = _channels[channel_index];
137 }
138 if (channel != nullptr) {
139 needs_update = control->channel_has_changed(channel, cdata->_frame_blend_flag);
140 }
141 }
142 }
143 }
144
145 if (needs_update) {
146 // Ok, get the latest value.
147 get_blend_value(root);
148 }
149
150 if (parent_changed || needs_update) {
151 any_changed = update_internals(root, parent, needs_update, parent_changed,
152 current_thread);
153 }
154
155 // Now recurse.
156 Children::iterator ci;
157 for (ci = _children.begin(); ci != _children.end(); ++ci) {
158 if ((*ci)->do_update(root, root_cdata, this,
159 parent_changed || needs_update,
160 anim_changed, current_thread)) {
161 any_changed = true;
162 }
163 }
164
165 return any_changed;
166}

187void MovingPartBase::
188pick_channel_index(plist<int> &holes, int &next) const {
189 // Verify each of the holes.
190
191 plist<int>::iterator ii, ii_next;
192 ii = holes.begin();
193 while (ii != holes.end()) {
194 ii_next = ii;
195 ++ii_next;
196
197 int hole = (*ii);
198 nassertv(hole >= 0 && hole < next);
199 if (hole < (int)_channels.size() ||
200 _channels[hole] != nullptr) {
201 // We can't accept this hole; we're using it!
202 holes.erase(ii);
203 }
204 ii = ii_next;
205 }
206
207 // Now do we have any more to restrict?
208 if (next < (int)_channels.size()) {
209 int i;
210 for (i = next; i < (int)_channels.size(); i++) {
211 if (_channels[i] == nullptr) {
212 // Here's a hole we do have.
213 holes.push_back(i);
214 }
215 }
216 next = _channels.size();
217 }
218
219 PartGroup::pick_channel_index(holes, next);
220}

228void MovingPartBase::
229bind_hierarchy(AnimGroup *anim, int channel_index, int &joint_index,
230 bool is_included, BitArray &bound_joints,
231 const PartSubset &subset) {
232 if (subset.matches_include(get_name())) {
233 is_included = true;
234 } else if (subset.matches_exclude(get_name())) {
235 is_included = false;
236 }
237
238 if (chan_cat.is_debug()) {
239 if (anim == nullptr) {
240 chan_cat.debug()
241 << "binding " << *this << " to NULL, is_included = "
242 << is_included << "\n";
243 } else {
244 chan_cat.debug()
245 << "binding " << *this << " to " << *anim << ", is_included = "
246 << is_included << "\n";
247 }
248 }
249 while ((int)_channels.size() <= channel_index) {
250 _channels.push_back(nullptr);
251 }
252
253 nassertv(_channels[channel_index] == nullptr);
254
255 if (is_included) {
256 if (anim == nullptr) {
257 // If we're binding to the NULL anim, it means actually to create a
258 // default AnimChannel that just returns the part's initial value.
259 _channels[channel_index] = make_default_channel();
260 } else {
261 _channels[channel_index] = DCAST(AnimChannelBase, anim);
262 }
263
264 // Record that we have bound this joint in the bound_joints BitArray.
265 bound_joints.set_bit(joint_index);
266 } else {
267 // Record that we have *not* bound this particular joint.
268 bound_joints.clear_bit(joint_index);
269 }
270 ++joint_index;
271
272 PartGroup::bind_hierarchy(anim, channel_index, joint_index,
273 is_included, bound_joints, subset);
274}

302void MovingPartBase::
303determine_effective_channels(const CycleData *root_cdata) {
304 _effective_control = nullptr;
305 _effective_channel = nullptr;
306
307 AnimControl *effective_control = nullptr;
308 AnimChannelBase *effective_channel = nullptr;
309 int num_effective_channels = 0;
310
311 const PartBundle::CData *cdata = (const PartBundle::CData *)root_cdata;
312 PartBundle::ChannelBlend::const_iterator cbi;
313 for (cbi = cdata->_blend.begin();
314 cbi != cdata->_blend.end();
315 ++cbi) {
316 AnimControl *control = (*cbi).first;
317 int channel_index = control->get_channel_index();
318 if (channel_index >= 0 && channel_index < (int)_channels.size()) {
319 if (_channels[channel_index] != nullptr) {
320 effective_control = control;
321 effective_channel = _channels[channel_index];
322 ++num_effective_channels;
323 }
324 }
325 }
326
327 if (num_effective_channels == 1) {
328 _effective_control = effective_control;
329 _effective_channel = effective_channel;
330 }
331
332 PartGroup::determine_effective_channels(root_cdata);
333}
