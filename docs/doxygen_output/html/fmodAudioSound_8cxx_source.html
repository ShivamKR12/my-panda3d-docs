40FmodAudioSound::
41FmodAudioSound(AudioManager *manager, VirtualFile *file, bool positional) : AudioSound(positional) {
42 ReMutexHolder holder(FmodAudioManager::_lock);
43 audio_debug("FmodAudioSound::FmodAudioSound() Creating new sound, filename: "
44 << file->get_original_filename());
45
46 _active = manager->get_active();
47 _paused = false;
48 _start_time = 0.0;
49
50 // Local Variables that are needed.
51 FMOD_RESULT result;
52
53 // Inits 3D Attributes
54 _location.x = 0;
55 _location.y = 0;
56 _location.z = 0;
57
58 _velocity.x = 0;
59 _velocity.y = 0;
60 _velocity.z = 0;
61
62 _min_dist = 1.0;
63 _max_dist = 1000000000.0;
64
65 // Play Rate Variable
66 _playrate = 1;
67
68 // These set the Speaker Levels to a default if you are using a MultiChannel
69 // Setup.
70 for (int i=0; i<AudioManager::SPK_COUNT; i++) {
71 _mix[i] = 1.0;
72 }
73
74 // Assign the values we need
75 FmodAudioManager *fmanager;
76 DCAST_INTO_V(fmanager, manager);
77 _manager = fmanager;
78
79 _channel = 0;
80 _file_name = file->get_original_filename();
81 _file_name.set_binary();
82
83 // Get the Speaker Mode [Important for later on.]
84 result = _manager->_system->getSpeakerMode( &_speakermode );
85 fmod_audio_errcheck("_system->getSpeakerMode()", result);
86
87 {
88 bool preload = (fmod_audio_preload_threshold < 0) || (file->get_file_size() < fmod_audio_preload_threshold);
89 int flags = FMOD_SOFTWARE;
90 flags |= positional ? FMOD_3D : FMOD_2D;
91
92 FMOD_CREATESOUNDEXINFO sound_info;
93 memset(&sound_info, 0, sizeof(sound_info));
94 sound_info.cbsize = sizeof(sound_info);
95
96 string ext = downcase(_file_name.get_extension());
97 if (ext == "mid") {
98 // Get the MIDI parameters.
99 memcpy(&sound_info, &_manager->_midi_info, sizeof(sound_info));
100 if (sound_info.dlsname != nullptr) {
101 audio_debug("Using DLS file " << sound_info.dlsname);
102 }
103 }
104
105 const char *name_or_data = _file_name.c_str();
106 string os_filename;
107
108 vector_uchar mem_buffer;
109 SubfileInfo info;
110 if (preload) {
111 // Pre-read the file right now, and pass it in as a memory buffer. This
112 // avoids threading issues completely, because all of the reading
113 // happens right here.
114 file->read_file(mem_buffer, true);
115 sound_info.length = mem_buffer.size();
116 if (mem_buffer.size() != 0) {
117 name_or_data = (const char *)&mem_buffer[0];
118 }
119 flags |= FMOD_OPENMEMORY;
120 if (fmodAudio_cat.is_debug()) {
121 fmodAudio_cat.debug()
122 << "Reading " << _file_name << " into memory (" << sound_info.length
123 << " bytes)\n";
124 }
125 result =
126 _manager->_system->createSound(name_or_data, flags, &sound_info, &_sound);
127 }
128 else {
129 result = FMOD_ERR_FILE_BAD;
130
131 if (file->get_system_info(info)) {
132 // The file exists on disk (or it's part of a multifile that exists on
133 // disk), so we can have FMod read the file directly. This is also
134 // safe, because FMod uses its own IO operations that don't involve
135 // Panda, so this can safely happen in an FMod thread.
136 os_filename = info.get_filename().to_os_specific();
137 name_or_data = os_filename.c_str();
138 sound_info.fileoffset = (unsigned int)info.get_start();
139 sound_info.length = (unsigned int)info.get_size();
140 flags |= FMOD_CREATESTREAM;
141 if (fmodAudio_cat.is_debug()) {
142 fmodAudio_cat.debug()
143 << "Streaming " << _file_name << " from disk (" << name_or_data
144 << ", " << sound_info.fileoffset << ", " << sound_info.length << ")\n";
145 }
146
147 result =
148 _manager->_system->createSound(name_or_data, flags, &sound_info, &_sound);
149 }
150
151 // If FMOD can't directly read the file (eg. if Panda is locking it for
152 // write, or it's compressed) we have to use the callback interface.
153 if (result == FMOD_ERR_FILE_BAD) {
154 #if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
155 // Otherwise, if the Panda threading system is compiled in, we can
156 // assign callbacks to read the file through the VFS.
157 name_or_data = (const char *)file;
158 sound_info.fileoffset = 0;
159 sound_info.length = (unsigned int)info.get_size();
160 sound_info.useropen = open_callback;
161 sound_info.userclose = close_callback;
162 sound_info.userread = read_callback;
163 sound_info.userseek = seek_callback;
164 flags |= FMOD_CREATESTREAM;
165 if (fmodAudio_cat.is_debug()) {
166 fmodAudio_cat.debug()
167 << "Streaming " << _file_name << " from disk using callbacks\n";
168 }
169 result =
170 _manager->_system->createSound(name_or_data, flags, &sound_info, &_sound);
171
172 #else // HAVE_THREADS && !SIMPLE_THREADS
173 // Without threads, we can't safely read this file.
174 name_or_data = "";
175
176 fmodAudio_cat.warning()
177 << "Cannot stream " << _file_name << "; file is not literally on disk.\n";
178 #endif
179 }
180 }
181 }
182
183 if (result != FMOD_OK) {
184 audio_error("createSound(" << _file_name << "): " << FMOD_ErrorString(result));
185
186 // We couldn't load the sound file. Create a blank sound record instead.
187 FMOD_CREATESOUNDEXINFO sound_info;
188 memset(&sound_info, 0, sizeof(sound_info));
189 char blank_data[100];
190 memset(blank_data, 0, sizeof(blank_data));
191 sound_info.cbsize = sizeof(sound_info);
192 sound_info.length = sizeof(blank_data);
193 sound_info.numchannels = 1;
194 sound_info.defaultfrequency = 8000;
195 sound_info.format = FMOD_SOUND_FORMAT_PCM16;
196 int flags = FMOD_SOFTWARE | FMOD_OPENMEMORY | FMOD_OPENRAW;
197
198 result = _manager->_system->createSound( blank_data, flags, &sound_info, &_sound);
199 fmod_audio_errcheck("createSound (blank)", result);
200 }
201
202 // Some WAV files contain a loop bit. This is not handled consistently.
203 // Override it.
204 _sound->setLoopCount(1);
205 _sound->setMode(FMOD_LOOP_OFF);
206
207 // This is just to collect the defaults of the sound, so we don't Have to
208 // query FMOD everytime for the info. It is also important we get the
209 // '_sampleFrequency' variable here, for the 'set_play_rate()' and
210 // 'get_play_rate()' methods later;
211
212 result = _sound->getDefaults( &_sampleFrequency, &_volume , &_balance, &_priority);
213 fmod_audio_errcheck("_sound->getDefaults()", result);
214}

442void FmodAudioSound::
443start_playing() {
444 ReMutexHolder holder(FmodAudioManager::_lock);
445 FMOD_RESULT result;
446
447 if (!_active) {
448 _paused = true;
449 return;
450 }
451
452 int startTime = (int)(_start_time * 1000);
453
454 if (_channel != 0) {
455 // try backing up current sound.
456 result = _channel->setPosition( startTime , FMOD_TIMEUNIT_MS );
457 if (result == FMOD_ERR_INVALID_HANDLE || result == FMOD_ERR_CHANNEL_STOLEN) {
458 _channel = 0;
459
460 } else {
461 fmod_audio_errcheck("_channel->setPosition()", result);
462
463 bool playing;
464 result = _channel->isPlaying(&playing);
465 fmod_audio_errcheck("_channel->isPlaying()", result);
466 if (result != FMOD_OK || !playing) {
467 _channel = 0;
468 }
469 }
470 }
471
472 if (_channel == 0) {
473 result = _manager->_system->playSound(FMOD_CHANNEL_FREE, _sound, true, &_channel);
474 fmod_audio_errcheck("_system->playSound()", result);
475 result = _channel->setChannelGroup(_manager->_channelgroup);
476 fmod_audio_errcheck("_channel->setChannelGroup()", result);
477 result = _channel->setUserData(this);
478 fmod_audio_errcheck("_channel->setUserData()", result);
479 result = _channel->setCallback(sound_end_callback);
480 fmod_audio_errcheck("_channel->setCallback()", result);
481 result = _channel->setPosition( startTime , FMOD_TIMEUNIT_MS );
482 fmod_audio_errcheck("_channel->setPosition()", result);
483
484 set_volume_on_channel();
485 set_play_rate_on_channel();
486 set_speaker_mix_or_balance_on_channel();
487 // add_dsp_on_channel();
488 set_3d_attributes_on_channel();
489
490 result = _channel->setPaused(false);
491 fmod_audio_errcheck("_channel->setPaused()", result);
492
493 _self_ref = this;
494 }
495}

857void FmodAudioSound::
858set_active(bool active) {
859 ReMutexHolder holder(FmodAudioManager::_lock);
860 if (_active != active) {
861 _active = active;
862 if (_active) {
863 // ...activate the sound.
864 if (_paused && get_loop_count()==0) {
865 // ...this sound was looping when it was paused.
866 _paused = false;
867 play();
868 }
869
870 } else {
871 // ...deactivate the sound.
872 if (status() == PLAYING) {
873 PN_stdfloat time = get_time();
874 stop();
875 if (get_loop_count() == 0) {
876 // ...we're pausing a looping sound.
877 _paused = true;
878 _start_time = time;
879 }
880 }
881 }
