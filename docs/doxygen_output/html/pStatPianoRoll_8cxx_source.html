45void PStatPianoRoll::BarBuilder::
46add_data_point(double time, bool is_start) {
47 if (is_start) {
48 // This is a "start" data point: start the bar.
49 if (_color_bars.empty() || _color_bars.back()._end >= 0.0) {
50 ColorBar bar;
51 bar._start = time;
52 bar._end = -1.0;
53 _color_bars.push_back(bar);
54 }
55
56 } else {
57 // This is a "stop" data point: end the bar.
58 if (_color_bars.empty()) {
59 // A "stop" in the middle of the frame implies a "start" at time 0.
60 ColorBar bar;
61 bar._start = 0.0;
62 bar._end = time;
63 _color_bars.push_back(bar);
64
65 } else {
66 _color_bars.back()._end = time;
67 }
68 }

123void PStatPianoRoll::
124update() {
125 const PStatClientData *client_data = _monitor->get_client_data();
126
127 // Don't bother to update the thread data until we know at least something
128 // about the collectors and threads.
129 if (client_data->get_num_collectors() != 0 &&
130 client_data->get_num_threads() != 0) {
131 const PStatThreadData *thread_data =
132 client_data->get_thread_data(_thread_index);
133 if (!thread_data->is_empty()) {
134 int frame_number = thread_data->get_latest_frame_number();
135 if (frame_number != _current_frame) {

303void PStatPianoRoll::
304compute_page(const PStatFrameData &frame_data) {
305 _start_time = frame_data.get_start();
306
307 // Clear out the page data and copy it to previous, so we can fill it up
308 // again and then check to see if we changed the set of bars this frame.
309 PageData previous;
310 _page_data.swap(previous);
311
312 int num_events = frame_data.get_num_events();
313 for (int i = 0; i < num_events; i++) {
314 int collector_index = frame_data.get_time_collector(i);
315 double time = frame_data.get_time(i);
316 bool is_start = frame_data.is_start(i);
317 _page_data[collector_index].add_data_point(time, is_start);
318 }
319
320 // Now check to see if the set of bars has changed.
321 bool changed_bars = (_page_data.size() != previous.size());
322
323 if (!changed_bars) {
324 PageData::const_iterator ai, bi;
325 ai = _page_data.begin();
326 bi = previous.begin();
327 while (ai != _page_data.end() && !changed_bars) {
328 changed_bars = ((*ai).first == (*bi).first);
329 ++ai;
330 ++bi;
331 }
332 }
333
334 if (changed_bars) {
335 // If we added or removed some new bars this time, we'll have to update
336 // our list.
337 const PStatClientData *client_data = _monitor->get_client_data();
338
339 _labels.clear();
340 PageData::const_iterator pi;
341 for (pi = _page_data.begin(); pi != _page_data.end(); ++pi) {
342 int collector_index = (*pi).first;
343 if (client_data->has_collector(collector_index)) {
344 _labels.push_back(collector_index);
345 }
346 }
347
348 SortCollectorLabels1 sort_labels(client_data);
349 sort(_labels.begin(), _labels.end(), sort_labels);
350
351 _labels_changed = true;
352 }
353
354 // Finally, make sure all of the bars are closed.
355 double time = frame_data.get_end();
356 PageData::iterator pi;
357 for (pi = _page_data.begin(); pi != _page_data.end(); ++pi) {
358 (*pi).second.finish(time);
359 }
