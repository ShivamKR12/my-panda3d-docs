438int FrameBufferProperties::
439get_quality(const FrameBufferProperties &reqs) const {
440
441 if (!get_indexed_color() && !get_rgb_color()) {
442 // Nonfunctioning window.
443 return 0;
444 }
445
446 if ((reqs.get_rgb_color() && !get_rgb_color()) ||
447 (reqs.get_indexed_color() && !get_indexed_color())) {
448 // These properties are nonnegotiable.
449 return 0;
450 }
451
452 int quality = 100000000;
453
454 // Deduct for using the wrong kind of renderer (hardware or software). Cost:
455 // 10,000,000
456
457 if ((reqs._flags & FBF_force_hardware) > (_flags & FBF_force_hardware) ||
458 (reqs._flags & FBF_force_software) > (_flags & FBF_force_software)) {
459 quality -= 10000000;
460 }
461
462 // Deduct for software-only renderers in absence of a special request.
463 // Cost: 2,000,000
464
465 if (get_force_software() && !reqs.get_force_software()) {
466 quality -= 2000000;
467 }
468
469 // Deduct for missing depth, color, alpha, stencil, or accum. Cost:
470 // 1,000,000
471
472 for (int prop = FBP_depth_bits; prop <= FBP_accum_bits; ++prop) {
473 if (reqs._property[prop] && _property[prop] == 0) {
474 quality -= 1000000;
475 }
476 }
477
478 // Deduct for missing aux bitplanes. Cost: 100,000
479
480 for (int prop = FBP_aux_rgba; prop <= FBP_aux_float; ++prop) {
481 if (reqs._property[prop] > _property[prop]) {
482 quality -= 100000;
483 }
484 }
485
486 // Deduct for stereo not enabled. Cost: 100,000
487
488 if (reqs.get_stereo() && !get_stereo()) {
489 quality -= 100000;
490 }
491
492 // Deduct for not being sRGB-capable. Cost: 100,000
493
494 if (reqs.get_srgb_color() && !get_srgb_color()) {
495 quality -= 100000;
496 }
497
498 // Deduct for not having a floating-point format if we requested it. Cost:
499 // 100,000
500
501 if (reqs.get_float_color() && !get_float_color()) {
502 quality -= 100000;
503 }
504
505 if (reqs.get_float_depth() && !get_float_depth()) {
506 quality -= 100000;
507 }
508
509 // Deduct for insufficient back-buffers. Cost: 100,000
510
511 if (reqs._property[FBP_back_buffers] > _property[FBP_back_buffers]) {
512 quality -= 100000;
513 }
514
515 // Deduct for lacking multisamples altogether. Cost: 100,000
516 if (reqs._property[FBP_multisamples] != 0 && _property[FBP_multisamples] == 0) {
517 quality -= 100000;
518 }
519
520 // Deduct for not enough bits in depth, color, alpha, stencil, or accum.
521 // Cost: 10,000
522
523 for (int prop = FBP_depth_bits; prop <= FBP_accum_bits; ++prop) {
524 if (_property[prop] != 0 && reqs._property[prop] > _property[prop]) {
525 quality -= 10000;
526 }
527 }
528
529 // Deduct for insufficient multisamples. Cost: 1,000
530
531 if (_property[FBP_multisamples] != 0 &&
532 reqs._property[FBP_multisamples] > _property[FBP_multisamples]) {
533 quality -= 1000;
534 }
535
536 // Deduct for having more multisamples than requested. Cost: 2 per sample
537 // In case the special value of 1 multisample is requested, nothing gets deducted.
538
539 if (_property[FBP_multisamples] != 0 &&
540 reqs._property[FBP_multisamples] != 1 &&
541 reqs._property[FBP_multisamples] < _property[FBP_multisamples]) {
542 quality -= 2 * _property[FBP_multisamples];
543 }
544
545 // Deduct for insufficient coverage samples. Cost: 1,000
546
547 if (_property[FBP_coverage_samples] != 0 &&
548 reqs._property[FBP_coverage_samples] > _property[FBP_coverage_samples]) {
549 quality -= 1000;
550 }
551
552 // Deduct for having more coverage samples than requested. Cost: 2 per sample
553 // In case the special value of 1 sample is requested, nothing gets deducted.
554
555 if (_property[FBP_coverage_samples] != 0 &&
556 reqs._property[FBP_coverage_samples] != 1 &&
557 reqs._property[FBP_coverage_samples] < _property[FBP_coverage_samples]) {
558 quality -= 2 * _property[FBP_coverage_samples];
559 }
560
561 // Deduct for unrequested bitplanes. Cost: 50
562
563 for (int prop = FBP_depth_bits; prop <= FBP_accum_bits; ++prop) {
564 if ((_property[prop]) && (reqs._property[prop] == 0)) {
565 quality -= 50;
566 }
567 }
568 for (int prop = FBP_aux_rgba; prop <= FBP_aux_float; ++prop) {
569 int extra = _property[prop] > reqs._property[prop];
570 if (extra > 0) {
571 extra = std::min(extra, 3);
572 quality -= extra*50;
573 }
574 }
575
576 // Deduct for excessive resolution in any bitplane (unless we asked for only
577 // 1 bit, which is the convention for any amount).
578
579 // Cost: 50
580
581 for (int prop = FBP_depth_bits; prop <= FBP_accum_bits; ++prop) {
582 if (reqs._property[prop] > 1 &&
583 _property[prop] > reqs._property[prop]) {
584 quality -= 50;
585 }
586 }
587
588 // However, deduct for color bits above 24, if we are requesting only 1.
589 // This is to prevent choosing a 64-bit color mode in NVIDIA cards that
590 // is linear and therefore causes the gamma to be off in non-sRGB pipelines.
591 if (reqs._property[FBP_color_bits] <= 3 && _property[FBP_color_bits] > 24) {
592 quality -= 100;
593 }
594
595 // Bonus for each multisample in case the special value of 1 multisample
596 // is requested. Extra: 2 per sample.
597 if (_property[FBP_multisamples] != 0 &&
598 reqs._property[FBP_multisamples] == 1) {
599 quality += 2 * _property[FBP_multisamples];
600 }
601
602 // Bonus for each depth bit. Extra: 8 per bit.
603 // Please note that the Intel Windows driver only gives extra depth in
604 // combination with a stencil buffer, so we need 8 extra depth bits to
605 // outweigh the penalty of 50 for the unwanted stencil buffer, otherwise we
606 // will end up only getting 16-bit depth.
607 if (reqs._property[FBP_depth_bits] != 0) {
608 quality += 8 * _property[FBP_depth_bits];
609 }
610
611 // Bonus for each coverage sample in case the special value of 1 sample
612 // is requested. Extra: 2 per sample.
613 if (_property[FBP_coverage_samples] != 0 &&
614 reqs._property[FBP_coverage_samples] == 1) {
615 quality += 2 * _property[FBP_coverage_samples];
616 }
617
618 // Bonus for each color, alpha, stencil, and accum. Extra: 1 per bit.
619 for (int prop=FBP_color_bits; prop<=FBP_accum_bits; prop++) {
620 if (reqs._property[prop] != 0) {
621 quality += _property[prop];
622 }
623 }
624
625 return quality;
626}

673bool FrameBufferProperties::
674setup_color_texture(Texture *tex) const {
675 // Note by rdb: I'm not entirely happy about this system. I'd eventually
676 // like to move to a system in which framebuffer color formats and texture
677 // formats are unified (like in Direct3D and OpenGL) and where a table such
678 // as the below one would be generated dynamically by the GSG to reflect the
679 // formats that are supported for render-to-texture.
680
681 static const int num_formats = 17;
682 static const struct {
683 unsigned char color_bits, red_bits, green_bits, blue_bits, alpha_bits;
684 bool has_float;
685 Texture::Format format;
686 } formats[num_formats] = {
687 { 1, 1, 0, 0, 0, false, Texture::F_red },
688 { 1, 1, 1, 0, 0, false, Texture::F_rg },
689 { 1, 1, 1, 1, 0, false, Texture::F_rgb },
690 { 1, 1, 1, 1, 1, false, Texture::F_rgba },
691 { 8, 8, 0, 0, 0, false, Texture::F_red },
692 { 16, 8, 8, 0, 0, false, Texture::F_rg },
693 { 24, 8, 8, 8, 0, false, Texture::F_rgb8 },
694 { 32, 8, 8, 8, 8, false, Texture::F_rgba8 },
695 { 16, 16, 0, 0, 0, true, Texture::F_r16 },
696 { 32, 16, 16, 0, 0, true, Texture::F_rg16 },
697 { 32, 11, 11, 10, 0, true, Texture::F_r11_g11_b10 },
698 { 48, 16, 16, 16, 0, true, Texture::F_rgb16 },
699 { 48, 16, 16, 16, 16, true, Texture::F_rgba16 },
700 { 32, 32, 0, 0, 0, true, Texture::F_r32 },
701 { 64, 32, 32, 0, 0, true, Texture::F_rg32 },
702 { 96, 32, 32, 32, 0, true, Texture::F_rgb32 },
703 { 96, 32, 32, 32, 32, true, Texture::F_rgba32 },
704 };
705
706 if (get_srgb_color()) {
707 // These are the only sRGB formats. Deal with it.
708 if (get_alpha_bits() == 0) {
709 tex->set_format(Texture::F_srgb);
710 } else {
711 tex->set_format(Texture::F_srgb_alpha);
712 }
713
714 return (get_color_bits() <= 24 &&
715 get_red_bits() <= 8 &&
716 get_green_bits() <= 8 &&
717 get_blue_bits() <= 8 &&
718 get_alpha_bits() <= 8);
719
720 } else {
721 if (get_float_color()) {
722 tex->set_component_type(Texture::T_float);
723 }
724
725 for (int i = 0; i < num_formats; ++i) {
726 if (get_color_bits() <= (int)formats[i].color_bits &&
727 get_red_bits() <= (int)formats[i].red_bits &&
728 get_green_bits() <= (int)formats[i].green_bits &&
729 get_blue_bits() <= (int)formats[i].blue_bits &&
730 get_alpha_bits() <= (int)formats[i].alpha_bits &&
731 get_float_color() <= formats[i].has_float) {
732
733 tex->set_format(formats[i].format);
734 return true;
735 }
736 }
737
738 // Can't get requested bits. Choose a generic format and return.
739 tex->set_format((get_alpha_bits() == 0) ? Texture::F_rgb : Texture::F_rgba);
740 return false;
741 }
