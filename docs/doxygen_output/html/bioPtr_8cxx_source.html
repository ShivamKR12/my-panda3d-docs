49static string format_error() {
50 PVOID buffer;
51 DWORD len;
52 len = FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
53 nullptr, WSAGetLastError(), 0, (LPTSTR)&buffer, 0, nullptr);
54 if (len == 0) {
55 return string("Unknown error message");
56 }
57
58 const char *text = (const char *)buffer;
59 while (len > 0 && isspace(text[len - 1])) {
60 --len;
61 }
62
63 string result(text, len);
64 LocalFree(buffer);
65 return result;
66}
67#else
68#define format_error() strerror(errno)

78sock_should_retry(int i) {
79 if (i == 0 || i == -1) {
80 int err = WSAGetLastError();

83 case WSAEWOULDBLOCK:
84 case ENOTCONN:

90 }

103BioPtr::
104BioPtr(const URLSpec &url) : _connecting(false) {
105 if (url.get_scheme() == "file") {
106 // We're just reading a disk file.
107 string filename = URLSpec::unquote(url.get_path());
108#ifdef _WIN32
109 // On Windows, we have to munge the filename specially, because it's been
110 // URL-munged. It might begin with a leading slash as well as a drive
111 // letter. Clean up that nonsense.
112 if (!filename.empty()) {
113 if (filename[0] == '/' || filename[0] == '\\') {
114 Filename fname = Filename::from_os_specific(filename.substr(1));
115 if (fname.is_local()) {
116 // Put the slash back on.
117 fname = string("/") + fname.get_fullpath();
118 }
119 filename = fname.to_os_specific();
120 }
121 }
122#endif // _WIN32
123 _server_name = "";
124 _port = 0;
125 _bio = BIO_new_file(filename.c_str(), "rb");
126
127 } else {
128 // A normal network-based URL. We don't use BIO_new_connect since it
129 // doesn't handle IPv6 properly.
130 _server_name = url.get_server();
131 _port = url.get_port();
132 _bio = nullptr;
133
134 // These hints tell getaddrinfo what kind of address to return.
135 struct addrinfo hints, *res = nullptr;
136 memset(&hints, 0, sizeof(hints));
137
138 hints.ai_flags = AI_V4MAPPED | AI_ADDRCONFIG;
139 hints.ai_family = support_ipv6 ? AF_UNSPEC : AF_INET;
140 hints.ai_socktype = SOCK_STREAM;
141
142 // Resolve the hostname or address string.
143 int result = getaddrinfo(_server_name.c_str(), nullptr, &hints, &res);
144 if (result != 0) {
145 const char *errmsg;
146#ifndef _WIN32
147 if (result == EAI_SYSTEM && errno != 0) {
148 errmsg = strerror(errno);
149 } else
150#endif
151 {
152 errmsg = gai_strerror(result);
153 }
154 downloader_cat.error()
155 << "Failed to resolve " << url.get_server() << ": " << errmsg << "\n";
156 return;
157 }
158 nassertv(res != nullptr && res->ai_addr != nullptr);
159
160 // Store the real resolved address.
161 char buf[48];
162 buf[0] = 0;
163#ifdef _WIN32
164 DWORD bufsize = sizeof(buf);
165 WSAAddressToStringA(res->ai_addr, res->ai_addrlen, nullptr, buf, &bufsize);
166#else
167 if (res->ai_addr->sa_family == AF_INET) {
168 inet_ntop(AF_INET, (char *)&((sockaddr_in *)res->ai_addr)->sin_addr, buf, sizeof(buf));
169
170 } else if (res->ai_addr->sa_family == AF_INET6) {
171 inet_ntop(AF_INET6, (char *)&((sockaddr_in6 *)res->ai_addr)->sin6_addr, buf, sizeof(buf));
172 }
173#endif
174
175 if (buf[0]) {
176 _server_name = buf;
177 }
178 if (downloader_cat.is_debug()) {
179 downloader_cat.debug()
180 << "Resolved " << url.get_server() << " to " << buf << "\n";
181 }
182
183 // Create the socket.
184 int fd = socket(res->ai_family, SOCK_STREAM, IPPROTO_TCP);
185 if (fd < 0) {
186 downloader_cat.error()
187 << "Failed to create socket: " << format_error() << "\n";
188 _bio = nullptr;
189 freeaddrinfo(res);
190 return;
191 }
192
193 // Store the address and length for later use in connect().
194 nassertv(res->ai_addrlen <= sizeof(_addr));
195 memcpy(&_addr, res->ai_addr, res->ai_addrlen);
196 _addrlen = res->ai_addrlen;
197 freeaddrinfo(res);
198
199 // Also set the port we'd like to connect to.
200 if (_addr.ss_family == AF_INET) {
201 ((sockaddr_in &)_addr).sin_port = htons(_port);
202 } else if (_addr.ss_family == AF_INET6) {
203 ((sockaddr_in6 &)_addr).sin6_port = htons(_port);
204 }
205
206 _bio = BIO_new_socket(fd, 1);
207 }
208}

229bool BioPtr::
230connect() {
231 if (_bio == nullptr) {
232 return false;
233 }
234
235 int fd = -1;
236 BIO_get_fd(_bio, &fd);
237 nassertr(fd >= 0, false);
238
239 int result;
240 if (_connecting) {
241 result = BIO_sock_error(fd);
242 } else {
243 result = ::connect(fd, (sockaddr *)&_addr, _addrlen);
244 if (result != 0 && sock_should_retry(-1)) {
245 // It's still in progress; we should retry later. This causes
246 // should_retry() to return true.
247 BIO_set_flags(_bio, BIO_FLAGS_SHOULD_RETRY);
248 _connecting = true;
249 return false;
250 }
251 }
252 BIO_clear_retry_flags(_bio);
253 _connecting = false;
254
255 if (result != 0) {
256 downloader_cat.warning()
257 << "Failed to connect to " << _server_name << " port " << _port
258 << ": " << format_error() << "\n";
259 return false;
260 }
261
262 return true;
263}
