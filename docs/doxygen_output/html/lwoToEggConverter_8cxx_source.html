102bool LwoToEggConverter::
103convert_file(const Filename &filename) {
104 LwoInputFile in;
105
106 nout << "Reading " << filename << "\n";
107 if (!in.open_read(filename)) {
108 nout << "Unable to open " << filename << "\n";
109 return false;
110 }
111
112 PT(IffChunk) chunk = in.get_chunk();
113 if (chunk == nullptr) {
114 nout << "Unable to read " << filename << "\n";
115 return false;
116 }
117
118 if (!chunk->is_of_type(LwoHeader::get_class_type())) {
119 nout << "File " << filename << " is not a Lightwave Object file.\n";
120 return false;
121 }
122
123 LwoHeader *header = DCAST(LwoHeader, chunk);
124 if (!header->is_valid()) {
125 nout << "File " << filename
126 << " is not recognized as a Lightwave Object file. "
127 << "Perhaps the version is too recent.\n";
128 return false;
129 }
130
131 return convert_lwo(header);
132}

237void LwoToEggConverter::
238collect_lwo() {
239 CLwoLayer *last_layer = nullptr;
240 CLwoPoints *last_points = nullptr;
241 CLwoPolygons *last_polygons = nullptr;
242
243 const LwoTags *tags = nullptr;
244
245 int num_chunks = _lwo_header->get_num_chunks();
246 for (int i = 0; i < num_chunks; i++) {
247 const IffChunk *chunk = _lwo_header->get_chunk(i);
248
249 if (chunk->is_of_type(LwoLayer::get_class_type())) {
250 const LwoLayer *lwo_layer = DCAST(LwoLayer, chunk);
251 CLwoLayer *layer = new CLwoLayer(this, lwo_layer);
252 int number = layer->get_number();
253 slot_layer(number);
254
255 if (_layers[number] != nullptr) {
256 nout << "Warning: multiple layers with number " << number << "\n";
257 }
258 _layers[number] = layer;
259 last_layer = layer;
260 last_points = nullptr;
261 last_polygons = nullptr;
262
263 } else if (chunk->is_of_type(LwoClip::get_class_type())) {
264 const LwoClip *lwo_clip = DCAST(LwoClip, chunk);
265 CLwoClip *clip = new CLwoClip(this, lwo_clip);
266
267 int index = clip->get_index();
268 slot_clip(index);
269
270 if (_clips[index] != nullptr) {
271 nout << "Warning: multiple clips with index " << index << "\n";
272 }
273 _clips[index] = clip;
274
275 } else if (chunk->is_of_type(LwoPoints::get_class_type())) {
276 if (last_layer == nullptr) {
277 last_layer = make_generic_layer();
278 }
279
280 const LwoPoints *lwo_points = DCAST(LwoPoints, chunk);
281 CLwoPoints *points = new CLwoPoints(this, lwo_points, last_layer);
282 _points.push_back(points);
283 last_points = points;
284
285 } else if (chunk->is_of_type(LwoVertexMap::get_class_type())) {
286 if (last_points == nullptr) {
287 nout << "Vertex map chunk encountered without a preceding points chunk.\n";
288 } else {
289 const LwoVertexMap *lwo_vmap = DCAST(LwoVertexMap, chunk);
290 last_points->add_vmap(lwo_vmap);
291 }
292
293 } else if (chunk->is_of_type(LwoDiscontinuousVertexMap::get_class_type())) {
294 if (last_polygons == nullptr) {
295 nout << "Discontinous vertex map chunk encountered without a preceding polygons chunk.\n";
296 } else {
297 const LwoDiscontinuousVertexMap *lwo_vmad = DCAST(LwoDiscontinuousVertexMap, chunk);
298 last_polygons->add_vmad(lwo_vmad);
299 }
300
301 } else if (chunk->is_of_type(LwoTags::get_class_type())) {
302 tags = DCAST(LwoTags, chunk);
303
304 } else if (chunk->is_of_type(LwoPolygons::get_class_type())) {
305 if (last_points == nullptr) {
306 nout << "Polygon chunk encountered without a preceding points chunk.\n";
307 } else {
308 const LwoPolygons *lwo_polygons = DCAST(LwoPolygons, chunk);
309 CLwoPolygons *polygons =
310 new CLwoPolygons(this, lwo_polygons, last_points);
311 _polygons.push_back(polygons);
312 last_polygons = polygons;
313 }
314
315 } else if (chunk->is_of_type(LwoPolygonTags::get_class_type())) {
316 if (last_polygons == nullptr) {
317 nout << "Polygon tags chunk encountered without a preceding polygons chunk.\n";
318 } else if (tags == nullptr) {
319 nout << "Polygon tags chunk encountered without a preceding tags chunk.\n";
320 } else {
321 const LwoPolygonTags *lwo_ptags = DCAST(LwoPolygonTags, chunk);
322 last_polygons->add_ptags(lwo_ptags, tags);
323 }
324
325 } else if (chunk->is_of_type(LwoSurface::get_class_type())) {
326 if (last_layer == nullptr) {
327 last_layer = make_generic_layer();
328 }
329
330 const LwoSurface *lwo_surface = DCAST(LwoSurface, chunk);
331 CLwoSurface *surface = new CLwoSurface(this, lwo_surface);
332
333 bool inserted = _surfaces.insert(Surfaces::value_type(surface->get_name(), surface)).second;
334 if (!inserted) {
335 nout << "Multiple surface definitions named " << surface->get_name() << "\n";
336 delete surface;
337 }
338 }
339 }
