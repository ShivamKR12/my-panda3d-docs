436gboolean GtkStatsPianoRoll::
437handle_motion(int graph_x, int graph_y) {
438 if (_drag_mode == DM_none && _potential_drag_mode == DM_none) {
439 // When the mouse is over a color bar, highlight it.
440 int collector_index = get_collector_under_pixel(graph_x, graph_y);
441 _label_stack.highlight_label(collector_index);
442 on_enter_label(collector_index);
443
444 /*
445 // Now we want to get a WM_MOUSELEAVE when the mouse leaves the graph
446 // window.
447 TRACKMOUSEEVENT tme = {
448 sizeof(TRACKMOUSEEVENT),
449 TME_LEAVE,
450 _graph_window,
451 0
452 };
453 TrackMouseEvent(&tme);
454 */
455 }
456 else {
457 // If the mouse is in some drag mode, stop highlighting.
458 _label_stack.highlight_label(-1);
459 on_leave_label(_highlighted_index);
460 }
461
462 if (_drag_mode == DM_scale) {
463 double ratio = (double)graph_x / (double)get_xsize();
464 if (ratio > 0.0f) {
465 set_horizontal_scale(_drag_scale_start / ratio);
466 }
467 return TRUE;
468 }
469 else if (_drag_mode == DM_new_guide_bar) {
470 // We haven't created the new guide bar yet; we won't until the mouse
471 // comes within the graph's region.
472 if (graph_x >= 0 && graph_x < get_xsize()) {
473 set_drag_mode(DM_guide_bar);
474 _drag_guide_bar = add_user_guide_bar(pixel_to_height(graph_x));
475 return TRUE;
476 }
477
478 } else if (_drag_mode == DM_guide_bar) {
479 move_user_guide_bar(_drag_guide_bar, pixel_to_height(graph_x));
480 return TRUE;
481 }
482
483 return GtkStatsGraph::handle_motion(graph_x, graph_y);
484}

533void GtkStatsPianoRoll::
534draw_guide_bar(cairo_t *cr, const PStatGraph::GuideBar &bar) {
535 int x = height_to_pixel(bar._height);
536
537 if (x > 0 && x < get_xsize() - 1) {
538 // Only draw it if it's not too close to the top.
539 switch (bar._style) {
540 case GBS_target:
541 cairo_set_source_rgb(cr, rgb_light_gray[0], rgb_light_gray[1], rgb_light_gray[2]);
542 break;
543
544 case GBS_user:
545 cairo_set_source_rgb(cr, rgb_user_guide_bar[0], rgb_user_guide_bar[1], rgb_user_guide_bar[2]);
546 break;
547
548 default:
549 cairo_set_source_rgb(cr, rgb_dark_gray[0], rgb_dark_gray[1], rgb_dark_gray[2]);
550 break;
551 }
552 cairo_move_to(cr, x, 0);
553 cairo_line_to(cr, x, get_ysize());
554 cairo_stroke(cr);
555 }

578void GtkStatsPianoRoll::
579draw_guide_label(cairo_t *cr, const PStatGraph::GuideBar &bar) {
580 switch (bar._style) {
581 case GBS_target:
582 cairo_set_source_rgb(cr, rgb_light_gray[0], rgb_light_gray[1], rgb_light_gray[2]);
583 break;
584
585 case GBS_user:
586 cairo_set_source_rgb(cr, rgb_user_guide_bar[0], rgb_user_guide_bar[1], rgb_user_guide_bar[2]);
587 break;
588
589 default:
590 cairo_set_source_rgb(cr, rgb_dark_gray[0], rgb_dark_gray[1], rgb_dark_gray[2]);
591 break;
592 }
593
594 int x = height_to_pixel(bar._height);
595 const std::string &label = bar._label;
596
597 PangoLayout *layout = gtk_widget_create_pango_layout(_scale_area, label.c_str());
598 int width, height;
599 pango_layout_get_pixel_size(layout, &width, &height);
600
601 if (bar._style != GBS_user) {
602 double from_height = pixel_to_height(x - width * _cr_scale);
603 double to_height = pixel_to_height(x + width * _cr_scale);
604 if (find_user_guide_bar(from_height, to_height) >= 0) {
605 // Omit the label: there's a user-defined guide bar in the same space.
606 g_object_unref(layout);
607 return;
608 }
609 }
610
611 if (x >= 0 && x < get_xsize()) {
612 // Now convert our x to a coordinate within our drawing area.
613 int junk_y;
614
615 x /= _cr_scale;
616
617 // The x coordinate comes from the graph_window.
618 gtk_widget_translate_coordinates(_graph_window, _scale_area,
619 x, 0,
620 &x, &junk_y);
621
622 GtkAllocation allocation;
623 gtk_widget_get_allocation(_scale_area, &allocation);
624
625 int this_x = x - width / 2;
626 cairo_move_to(cr, this_x, allocation.height - height);
627 pango_cairo_show_layout(cr, layout);
628 }
629
630 g_object_unref(layout);
631}
