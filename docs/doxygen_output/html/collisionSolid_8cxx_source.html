133PT(PandaNode) CollisionSolid::
134get_viz(const CullTraverser *, const CullTraverserData &, bool bounds_only) const {
135 LightMutexHolder holder(_lock);
136 if ((_flags & F_viz_geom_stale) != 0) {
137 if (_viz_geom == nullptr) {
138 ((CollisionSolid *)this)->_viz_geom = new GeomNode("viz");
139 ((CollisionSolid *)this)->_bounds_viz_geom = new GeomNode("bounds_viz");
140 } else {
141 _viz_geom->remove_all_geoms();
142 _viz_geom->clear_effects();
143 _viz_geom->clear_transform();
144 _bounds_viz_geom->remove_all_geoms();
145 _bounds_viz_geom->clear_effects();
146 _bounds_viz_geom->clear_transform();
147 }
148 ((CollisionSolid *)this)->fill_viz_geom();
149 ((CollisionSolid *)this)->_flags &= ~F_viz_geom_stale;
150 }
151
152 if (bounds_only) {
153 return _bounds_viz_geom;
154 } else {
155 return _viz_geom;
156 }

301void CollisionSolid::
302report_undefined_intersection_test(TypeHandle from_type, TypeHandle into_type) {
303#ifndef NDEBUG
304 typedef pset<CollisionSolidUndefinedPair> Reported;
305 static Reported reported;
306
307 if (reported.insert(CollisionSolidUndefinedPair(from_type, into_type)).second) {
308 collide_cat.error()
309 << "Invalid attempt to detect collision from " << from_type << " into "
310 << into_type << "!\n\n"
311
312 "This means that a " << from_type << " object attempted to test for an\n"
313 "intersection into a " << into_type << " object. This intersection\n"
314 "test has not yet been defined; it is possible the " << into_type << "\n"
315 "object is not intended to be collidable. Consider calling\n"
316 "set_into_collide_mask(0) on the " << into_type << " object, or\n"
317 "set_from_collide_mask(0) on the " << from_type << " object.\n\n";
318 }
319#endif // NDEBUG
320}

392CPT(RenderState) CollisionSolid::
393get_solid_viz_state() {
394 // Once someone asks for this pointer, we hold its reference count and never
395 // free it.
396 static CPT(RenderState) base_state = nullptr;
397 if (base_state == nullptr) {
398 base_state = RenderState::make
399 (CullFaceAttrib::make(CullFaceAttrib::M_cull_clockwise),
400 RenderModeAttrib::make(RenderModeAttrib::M_filled),
401 TransparencyAttrib::make(TransparencyAttrib::M_alpha));
402 }
403
404 if (!do_is_tangible()) {
405 static CPT(RenderState) intangible_state = nullptr;
406 if (intangible_state == nullptr) {
407 intangible_state = base_state->add_attrib
408 (ColorAttrib::make_flat(LColor(1.0f, 0.3, 0.5f, 0.5f)));
409 }
410 return intangible_state;
411
412 } else if (do_has_effective_normal()) {
413 static CPT(RenderState) fakenormal_state = nullptr;
414 if (fakenormal_state == nullptr) {
415 fakenormal_state = base_state->add_attrib
416 (ColorAttrib::make_flat(LColor(0.5f, 0.5f, 1.0f, 0.5f)));
417 }
418 return fakenormal_state;
419
420 } else {
421 static CPT(RenderState) tangible_state = nullptr;
422 if (tangible_state == nullptr) {
423 tangible_state = base_state->add_attrib
424 (ColorAttrib::make_flat(LColor(1.0f, 1.0f, 1.0f, 0.5f)));
425 }
426 return tangible_state;
427 }
