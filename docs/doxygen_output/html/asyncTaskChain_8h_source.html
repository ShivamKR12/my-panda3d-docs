147 class AsyncTaskSortPriority {
148 public:
149 bool operator () (AsyncTask *a, AsyncTask *b) const {
150 if (a->get_sort() != b->get_sort()) {
151 return a->get_sort() > b->get_sort();
152 }
153 if (a->get_priority() != b->get_priority()) {
154 return a->get_priority() < b->get_priority();
155 }
156 if (a->get_start_time() != b->get_start_time()) {
157 return a->get_start_time() > b->get_start_time();
158 }
159 // Failing any other ordering criteria, we sort the tasks based on the
160 // order in which they were added to the task chain.
161 return a->_implicit_sort > b->_implicit_sort;
162 }
163 };
