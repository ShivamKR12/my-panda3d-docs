94void PStatServer::
95poll() {
96 // Delete all the readers that we couldn't delete before.
97 while (!_lost_readers.empty()) {
98 PStatReader *reader = _lost_readers.back();
99 _lost_readers.pop_back();
100
101 reader->lost_connection();
102 delete reader;
103 }
104 while (!_removed_readers.empty()) {
105 PStatReader *reader = _removed_readers.back();
106 _removed_readers.pop_back();
107 delete reader;
108 }
109
110 _listener->poll();
111
112 Readers::const_iterator ri = _readers.begin();
113 while (ri != _readers.end()) {
114 // Preincrement the iterator, in case we remove it as a result of calling
115 // poll().
116 Readers::const_iterator rnext = ri;
117 ++rnext;
118 PStatReader *reader = (*ri).second;
119
120 reader->poll();
121 reader->idle();
122
123 ri = rnext;
124 }
125}

294
