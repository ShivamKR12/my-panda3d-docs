104PyObject *Extension<NodePath>::
105__reduce_persist__(PyObject *self, PyObject *pickler) const {
106 // We should return at least a 2-tuple, (Class, (args)): the necessary class
107 // object whose constructor we should call (e.g. this), and the arguments
108 // necessary to reconstruct this object.
109
110 BamWriter *writer = nullptr;
111 if (pickler != nullptr) {
112 PyObject *py_writer = PyObject_GetAttrString(pickler, "bamWriter");
113 if (py_writer == nullptr) {
114 // It's OK if there's no bamWriter.
115 PyErr_Clear();
116 } else {
117 DtoolInstance_GetPointer(py_writer, writer, Dtool_BamWriter);
118 Py_DECREF(py_writer);
119 }
120 }
121
122 // We have a non-empty NodePath.
123
124 vector_uchar bam_stream;
125 if (!_this->encode_to_bam_stream(bam_stream, writer)) {
126 std::ostringstream stream;
127 stream << "Could not bamify " << _this;
128 std::string message = stream.str();
129 PyErr_SetString(PyExc_TypeError, message.c_str());
130 return nullptr;
131 }
132
133 // Start by getting this class object.
134 PyObject *this_class = (PyObject *)Py_TYPE(self);
135 if (this_class == nullptr) {
136 return nullptr;
137 }
138
139 PyObject *func;
140 if (writer != nullptr) {
141 // The modified pickle support: call the "persistent" version of this
142 // function, which receives the unpickler itself as an additional
143 // parameter.
144 func = Extension<TypedWritable>::find_global_decode(this_class, "py_decode_NodePath_from_bam_stream_persist");
145 if (func == nullptr) {
146 PyErr_SetString(PyExc_TypeError, "Couldn't find py_decode_NodePath_from_bam_stream_persist()");
147 return nullptr;
148 }
149
150 } else {
151 // The traditional pickle support: call the non-persistent version of this
152 // function.
153 func = Extension<TypedWritable>::find_global_decode(this_class, "py_decode_NodePath_from_bam_stream");
154 if (func == nullptr) {
155 PyErr_SetString(PyExc_TypeError, "Couldn't find py_decode_NodePath_from_bam_stream()");
156 return nullptr;
157 }
158 }
159
160 // PyTuple_SET_ITEM conveniently borrows the reference it is passed.
161 PyObject *args = PyTuple_New(1);
162 PyTuple_SET_ITEM(args, 0, Dtool_WrapValue(bam_stream));
163
164 PyObject *tuple = PyTuple_New(2);
165 PyTuple_SET_ITEM(tuple, 0, func);
166 PyTuple_SET_ITEM(tuple, 1, args);
167 return tuple;
