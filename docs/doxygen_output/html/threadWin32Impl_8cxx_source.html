60static __declspec(noinline) Thread *
61init_current_thread() {
62 Thread *thread = _current_thread;
63 if (!_main_thread_known.test_and_set(std::memory_order_relaxed)) {
64 // Assume that we must be in the main thread, since this method must be
65 // called before the first thread is spawned.
66 thread = Thread::get_main_thread();
67 _current_thread = thread;
68 }
69 // If this assertion triggers, you are making Panda calls from a thread
70 // that has not first been registered using Thread::bind_thread().
71 nassertr(thread != nullptr, nullptr);

104bool ThreadWin32Impl::
105start(ThreadPriority priority, bool joinable) {
106 _mutex.lock();
107 if (thread_cat->is_debug()) {
108 thread_cat.debug() << "Starting " << *_parent_obj << "\n";
109 }
110
111 nassertd(_status == S_new && _thread == 0) {
112 _mutex.unlock();
113 return false;
114 }
115
116 _joinable = joinable;
117 _status = S_start_called;
118
119 // Increment the parent object's reference count first. The thread will
120 // eventually decrement it when it terminates.
121 _parent_obj->ref();
122 _thread =
123 CreateThread(nullptr, 0, &root_func, (void *)this, 0, &_thread_id);
124
125 if (_thread_id == 0) {
126 // Oops, we couldn't start the thread. Be sure to decrement the reference
127 // count we incremented above, and return false to indicate failure.
128 unref_delete(_parent_obj);
129 _mutex.unlock();
130 return false;
131 }
132
133 // Thread was successfully started. Set the priority as specified.
134 switch (priority) {
135 case TP_low:
136 SetThreadPriority(_thread, THREAD_PRIORITY_BELOW_NORMAL);
137 break;
138
139 case TP_high:
140 SetThreadPriority(_thread, THREAD_PRIORITY_ABOVE_NORMAL);
141 break;
142
143 case TP_urgent:
144 SetThreadPriority(_thread, THREAD_PRIORITY_HIGHEST);
145 break;
146
147 case TP_normal:
148 default:
149 SetThreadPriority(_thread, THREAD_PRIORITY_NORMAL);
150 break;
151 }
152
153 _mutex.unlock();
154 return true;
155}

216void ThreadWin32Impl::
217sleep(double seconds) {
218 Thread *thread = get_current_thread();
219 ThreadWin32Impl *self = &thread->_impl;
220
221 HANDLE timer = self->_timer;
222 if (timer == nullptr) {
223 timer = CreateWaitableTimerExW(nullptr, nullptr, CREATE_WAITABLE_TIMER_MANUAL_RESET | CREATE_WAITABLE_TIMER_HIGH_RESOLUTION, TIMER_ALL_ACCESS);
224 self->_timer = timer;
225 }
226
227 LARGE_INTEGER ft;
228 ft.QuadPart = seconds * -10000000LL;
229 SetWaitableTimer(timer, &ft, 0, nullptr, nullptr, 0);
230 WaitForSingleObject(timer, INFINITE);
231}

265DWORD ThreadWin32Impl::
266root_func(LPVOID data) {
267 TAU_REGISTER_THREAD();
268 {
269 // TAU_PROFILE("void ThreadWin32Impl::root_func()", " ", TAU_USER);
270
271 ThreadWin32Impl *self = (ThreadWin32Impl *)data;
272 _current_thread = self->_parent_obj;
273
274 {
275 self->_mutex.lock();
276 nassertd(self->_status == S_start_called) {
277 self->_mutex.unlock();
278 return 1;
279 }
280 self->_status = S_running;
281 self->_cv.notify();
282 self->_mutex.unlock();
283 }
284
285 self->_parent_obj->thread_main();
286
287 if (thread_cat->is_debug()) {
288 thread_cat.debug()
289 << "Terminating thread " << self->_parent_obj->get_name()
290 << ", count = " << self->_parent_obj->get_ref_count() << "\n";
291 }
292
293 {
294 self->_mutex.lock();
295 nassertd(self->_status == S_running) {
296 self->_mutex.unlock();
297 return 1;
298 }
299 self->_status = S_finished;
300 self->_cv.notify();
301 self->_mutex.unlock();
302 }
303
304 if (self->_profiling != 0) {
305 DisableThreadProfiling(self->_profiling);
306 self->_profiling = 0;
307 }
308
309 // Now drop the parent object reference that we grabbed in start(). This
310 // might delete the parent object, and in turn, delete the ThreadWin32Impl
311 // object.
312 unref_delete(self->_parent_obj);
313 }
314
315 return 0;
316}
