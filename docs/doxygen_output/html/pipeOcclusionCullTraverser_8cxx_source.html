207void PipeOcclusionCullTraverser::
208end_traverse() {
209 if (!_live) {
210 return;
211 }
212
213 PStatTimer timer(_finish_occlusion_pcollector);
214 GraphicsStateGuardian *gsg = _buffer->get_gsg();
215 Thread *current_thread = get_current_thread();
216
217 _current_query = nullptr;
218 _next_query = nullptr;
219
220 for (PendingObject &pobj : _pending_objects) {
221 if (pobj._query == nullptr) {
222 _occlusion_untested_pcollector.add_level(1);
223 _true_cull_handler->record_object(std::move(pobj._object), this);
224 } else {

226 if (num_fragments != 0) {

229 } else {

241 _buffer->end_flip();
242
243 delete _internal_cull_handler;
244 _internal_cull_handler = nullptr;
245

247 _occlusion_passed_pcollector.flush_level();

280bool PipeOcclusionCullTraverser::
281is_in_view(CullTraverserData &data) {
282 _next_query = nullptr;
283
284 if (!data.is_in_view(get_camera_mask())) {
285 return false;
286 }
287 if (!_live) {
288 return true;
289 }
290
291 if (_current_query != nullptr) {
292 // We've already performed an occlusion test for some ancestor of this
293 // node; no need to perform another.
294 return true;
295 }
296
297 PandaNode *node = data.node();
298 PandaNodePipelineReader *node_reader = data.node_reader();
299
300 if (node_reader->get_nested_vertices() < min_occlusion_vertices) {
301 // Never mind; let this puny one slide.
302 return true;
303 }
304
305 if (node->is_geom_node() ||
306 node_reader->is_final() ||
307 node_reader->get_effects()->has_show_bounds() ||
308 node_reader->get_nested_vertices() <= max_occlusion_vertices) {
309 // In any of these cases, there's sufficient reason to perform an
310 // occlusion test on this particular node. Do it.
311
312 CPT(BoundingVolume) vol = node_reader->get_bounds();
313 CPT(TransformState) net_transform = data.get_net_transform(this);
314 CPT(TransformState) internal_transform;
315
316 CPT(Geom) geom;
317 if (get_volume_viz(vol, geom, net_transform, internal_transform)) {
318 _next_query =
319 perform_occlusion_test(geom, net_transform, internal_transform);
320 }
321 }
322
323 return true;
324}

356void PipeOcclusionCullTraverser::
357record_object(CullableObject &&object, const CullTraverser *traverser) {
358 nassertv(traverser == this);
359 PendingObject pobj(std::move(object));
360
361 Thread *current_thread = get_current_thread();
362
363 if (_next_query != nullptr) {
364 // We have just performed an occlusion query for this node. Don't perform
365 // another one.
366 pobj._query = _next_query;
367
368 } else if (_current_query != nullptr) {
369 // We have previously performed an occlusion query for this node or some
370 // ancestor. Don't perform another one.
371 pobj._query = _current_query;
372
373 } else if (pobj._object._geom->get_nested_vertices(current_thread) < min_occlusion_vertices) {
374 // This object is too small to bother testing for occlusions.
375
376 } else {
377 // Issue an occlusion test for this object.
378 CPT(BoundingVolume) vol = pobj._object._geom->get_bounds(current_thread);
379 CPT(TransformState) net_transform = _inv_cs_world_transform->compose(pobj._object._internal_transform);
380 CPT(TransformState) internal_transform;
381 CPT(Geom) geom;
382 if (get_volume_viz(vol, geom, net_transform, internal_transform)) {
383 pobj._query =
384 perform_occlusion_test(geom, net_transform, internal_transform);
385 }
386 }
387
388 _pending_objects.push_back(std::move(pobj));
389}

394void PipeOcclusionCullTraverser::
395make_sphere() {
396 ConfigVariableInt num_slices("num-slices", 16);
397 ConfigVariableInt num_stacks("num-stacks", 8);
398
399 // static const int num_slices = 16; static const int num_stacks = 8;
400
401 PT(GeomVertexData) vdata = new GeomVertexData
402 ("occlusion_sphere", GeomVertexFormat::get_v3(), Geom::UH_static);
403 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
404
405 PT(GeomTristrips) strip = new GeomTristrips(Geom::UH_stream);
406 for (int sl = 0; sl < num_slices; ++sl) {
407 PN_stdfloat longitude0 = (PN_stdfloat)sl / (PN_stdfloat)num_slices;
408 PN_stdfloat longitude1 = (PN_stdfloat)(sl + 1) / (PN_stdfloat)num_slices;
409 vertex.add_data3(compute_sphere_point(0.0, longitude0));
410 for (int st = 1; st < num_stacks; ++st) {
411 PN_stdfloat latitude = (PN_stdfloat)st / (PN_stdfloat)num_stacks;
412 vertex.add_data3(compute_sphere_point(latitude, longitude0));
413 vertex.add_data3(compute_sphere_point(latitude, longitude1));
414 }
415 vertex.add_data3(compute_sphere_point(1.0, longitude0));
416
417 strip->add_next_vertices(num_stacks * 2);
418 strip->close_primitive();
419 }
420
421 _sphere_geom = new Geom(vdata);
422 _sphere_geom->add_primitive(strip);
423}

429LVertex PipeOcclusionCullTraverser::
430compute_sphere_point(PN_stdfloat latitude, PN_stdfloat longitude) {
431 PN_stdfloat s1, c1;
432 csincos(latitude * MathNumbers::pi, &s1, &c1);
433
434 PN_stdfloat s2, c2;
435 csincos(longitude * 2.0f * MathNumbers::pi, &s2, &c2);
436
437 LVertex p(s1 * c2, s1 * s2, c1);
438 return p;
439}

503bool PipeOcclusionCullTraverser::
504get_volume_viz(const BoundingVolume *vol,
505 CPT(Geom) &geom, // OUT
506 CPT(TransformState) &net_transform, // IN-OUT
507 CPT(TransformState) &internal_transform // OUT
508 ) {
509 if (vol->is_infinite() || vol->is_empty()) {
510 return false;
511 }
512
513 if (vol->is_exact_type(BoundingSphere::get_class_type())) {
514 const BoundingSphere *sphere = DCAST(BoundingSphere, vol);
515 CPT(TransformState) local_transform =
516 TransformState::make_pos_hpr_scale(sphere->get_center(),
517 LVecBase3(0, 0, 0),
518 sphere->get_radius());
519 net_transform = net_transform->compose(local_transform);
520
521 CPT(TransformState) modelview_transform =
522 _internal_trav->get_world_transform()->compose(net_transform);
523
524 // See if the bounding sphere is clipped by the near plane. If it is, the
525 // occlusion test may fail, so we won't bother performing it for this
526 // object. Anyway, it's not occluded by anything, since it's intersecting
527 // the near plane.
528 const LPoint3 &center = modelview_transform->get_pos();
529 const LVecBase3 &radius = modelview_transform->get_scale();
530 if (center[1] - radius[1] < 0.0f) {
531 return false;
532 }
533
534 // Construct the internal transform for the internal traverser.
535 internal_transform = _internal_trav->get_scene()->
536 get_cs_transform()->compose(modelview_transform);
537
538 // The sphere looks good.
539 geom = _sphere_geom;
540 return true;
541
542 } else if (vol->is_exact_type(BoundingBox::get_class_type())) {
543 const BoundingBox *box = DCAST(BoundingBox, vol);
544 CPT(TransformState) local_transform =
545 TransformState::make_pos_hpr_scale(box->get_minq(),
546 LVecBase3(0, 0, 0),
547 box->get_maxq() - box->get_minq());
548 net_transform = net_transform->compose(local_transform);
549
550 CPT(TransformState) modelview_transform =
551 _internal_trav->get_world_transform()->compose(net_transform);
552
553 // See if the bounding box is clipped by the near plane. If it is, the
554 // occlusion test may fail, so we won't bother performing it for this
555 // object. Anyway, it's not occluded by anything, since it's intersecting
556 // the near plane.
557 static const LPoint3 points[8] = {
558 LPoint3(0.0f, 0.0f, 0.0f),
559 LPoint3(0.0f, 0.0f, 1.0f),
560 LPoint3(0.0f, 1.0f, 0.0f),
561 LPoint3(0.0f, 1.0f, 1.0f),
562 LPoint3(1.0f, 0.0f, 0.0f),
563 LPoint3(1.0f, 0.0f, 1.0f),
564 LPoint3(1.0f, 1.0f, 0.0f),
565 LPoint3(1.0f, 1.0f, 1.0f),
566 };
567 const LMatrix4 &mat = modelview_transform->get_mat();
568 for (int i = 0; i < 8; ++i) {
569 LPoint3 p = points[i] * mat;
570 if (p[1] < 0.0f) {
571 return false;
572 }
573 }
574
575 // Construct the internal transform for the internal traverser.
576 internal_transform = _internal_trav->get_scene()->
577 get_cs_transform()->compose(modelview_transform);
578
579 // The box looks good.
580 geom = _box_geom;
581 return true;
582 }
583
584 // Don't have a suitable representation for this bounding volume.
585 return false;
586}

592PT(OcclusionQueryContext) PipeOcclusionCullTraverser::
593perform_occlusion_test(const Geom *geom, const TransformState *net_transform,
594 const TransformState *internal_transform) {
595 _occlusion_tests_pcollector.add_level(1);
596 PStatTimer timer(_test_occlusion_pcollector);
597
598 GraphicsStateGuardian *gsg = _buffer->get_gsg();
599
600 gsg->begin_occlusion_query();
601
602 CullableObject viz(geom, _solid_test_state, internal_transform);
603
604 static ConfigVariableBool test_occlude("test-occlude", false);
605 if (test_occlude) {
606 _true_cull_handler->record_object(std::move(viz), _internal_trav);
607 } else {
608 _internal_cull_handler->record_object(std::move(viz), _internal_trav);
609 }
610
611 PT(OcclusionQueryContext) query = gsg->end_occlusion_query();
612
613 if (show_occlusion) {
614 // Show the results of the occlusion. To do this, we need to get the
615 // results of the query immediately. This will stall the pipe, but we're
616 // rendering a debug effect, so we don't mind too much.
617 int num_fragments = query->get_num_fragments();
618 show_results(num_fragments, geom, net_transform, internal_transform);
619 }
620
621 return query;
622}
