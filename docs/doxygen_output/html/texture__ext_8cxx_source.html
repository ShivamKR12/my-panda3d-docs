40 } else if (DtoolInstance_TYPE(image) == &Dtool_PointerToArray_unsigned_char) {
41 _this->set_ram_image(*(const PTA_uchar *)DtoolInstance_VOID_PTR(image), compression, page_size);
42 return;
43 }
44 }
45
46 if (PyObject_CheckBuffer(image)) {
47 // User passed a buffer object.
48 Py_buffer view;
49 if (PyObject_GetBuffer(image, &view, PyBUF_CONTIG_RO) == -1) {
50 PyErr_SetString(PyExc_TypeError,
51 "Texture.set_ram_image() requires a contiguous buffer");
52 return;
53 }
54
55 int component_width = _this->get_component_width();
56 if (compression == Texture::CM_off) {
57 if (view.itemsize != 1 && view.itemsize != component_width) {
58 PyErr_SetString(PyExc_TypeError,
59 "buffer.itemsize does not match Texture component size");
60 return;
61 }
62
63 if (view.len % component_width != 0) {
64 PyErr_Format(PyExc_ValueError,
65 "byte buffer is not a multiple of %d bytes",
66 component_width);
67 return;

69 } else {

71 PyErr_SetString(PyExc_TypeError,
72 "buffer.itemsize should be 1 for compressed images");
73 return;
74 }
75 }
76
77 PTA_uchar data = PTA_uchar::empty_array(view.len, Texture::get_class_type());
78 memcpy(data.p(), view.buf, view.len);
79 _this->set_ram_image(std::move(data), compression, page_size);
80
81 PyBuffer_Release(&view);
82 return;
83 }
