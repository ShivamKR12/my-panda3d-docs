127void GtkStatsLabel::
128update_color() {
129 // Set the fg and bg colors on the label.
130 LRGBColor rgb = _monitor->get_collector_color(_collector_index);
131 _bg_color = LRGBColor(
132 encode_sRGB_float((float)rgb[0]),
133 encode_sRGB_float((float)rgb[1]),
134 encode_sRGB_float((float)rgb[2]));
135
136 _highlight_bg_color = LRGBColor(
137 encode_sRGB_float((float)rgb[0] * 0.75f),
138 encode_sRGB_float((float)rgb[1] * 0.75f),
139 encode_sRGB_float((float)rgb[2] * 0.75f));
140
141 // Should our foreground be black or white?
142 PN_stdfloat bright = _bg_color.dot(LRGBColor(0.2126, 0.7152, 0.0722));
143 if (bright >= 0.5) {
144 _fg_color = LRGBColor(0);
145 } else {
146 _fg_color = LRGBColor(1);
147 }
148 if (bright * 0.75 >= 0.5) {
149 _highlight_fg_color = LRGBColor(0);
150 } else {
151 _highlight_fg_color = LRGBColor(1);
152 }
153
154 gtk_widget_queue_draw(_widget);
155}

160void GtkStatsLabel::
161update_text(bool use_fullname) {
162 const PStatClientData *client_data = _monitor->get_client_data();
163 if (use_fullname) {
164 _text = client_data->get_collector_fullname(_collector_index);
165 } else {
166 _text = client_data->get_collector_name(_collector_index);
167 }
168
169 // Make up a PangoLayout to represent the text.
170 if (_layout) {
171 g_object_unref(_layout);
172 }
173 _layout = gtk_widget_create_pango_layout(_widget, _text.c_str());
174
175 // What are the extents of the text? This determines the minimum size of
176 // our widget.
177 int width, height;
178 pango_layout_get_pixel_size(_layout, &width, &height);
179 _ideal_width = width + _left_margin + _right_margin;
180 _height = height + _top_margin + _bottom_margin;
181 gtk_widget_set_size_request(_widget, _ideal_width, _height);
182}

198gboolean GtkStatsLabel::
199draw_callback(GtkWidget *widget, cairo_t *cr, gpointer data) {
200 GtkStatsLabel *self = (GtkStatsLabel *)data;
201
202 LRGBColor bg, fg;
203 if (self->_highlight || self->_mouse_within) {
204 bg = self->_highlight_bg_color;
205 fg = self->_highlight_fg_color;
206 } else {
207 bg = self->_bg_color;
208 fg = self->_fg_color;
209
210 }
211 cairo_set_source_rgb(cr, bg[0], bg[1], bg[2]);
212
213 GtkAllocation allocation;
214 gtk_widget_get_allocation(widget, &allocation);
215
216 cairo_rectangle(cr, 0, 0, allocation.width, allocation.height);
217 cairo_fill(cr);
218
219 int width, height;
220 pango_layout_get_pixel_size(self->_layout, &width, &height);
221
222 cairo_set_source_rgb(cr, fg[0], fg[1], fg[2]);
223 if (self->_align_right) {
224 cairo_move_to(cr, allocation.width - width - _right_margin, _top_margin);
225 } else {
226 cairo_move_to(cr, _left_margin, _top_margin);
227 }
228 pango_cairo_show_layout(cr, self->_layout);
229
230 return TRUE;
231}
