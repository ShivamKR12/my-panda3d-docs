297CPT(TransformState) TinyGraphicsStateGuardian::
298calc_projection_mat(const Lens *lens) {
299 if (lens == nullptr) {
300 return nullptr;
301 }
302
303 if (!lens->is_linear()) {
304 return nullptr;
305 }
306
307 // The projection matrix must always be right-handed Y-up, even if our
308 // coordinate system of choice is otherwise, because certain GL calls
309 // (specifically glTexGen(GL_SPHERE_MAP)) assume this kind of a coordinate
310 // system. Sigh. In order to implement a Z-up (or other arbitrary)
311 // coordinate system, we'll use a Y-up projection matrix, and store the
312 // conversion to our coordinate system of choice in the modelview matrix.
313
314 LMatrix4 result =
315 LMatrix4::convert_mat(CS_yup_right, _current_lens->get_coordinate_system()) *
316 lens->get_projection_mat(_current_stereo_channel);
317
318 if (_scene_setup->get_inverted()) {
319 // If the scene is supposed to be inverted, then invert the projection
320 // matrix.
321 result *= LMatrix4::scale_mat(1.0f, -1.0f, 1.0f);
322 }
323
324 return TransformState::make_mat(result);
325}

472bool TinyGraphicsStateGuardian::
473begin_draw_primitives(const GeomPipelineReader *geom_reader,
474 const GeomVertexDataPipelineReader *data_reader,
475 size_t num_instances, bool force) {
476#ifndef NDEBUG
477 if (tinydisplay_cat.is_spam()) {
478 tinydisplay_cat.spam() << "begin_draw_primitives: " << *(data_reader->get_object()) << "\n";
479 }
480#endif // NDEBUG
481
482 if (!GraphicsStateGuardian::begin_draw_primitives(geom_reader, data_reader, num_instances, force)) {
483 return false;
484 }
485 nassertr(_data_reader != nullptr, false);
486
487 PStatTimer timer(_draw_transform_pcollector);
488
489 // Set up the proper transform.
490 if (_data_reader->is_vertex_transformed()) {
491 // If the vertex data claims to be already transformed into clip
492 // coordinates, wipe out the current projection and modelview matrix (so
493 // we don't attempt to transform it again).
494 const TransformState *ident = TransformState::make_identity();
495 load_matrix(&_c->matrix_model_view, ident);
496 load_matrix(&_c->matrix_projection, _scissor_mat);
497 load_matrix(&_c->matrix_model_view_inv, ident);
498 load_matrix(&_c->matrix_model_projection, _scissor_mat);
499 _c->matrix_model_projection_no_w_transform = 1;
500 _transform_stale = true;
501
502 } else if (_transform_stale) {
503 // Load the actual transform.
504
505 CPT(TransformState) scissor_proj_mat = _scissor_mat->compose(_projection_mat);
506
507 if (_c->lighting_enabled) {
508 // With the lighting equation, we need to keep the modelview and
509 // projection matrices separate.
510
511 load_matrix(&_c->matrix_model_view, _internal_transform);
512 load_matrix(&_c->matrix_projection, scissor_proj_mat);
513
514 /* precompute inverse modelview */
515 M4 tmp;
516 gl_M4_Inv(&tmp, &_c->matrix_model_view);
517 gl_M4_Transpose(&_c->matrix_model_view_inv, &tmp);
518
519 }
520
521 // Compose the modelview and projection matrices.
522 load_matrix(&_c->matrix_model_projection,
523 scissor_proj_mat->compose(_internal_transform));
524
525 /* test to accelerate computation */
526 _c->matrix_model_projection_no_w_transform = 0;
527 PN_stdfloat *m = &_c->matrix_model_projection.m[0][0];
528 if (m[12] == 0.0 && m[13] == 0.0 && m[14] == 0.0) {
529 _c->matrix_model_projection_no_w_transform = 1;
530 }
531 _transform_stale = false;
532 }
533
534 // Figure out the subset of vertices we will be using in this operation.
535 int num_vertices = data_reader->get_num_rows();
536 _min_vertex = num_vertices;
537 _max_vertex = 0;
538 int num_prims = geom_reader->get_num_primitives();
539 int i;
540 for (i = 0; i < num_prims; ++i) {
541 CPT(GeomPrimitive) prim = geom_reader->get_primitive(i);
542 int nv = prim->get_min_vertex();
543 _min_vertex = min(_min_vertex, nv);
544 int xv = prim->get_max_vertex();
545 _max_vertex = max(_max_vertex, xv);
546 }
547 if (_min_vertex > _max_vertex) {
548 return false;
549 }
550
551 // Now copy all of those vertices into our working table, transforming into
552 // screen space them as we go.
553 int num_used_vertices = _max_vertex - _min_vertex + 1;
554 if (_vertices_size < num_used_vertices) {
555 if (_vertices_size == 0) {
556 _vertices_size = 1;
557 }
558 while (_vertices_size < num_used_vertices) {
559 _vertices_size *= 2;
560 }
561 if (_vertices != nullptr) {
562 PANDA_FREE_ARRAY(_vertices);
563 }
564 _vertices = (GLVertex *)PANDA_MALLOC_ARRAY(_vertices_size * sizeof(GLVertex));
565 }
566
567 GeomVertexReader rcolor, rnormal;
568
569 // We now support up to 3-stage multitexturing.
570 GenTexcoordFunc *texgen_func[MAX_TEXTURE_STAGES];
571 TexCoordData tcdata[MAX_TEXTURE_STAGES];
572
573 const TexGenAttrib *target_tex_gen = DCAST(TexGenAttrib, _target_rs->get_attrib_def(TexGenAttrib::get_class_slot()));
574 const TexMatrixAttrib *target_tex_matrix = DCAST(TexMatrixAttrib, _target_rs->get_attrib_def(TexMatrixAttrib::get_class_slot()));
575
576 int max_stage_index = _target_texture->get_num_on_ff_stages();
577 for (int si = 0; si < max_stage_index; ++si) {
578 TextureStage *stage = _target_texture->get_on_ff_stage(si);
579
580 switch (target_tex_gen->get_mode(stage)) {
581 case TexGenAttrib::M_eye_sphere_map:
582 tcdata[si]._r1 = GeomVertexReader(data_reader, InternalName::get_normal(),
583 force);
584 tcdata[si]._r2 = GeomVertexReader(data_reader, InternalName::get_vertex(),
585 force);
586 texgen_func[si] = &texgen_sphere_map;
587 tcdata[si]._mat = _internal_transform->get_mat();
588 break;
589
590 case TexGenAttrib::M_eye_position:
591 tcdata[si]._r1 = GeomVertexReader(data_reader, InternalName::get_vertex(),
592 force);
593 texgen_func[si] = &texgen_texmat;
594 {
595 CPT(TransformState) eye_transform =
596 _cs_transform->invert_compose(_internal_transform);
597 tcdata[si]._mat = eye_transform->get_mat();
598 }
599 if (target_tex_matrix->has_stage(stage)) {
600 tcdata[si]._mat = tcdata[si]._mat * target_tex_matrix->get_mat(stage);
601 }
602 break;
603
604 case TexGenAttrib::M_world_position:
605 tcdata[si]._r1 = GeomVertexReader(data_reader, InternalName::get_vertex(),
606 force);
607 texgen_func[si] = &texgen_texmat;
608 {
609 CPT(TransformState) render_transform =
610 _cs_transform->compose(_scene_setup->get_world_transform());
611 CPT(TransformState) world_inv_transform =
612 render_transform->invert_compose(_internal_transform);
613 tcdata[si]._mat = world_inv_transform->get_mat();
614 }
615 if (target_tex_matrix->has_stage(stage)) {
616 tcdata[si]._mat = tcdata[si]._mat * target_tex_matrix->get_mat(stage);
617 }
618 break;
619
620 default:
621 // Fall through: use the standard texture coordinates.
622 tcdata[si]._r1 = GeomVertexReader(data_reader, stage->get_texcoord_name(),
623 force);
624 texgen_func[si] = &texgen_simple;
625 if (target_tex_matrix->has_stage(stage)) {
626 texgen_func[si] = &texgen_texmat;
627 tcdata[si]._mat = target_tex_matrix->get_mat(stage);
628 }
629
630 break;
631 }
632 tcdata[si]._r1.set_row_unsafe(_min_vertex);
633 tcdata[si]._r2.set_row_unsafe(_min_vertex);
634 if (!tcdata[si]._r1.has_column()) {
635 texgen_func[si] = &texgen_null;
636 }
637 }
638
639 bool needs_color = false;
640 if (_vertex_colors_enabled) {
641 rcolor = GeomVertexReader(data_reader, InternalName::get_color(), force);
642 rcolor.set_row_unsafe(_min_vertex);
643 needs_color = rcolor.has_column();
644 }
645
646 if (!needs_color) {
647 const LColor &d = _scene_graph_color;
648 const LColor &s = _current_color_scale;
649 _c->current_color.v[0] = max(d[0] * s[0], (PN_stdfloat)0);
650 _c->current_color.v[1] = max(d[1] * s[1], (PN_stdfloat)0);
651 _c->current_color.v[2] = max(d[2] * s[2], (PN_stdfloat)0);
652 _c->current_color.v[3] = max(d[3] * s[3], (PN_stdfloat)0);
653 }
654
655 bool needs_normal = false;
656 if (_c->lighting_enabled) {
657 rnormal = GeomVertexReader(data_reader, InternalName::get_normal(), force);
658 rnormal.set_row_unsafe(_min_vertex);
659 needs_normal = rnormal.has_column();
660 }
661
662 GeomVertexReader rvertex(data_reader, InternalName::get_vertex(), force);
663 rvertex.set_row_unsafe(_min_vertex);
664
665 if (!rvertex.has_column()) {
666 // Whoops, guess the vertex data isn't resident.
667 return false;
668 }
669
670 if (!needs_color && _color_material_flags) {
671 if (_color_material_flags & CMF_ambient) {
672 _c->materials[0].ambient = _c->current_color;
673 _c->materials[1].ambient = _c->current_color;
674 }
675 if (_color_material_flags & CMF_diffuse) {
676 _c->materials[0].diffuse = _c->current_color;
677 _c->materials[1].diffuse = _c->current_color;
678 }
679 }
680
681 if (_texturing_state != 0 && _texture_replace) {
682 // We don't need the vertex color or lighting calculation after all, since
683 // the current texture will just hide all of that.
684 needs_color = false;
685 needs_normal = false;
686 }
687
688 for (i = 0; i < num_used_vertices; ++i) {
689 GLVertex *v = &_vertices[i];
690 const LVecBase4 &d = rvertex.get_data4();
691
692 v->coord.v[0] = d[0];
693 v->coord.v[1] = d[1];
694 v->coord.v[2] = d[2];
695 v->coord.v[3] = d[3];
696
697 // Texture coordinates.
698 for (int si = 0; si < max_stage_index; ++si) {
699 (*texgen_func[si])(v->tex_coord[si], tcdata[si]);
700 }
701
702 if (needs_color) {
703 const LColor &d = rcolor.get_data4();
704 const LColor &s = _current_color_scale;
705 _c->current_color.v[0] = max(d[0] * s[0], (PN_stdfloat)0);
706 _c->current_color.v[1] = max(d[1] * s[1], (PN_stdfloat)0);
707 _c->current_color.v[2] = max(d[2] * s[2], (PN_stdfloat)0);
708 _c->current_color.v[3] = max(d[3] * s[3], (PN_stdfloat)0);
709
710 if (_color_material_flags) {
711 if (_color_material_flags & CMF_ambient) {
712 _c->materials[0].ambient = _c->current_color;
713 _c->materials[1].ambient = _c->current_color;
714 }
715 if (_color_material_flags & CMF_diffuse) {
716 _c->materials[0].diffuse = _c->current_color;
717 _c->materials[1].diffuse = _c->current_color;
718 }
719 }
720 }
721
722 v->color = _c->current_color;
723
724 if (_c->lighting_enabled) {
725 if (needs_normal) {
726 const LVecBase3 &d = rnormal.get_data3();
727 _c->current_normal.v[0] = d[0];
728 _c->current_normal.v[1] = d[1];
729 _c->current_normal.v[2] = d[2];
730 _c->current_normal.v[3] = 0.0f;
731 }
732
733 gl_vertex_transform(_c, v);
734 gl_shade_vertex(_c, v);
735
736 } else {
737 gl_vertex_transform(_c, v);
738 }
739
740 if (v->clip_code == 0) {
741 gl_transform_to_viewport(_c, v);
742 }
743
744 v->edge_flag = 1;
745 }
746
747 // Set up the appropriate function callback for filling triangles, according
748 // to the current state.
749
750 bool srgb_blend = _current_properties->get_srgb_color();
751
752 int depth_write_state = 0; // zon
753 const DepthWriteAttrib *target_depth_write = DCAST(DepthWriteAttrib, _target_rs->get_attrib_def(DepthWriteAttrib::get_class_slot()));
754 if (target_depth_write->get_mode() != DepthWriteAttrib::M_on) {
755 depth_write_state = 1; // zoff
756 }
757
758 int color_write_state = 0; // cstore
759
760 const ColorWriteAttrib *target_color_write = DCAST(ColorWriteAttrib, _target_rs->get_attrib_def(ColorWriteAttrib::get_class_slot()));
761 unsigned int color_channels =
762 target_color_write->get_channels() & _color_write_mask;
763
764 if (color_channels == ColorWriteAttrib::C_all) {
765 if (srgb_blend) {
766 color_write_state = 4; // csstore
767 } else {
768 color_write_state = 0; // cstore
769 }
770 } else {
771 // Implement a color mask.
772 int op_a = get_color_blend_op(ColorBlendAttrib::O_one);
773 int op_b = get_color_blend_op(ColorBlendAttrib::O_zero);
774
775 if (srgb_blend) {
776 _c->zb->store_pix_func = store_pixel_funcs_sRGB[op_a][op_b][color_channels];
777 } else {
778 _c->zb->store_pix_func = store_pixel_funcs[op_a][op_b][color_channels];
779 }
780 color_write_state = 2; // cgeneral
781 }
782
783 const TransparencyAttrib *target_transparency = DCAST(TransparencyAttrib, _target_rs->get_attrib_def(TransparencyAttrib::get_class_slot()));
784 switch (target_transparency->get_mode()) {
785 case TransparencyAttrib::M_alpha:
786 case TransparencyAttrib::M_dual:
787 if (color_channels == ColorWriteAttrib::C_all) {
788 if (srgb_blend) {
789 color_write_state = 5; // csblend
790 } else {
791 color_write_state = 1; // cblend
792 }
793 } else {
794 // Implement a color mask, with alpha blending.
795 int op_a = get_color_blend_op(ColorBlendAttrib::O_incoming_alpha);
796 int op_b = get_color_blend_op(ColorBlendAttrib::O_one_minus_incoming_alpha);
797
798 if (srgb_blend) {
799 _c->zb->store_pix_func = store_pixel_funcs_sRGB[op_a][op_b][color_channels];
800 } else {
801 _c->zb->store_pix_func = store_pixel_funcs[op_a][op_b][color_channels];
802 }
803 color_write_state = 2; // cgeneral
804 }
805 break;
806
807 case TransparencyAttrib::M_premultiplied_alpha:
808 {
809 // Implement a color mask, with pre-multiplied alpha blending.
810 int op_a = get_color_blend_op(ColorBlendAttrib::O_one);
811 int op_b = get_color_blend_op(ColorBlendAttrib::O_one_minus_incoming_alpha);
812
813 if (srgb_blend) {
814 _c->zb->store_pix_func = store_pixel_funcs_sRGB[op_a][op_b][color_channels];
815 } else {
816 _c->zb->store_pix_func = store_pixel_funcs[op_a][op_b][color_channels];
817 }
818 color_write_state = 2; // cgeneral
819 }
820 break;
821
822 default:
823 break;
824 }
825
826 const ColorBlendAttrib *target_color_blend = DCAST(ColorBlendAttrib, _target_rs->get_attrib_def(ColorBlendAttrib::get_class_slot()));
827 if (target_color_blend->get_mode() == ColorBlendAttrib::M_add) {
828 // If we have a color blend set that we can support, it overrides the
829 // transparency set.
830 LColor c = target_color_blend->get_color();
831 _c->zb->blend_r = (int)(c[0] * ZB_POINT_RED_MAX);
832 _c->zb->blend_g = (int)(c[1] * ZB_POINT_GREEN_MAX);
833 _c->zb->blend_b = (int)(c[2] * ZB_POINT_BLUE_MAX);
834 _c->zb->blend_a = (int)(c[3] * ZB_POINT_ALPHA_MAX);
835
836 int op_a = get_color_blend_op(target_color_blend->get_operand_a());
837 int op_b = get_color_blend_op(target_color_blend->get_operand_b());
838
839 if (srgb_blend) {
840 _c->zb->store_pix_func = store_pixel_funcs_sRGB[op_a][op_b][color_channels];
841 } else {
842 _c->zb->store_pix_func = store_pixel_funcs[op_a][op_b][color_channels];
843 }
844 color_write_state = 2; // cgeneral
845 }
846
847 if (color_channels == ColorWriteAttrib::C_off) {
848 color_write_state = 3; // coff
849 }
850
851 int alpha_test_state = 0; // anone
852 const AlphaTestAttrib *target_alpha_test = DCAST(AlphaTestAttrib, _target_rs->get_attrib_def(AlphaTestAttrib::get_class_slot()));
853 switch (target_alpha_test->get_mode()) {
854 case AlphaTestAttrib::M_none:
855 case AlphaTestAttrib::M_never:
856 case AlphaTestAttrib::M_always:
857 case AlphaTestAttrib::M_equal:
858 case AlphaTestAttrib::M_not_equal:
859 alpha_test_state = 0; // anone
860 break;
861
862 case AlphaTestAttrib::M_less:
863 case AlphaTestAttrib::M_less_equal:
864 alpha_test_state = 1; // aless
865 _c->zb->reference_alpha = (int)(target_alpha_test->get_reference_alpha() * ZB_POINT_ALPHA_MAX);
866 break;
867
868 case AlphaTestAttrib::M_greater:
869 case AlphaTestAttrib::M_greater_equal:
870 alpha_test_state = 2; // amore
871 _c->zb->reference_alpha = (int)(target_alpha_test->get_reference_alpha() * ZB_POINT_ALPHA_MAX);
872 break;
873 }
874
875 int depth_test_state = 1; // zless
876 _c->depth_test = 1; // set this for ZB_line
877 const DepthTestAttrib *target_depth_test = DCAST(DepthTestAttrib, _target_rs->get_attrib_def(DepthTestAttrib::get_class_slot()));
878 if (target_depth_test->get_mode() == DepthTestAttrib::M_none) {
879 depth_test_state = 0; // zless
880 _c->depth_test = 0;
881 }
882
883 const ShadeModelAttrib *target_shade_model = DCAST(ShadeModelAttrib, _target_rs->get_attrib_def(ShadeModelAttrib::get_class_slot()));
884 ShadeModelAttrib::Mode shade_model = target_shade_model->get_mode();
885 if (!needs_color && !_c->lighting_enabled) {
886 // With no per-vertex lighting, and no per-vertex colors, we might as well
887 // use the flat shading model.
888 shade_model = ShadeModelAttrib::M_flat;
889 }
890 int shade_model_state = 2; // smooth
891 _c->smooth_shade_model = true;
892
893 if (shade_model == ShadeModelAttrib::M_flat) {
894 _c->smooth_shade_model = false;
895 shade_model_state = 1; // flat
896 if (_c->current_color.v[0] == 1.0f &&
897 _c->current_color.v[1] == 1.0f &&
898 _c->current_color.v[2] == 1.0f &&
899 _c->current_color.v[3] == 1.0f) {
900 shade_model_state = 0; // white
901 }
902 }
903
904 int texturing_state = _texturing_state;
905 int texfilter_state = 0; // tnearest
906 if (texturing_state > 0) {
907 texfilter_state = _texfilter_state;
908
909 if (texturing_state < 3 &&
910 (_c->matrix_model_projection_no_w_transform || _filled_flat)) {
911 // Don't bother with the perspective-correct algorithm if we're under an
912 // orthonormal lens, e.g. render2d; or if RenderMode::M_filled_flat is
913 // in effect.
914 texturing_state = 1; // textured (not perspective correct)
915 }
916
917 if (_texture_replace) {
918 // If we're completely replacing the underlying color, then it doesn't
919 // matter what the color is.
920 shade_model_state = 0;
921 }
922 }
923
924 _c->zb_fill_tri = fill_tri_funcs[depth_write_state][color_write_state][alpha_test_state][depth_test_state][texfilter_state][shade_model_state][texturing_state];
925
926#ifdef DO_PSTATS
927 pixel_count_white_untextured = 0;
928 pixel_count_flat_untextured = 0;
929 pixel_count_smooth_untextured = 0;
930 pixel_count_white_textured = 0;
931 pixel_count_flat_textured = 0;
932 pixel_count_smooth_textured = 0;
933 pixel_count_white_perspective = 0;
934 pixel_count_flat_perspective = 0;
935 pixel_count_smooth_perspective = 0;
936 pixel_count_smooth_multitex2 = 0;
937 pixel_count_smooth_multitex3 = 0;
938#endif // DO_PSTATS
939
940 return true;
941}

1399bool TinyGraphicsStateGuardian::
1400framebuffer_copy_to_ram(Texture *tex, int view, int z,
1401 const DisplayRegion *dr, const RenderBuffer &rb,
1402 ScreenshotRequest *request) {
1403 nassertr(tex != nullptr && dr != nullptr, false);
1404
1405 int xo, yo, w, h;
1406 dr->get_region_pixels_i(xo, yo, w, h);
1407
1408 Texture::TextureType texture_type;
1409 int z_size;
1410 if (z >= 0) {
1411 texture_type = Texture::TT_cube_map;
1412 z_size = 6;
1413 } else {
1414 texture_type = Texture::TT_2d_texture;
1415 z_size = 1;
1416 }
1417
1418 Texture::ComponentType component_type = Texture::T_unsigned_byte;
1419 Texture::Format format = Texture::F_rgba;
1420 if (_current_properties->get_srgb_color()) {
1421 format = Texture::F_srgb_alpha;
1422 }
1423
1424 if (tex->get_x_size() != w || tex->get_y_size() != h ||
1425 tex->get_z_size() != z_size ||
1426 tex->get_component_type() != component_type ||
1427 tex->get_format() != format ||
1428 tex->get_texture_type() != texture_type) {
1429 // Re-setup the texture; its properties have changed.
1430 tex->setup_texture(texture_type, w, h, z_size,
1431 component_type, format);
1432 }
1433
1434 nassertr(z < tex->get_z_size(), false);
1435
1436 unsigned char *image_ptr = tex->modify_ram_image();
1437 size_t image_size = tex->get_ram_image_size();
1438 if (z >= 0 || view > 0) {
1439 image_size = tex->get_expected_ram_page_size();
1440 if (z >= 0) {
1441 image_ptr += z * image_size;
1442 }
1443 if (view > 0) {
1444 image_ptr += (view * tex->get_z_size()) * image_size;
1445 }
1446 }
1447
1448 PIXEL *ip = (PIXEL *)(image_ptr + image_size);
1449 PIXEL *fo = _c->zb->pbuf + xo + yo * _c->zb->linesize / PSZB;
1450 for (int y = 0; y < h; ++y) {
1451 ip -= w;
1452#ifndef WORDS_BIGENDIAN
1453 // On a little-endian machine, we can copy the whole row at a time.
1454 memcpy(ip, fo, w * PSZB);
1455#else
1456 // On a big-endian machine, we have to reverse the color-component order.
1457 const char *source = (const char *)fo;
1458 const char *stop = (const char *)fo + w * PSZB;
1459 char *dest = (char *)ip;
1460 while (source < stop) {
1461 char b = source[0];
1462 char g = source[1];
1463 char r = source[2];
1464 char a = source[3];
1465 dest[0] = a;
1466 dest[1] = r;
1467 dest[2] = g;
1468 dest[3] = b;
1469 dest += 4;
1470 source += 4;
1471 }
1472#endif
1473 fo += _c->zb->linesize / PSZB;
1474 }
1475
1476 if (request != nullptr) {
1477 request->finish();
1478 }
1479 return true;

2154 if (num_stages == 0) {
2155 // No texturing.
2156 return;
2157 }
2158 nassertv(num_stages <= MAX_TEXTURE_STAGES);
2159
2160 bool all_replace = true;
2161 bool all_nearest = true;
2162 bool all_mipmap_nearest = true;
2163 bool any_mipmap = false;
2164 bool needs_general = false;
2165 Texture::QualityLevel best_quality_level = Texture::QL_default;
2166
2167 for (int si = 0; si < num_stages; ++si) {
2168 TextureStage *stage = _target_texture->get_on_ff_stage(si);
2169 Texture *texture = _target_texture->get_on_texture(stage);
2170 nassertv(texture != nullptr);
2171
2172 int view = get_current_tex_view_offset() + stage->get_tex_view_offset();
2173 TextureContext *tc = texture->prepare_now(_prepared_objects, this);
2174 if (tc == nullptr) {
2175 // Something wrong with this texture; skip it.
2176 return;
2177 }
2178
2179 // Get the sampler state that we are supposed to use.
2180 const SamplerState &sampler = _target_texture->get_on_sampler(stage);
2181
2182 // Then, turn on the current texture mode.
2183 if (!update_texture(tc, view, false, si, sampler.uses_mipmaps())) {
2184 return;
2185 }
2186

2191 }

2197 if (quality_level == Texture::QL_default) {

2202
2203 ZTextureDef *texture_def = &_c->zb->current_textures[si];
2204

2206 // if we decide below we can inline the filters.
2207 SamplerState::FilterType minfilter = sampler.get_minfilter();
2208 SamplerState::FilterType magfilter = sampler.get_magfilter();
2209
2210 if (td_ignore_mipmaps && SamplerState::is_mipmap(minfilter)) {
2211 // Downgrade mipmaps.

2213 minfilter = SamplerState::FT_nearest;
2214 } else {
2215 minfilter = SamplerState::FT_linear;
2216 }
2217 }
2218
2219 // Depending on this particular texture's quality level, we may downgrade
2220 // the requested filters.
2221 if (quality_level == Texture::QL_fastest) {
2222 minfilter = SamplerState::FT_nearest;
2223 magfilter = SamplerState::FT_nearest;
2224
2225 } else if (quality_level == Texture::QL_normal) {
2226 if (SamplerState::is_mipmap(minfilter)) {
2227 minfilter = SamplerState::FT_nearest_mipmap_nearest;
2228 } else {
2229 minfilter = SamplerState::FT_nearest;
2230 }
2231 magfilter = SamplerState::FT_nearest;
2232

2240

2242 SamplerState::WrapMode wrap_v = sampler.get_wrap_v();

2252

2264
2265 texture_def->tex_wrap_u_func = get_tex_wrap_func(wrap_u);
2266 texture_def->tex_wrap_v_func = get_tex_wrap_func(wrap_v);
2267
2268 // The following special cases are handled inline, rather than relying
2269 // on the above wrap function pointers.
2270 if (wrap_u == SamplerState::WM_border_color && wrap_v == SamplerState::WM_border_color) {
2271 texture_def->tex_minfilter_func = apply_wrap_border_color_minfilter;
2272 texture_def->tex_magfilter_func = apply_wrap_border_color_magfilter;
2273 } else if (wrap_u == SamplerState::WM_clamp && wrap_v == SamplerState::WM_clamp) {
2274 texture_def->tex_minfilter_func = apply_wrap_clamp_minfilter;
2275 texture_def->tex_magfilter_func = apply_wrap_clamp_magfilter;
2276 }
2277 }
2278
2279 if (minfilter != SamplerState::FT_nearest || magfilter != SamplerState::FT_nearest) {
2280 all_nearest = false;
2281 }
2282
2283 if (minfilter != SamplerState::FT_nearest_mipmap_nearest ||
2284 magfilter != SamplerState::FT_nearest) {
2285 all_mipmap_nearest = false;
2286 }
2287
2288 if (SamplerState::is_mipmap(minfilter)) {
2289 any_mipmap = true;
2290 }
2291 }
2292
2293 // Set a few state cache values.
2294 _c->num_textures_enabled = num_stages;
2295 _texture_replace = all_replace;
2296
2297 _texturing_state = 2; // perspective (perspective-correct texturing)
2298 if (num_stages >= 3) {
2299 _texturing_state = 4; // multitex3
2300 } else if (num_stages == 2) {
2301 _texturing_state = 3; // multitex2
2302 } else if (!td_perspective_textures) {
2303 _texturing_state = 1; // textured (not perspective correct)
2304 }
2305
2306 if (best_quality_level == Texture::QL_best) {
2307 // This is the most generic texture filter. Slow, but pretty.
2308 _texfilter_state = 2; // tgeneral
2309
2310 if (!needs_general) {
2311 if (all_nearest) {
2312 // This case is inlined.
2313 _texfilter_state = 0; // tnearest
2314 } else if (all_mipmap_nearest) {
2315 // So is this case.
2316 _texfilter_state = 1; // tmipmap
2317 }
2318 }
2319
2320 } else if (best_quality_level == Texture::QL_fastest) {
2321 // This is the cheapest texture filter. We disallow mipmaps and
2322 // perspective correctness.
2323 _texfilter_state = 0; // tnearest
2324 _texturing_state = 1; // textured (not perspective correct, no multitexture)
2325
2326 } else {
2327 // This is the default texture filter. We use nearest sampling if there
2328 // are no mipmaps, and mipmap_nearest if there are any mipmaps--these are
2329 // the two inlined filters.

2331 if (any_mipmap) {
2332 _texfilter_state = 1; // tmipmap
2333 }
2334
2335 if (needs_general) {
2336 // To support nonstandard texcoord wrapping etc, we need to force the
2337 // general texfilter mode.
2338 _texfilter_state = 2; // tgeneral
2339 }
2340 }

2589setup_gltex(GLTexture *gltex, int x_size, int y_size, int num_views, int num_levels) {

2591 // A texture without pixels gets turned into a 1x1 texture.
2592 x_size = 1;
2593 y_size = 1;
2594 num_levels = 1;
2595 }

2597 int s_bits = get_tex_shift(x_size);

2599
2600 if (s_bits < 0 || t_bits < 0) {
2601 tinydisplay_cat.error()
2602 << "Texture size " << x_size << 'x' << y_size
2603 << " unsupported: dimensions must be power of two"
2604 << " and smaller than " << _max_texture_dimension << '\n';

2609 num_levels = min(num_levels, MAX_MIPMAP_LEVELS);
2610
2611 gltex->xsize = x_size;

2621 // number of individual alloc calls we have to make for each texture.

2627 int y = y_size;

2633 }
2634 }

2643 gltex->total_bytecount = total_bytecount;
2644 }
2645
2646 if (num_views != gltex->num_views) {
2647 if (gltex->views != nullptr) {
2648 TinyTextureContext::get_class_type().deallocate_array(gltex->views);
2649 }

2651 gltex->views = (ZTextureView *)TinyTextureContext::get_class_type().allocate_array(sizeof(ZTextureView) * num_views);
2652 gltex->num_views = num_views;
2653 }
2654
2655 char *next_buffer = (char *)gltex->allocated_buffer;
2656 char *end_of_buffer = next_buffer + total_bytecount;
2657
2658 for (int view = 0; view < num_views; ++view) {
2659 int level = 0;
2660 ZTextureLevel *dest = nullptr;
2661 while (level < num_levels) {
2662 dest = &gltex->views[view].levels[level];
2663 int bytecount = x_size * y_size * 4;
2664 dest->pixmap = (PIXEL *)next_buffer;
2665 next_buffer += bytecount;
2666 nassertr(next_buffer <= end_of_buffer, false);
2667
2668 dest->s_mask = ((1 << (s_bits + ZB_POINT_ST_FRAC_BITS)) - (1 << ZB_POINT_ST_FRAC_BITS)) << level;
2669 dest->t_mask = ((1 << (t_bits + ZB_POINT_ST_FRAC_BITS)) - (1 << ZB_POINT_ST_FRAC_BITS)) << level;
2670 dest->s_shift = (ZB_POINT_ST_FRAC_BITS + level);
2671 dest->t_shift = (ZB_POINT_ST_FRAC_BITS - s_bits + level);

2673 x_size = max((x_size >> 1), 1);
2674 y_size = max((y_size >> 1), 1);
2675 s_bits = max(s_bits - 1, 0);
2676 t_bits = max(t_bits - 1, 0);

2679 }

2682 // don't have to be concerned with running off the end of this array while
2683 // scanning out triangles.

2687 }
2688 }
2689
2690 return true;
