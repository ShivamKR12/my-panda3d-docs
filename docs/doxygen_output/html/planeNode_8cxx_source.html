140 // Normally, a PlaneNode is invisible. But if someone shows it, we will

168PT(Geom) PlaneNode::
169get_viz(CullTraverser *trav, CullTraverserData &data) {
170 CDLockedReader cdata(_cycler);
171
172 // Figure out whether we are looking at the front or the back of the plane.
173 const Lens *lens = trav->get_scene()->get_lens();
174 LPlane eye_plane = cdata->_plane * data.get_modelview_transform(trav)->get_mat();
175 bool front = (eye_plane.dist_to_plane(lens->get_nodal_point()) >= 0.0f);
176
177 if (cdata->_front_viz != nullptr) {
178 return front ? cdata->_front_viz : cdata->_back_viz;
179 }
180
181 if (pgraph_cat.is_debug()) {
182 pgraph_cat.debug()
183 << "Recomputing viz for " << *this << "\n";
184 }
185
186 CDWriter cdataw(_cycler, cdata, false);
187 const LPlane &plane = cdataw->_plane;
188
189 PT(GeomVertexData) vdata = new GeomVertexData
190 (get_name(), GeomVertexFormat::get_v3c(), Geom::UH_static);
191
192 GeomVertexWriter vertex(vdata, InternalName::get_vertex());
193 PT(GeomLines) lines = new GeomLines(Geom::UH_static);
194
195 LVector3 a, b;
196
197 if (fabs(plane[0]) > fabs(plane[1])) {
198 // X > Y
199 if (fabs(plane[0]) > fabs(plane[2])) {
200 // X > Y && X > Z. X is the largest.
201 a.set(0, 1, 0);
202 b.set(0, 0, 1);
203 } else {
204 // X > Y && Z > X. Z is the largest.
205 a.set(1, 0, 0);
206 b.set(0, 1, 0);
207 }
208 } else {
209 // Y > X
210 if (fabs(plane[1]) > fabs(plane[2])) {
211 // Y > X && Y > Z. Y is the largest.
212 a.set(1, 0, 0);
213 b.set(0, 0, 1);
214 } else {
215 // Y > X && Z > Y. Z is the largest.
216 a.set(1, 0, 0);
217 b.set(0, 1, 0);
218 }
219 }
220
221 static const int num_segs = 10;
222 a *= cdataw->_viz_scale / (num_segs * 2);
223 b *= cdataw->_viz_scale / (num_segs * 2);
224
225 for (int x = -num_segs; x <= num_segs; ++x) {
226 vertex.add_data3(plane.project(a * x - b * num_segs));
227 vertex.add_data3(plane.project(a * x + b * num_segs));
228 lines->add_next_vertices(2);
229 lines->close_primitive();
230 }
231 for (int y = -num_segs; y <= num_segs; ++y) {
232 vertex.add_data3(plane.project(b * y - a * num_segs));
233 vertex.add_data3(plane.project(b * y + a * num_segs));
234 lines->add_next_vertices(2);
235 lines->close_primitive();
236 }
237
238 cdataw->_front_viz = new Geom(vdata->set_color(LColor(1.0f, 1.0f, 0.0f, 1.0f)));
239 cdataw->_front_viz->add_primitive(lines);
240
241 cdataw->_back_viz = new Geom(vdata->set_color(LColor(0.4, 0.4, 0.0f, 1.0f)));
242 cdataw->_back_viz->add_primitive(lines);
243
244 return front ? cdataw->_front_viz : cdataw->_back_viz;
245}
