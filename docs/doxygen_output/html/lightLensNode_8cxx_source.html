112set_shadow_caster(bool caster, int buffer_xsize, int buffer_ysize, int buffer_sort) {
113 if ((_shadow_caster && !caster) || buffer_xsize != _sb_size[0] || buffer_ysize != _sb_size[1]) {
114 clear_shadow_buffers();

116 if (_shadow_caster != caster && _used_by_auto_shader) {
117 // Make sure any shaders using this light are regenerated.

120 _shadow_caster = caster;
121 _sb_size.set(buffer_xsize, buffer_ysize);

123 if (buffer_sort != _sb_sort) {

126 (*it).second->set_sort(buffer_sort);

128 _sb_sort = buffer_sort;

140void LightLensNode::
141clear_shadow_buffers() {
142 if (_shadow_map) {
143 // Clear it to all ones, so that any shaders that might still be using
144 // it will see the shadows being disabled.
145 _shadow_map->clear_image();
146 }
147
148 ShadowBuffers::iterator it;
149 for(it = _sbuffers.begin(); it != _sbuffers.end(); ++it) {
150 (*it).first->remove_window((*it).second);
151 }
152 _sbuffers.clear();
153}

158void LightLensNode::

160 if (_shadow_map != nullptr &&
161 _shadow_map->get_x_size() == _sb_size[0] &&
162 _shadow_map->get_y_size() == _sb_size[1]) {

167 if (_shadow_map == nullptr) {

172 _shadow_map->set_clear_color(LColor(1));

174 _shadow_map->set_wrap_v(SamplerState::WM_border_color);
175 _shadow_map->set_border_color(LColor(1));
176 _shadow_map->set_minfilter(SamplerState::FT_shadow);

191void LightLensNode::

193 // When it is removed from the last LightAttrib, destroy the shadow buffers.
194 // This is necessary to break the circular reference that the buffer holds
195 // on this node, via the display region's camera.
196 if (!AtomicAdjust::dec(_attrib_count)) {
