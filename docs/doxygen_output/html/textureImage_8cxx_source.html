75void TextureImage::
76assign_groups() {
77 if (_egg_files.empty()) {
78 // If we're not referenced by any egg files any more, assign us to no
79 // groups.
80 PaletteGroups empty;
81 assign_to_groups(empty);
82 return;
83 }
84
85 PaletteGroups definitely_in;
86
87 // First, we need to eliminate from consideration all the egg files that are
88 // already taken care of by the user's explicit group assignments for this
89 // texture.
90 WorkingEggs needed_eggs;
91
92 if (_explicitly_assigned_groups.empty()) {
93 // If we have no explicit group assignments, we must consider all the egg
94 // files.
95 std::copy(_egg_files.begin(), _egg_files.end(), std::back_inserter(needed_eggs));
96
97 } else {
98 // Otherwise, we only need to consider the egg files that don't have any
99 // groups in common with our explicit assignments.
100
101 EggFiles::const_iterator ei;
102 for (ei = _egg_files.begin(); ei != _egg_files.end(); ++ei) {
103 PaletteGroups intersect;
104 intersect.make_intersection(_explicitly_assigned_groups, (*ei)->get_complete_groups());
105 if (!intersect.empty()) {
106 // This egg file is satisfied by one of the texture's explicit
107 // assignments.
108
109 // We must use at least one of the explicitly-assigned groups that
110 // satisfied the egg file. We don't need to use all of them, however,
111 // and we choose the first one arbitrarily.
112 definitely_in.insert(*intersect.begin());
113
114 } else {
115 // This egg file was not satisfied by any of the texture's explicit
116 // assignments. Therefore, we'll need to choose some additional group
117 // to assign the texture to, to make the egg file happy. Defer this a
118 // bit.
119 needed_eggs.push_back(*ei);
120 }
121 }
122 }
123
124 while (!needed_eggs.empty()) {
125 // We need to know the complete set of groups that we need to consider
126 // adding the texture to. This is the union of all the egg files'
127 // requested groups.
128 PaletteGroups total;
129 WorkingEggs::const_iterator ei;
130 for (ei = needed_eggs.begin(); ei != needed_eggs.end(); ++ei) {
131 total.make_union(total, (*ei)->get_complete_groups());
132 }
133
134 // We don't count the "null" group for texture assignment.
135 total.remove_null();
136 if (total.empty()) {
137 break;
138 }
139
140 // Now, find the group that will satisfy the most egg files. If two
141 // groups satisfy the same number of egg files, choose (a) the most
142 // specific one, i.e. with the lowest dirname_level, or the lowest
143 // dependency_level if the dirname_levels are equal, and (b) the one that
144 // has the fewest egg files sharing it.
145 PaletteGroups::iterator gi = total.begin();
146 PaletteGroup *best = (*gi);
147 int best_egg_count = compute_egg_count(best, needed_eggs);
148 ++gi;
149 while (gi != total.end()) {
150 PaletteGroup *group = (*gi);
151
152 // Do we prefer this group to our current 'best'?
153 bool prefer_group = false;
154 int group_egg_count = compute_egg_count(group, needed_eggs);
155 if (group_egg_count != best_egg_count) {
156 prefer_group = (group_egg_count > best_egg_count);
157
158 } else {
159 prefer_group = group->is_preferred_over(*best);
160 }
161
162 if (prefer_group) {
163 best = group;
164 best_egg_count = group_egg_count;
165 }
166 ++gi;
167 }
168
169 // Okay, now we've picked the best group. Eliminate all the eggs from
170 // consideration that are satisfied by this group, and repeat.
171 definitely_in.insert(best);
172
173 WorkingEggs next_needed_eggs;
174 for (ei = needed_eggs.begin(); ei != needed_eggs.end(); ++ei) {
175 if ((*ei)->get_complete_groups().count(best) == 0) {
176 // This one wasn't eliminated.
177 next_needed_eggs.push_back(*ei);
178 }
179 }
180 needed_eggs.swap(next_needed_eggs);
181 }
182
183 // Finally, now that we've computed the set of groups we need to assign the
184 // texture to, we need to reconcile this with the set of groups we've
185 // assigned the texture to previously.
186 assign_to_groups(definitely_in);
187}

288void TextureImage::
289post_txa_file() {
290 _got_txa_file = true;
291
292 // First, get the actual size of the texture.
293 SourceTextureImage *source = get_preferred_source();
294 if (source != nullptr) {
295 if (source->get_size()) {
296 _size_known = true;
297 _x_size = source->get_x_size();
298 _y_size = source->get_y_size();
299 _properties.set_num_channels(source->get_num_channels());
300 }
301 }
302
303 // Now update this with a particularly requested size.
304 if (_request._got_size) {
305 _size_known = true;
306 _x_size = _request._x_size;
307 _y_size = _request._y_size;
308 }
309
310 if (_txa_wrap_u != _request._wrap_u ||
311 _txa_wrap_v != _request._wrap_v) {
312 _txa_wrap_u = _request._wrap_u;
313 _txa_wrap_v = _request._wrap_v;
314
315 // If the explicit wrap mode changes, we may need to regenerate the egg
316 // files, andor refill the palettes.
317 mark_eggs_stale();
318
319 Placement::iterator pi;
320 for (pi = _placement.begin(); pi != _placement.end(); ++pi) {
321 TexturePlacement *placement = (*pi).second;
322 placement->mark_unfilled();
323 }
324 }
325
326 if (_properties.has_num_channels() && !_request._keep_format) {
327 int num_channels = _properties.get_num_channels();
328 // Examine the image to determine if we can downgrade the number of
329 // channels, for instance from color to grayscale.
330 if (num_channels == 3 || num_channels == 4) {
331 consider_grayscale();
332 }
333
334 // Also consider the alpha properties, and whether we should downgrade
335 // from alpha to non-alpha.
336 if (num_channels == 2 || num_channels == 4) {
337 consider_alpha();
338 }
339 }
340
341 // However, if we got an explicit request for channels, honor that.
342 if (_request._got_num_channels) {
343 _properties.set_num_channels(_request._num_channels);
344 }
345
346 _properties._generic_format = _request._generic_format;
347 _properties._keep_format = _request._keep_format;
348
349 if (_request._format != EggTexture::F_unspecified) {
350 _properties._format = _request._format;
351 _properties._force_format = _request._force_format;
352 }
353
354 if (_request._minfilter != EggTexture::FT_unspecified) {
355 _properties._minfilter = _request._minfilter;
356 }
357 if (_request._magfilter != EggTexture::FT_unspecified) {
358 _properties._magfilter = _request._magfilter;
359 }
360
361 _properties._anisotropic_degree = _request._anisotropic_degree;
362
363 _properties._srgb = _request._srgb;
364
365 if (_properties._color_type == nullptr) {
366 _properties._color_type = _request._properties._color_type;
367 _properties._alpha_type = _request._properties._alpha_type;
368 }
369
370 // Finally, make sure our properties are fully defined.
371 _properties.fully_define();
372
373 // Now, if our properties have changed in all that from our previous
374 // session, we need to re-place ourself in all palette groups.
375 if (_properties != _pre_txa_properties) {
376 force_replace();
377
378 // The above will mark the egg files stale when the texture is palettized
379 // (since the UV's will certainly need to be recomputed), but sometimes we
380 // need to mark the egg files stale even when the texture is not
381 // palettized (if a critical property has changed). The following
382 // accomplishes this:
383 if (!_properties.egg_properties_match(_pre_txa_properties)) {
384 mark_eggs_stale();
385 }
386 }
387
388 // The alpha mode isn't stored in the properties, because it doesn't affect
389 // which textures may be associated into a common palette.
390 if (_request._alpha_mode != EggRenderMode::AM_unspecified) {
391 _alpha_mode = _request._alpha_mode;
392 }
393
394 // On the other hand, if we don't have an alpha channel, we shouldn't have
395 // an alpha mode.
396 if (_properties.has_num_channels()) {
397 int num_channels = _properties.get_num_channels();
398 if (num_channels == 1 || num_channels == 3) {
399 _alpha_mode = EggRenderMode::AM_unspecified;
400 }
401 }
402
403 // If we've changed the alpha mode, we should also mark the eggs stale.
404 if (_pre_txa_alpha_mode != _alpha_mode) {
405 mark_eggs_stale();
406 }
407}

548SourceTextureImage *TextureImage::
549get_preferred_source() {
550 if (_preferred_source != nullptr) {
551 return _preferred_source;
552 }
553
554 // Now examine all of the various source images available to us and pick the
555 // most suitable. We base this on the following criteria:
556
557 // (1) A suitable source image must be referenced by at least one egg file,
558 // unless no source images are referenced by any egg file.
559
560 // (2) A larger source image is preferable to a smaller one.
561
562 // (3) Given two source images of the same size, the more recent one is
563 // preferable.
564
565 // Are any source images referenced by an egg file?
566
567 bool any_referenced = false;
568 Sources::iterator si;
569 for (si = _sources.begin(); si != _sources.end() && !any_referenced; ++si) {
570 SourceTextureImage *source = (*si).second;
571 if (source->get_egg_count() > 0) {
572 any_referenced = true;
573 }
574 }
575
576 SourceTextureImage *best = nullptr;
577 int best_size = 0;
578
579 for (si = _sources.begin(); si != _sources.end(); ++si) {
580 SourceTextureImage *source = (*si).second;
581
582 if (source->get_egg_count() > 0 || !any_referenced) {
583 // Rule (1) passes.
584
585 if (source->exists() && source->get_size()) {
586 int source_size = source->get_x_size() * source->get_y_size();
587 if (best == nullptr) {
588 best = source;
589 best_size = source_size;
590
591 } else if (source_size > best_size) {
592 // Rule (2) passes.
593 best = source;
594 best_size = source_size;
595
596 } else if (source_size == best_size &&
597 source->get_filename().compare_timestamps(best->get_filename()) > 0) {
598 // Rule (3) passes.
599 best = source;
600 best_size = source_size;
601 }
602 }
603 }
604 }
605
606 if (best == nullptr && !_sources.empty()) {
607 // If we didn't pick any that pass, it must be that all of them are
608 // unreadable. In this case, it really doesn't matter which one we pick,
609 // but we should at least pick one that has an egg reference, if any of
610 // them do.
611 if (any_referenced) {
612 for (si = _sources.begin();
613 si != _sources.end() && best == nullptr;
614 ++si) {
615 SourceTextureImage *source = (*si).second;
616 if (source->get_egg_count() > 0) {
617 best = source;
618 }
619 }
620 } else {
621 best = (*_sources.begin()).second;
622 }
623 }
624
625 _preferred_source = best;
626 return _preferred_source;
627}

650void TextureImage::
651copy_unplaced(bool redo_all) {
652 // First, we need to build up the set of DestTextureImages that represents
653 // the files we need to generate.
654 Dests generate;
655
656 // Go through all the TexturePlacements and note the ones for which we're
657 // unplaced. We check get_omit_reason() and not is_placed(), because we
658 // want to consider solitary images to be unplaced in this case.
659 Placement::iterator pi;
660 for (pi = _placement.begin(); pi != _placement.end(); ++pi) {
661 TexturePlacement *placement = (*pi).second;
662 if (placement->get_omit_reason() != OR_none &&
663 placement->get_omit_reason() != OR_unknown) {
664 DestTextureImage *dest = new DestTextureImage(placement);
665 Filename filename = dest->get_filename();
666 FilenameUnifier::make_canonical(filename);
667
668 std::pair<Dests::iterator, bool> insert_result = generate.insert
669 (Dests::value_type(filename, dest));
670 if (!insert_result.second) {
671 // At least two DestTextureImages map to the same filename, no sweat.
672 delete dest;
673 dest = (*insert_result.first).second;
674 }
675
676 placement->set_dest(dest);
677
678 } else {
679 placement->set_dest(nullptr);
680 }
681 }
682
683 if (redo_all) {
684 // If we're redoing everything, we remove everything first and then recopy
685 // it again.
686 Dests empty;
687 remove_old_dests(empty, _dests);
688 copy_new_dests(generate, empty);
689
690 } else {
691 // Otherwise, we only remove and recopy the things that changed between
692 // this time and last time.
693 remove_old_dests(generate, _dests);
694 copy_new_dests(generate, _dests);
695 }
696
697 // Clean up the old set.
698 Dests::iterator di;
699 for (di = _dests.begin(); di != _dests.end(); ++di) {
700 delete (*di).second;
701 }
702
703 _dests.swap(generate);
704}

840 }

842 if (!_explicitly_assigned_groups.empty()) {

845 }

849 << "Not used.\n";

860write_scale_info(std::ostream &out, int indent_level) {
861 SourceTextureImage *source = get_preferred_source();
862 indent(out, indent_level) << get_name();
863
864 // Write the list of groups we're placed in.
865 if (_placement.empty()) {
866 out << " (not used)";
867 } else {
868 Placement::const_iterator pi;
869 pi = _placement.begin();
870 out << " (" << (*pi).second->get_group()->get_name();
871 ++pi;
872 while (pi != _placement.end()) {
873 out << " " << (*pi).second->get_group()->get_name();
874 ++pi;
875 }
876 out << ")";
877 }
878
879 out << " orig ";
880
881 if (source == nullptr ||
882 !source->is_size_known()) {
883 out << "unknown";
884 } else {
885 out << source->get_x_size() << " " << source->get_y_size()
886 << " " << source->get_num_channels();
887 }
888
889 if (!_placement.empty() && is_size_known()) {
890 out << " new " << get_x_size() << " " << get_y_size()
891 << " " << get_num_channels();
892
893 if (source != nullptr &&
894 source->is_size_known()) {
895 double scale =
896 100.0 * (((double)get_x_size() / (double)source->get_x_size()) +
897 ((double)get_y_size() / (double)source->get_y_size())) / 2.0;
898 out << " scale " << scale << "%";
899 }
900 }
901 out << "\n";
902
903 // Also cross-reference the placed and unplaced information.
904 Placement::iterator pi;
905 for (pi = _placement.begin(); pi != _placement.end(); ++pi) {
906 TexturePlacement *placement = (*pi).second;
907 if (placement->get_omit_reason() == OR_none) {
908 PaletteImage *image = placement->get_image();
909 nassertv(image != nullptr);
910 indent(out, indent_level + 2)
911 << "placed on "
912 << FilenameUnifier::make_user_filename(image->get_filename())
913 << "\n";
914
915 } else if (placement->get_omit_reason() == OR_unknown) {
916 indent(out, indent_level + 2)
917 << "not placed because unknown.\n";
918
919 } else {
920 DestTextureImage *image = placement->get_dest();

1032void TextureImage::
1033consider_grayscale() {
1034 // Since this isn't likely to change for a particular texture after its
1035 // creation, we save a bit of time by not performing this check unless this
1036 // is the first time we've ever seen this texture. This will save us from
1037 // having to load the texture images each time we look at them. On the
1038 // other hand, if we've already loaded up the image, then go ahead.
1039 if (!_read_source_image && _ever_read_image) {
1040 if (_forced_grayscale) {
1041 _properties.force_grayscale();
1042 }
1043 return;
1044 }
1045
1046 const PNMImage &source = read_source_image();
1047 if (!source.is_valid()) {
1048 return;
1049 }
1050
1051 for (int y = 0; y < source.get_y_size(); y++) {
1052 for (int x = 0; x < source.get_x_size(); x++) {
1053 const xel &v = source.get_xel_val(x, y);
1054 if (PPM_GETR(v) != PPM_GETG(v) || PPM_GETR(v) != PPM_GETB(v)) {
1055 // Here's a colored pixel. We can't go grayscale.
1056 _forced_grayscale = false;
1057 return;
1058 }
1059 }
1060 }
1061
1062 // All pixels in the image were grayscale!
1063 _properties.force_grayscale();
1064 _forced_grayscale = true;
1065}

1071void TextureImage::
1072consider_alpha() {
1073 // As above, we don't bother doing this if we've already done this in a
1074 // previous session.
1075
1076 // _alpha_bits == -1 indicates we have read an older textures.boo file that
1077 // didn't define these bits.
1078 if (_read_source_image || !_ever_read_image || _alpha_bits == -1) {
1079 _alpha_bits = 0;
1080 int num_mid_pixels = 0;
1081
1082 const PNMImage &source = read_source_image();
1083 if (source.is_valid() && source.has_alpha()) {
1084 xelval maxval = source.get_maxval();
1085 for (int y = 0; y < source.get_y_size(); y++) {
1086 for (int x = 0; x < source.get_x_size(); x++) {
1087 xelval alpha_val = source.get_alpha_val(x, y);
1088 if (alpha_val == 0) {
1089 _alpha_bits |= AB_zero;
1090 } else if (alpha_val == maxval) {
1091 _alpha_bits |= AB_one;
1092 } else {
1093 _alpha_bits |= AB_mid;
1094 ++num_mid_pixels;
1095 }
1096 }
1097 }
1098 }
1099
1100 int num_pixels = source.get_x_size() * source.get_y_size();
1101 _mid_pixel_ratio = 0.0;
1102 if (num_pixels != 0) {
1103 _mid_pixel_ratio = (double)num_mid_pixels / (double)num_pixels;
1104 }
1105 }
1106
1107 _is_cutout = false;
1108
1109 if (_alpha_bits != 0) {
1110 if (_alpha_bits == AB_one) {
1111 // All alpha pixels are white; drop the alpha channel.
1112 _properties.force_nonalpha();
1113
1114 } else if (_alpha_bits == AB_zero) {
1115 // All alpha pixels are invisible; this is probably a mistake. Drop the
1116 // alpha channel and complain.
1117 _properties.force_nonalpha();
1118 if (_read_source_image) {
1119 nout << *this << " has an all-zero alpha channel; dropping alpha.\n";
1120 }
1121
1122 } else if (_alpha_mode == EggRenderMode::AM_unspecified) {
1123 // Consider fiddling with the alpha mode, if the user hasn't specified a
1124 // particular alpha mode in the txa file.
1125 if ((_alpha_bits & AB_mid) == 0) {
1126 // No middle range bits: a binary alpha image.
1127 _alpha_mode = EggRenderMode::AM_binary;
1128
1129 } else if ((_alpha_bits & AB_one) != 0 && _mid_pixel_ratio < pal->_cutout_ratio) {
1130 // At least some opaque bits, and relatively few middle range bits: a
1131 // cutout image.
1132 _alpha_mode = pal->_cutout_mode;
1133 _is_cutout = true;
1134
1135 } else {
1136 // No opaque bits; just use regular alpha blending.
1137 _alpha_mode = EggRenderMode::AM_blend;
1138 }
1139 }
1140 }
1141}
