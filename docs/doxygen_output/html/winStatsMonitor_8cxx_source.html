620void WinStatsMonitor::
621update_status_bar() {
622 const PStatClientData *client_data = get_client_data();
623 if (client_data == nullptr) {
624 return;
625 }
626
627 const PStatThreadData *thread_data = get_client_data()->get_thread_data(0);
628 if (thread_data == nullptr || thread_data->is_empty()) {
629 return;
630 }
631 int frame_number = thread_data->get_latest_frame_number();
632 const PStatFrameData &frame_data = thread_data->get_latest_frame();
633
634 // Gather the top-level collector list.
635 pvector<std::string> parts;
636 pvector<int> collectors;
637 size_t total_chars = 0;
638
639 int num_toplevel_collectors = client_data->get_num_toplevel_collectors();
640 for (int tc = 0; tc < num_toplevel_collectors; tc++) {
641 int collector = client_data->get_toplevel_collector(tc);
642 if (client_data->has_collector(collector) &&
643 client_data->get_collector_has_level(collector, 0)) {
644 PStatView &view = get_level_view(collector, 0);
645 view.set_to_frame(frame_data);
646 double value = view.get_net_value();
647 if (value == 0.0) {
648 // Don't include it unless we've included it before.
649 if (std::find(_status_bar_collectors.begin(), _status_bar_collectors.end(), collector) == _status_bar_collectors.end()) {
650 continue;
651 }
652 }
653
654 // Add the value for other threads that have this collector.
655 for (int thread_index = 1; thread_index < client_data->get_num_threads(); ++thread_index) {
656 PStatView &view = get_level_view(collector, thread_index);
657 view.set_to_frame(frame_number);
658 value += view.get_net_value();
659 }
660
661 const PStatCollectorDef &def = client_data->get_collector_def(collector);
662 std::string text = "\t" + def._name;
663 text += ": " + PStatGraph::format_number(value, PStatGraph::GBU_named | PStatGraph::GBU_show_units, def._level_units);
664 total_chars += text.size();
665 parts.push_back(text);
666 collectors.push_back(collector);
667 }
668 }
669
670 size_t cur_size = _status_bar_collectors.size();
671 _status_bar_collectors = std::move(collectors);
672
673 // Allocate an array for holding the right edge coordinates.
674 HLOCAL hloc = LocalAlloc(LHND, sizeof(int) * (parts.size() + 1));
675 PINT sizes = (PINT)LocalLock(hloc);
676
677 int pixel_scale = get_pixel_scale();
678
679 // Allocate the left-most slot for the framerate indicator.
680 double offset = 28.0 * pixel_scale;
681 sizes[0] = (int)(offset + 0.5);
682
683 if (!parts.empty()) {
684 // Distribute the sizes roughly based on the number of characters. It's not
685 // as good as measuring the text, but it's good enough.
686 RECT rect;
687 double width_per_char = 0;
688 GetClientRect(_status_bar, &rect);
689 // Leave room for the grip.
690 rect.right -= pixel_scale * 4;
691 width_per_char = (rect.right - rect.left - offset) / (double)total_chars;
692
693 // If we get below a minimum width, start chopping parts off.
694 while (!parts.empty() && width_per_char < pixel_scale * 1.5) {
695 total_chars -= parts.back().size();
696 parts.pop_back();
697 width_per_char = (rect.right - rect.left - offset) / (double)total_chars;
698 }
699
700 if (!parts.empty()) {
701 for (size_t i = 0; i < parts.size(); ++i) {
702 offset += (parts[i].size()) * width_per_char;
703 sizes[i + 1] = (int)(offset + 0.5);
704 }
705 } else {
706 // No room for any collectors; the framerate can take up the whole width.
707 sizes[0] = rect.right - rect.left;
708 }
709 }
710
711 SendMessage(_status_bar, SB_SETPARTS, (WPARAM)(parts.size() + 1), (LPARAM)sizes);
712
713 LocalUnlock(hloc);
714 LocalFree(hloc);
715
716 for (size_t i = 0; i < parts.size(); ++i) {
717 SendMessage(_status_bar, SB_SETTEXT, i + 1, (LPARAM)parts[i].c_str());
718 }
719}

724void WinStatsMonitor::
725show_popup_menu(int collector) {
726 POINT point;
727 if (!GetCursorPos(&point)) {
728 return;
729 }
730
731 const PStatClientData *client_data = get_client_data();
732 if (client_data == nullptr) {
733 return;
734 }
735
736 PStatView &level_view = get_level_view(collector, 0);
737 const PStatViewLevel *view_level = level_view.get_top_level();
738 int num_children = view_level->get_num_children();
739 if (num_children == 0) {
740 return;
741 }
742
743 HMENU popup = CreatePopupMenu();
744
745 // Reverse the order since the menus are listed from the top down; we want
746 // to be visually consistent with the graphs, which list these labels from
747 // the bottom up.
748 for (int c = num_children - 1; c >= 0; c--) {
749 const PStatViewLevel *child_level = view_level->get_child(c);
750
751 int child_collector = child_level->get_collector();
752 MenuDef menu_def(0, child_collector, CT_strip_chart, true);
753 int menu_id = get_menu_id(menu_def);
754
755 double value = child_level->get_net_value();
756
757 const PStatCollectorDef &def = client_data->get_collector_def(child_collector);
758 std::string text = def._name;
759 text += ": " + PStatGraph::format_number(value, PStatGraph::GBU_named | PStatGraph::GBU_show_units, def._level_units);
760 AppendMenu(popup, MF_STRING, menu_id, text.c_str());
761 }
762
763 TrackPopupMenu(popup, TPM_LEFTBUTTON, point.x, point.y, 0, _window, nullptr);
764}

769void WinStatsMonitor::
770calc_iconic_graph_window_pos(WinStatsGraph *moved_graph, int &x, int &y) {
771 MINIMIZEDMETRICS metrics;
772 metrics.cbSize = sizeof(metrics);
773 SystemParametersInfoA(SPI_GETMINIMIZEDMETRICS, 0, &metrics, 0);
774
775 int height = GetThemeSysSize(nullptr, SM_CYSIZE) + GetThemeSysSize(nullptr, SM_CXPADDEDBORDER) * 2;
776
777 RECT client_rect;
778 GetClientRect(_window, &client_rect);
779 MapWindowPoints(_window, nullptr, (POINT *)&client_rect, 2);
780
781 // Remove the status bar from the client rectangle.
782 RECT status_bar_rect;
783 GetWindowRect(_status_bar, &status_bar_rect);
784 client_rect.bottom -= (status_bar_rect.bottom - status_bar_rect.top);
785
786 int iconic_offset = 0;
787
788 for (WinStatsGraph *graph : _graphs) {
789 RECT child_rect;
790 HWND window = graph->get_window();
791 if (graph == moved_graph) {
792 x = client_rect.left + iconic_offset;
793 y = client_rect.bottom - height;
794 iconic_offset += metrics.iWidth + metrics.iHorzGap;
795 }
796 else if (IsIconic(window) && GetWindowRect(window, &child_rect)) {
797 // Keep it glued to the bottom-left corner of the parent window.
798 child_rect.left = client_rect.left + iconic_offset;
799 child_rect.top = client_rect.bottom - height;
800 iconic_offset += metrics.iWidth + metrics.iHorzGap;
801
802 SetWindowPos(window, 0, child_rect.left, child_rect.top, 0, 0,
803 SWP_NOOWNERZORDER | SWP_NOZORDER | SWP_NOSIZE | SWP_NOACTIVATE | SWP_NOSENDCHANGING);
804 }

811void WinStatsMonitor::
812handle_window_moved(const RECT &client_rect, int delta_x, int delta_y) {
813 if (_graphs.empty()) {
814 return;
815 }
816
817 MINIMIZEDMETRICS metrics;
818 metrics.cbSize = sizeof(metrics);
819 SystemParametersInfoA(SPI_GETMINIMIZEDMETRICS, 0, &metrics, 0);
820
821 int height = GetThemeSysSize(nullptr, SM_CYSIZE) + GetThemeSysSize(nullptr, SM_CXPADDEDBORDER) * 2;
822
823 int iconic_offset = 0;
824
825 for (WinStatsGraph *graph : _graphs) {
826 WINDOWPLACEMENT wp;
827 HWND window = graph->get_window();
828 if (GetWindowPlacement(window, &wp)) {

830 // Keep the minimized window title bar glued to the bottom-left
831 // corner of the parent window.

834 wp.ptMinPosition.y = client_rect.bottom - height;
835 iconic_offset += metrics.iWidth + metrics.iHorzGap;
836 }
837
838 // Move the "restored" window (even when it's currently min/maximized).
839 wp.rcNormalPosition.left += delta_x;
840 wp.rcNormalPosition.top += delta_y;
841 wp.rcNormalPosition.right += delta_x;
842 wp.rcNormalPosition.bottom += delta_y;
843 SetWindowPlacement(window, &wp);
844 }
