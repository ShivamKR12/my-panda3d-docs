100void AsyncFuture::
101wait() {
102 if (!done()) {
103 PStatTimer timer(AsyncTaskChain::_wait_pcollector);
104 if (task_cat.is_debug()) {
105 task_cat.debug()
106 << "Waiting for future " << *this << "\n";
107 }
108
109 FutureState state = get_future_state();
110 while (state < FS_finished) {
111 if (state == FS_locked_pending) {
112 // If it's locked, someone is probably about to finish it, so don't
113 // go to sleep.
114 do {
115 Thread::relax();
116 state = get_future_state();
117 }
118 while (state == FS_locked_pending);
119
120 if (state >= FS_finished) {
121 return;
122 }
123 }
124
125 // Go to sleep.
126 _future_state.wait(state, std::memory_order_relaxed);
127 state = get_future_state();
128 }
129 }
130
131 // Let's make wait() an acquire op, so that people can use a future for
132 // synchronization.
133 patomic_thread_fence(std::memory_order_acquire);
134}

141void AsyncFuture::
142wait(double timeout) {
143 if (!done()) {
144 PStatTimer timer(AsyncTaskChain::_wait_pcollector);
145 if (task_cat.is_debug()) {
146 task_cat.debug()
147 << "Waiting up to " << timeout << " seconds for future " << *this << "\n";
148 }
149
150 // Continue to yield while the future isn't done. It may be more efficient
151 // to use a condition variable, but let's not add the extra complexity
152 // unless we're sure that we need it.
153 TrueClock *clock = TrueClock::get_global_ptr();
154 double end = clock->get_short_time() + timeout;
155 do {
156 Thread::relax();
157 }
158 while (!done() && clock->get_short_time() < end);
159
160 patomic_thread_fence(std::memory_order_acquire);
161 }

169void AsyncFuture::
170notify_done(bool clean_exit) {
171 patomic_thread_fence(std::memory_order_acquire);
172 nassertv(done());
173
174 // Let any calls to wait() know that we're done.
175 _future_state.notify_all();
176
177 // This will only be called by the thread that managed to set the
178 // _future_state away from the "pending" state, so this is thread safe.
179
180 // Go through the futures that are waiting for this to finish.
181 for (AsyncFuture *fut : _waiting) {
182 if (fut->is_task()) {
183 // It's a task. Make it active again.
184 wake_task((AsyncTask *)fut);
185 }
186 else if (fut->get_type() == AsyncGatheringFuture::get_class_type()) {
187 // It's a gathering future. Decrease the pending count on it, and if
188 // we're the last one, call notify_done() on it.
189 AsyncGatheringFuture *gather = (AsyncGatheringFuture *)fut;
190 if (gather->_num_pending.fetch_sub(1, std::memory_order_relaxed) == 1) {
191 if (gather->set_future_state(FS_finished)) {
192 gather->notify_done(true);
193 }
194 }
195 }
196 else {
197 // It's a shielding future. The shielding only protects the inner future
198 // when the outer is cancelled, not the other way around, so we have to
199 // propagate any cancellation here as well.
200 if (clean_exit && _result != nullptr) {
201 // Propagate the result, if any.
202 if (fut->try_lock_pending()) {
203 fut->_result = _result;
204 fut->_result_ref = _result_ref;
205 fut->unlock(FS_finished);
206 fut->notify_done(true);
207 }
208 }
209 else if (fut->set_future_state(clean_exit ? FS_finished : FS_cancelled)) {
210 fut->notify_done(clean_exit);
211 }
212 }
213 }
214 _waiting.clear();
215
216 // For historical reasons, we don't send the "done event" if the future was
217 // cancelled.
218 if (clean_exit && !_done_event.empty()) {
219 PT_Event event = new Event(_done_event);
220 event->add_parameter(EventParameter(this));
221 throw_event(std::move(event));
222 }
223}

250void AsyncFuture::
251set_result(TypedObject *ptr, ReferenceCount *ref_ptr) {
252 // We don't strictly need to lock the future since only one thread is
253 // allowed to call set_result(), but we might as well.
254 patomic_unsigned_lock_free::value_type orig_state = FS_pending;
255 if (!_future_state.compare_exchange_strong(orig_state, FS_locked_pending,
256 std::memory_order_relaxed)) {
257#if defined(HAVE_THREADS) && !defined(SIMPLE_THREADS)
258 while (orig_state == FS_locked_pending) {
259 Thread::relax();
260 orig_state = FS_pending;
261 _future_state.compare_exchange_weak(orig_state, FS_locked_pending,
262 std::memory_order_relaxed);
263 }
264#else
265 // We can't lose control between now and calling unlock() if we're using a
266 // cooperative threading model.
267 nassertv(false);
268#endif
269 }
270
271 if (orig_state == FS_pending) {
272 _result = ptr;
273 _result_ref = ref_ptr;
274 unlock(FS_finished);
275
276 // OK, now our thread owns the _waiting vector et al.
277 notify_done(true);
278 }
279 else if (orig_state == FS_cancelled) {
280 // This was originally illegal, but there is a chance that the future was
281 // cancelled while another thread was setting the result. So, we drop
282 // this, but we can issue a warning.
283 task_cat.warning()
284 << "Ignoring set_result() called on cancelled " << *this << "\n";
285 }
286 else {
287 task_cat.error()
288 << "set_result() was called on finished " << *this << "\n";
289 }
290}

299bool AsyncFuture::
300add_waiting_task(AsyncTask *task) {
301 nassertr(task->is_runnable(), false);
302
303 // We have to make sure we're not going to change state while we're in the
304 // process of adding the task.
305 if (try_lock_pending()) {
306 if (_manager == nullptr) {
307 _manager = task->_manager;
308 }
309
310 _waiting.push_back(task);
311
312 // Unlock the state.
313 unlock();
314 nassertr(task->_manager == nullptr || task->_manager == _manager, true);
315 return true;
316 } else {
317 // It's already done. Wake the task immediately.
318 wake_task(task);
319 return false;

326void AsyncFuture::
327wake_task(AsyncTask *task) {
328 AsyncTaskManager *manager = task->_manager;
329 if (manager == nullptr) {
330 // If it's an unscheduled task, schedule it on the same manager as the
331 // rest of the waiting tasks.
332 manager = _manager;
333 if (manager == nullptr) {
334 manager = AsyncTaskManager::get_global_ptr();
335 }
336 }
337
338 MutexHolder holder(manager->_lock);
339 switch (task->_state) {
340 case AsyncTask::S_servicing_removed:
341 nassertv(task->_manager == _manager);
342 // Re-adding a self-removed task; this just means clearing the removed
343 // flag.
344 task->_state = AsyncTask::S_servicing;
345 return;
346
347 case AsyncTask::S_active:
348 // It could have already been activated, such as by a cancel() which then
349 // indirectly caused the awaiting future to be cancelled. Do nothing.
350 return;
351
352 case AsyncTask::S_inactive:
353 // Schedule it immediately.
354 nassertv(task->_manager == nullptr);
355
356 if (task_cat.is_debug()) {
357 task_cat.debug()
358 << "Adding " << *task << " (woken by future " << *this << ")\n";
359 }
360
361 {
362 manager->_lock.unlock();
363 task->upon_birth(manager);
364 manager->_lock.lock();
365 nassertv(task->_manager == nullptr &&
366 task->_state == AsyncTask::S_inactive);
367
368 AsyncTaskChain *chain = manager->do_find_task_chain(task->_chain_name);
369 if (chain == nullptr) {
370 task_cat.warning()
371 << "Creating implicit AsyncTaskChain " << task->_chain_name
372 << " for " << manager->get_type() << " " << manager->get_name() << "\n";
373 chain = manager->do_make_task_chain(task->_chain_name);
374 }
375 chain->do_add(task);
376 }
377 return;
378
379 case AsyncTask::S_awaiting:
380 nassertv(task->_manager == _manager);
381 task->_state = AsyncTask::S_active;
382 task->_chain->_active.push_back(task);
383 --task->_chain->_num_awaiting_tasks;
384 return;
385
386 default:
387 nassert_raise("unexpected task state");
388 return;
389 }

406AsyncGatheringFuture::
407AsyncGatheringFuture(AsyncFuture::Futures futures) :
408 _futures(std::move(futures)),
409 _num_pending(0) {
410
411 bool any_pending = false;
412
413 for (AsyncFuture *fut : _futures) {
414 // If this returns true, the future is not yet done and we need to
415 // register ourselves with it. This creates a circular reference, but it
416 // is resolved when the future is completed or cancelled.
417 if (fut->try_lock_pending()) {
418 if (_manager == nullptr) {
419 _manager = fut->_manager;
420 }
421 fut->_waiting.push_back((AsyncFuture *)this);
422 _num_pending.fetch_add(1, std::memory_order_relaxed);
423 fut->unlock();
424 any_pending = true;
425 }
426 }
427 if (!any_pending) {
428 // Start in the done state if all the futures we were passed are done.
429 // Note that it is only safe to set this member in this manner if indeed
430 // no other future holds a reference to us.
431 _future_state.store(FS_finished, std::memory_order_relaxed);
432 }
433}

441bool AsyncGatheringFuture::
442cancel() {
443 if (!done()) {
444 if (task_cat.is_debug()) {
445 task_cat.debug()
446 << "Cancelling AsyncGatheringFuture (" << _futures.size() << " futures)\n";
447 }
448
449 // Temporarily increase the pending count so that the notify_done()
450 // callbacks won't end up causing it to be set to "finished".
451 _num_pending.fetch_add(1, std::memory_order_relaxed);
452
453 bool any_cancelled = false;
454 for (AsyncFuture *fut : _futures) {
455 if (fut->cancel()) {
456 any_cancelled = true;
457 }
458 }
459
460 // If all the futures were cancelled, change state of this future to
461 // "cancelled" and call the notify_done() callbacks.
462 if (_num_pending.fetch_sub(1, std::memory_order_relaxed) == 1) {
463 if (set_future_state(FS_cancelled)) {
464 notify_done(false);
465 }
466 }
467
468 return any_cancelled;
469 } else {
470 return false;
471 }
