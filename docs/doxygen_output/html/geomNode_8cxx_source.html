103void GeomNode::
104apply_attribs_to_vertices(const AccumulatedAttribs &attribs, int attrib_types,
105 GeomTransformer &transformer) {
106 if (pgraph_cat.is_debug()) {
107 pgraph_cat.debug()
108 << "Transforming geometry:\n";
109 attribs.write(pgraph_cat.debug(false), attrib_types, 2);
110 }
111
112 if ((attrib_types & SceneGraphReducer::TT_transform) != 0) {
113 if (!attribs._transform->is_identity()) {
114 transformer.transform_vertices(this, attribs._transform->get_mat());
115 }
116 }
117
118 Thread *current_thread = Thread::get_current_thread();
119 OPEN_ITERATE_CURRENT_AND_UPSTREAM(_cycler, current_thread) {
120 CDStageWriter cdata(_cycler, pipeline_stage, current_thread);
121 PT(GeomList) geoms = cdata->modify_geoms();
122
123 // Iterate based on the number of geoms, not using STL iterators. This
124 // allows us to append to the list in the code below (which we might do
125 // when doublesiding polys) without visiting those new nodes during the
126 // traversal.
127 size_t num_geoms = geoms->size();
128 for (size_t i = 0; i < num_geoms; ++i) {
129 GeomEntry *entry = &(*geoms)[i];
130 PT(Geom) new_geom = entry->_geom.get_read_pointer()->make_copy();
131
132 AccumulatedAttribs geom_attribs = attribs;
133 entry->_state = geom_attribs.collect(entry->_state, attrib_types);
134
135 bool any_changed = false;
136
137 if ((attrib_types & SceneGraphReducer::TT_color) != 0) {
138 CPT(RenderAttrib) ra = geom_attribs._color;
139 if (ra != nullptr) {
140 int override = geom_attribs._color_override;
141 entry->_state = entry->_state->add_attrib(ra, override);
142 }
143
144 ra = entry->_state->get_attrib_def(ColorAttrib::get_class_slot());
145 CPT (ColorAttrib) ca = DCAST(ColorAttrib, ra);
146 if (ca->get_color_type() != ColorAttrib::T_vertex) {
147 if(allow_flatten_color) {
148 if(transformer.set_color(new_geom, ca->get_color())) {
149 any_changed = true;
150 entry->_state = entry->_state->set_attrib(ColorAttrib::make_vertex());
151 }
152 } else {
153 if (transformer.remove_column(new_geom, InternalName::get_color())) {
154 any_changed = true;
155 }
156 }
157 }
158 }
159 if ((attrib_types & SceneGraphReducer::TT_color_scale) != 0) {
160 if (geom_attribs._color_scale != nullptr) {
161 CPT(ColorScaleAttrib) csa = DCAST(ColorScaleAttrib, geom_attribs._color_scale);
162 if (csa->get_scale() != LVecBase4(1.0f, 1.0f, 1.0f, 1.0f)) {
163
164
165 // Now, if we have an "off" or "flat" color attribute, we simply
166 // modify the color attribute, and leave the vertices alone.
167 CPT(RenderAttrib) ra = entry->_state->get_attrib_def(ColorAttrib::get_class_slot());
168 CPT(ColorAttrib) ca = DCAST(ColorAttrib, ra);
169 if(allow_flatten_color) {
170 if (transformer.transform_colors(new_geom, csa->get_scale())) {
171 any_changed = true;
172 }
173 } else {
174 if (ca->get_color_type() == ColorAttrib::T_off) {
175 entry->_state = entry->_state->set_attrib(ColorAttrib::make_vertex());
176 // ColorAttrib::T_off means the color scale becomes the new
177 // color.
178 entry->_state = entry->_state->set_attrib(ColorAttrib::make_flat(csa->get_scale()));
179
180 } else if (ca->get_color_type() == ColorAttrib::T_flat) {
181 // ColorAttrib::T_flat means the color scale modulates the
182 // specified color to produce a new color.
183 const LColor &c1 = ca->get_color();
184 const LVecBase4 &c2 = csa->get_scale();
185 LColor color(c1[0] * c2[0], c1[1] * c2[1],
186 c1[2] * c2[2], c1[3] * c2[3]);
187 entry->_state = entry->_state->set_attrib(ColorAttrib::make_flat(color));
188
189 } else {
190 // Otherwise, we have vertex color, and we just scale it
191 // normally.
192 if (transformer.transform_colors(new_geom, csa->get_scale())) {
193 any_changed = true;
194 }
195 entry->_state = entry->_state->set_attrib(ColorAttrib::make_vertex());
196 }
197 }
198 }
199 }
200 }
201
202 if ((attrib_types & SceneGraphReducer::TT_tex_matrix) != 0) {
203 if (geom_attribs._tex_matrix != nullptr) {
204 // Determine which texture coordinate names are used more than once.
205 // This assumes we have discovered all of the textures that are in
206 // effect on the GeomNode; this may not be true if there is a
207 // texture that has been applied at a node above that from which we
208 // started the flatten operation, but caveat programmer.
209 pmap<InternalName *, pset<CPT(TransformState)> > name_transforms;
210
211 const TexMatrixAttrib *tma =
212 DCAST(TexMatrixAttrib, geom_attribs._tex_matrix);
213 int num_stages = tma->get_num_stages();
214 for (int i = 0; i < num_stages; i++) {
215 TextureStage *stage = tma->get_stage(i);
216 InternalName *name = stage->get_texcoord_name();
217 name_transforms[name].insert(tma->get_transform(stage)->get_unique());
218 }
219
220 if (geom_attribs._texture != nullptr) {
221 // There may be stages without a TexMatrixAttrib, which implicitly
222 // use the identity transform.
223 const TextureAttrib *ta = DCAST(TextureAttrib, geom_attribs._texture);
224 int num_on_stages = ta->get_num_on_stages();
225 for (int si = 0; si < num_on_stages; si++) {
226 TextureStage *stage = ta->get_on_stage(si);
227 InternalName *name = stage->get_texcoord_name();
228 if (!tma->has_stage(stage)) {
229 name_transforms[name].insert(TransformState::make_identity());
230 }
231 }
232 }
233
234 CPT(TexMatrixAttrib) new_tma = DCAST(TexMatrixAttrib, TexMatrixAttrib::make());
235
236 for (int i = 0; i < num_stages; i++) {
237 TextureStage *stage = tma->get_stage(i);
238 InternalName *name = stage->get_texcoord_name();
239 auto it = name_transforms.find(name);
240 if (it == name_transforms.end()) {
241 // Already processed this set.
242 }
243 else if (it->second.size() != 1) {
244 // We can't transform these texcoords, since the name is used by
245 // more than one active stage.
246 new_tma = DCAST(TexMatrixAttrib, new_tma->add_stage(stage, tma->get_transform(stage)));
247 }
248 else {
249 // It's safe to transform these texcoords; the name is used by
250 // no more than one active stage, or all these stages have the
251 // same transform.
252 const TransformState *transform = *(it->second.begin());
253 if (!transform->is_identity() &&
254 transformer.transform_texcoords(new_geom, name, name, transform->get_mat())) {
255 any_changed = true;
256 }
257 // Now make sure we don't transform this set more than once.
258 name_transforms.erase(it);
259 }
260 }
261
262 if (!new_tma->is_empty()) {
263 entry->_state = entry->_state->add_attrib(new_tma);
264 }
265 }
266 }
267
268 if ((attrib_types & SceneGraphReducer::TT_other) != 0) {
269 entry->_state = geom_attribs._other->compose(entry->_state);
270 }
271
272 // We handle cull_face last, since that might involve duplicating the
273 // geom, and we'd also like to duplicate all of the changes we may have
274 // applied in the above.
275
276 if ((attrib_types & SceneGraphReducer::TT_cull_face) != 0) {
277 if (geom_attribs._cull_face != nullptr) {
278 const CullFaceAttrib *cfa = DCAST(CullFaceAttrib, geom_attribs._cull_face);
279 CullFaceAttrib::Mode mode = cfa->get_effective_mode();
280 switch (mode) {
281 case CullFaceAttrib::M_cull_none:
282 // Doublesided polys. Duplicate them.
283 {
284 bool has_normals = (new_geom->get_vertex_data()->has_column(InternalName::get_normal()));
285 if (has_normals) {
286 // If the geometry has normals, we have to duplicate it to
287 // reverse the normals on the duplicate copy.
288 PT(Geom) dup_geom = new_geom->reverse();
289 transformer.reverse_normals(dup_geom);
290
291 geoms->push_back(GeomEntry(dup_geom, entry->_state));
292
293 // The above push_back() operation might have invalidated our
294 // old pointer into the list, so we reassign it now.
295 entry = &(*geoms)[i];
296
297 } else {
298 // If there are no normals, we can just doubleside it in
299 // place. This is preferable because we can share vertices.
300 new_geom->doubleside_in_place();
301 any_changed = true;
302 }
303 }
304 break;
305
306 case CullFaceAttrib::M_cull_counter_clockwise:
307 // Reverse winding order.
308 new_geom->reverse_in_place();
309 transformer.reverse_normals(new_geom);
310 any_changed = true;
311 break;
312
313 default:
314 break;
315 }
316 }
317 }
318
319 if (any_changed) {
320 entry->_geom = new_geom;
321 }
322 }
323 }
324 CLOSE_ITERATE_CURRENT_AND_UPSTREAM(_cycler);
325
326 if ((attrib_types & SceneGraphReducer::TT_apply_texture_color) != 0) {
327 transformer.apply_texture_colors(this, attribs._other);
328 }
329
330 transformer.register_vertices(this, false);
331}

382void GeomNode::
383r_prepare_scene(GraphicsStateGuardianBase *gsg, const RenderState *node_state,
384 GeomTransformer &transformer, Thread *current_thread) {
385 PreparedGraphicsObjects *prepared_objects = gsg->get_prepared_objects();
386
387 CDReader cdata(_cycler, current_thread);
388 GeomList::const_iterator gi;
389 CPT(GeomList) geoms = cdata->get_geoms();
390 for (gi = geoms->begin(); gi != geoms->end(); ++gi) {
391 const GeomEntry &entry = (*gi);
392 CPT(RenderState) geom_state = node_state->compose(entry._state);
393 CPT(Geom) geom = entry._geom.get_read_pointer();
394
395 // Munge the geom as required by the GSG.
396 PT(GeomMunger) munger = gsg->get_geom_munger(geom_state, current_thread);
397 geom = transformer.premunge_geom(geom, munger);
398
399 // Prepare each of the vertex arrays in the munged Geom.
400 CPT(GeomVertexData) vdata = geom->get_animated_vertex_data(false, current_thread);
401 GeomVertexDataPipelineReader vdata_reader(vdata, current_thread);
402 int num_arrays = vdata_reader.get_num_arrays();
403 for (int i = 0; i < num_arrays; ++i) {
404 CPT(GeomVertexArrayData) array = vdata_reader.get_array(i);
405 prepared_objects->enqueue_vertex_buffer((GeomVertexArrayData *)array.p());
406 }
407
408 // And also each of the index arrays.
409 int num_primitives = geom->get_num_primitives();
410 for (int i = 0; i < num_primitives; ++i) {
411 CPT(GeomPrimitive) prim = geom->get_primitive(i);
412 prepared_objects->enqueue_index_buffer((GeomPrimitive *)prim.p());
413 }
414
415 // As well as the shaders.
416 const ShaderAttrib *sa;
417 if (geom_state->get_attrib(sa)) {
418 Shader *shader = (Shader *)sa->get_shader();
419 if (shader != nullptr) {
420 prepared_objects->enqueue_shader(shader);
421 }
422 else if (sa->auto_shader()) {
423 gsg->ensure_generated_shader(geom_state);
424 }
425 else if (munger->is_of_type(StateMunger::get_class_type())) {
426 // Premunge the state for the fixed-function pipeline.
427 StateMunger *state_munger = (StateMunger *)munger.p();
428 if (state_munger->should_munge_state()) {
429 geom_state = state_munger->munge_state(geom_state);
430 }
431 }
432 // TODO: prepare the shader inputs.
433 }
434
435 // And now prepare each of the textures.
436 const TextureAttrib *ta;
437 if (geom_state->get_attrib(ta)) {
438 int num_stages = ta->get_num_on_stages();
439 for (int i = 0; i < num_stages; ++i) {
440 Texture *texture = ta->get_on_texture(ta->get_on_stage(i));
441 // TODO: prepare the sampler states, if specified.
442 if (texture != nullptr) {
443 prepared_objects->enqueue_texture(texture);
444 }
445 }
446 }
447 }
448
449 PandaNode::r_prepare_scene(gsg, node_state, transformer, current_thread);
450}

515void GeomNode::
516add_for_draw(CullTraverser *trav, CullTraverserData &data) {
517 trav->_geom_nodes_pcollector.add_level(1);
518
519 if (pgraph_cat.is_spam()) {
520 pgraph_cat.spam()
521 << "Found " << *this << " in state " << *data._state
522 << " draw_mask = " << data._draw_mask << "\n";
523 }
524
525 Thread *current_thread = trav->get_current_thread();
526
527 // Get all the Geoms, with no decalling.
528 Geoms geoms = get_geoms(current_thread);
529 int num_geoms = geoms.get_num_geoms();
530 trav->_geoms_pcollector.add_level(num_geoms);
531 CPT(TransformState) internal_transform = data.get_internal_transform(trav);
532
533 if (num_geoms == 1) {
534 // If there's only one Geom, we don't need to bother culling each individual
535 // Geom bounding volume against the view frustum, since we've already
536 // checked the one on the GeomNode itself.
537 CPT(Geom) geom = geoms.get_geom(0);
538 if (!geom->is_empty()) {
539 CPT(RenderState) state = data._state->compose(geoms.get_geom_state(0));
540 if (!state->has_cull_callback() || state->cull_callback(trav, data)) {
541 CullableObject object(std::move(geom), std::move(state), std::move(internal_transform));
542 object._instances = data._instances;
543 trav->get_cull_handler()->record_object(std::move(object), trav);
544 }
545 }
546 }
547 else {
548 // More than one Geom.
549 for (int i = 0; i < num_geoms; i++) {
550 CPT(Geom) geom = geoms.get_geom(i);
551 if (geom->is_empty()) {
552 continue;
553 }
554
555 CPT(RenderState) state = data._state->compose(geoms.get_geom_state(i));
556 if (state->has_cull_callback() && !state->cull_callback(trav, data)) {
557 // Cull.
558 continue;
559 }
560
561 if (data._instances != nullptr) {
562 // Draw each individual instance. We don't bother culling each
563 // individual Geom for each instance; that is probably way too slow.
564 CullableObject object(std::move(geom), std::move(state), internal_transform);
565 object._instances = data._instances;
566 trav->get_cull_handler()->record_object(std::move(object), trav);
567 continue;
568 }
569
570 // Cull the individual Geom against the view frustum.
571 if (data._view_frustum != nullptr &&
572 !geom->is_in_view(data._view_frustum, current_thread)) {
573 // Cull this Geom.
574 continue;
575 }
576 if (data._cull_planes != nullptr) {
577 // Also cull the Geom against the cull planes.
578 CPT(BoundingVolume) geom_volume = geom->get_bounds(current_thread);
579 const GeometricBoundingVolume *geom_gbv =
580 geom_volume->as_geometric_bounding_volume();
581 int result;
582 data._cull_planes->do_cull(result, state, geom_gbv);
583 if (result == BoundingVolume::IF_no_intersection) {
584 // Cull.
585 continue;
586 }
587 }
588
589 trav->get_cull_handler()->record_object(CullableObject(
590 std::move(geom), std::move(state), internal_transform), trav);
591 }
592 }

741void GeomNode::
742unify(int max_indices, bool preserve_order) {
743 bool any_changed = false;
744
745 Thread *current_thread = Thread::get_current_thread();
746 OPEN_ITERATE_CURRENT_AND_UPSTREAM(_cycler, current_thread) {
747 CDStageWriter cdata(_cycler, pipeline_stage, current_thread);
748
749 PT(GeomList) new_geoms = new GeomList;
750
751 // Try to unify each Geom with each preceding Geom. This is an n^2
752 // operation, but usually there are only a handful of Geoms to consider,
753 // so that's not a big deal.
754 GeomList::const_iterator gi;
755 CPT(GeomList) old_geoms = cdata->get_geoms();
756 for (gi = old_geoms->begin(); gi != old_geoms->end(); ++gi) {
757 const GeomEntry &old_entry = (*gi);
758
759 bool unified = false;
760
761 // Go from back to front, to minimize damage to the primitive ordering.
762 GeomList::reverse_iterator gj;
763 for (gj = new_geoms->rbegin(); gj != new_geoms->rend() && !unified; ++gj) {
764 GeomEntry &new_entry = (*gj);
765 if (old_entry._state == new_entry._state) {
766 // Both states match, so try to combine the primitives.
767 CPT(Geom) old_geom = old_entry._geom.get_read_pointer();
768 PT(Geom) new_geom = new_entry._geom.get_write_pointer();
769 if (new_geom->copy_primitives_from(old_geom)) {
770 // Successfully combined!
771 unified = true;
772 any_changed = true;
773 }
774 }
775
776 if (preserve_order) {
777 // If we're insisting on preserving the order, we can only attempt
778 // to merge with the tail of the list.
779 break;
780 }
781 }
782
783 if (!unified) {
784 // Couldn't unify this Geom with anything, so just add it to the
785 // output list.
786 new_geoms->push_back(old_entry);
787 }
788 }
789
790 // Done! We'll keep whatever's left in the output list.
791 cdata->set_geoms(new_geoms);
792
793 // Finally, go back through and unify the resulting geom(s).
794 GeomList::iterator wgi;
795 for (wgi = new_geoms->begin(); wgi != new_geoms->end(); ++wgi) {
796 GeomEntry &entry = (*wgi);
797 nassertv(entry._geom.test_ref_count_integrity());
798 PT(Geom) geom = entry._geom.get_write_pointer();
799 geom->unify_in_place(max_indices, preserve_order);
800 }
801 }
802 CLOSE_ITERATE_CURRENT_AND_UPSTREAM(_cycler);
803
804 if (any_changed) {
805 mark_internal_bounds_stale();
806 }
807}

945void GeomNode::
946compute_internal_bounds(CPT(BoundingVolume) &internal_bounds,
947 int &internal_vertices,
948 int pipeline_stage,
949 Thread *current_thread) const {
950 int num_vertices = 0;
951
952 CDLockedStageReader cdata(_cycler, pipeline_stage, current_thread);
953
954 pvector<const BoundingVolume *> child_volumes;
955 pvector<CPT(BoundingVolume) > child_volumes_ref;
956 bool all_box = true;
957
958 GeomList::const_iterator gi;
959 CPT(GeomList) geoms = cdata->get_geoms();
960 child_volumes.reserve(geoms->size());
961 child_volumes_ref.reserve(geoms->size());
962
963 for (gi = geoms->begin(); gi != geoms->end(); ++gi) {
964 const GeomEntry &entry = (*gi);
965 CPT(Geom) geom = entry._geom.get_read_pointer();
966 CPT(BoundingVolume) volume = geom->get_bounds();
967
968 if (!volume->is_empty()) {
969 child_volumes.push_back(volume);
970 child_volumes_ref.push_back(volume);
971 if (!volume->is_exact_type(BoundingBox::get_class_type())) {
972 all_box = false;
973 }
974 }
975 num_vertices += geom->get_nested_vertices();
976 }
977
978 PT(GeometricBoundingVolume) gbv;
979
980 BoundingVolume::BoundsType btype = get_bounds_type();
981 if (btype == BoundingVolume::BT_default) {
982 btype = bounds_type;
983 }
984
985 if (btype == BoundingVolume::BT_box ||
986 (btype != BoundingVolume::BT_sphere && all_box)) {
987 // If all of the child volumes are a BoundingBox, then our volume is also
988 // a BoundingBox.
989 gbv = new BoundingBox;
990 } else {
991 // Otherwise, it's a sphere.
992 gbv = new BoundingSphere;
993 }
994
995 if (child_volumes.size() > 0) {
996 const BoundingVolume **child_begin = &child_volumes[0];
997 const BoundingVolume **child_end = child_begin + child_volumes.size();
998 ((BoundingVolume *)gbv)->around(child_begin, child_end);
999 }
1000
1001 internal_bounds = gbv;
1002 internal_vertices = num_vertices;
1003}

1028void GeomNode::
1029finalize(BamReader *manager) {
1030 if (manager->get_file_minor_ver() < 14) {
1031 // With version 6.14, we changed the default ColorAttrib behavior from
1032 // make_vertex() to make_flat(). This means that every Geom that contains
1033 // vertex colors now needs to have an explicit ColorAttrib::make_vertex()
1034 // on its state.
1035
1036 // Since we shouldn't override a different ColorAttrib inherited from
1037 // above, we create this new attrib with an override of -1.
1038
1039 CPT(InternalName) color = InternalName::get_color();
1040 CPT(RenderAttrib) vertex_color = ColorAttrib::make_vertex();
1041
1042 Thread *current_thread = Thread::get_current_thread();
1043 OPEN_ITERATE_CURRENT_AND_UPSTREAM(_cycler, current_thread) {
1044 CDStageWriter cdata(_cycler, pipeline_stage, current_thread);
1045
1046 GeomList::iterator gi;
1047 PT(GeomList) geoms = cdata->modify_geoms();
1048 for (gi = geoms->begin(); gi != geoms->end(); ++gi) {
1049 GeomEntry &entry = (*gi);
1050 CPT(Geom) geom = entry._geom.get_read_pointer();
1051
1052 // Force the various GeomVertexArrayFormat objects to finalize
1053 // themselves. We have to do this before we can reliably call
1054 // GeomVertexData::has_column().
1055 CPT(GeomVertexData) vdata = geom->get_vertex_data(current_thread);
1056 CPT(GeomVertexFormat) vformat = vdata->get_format();
1057 for (size_t i = 0; i < vformat->get_num_arrays(); ++i) {
1058 const GeomVertexArrayFormat *varray = vformat->get_array(i);
1059 manager->finalize_now((GeomVertexArrayFormat *)varray);
1060 }
1061
1062 if (vdata->has_column(color) &&
1063 !entry._state->has_attrib(ColorAttrib::get_class_slot())) {
1064 // We'll be reassigning the RenderState. Therefore, save it
1065 // temporarily to increment its reference count.
1066 PT(BamAuxData) aux_data = new BamAuxData;
1067 aux_data->_hold_state = entry._state;
1068 manager->set_aux_data((RenderState *)entry._state.p(), "hold_state", aux_data);
1069
1070 entry._state = entry._state->add_attrib(vertex_color, -1);
1071 }
1072 }
1073 }
1074 CLOSE_ITERATE_CURRENT_AND_UPSTREAM(_cycler);
1075 }
