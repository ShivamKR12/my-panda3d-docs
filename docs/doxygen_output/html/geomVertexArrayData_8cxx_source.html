122int GeomVertexArrayData::
123compare_to(const GeomVertexArrayData &other) const {
124 Thread *current_thread = Thread::get_current_thread();
125
126 CPT(GeomVertexArrayDataHandle) handle = get_handle(current_thread);
127 CPT(GeomVertexArrayDataHandle) other_handle = other.get_handle(current_thread);
128
129 if (handle->get_usage_hint() != other_handle->get_usage_hint()) {
130 return (int)handle->get_usage_hint() - (int)other_handle->get_usage_hint();
131 }
132 if (handle->get_array_format() != other_handle->get_array_format()) {
133 return handle->get_array_format() < other_handle->get_array_format() ? -1 : 1;
134 }
135 if (handle->get_data_size_bytes() != other_handle->get_data_size_bytes()) {
136 return (int)handle->get_data_size_bytes() - (int)other_handle->get_data_size_bytes();
137 }
138 return memcmp(handle->get_read_pointer(true),
139 other_handle->get_read_pointer(true),
140 handle->get_data_size_bytes());
141}

263int GeomVertexArrayData::
264release_all() {
265 int num_freed = 0;
266
267 if (_contexts != nullptr) {
268 // We have to traverse a copy of the _contexts list, because the
269 // PreparedGraphicsObjects object will call clear_prepared() in response
270 // to each release_vertex_buffer(), and we don't want to be modifying the
271 // _contexts list while we're traversing it.
272 Contexts temp = *_contexts;
273 num_freed = (int)_contexts->size();
274
275 Contexts::const_iterator ci;
276 for (ci = temp.begin(); ci != temp.end(); ++ci) {
277 PreparedGraphicsObjects *prepared_objects = (*ci).first;
278 VertexBufferContext *vbc = (*ci).second;
279 prepared_objects->release_vertex_buffer(vbc);
280 }
281
282 // Now that we've called release_vertex_buffer() on every known context,
283 // the _contexts list should have completely emptied itself.
284 nassertr(_contexts == nullptr, num_freed);
285 }
286
287 return num_freed;
288}

348void GeomVertexArrayData::
349reverse_data_endianness(unsigned char *dest, const unsigned char *source,
350 size_t size) {
351 int num_columns = _array_format->get_num_columns();
352
353 // Walk through each row of the data.
354 for (size_t pi = 0; pi < size; pi += _array_format->get_stride()) {
355 // For each row, visit all of the columns; and for each column, visit all
356 // of the components of that column.
357 for (int ci = 0; ci < num_columns; ++ci) {
358 const GeomVertexColumn *col = _array_format->get_column(ci);
359 int component_bytes = col->get_component_bytes();
360 if (component_bytes > 1) {
361 // Get the index of the beginning of the column.
362 size_t ci = pi + col->get_start();
363
364 int num_components = col->get_num_components();
365 for (int cj = 0; cj < num_components; ++cj) {
366 // Reverse the bytes of each component.
367 ReversedNumericData nd(source + ci, component_bytes);
368 nd.store_value(dest + ci, component_bytes);
369 ci += component_bytes;
370 }
371 }
372 }
373 }

430void GeomVertexArrayData::
431finalize(BamReader *manager) {
432 // Now we need to register the format that we have read from the bam file
433 // (since it doesn't come out of the bam file automatically registered).
434 // This may change the format's pointer, which we should then update our own
435 // data to reflect. But since this may cause the unregistered object to
436 // destruct, we have to also tell the BamReader to return the new object
437 // from now on.
438
439 CDWriter cdata(_cycler, true);
440
441 CPT(GeomVertexArrayFormat) new_array_format =
442 GeomVertexArrayFormat::register_format(_array_format);
443
444 manager->change_pointer(_array_format, new_array_format);
445 _array_format = new_array_format;
446
447 PT(BamAuxData) aux_data = (BamAuxData *)manager->get_aux_data(this, "");
448 if (aux_data != nullptr) {
449 if (aux_data->_endian_reversed) {
450 // Now is the time to endian-reverse the data.
451 VertexDataBuffer new_buffer(cdata->_buffer.get_size());
452 reverse_data_endianness(new_buffer.get_write_pointer(), cdata->_buffer.get_read_pointer(true), cdata->_buffer.get_size());
453 cdata->_buffer.swap(new_buffer);
454 }
455 }
456
457 set_lru_size(cdata->_buffer.get_size());
458}

532void GeomVertexArrayData::CData::
533fillin(DatagramIterator &scan, BamReader *manager, void *extra_data) {
534 GeomVertexArrayData *array_data = (GeomVertexArrayData *)extra_data;
535 _usage_hint = (UsageHint)scan.get_uint8();
536
537 if (manager->get_file_minor_ver() < 8) {
538 // Before bam version 6.8, the array data was a PTA_uchar.
539 PTA_uchar new_data;
540 READ_PTA(manager, scan, array_data->read_raw_data, new_data);
541 _buffer.unclean_realloc(new_data.size());
542 _buffer.set_size(new_data.size());
543 memcpy(_buffer.get_write_pointer(), &new_data[0], new_data.size());
544
545 } else {
546 // Now, the array data is just stored directly.
547 size_t size = scan.get_uint32();
548 _buffer.unclean_realloc(size);
549 _buffer.set_size(size);
550
551 const unsigned char *source_data =
552 (const unsigned char *)scan.get_datagram().get_data();
553 memcpy(_buffer.get_write_pointer(), source_data + scan.get_current_index(), size);
554 scan.skip_bytes(size);
555 }
556
557 bool endian_reversed = false;
558
559 if (manager->get_file_endian() != BamReader::BE_native) {
560 // For non-native endian files, we have to convert the data.
561
562 if (array_data->_array_format == nullptr) {
563 // But we can't do that until we've completed the _array_format pointer,
564 // which tells us how to convert it.
565 endian_reversed = true;
566 } else {
567 // Since we have the _array_format pointer now, we can reverse it
568 // immediately (and we should, to support threaded CData updates).
569 VertexDataBuffer new_buffer(_buffer.get_size());
570 array_data->reverse_data_endianness(new_buffer.get_write_pointer(), _buffer.get_read_pointer(true), _buffer.get_size());
571 _buffer.swap(new_buffer);
572 }
573 }
574
575 if (endian_reversed) {
576 PT(BamAuxData) aux_data = new BamAuxData;
577 aux_data->_endian_reversed = endian_reversed;
578 manager->set_aux_data(array_data, "", aux_data);
579 }
580
581 array_data->set_lru_size(_buffer.get_size());
582
583 _modified = Geom::get_next_modified();
584}

600bool GeomVertexArrayDataHandle::
601set_num_rows(int n) {
602 nassertr(_writable, false);
603 mark_used();
604
605 int stride = _object->_array_format->get_stride();
606 size_t new_size = n * stride;
607 size_t orig_size = _cdata->_buffer.get_size();
608
609 if (gobj_cat.is_spam()) {
610 gobj_cat.spam()
611 << _object << ".set_num_rows(" << n << "), size = " << new_size << "\n";
612 }
613
614 if (new_size != orig_size) {
615 size_t orig_reserved_size = _cdata->_buffer.get_reserved_size();
616 if (new_size > orig_reserved_size) {
617 // Add more rows. Go up to the next power of two bytes, mainly to
618 // reduce the number of allocs needed.
619 size_t new_reserved_size = (size_t)1 << get_next_higher_bit(new_size - 1);
620 nassertr(new_reserved_size >= new_size, false);
621
622 _cdata->_buffer.clean_realloc(new_reserved_size);
623
624 } else if (new_size == 0) {
625 // If we set the number of rows to 0, go ahead and clear the buffer
626 // altogether, and let the user build it up again from nothing, to try
627 // to reduce frivolous memory waste.
628 _cdata->_buffer.clear();
629 }
630
631 _cdata->_buffer.set_size(new_size);
632
633 // Now ensure that the newly-added rows are initialized to 0.
634 if (new_size > orig_size) {
635 memset(_cdata->_buffer.get_write_pointer() + orig_size, 0,
636 new_size - orig_size);
637 }
638
639 _cdata->_modified = Geom::get_next_modified();
640
641 if (get_current_thread()->get_pipeline_stage() == 0) {
642 _object->set_lru_size(_cdata->_buffer.get_size());
643 }
644
645 nassertr(get_num_rows() == n, true);
646 return true;
647 }
648
649 nassertr(get_num_rows() == n, false);
650 return false;
651}

656bool GeomVertexArrayDataHandle::
657unclean_set_num_rows(int n) {
658 nassertr(_writable, false);
659 mark_used();
660
661 int stride = _object->_array_format->get_stride();
662 size_t new_size = n * stride;
663 size_t orig_size = _cdata->_buffer.get_size();
664 size_t orig_reserved_size = _cdata->_buffer.get_reserved_size();
665
666 if (new_size != orig_size || new_size != orig_reserved_size) {
667 // Since this is unclean_set_num_rows(), we won't be using it to
668 // incrementally increase the array; instead, it will generally be used
669 // only to create an array initially. So it makes sense to set the
670 // reserved size to precisely the same as the target size.
671
672 _cdata->_buffer.unclean_realloc(new_size);
673 _cdata->_buffer.set_size(new_size);
674
675 // No need to fill to zero or copy the old buffer, since this is
676 // unclean_set_num_rows().
677
678 if (new_size != orig_size) {
679 _cdata->_modified = Geom::get_next_modified();
680
681 if (get_current_thread()->get_pipeline_stage() == 0) {
682 _object->set_lru_size(_cdata->_buffer.get_size());
683 }
684 }
685 return true;
686 }
687
688 return false;
689}

694bool GeomVertexArrayDataHandle::
695reserve_num_rows(int n) {
696 nassertr(_writable, false);
697 mark_used();
698
699 int stride = _object->_array_format->get_stride();
700 size_t new_reserved_size = n * stride;
701 new_reserved_size = max(_cdata->_buffer.get_size(), new_reserved_size);
702 size_t orig_reserved_size = _cdata->_buffer.get_reserved_size();
703
704 if (gobj_cat.is_debug()) {
705 gobj_cat.debug()
706 << _object << ".reserve_num_rows(" << n << "), size = " << new_reserved_size << "\n";
707 }
708
709 if (new_reserved_size != orig_reserved_size) {
710 // We allow the user to set the alloc point smaller with this call,
711 // assuming the user knows what he's doing. This allows the user to
712 // reduce wasted memory after completely filling up a buffer.
713 _cdata->_buffer.clean_realloc(new_reserved_size);
714 return true;
715 }
716
717 return false;

781void GeomVertexArrayDataHandle::
782copy_subdata_from(size_t to_start, size_t to_size,
783 const unsigned char *source,
784 size_t from_start, size_t from_size) {
785 nassertv(_writable);
786 mark_used();
787
788 VertexDataBuffer &to_buffer = _cdata->_buffer;
789 size_t to_buffer_orig_size = to_buffer.get_size();
790 to_start = min(to_start, to_buffer_orig_size);
791 to_size = min(to_size, to_buffer_orig_size - to_start);
792
793 if (from_size < to_size) {
794 // Reduce the array.
795 unsigned char *pointer = to_buffer.get_write_pointer();
796 memmove(pointer + to_start + to_size,
797 pointer + to_start + from_size,
798 to_buffer_orig_size - (to_start + to_size));
799 to_buffer.set_size(to_buffer_orig_size + from_size - to_size);
800
801 } else if (to_size < from_size) {
802 // Expand the array.
803 size_t needed_size = to_buffer_orig_size + from_size - to_size;
804 size_t to_buffer_orig_reserved_size = to_buffer.get_reserved_size();
805 if (needed_size > to_buffer_orig_reserved_size) {
806 size_t new_reserved_size = (size_t)1 << get_next_higher_bit(needed_size - 1);
807 to_buffer.clean_realloc(new_reserved_size);
808 }
809 to_buffer.set_size(needed_size);
810
811 unsigned char *pointer = to_buffer.get_write_pointer();
812 memmove(pointer + to_start + to_size,
813 pointer + to_start + from_size,
814 to_buffer_orig_size - (to_start + to_size));
815 }
816
817 // Now copy the data.
818 memcpy(to_buffer.get_write_pointer() + to_start,
819 source + from_start, from_size);
820 _cdata->_modified = Geom::get_next_modified();
821
822 if (get_current_thread()->get_pipeline_stage() == 0) {
823 _object->set_lru_size(_cdata->_buffer.get_size());
824 }

867 // Reduce the array.
868 unsigned char *pointer = to_buffer.get_write_pointer();

871 to_buffer_orig_size - (start + size));

874 } else if (size < from_size) {

885 memmove(pointer + start + from_size,
886 pointer + start + size,
887 to_buffer_orig_size - (start + size));
888 }

891 memcpy(to_buffer.get_write_pointer() + start, data.data(), from_size);
