121 // Initialization
122 audio_cat.init();

124 _device = nullptr;
125 string dev_name = select_audio_device();
126
127 if (!dev_name.empty()) {
128 // Open a specific device by name.
129 audio_cat.info() << "Using OpenAL device " << dev_name << "\n";
130 _device = alcOpenDevice(dev_name.c_str());

132 if (_device == nullptr) {
133 audio_cat.error()
134 << "Couldn't open OpenAL device \"" << dev_name << "\", falling back to default device\n";

153 }

156 // We managed to get a device open.
157 alcGetError(_device); // clear errors
158

166 attrs[1] = audio_want_hrtf.get_value() ? ALC_TRUE : ALC_FALSE;

170 }
171#endif // HAVE_OPENAL_FRAMEWORK

178 }
179 }

189 } else {

202 }
203 }
204
205 if (audio_cat.is_debug()) {
206 audio_cat.debug() << "AL_RENDERER:" << alGetString(AL_RENDERER) << endl;
207 audio_cat.debug() << "AL_VENDOR:" << alGetString(AL_VENDOR) << endl;

258string OpenALAudioManager::
259select_audio_device() {
260 string selected_device = openal_device;
261
262 const char *devices = nullptr;
263
264 // This extension gives us all audio paths on all drivers.
265 if (alcIsExtensionPresent(nullptr, "ALC_ENUMERATE_ALL_EXT") == AL_TRUE) {
266 string default_device = alcGetString(nullptr, ALC_DEFAULT_ALL_DEVICES_SPECIFIER);
267 devices = (const char *)alcGetString(nullptr, ALC_ALL_DEVICES_SPECIFIER);
268
269 if (devices) {
270 if (audio_cat.is_debug()) {
271 audio_cat.debug() << "All OpenAL devices:\n";
272 }
273
274 while (*devices) {
275 string device(devices);
276 devices += device.size() + 1;
277
278 if (audio_cat.is_debug()) {
279 if (device == selected_device) {
280 audio_cat.debug() << " " << device << " [selected]\n";
281 } else if (device == default_device) {
282 audio_cat.debug() << " " << device << " [default]\n";
283 } else {
284 audio_cat.debug() << " " << device << "\n";
285 }
286 }
287 }
288 }
289 } else {
290 audio_cat.debug() << "ALC_ENUMERATE_ALL_EXT not supported\n";
291 }
292
293 // This extension just gives us generic driver names, like "OpenAL Soft" and
294 // "Generic Software", rather than individual outputs.
295 if (alcIsExtensionPresent(nullptr, "ALC_ENUMERATION_EXT") == AL_TRUE) {
296 string default_device = alcGetString(nullptr, ALC_DEFAULT_DEVICE_SPECIFIER);
297 devices = (const char *)alcGetString(nullptr, ALC_DEVICE_SPECIFIER);
298
299 if (devices) {
300 if (audio_cat.is_debug()) {
301 audio_cat.debug() << "OpenAL drivers:\n";
302 }
303
304 while (*devices) {
305 string device(devices);
306 devices += device.size() + 1;
307
308 if (selected_device.empty() && device == "OpenAL Soft" &&
309 default_device == "Generic Software") {
310 // Prefer OpenAL Soft over the buggy Generic Software driver.
311 selected_device = "OpenAL Soft";
312 }
313
314 if (audio_cat.is_debug()) {
315 if (device == selected_device) {
316 audio_cat.debug() << " " << device << " [selected]\n";
317 } else if (device == default_device) {
318 audio_cat.debug() << " " << device << " [default]\n";
319 } else {
320 audio_cat.debug() << " " << device << "\n";
321 }
322 }
323 }
324 }
325 } else {
326 audio_cat.debug() << "ALC_ENUMERATION_EXT not supported\n";
327 }
328
329 return selected_device;
330}

362bool OpenALAudioManager::
363should_load_audio(MovieAudioCursor *source, int mode) {
364 ReMutexHolder holder(_lock);
365 if (mode == SM_stream) {
366 // If the user asked for streaming, give him streaming.
367 return false;
368 }
369 if (source->get_source()->get_filename().empty()) {
370 // Non-files cannot be preloaded.
371 return false;
372 }
373 if (source->ready() != 0x40000000) {
374 // Streaming sources cannot be preloaded.
375 return false;
376 }
377 if (source->length() > 3600.0) {
378 // Anything longer than an hour cannot be preloaded.
379 return false;
380 }
381 int channels = source->audio_channels();
382 int samples = (int)(source->length() * source->audio_rate());
383 int bytes = samples * channels * 2;
384 if ((mode == SM_heuristic)&&(bytes > audio_preload_threshold)) {
385 // In heuristic mode, if file is long, stream it.
386 return false;
387 }
388 return true;
389}

397OpenALAudioManager::SoundData *OpenALAudioManager::
398get_sound_data(MovieAudio *movie, int mode) {
399 ReMutexHolder holder(_lock);
400 const Filename &path = movie->get_filename();
401
402 // Search for an already-cached sample or an already-opened stream.
403 if (!path.empty()) {
404
405 if (mode != SM_stream) {
406 SampleCache::iterator lsmi=_sample_cache.find(path);
407 if (lsmi != _sample_cache.end()) {
408 SoundData *sd = (*lsmi).second;
409 increment_client_count(sd);
410 return sd;
411 }
412 }
413
414 if (mode != SM_sample) {
415 ExpirationQueue::iterator exqi;
416 for (exqi=_expiring_streams.begin(); exqi!=_expiring_streams.end(); exqi++) {
417 SoundData *sd = (SoundData*)(*exqi);
418 if (sd->_movie->get_filename() == path) {
419 increment_client_count(sd);
420 return sd;
421 }
422 }
423 }
424 }
425
426 PT(MovieAudioCursor) stream = movie->open();
427 if (stream == nullptr) {
428 audio_error("Cannot open file: "<<path);
429 return nullptr;
430 }
431
432 if (!can_use_audio(stream)) {
433 audio_error("File is not in usable format: "<<path);
434 return nullptr;
435 }
436
437 SoundData *sd = new SoundData();
438 sd->_client_count = 1;
439 sd->_manager = this;
440 sd->_movie = movie;
441 sd->_rate = stream->audio_rate();
442 sd->_channels = stream->audio_channels();
443 sd->_length = stream->length();
444 audio_debug("Creating: " << sd->_movie->get_filename().get_basename());
445 audio_debug(" - Rate: " << sd->_rate);
446 audio_debug(" - Channels: " << sd->_channels);
447 audio_debug(" - Length: " << sd->_length);
448
449 if (should_load_audio(stream, mode)) {
450 audio_debug(path.get_basename() << ": loading as sample");
451 make_current();
452 alGetError(); // clear errors
453 sd->_sample = 0;
454 alGenBuffers(1, &sd->_sample);
455 al_audio_errcheck("alGenBuffers");
456 if (sd->_sample == 0) {
457 audio_error("Could not create an OpenAL buffer object");
458 delete sd;
459 return nullptr;
460 }
461 int channels = stream->audio_channels();
462 int samples = (int)(stream->length() * stream->audio_rate());
463 int16_t *data = new int16_t[samples * channels];
464 samples = stream->read_samples(samples, data);
465 alBufferData(sd->_sample,
466 (channels>1) ? AL_FORMAT_STEREO16 : AL_FORMAT_MONO16,
467 data, samples * channels * 2, stream->audio_rate());
468 delete[] data;
469 int err = alGetError();
470 if (err != AL_NO_ERROR) {
471 audio_error("could not fill OpenAL buffer object with data");
472 delete sd;
473 return nullptr;
474 }
475 _sample_cache.insert(SampleCache::value_type(path, sd));
476 } else {
477 audio_debug(path.get_basename() << ": loading as stream");
478 sd->_stream = stream;
479 }
480
481 return sd;
482}

546void OpenALAudioManager::
547uncache_sound(const Filename &file_name) {
548 ReMutexHolder holder(_lock);
549 nassertv(is_valid());
550 Filename path = file_name;
551
552 VirtualFileSystem *vfs = VirtualFileSystem::get_global_ptr();
553 vfs->resolve_filename(path, get_model_path());
554
555 SampleCache::iterator sci = _sample_cache.find(path);
556 if (sci == _sample_cache.end()) {
557 sci = _sample_cache.find(file_name);
558 }
559 if (sci != _sample_cache.end()) {
560 SoundData *sd = (*sci).second;
561 if (sd->_client_count == 0) {
562 _expiring_samples.erase(sd->_expire);
563 _sample_cache.erase(sci);
564 delete sd;
565 }
566 }
567
568 ExpirationQueue::iterator exqi;
569 for (exqi = _expiring_streams.begin(); exqi != _expiring_streams.end();) {
570 SoundData *sd = (SoundData *)(*exqi);
571 if (sd->_client_count == 0) {
572 if (sd->_movie->get_filename() == path ||
573 sd->_movie->get_filename() == file_name) {
574 exqi = _expiring_streams.erase(exqi);
575 delete sd;
576 continue;
577 }
578 }
579 ++exqi;
580 }

863void OpenALAudioManager::
864starting_sound(OpenALAudioSound* audio) {
865 ReMutexHolder holder(_lock);
866 ALuint source=0;
867
868 // If the sound already has a source, we don't need to do anything.
869 if (audio->_source) {
870 return;
871 }
872
873 // first give all sounds that have finished a chance to stop, so that these
874 // get stopped first
875 update();
876
877 if (_concurrent_sound_limit) {
878 reduce_sounds_playing_to(_concurrent_sound_limit-1); // because we're about to add one
879 }
880
881 // get a source from the source pool or create a new source
882 if (_al_sources->empty()) {
883 make_current();
884 alGetError(); // clear errors
885 alGenSources(1,&source);
886 ALenum result = alGetError();
887 if (result!=AL_NO_ERROR) {
888 audio_error("alGenSources(): " << alGetString(result) );
889 // if we can't create any more sources, set stop a sound to free a
890 // source
891 reduce_sounds_playing_to(_sounds_playing.size()-1);
892 source = 0;
893 }
894 }
895 // get a source from the source bool if we didn't just allocate one
896 if (!source && !_al_sources->empty()) {
897 source = *(_al_sources->begin());
898 _al_sources->erase(source);
899 }
900
901 audio->_source = source;
902
903 if (source)
904 _sounds_playing.insert(audio);
905}

942void OpenALAudioManager::
943reduce_sounds_playing_to(unsigned int count) {
944 ReMutexHolder holder(_lock);
945 // first give all sounds that have finished a chance to stop, so that these
946 // get stopped first
947 update();
948
949 int limit = _sounds_playing.size() - count;
950 while (limit-- > 0) {
951 SoundsPlaying::iterator sound = _sounds_playing.begin();
952 nassertv(sound != _sounds_playing.end());
953 // When the user stops a sound, there is still a PT in the user's hand.
954 // When we stop a sound here, however, this can remove the last PT. This
955 // can cause an ugly recursion where stop calls the destructor, and the
956 // destructor calls stop. To avoid this, we create a temporary PT, stop
957 // the sound, and then release the PT.
958 PT(OpenALAudioSound) s = (*sound);
959 s->stop();
960 }
961}

975void OpenALAudioManager::
976update() {
977 ReMutexHolder const holder(_lock);
978
979 // See if any of our playing sounds have ended.
980 double const rtc = TrueClock::get_global_ptr()->get_short_time();
981 SoundsPlaying::iterator i = _sounds_playing.begin();
982 while (i != _sounds_playing.end()) {
983 // The post-increment syntax is *very* important here.
984 // As both OpenALAudioSound::pull_used_buffers and OpenALAudioSound::finished can modify the list of sounds playing
985 // by erasing 'sound' from the list, thus invaliding the iterator.
986 PT(OpenALAudioSound) sound = *(i++);
987 sound->pull_used_buffers();
988
989 // If pull_used_buffers() encountered an error, the sound was cleaned up.
990 // No need to do anymore work.
991 if (!sound->is_valid()) {
992 continue;
993 }
994
995 sound->push_fresh_buffers();
996 sound->restart_stalled_audio();
997 sound->cache_time(rtc);
998 if (sound->_source == 0 ||
999 (sound->_stream_queued.empty() &&
1000 (sound->_loops_completed >= sound->_playing_loops))) {
1001 sound->finished();
1002 }
1003 }
1004}

1179void OpenALAudioManager::
1180delete_buffer(ALuint buffer) {
1181 ReMutexHolder holder(_lock);
1182 int tries = 0;
1183 ALuint error;
1184
1185 // Keep trying until we succeed (or give up).
1186 while (true) {
1187 alDeleteBuffers(1, &buffer);
1188 error = alGetError();
1189
1190 if (error == AL_NO_ERROR) {
1191 // Success! This will happen right away 99% of the time.
1192 return;
1193 } else if (error != AL_INVALID_OPERATION) {
1194 // We weren't expecting that. This should be reported.
1195 break;
1196 } else if (tries >= openal_buffer_delete_retries.get_value()) {
1197 // We ran out of retries. Give up.
1198 break;
1199 } else {
1200 // Make another try after (delay * 2^n) seconds.
1201 Thread::sleep(openal_buffer_delete_delay.get_value() * (1 << tries));
1202 tries++;
1203 }
1204 }
1205
1206 // If we got here, one of the breaks above happened, indicating an error.
1207 audio_error("failed to delete a buffer: " << alGetString(error) );
