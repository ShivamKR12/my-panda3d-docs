166void EggLoader::
167build_graph() {
168 _deferred_nodes.clear();
169
170 // Expand all of the ObjectType flags before we do anything else; that might
171 // prune out large portions of the scene.
172 if (!expand_all_object_types(_data)) {
173 return;
174 }
175
176 // Now, load up all of the textures.
177 load_textures();
178
179 // Clean up the vertices.
180 _data->clear_connected_shading();
181 _data->remove_unused_vertices(true);
182 _data->get_connected_shading();
183 _data->unify_attributes(true, egg_flat_shading, true);
184
185 // Now we need to get the connected shading again, since in unifying the
186 // attributes we may have made vertices suddenly become identical to each
187 // other, thereby connecting more primitives than before.
188 _data->clear_connected_shading();
189 _data->remove_unused_vertices(true);
190 _data->get_connected_shading();
191
192 // Sequences and switches have special needs. Make sure that primitives
193 // parented directly to a sequence or switch are sorted into sub-groups
194 // first, to prevent them being unified into a single polyset.
195 separate_switches(_data);
196
197 if (egg_emulate_bface) {
198 emulate_bface(_data);
199 }
200
201 // Then bin up the polysets and LOD nodes.
202 _data->remove_invalid_primitives(true);
203 EggBinner binner(*this);
204 binner.make_bins(_data);
205
206 // ((EggGroupNode *)_data)->write(cerr, 0);
207
208 // Now build up the scene graph.
209 _root = new ModelRoot(_data->get_egg_filename(), _data->get_egg_timestamp());
210
211 EggGroupNode::const_iterator ci;
212 for (ci = _data->begin(); ci != _data->end(); ++ci) {
213 make_node(*ci, _root);
214 }
215
216 reparent_decals();
217 start_sequences();
218
219 apply_deferred_nodes(_root, DeferredNodeProperty());
220}

227void EggLoader::
228reparent_decals() {
229 ExtraNodes::const_iterator di;
230 for (di = _decals.begin(); di != _decals.end(); ++di) {
231 PandaNode *node = (*di);
232 nassertv(node != nullptr);
233
234 // The NodePath interface is best for this.
235 NodePath parent(node);
236
237 // First, search for the GeomNode.
238 NodePath geom_parent;
239 int num_children = parent.get_num_children();
240 for (int i = 0; i < num_children; i++) {
241 NodePath child = parent.get_child(i);
242
243 if (child.node()->is_of_type(GeomNode::get_class_type())) {
244 if (!geom_parent.is_empty()) {
245 // Oops, too many GeomNodes.
246 egg2pg_cat.error()
247 << "Decal onto " << parent.node()->get_name()
248 << " uses base geometry with multiple GeomNodes.\n";
249 _error = true;
250 }
251 geom_parent = child;
252 }
253 }
254
255 if (geom_parent.is_empty()) {
256 // No children were GeomNodes.
257 egg2pg_cat.error()
258 << "Ignoring decal onto " << parent.node()->get_name()
259 << "; no geometry within group.\n";
260 _error = true;
261 } else {
262 // Now reparent all of the non-GeomNodes to this node. We have to be
263 // careful so we don't get lost as we self-modify this list.
264 int i = 0;
265 while (i < num_children) {
266 NodePath child = parent.get_child(i);
267
268 if (child.node()->is_of_type(GeomNode::get_class_type())) {
269 i++;
270 } else {
271 child.reparent_to(geom_parent);
272 num_children--;
273 }
274 }
275
276 // Finally, set the DecalEffect on the base geometry.
277 geom_parent.node()->set_effect(DecalEffect::make());
278 }
279 }

302void EggLoader::
303make_polyset(EggBin *egg_bin, PandaNode *parent, const LMatrix4d *transform,
304 bool is_dynamic, CharacterMaker *character_maker) {
305 if (egg_bin->empty()) {
306 // If there are no children--no primitives--never mind.
307 return;
308 }
309
310 // We know that all of the primitives in the bin have the same render state,
311 // so we can get that information from the first primitive.
312 EggGroupNode::const_iterator ci = egg_bin->begin();
313 nassertv(ci != egg_bin->end());
314 CPT(EggPrimitive) first_prim = DCAST(EggPrimitive, (*ci));
315 nassertv(first_prim != nullptr);
316 const EggRenderState *render_state;
317 DCAST_INTO_V(render_state, first_prim->get_user_data(EggRenderState::get_class_type()));
318
319 if (render_state->_hidden && egg_suppress_hidden) {
320 // Eat this polyset.
321 return;
322 }
323
324 // Generate an optimal vertex pool (or multiple vertex pools, if we have a
325 // lot of vertex) for the polygons within just the bin. Each EggVertexPool
326 // translates directly to an optimal GeomVertexData structure.
327 EggVertexPools vertex_pools;
328 egg_bin->rebuild_vertex_pools(vertex_pools, (unsigned int)egg_max_vertices,
329 false);
330
331 if (egg_mesh) {
332 // If we're using the mesher, mesh now.
333 egg_bin->mesh_triangles(render_state->_flat_shaded ? EggGroupNode::T_flat_shaded : 0);
334
335 } else {
336 // If we're not using the mesher, at least triangulate any higher-order
337 // polygons we might have.
338 egg_bin->triangulate_polygons(EggGroupNode::T_polygon | EggGroupNode::T_convex);
339 }
340
341 // Now that we've meshed, apply the per-prim attributes onto the vertices,
342 // so we can copy them to the GeomVertexData.
343 egg_bin->apply_first_attribute(false);
344 egg_bin->post_apply_flat_attribute(false);
345
346 // egg_bin->write(cerr, 0);
347
348 PT(GeomNode) geom_node;
349
350 // Now iterate through each EggVertexPool. Normally, there's only one, but
351 // if we have a really big mesh, it might have been split into multiple
352 // vertex pools (to keep each one within the egg_max_vertices constraint).
353 EggVertexPools::iterator vpi;
354 for (vpi = vertex_pools.begin(); vpi != vertex_pools.end(); ++vpi) {
355 EggVertexPool *vertex_pool = (*vpi);
356 vertex_pool->remove_unused_vertices();
357 // vertex_pool->write(cerr, 0);
358
359 bool has_overall_color;
360 LColor overall_color;
361 vertex_pool->check_overall_color(has_overall_color, overall_color);
362 if (!egg_flat_colors) {
363 // If flat colors aren't allowed, then we don't care whether there is an
364 // overall color. In that case, treat all vertex pools as if they
365 // contain a combination of multiple colors.
366 has_overall_color = false;
367 }
368
369 PT(TransformBlendTable) blend_table;
370 if (is_dynamic) {
371 // Dynamic vertex pools will require a TransformBlendTable to indicate
372 // how the vertices are to be animated.
373 blend_table = make_blend_table(vertex_pool, egg_bin, character_maker);
374
375 // Now that we've created the blend table, we can re-order the vertices
376 // in the pool to efficiently group vertices together that will share
377 // the same transform matrix. (We have to re-order them before we
378 // create primitives, below, because this will change the vertex index
379 // numbers.)
380 vertex_pool->sort_by_external_index();
381 }
382
383 // Create a handful of GeomPrimitives corresponding to the various types
384 // of primitives that reference this vertex pool.
385 UniquePrimitives unique_primitives;
386 Primitives primitives;
387 for (ci = egg_bin->begin(); ci != egg_bin->end(); ++ci) {
388 EggPrimitive *egg_prim;
389 DCAST_INTO_V(egg_prim, (*ci));
390 if (egg_prim->get_pool() == vertex_pool) {
391 make_primitive(render_state, egg_prim, unique_primitives, primitives,
392 has_overall_color, overall_color);
393 }
394 }
395
396 if (!primitives.empty()) {
397 LMatrix4d mat;
398 if (transform != nullptr) {
399 mat = (*transform);
400 } else {
401 mat = egg_bin->get_vertex_to_node();
402 }
403
404 // Now convert this vertex pool to a GeomVertexData.
405 PT(GeomVertexData) vertex_data =
406 make_vertex_data(render_state, vertex_pool, egg_bin, mat, blend_table,
407 is_dynamic, character_maker, has_overall_color);
408 nassertv(vertex_data != nullptr);
409
410 // And create a Geom to hold the primitives.
411 PT(Geom) geom = new Geom(vertex_data);
412
413 // Add each new primitive to the Geom.
414 Primitives::const_iterator pi;
415 for (pi = primitives.begin(); pi != primitives.end(); ++pi) {
416 PT(GeomPrimitive) primitive = (*pi);
417
418 if (primitive->is_indexed()) {
419 // Since we may have over-allocated while we were filling up the
420 // primitives, down-allocate now.
421 primitive->reserve_num_vertices(primitive->get_num_vertices());
422 }
423
424 geom->add_primitive(primitive);
425 }
426
427 // vertex_data->write(cerr); geom->write(cerr);
428 // render_state->_state->write(cerr, 0);
429
430 // Create a new GeomNode if we haven't already.
431 if (geom_node == nullptr) {
432 // Now, is our parent node a GeomNode, or just an ordinary PandaNode?
433 // If it's a GeomNode, we can add the new Geom directly to our parent;
434 // otherwise, we need to create a new node.
435 if (parent->is_geom_node() && !render_state->_hidden) {
436 geom_node = DCAST(GeomNode, parent);
437
438 } else {
439 geom_node = new GeomNode(egg_bin->get_name());
440 if (render_state->_hidden) {
441 parent->add_stashed(geom_node);
442 } else {
443 parent->add_child(geom_node);
444 }
445 }
446 }
447
448 CPT(RenderState) geom_state = render_state->_state;
449 if (has_overall_color) {
450 if (!overall_color.almost_equal(LColor(1.0f, 1.0f, 1.0f, 1.0f))) {
451 geom_state = geom_state->add_attrib(ColorAttrib::make_flat(overall_color), -1);
452 }
453 } else {
454 geom_state = geom_state->add_attrib(ColorAttrib::make_vertex(), -1);
455 }
456
457 geom_node->add_geom(geom, geom_state);
458 }
459 }
460
461 if (geom_node != nullptr && egg_show_normals) {
462 // Create some more geometry to visualize each normal.
463 for (vpi = vertex_pools.begin(); vpi != vertex_pools.end(); ++vpi) {
464 EggVertexPool *vertex_pool = (*vpi);
465 show_normals(vertex_pool, geom_node);
466 }
467 }

474CPT(TransformState) EggLoader::
475make_transform(const EggTransform *egg_transform) {
476 // We'll build up the transform componentwise, so we preserve any
477 // componentwise properties of the egg transform.
478
479 CPT(TransformState) ts = TransformState::make_identity();
480 int num_components = egg_transform->get_num_components();
481 for (int i = 0; i < num_components; i++) {
482 switch (egg_transform->get_component_type(i)) {
483 case EggTransform::CT_translate2d:
484 {
485 LVecBase2 trans2d(LCAST(PN_stdfloat, egg_transform->get_component_vec2(i)));
486 LVecBase3 trans3d(trans2d[0], trans2d[1], 0.0f);
487 ts = TransformState::make_pos(trans3d)->compose(ts);
488 }
489 break;
490
491 case EggTransform::CT_translate3d:
492 {
493 LVecBase3 trans3d(LCAST(PN_stdfloat, egg_transform->get_component_vec3(i)));
494 ts = TransformState::make_pos(trans3d)->compose(ts);
495 }
496 break;
497
498 case EggTransform::CT_rotate2d:
499 {
500 LRotation rot(LVector3(0.0f, 0.0f, 1.0f),
501 (PN_stdfloat)egg_transform->get_component_number(i));
502 ts = TransformState::make_quat(rot)->compose(ts);
503 }
504 break;
505
506 case EggTransform::CT_rotx:
507 {
508 LRotation rot(LVector3(1.0f, 0.0f, 0.0f),
509 (PN_stdfloat)egg_transform->get_component_number(i));
510 ts = TransformState::make_quat(rot)->compose(ts);
511 }
512 break;
513
514 case EggTransform::CT_roty:
515 {
516 LRotation rot(LVector3(0.0f, 1.0f, 0.0f),
517 (PN_stdfloat)egg_transform->get_component_number(i));
518 ts = TransformState::make_quat(rot)->compose(ts);
519 }
520 break;
521
522 case EggTransform::CT_rotz:
523 {
524 LRotation rot(LVector3(0.0f, 0.0f, 1.0f),
525 (PN_stdfloat)egg_transform->get_component_number(i));
526 ts = TransformState::make_quat(rot)->compose(ts);
527 }
528 break;
529
530 case EggTransform::CT_rotate3d:
531 {
532 LRotation rot(LCAST(PN_stdfloat, egg_transform->get_component_vec3(i)),
533 (PN_stdfloat)egg_transform->get_component_number(i));
534 ts = TransformState::make_quat(rot)->compose(ts);
535 }
536 break;
537
538 case EggTransform::CT_scale2d:
539 {
540 LVecBase2 scale2d(LCAST(PN_stdfloat, egg_transform->get_component_vec2(i)));
541 LVecBase3 scale3d(scale2d[0], scale2d[1], 1.0f);
542 ts = TransformState::make_scale(scale3d)->compose(ts);
543 }
544 break;
545
546 case EggTransform::CT_scale3d:
547 {
548 LVecBase3 scale3d(LCAST(PN_stdfloat, egg_transform->get_component_vec3(i)));
549 ts = TransformState::make_scale(scale3d)->compose(ts);
550 }
551 break;
552
553 case EggTransform::CT_uniform_scale:
554 {
555 PN_stdfloat scale = (PN_stdfloat)egg_transform->get_component_number(i);
556 ts = TransformState::make_scale(scale)->compose(ts);
557 }
558 break;
559
560 case EggTransform::CT_matrix3:
561 {
562 LMatrix3 m(LCAST(PN_stdfloat, egg_transform->get_component_mat3(i)));
563 LMatrix4 mat4(m(0, 0), m(0, 1), 0.0, m(0, 2),
564 m(1, 0), m(1, 1), 0.0, m(1, 2),
565 0.0, 0.0, 1.0, 0.0,
566 m(2, 0), m(2, 1), 0.0, m(2, 2));
567
568 ts = TransformState::make_mat(mat4)->compose(ts);
569 }
570 break;
571
572 case EggTransform::CT_matrix4:
573 {
574 LMatrix4 mat4(LCAST(PN_stdfloat, egg_transform->get_component_mat4(i)));
575 ts = TransformState::make_mat(mat4)->compose(ts);
576 }
577 break;
578
579 case EggTransform::CT_invalid:
580 nassertr(false, ts);
581 break;
582 }
583 }
584
585 if (ts->components_given()) {
586 return ts;
587 }
588
589 // Finally, we uniquify all the matrix-based TransformStates we create by
590 // complete matrix value. The TransformState class doesn't normally go this
591 // far, because of the cost of this additional uniquification step, but this
592 // is the egg loader so we don't mind spending a little bit of extra time
593 // here to get a more optimal result.
594 TransformStates::iterator tsi = _transform_states.insert(TransformStates::value_type(ts->get_mat(), ts)).first;
595
596 return (*tsi).second;
597}

603void EggLoader::
604show_normals(EggVertexPool *vertex_pool, GeomNode *geom_node) {
605 PT(GeomPrimitive) primitive = new GeomLines(Geom::UH_static);
606 CPT(GeomVertexFormat) format = GeomVertexFormat::get_v3c();
607 PT(GeomVertexData) vertex_data =
608 new GeomVertexData(vertex_pool->get_name(), format, Geom::UH_static);
609
610 GeomVertexWriter vertex(vertex_data, InternalName::get_vertex());
611 GeomVertexWriter color(vertex_data, InternalName::get_color());
612
613 EggVertexPool::const_iterator vi;
614 for (vi = vertex_pool->begin(); vi != vertex_pool->end(); ++vi) {
615 EggVertex *vert = (*vi);
616 LPoint3d pos = vert->get_pos3();
617
618 if (vert->has_normal()) {
619 vertex.add_data3d(pos);
620 vertex.add_data3d(pos + vert->get_normal() * egg_normal_scale);
621 color.add_data4(1.0f, 0.0f, 0.0f, 1.0f);
622 color.add_data4(1.0f, 0.0f, 0.0f, 1.0f);
623 primitive->add_next_vertices(2);
624 primitive->close_primitive();
625 }
626
627 // Also look for tangents and binormals in each texture coordinate set.
628 EggVertex::const_uv_iterator uvi;
629 for (uvi = vert->uv_begin(); uvi != vert->uv_end(); ++uvi) {
630 EggVertexUV *uv_obj = (*uvi);
631 if (uv_obj->has_tangent()) {
632 vertex.add_data3d(pos);
633 vertex.add_data3d(pos + uv_obj->get_tangent() * egg_normal_scale);
634 color.add_data4(0.0f, 1.0f, 0.0f, 1.0f);
635 color.add_data4(0.0f, 1.0f, 0.0f, 1.0f);
636 primitive->add_next_vertices(2);
637 primitive->close_primitive();
638 }
639 if (uv_obj->has_binormal()) {
640 vertex.add_data3d(pos);
641 vertex.add_data3d(pos + uv_obj->get_binormal() * egg_normal_scale);
642 color.add_data4(0.0f, 0.0f, 1.0f, 1.0f);
643 color.add_data4(0.0f, 0.0f, 1.0f, 1.0f);
644 primitive->add_next_vertices(2);
645 primitive->close_primitive();
646 }
647 }
648 }
649
650 PT(Geom) geom = new Geom(vertex_data);
651 geom->add_primitive(primitive);
652 geom_node->add_geom(geom);
653}

658void EggLoader::
659make_nurbs_curve(EggNurbsCurve *egg_curve, PandaNode *parent,
660 const LMatrix4d &mat) {
661 if (egg_load_old_curves) {
662 // Make a NurbsCurve instead of a RopeNode (old interface).
663 make_old_nurbs_curve(egg_curve, parent, mat);
664 return;
665 }
666
667 assert(parent != nullptr);
668 assert(!parent->is_geom_node());
669
670 PT(NurbsCurveEvaluator) nurbs = ::make_nurbs_curve(egg_curve, mat);
671 if (nurbs == nullptr) {
672 _error = true;
673 return;
674 }
675
676 /*
677 switch (egg_curve->get_curve_type()) {
678 case EggCurve::CT_xyz:
679 curve->set_curve_type(PCT_XYZ);
680 break;
681
682 case EggCurve::CT_hpr:
683 curve->set_curve_type(PCT_HPR);
684 break;
685
686 case EggCurve::CT_t:
687 curve->set_curve_type(PCT_T);
688 break;
689
690 default:
691 break;
692 }
693 */
694
695 PT(RopeNode) rope = new RopeNode(egg_curve->get_name());
696 rope->set_curve(nurbs);
697
698 // Respect the subdivision values in the egg file, if any.
699 if (egg_curve->get_subdiv() != 0) {
700 int subdiv_per_segment =
701 (int)((egg_curve->get_subdiv() + 0.5) / nurbs->get_num_segments());
702 rope->set_num_subdiv(max(subdiv_per_segment, 1));
703 }
704
705 const EggRenderState *render_state;
706 DCAST_INTO_V(render_state, egg_curve->get_user_data(EggRenderState::get_class_type()));
707 if (render_state->_hidden && egg_suppress_hidden) {
708 // Eat this primitive.
709 return;
710 }
711
712 rope->set_state(render_state->_state);
713 rope->set_uv_mode(RopeNode::UV_parametric);
714
715 if (egg_curve->has_vertex_color()) {
716 // If the curve had individual vertex color, enable it.
717 rope->set_use_vertex_color(true);
718 } else if (egg_curve->has_color()) {
719 // Otherwise, if the curve has overall color, apply it.

721 }

723 parent->add_child(rope);

733void EggLoader::
734make_old_nurbs_curve(EggNurbsCurve *egg_curve, PandaNode *parent,
735 const LMatrix4d &mat) {
736 assert(parent != nullptr);
737 assert(!parent->is_geom_node());
738
739 PT(ParametricCurve) curve;
740 curve = new NurbsCurve;
741
742 NurbsCurveInterface *nurbs = curve->get_nurbs_interface();
743 nassertv(nurbs != nullptr);
744
745 if (egg_curve->get_order() < 1 || egg_curve->get_order() > 4) {
746 egg2pg_cat.error()
747 << "Invalid NURBSCurve order for " << egg_curve->get_name() << ": "
748 << egg_curve->get_order() << "\n";
749 _error = true;
750 return;
751 }
752
753 nurbs->set_order(egg_curve->get_order());
754
755 EggPrimitive::const_iterator pi;
756 for (pi = egg_curve->begin(); pi != egg_curve->end(); ++pi) {
757 nurbs->append_cv(LCAST(PN_stdfloat, (*pi)->get_pos4() * mat));
758 }
759
760 int num_knots = egg_curve->get_num_knots();
761 if (num_knots != nurbs->get_num_knots()) {
762 egg2pg_cat.error()
763 << "Invalid NURBSCurve number of knots for "
764 << egg_curve->get_name() << ": got " << num_knots
765 << " knots, expected " << nurbs->get_num_knots() << "\n";
766 _error = true;
767 return;
768 }
769
770 for (int i = 0; i < num_knots; i++) {
771 nurbs->set_knot(i, egg_curve->get_knot(i));
772 }
773
774 switch (egg_curve->get_curve_type()) {
775 case EggCurve::CT_xyz:
776 curve->set_curve_type(PCT_XYZ);
777 break;
778
779 case EggCurve::CT_hpr:
780 curve->set_curve_type(PCT_HPR);
781 break;
782
783 case EggCurve::CT_t:
784 curve->set_curve_type(PCT_T);
785 break;
786
787 default:
788 break;
789 }
790 curve->set_name(egg_curve->get_name());
791
792 if (!curve->recompute()) {
793 egg2pg_cat.error()
794 << "Invalid NURBSCurve " << egg_curve->get_name() << "\n";
795 _error = true;
796 return;
797 }
798
799 parent->add_child(curve);
800}

805void EggLoader::
806make_nurbs_surface(EggNurbsSurface *egg_surface, PandaNode *parent,
807 const LMatrix4d &mat) {
808 assert(parent != nullptr);
809 assert(!parent->is_geom_node());
810
811 PT(NurbsSurfaceEvaluator) nurbs = ::make_nurbs_surface(egg_surface, mat);
812 if (nurbs == nullptr) {
813 _error = true;
814 return;
815 }
816
817 PT(SheetNode) sheet = new SheetNode(egg_surface->get_name());
818 sheet->set_surface(nurbs);
819
820 // Respect the subdivision values in the egg file, if any.
821 if (egg_surface->get_u_subdiv() != 0) {
822 int u_subdiv_per_segment =
823 (int)((egg_surface->get_u_subdiv() + 0.5) / nurbs->get_num_u_segments());
824 sheet->set_num_u_subdiv(max(u_subdiv_per_segment, 1));
825 }
826 if (egg_surface->get_v_subdiv() != 0) {
827 int v_subdiv_per_segment =
828 (int)((egg_surface->get_v_subdiv() + 0.5) / nurbs->get_num_v_segments());
829 sheet->set_num_v_subdiv(max(v_subdiv_per_segment, 1));
830 }
831
832 const EggRenderState *render_state;
833 DCAST_INTO_V(render_state, egg_surface->get_user_data(EggRenderState::get_class_type()));
834 if (render_state->_hidden && egg_suppress_hidden) {
835 // Eat this primitive.
836 return;
837 }
838
839 sheet->set_state(render_state->_state);
840
841 if (egg_surface->has_vertex_color()) {
842 // If the surface had individual vertex color, enable it.
843 sheet->set_use_vertex_color(true);
844 } else if (egg_surface->has_color()) {
845 // Otherwise, if the surface has overall color, apply it.
846 sheet->set_attrib(ColorAttrib::make_flat(egg_surface->get_color()));
847 }
848
849 parent->add_child(sheet);
850}

878bool EggLoader::
879load_texture(TextureDef &def, EggTexture *egg_tex) {
880 // Check to see if we should reduce the number of channels in the texture.
881 int wanted_channels = 0;
882 bool wanted_alpha = false;
883
884 switch (egg_tex->get_format()) {
885 case EggTexture::F_red:
886 case EggTexture::F_green:
887 case EggTexture::F_blue:
888 case EggTexture::F_alpha:
889 case EggTexture::F_luminance:
890 case EggTexture::F_sluminance:
891 wanted_channels = 1;
892 wanted_alpha = false;
893 break;
894
895 case EggTexture::F_luminance_alpha:
896 case EggTexture::F_luminance_alphamask:
897 case EggTexture::F_sluminance_alpha:
898 wanted_channels = 2;
899 wanted_alpha = true;
900 break;
901
902 case EggTexture::F_rgb:
903 case EggTexture::F_rgb12:
904 case EggTexture::F_rgb8:
905 case EggTexture::F_rgb5:
906 case EggTexture::F_rgb332:
907 case EggTexture::F_srgb:
908 wanted_channels = 3;
909 wanted_alpha = false;
910 break;
911
912 case EggTexture::F_rgba:
913 case EggTexture::F_rgbm:
914 case EggTexture::F_rgba12:
915 case EggTexture::F_rgba8:
916 case EggTexture::F_rgba4:
917 case EggTexture::F_rgba5:
918 case EggTexture::F_srgb_alpha:
919 wanted_channels = 4;
920 wanted_alpha = true;
921 break;
922
923 case EggTexture::F_unspecified:
924 wanted_alpha = egg_tex->has_alpha_filename();
925 }
926
927 // Since some properties of the textures are inferred from the texture files
928 // themselves (if the properties are not explicitly specified in the egg
929 // file), then we add the textures as dependents for the egg file.
930 if (_record != nullptr) {
931 _record->add_dependent_file(egg_tex->get_fullpath());
932 if (egg_tex->has_alpha_filename() && wanted_alpha) {
933 _record->add_dependent_file(egg_tex->get_alpha_fullpath());
934 }
935 }
936
937 // By convention, the egg loader will preload the simple texture images.
938 LoaderOptions options;
939
940 if (egg_preload_simple_textures) {
941 options.set_texture_flags(options.get_texture_flags() | LoaderOptions::TF_preload_simple);
942 }
943
944 if (!egg_ignore_filters && !egg_ignore_mipmaps) {
945 switch (egg_tex->get_minfilter()) {
946 case EggTexture::FT_nearest:
947 case EggTexture::FT_linear:
948 case EggTexture::FT_unspecified:
949 break;
950
951 case EggTexture::FT_nearest_mipmap_nearest:
952 case EggTexture::FT_linear_mipmap_nearest:
953 case EggTexture::FT_nearest_mipmap_linear:
954 case EggTexture::FT_linear_mipmap_linear:
955 options.set_texture_flags(options.get_texture_flags() | LoaderOptions::TF_generate_mipmaps);
956 }
957 }
958
959 if (egg_tex->get_multiview()) {
960 options.set_texture_flags(options.get_texture_flags() | LoaderOptions::TF_multiview);
961 if (egg_tex->has_num_views()) {
962 options.set_texture_num_views(egg_tex->get_num_views());
963 }
964 }
965
966 // Allow the texture loader to pre-compress the texture.
967 if (egg_tex->get_compression_mode() == EggTexture::CM_on) {
968 options.set_texture_flags(options.get_texture_flags() | LoaderOptions::TF_allow_compression);
969 }
970
971 if (egg_force_srgb_textures) {
972 options.set_texture_flags(options.get_texture_flags() | LoaderOptions::TF_force_srgb);
973 }
974
975 //The following code sets up all the options for the textures
976 //so that they can be differentiated later in texturePool
977 SamplerState sampler;
978 set_up_loader_options(egg_tex, options, sampler);
979
980 PT(Texture) tex;
981 switch (egg_tex->get_texture_type()) {
982 case EggTexture::TT_unspecified:
983 case EggTexture::TT_1d_texture:
984 options.set_texture_flags(options.get_texture_flags() | LoaderOptions::TF_allow_1d);
985 // Fall through.
986
987 case EggTexture::TT_2d_texture:
988 if (egg_tex->has_alpha_filename() && wanted_alpha) {
989 tex = TexturePool::load_texture(egg_tex->get_fullpath(),
990 egg_tex->get_alpha_fullpath(),
991 wanted_channels,
992 egg_tex->get_alpha_file_channel(),
993 egg_tex->get_read_mipmaps(),
994 options, sampler);
995 } else {
996 tex = TexturePool::load_texture(egg_tex->get_fullpath(),
997 wanted_channels,
998 egg_tex->get_read_mipmaps(),
999 options, sampler);
1000 }
1001 break;
1002
1003 case EggTexture::TT_3d_texture:
1004 tex = TexturePool::load_3d_texture(egg_tex->get_fullpath(),
1005 egg_tex->get_read_mipmaps(),
1006 options, sampler);
1007 break;
1008
1009 case EggTexture::TT_cube_map:
1010 tex = TexturePool::load_cube_map(egg_tex->get_fullpath(),
1011 egg_tex->get_read_mipmaps(),
1012 options, sampler);
1013 break;
1014 }
1015
1016 if (tex == nullptr) {
1017 return false;
1018 }
1019
1020 // Record the original filenames in the textures (as loaded from the egg
1021 // file). These filenames will be written back to the bam file if the bam
1022 // file is written out.
1023 tex->set_filename(egg_tex->get_filename());
1024 if (egg_tex->has_alpha_filename() && wanted_alpha) {
1025 tex->set_alpha_filename(egg_tex->get_alpha_filename());
1026 }
1027
1028 // See if there is some egg data hanging on the texture. In particular, the
1029 // TxaFileFilter might have left that here for us.
1030 TypedReferenceCount *aux = tex->get_aux_data("egg");
1031 if (aux != nullptr &&
1032 aux->is_of_type(EggTexture::get_class_type())) {
1033 EggTexture *aux_egg_tex = DCAST(EggTexture, aux);
1034
1035 if (aux_egg_tex->get_alpha_mode() != EggTexture::AM_unspecified) {
1036 egg_tex->set_alpha_mode(aux_egg_tex->get_alpha_mode());
1037 }
1038 if (aux_egg_tex->get_format() != EggTexture::F_unspecified) {
1039 egg_tex->set_format(aux_egg_tex->get_format());
1040 }
1041 if (aux_egg_tex->get_minfilter() != EggTexture::FT_unspecified) {
1042 egg_tex->set_minfilter(aux_egg_tex->get_minfilter());
1043 }
1044 if (aux_egg_tex->get_magfilter() != EggTexture::FT_unspecified) {
1045 egg_tex->set_magfilter(aux_egg_tex->get_magfilter());
1046 }
1047 if (aux_egg_tex->has_anisotropic_degree()) {
1048 egg_tex->set_anisotropic_degree(aux_egg_tex->get_anisotropic_degree());
1049 }
1050 }
1051 check_texture_attributes(tex, sampler, egg_tex);
1052
1053 // Make a texture stage for the texture.
1054 PT(TextureStage) stage = make_texture_stage(egg_tex);
1055 def._texture = DCAST(TextureAttrib, TextureAttrib::make())->add_on_stage(stage, tex);
1056 def._stage = stage;
1057 def._egg_tex = egg_tex;
1058
1059 return true;
1060}

1701void EggLoader::
1702separate_switches(EggNode *egg_node) {
1703 bool parent_has_switch = false;
1704 if (egg_node->is_of_type(EggGroup::get_class_type())) {
1705 EggGroup *egg_group = DCAST(EggGroup, egg_node);
1706 parent_has_switch = egg_group->get_switch_flag();
1707 }
1708
1709 if (egg_node->is_of_type(EggGroupNode::get_class_type())) {
1710 EggGroupNode *egg_group = DCAST(EggGroupNode, egg_node);
1711
1712 EggGroupNode::iterator ci;
1713 ci = egg_group->begin();
1714 while (ci != egg_group->end()) {
1715 EggGroupNode::iterator cnext;
1716 cnext = ci;
1717 ++cnext;
1718
1719 PT(EggNode) child = (*ci);
1720 if (parent_has_switch &&
1721 child->is_of_type(EggPrimitive::get_class_type())) {
1722 // Move this child under a new node.
1723 PT(EggGroup) new_group = new EggGroup(child->get_name());
1724 egg_group->replace(ci, new_group.p());
1725 new_group->add_child(child);
1726 }
1727
1728 separate_switches(child);
1729
1730 ci = cnext;
1731 }
1732 }

1740void EggLoader::
1741emulate_bface(EggNode *egg_node) {
1742 if (egg_node->is_of_type(EggGroupNode::get_class_type())) {
1743 EggGroupNode *egg_group = DCAST(EggGroupNode, egg_node);
1744 PT(EggGroupNode) dup_prims = new EggGroupNode;
1745
1746 EggGroupNode::iterator ci;
1747 for (ci = egg_group->begin(); ci != egg_group->end(); ++ci) {
1748 PT(EggNode) child = (*ci);
1749 if (child->is_of_type(EggPolygon::get_class_type())) {
1750 EggPolygon *poly = DCAST(EggPolygon, child);
1751 if (poly->get_bface_flag()) {
1752 poly->set_bface_flag(false);
1753
1754 PT(EggPolygon) dup_poly = new EggPolygon(*poly);
1755 dup_poly->reverse_vertex_ordering();
1756 if (dup_poly->has_normal()) {
1757 dup_poly->set_normal(-dup_poly->get_normal());
1758 }
1759
1760 // Also reverse the normal on any vertices.
1761 EggPolygon::iterator vi;
1762 for (vi = dup_poly->begin(); vi != dup_poly->end(); ++vi) {
1763 EggVertex *vertex = (*vi);
1764 if (vertex->has_normal()) {
1765 EggVertex dup_vertex(*vertex);
1766 dup_vertex.set_normal(-dup_vertex.get_normal());
1767 EggVertex *new_vertex = vertex->get_pool()->create_unique_vertex(dup_vertex);
1768 if (new_vertex != vertex) {
1769 new_vertex->copy_grefs_from(*vertex);
1770 dup_poly->replace(vi, new_vertex);
1771 }
1772 }
1773 }
1774 dup_prims->add_child(dup_poly);
1775 }
1776 }
1777
1778 emulate_bface(child);
1779 }
1780
1781 // Now that we've iterated through all the children, add in any duplicated
1782 // polygons we generated.
1783 egg_group->steal_children(*dup_prims);
1784 }

1855PandaNode *EggLoader::
1856make_lod(EggBin *egg_bin, PandaNode *parent) {
1857 PT(LODNode) lod_node = LODNode::make_default_lod(egg_bin->get_name());
1858
1859 pvector<LODInstance> instances;
1860
1861 EggGroup::const_iterator ci;
1862 for (ci = egg_bin->begin(); ci != egg_bin->end(); ++ci) {
1863 LODInstance instance(*ci);
1864 instances.push_back(instance);
1865 }
1866
1867 // Now that we've created all of our children, put them in the proper order
1868 // and tell the LOD node about them.
1869 std::sort(instances.begin(), instances.end());
1870
1871 if (!instances.empty()) {
1872 // Set up the LOD node's center. All of the children should have the same
1873 // center, because that's how we binned them.
1874 lod_node->set_center(LCAST(PN_stdfloat, instances[0]._d->_center));
1875 }
1876
1877 for (size_t i = 0; i < instances.size(); i++) {
1878 // Create the children in the proper order within the scene graph.
1879 const LODInstance &instance = instances[i];
1880 make_node(instance._egg_node, lod_node);
1881
1882 // All of the children should have the same center, because that's how we
1883 // binned them.
1884 nassertr(lod_node->get_center().almost_equal
1885 (LCAST(PN_stdfloat, instance._d->_center), 0.01), nullptr);
1886
1887 // Tell the LOD node about this child's switching distances.
1888 lod_node->add_switch(instance._d->_switch_in, instance._d->_switch_out);
1889 }
1890
1891 _groups[egg_bin] = lod_node;
1892 return create_group_arc(egg_bin, parent, lod_node);
1893}

1898PandaNode *EggLoader::
1899make_node(EggGroup *egg_group, PandaNode *parent) {
1900 PT(PandaNode) node = nullptr;
1901
1902 if (egg_group->get_dart_type() != EggGroup::DT_none) {
1903 // A group with the <Dart> flag set means to create a character.
1904 bool structured = (egg_group->get_dart_type() == EggGroup::DT_structured);
1905
1906 CharacterMaker char_maker(egg_group, *this, structured);
1907
1908 node = char_maker.make_node();
1909 if(structured) {
1910 // we're going to generate the rest of the children normally except
1911 // we'll be making dynamic geometry
1912 _dynamic_override = true;
1913 _dynamic_override_char_maker = &char_maker;
1914 EggGroup::const_iterator ci;
1915 for (ci = egg_group->begin(); ci != egg_group->end(); ++ci) {
1916 make_node(*ci, node);
1917 }
1918 _dynamic_override_char_maker = nullptr;
1919 _dynamic_override = false;
1920 }
1921
1922 } else if (egg_group->get_cs_type() != EggGroup::CST_none) {
1923 // A collision group: create collision geometry.
1924 node = new CollisionNode(egg_group->get_name());
1925
1926 // Piggy-back the desired transform to apply onto the node, since we can't
1927 // break the ABI in 1.10.
1928 node->set_transform(TransformState::make_mat(LCAST(PN_stdfloat, egg_group->get_vertex_to_node())));
1929 make_collision_solids(egg_group, egg_group, (CollisionNode *)node.p());
1930 node->clear_transform();
1931
1932 if ((egg_group->get_collide_flags() & EggGroup::CF_keep) != 0) {
1933 // If we also specified to keep the geometry, continue the traversal.
1934 // In this case, we create a new PandaNode to be the parent of the
1935 // visible geometry and the collision geometry.
1936 PandaNode *combined = new PandaNode("");
1937 parent->add_child(combined);
1938 combined->add_child(node);
1939 node = combined;
1940
1941 EggGroup::const_iterator ci;
1942 for (ci = egg_group->begin(); ci != egg_group->end(); ++ci) {
1943 make_node(*ci, combined);
1944 }
1945 }
1946
1947 node = create_group_arc(egg_group, parent, node);
1948 return node;
1949
1950 } else if (egg_group->get_portal_flag()) {
1951 // Create a portal instead of a regular polyset. Scan the children of
1952 // this node looking for a polygon, similar to the collision polygon case,
1953 // above.
1954 PortalNode *pnode = new PortalNode(egg_group->get_name());
1955 node = pnode;
1956
1957 set_portal_polygon(egg_group, pnode);
1958 if (pnode->get_num_vertices() == 0) {
1959 egg2pg_cat.warning()
1960 << "Portal " << egg_group->get_name() << " has no vertices!\n";
1961 }
1962
1963 } else if (egg_group->get_occluder_flag()) {
1964 // Create an occluder instead of a regular polyset. Scan the children of
1965 // this node looking for a polygon, the same as the portal polygon case,
1966 // above.
1967 OccluderNode *pnode = new OccluderNode(egg_group->get_name());
1968 node = pnode;
1969
1970 set_occluder_polygon(egg_group, pnode);
1971 if (pnode->get_num_vertices() == 0) {
1972 egg2pg_cat.warning()
1973 << "Occluder " << egg_group->get_name() << " has no vertices!\n";
1974 }
1975
1976 } else if (egg_group->get_polylight_flag()) {
1977 // Create a polylight instead of a regular polyset. use make_sphere to
1978 // get the center, radius and color egg2pg_cat.debug() << "polylight
1979 // node\n";
1980 LPoint3 center;
1981 LColor color;
1982 PN_stdfloat radius;
1983
1984 if (!make_sphere(egg_group, EggGroup::CF_none, center, radius, color)) {
1985 egg2pg_cat.warning()
1986 << "Polylight " << egg_group->get_name() << " make_sphere failed!\n";
1987 }
1988 PolylightNode *pnode = new PolylightNode(egg_group->get_name());
1989 pnode->set_pos(center);
1990 pnode->set_color(color);
1991 pnode->set_radius(radius);
1992
1993 pnode->xform(LCAST(PN_stdfloat, egg_group->get_vertex_to_node()));
1994
1995 node = pnode;
1996
1997 } else if (egg_group->get_switch_flag()) {
1998 if (egg_group->get_switch_fps() != 0.0) {
1999 // Create a sequence node.
2000 node = new SequenceNode(egg_group->get_name());
2001 ((SequenceNode *)node.p())->set_frame_rate(egg_group->get_switch_fps());
2002 _sequences.insert(node);
2003 } else {
2004 // Create a switch node.
2005 node = new SwitchNode(egg_group->get_name());
2006 }
2007
2008 EggGroup::const_iterator ci;
2009 for (ci = egg_group->begin(); ci != egg_group->end(); ++ci) {
2010 make_node(*ci, node);
2011 }
2012 } else if (egg_group->has_scrolling_uvs()) {
2013 node = new UvScrollNode(egg_group->get_name(), egg_group->get_scroll_u(), egg_group->get_scroll_v(), egg_group->get_scroll_w(), egg_group->get_scroll_r());
2014
2015 EggGroup::const_iterator ci;
2016 for (ci = egg_group->begin(); ci != egg_group->end(); ++ci) {
2017 make_node(*ci, node);
2018 }
2019
2020 } else if (egg_group->get_model_flag() || egg_group->has_dcs_type()) {
2021 // A model or DCS flag; create a model node.
2022 node = new ModelNode(egg_group->get_name());
2023 switch (egg_group->get_dcs_type()) {
2024 case EggGroup::DC_net:
2025 DCAST(ModelNode, node)->set_preserve_transform(ModelNode::PT_net);
2026 break;
2027
2028 case EggGroup::DC_no_touch:
2029 DCAST(ModelNode, node)->set_preserve_transform(ModelNode::PT_no_touch);
2030 break;
2031
2032 case EggGroup::DC_local:
2033 case EggGroup::DC_default:
2034 DCAST(ModelNode, node)->set_preserve_transform(ModelNode::PT_local);
2035 break;
2036
2037 case EggGroup::DC_none:
2038 case EggGroup::DC_unspecified:
2039 break;
2040 }
2041
2042 EggGroup::const_iterator ci;
2043 for (ci = egg_group->begin(); ci != egg_group->end(); ++ci) {
2044 make_node(*ci, node);
2045 }
2046
2047 } else {

2049 // the children of this group are polysets, anticipate this for the

2578PT(TransformBlendTable) EggLoader::
2579make_blend_table(EggVertexPool *vertex_pool, EggNode *primitive_home,
2580 CharacterMaker *character_maker) {
2581 PT(TransformBlendTable) blend_table;
2582 blend_table = new TransformBlendTable;
2583 blend_table->set_rows(SparseArray::lower_on(vertex_pool->size()));
2584
2585 EggVertexPool::const_iterator vi;
2586 for (vi = vertex_pool->begin(); vi != vertex_pool->end(); ++vi) {
2587 EggVertex *vertex = (*vi);
2588
2589 // Figure out the transforms affecting this particular vertex.
2590 TransformBlend blend;
2591 if (vertex->gref_size() == 0) {
2592 // If the vertex has no explicit membership, it belongs right where it
2593 // is.
2594 PT(VertexTransform) vt = character_maker->egg_to_transform(primitive_home);
2595 nassertr(vt != nullptr, nullptr);
2596 blend.add_transform(vt, 1.0f);
2597 } else {
2598 // If the vertex does have an explicit membership, ignore its parentage
2599 // and assign it where it wants to be.
2600 double quantize = egg_vertex_membership_quantize;
2601 EggVertex::GroupRef::const_iterator gri;
2602 for (gri = vertex->gref_begin(); gri != vertex->gref_end(); ++gri) {
2603 EggGroup *egg_joint = (*gri);
2604 double membership = egg_joint->get_vertex_membership(vertex);
2605 if (quantize != 0.0) {
2606 membership = cfloor(membership / quantize + 0.5) * quantize;
2607 }
2608
2609 PT(VertexTransform) vt = character_maker->egg_to_transform(egg_joint);
2610 nassertr(vt != nullptr, nullptr);
2611 blend.add_transform(vt, membership);
2612 }
2613 }
2614 if (egg_vertex_max_num_joints >= 0) {
2615 blend.limit_transforms(egg_vertex_max_num_joints);
2616 }
2617 blend.normalize_weights();
2618
2619 int table_index = blend_table->add_blend(blend);
2620
2621 // We take advantage of the "external index" field of the EggVertex to
2622 // temporarily store the transform blend index.
2623 vertex->set_external_index(table_index);
2624 }

2650void EggLoader::
2651make_primitive(const EggRenderState *render_state, EggPrimitive *egg_prim,
2652 EggLoader::UniquePrimitives &unique_primitives,
2653 EggLoader::Primitives &primitives,
2654 bool has_overall_color, const LColor &overall_color) {
2655 PT(GeomPrimitive) primitive;
2656 if (egg_prim->is_of_type(EggPolygon::get_class_type())) {
2657 if (egg_prim->size() == 3) {
2658 primitive = new GeomTriangles(Geom::UH_static);
2659 }
2660
2661 } else if (egg_prim->is_of_type(EggTriangleStrip::get_class_type())) {
2662 primitive = new GeomTristrips(Geom::UH_static);
2663
2664 } else if (egg_prim->is_of_type(EggTriangleFan::get_class_type())) {
2665 primitive = new GeomTrifans(Geom::UH_static);
2666
2667 } else if (egg_prim->is_of_type(EggLine::get_class_type())) {
2668 if (egg_prim->size() == 2) {
2669 primitive = new GeomLines(Geom::UH_static);
2670 } else {
2671 primitive = new GeomLinestrips(Geom::UH_static);
2672 }
2673
2674 } else if (egg_prim->is_of_type(EggPoint::get_class_type())) {
2675 primitive = new GeomPoints(Geom::UH_static);
2676
2677 } else if (egg_prim->is_of_type(EggPatch::get_class_type())) {
2678 int num_vertices = egg_prim->size();
2679 primitive = new GeomPatches(num_vertices, Geom::UH_static);
2680 }
2681
2682 if (primitive == nullptr) {
2683 // Don't know how to make this kind of primitive.
2684 egg2pg_cat.warning()
2685 << "Ignoring " << egg_prim->get_type() << "\n";
2686 return;
2687 }
2688
2689 if (render_state->_flat_shaded) {
2690 primitive->set_shade_model(GeomPrimitive::SM_flat_first_vertex);
2691
2692 } else if (egg_prim->get_shading() == EggPrimitive::S_overall) {
2693 primitive->set_shade_model(GeomPrimitive::SM_uniform);
2694
2695 } else {
2696 primitive->set_shade_model(GeomPrimitive::SM_smooth);
2697 }
2698
2699 // Insert the primitive into the set, but if we already have a primitive of
2700 // that type, reset the pointer to that one instead.
2701 PrimitiveUnifier pu(primitive);
2702 std::pair<UniquePrimitives::iterator, bool> result =
2703 unique_primitives.insert(UniquePrimitives::value_type(pu, primitive));
2704
2705 if (result.second) {
2706 // This was the first primitive of this type. Store it.
2707 primitives.push_back(primitive);
2708
2709 if (egg2pg_cat.is_debug()) {
2710 egg2pg_cat.debug()
2711 << "First primitive of type " << primitive->get_type()
2712 << ": " << primitive << "\n";
2713 }
2714 }
2715
2716 GeomPrimitive *orig_prim = (*result.first).second;
2717
2718 // Make sure we don't try to put more than egg_max_indices into any one
2719 // GeomPrimitive.
2720 if (orig_prim->get_num_vertices() + egg_prim->size() <= (unsigned int)egg_max_indices) {
2721 primitive = orig_prim;
2722
2723 } else if (orig_prim != primitive) {
2724 // If the old primitive is full, keep the new primitive from now on.
2725 (*result.first).second = primitive;
2726
2727 if (egg2pg_cat.is_debug()) {
2728 egg2pg_cat.debug()
2729 << "Next primitive of type " << primitive->get_type()
2730 << ": " << primitive << "\n";
2731 }
2732 primitives.push_back(primitive);
2733 }
2734
2735 // Now add the vertices.
2736 EggPrimitive::const_iterator vi;
2737 for (vi = egg_prim->begin(); vi != egg_prim->end(); ++vi) {
2738 primitive->add_vertex((*vi)->get_index());
2739 }
2740 primitive->close_primitive();
2741}

2825bool EggLoader::
2826make_sphere(EggGroup *egg_group, EggGroup::CollideFlags flags,
2827 LPoint3 &center, PN_stdfloat &radius, LColor &color) {
2828 EggGroup *geom_group = find_collision_geometry(egg_group, flags);
2829 if (geom_group != nullptr) {
2830 // Collect all of the vertices.
2831 pset<EggVertex *> vertices;
2832
2833 EggGroup::const_iterator ci;
2834 for (ci = geom_group->begin(); ci != geom_group->end(); ++ci) {
2835 if ((*ci)->is_of_type(EggPrimitive::get_class_type())) {
2836 EggPrimitive *prim = DCAST(EggPrimitive, *ci);
2837 EggPrimitive::const_iterator pi;
2838 for (pi = prim->begin(); pi != prim->end(); ++pi) {
2839 vertices.insert(*pi);
2840 }
2841 }
2842 }
2843
2844 // Now average together all of the vertices to get a center.
2845 int num_vertices = 0;
2846 LPoint3d d_center(0.0, 0.0, 0.0);
2847 pset<EggVertex *>::const_iterator vi;
2848
2849 for (vi = vertices.begin(); vi != vertices.end(); ++vi) {
2850 EggVertex *vtx = (*vi);
2851 d_center += vtx->get_pos3();
2852 num_vertices++;
2853 }
2854
2855 if (num_vertices > 0) {
2856 d_center /= (double)num_vertices;
2857 // egg2pg_cat.debug() << "make_sphere d_center: " << d_center << "\n";
2858
2859 // And the furthest vertex determines the radius.
2860 double radius2 = 0.0;
2861 for (vi = vertices.begin(); vi != vertices.end(); ++vi) {
2862 EggVertex *vtx = (*vi);
2863 LPoint3d p3 = vtx->get_pos3();
2864 LVector3d v = p3 - d_center;
2865 radius2 = max(radius2, v.length_squared());
2866 }
2867
2868 center = LCAST(PN_stdfloat, d_center);
2869 radius = sqrtf(radius2);
2870
2871 // egg2pg_cat.debug() << "make_sphere radius: " << radius << "\n";
2872 vi = vertices.begin();
2873 EggVertex *clr_vtx = (*vi);
2874 color = clr_vtx->get_color();
2875 return true;
2876 }
2877 }
2878 return false;
2879}

2885bool EggLoader::
2886make_box(EggGroup *egg_group, EggGroup::CollideFlags flags,
2887 const LMatrix4 &xform, LPoint3 &min_p, LPoint3 &max_p) {
2888 EggGroup *geom_group = find_collision_geometry(egg_group, flags);
2889 if (geom_group != nullptr) {
2890 // Collect all of the vertices.
2891 pset<EggVertex *> vertices;
2892
2893 EggGroup::const_iterator ci;
2894 for (ci = geom_group->begin(); ci != geom_group->end(); ++ci) {
2895 if ((*ci)->is_of_type(EggPrimitive::get_class_type())) {
2896 EggPrimitive *prim = DCAST(EggPrimitive, *ci);
2897 EggPrimitive::const_iterator pi;
2898 for (pi = prim->begin(); pi != prim->end(); ++pi) {
2899 vertices.insert(*pi);
2900 }
2901 }
2902 }
2903
2904 // Now find the minmax points
2905 pset<EggVertex *>::const_iterator vi;
2906 vi = vertices.begin();
2907
2908 if (vi == vertices.end()) {
2909 // No vertices, no bounding box.
2910 min_p.set(0, 0, 0);
2911 max_p.set(0, 0, 0);
2912 return false;
2913 }
2914
2915 EggVertex *vertex = (*vi);
2916 LPoint3 min_pd = LCAST(PN_stdfloat, vertex->get_pos3()) * xform;
2917 LPoint3 max_pd = min_pd;
2918
2919 for (++vi; vi != vertices.end(); ++vi) {
2920 vertex = (*vi);
2921 LPoint3 pos = LCAST(PN_stdfloat, vertex->get_pos3()) * xform;
2922 min_pd.set(min(min_pd[0], pos[0]),
2923 min(min_pd[1], pos[1]),
2924 min(min_pd[2], pos[2]));
2925 max_pd.set(max(max_pd[0], pos[0]),
2926 max(max_pd[1], pos[1]),
2927 max(max_pd[2], pos[2]));
2928 }
2929
2930 min_p = min_pd;
2931 max_p = max_pd;
2932 return (min_pd != max_pd);
2933 }
2934 return false;
2935}

3177void EggLoader::
3178make_collision_capsule(EggGroup *egg_group, CollisionNode *cnode,
3179 EggGroup::CollideFlags flags) {
3180 EggGroup *geom_group = find_collision_geometry(egg_group, flags);
3181 if (geom_group != nullptr) {
3182 // Collect all of the vertices.
3183 pset<EggVertex *> vertices;
3184
3185 EggGroup::const_iterator ci;
3186 for (ci = geom_group->begin(); ci != geom_group->end(); ++ci) {
3187 if ((*ci)->is_of_type(EggPrimitive::get_class_type())) {
3188 EggPrimitive *prim = DCAST(EggPrimitive, *ci);
3189 EggPrimitive::const_iterator pi;
3190 for (pi = prim->begin(); pi != prim->end(); ++pi) {
3191 vertices.insert(*pi);
3192 }
3193 }
3194 }
3195
3196 // Now store the 3-d values in a vector for convenient access (and also
3197 // determine the centroid). We compute this in node space.
3198 size_t num_vertices = vertices.size();
3199 if (num_vertices != 0) {
3200 pvector<LPoint3d> vpos;
3201 vpos.reserve(num_vertices);
3202
3203 LPoint3d center(0.0, 0.0, 0.0);
3204 pset<EggVertex *>::const_iterator vi;
3205 for (vi = vertices.begin(); vi != vertices.end(); ++vi) {
3206 EggVertex *vtx = (*vi);
3207 const LPoint3d &pos = vtx->get_pos3();
3208 vpos.push_back(pos);
3209 center += pos;
3210 }
3211 center /= (double)num_vertices;
3212
3213 // Now that we have the centroid, we have to try to figure out the
3214 // cylinder's major axis. Start by finding a point farthest from the
3215 // centroid.
3216 size_t i;
3217 double radius2 = 0.0;
3218 LPoint3d far_a = center;
3219 for (i = 0; i < num_vertices; i++) {
3220 double dist2 = (vpos[i] - center).length_squared();
3221 if (dist2 > radius2) {
3222 radius2 = dist2;
3223 far_a = vpos[i];
3224 }
3225 }
3226
3227 // The point we have found above, far_a, must be one one of the endcaps.
3228 // Now find another point, far_b, that is the farthest from far_a. This
3229 // will be a point on the other endcap.
3230 radius2 = 0.0;
3231 LPoint3d far_b = center;
3232 for (i = 0; i < num_vertices; i++) {
3233 double dist2 = (vpos[i] - far_a).length_squared();
3234 if (dist2 > radius2) {
3235 radius2 = dist2;
3236 far_b = vpos[i];
3237 }
3238 }
3239
3240 // Now we have far_a and far_b, one point on each endcap. However,
3241 // these points are not necessarily centered on the endcaps, so we
3242 // haven't figured out the cylinder's axis yet (the line between far_a
3243 // and far_b will probably pass through the cylinder at an angle).
3244
3245 // So we still need to determine the full set of points in each endcap.
3246 // To do this, we pass back through the set of points, categorizing each
3247 // point into either "endcap a" or "endcap b". We also leave a hefty
3248 // chunk of points in the middle uncategorized; this helps prevent us
3249 // from getting a little bit lopsided with points near the middle that
3250 // may appear to be closer to the wrong endcap.
3251 LPoint3d cap_a_center(0.0, 0.0, 0.0);
3252 LPoint3d cap_b_center(0.0, 0.0, 0.0);
3253 int num_a = 0;
3254 int num_b = 0;
3255
3256 // This is the threshold length; points farther away from the center
3257 // than this are deemed to be in one endcap or the other.
3258 double center_length = (far_a - far_b).length() / 4.0;
3259 double center_length2 = center_length * center_length;
3260
3261 for (i = 0; i < num_vertices; i++) {
3262 double dist2 = (vpos[i] - center).length_squared();
3263 if (dist2 > center_length2) {
3264 // This point is farther away from the center than center_length;
3265 // therefore it belongs in an endcap.
3266 double dist_a2 = (vpos[i] - far_a).length_squared();
3267 double dist_b2 = (vpos[i] - far_b).length_squared();
3268 if (dist_a2 < dist_b2) {
3269 // It's in endcap a.
3270 cap_a_center += vpos[i];
3271 num_a++;
3272 } else {
3273 // It's in endcap b.
3274 cap_b_center += vpos[i];
3275 num_b++;
3276 }
3277 }
3278 }
3279
3280 if (num_a > 0 && num_b > 0) {
3281 cap_a_center /= (double)num_a;
3282 cap_b_center /= (double)num_b;
3283
3284
3285 // Now we finally have the major axis of the cylinder.
3286 LVector3d axis = cap_b_center - cap_a_center;
3287 axis.normalize();
3288
3289 // If the axis is *almost* parallel with a major axis, assume it is
3290 // meant to be exactly parallel.
3291 if (IS_THRESHOLD_ZERO(axis[0], 0.01)) {
3292 axis[0] = 0.0;
3293 }
3294 if (IS_THRESHOLD_ZERO(axis[1], 0.01)) {
3295 axis[1] = 0.0;
3296 }
3297 if (IS_THRESHOLD_ZERO(axis[2], 0.01)) {
3298 axis[2] = 0.0;
3299 }
3300 axis.normalize();
3301

3303 // axis, and the origin is the center. This is very similar to the
3304 // CollisionCapsule's idea of its canonical orientation (although not
3305 // exactly the same, since it is centered on the origin instead of
3306 // having point_a on the origin). It makes it easier to determine the
3307 // length and radius of the cylinder.
3308 LMatrix4d mat;
3309 look_at(mat, axis, LVector3d(0.0, 0.0, 1.0), CS_zup_right);
3310 mat.set_row(3, center);
3311 LMatrix4d inv_mat;
3312 inv_mat.invert_from(mat);
3313
3314 for (i = 0; i < num_vertices; i++) {
3315 vpos[i] = vpos[i] * inv_mat;
3316 }

3319
3320 // Now determine the radius.
3321 for (i = 0; i < num_vertices; i++) {
3322 LVector2d v(vpos[i][0], vpos[i][2]);
3323 double radius2 = v.length_squared();

3330 // the radius first because we want the round endcaps to enclose all
