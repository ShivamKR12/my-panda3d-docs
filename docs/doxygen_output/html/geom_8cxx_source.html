213int Geom::
214make_nonindexed(bool composite_only) {
215 Thread *current_thread = Thread::get_current_thread();
216 int num_changed = 0;
217
218 CDWriter cdata(_cycler, true, current_thread);
219 CPT(GeomVertexData) orig_data = cdata->_data.get_read_pointer(current_thread);
220 PT(GeomVertexData) new_data = new GeomVertexData(*orig_data);
221 new_data->clear_rows();
222
223#ifndef NDEBUG
224 bool all_is_valid = true;
225#endif
226 Primitives::iterator pi;
227 Primitives new_prims;
228 new_prims.reserve(cdata->_primitives.size());
229 for (pi = cdata->_primitives.begin(); pi != cdata->_primitives.end(); ++pi) {
230 PT(GeomPrimitive) primitive = (*pi).get_read_pointer(current_thread)->make_copy();
231 new_prims.push_back(primitive.p());
232
233 // GeomPoints are considered "composite" for the purposes of making
234 // nonindexed, since there's no particular advantage to having indexed
235 // points (as opposed to, say, indexed triangles or indexed lines).
236 if (primitive->is_indexed() &&
237 (primitive->is_composite() ||
238 primitive->is_exact_type(GeomPoints::get_class_type()) ||
239 !composite_only)) {
240 primitive->make_nonindexed(new_data, orig_data);
241 ++num_changed;
242 } else {
243 // If it's a simple primitive, pack it anyway, so it can share the same
244 // GeomVertexData.
245 primitive->pack_vertices(new_data, orig_data);
246 }
247
248#ifndef NDEBUG
249 if (!primitive->check_valid(new_data)) {
250 all_is_valid = false;
251 }
252#endif
253 }
254
255 nassertr(all_is_valid, 0);
256
257 if (num_changed != 0) {
258 // If any at all were changed, then keep the result (otherwise, discard
259 // it, since we might have de-optimized the indexed geometry a bit).
260 cdata->_data = new_data;
261 cdata->_primitives.swap(new_prims);
262 cdata->_modified = Geom::get_next_modified();
263 clear_cache_stage(current_thread);
264 }
265
266 return num_changed;
267}

592void Geom::
593unify_in_place(int max_indices, bool preserve_order) {
594 if (gobj_cat.is_debug()) {
595 gobj_cat.debug()
596 << "unify_in_place(" << max_indices << ", " << preserve_order
597 << "): " << *this << "\n";
598 }
599
600 Thread *current_thread = Thread::get_current_thread();
601 if (get_num_primitives() <= 1) {
602 // If we don't have more than one primitive to start with, no need to do
603 // anything.
604 return;
605 }
606
607 CDWriter cdata(_cycler, true, current_thread);
608
609 typedef pmap<TypeHandle, PT(GeomPrimitive) > NewPrims;
610
611 NewPrims new_prims;
612
613 bool keep_different_types = preserve_triangle_strips && !preserve_order;
614
615 Primitives::const_iterator pi;
616 for (pi = cdata->_primitives.begin(); pi != cdata->_primitives.end(); ++pi) {
617 CPT(GeomPrimitive) primitive = (*pi).get_read_pointer(current_thread);
618 NewPrims::iterator npi = new_prims.find(primitive->get_type());
619 if (npi == new_prims.end()) {
620 // This is the first primitive of this type.
621 if (!keep_different_types && !new_prims.empty()) {
622 // Actually, since we aren't trying to keep the different types of
623 // primitives, we should try to combine this type and the other type
624 // by decomposing them both (into triangles, segments, or whatever).
625
626 // First, decompose the incoming one.
627 primitive = primitive->decompose();
628 npi = new_prims.find(primitive->get_type());
629 if (npi == new_prims.end()) {
630 // That didn't help, so decompose the one already in the table.
631 nassertv(new_prims.size() == 1);
632 npi = new_prims.begin();
633 CPT(GeomPrimitive) np = (*npi).second->decompose();
634 new_prims.clear();
635 new_prims.insert(NewPrims::value_type(np->get_type(), np->make_copy()));
636 npi = new_prims.find(primitive->get_type());
637 }
638 }
639 }
640
641 if (npi == new_prims.end()) {
642 // This is the first primitive of this type. Just store it.
643 new_prims.insert(NewPrims::value_type(primitive->get_type(), primitive->make_copy()));
644
645 } else {
646 // We have already encountered another primitive of this type. Combine
647 // them.
648 combine_primitives((*npi).second, std::move(primitive), current_thread);
649 }
650 }
651
652 // Now, we have one or more primitives, but only one of each type.
653#ifndef NDEBUG
654 if (!keep_different_types && new_prims.size() > 1) {
655 // This shouldn't be possible, because we decompose as we go, in the loop
656 // above. (We have to decompose as we go to preserve the ordering of the
657 // primitives.)
658 nassertv(false);
659 }
660
661 GeomVertexDataPipelineReader data_reader(cdata->_data.get_read_pointer(current_thread), current_thread);
662 data_reader.check_array_readers();
663#endif
664
665 // Finally, iterate through the remaining primitives, and copy them to the
666 // output list.
667 cdata->_primitives.clear();
668 NewPrims::iterator npi;
669 for (npi = new_prims.begin(); npi != new_prims.end(); ++npi) {
670 GeomPrimitive *prim = (*npi).second;
671
672 nassertv(prim->check_valid(&data_reader));
673
674 // Each new primitive, naturally, inherits the Geom's overall shade model.
675 prim->set_shade_model(cdata->_shade_model);
676
677 // Should we split it up again to satisfy max_indices?
678 if (prim->get_num_vertices() > max_indices) {
679 // Copy prim into smaller prims, no one of which has more than
680 // max_indices vertices.
681 GeomPrimitivePipelineReader reader(prim, current_thread);
682
683 // Copy prim into smaller prims, no one of which has more than
684 // max_indices vertices.
685 int i = 0;
686 int num_primitives = reader.get_num_primitives();
687 int num_vertices_per_primitive = prim->get_num_vertices_per_primitive();
688 int num_unused_vertices_per_primitive = prim->get_num_unused_vertices_per_primitive();
689 if (num_vertices_per_primitive != 0) {
690 // This is a simple primitive type like a triangle, where all the
691 // primitives share the same number of vertices.
692 int total_vertices_per_primitive = num_vertices_per_primitive + num_unused_vertices_per_primitive;
693 int max_primitives = max_indices / total_vertices_per_primitive;
694 const unsigned char *ptr = reader.get_read_pointer(true);
695 size_t stride = reader.get_index_stride();
696
697 while (i < num_primitives) {
698 PT(GeomPrimitive) smaller = prim->make_copy();
699 smaller->clear_vertices();
700
701 // Since the number of vertices is consistent, we can calculate how
702 // many primitives will fit, and copy them all in one go.
703 int copy_primitives = min((num_primitives - i), max_primitives);
704 int num_vertices = copy_primitives * total_vertices_per_primitive;
705 nassertv(num_vertices > 0);
706 {
707 smaller->set_index_type(reader.get_index_type());
708 GeomVertexArrayDataHandle writer(smaller->modify_vertices(), current_thread);
709 writer.unclean_set_num_rows(num_vertices);
710 memcpy(writer.get_write_pointer(), ptr, stride * (size_t)(num_vertices - num_unused_vertices_per_primitive));
711 }
712
713 cdata->_primitives.push_back(smaller.p());
714
715 ptr += stride * (size_t)num_vertices;
716 i += copy_primitives;
717 }
718 } else {
719 // This is a complex primitive type like a triangle strip.
720 CPTA_int ends = reader.get_ends();
721 int start = 0;
722 int end = ends[0];
723
724 while (i < num_primitives) {
725 PT(GeomPrimitive) smaller = prim->make_copy();
726 smaller->clear_vertices();
727
728 while (smaller->get_num_vertices() + (end - start) < max_indices) {
729 for (int n = start; n < end; ++n) {
730 smaller->add_vertex(reader.get_vertex(n));
731 }
732 smaller->close_primitive();
733
734 ++i;
735 if (i >= num_primitives) {
736 break;
737 }
738
739 start = end + num_unused_vertices_per_primitive;
740 end = ends[i];
741 }
742
743 cdata->_primitives.push_back(smaller.p());
744 }
745 }
746 } else {
747 // The prim has few enough vertices; keep it.
748 cdata->_primitives.push_back(prim);
749 }
750 }
751
752 cdata->_modified = Geom::get_next_modified();
753 clear_cache_stage(current_thread);
754 reset_geom_rendering(cdata);
755}

1351void Geom::
1352compute_internal_bounds(Geom::CData *cdata, Thread *current_thread) const {
1353 // Get the vertex data, after animation.
1354 CPT(GeomVertexData) vertex_data = get_animated_vertex_data(true, current_thread);
1355
1356 // Now actually compute the bounding volume. We do this by using
1357 // calc_tight_bounds to determine our box first.
1358 LPoint3 pmin, pmax;
1359 PN_stdfloat sq_center_dist = 0.0f;
1360 bool found_any = false;
1361 do_calc_tight_bounds(pmin, pmax, sq_center_dist, found_any,
1362 vertex_data, false, LMatrix4::ident_mat(),
1363 InternalName::get_vertex(),
1364 cdata, current_thread);
1365
1366 BoundingVolume::BoundsType btype = cdata->_bounds_type;
1367 if (btype == BoundingVolume::BT_default) {
1368 btype = bounds_type;
1369 }
1370
1371 if (found_any) {
1372 nassertv(!pmin.is_nan());
1373 nassertv(!pmax.is_nan());
1374
1375 // Then we put the bounding volume around both of those points.
1376 PN_stdfloat avg_box_area;
1377 switch (btype) {
1378 case BoundingVolume::BT_best:
1379 case BoundingVolume::BT_fastest:
1380 case BoundingVolume::BT_default:
1381 {
1382 // When considering a box, calculate (roughly) the average area of the
1383 // sides. We will use this to determine whether a sphere or box is a
1384 // better fit.
1385 PN_stdfloat min_extent = min(pmax[0] - pmin[0],
1386 min(pmax[1] - pmin[1],
1387 pmax[2] - pmin[2]));
1388 PN_stdfloat max_extent = max(pmax[0] - pmin[0],
1389 max(pmax[1] - pmin[1],
1390 pmax[2] - pmin[2]));
1391 avg_box_area = ((min_extent * min_extent) + (max_extent * max_extent)) / 2;
1392 }
1393 // Fall through
1394 case BoundingVolume::BT_sphere:
1395 {
1396 // Determine the best radius for a bounding sphere.
1397 LPoint3 aabb_center = (pmin + pmax) * 0.5f;
1398 PN_stdfloat best_sq_radius = (pmax - aabb_center).length_squared();
1399
1400 if (btype != BoundingVolume::BT_fastest && best_sq_radius > 0.0f &&
1401 aabb_center.length_squared() / best_sq_radius >= (0.2f * 0.2f)) {
1402 // Hmm, this is an off-center model. Maybe we can do a better job
1403 // by calculating the bounding sphere from the AABB center.
1404
1405 PN_stdfloat better_sq_radius;
1406 bool found_any = false;
1407 do_calc_sphere_radius(aabb_center, better_sq_radius, found_any,
1408 vertex_data, cdata, current_thread);
1409
1410 if (found_any && better_sq_radius > 0.0f &&
1411 better_sq_radius <= best_sq_radius) {
1412 // Great. This is as good a sphere as we're going to get.
1413 if (btype == BoundingVolume::BT_best &&
1414 avg_box_area < better_sq_radius * MathNumbers::pi) {
1415 // But the box is better, anyway. Use that instead.
1416 cdata->_internal_bounds = new BoundingBox(pmin, pmax);
1417 break;
1418 }
1419 cdata->_internal_bounds =
1420 new BoundingSphere(aabb_center, csqrt(better_sq_radius));
1421 break;
1422 }
1423 }
1424
1425 if (btype != BoundingVolume::BT_sphere &&
1426 avg_box_area < sq_center_dist * MathNumbers::pi) {
1427 // A box is probably a tighter fit.
1428 cdata->_internal_bounds = new BoundingBox(pmin, pmax);
1429 break;
1430
1431 } else if (sq_center_dist >= 0.0f && sq_center_dist <= best_sq_radius) {
1432 // No, but a sphere centered on the origin is apparently still
1433 // better than a sphere around the bounding box.
1434 cdata->_internal_bounds =
1435 new BoundingSphere(LPoint3::origin(), csqrt(sq_center_dist));
1436 break;
1437
1438 } else if (btype == BoundingVolume::BT_sphere) {
1439 // This is the worst sphere we can make, which is why we will only
1440 // do it when the user specifically requests a sphere.
1441 cdata->_internal_bounds =
1442 new BoundingSphere(aabb_center,
1443 (best_sq_radius > 0.0f) ? csqrt(best_sq_radius) : 0.0f);
1444 break;
1445 }
1446 }
1447 // Fall through.
1448
1449 case BoundingVolume::BT_box:
1450 cdata->_internal_bounds = new BoundingBox(pmin, pmax);
1451 }
1452 }
1453 else {
1454 // No points; empty bounding volume.
1455 if (btype == BoundingVolume::BT_sphere) {
1456 cdata->_internal_bounds = new BoundingSphere;
1457 } else {
1458 cdata->_internal_bounds = new BoundingBox;
1459 }
1460 }
1461 cdata->_internal_bounds_stale = false;
1462}

1613void Geom::
1614combine_primitives(GeomPrimitive *a_prim, CPT(GeomPrimitive) b_prim,
1615 Thread *current_thread) {
1616 nassertv(a_prim != b_prim);
1617 nassertv(a_prim->get_type() == b_prim->get_type());
1618
1619 if (a_prim->get_index_type() != b_prim->get_index_type()) {
1620 GeomPrimitive::NumericType index_type = max(a_prim->get_index_type(), b_prim->get_index_type());
1621 a_prim->set_index_type(index_type);
1622 if (b_prim->get_index_type() != index_type) {
1623 PT(GeomPrimitive) b_prim_copy = b_prim->make_copy();
1624 b_prim_copy->set_index_type(index_type);
1625 b_prim = b_prim_copy;
1626 }
1627 }
1628
1629 if (!b_prim->is_indexed()) {
1630 PT(GeomPrimitive) b_prim_copy = b_prim->make_copy();
1631 b_prim_copy->make_indexed();
1632 b_prim = b_prim_copy;
1633 }
1634
1635 PT(GeomVertexArrayData) a_vertices = a_prim->modify_vertices();
1636 CPT(GeomVertexArrayData) b_vertices = b_prim->get_vertices();
1637
1638 if (a_prim->requires_unused_vertices()) {
1639 GeomVertexReader index(b_vertices, 0);
1640 int b_vertex = index.get_data1i();
1641 a_prim->append_unused_vertices(a_vertices, b_vertex);
1642 }
1643
1644 PT(GeomVertexArrayDataHandle) a_handle =
1645 new GeomVertexArrayDataHandle(std::move(a_vertices), current_thread);
1646 CPT(GeomVertexArrayDataHandle) b_handle =
1647 new GeomVertexArrayDataHandle(std::move(b_vertices), current_thread);
1648
1649 size_t orig_a_vertices = a_handle->get_num_rows();
1650
1651 a_handle->copy_subdata_from(a_handle->get_data_size_bytes(), 0,
1652 b_handle, 0, b_handle->get_data_size_bytes());
1653 a_prim->clear_minmax();
