93advance(bool ok_eof) {
94 if (_state == S_eof) {
95 assert(!flt_error_abort);
96 return FE_end_of_file;
97 }
98 if (_state == S_error) {
99 assert(!flt_error_abort);
100 return FE_read_error;
101 }
102 if (_iterator != nullptr) {
103 delete _iterator;
104 _iterator = nullptr;
105 }
106
107 if (_next_error == FE_end_of_file) {
108 _state = S_eof;
109 if (ok_eof) {
110 return FE_ok;
111 }
112 assert(!flt_error_abort);
113 return FE_end_of_file;
114
115 } else if (_next_error != FE_ok) {
116 _state = S_error;
117 assert(!flt_error_abort);
118 return _next_error;
119 }
120
121 _opcode = _next_opcode;
122 _record_length = _next_record_length;
123
124 if (flt_cat.is_debug()) {
125 flt_cat.debug()
126 << "Reading " << _opcode
127 << " of length " << _record_length << "\n";
128 }
129
130 // And now read the full record based on the length.
131 int length = _next_record_length - header_size;
132 if (length > 0) {
133 vector_uchar data((size_t)length);
134 _in.read((char *)&data[0], length);
135 _datagram = Datagram(std::move(data));
136 } else {
137 _datagram = Datagram();
138 }
139
140 if (_in.fail()) {
141 if (_in.eof()) {
142 _state = S_eof;
143 assert(!flt_error_abort);
144 return FE_end_of_file;
145 }
146
147 _state = S_error;
148 assert(!flt_error_abort);
149 return FE_read_error;
150 }
151
152 // Check out the next header in case it's a continuation.
153 read_next_header();
154 while (_next_error == FE_ok && _next_opcode == FO_continuation) {
155 if (flt_cat.is_debug()) {
156 flt_cat.debug()
157 << "Reading continuation of length " << _next_record_length << "\n";
158 }
159
160 // Read the continuation and tack it on.
161 _record_length += _next_record_length;
162 length = _next_record_length - header_size;
163
164 if (length > 0) {
165 char *buffer = new char[length];
166 _in.read(buffer, length);
167 _datagram.append_data(buffer, length);
168 delete[] buffer;
169 }
170
171 if (_in.fail()) {
172 if (_in.eof()) {
173 _state = S_eof;
174 assert(!flt_error_abort);
175 return FE_end_of_file;
176 }
177
178 _state = S_error;
179 assert(!flt_error_abort);
180 return FE_read_error;
181 }
182
183 read_next_header();
184 }
185
186 // Finally, create a new iterator to read this record.
187 _iterator = new DatagramIterator(_datagram);
188 _state = S_normal;
189
190 return FE_ok;
191}

222
