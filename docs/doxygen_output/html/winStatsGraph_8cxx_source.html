347HBRUSH WinStatsGraph::
348get_collector_brush(int collector_index, bool highlight) {
349 Brushes::iterator bi;
350 bi = _brushes.find(collector_index);
351 if (bi != _brushes.end()) {
352 return highlight ? (*bi).second.second : (*bi).second.first;
353 }
354
355 // Ask the monitor what color this guy should be.
356 LRGBColor rgb = _monitor->get_collector_color(collector_index);
357 int r = (int)encode_sRGB_uchar((float)rgb[0]);
358 int g = (int)encode_sRGB_uchar((float)rgb[1]);
359 int b = (int)encode_sRGB_uchar((float)rgb[2]);
360 HBRUSH brush = CreateSolidBrush(RGB(r, g, b));
361
362 int hr = (int)encode_sRGB_uchar((float)rgb[0] * 0.75f);
363 int hg = (int)encode_sRGB_uchar((float)rgb[1] * 0.75f);
364 int hb = (int)encode_sRGB_uchar((float)rgb[2] * 0.75f);
365 HBRUSH hbrush = CreateSolidBrush(RGB(hr, hg, hb));
366
367 _brushes[collector_index] = std::make_pair(brush, hbrush);
368 return highlight ? hbrush : brush;
369}

374COLORREF WinStatsGraph::
375get_collector_text_color(int collector_index, bool highlight) {
376 TextColors::iterator tci;
377 tci = _text_colors.find(collector_index);
378 if (tci != _text_colors.end()) {
379 return highlight ? (*tci).second.second : (*tci).second.first;
380 }
381
382 LRGBColor rgb = _monitor->get_collector_color(collector_index);
383 double bright =
384 rgb[0] * 0.2126 +
385 rgb[1] * 0.7152 +
386 rgb[2] * 0.0722;
387 COLORREF color = bright >= 0.5 ? RGB(0, 0, 0) : RGB(255, 255, 255);
388 COLORREF hcolor = bright * 0.75 >= 0.5 ? RGB(0, 0, 0) : RGB(255, 255, 255);
389
390 _text_colors[collector_index] = std::make_pair(color, hcolor);
391 return highlight ? hcolor : color;
392}

409LONG WinStatsGraph::
410window_proc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam) {
411 switch (msg) {
412 case WM_DESTROY:
413 close();
414 break;
415
416 case WM_APPCOMMAND:
417 if (GET_APPCOMMAND_LPARAM(lparam) == APPCOMMAND_CLOSE) {
418 close();
419 return TRUE;
420 }
421 break;
422
423 case WM_GETMINMAXINFO:
424 {
425 WINDOWINFO winfo;
426 GetWindowInfo(hwnd, &winfo);
427 MINMAXINFO &minmax = *(MINMAXINFO *)lparam;
428 int vminheight = _bottom_margin + _top_margin;
429 if (_label_stack.get_num_labels() > 0) {
430 // If we have a label, make sure at least one can be shown.
431 vminheight = (std::max)(vminheight, _top_label_stack_margin + _label_stack.get_label_height(0) + _bottom_margin);
432 }
433 minmax.ptMinTrackSize.x = (winfo.rcClient.left - winfo.rcWindow.left)
434 + (winfo.rcWindow.right - winfo.rcClient.right)
435 + (_right_margin + _left_margin);
436 minmax.ptMinTrackSize.y = (winfo.rcClient.top - winfo.rcWindow.top)
437 + (winfo.rcWindow.bottom - winfo.rcClient.bottom)
438 + vminheight;
439 return 0;
440 }
441
442 case WM_WINDOWPOSCHANGING:
443 {
444 WINDOWPOS &pos = *(WINDOWPOS *)lparam;
445 if ((pos.flags & (SWP_NOMOVE | SWP_NOSIZE)) == 0 && IsIconic(hwnd)) {
446 _monitor->calc_iconic_graph_window_pos(this, pos.x, pos.y);
447 }
448 }
449 break;
450
451 case WM_SYSCOMMAND:
452 if (wparam == SC_MINIMIZE) {
453 WINDOWPLACEMENT wp;
454 if (GetWindowPlacement(hwnd, &wp)) {
455 int x, y;
456 _monitor->calc_iconic_graph_window_pos(this, x, y);
457 wp.showCmd = SW_SHOWMINIMIZED;
458 wp.flags |= WPF_SETMINPOSITION;
459 wp.ptMinPosition.x = x;
460 wp.ptMinPosition.y = y;
461 SetWindowPlacement(hwnd, &wp);
462 }
463 return 0;
464 }
465 break;
466
467 case WM_SIZE:
468 move_label_stack();
469 InvalidateRect(hwnd, nullptr, TRUE);
470 break;
471
472 case WM_SIZING:
473 set_drag_mode(DM_sizing);
474 break;
475
476 case WM_EXITSIZEMOVE:
477 set_drag_mode(DM_none);
478 break;
479
480 case WM_SETCURSOR:
481 {
482 // Why is it so hard to ask for the cursor position within the window's
483 // client area?
484 POINT point;
485 GetCursorPos(&point);
486 WINDOWINFO winfo;
487 GetWindowInfo(hwnd, &winfo);
488 const RECT &rect = winfo.rcClient;
489 int x = point.x - rect.left;
490 int y = point.y - rect.top;
491 int width = rect.right - rect.left;
492 int height = rect.bottom - rect.top;
493
494 _potential_drag_mode = consider_drag_start(x, y, width, height);
495
496 switch (_potential_drag_mode) {
497 case DM_left_margin:
498 case DM_right_margin:
499 SetCursor(_sizewe_cursor);
500 return TRUE;
501
502 case DM_guide_bar:
503 SetCursor(_hand_cursor);
504 return TRUE;
505
506 default:
507 case DM_none:
508 break;
509 }
510 }
511 break;
512
513 case WM_LBUTTONDOWN:
514 if (_potential_drag_mode != DM_none) {
515 set_drag_mode(_potential_drag_mode);
516 _drag_start_x = (int16_t)LOWORD(lparam);
517 _drag_start_y = (int16_t)HIWORD(lparam);
518 SetCapture(_window);
519 }
520 return 0;
521
522 case WM_MOUSEMOVE:
523 if (_drag_mode == DM_left_margin) {
524 int16_t x = LOWORD(lparam);
525 int new_left_margin = _left_margin + (x - _drag_start_x);
526 _left_margin = std::max(new_left_margin, _pixel_scale * 2);
527 _drag_start_x = x - (new_left_margin - _left_margin);
528 InvalidateRect(hwnd, nullptr, TRUE);
529 move_label_stack();
530 return 0;
531
532 } else if (_drag_mode == DM_right_margin) {
533 int16_t x = LOWORD(lparam);
534 _right_margin += (_drag_start_x - x);
535 _drag_start_x = x;
536 InvalidateRect(hwnd, nullptr, TRUE);
537 return 0;
538 }
539 break;
540
541 case WM_LBUTTONUP:
542 set_drag_mode(DM_none);
543 ReleaseCapture();
544 break;
545
546 case WM_PAINT:
547 {
548 PAINTSTRUCT ps;
549 HDC hdc = BeginPaint(hwnd, &ps);
550
551 // First, draw a frame around the graph.
552 RECT rect;
553 GetClientRect(hwnd, &rect);
554
555 rect.left += _left_margin;
556 rect.top += _top_margin;
557 rect.right -= _right_margin;
558 rect.bottom -= _bottom_margin;
559
560 if (rect.right > rect.left && rect.bottom > rect.top) {
561 int graph_xsize = rect.right - rect.left;
562 int graph_ysize = rect.bottom - rect.top;
563 if (_bitmap_dc == 0 ||
564 graph_xsize != _bitmap_xsize ||
565 graph_ysize != _bitmap_ysize) {
566 // Oops, we need to change the bitmap (and graph) size.
567 changed_graph_size(graph_xsize, graph_ysize);
568 move_graph_window(rect.left, rect.top, graph_xsize, graph_ysize);
569 force_redraw();
570 }
571 }
572
573 additional_window_paint(hdc);
574
575 EndPaint(hwnd, &ps);
576 return 0;
577 }
578
579 case WM_DRAWITEM:
580 if (wparam == IDC_GRAPH) {
581 const DRAWITEMSTRUCT &dis = *(DRAWITEMSTRUCT *)lparam;
582
583 // Repaint the graph by copying the backing pixmap in.
584 BitBlt(dis.hDC, 0, 0,
585 _bitmap_xsize, _bitmap_ysize,
586 _bitmap_dc, 0, 0,
587 SRCCOPY);
588
589 additional_graph_window_paint(dis.hDC);
590 return 0;
591 }
592 break;
593
594 case WM_TIMER:
595 {
596 TrueClock *clock = TrueClock::get_global_ptr();
597 double new_time = clock->get_short_time();
598 if (!animate(new_time, new_time - _time)) {
599 KillTimer(hwnd, 0x100);
600 _timer_running = false;
601 }
602 _time = new_time;
603 }
604 return 0;
605
606 case WM_NOTIFY:
607 switch (((LPNMHDR)lparam)->code) {
608 case TTN_GETDISPINFO:
609 {
610 NMTTDISPINFO &info = *(NMTTDISPINFO *)lparam;
611 POINT point;
612 if (GetCursorPos(&point) && ScreenToClient(_graph_window, &point)) {
613 _tooltip_text = get_graph_tooltip(point.x, point.y);
614 info.lpszText = (char *)_tooltip_text.c_str();
615 }
616 }
617 return 0;
618 }
619 break;
620
621 default:
622 break;
623 }
624
625 return DefWindowProc(hwnd, msg, wparam, lparam);
626}

631LONG WinStatsGraph::
632graph_window_proc(HWND hwnd, UINT msg, WPARAM wparam, LPARAM lparam) {
633 switch (msg) {
634 case WM_DISPLAYCHANGE:
635 setup_bitmap(_bitmap_xsize, _bitmap_ysize);
636 force_redraw();
637 break;
638
639 case WM_NCHITTEST:
640 // Necessary for mouse events to work; default returns HTTRANSPARENT
641 return HTCLIENT;
642
643 case WM_LBUTTONDOWN:
644 // Vector any uncaught WM_LBUTTONDOWN into the main window, so we can drag
645 // margins, etc.
646 if (_potential_drag_mode != DM_none) {
647 int16_t x = LOWORD(lparam) + _graph_left;
648 int16_t y = HIWORD(lparam) + _graph_top;
649 return window_proc(_window, msg, wparam, MAKELPARAM(x, y));
650 }
651 break;
652
653 case WM_LBUTTONUP:
654 set_drag_mode(DM_none);
655 ReleaseCapture();
656 break;
657
658 default:
659 break;
660 }
661
662 return DefSubclassProc(hwnd, msg, wparam, lparam);
663}
