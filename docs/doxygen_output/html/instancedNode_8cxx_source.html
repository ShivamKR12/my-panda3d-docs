133PandaNode *InstancedNode::
134combine_with(PandaNode *other) {
135 if (is_exact_type(get_class_type()) && other->is_exact_type(get_class_type())) {
136 InstancedNode *iother = DCAST(InstancedNode, other);
137
138 // Only combine them if the instance lists for both are identical.
139 Thread *current_thread = Thread::get_current_thread();
140 CDReader this_cdata(_cycler, current_thread);
141 CDReader other_cdata(iother->_cycler, current_thread);
142 CPT(InstanceList) this_instances = this_cdata->_instances.get_read_pointer(current_thread);
143 CPT(InstanceList) other_instances = other_cdata->_instances.get_read_pointer(current_thread);
144 if (this_instances == other_instances) {
145 return this;
146 }
147 }
148
149 return nullptr;
150}

163CPT(TransformState) InstancedNode::
164calc_tight_bounds(LPoint3 &min_point, LPoint3 &max_point, bool &found_any,
165 const TransformState *transform, Thread *current_thread) const {
166
167 CPT(InstanceList) instances = get_instances(current_thread);
168 CPT(TransformState) next_transform = transform->compose(get_transform(current_thread));
169
170 for (size_t ii = 0; ii < instances->size(); ++ii) {
171 CPT(TransformState) instance_transform = next_transform->compose((*instances)[ii].get_transform());
172
173 Children cr = get_children(current_thread);
174 size_t num_children = cr.get_num_children();
175 for (size_t ci = 0; ci < num_children; ++ci) {
176 cr.get_child(ci)->calc_tight_bounds(min_point, max_point,
177 found_any, instance_transform,
178 current_thread);
179 }
180 }
181
182 return next_transform;
183}

203bool InstancedNode::
204cull_callback(CullTraverser *trav, CullTraverserData &data) {
205 Thread *current_thread = trav->get_current_thread();
206
207 CPT(InstanceList) instances = get_instances(current_thread);
208
209 if (data._instances != nullptr) {
210 // We are already under an instanced node. Create a new combined list.
211 InstanceList *new_list = new InstanceList();
212 new_list->reserve(data._instances->size() * instances->size());
213 for (const InstanceList::Instance &parent_instance : *data._instances) {
214 for (const InstanceList::Instance &this_instance : *instances) {
215 new_list->append(parent_instance.get_transform()->compose(this_instance.get_transform()));
216 }
217 }
218 instances = new_list;
219 }
220
221 if (data._view_frustum != nullptr || data._cull_planes != nullptr) {
222 // Culling is on, so we need to figure out which instances should be culled.
223 BitArray culled_instances;
224 culled_instances.set_range(0, instances->size());
225
226 for (size_t ii = 0; ii < instances->size(); ++ii) {
227 if (data.is_instance_in_view((*instances)[ii].get_transform(), trav->get_camera_mask())) {
228 culled_instances.clear_bit(ii);
229 }
230 }
231
232 if (!culled_instances.is_zero() && trav->get_fake_view_frustum_cull()) {
233 // The culled instances are drawn with the fake-view-frustum-cull effect.
234 data._instances = instances->without(culled_instances ^ BitArray::range(0, instances->size()));
235
236 Children children = data.node_reader()->get_children();
237 int num_children = children.get_num_children();
238 for (int i = 0; i < num_children; ++i) {
239 trav->do_fake_cull(data, children.get_child(i), data._net_transform, data._state);
240 }
241 }
242
243 instances = instances->without(culled_instances);
244 }
245
246 if (instances->empty()) {
247 // There are no instances, or they are all culled away.
248 return false;
249 }
250
251 data._instances = std::move(instances);
252
253 // Disable culling from this point on, for now. It's probably not worth it
254 // to keep lists of transformed bounding volumes for each instance.
255 data._view_frustum = nullptr;
256 data._cull_planes = CullPlanes::make_empty();
257
258 return true;
259}

275void InstancedNode::
276compute_external_bounds(CPT(BoundingVolume) &external_bounds,
277 BoundingVolume::BoundsType btype,
278 const BoundingVolume **volumes, size_t num_volumes,
279 int pipeline_stage, Thread *current_thread) const {
280
281 CPT(InstanceList) instances = get_instances(current_thread);
282
283 PT(GeometricBoundingVolume) gbv;
284 if (btype == BoundingVolume::BT_sphere) {
285 gbv = new BoundingSphere;
286 } else {
287 gbv = new BoundingBox;
288 }
289
290 if (num_volumes == 0 || instances->empty()) {
291 external_bounds = gbv;
292 return;
293 }
294
295 // Compute a sphere at the origin, encompassing the children. This may not
296 // be the most optimal shape, but it allows us to easily estimate a bounding
297 // volume without having to take each instance transform into account.
298 PN_stdfloat max_radius = 0;
299 LVector3 max_abs_box(0);
300
301 for (size_t i = 0; i < num_volumes; ++i) {
302 const BoundingVolume *child_volume = volumes[i];
303 if (child_volume->is_empty()) {
304 continue;
305 }
306 if (child_volume->is_infinite()) {
307 gbv->set_infinite();
308 break;
309 }
310 if (const BoundingSphere *child_sphere = child_volume->as_bounding_sphere()) {
311 max_radius = child_sphere->get_center().length() + child_sphere->get_radius();
312 }
313 else if (const FiniteBoundingVolume *child_finite = child_volume->as_finite_bounding_volume()) {
314 LPoint3 min1 = child_finite->get_min();
315 LPoint3 max1 = child_finite->get_max();
316 max_abs_box.set(
317 std::max(max_abs_box[0], std::max(std::fabs(min1[0]), std::fabs(max1[0]))),
318 std::max(max_abs_box[1], std::max(std::fabs(min1[1]), std::fabs(max1[1]))),
319 std::max(max_abs_box[2], std::max(std::fabs(min1[2]), std::fabs(max1[2]))));
320 }
321 else {
322 gbv->set_infinite();
323 break;
324 }
325 }
326
327 max_radius = std::max(max_radius, max_abs_box.length());
328 if (max_radius == 0 || gbv->is_infinite()) {
329 external_bounds = gbv;
330 return;
331 }
332
333 // Now that we have a sphere encompassing the children, we will make a box
334 // surrounding all the instances, extended by the computed radius.
335 LPoint3 min_point = (*instances)[0].get_pos();
336 LPoint3 max_point(min_point);
337
338 for (const InstanceList::Instance &instance : *instances) {
339 // To make the math easier and not have to take rotations into account, we
340 // take the highest scale component and multiply it by the radius of the
341 // bounding sphere on the origin we just calculated.
342 LVecBase3 scale = instance.get_scale();
343 PN_stdfloat max_scale = std::max(std::fabs(scale[0]), std::max(std::fabs(scale[1]), std::fabs(scale[2])));
344 PN_stdfloat inst_radius = max_scale * max_radius;
345 LVector3 extends_by(inst_radius);
346 LPoint3 pos = instance.get_pos();
347 min_point = min_point.fmin(pos - extends_by);
348 max_point = max_point.fmax(pos + extends_by);
349 }
350
351 if (min_point == max_point) {
352 external_bounds = gbv;
353 return;
354 }
355
356 // If we really need to make a sphere, we use the center of the bounding box
357 // as our sphere center, and iterate again to find the furthest instance.
358 if (btype == BoundingVolume::BT_sphere) {
359 LPoint3 center = (min_point + max_point) * 0.5;
360
361 PN_stdfloat max_distance = 0;
362 for (const InstanceList::Instance &instance : *instances) {
363 LVecBase3 scale = instance.get_scale();
364 PN_stdfloat max_scale = std::max(std::fabs(scale[0]), std::max(std::fabs(scale[1]), std::fabs(scale[2])));
365 PN_stdfloat inst_radius = max_scale * max_radius;
366 PN_stdfloat distance = (instance.get_pos() - center).length() + inst_radius;
367 max_distance = std::max(max_distance, distance);
368 }
369
370 if (max_distance == 0) {
371 external_bounds = gbv;
372 return;
373 }
374 ((BoundingSphere *)gbv.p())->set_center(center);
375 ((BoundingSphere *)gbv.p())->set_radius(max_distance);
376 } else {
377 ((BoundingBox *)gbv.p())->set_min_max(min_point, max_point);
378 }
379
380 // If we have a transform, apply it to the bounding volume we just
381 // computed.
382 CPT(TransformState) transform = get_transform(current_thread);
383 if (!transform->is_identity()) {
384 gbv->xform(transform->get_mat());
385 }
386
387 external_bounds = gbv;
388}
