341CPT(RenderState) RenderState::
342compose(const RenderState *other) const {
343 // This method isn't strictly const, because it updates the cache, but we
344 // pretend that it is because it's only a cache which is transparent to the
345 // rest of the interface.
346
347 // We handle empty state (identity) as a trivial special case.
348 if (is_empty()) {
349 return other;
350 }
351 if (other->is_empty()) {
352 return this;
353 }
354
355 if (!state_cache) {
356 return do_compose(other);
357 }
358
359 LightReMutexHolder holder(*_states_lock);
360
361 // Is this composition already cached?
362 int index = _composition_cache.find(other);
363 if (index != -1) {
364 Composition &comp = ((RenderState *)this)->_composition_cache.modify_data(index);
365 if (comp._result == nullptr) {
366 // Well, it wasn't cached already, but we already had an entry (probably
367 // created for the reverse direction), so use the same entry to store
368 // the new result.
369 CPT(RenderState) result = do_compose(other);
370 comp._result = result;
371
372 if (result != (const RenderState *)this) {
373 // See the comments below about the need to up the reference count
374 // only when the result is not the same as this.
375 result->cache_ref();
376 }
377 }
378 // Here's the cache!
379 _cache_stats.inc_hits();
380 return comp._result;
381 }
382 _cache_stats.inc_misses();
383
384 // We need to make a new cache entry, both in this object and in the other
385 // object. We make both records so the other RenderState object will know
386 // to delete the entry from this object when it destructs, and vice-versa.
387
388 // The cache entry in this object is the only one that indicates the result;
389 // the other will be NULL for now.
390 CPT(RenderState) result = do_compose(other);
391
392 _cache_stats.add_total_size(1);
393 _cache_stats.inc_adds(_composition_cache.is_empty());
394
395 ((RenderState *)this)->_composition_cache[other]._result = result;
396
397 if (other != this) {
398 _cache_stats.add_total_size(1);
399 _cache_stats.inc_adds(other->_composition_cache.is_empty());
400 ((RenderState *)other)->_composition_cache[this]._result = nullptr;
401 }
402
403 if (result != (const RenderState *)this) {
404 // If the result of compose() is something other than this, explicitly
405 // increment the reference count. We have to be sure to decrement it
406 // again later, when the composition entry is removed from the cache.
407 result->cache_ref();
408
409 // (If the result was just this again, we still store the result, but we
410 // don't increment the reference count, since that would be a self-
411 // referential leak.)
412 }
413
414 _cache_stats.maybe_report("RenderState");
415
416 return result;
417}

608bool RenderState::
609unref() const {
610 if (garbage_collect_states || !state_cache) {
611 // If we're not using the cache at all, or if we're relying on garbage
612 // collection, just allow the pointer to unref normally.
613 return ReferenceCount::unref();
614 }
615
616 // Here is the normal refcounting case, with a normal cache, and without
617 // garbage collection in effect. In this case we will pull the object out
618 // of the cache when its reference count goes to 0.
619
620 // We always have to grab the lock, since we will definitely need to be
621 // holding it if we happen to drop the reference count to 0. Having to grab
622 // the lock at every call to unref() is a big limiting factor on
623 // parallelization.
624 LightReMutexHolder holder(*_states_lock);
625
626 if (auto_break_cycles && uniquify_states) {
627 if (get_cache_ref_count() > 0 &&
628 get_ref_count() == get_cache_ref_count() + 1) {
629 // If we are about to remove the one reference that is not in the cache,
630 // leaving only references in the cache, then we need to check for a
631 // cycle involving this RenderState and break it if it exists.
632 ((RenderState *)this)->detect_and_break_cycles();
633 }
634 }
635
636 if (ReferenceCount::unref()) {
637 // The reference count is still nonzero.
638 return true;
639 }
640
641 // The reference count has just reached zero. Make sure the object is
642 // removed from the global object pool, before anyone else finds it and
643 // tries to ref it.
644 ((RenderState *)this)->release_new();
645 ((RenderState *)this)->remove_cache_pointers();
646
647 return false;
648}

734int RenderState::
735get_num_unused_states() {
736 LightReMutexHolder holder(*_states_lock);
737
738 // First, we need to count the number of times each RenderState object is
739 // recorded in the cache.
740 typedef pmap<const RenderState *, int> StateCount;
741 StateCount state_count;
742
743 size_t size = _states.get_num_entries();
744 for (size_t si = 0; si < size; ++si) {
745 const RenderState *state = _states.get_key(si);
746
747 std::pair<StateCount::iterator, bool> ir =
748 state_count.insert(StateCount::value_type(state, 1));
749 if (!ir.second) {
750 // If the above insert operation fails, then it's already in the
751 // cache; increment its value.
752 (*(ir.first)).second++;
753 }
754
755 size_t i;
756 size_t cache_size = state->_composition_cache.get_num_entries();
757 for (i = 0; i < cache_size; ++i) {
758 const RenderState *result = state->_composition_cache.get_data(i)._result;
759 if (result != nullptr && result != state) {
760 // Here's a RenderState that's recorded in the cache. Count it.
761 std::pair<StateCount::iterator, bool> ir =
762 state_count.insert(StateCount::value_type(result, 1));
763 if (!ir.second) {
764 // If the above insert operation fails, then it's already in the
765 // cache; increment its value.
766 (*(ir.first)).second++;
767 }
768 }
769 }
770 cache_size = state->_invert_composition_cache.get_num_entries();
771 for (i = 0; i < cache_size; ++i) {
772 const RenderState *result = state->_invert_composition_cache.get_data(i)._result;
773 if (result != nullptr && result != state) {
774 std::pair<StateCount::iterator, bool> ir =
775 state_count.insert(StateCount::value_type(result, 1));
776 if (!ir.second) {
777 (*(ir.first)).second++;
778 }
779 }
780 }
781 }
782
783 // Now that we have the appearance count of each RenderState object, we can
784 // tell which ones are unreferenced outside of the RenderState cache, by
785 // comparing these to the reference counts.
786 int num_unused = 0;
787
788 StateCount::iterator sci;
789 for (sci = state_count.begin(); sci != state_count.end(); ++sci) {
790 const RenderState *state = (*sci).first;
791 int count = (*sci).second;
792 nassertr(count == state->get_cache_ref_count(), num_unused);
793 nassertr(count <= state->get_ref_count(), num_unused);
794 if (count == state->get_ref_count()) {
795 num_unused++;
796
797 if (pgraph_cat.is_debug()) {
798 pgraph_cat.debug()
799 << "Unused state: " << (void *)state << ":"
800 << state->get_ref_count() << " =\n";
801 state->write(pgraph_cat.debug(false), 2);
802 }
803 }
804 }
805
806 return num_unused;
807}

822int RenderState::
823clear_cache() {
824 LightReMutexHolder holder(*_states_lock);
825
826 PStatTimer timer(_cache_update_pcollector);
827 int orig_size = _states.get_num_entries();
828
829 // First, we need to copy the entire set of states to a temporary vector,
830 // reference-counting each object. That way we can walk through the copy,
831 // without fear of dereferencing (and deleting) the objects in the map as we
832 // go.
833 {
834 typedef pvector< CPT(RenderState) > TempStates;
835 TempStates temp_states;
836 temp_states.reserve(orig_size);
837
838 size_t size = _states.get_num_entries();
839 for (size_t si = 0; si < size; ++si) {
840 const RenderState *state = _states.get_key(si);
841 temp_states.push_back(state);
842 }
843
844 // Now it's safe to walk through the list, destroying the cache within
845 // each object as we go. Nothing will be destructed till we're done.
846 TempStates::iterator ti;
847 for (ti = temp_states.begin(); ti != temp_states.end(); ++ti) {
848 RenderState *state = (RenderState *)(*ti).p();
849
850 size_t i;
851 size_t cache_size = (int)state->_composition_cache.get_num_entries();
852 for (i = 0; i < cache_size; ++i) {
853 const RenderState *result = state->_composition_cache.get_data(i)._result;
854 if (result != nullptr && result != state) {
855 result->cache_unref();
856 nassertr(result->get_ref_count() > 0, 0);
857 }
858 }
859 _cache_stats.add_total_size(-(int)state->_composition_cache.get_num_entries());
860 state->_composition_cache.clear();
861
862 cache_size = (int)state->_invert_composition_cache.get_num_entries();
863 for (i = 0; i < cache_size; ++i) {
864 const RenderState *result = state->_invert_composition_cache.get_data(i)._result;
865 if (result != nullptr && result != state) {
866 result->cache_unref();
867 nassertr(result->get_ref_count() > 0, 0);
868 }
869 }
870 _cache_stats.add_total_size(-(int)state->_invert_composition_cache.get_num_entries());
871 state->_invert_composition_cache.clear();
872 }
873
874 // Once this block closes and the temp_states object goes away, all the
875 // destruction will begin. Anything whose reference was held only within
876 // the various objects' caches will go away.
877 }
878
879 int new_size = _states.get_num_entries();
880 return orig_size - new_size;
881}

891int RenderState::
892garbage_collect() {
893 int num_attribs = RenderAttrib::garbage_collect();
894
895 if (!garbage_collect_states) {
896 return num_attribs;
897 }
898
899 LightReMutexHolder holder(*_states_lock);
900
901 PStatTimer timer(_garbage_collect_pcollector);
902 size_t orig_size = _states.get_num_entries();
903
904 // How many elements to process this pass?
905 size_t size = orig_size;
906 size_t num_this_pass = std::max(0, int(size * garbage_collect_states_rate));
907 if (num_this_pass <= 0) {
908 return num_attribs;
909 }
910
911 bool break_and_uniquify = (auto_break_cycles && uniquify_transforms);
912
913 size_t si = _garbage_index;
914 if (si >= size) {
915 si = 0;
916 }
917
918 num_this_pass = std::min(num_this_pass, size);
919 size_t stop_at_element = (si + num_this_pass) % size;
920
921 do {
922 RenderState *state = (RenderState *)_states.get_key(si);
923 if (break_and_uniquify) {
924 if (state->get_cache_ref_count() > 0 &&
925 state->get_ref_count() == state->get_cache_ref_count()) {
926 // If we have removed all the references to this state not in the
927 // cache, leaving only references in the cache, then we need to
928 // check for a cycle involving this RenderState and break it if it
929 // exists.
930 state->detect_and_break_cycles();
931 }
932 }
933
934 if (!state->unref_if_one()) {
935 // This state has recently been unreffed to 1 (the one we added when
936 // we stored it in the cache). Now it's time to delete it. This is
937 // safe, because we're holding the _states_lock, so it's not possible
938 // for some other thread to find the state in the cache and ref it
939 // while we're doing this. Also, we've just made sure to unref it to 0,
940 // to ensure that another thread can't get it via a weak pointer.
941
942 state->release_new();
943 state->remove_cache_pointers();
944 state->cache_unref_only();
945 delete state;
946
947 // When we removed it from the hash map, it swapped the last element
948 // with the one we just removed. So the current index contains one we
949 // still need to visit.
950 --size;
951 --si;
952 if (stop_at_element > 0) {
953 --stop_at_element;
954 }
955 }
956
957 si = (si + 1) % size;
958 } while (si != stop_at_element);
959 _garbage_index = si;
960
961 nassertr(_states.get_num_entries() == size, 0);
962
963#ifdef _DEBUG
964 nassertr(_states.validate(), 0);
965#endif
966
967 // If we just cleaned up a lot of states, see if we can reduce the table in
968 // size. This will help reduce iteration overhead in the future.
969 _states.consider_shrink_table();
970
971 return (int)orig_size - (int)size + num_attribs;
972}

1101 if (_states.is_empty()) {
1102 return true;
1103 }
1104

1120 const RenderState *ssnext = _states.get_key(snext);
1121 int c = ssi->compare_to(*ssnext);
1122 int ci = ssnext->compare_to(*ssi);
1123 if ((ci < 0) != (c > 0) ||
1124 (ci > 0) != (c < 0) ||
1125 (ci == 0) != (c == 0)) {
1126 pgraph_cat.error()

1128 pgraph_cat.error(false)

1130 pgraph_cat.error(false)
1131 << "(b, a): " << ci << "\n";
1132 ssi->write(pgraph_cat.error(false), 2);
1133 ssnext->write(pgraph_cat.error(false), 2);
1134 return false;
1135 }
1136 si = snext;
1137 ++snext;

1227CPT(RenderState) RenderState::
1228return_new(RenderState *state) {
1229 nassertr(state != nullptr, state);
1230
1231 // Make sure we don't have anything in the 0 slot. If we did, that would
1232 // indicate an uninitialized slot number.
1233#ifndef NDEBUG
1234 if (state->_attributes[0]._attrib != nullptr) {
1235 const RenderAttrib *attrib = state->_attributes[0]._attrib;
1236 if (attrib->get_type() == TypeHandle::none()) {
1237 ((RenderAttrib *)attrib)->force_init_type();
1238 pgraph_cat->error()
1239 << "Uninitialized RenderAttrib type: " << attrib->get_type()
1240 << "\n";
1241
1242 } else {
1243 static pset<TypeHandle> already_reported;
1244 if (already_reported.insert(attrib->get_type()).second) {
1245 pgraph_cat->error()
1246 << attrib->get_type() << " did not initialize its slot number.\n";
1247 }
1248 }
1249 }
1250#endif
1251 state->_attributes[0]._attrib = nullptr;
1252 state->_filled_slots.clear_bit(0);
1253
1254#ifndef NDEBUG
1255 nassertr(state->validate_filled_slots(), state);
1256#endif
1257
1258 if (!uniquify_states && !state->is_empty()) {
1259 return state;
1260 }
1261

1273CPT(RenderState) RenderState::
1274return_unique(RenderState *state) {
1275 nassertr(state != nullptr, nullptr);
1276
1277 if (!state_cache) {
1278 return state;
1279 }
1280
1281#ifndef NDEBUG
1282 if (paranoid_const) {
1283 nassertr(validate_states(), state);
1284 }
1285#endif
1286
1287 LightReMutexHolder holder(*_states_lock);
1288
1289 if (state->_saved_entry != -1) {
1290 // This state is already in the cache.
1291 //nassertr(_states.find(state) == state->_saved_entry, pt_state);
1292 return state;
1293 }
1294
1295 // Ensure each of the individual attrib pointers has been uniquified before
1296 // we add the state to the cache.
1297 if (!uniquify_attribs && !state->is_empty()) {
1298 SlotMask mask = state->_filled_slots;
1299 int slot = mask.get_lowest_on_bit();
1300 while (slot >= 0) {
1301 Attribute &attrib = state->_attributes[slot];
1302 nassertd(attrib._attrib != nullptr) continue;
1303 attrib._attrib = RenderAttrib::do_uniquify(attrib._attrib);
1304 mask.clear_bit(slot);
1305 slot = mask.get_lowest_on_bit();
1306 }
1307 }
1308
1309 int si = _states.find(state);
1310 if (si != -1) {
1311 // There's an equivalent state already in the set. Return it. The state
1312 // that was passed may be newly created and therefore may not be
1313 // automatically deleted. Do that if necessary.
1314 if (state->get_ref_count() == 0) {
1315 delete state;
1316 }
1317 return _states.get_key(si);
1318 }
1319
1320 // Not already in the set; add it.
1321 if (garbage_collect_states) {
1322 // If we'll be garbage collecting states explicitly, we'll increment the
1323 // reference count when we store it in the cache, so that it won't be
1324 // deleted while it's in it.
1325 state->cache_ref();
1326 }
1327 si = _states.store(state, nullptr);
1328
1329 // Save the index and return the input state.
1330 state->_saved_entry = si;
1331 return state;
1332}

1338CPT(RenderState) RenderState::
1339do_compose(const RenderState *other) const {
1340 PStatTimer timer(_state_compose_pcollector);
1341
1342 RenderState *new_state = new RenderState;
1343
1344 SlotMask mask = _filled_slots | other->_filled_slots;
1345 new_state->_filled_slots = mask;
1346
1347 int slot = mask.get_lowest_on_bit();
1348 while (slot >= 0) {
1349 const Attribute &a = _attributes[slot];
1350 const Attribute &b = other->_attributes[slot];
1351 Attribute &result = new_state->_attributes[slot];
1352
1353 if (a._attrib == nullptr) {
1354 nassertr(b._attrib != nullptr, this);
1355 // B wins.
1356 result = b;
1357
1358 } else if (b._attrib == nullptr) {
1359 // A wins.
1360 result = a;
1361
1362 } else if (b._override < a._override) {
1363 // A, the higher RenderAttrib, overrides.
1364 result = a;
1365
1366 } else if (a._override < b._override &&
1367 a._attrib->lower_attrib_can_override()) {
1368 // B, the higher RenderAttrib, overrides. This is a special case;
1369 // normally, a lower RenderAttrib does not override a higher one, even
1370 // if it has a higher override value. But certain kinds of
1371 // RenderAttribs redefine lower_attrib_can_override() to return true,
1372 // allowing this override.
1373 result = b;
1374
1375 } else {
1376 // Either they have the same override value, or B is higher. In either
1377 // case, the result is the composition of the two, with B's override
1378 // value.
1379 result.set(a._attrib->compose(b._attrib), b._override);
1380 }
1381
1382 mask.clear_bit(slot);
1383 slot = mask.get_lowest_on_bit();
1384 }
1385
1386 return return_new(new_state);
1387}

1467r_detect_cycles(const RenderState *start_state,
1468 const RenderState *current_state,
1469 int length, UpdateSeq this_seq,
1470 RenderState::CompositionCycleDesc *cycle_desc) {
1471 if (current_state->_cycle_detect == this_seq) {
1472 // We've already seen this state; therefore, we've found a cycle.
1473
1474 // However, we only care about cycles that return to the starting state
1475 // and involve more than two steps. If only one or two nodes are
1476 // involved, it doesn't represent a memory leak, so no problem there.
1477 return (current_state == start_state && length > 2);
1478 }
1479 ((RenderState *)current_state)->_cycle_detect = this_seq;
1480
1481 size_t i;
1482 size_t cache_size = current_state->_composition_cache.get_num_entries();
1483 for (i = 0; i < cache_size; ++i) {
1484 const RenderState *result = current_state->_composition_cache.get_data(i)._result;
1485 if (result != nullptr) {
1486 if (r_detect_cycles(start_state, result, length + 1,
1487 this_seq, cycle_desc)) {
1488 // Cycle detected.
1489 if (cycle_desc != nullptr) {
1490 const RenderState *other = current_state->_composition_cache.get_key(i);
1491 CompositionCycleDescEntry entry(other, result, false);
1492 cycle_desc->push_back(entry);
1493 }
1494 return true;
1495 }
1496 }
1497 }
1498
1499 cache_size = current_state->_invert_composition_cache.get_num_entries();
1500 for (i = 0; i < cache_size; ++i) {
1501 const RenderState *result = current_state->_invert_composition_cache.get_data(i)._result;
1502 if (result != nullptr) {
1503 if (r_detect_cycles(start_state, result, length + 1,
1504 this_seq, cycle_desc)) {
1505 // Cycle detected.
1506 if (cycle_desc != nullptr) {
1507 const RenderState *other = current_state->_invert_composition_cache.get_key(i);
1508 CompositionCycleDescEntry entry(other, result, true);
1509 cycle_desc->push_back(entry);
1510 }
1511 return true;
1512 }
1513 }
1514 }
1515
1516 // No cycle detected.
1517 return false;

1614void RenderState::
1615remove_cache_pointers() {
1616 nassertv(_states_lock->debug_is_locked());
1617
1618 // Fortunately, since we added CompositionCache records in pairs, we know
1619 // exactly the set of RenderState objects that have us in their cache: it's
1620 // the same set of RenderState objects that we have in our own cache.
1621
1622/*
1623 * We do need to put considerable thought into this loop, because as we clear
1624 * out cache entries we'll cause other RenderState objects to destruct, which
1625 * could cause things to get pulled out of our own _composition_cache map. We
1626 * want to allow this (so that we don't encounter any just-destructed pointers
1627 * in our cache), but we don't want to get bitten by this cascading effect.
1628 * Instead of walking through the map from beginning to end, therefore, we
1629 * just pull out the first one each time, and erase it.
1630 */
1631
1632#ifdef DO_PSTATS
1633 if (_composition_cache.is_empty() && _invert_composition_cache.is_empty()) {
1634 return;
1635 }
1636 PStatTimer timer(_cache_update_pcollector);
1637#endif // DO_PSTATS
1638
1639 // There are lots of ways to do this loop wrong. Be very careful if you
1640 // need to modify it for any reason.
1641 size_t i = 0;
1642 while (!_composition_cache.is_empty()) {
1643 // It is possible that the "other" RenderState object is currently within
1644 // its own destructor. We therefore can't use a PT() to hold its pointer;
1645 // that could end up calling its destructor twice. Fortunately, we don't
1646 // need to hold its reference count to ensure it doesn't destruct while we
1647 // process this loop; as long as we ensure that no *other* RenderState
1648 // objects destruct, there will be no reason for that one to.
1649 RenderState *other = (RenderState *)_composition_cache.get_key(i);
1650
1651 // We hold a copy of the composition result so we can dereference it
1652 // later.
1653 Composition comp = _composition_cache.get_data(i);
1654
1655 // Now we can remove the element from our cache. We do this now, rather
1656 // than later, before any other RenderState objects have had a chance to
1657 // destruct, so we are confident that our iterator is still valid.
1658 _composition_cache.remove_element(i);
1659 _cache_stats.add_total_size(-1);
1660 _cache_stats.inc_dels();
1661
1662 if (other != this) {
1663 int oi = other->_composition_cache.find(this);
1664
1665 // We may or may not still be listed in the other's cache (it might be
1666 // halfway through pulling entries out, from within its own destructor).
1667 if (oi != -1) {
1668 // Hold a copy of the other composition result, too.
1669 Composition ocomp = other->_composition_cache.get_data(oi);
1670
1671 other->_composition_cache.remove_element(oi);
1672 _cache_stats.add_total_size(-1);
1673 _cache_stats.inc_dels();
1674
1675 // It's finally safe to let our held pointers go away. This may have
1676 // cascading effects as other RenderState objects are destructed, but
1677 // there will be no harm done if they destruct now.
1678 if (ocomp._result != nullptr && ocomp._result != other) {
1679 cache_unref_delete(ocomp._result);
1680 }
1681 }
1682 }
1683
1684 // It's finally safe to let our held pointers go away. (See comment
1685 // above.)
1686 if (comp._result != nullptr && comp._result != this) {
1687 cache_unref_delete(comp._result);
1688 }
1689 }
1690
1691 // A similar bit of code for the invert cache.
1692 i = 0;
1693 while (!_invert_composition_cache.is_empty()) {
1694 RenderState *other = (RenderState *)_invert_composition_cache.get_key(i);
1695 nassertv(other != this);
1696 Composition comp = _invert_composition_cache.get_data(i);
1697 _invert_composition_cache.remove_element(i);
1698 _cache_stats.add_total_size(-1);
1699 _cache_stats.inc_dels();
1700 if (other != this) {
1701 int oi = other->_invert_composition_cache.find(this);
1702 if (oi != -1) {
1703 Composition ocomp = other->_invert_composition_cache.get_data(oi);
1704 other->_invert_composition_cache.remove_element(oi);
1705 _cache_stats.add_total_size(-1);
1706 _cache_stats.inc_dels();
1707 if (ocomp._result != nullptr && ocomp._result != other) {
1708 cache_unref_delete(ocomp._result);
1709 }
1710 }
1711 }
1712 if (comp._result != nullptr && comp._result != this) {
1713 cache_unref_delete(comp._result);
1714 }
1715 }
