95void ThreadSimpleManager::
96enqueue_ready(ThreadSimpleImpl *thread, bool volunteer) {
97 // We actually add it to _next_ready, so that we can tell when we have
98 // processed every thread in a given epoch.
99 if (!volunteer) {
100 _next_ready.push_back(thread);
101
102 } else {
103 // Unless it's a volunteer, in which case we actually put it to sleep for
104 // the duration of the timeslice, so it won't interfere with timeslice
105 // accounting for the remaining ready threads.
106 double now = get_current_time();
107 thread->_wake_time = now + _simple_thread_volunteer_delay;
108 _volunteers.push_back(thread);
109 push_heap(_volunteers.begin(), _volunteers.end(), CompareStartTime());
110 }

150 if (bi != _blocked.end()) {
151 nassertr(blocker->_flags & BlockerSimple::F_has_waiters, false);

226void ThreadSimpleManager::
227next_context() {
228 // Delete any threads that need it. We can't delete the current thread,
229 // though.
230 while (!_finished.empty() && _finished.front() != _current_thread) {
231 ThreadSimpleImpl *finished_thread = _finished.front();
232 _finished.pop_front();
233 unref_delete(finished_thread->_parent_obj);
234 }
235
236 ++_current_thread->_context_switches;
237
238 // Mark the current thread's resume point.
239
240#ifdef HAVE_PYTHON
241 // Save the current Python thread state.
242 _current_thread->_python_state = thread_state_swap(nullptr);
243#endif // HAVE_PYTHON
244
245#ifdef DO_PSTATS
246 Thread::PStatsCallback *pstats_callback = _current_thread->_parent_obj->get_pstats_callback();
247 if (pstats_callback != nullptr) {
248 pstats_callback->deactivate_hook(_current_thread->_parent_obj);
249 }
250#endif // DO_PSTATS
251
252 save_thread_context(_current_thread->_context, st_choose_next_context, this);
253 // Pass 2: we have returned into the context, and are now resuming the
254 // current thread.
255
256#ifdef DO_PSTATS
257 if (pstats_callback != nullptr) {
258 pstats_callback->activate_hook(_current_thread->_parent_obj);
259 }
260#endif // DO_PSTATS
261
262#ifdef HAVE_PYTHON
263 thread_state_swap(_current_thread->_python_state);
264#endif // HAVE_PYTHON

272void ThreadSimpleManager::
273prepare_for_exit() {
274 if (!_current_thread->_parent_obj->is_exact_type(MainThread::get_class_type())) {
275 if (thread_cat->is_debug()) {
276 thread_cat.debug()
277 << "Ignoring prepare_for_exit called from "
278 << *(_current_thread->_parent_obj) << "\n";
279 }
280 return;
281 }
282
283 if (thread_cat->is_debug()) {
284 thread_cat.debug()
285 << "prepare_for_exit\n";
286 }
287
288 nassertv(_waiting_for_exit == nullptr);
289 _waiting_for_exit = _current_thread;
290
291 // At this point, any non-joinable threads on any of the queues are
292 // automatically killed.
293 kill_non_joinable(_ready);
294
295 Blocked::iterator bi = _blocked.begin();
296 while (bi != _blocked.end()) {
297 Blocked::iterator bnext = bi;
298 ++bnext;
299 BlockerSimple *blocker = (*bi).first;
300 FifoThreads &threads = (*bi).second;
301 kill_non_joinable(threads);
302 if (threads.empty()) {
303 blocker->_flags &= ~BlockerSimple::F_has_waiters;
304 _blocked.erase(bi);
305 }
306 bi = bnext;
307 }
308
309 kill_non_joinable(_sleeping);
310 kill_non_joinable(_volunteers);
311
312 next_context();
313
314 // Delete any remaining threads.
315 while (!_finished.empty() && _finished.front() != _current_thread) {
316 ThreadSimpleImpl *finished_thread = _finished.front();
317 _finished.pop_front();
318 unref_delete(finished_thread->_parent_obj);
319 }
320}

336void ThreadSimpleManager::
337remove_thread(ThreadSimpleImpl *thread) {
338 TickRecords new_records;
339 TickRecords::iterator ri;
340 for (ri = _tick_records.begin(); ri != _tick_records.end(); ++ri) {
341 if ((*ri)._thread != thread) {
342 // Keep this record.
343 new_records.push_back(*ri);
344 } else {
345 // Lose this record.
346 nassertv(_total_ticks >= (*ri)._tick_count);
347 _total_ticks -= (*ri)._tick_count;
348 }
349 }
350

358void ThreadSimpleManager::
359system_sleep(double seconds) {
360#ifdef WIN32
361 Sleep((int)(seconds * 1000 + 0.5));
362
363#else
364 /*
365 struct timespec rqtp;
366 rqtp.tv_sec = time_t(seconds);
367 rqtp.tv_nsec = long((seconds - (double)rqtp.tv_sec) * 1000000000.0 + 0.5);
368 nanosleep(&rqtp, NULL);
369 */
370
371 // We use select() as the only way that seems to actually yield the
372 // timeslice. sleep() and nanosleep() don't appear to do the trick.
373 struct timeval tv;
374 tv.tv_sec = time_t(seconds);
375 tv.tv_usec = long((seconds - (double)tv.tv_sec) * 1000000.0 + 0.5);
376 select(0, nullptr, nullptr, nullptr, &tv);
377#endif // WIN32
378}

437system_yield() {

440 return;

490void ThreadSimpleManager::
491choose_next_context(struct ThreadContext *from_context) {
492 double now = get_current_time();
493
494 do_timeslice_accounting(_current_thread, now);
495 _current_thread = nullptr;
496
497 if (!_sleeping.empty() || !_volunteers.empty()) {
498 if (_ready.empty() && _next_ready.empty()) {
499 // All of our threads are currently sleeping. Therefore, wake the
500 // volunteer(s) immediately.
501 wake_all_sleepers(_volunteers);
502
503 // We should also yield the whole process now, to be polite to the rest
504 // of the system.
505 system_yield();
506 now = get_current_time();
507 }
508 wake_sleepers(_sleeping, now);
509 wake_sleepers(_volunteers, now);
510 }
511
512 bool new_epoch = !_ready.empty() && _next_ready.empty();
513
514 // Choose a new thread to execute.
515 while (true) {
516 // If there are no threads, sleep.
517 while (_ready.empty()) {
518 if (!_next_ready.empty()) {
519 // We've finished an epoch.
520 _ready.swap(_next_ready);
521
522 if (new_epoch && !_tick_records.empty()) {
523 // Pop the oldest timeslice record off when we finish an epoch
524 // without executing any threads, to ensure we don't get caught in
525 // an "all threads reached budget" loop.
526 if (thread_cat->is_debug()) {
527 thread_cat.debug()
528 << "All threads exceeded budget.\n";
529 }
530 TickRecord &record = _tick_records.front();
531 _total_ticks -= record._tick_count;
532
533 if (record._thread->_run_ticks >= record._tick_count) {
534 // Ensure we don't go negative.
535 record._thread->_run_ticks -= record._tick_count;
536 } else {
537 // It is possible for this to happen if the application has been
538 // paused for more than 2^31 ticks.
539 record._thread->_run_ticks = 0;
540 }
541 _tick_records.pop_front();
542 }
543 new_epoch = true;
544
545 } else if (!_volunteers.empty()) {
546 // There are some volunteers. Wake them. Also wake any sleepers that
547 // need it.
548 if (thread_cat->is_debug()) {
549 thread_cat.debug()
550 << "Waking volunteers.\n";
551 }
552 // We should yield the whole process now, to be polite to the rest of
553 // the system.
554 system_yield();
555 now = get_current_time();
556 wake_all_sleepers(_volunteers);
557 wake_sleepers(_sleeping, now);
558
559 } else if (!_sleeping.empty()) {
560 // All threads are sleeping.
561 double wait = _sleeping.front()->_wake_time - now;
562 if (wait > 0.0) {
563 if (thread_cat->is_debug()) {
564 thread_cat.debug()
565 << "Sleeping all threads " << wait << " seconds\n";
566 }
567 system_sleep(wait);
568 }
569 now = get_current_time();
570 wake_sleepers(_sleeping, now);
571 wake_sleepers(_volunteers, now);
572
573 } else {
574 // No threads are ready!
575 if (_waiting_for_exit != nullptr) {
576 // This is a shutdown situation. In this case, we quietly abandoned
577 // the remaining blocked threads, if any, and switch back to the
578 // main thread to finish shutting down.
579 _ready.push_back(_waiting_for_exit);
580 _waiting_for_exit = nullptr;
581 break;
582 }
583
584 // No threads are ready to run, but we're not explicitly shutting
585 // down. This is an error condition, an unintentional deadlock.
586 if (!_blocked.empty()) {
587 thread_cat->error()
588 << "Deadlock! All threads blocked.\n";
589 report_deadlock();
590 abort();
591 }
592
593 // No threads are queued anywhere. This is some kind of internal
594 // error, since normally the main thread, at least, should be queued
595 // somewhere.
596 thread_cat->error()
597 << "All threads disappeared!\n";
598 exit(0);
599 }
600 }
601
602 ThreadSimpleImpl *chosen_thread = _ready.front();
603 _ready.pop_front();
604
605 double timeslice = determine_timeslice(chosen_thread);
606 if (timeslice > 0.0) {
607 // This thread is ready to roll. Break out of the loop.
608 chosen_thread->_start_time = now;
609 chosen_thread->_stop_time = now + timeslice;
610 _current_thread = chosen_thread;
611 break;
612 }
613
614 // This thread is not ready to wake up yet. Put it back for next epoch.
615 // It doesn't count as a volunteer, though--its timeslice was used up.
616 _next_ready.push_back(chosen_thread);
617 }
618
619 // All right, the thread is ready to roll. Begin.
620 if (thread_cat->is_debug()) {
621 size_t blocked_count = 0;
622 Blocked::const_iterator bi;
623 for (bi = _blocked.begin(); bi != _blocked.end(); ++bi) {
624 const FifoThreads &threads = (*bi).second;
625 blocked_count += threads.size();
626 }
627
628 double timeslice = _current_thread->_stop_time - _current_thread->_start_time;
629 thread_cat.debug()
630 << "Switching to " << *_current_thread->_parent_obj
631 << " for " << timeslice << " s ("
632 << _ready.size() << " + " << _next_ready.size()
633 << " + " << _volunteers.size()
634 << " other threads ready, " << blocked_count
635 << " blocked, " << _sleeping.size() << " sleeping)\n";
636 }
637
638 switch_to_thread_context(from_context, _current_thread->_context);
639
640 // Shouldn't get here.
641 nassertv(false);
642 abort();
643}

649void ThreadSimpleManager::
650do_timeslice_accounting(ThreadSimpleImpl *thread, double now) {
651 double elapsed = now - thread->_start_time;
652 if (thread_cat.is_debug()) {
653 thread_cat.debug()
654 << *thread->_parent_obj << " ran for " << elapsed << " s of "
655 << thread->_stop_time - thread->_start_time << " requested.\n";
656 }
657
658 // Clamp the elapsed time at 0. (If it's less than 0, the clock is running
659 // backwards, ick.)
660 elapsed = std::max(elapsed, 0.0);
661
662 unsigned int ticks = (unsigned int)(elapsed * _tick_scale + 0.5);
663 thread->_run_ticks += ticks;
664
665 // Now remove any old records.
666 unsigned int ticks_window = (unsigned int)(_simple_thread_window * _tick_scale + 0.5);
667 while (_total_ticks > ticks_window) {
668 nassertv(!_tick_records.empty());
669 TickRecord &record = _tick_records.front();
670 _total_ticks -= record._tick_count;
671 if (record._thread->_run_ticks >= record._tick_count) {
672 // Ensure we don't go negative.
673 record._thread->_run_ticks -= record._tick_count;
674 } else {
675 // It is possible for this to happen if the application has been paused
676 // for more than 2^31 ticks.
677 record._thread->_run_ticks = 0;
678 }
679 _tick_records.pop_front();
680 }
681
682 // Finally, record the new record.
683 TickRecord record;
684 record._tick_count = ticks;
685 record._thread = thread;
686 _tick_records.push_back(record);
687 _total_ticks += ticks;
688}

750double ThreadSimpleManager::
751determine_timeslice(ThreadSimpleImpl *chosen_thread) {
752 if (_ready.empty() && _next_ready.empty()) {
753 // This is the only ready thread. It gets the full timeslice.
754 return _simple_thread_epoch_timeslice;
755 }
756
757 // Count up the total runtime and weight of all ready threads.
758 unsigned int total_ticks = chosen_thread->_run_ticks;
759 double total_weight = chosen_thread->_priority_weight;
760
761 FifoThreads::const_iterator ti;
762 for (ti = _ready.begin(); ti != _ready.end(); ++ti) {
763 total_ticks += (*ti)->_run_ticks;
764 total_weight += (*ti)->_priority_weight;
765 }
766 for (ti = _next_ready.begin(); ti != _next_ready.end(); ++ti) {
767 total_ticks += (*ti)->_run_ticks;
768 total_weight += (*ti)->_priority_weight;
769 }
770
771 nassertr(total_weight != 0.0, 0.0);
772 double budget_ratio = chosen_thread->_priority_weight / total_weight;
773
774 if (total_ticks == 0) {
775 // This must be the first thread. Special case.
776 return budget_ratio * _simple_thread_epoch_timeslice;
777 }
778
779 double run_ratio = (double)chosen_thread->_run_ticks / (double)total_ticks;
780 double remaining_ratio = budget_ratio - run_ratio;
781
782 if (thread_cat->is_debug()) {
783 thread_cat.debug()
784 << *chosen_thread->_parent_obj << " accrued "
785 << chosen_thread->_run_ticks / _tick_scale << " s of "
786 << total_ticks / _tick_scale << "; budget is "
787 << budget_ratio * total_ticks / _tick_scale << ".\n";
788 if (remaining_ratio <= 0.0) {
789 thread_cat.debug()
790 << "Exceeded budget.\n";
791 }
792 }
793
794 return remaining_ratio * _simple_thread_epoch_timeslice;
795}
